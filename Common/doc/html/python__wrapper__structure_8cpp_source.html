334 bool CDriver::ComputeVertexForces(unsigned short iMarker, unsigned short iVertex) {
335 
336  unsigned long iPoint;
337  unsigned short iDim, jDim;
338  su2double *Normal, AreaSquare, Area;
339  bool halo;
340 
341  unsigned short FinestMesh = config_container[ZONE_0]->GetFinestMesh();
342 
343  /*--- Check the kind of fluid problem ---*/
344  bool compressible = (config_container[ZONE_0]->GetKind_Regime() == COMPRESSIBLE);
345  bool incompressible = (config_container[ZONE_0]->GetKind_Regime() == INCOMPRESSIBLE);
346  bool viscous_flow = ((config_container[ZONE_0]->GetKind_Solver() == NAVIER_STOKES) ||
347  (config_container[ZONE_0]->GetKind_Solver() == RANS) );
348 
349  /*--- Parameters for the calculations ---*/
350  // Pn: Pressure
351  // Pinf: Pressure_infinite
352  // div_vel: Velocity divergence
353  // Dij: Dirac delta
354  su2double Pn = 0.0, div_vel = 0.0, Dij = 0.0;
355  su2double Viscosity = 0.0;
356  su2double Grad_Vel[3][3] = { {0.0, 0.0, 0.0} ,
357  {0.0, 0.0, 0.0} ,
358  {0.0, 0.0, 0.0} } ;
359  su2double Tau[3][3] = { {0.0, 0.0, 0.0} ,
360  {0.0, 0.0, 0.0} ,
361  {0.0, 0.0, 0.0} } ;
362 
363  su2double Pinf = solver_container[ZONE_0][INST_0][FinestMesh][FLOW_SOL]->GetPressure_Inf();
364 
365  iPoint = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNode();
366 
367  /*--- It is necessary to distinguish the halo nodes from the others, since they introduce non physical forces. ---*/
368  if(geometry_container[ZONE_0][INST_0][MESH_0]->node[iPoint]->GetDomain()) {
369  /*--- Get the normal at the vertex: this normal goes inside the fluid domain. ---*/
370  Normal = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNormal();
371  AreaSquare = 0.0;
372  for(iDim = 0; iDim < nDim; iDim++) {
373  AreaSquare += Normal[iDim]*Normal[iDim];
374  }
375  Area = sqrt(AreaSquare);
376 
377  /*--- Get the values of pressure and viscosity ---*/
378  Pn = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->node[iPoint]->GetPressure();
379  if (viscous_flow) {
380  for(iDim=0; iDim<nDim; iDim++) {
381  for(jDim=0; jDim<nDim; jDim++) {
382  Grad_Vel[iDim][jDim] = solver_container[ZONE_0][INST_0][FinestMesh][FLOW_SOL]->node[iPoint]->GetGradient_Primitive(iDim+1, jDim);
383  }
384  }
385  Viscosity = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->node[iPoint]->GetLaminarViscosity();
386  }
387 
388  /*--- Calculate the inviscid (pressure) part of tn in the fluid nodes (force units) ---*/
389  for (iDim = 0; iDim < nDim; iDim++) {
390  PyWrapNodalForce[iDim] = -(Pn-Pinf)*Normal[iDim]; //NB : norm(Normal) = Area
391  }
392 
393  /*--- Calculate the viscous (shear stress) part of tn in the fluid nodes (force units) ---*/
394  if ((incompressible || compressible) && viscous_flow) {
395  div_vel = 0.0;
396  for (iDim = 0; iDim < nDim; iDim++)
397  div_vel += Grad_Vel[iDim][iDim];
398  if (incompressible) div_vel = 0.0;
399 
400  for (iDim = 0; iDim < nDim; iDim++) {
401  for (jDim = 0 ; jDim < nDim; jDim++) {
402  Dij = 0.0; if (iDim == jDim) Dij = 1.0;
403  Tau[iDim][jDim] = Viscosity*(Grad_Vel[jDim][iDim] + Grad_Vel[iDim][jDim]) - TWO3*Viscosity*div_vel*Dij;
404  PyWrapNodalForce[iDim] += Tau[iDim][jDim]*Normal[jDim];
405  }
406  }
407  }
408 
409  //Divide by local are in case of force density communication.
410  for(iDim = 0; iDim < nDim; iDim++) {
411  PyWrapNodalForceDensity[iDim] = PyWrapNodalForce[iDim]/Area;
412  }
413 
414  halo = false;
415  }
416  else {
417  halo = true;
418  }
419 
420  return halo;
421 
422 }
