1104  COptionBase* val = new COptionDouble(name, option_field, default_value);

1107  // During configuration, the parsing script will get the option name, and use this map
1108  // to find how to parse that option.
1109  option_map.insert(pair<string, COptionBase *>(name, val));
1110  }
