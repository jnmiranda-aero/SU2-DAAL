1818 void CSolver::SetUpTypicalSectionWingModel(vector<vector<su2double> >& Phi, vector<su2double>& omega, CConfig *config) {
1819 
1820  /*--- Retrieve values from the config file ---*/
1821  su2double w_h = config->GetAeroelastic_Frequency_Plunge();
1822  su2double w_a = config->GetAeroelastic_Frequency_Pitch();
1823  su2double x_a = config->GetAeroelastic_CG_Location();
1824  su2double r_a = sqrt(config->GetAeroelastic_Radius_Gyration_Squared());
1825  su2double w = w_h/w_a;
1826 
1827  // Mass Matrix
1828  vector<vector<su2double> > M(2,vector<su2double>(2,0.0));
1829  M[0][0] = 1;
1830  M[0][1] = x_a;
1831  M[1][0] = x_a;
1832  M[1][1] = r_a*r_a;
1833 
1834  // Stiffness Matrix
1835  // vector<vector<su2double> > K(2,vector<su2double>(2,0.0));
1836  // K[0][0] = (w_h/w_a)*(w_h/w_a);
1837  // K[0][1] = 0.0;
1838  // K[1][0] = 0.0;
1839  // K[1][1] = r_a*r_a;
1840 
1841  /* Eigenvector and Eigenvalue Matrices of the Generalized EigenValue Problem. */
1842 
1843  vector<vector<su2double> > Omega2(2,vector<su2double>(2,0.0));
1844  su2double aux; // auxiliary variable
1845  aux = sqrt(pow(r_a,2)*pow(w,4) - 2*pow(r_a,2)*pow(w,2) + pow(r_a,2) + 4*pow(x_a,2)*pow(w,2));
1846  Phi[0][0] = (r_a * (r_a - r_a*pow(w,2) + aux)) / (2*x_a*pow(w, 2));
1847  Phi[0][1] = (r_a * (r_a - r_a*pow(w,2) - aux)) / (2*x_a*pow(w, 2));
1848  Phi[1][0] = 1.0;
1849  Phi[1][1] = 1.0;
1850 
1851  Omega2[0][0] = (r_a * (r_a + r_a*pow(w,2) - aux)) / (2*(pow(r_a, 2) - pow(x_a, 2)));
1852  Omega2[0][1] = 0;
1853  Omega2[1][0] = 0;
1854  Omega2[1][1] = (r_a * (r_a + r_a*pow(w,2) + aux)) / (2*(pow(r_a, 2) - pow(x_a, 2)));
1855 
1856  /* Nondimesionalize the Eigenvectors such that Phi'*M*Phi = I and PHI'*K*PHI = Omega */
1857  // Phi'*M*Phi = D
1858  // D^(-1/2)*Phi'*M*Phi*D^(-1/2) = D^(-1/2)*D^(1/2)*D^(1/2)*D^(-1/2) = I, D^(-1/2) = inv(sqrt(D))
1859  // Phi = Phi*D^(-1/2)
1860 
1861  vector<vector<su2double> > Aux(2,vector<su2double>(2,0.0));
1862  vector<vector<su2double> > D(2,vector<su2double>(2,0.0));
1863  // Aux = M*Phi
1864  for (int i=0; i<2; i++) {
1865  for (int j=0; j<2; j++) {
1866  Aux[i][j] = 0;
1867  for (int k=0; k<2; k++) {
1868  Aux[i][j] += M[i][k]*Phi[k][j];
1869  }
1870  }
1871  }
1872 
1873  // D = Phi'*Aux
1874  for (int i=0; i<2; i++) {
1875  for (int j=0; j<2; j++) {
1876  D[i][j] = 0;
1877  for (int k=0; k<2; k++) {
1878  D[i][j] += Phi[k][i]*Aux[k][j]; //PHI transpose
1879  }
1880  }
1881  }
1882 
1883  //Modify the first column
1884  Phi[0][0] = Phi[0][0] * 1/sqrt(D[0][0]);
1885  Phi[1][0] = Phi[1][0] * 1/sqrt(D[0][0]);
1886  //Modify the second column
1887  Phi[0][1] = Phi[0][1] * 1/sqrt(D[1][1]);
1888  Phi[1][1] = Phi[1][1] * 1/sqrt(D[1][1]);
1889 
1890  // Sqrt of the eigenvalues (frequency of vibration of the modes)
1891  omega[0] = sqrt(Omega2[0][0]);
1892  omega[1] = sqrt(Omega2[1][1]);
1893 
1894 }

1896 void CSolver::SolveTypicalSectionWingModel(CGeometry *geometry, su2double Cl, su2double Cm, CConfig *config, unsigned short iMarker, vector<su2double>& displacements) {
1897 
1898  /*--- The aeroelastic model solved in this routine is the typical section wing model
1899  The details of the implementation are similar to those found in J.J. Alonso
1900  "Fully-Implicit Time-Marching Aeroelastic Solutions" 1994. ---*/
1901 
1902  /*--- Retrieve values from the config file ---*/
1903  su2double w_alpha = config->GetAeroelastic_Frequency_Pitch();
1904  su2double vf = config->GetAeroelastic_Flutter_Speed_Index();
1905  su2double b = config->GetLength_Reynolds()/2.0; // airfoil semichord, Reynolds length is by defaul 1.0
1906  su2double dt = config->GetDelta_UnstTimeND();
1907  dt = dt*w_alpha; //Non-dimensionalize the structural time.
1908 
1909  /*--- Structural Equation damping ---*/
1910  vector<su2double> xi(2,0.0);
1911 
1912  /*--- Eigenvectors and Eigenvalues of the Generalized EigenValue Problem. ---*/
1913  vector<vector<su2double> > Phi(2,vector<su2double>(2,0.0)); // generalized eigenvectors.
1914  vector<su2double> w(2,0.0); // sqrt of the generalized eigenvalues (frequency of vibration of the modes).
1915  SetUpTypicalSectionWingModel(Phi, w, config);
1916 
1917  /*--- Solving the Decoupled Aeroelastic Problem with second order time discretization Eq (9) ---*/
1918 
1919  /*--- Solution variables description. //x[j][i], j-entry, i-equation. // Time (n+1)->np1, n->n, (n-1)->n1 ---*/
1920  vector<vector<su2double> > x_np1(2,vector<su2double>(2,0.0));
1921 
1922  /*--- Values from previous movement of spring at true time step n+1
1923  We use this values because we are solving for delta changes not absolute changes ---*/
1924  vector<vector<su2double> > x_np1_old = config->GetAeroelastic_np1(iMarker);
1925 
1926  /*--- Values at previous timesteps. ---*/
1927  vector<vector<su2double> > x_n = config->GetAeroelastic_n(iMarker);
1928  vector<vector<su2double> > x_n1 = config->GetAeroelastic_n1(iMarker);
1929 
1930  /*--- Set up of variables used to solve the structural problem. ---*/
1931  vector<su2double> f_tilde(2,0.0);
1932  vector<vector<su2double> > A_inv(2,vector<su2double>(2,0.0));
1933  su2double detA;
1934  su2double s1, s2;
1935  vector<su2double> rhs(2,0.0); //right hand side
1936  vector<su2double> eta(2,0.0);
1937  vector<su2double> eta_dot(2,0.0);
1938 
1939  /*--- Forcing Term ---*/
1940  su2double cons = vf*vf/PI_NUMBER;
1941  vector<su2double> f(2,0.0);
1942  f[0] = cons*(-Cl);
1943  f[1] = cons*(2*-Cm);
1944 
1945  //f_tilde = Phi'*f
1946  for (int i=0; i<2; i++) {
1947  f_tilde[i] = 0;
1948  for (int k=0; k<2; k++) {
1949  f_tilde[i] += Phi[k][i]*f[k]; //PHI transpose
1950  }
1951  }
1952 
1953  /*--- solve each decoupled equation (The inverse of the 2x2 matrix is provided) ---*/
1954  for (int i=0; i<2; i++) {
1955  /* Matrix Inverse */
1956  detA = 9.0/(4.0*dt*dt) + 3*w[i]*xi[i]/(dt) + w[i]*w[i];
1957  A_inv[0][0] = 1/detA * (3/(2.0*dt) + 2*xi[i]*w[i]);
1958  A_inv[0][1] = 1/detA * 1;
1959  A_inv[1][0] = 1/detA * -w[i]*w[i];
1960  A_inv[1][1] = 1/detA * 3/(2.0*dt);
1961 
1962  /* Source Terms from previous iterations */
1963  s1 = (-4*x_n[0][i] + x_n1[0][i])/(2.0*dt);
1964  s2 = (-4*x_n[1][i] + x_n1[1][i])/(2.0*dt);
1965 
1966  /* Problem Right Hand Side */
1967  rhs[0] = -s1;
1968  rhs[1] = f_tilde[i]-s2;
1969 
1970  /* Solve the equations */
1971  x_np1[0][i] = A_inv[0][0]*rhs[0] + A_inv[0][1]*rhs[1];
1972  x_np1[1][i] = A_inv[1][0]*rhs[0] + A_inv[1][1]*rhs[1];
1973 
1974  eta[i] = x_np1[0][i]-x_np1_old[0][i]; // For displacements, the change(deltas) is used.
1975  eta_dot[i] = x_np1[1][i]; // For velocities, absolute values are used.
1976  }
1977 
1978  /*--- Transform back from the generalized coordinates to get the actual displacements in plunge and pitch q = Phi*eta ---*/
1979  vector<su2double> q(2,0.0);
1980  vector<su2double> q_dot(2,0.0);
1981  for (int i=0; i<2; i++) {
1982  q[i] = 0;
1983  q_dot[i] = 0;
1984  for (int k=0; k<2; k++) {
1985  q[i] += Phi[i][k]*eta[k];
1986  q_dot[i] += Phi[i][k]*eta_dot[k];
1987  }
1988  }
1989 
1990  su2double dh = b*q[0];
1991  su2double dalpha = q[1];
1992 
1993  su2double h_dot = w_alpha*b*q_dot[0]; //The w_a brings it back to actual time.
1994  su2double alpha_dot = w_alpha*q_dot[1];
1995 
1996  /*--- Set the solution of the structural equations ---*/
1997  displacements[0] = dh;
1998  displacements[1] = dalpha;
1999  displacements[2] = h_dot;
2000  displacements[3] = alpha_dot;
2001 
2002  /*--- Calculate the total plunge and total pitch displacements for the unsteady step by summing the displacement at each sudo time step ---*/
2003  su2double pitch, plunge;
2004  pitch = config->GetAeroelastic_pitch(iMarker);
2005  plunge = config->GetAeroelastic_plunge(iMarker);
2006 
2007  config->SetAeroelastic_pitch(iMarker , pitch+dalpha);
2008  config->SetAeroelastic_plunge(iMarker , plunge+dh/b);
2009 
2010  /*--- Set the Aeroelastic solution at time n+1. This gets update every sudo time step
2011  and after convering the sudo time step the solution at n+1 get moved to the solution at n
2012  in SetDualTime_Solver method ---*/
2013 
2014  config->SetAeroelastic_np1(iMarker, x_np1);
2015 
2016 }
