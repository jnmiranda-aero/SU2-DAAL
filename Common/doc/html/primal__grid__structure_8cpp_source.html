66 void CPrimalGrid::SetCoord_CG(su2double **val_coord) {
67  unsigned short iDim, iNode, NodeFace, iFace;
68 
69  AD::StartPreacc();
70  AD::SetPreaccIn(val_coord, GetnNodes(), nDim);
71 
72  for (iDim = 0; iDim < nDim; iDim++) {
73  Coord_CG[iDim] = 0.0;
74  for (iNode = 0; iNode < GetnNodes(); iNode++)
75  Coord_CG[iDim] += val_coord[iNode][iDim]/su2double(GetnNodes());
76  }
77 
78  for (iFace = 0; iFace < GetnFaces(); iFace++)
79  for (iDim = 0; iDim < nDim; iDim++) {
80  Coord_FaceElems_CG[iFace][iDim] = 0.0;
81  for (iNode = 0; iNode < GetnNodesFace(iFace); iNode++) {
82  NodeFace = GetFaces(iFace, iNode);
83  Coord_FaceElems_CG[iFace][iDim] += val_coord[NodeFace][iDim]/su2double(GetnNodesFace(iFace));
84  }
85  }
86 
87  AD::SetPreaccOut(Coord_CG, nDim);
88  AD::SetPreaccOut(Coord_FaceElems_CG, GetnFaces(), nDim);
89  AD::EndPreacc();

727 void CPrimalGridFEM::GetLocalCornerPointsAllFaces(unsigned short elementType,
728  unsigned short nPoly,
729  unsigned short nDOFs,
730  unsigned short &numFaces,
731  unsigned short nPointsPerFace[],
732  unsigned long faceConn[6][4]) {
733 
734  /*--- Determine the element type and set the face data accordingly.
735  The faceConn values are local to the element. ---*/
736 
737  unsigned short nn2, nn3, nn4;
738  switch( elementType ) {
739  case TRIANGLE:
740  numFaces = 3;
741  nPointsPerFace[0] = 2; faceConn[0][0] = 0; faceConn[0][1] = nPoly;
742  nPointsPerFace[1] = 2; faceConn[1][0] = nPoly; faceConn[1][1] = nDOFs -1;
743  nPointsPerFace[2] = 2; faceConn[2][0] = nDOFs -1; faceConn[2][1] = 0;
744  break;
745 
746  case QUADRILATERAL:
747  numFaces = 4; nn2 = nPoly*(nPoly+1);
748  nPointsPerFace[0] = 2; faceConn[0][0] = 0; faceConn[0][1] = nPoly;
749  nPointsPerFace[1] = 2; faceConn[1][0] = nPoly; faceConn[1][1] = nDOFs -1;
750  nPointsPerFace[2] = 2; faceConn[2][0] = nDOFs -1; faceConn[2][1] = nn2;
751  nPointsPerFace[3] = 2; faceConn[3][0] = nn2; faceConn[3][1] = 0;
752  break;
753 
754  case TETRAHEDRON:
755  numFaces = 4; nn2 = (nPoly+1)*(nPoly+2)/2 -1; nn3 = nDOFs -1;
756  nPointsPerFace[0] = 3; faceConn[0][0] = 0; faceConn[0][1] = nPoly; faceConn[0][2] = nn2;
757  nPointsPerFace[1] = 3; faceConn[1][0] = 0; faceConn[1][1] = nn3; faceConn[1][2] = nPoly;
758  nPointsPerFace[2] = 3; faceConn[2][0] = 0; faceConn[2][1] = nn2; faceConn[2][2] = nn3;
759  nPointsPerFace[3] = 3; faceConn[3][0] = nPoly; faceConn[3][1] = nn3; faceConn[3][2] = nn2;
760  break;
761 
762  case PYRAMID:
763  numFaces = 5; nn2 = (nPoly+1)*(nPoly+1) -1; nn3 = nn2 - nPoly;
764  nPointsPerFace[0] = 4; faceConn[0][0] = 0; faceConn[0][1] = nPoly; faceConn[0][2] = nn2; faceConn[0][3] = nn3;
765  nPointsPerFace[1] = 3; faceConn[1][0] = 0; faceConn[1][1] = nDOFs -1; faceConn[1][2] = nPoly;
766  nPointsPerFace[2] = 3; faceConn[2][0] = nn3; faceConn[2][1] = nn2; faceConn[2][2] = nDOFs -1;
767  nPointsPerFace[3] = 3; faceConn[3][0] = 0; faceConn[3][1] = nn3; faceConn[3][2] = nDOFs -1;
768  nPointsPerFace[4] = 3; faceConn[4][0] = nPoly; faceConn[4][1] = nDOFs -1; faceConn[4][2] = nn2;
769  break;
770 
771  case PRISM:
772  numFaces = 5; nn2 = (nPoly+1)*(nPoly+2)/2; nn3 = nPoly*nn2; --nn2;
773  nPointsPerFace[0] = 3; faceConn[0][0] = 0; faceConn[0][1] = nPoly; faceConn[0][2] = nn2;
774  nPointsPerFace[1] = 3; faceConn[1][0] = nn3; faceConn[1][1] = nn2+nn3; faceConn[1][2] = nPoly+nn3;
775  nPointsPerFace[2] = 4; faceConn[2][0] = 0; faceConn[2][1] = nn3; faceConn[2][2] = nPoly+nn3; faceConn[2][3] = nPoly;
776  nPointsPerFace[3] = 4; faceConn[3][0] = 0; faceConn[3][1] = nn2; faceConn[3][2] = nn2+nn3; faceConn[3][3] = nn3;
777  nPointsPerFace[4] = 4; faceConn[4][0] = nPoly; faceConn[4][1] = nPoly+nn3; faceConn[4][2] = nn2+nn3; faceConn[4][3] = nn2;
778  break;
779 
780  case HEXAHEDRON:
781  numFaces = 6; nn2 = (nPoly+1)*(nPoly+1); nn4 = nPoly*nn2; --nn2; nn3 = nn2 - nPoly;
782  nPointsPerFace[0] = 4; faceConn[0][0] = 0; faceConn[0][1] = nPoly; faceConn[0][2] = nn2; faceConn[0][3] = nn3;
783  nPointsPerFace[1] = 4; faceConn[1][0] = nn4; faceConn[1][1] = nn3+nn4; faceConn[1][2] = nn2+nn4; faceConn[1][3] = nPoly+nn4;
784  nPointsPerFace[2] = 4; faceConn[2][0] = 0; faceConn[2][1] = nn4; faceConn[2][2] = nPoly+nn4; faceConn[2][3] = nPoly;
785  nPointsPerFace[3] = 4; faceConn[3][0] = nn3; faceConn[3][1] = nn2; faceConn[3][2] = nn2+nn4; faceConn[3][3] = nn3+nn4;
786  nPointsPerFace[4] = 4; faceConn[4][0] = 0; faceConn[4][1] = nn3; faceConn[4][2] = nn3+nn4; faceConn[4][3] = nn4;
787  nPointsPerFace[5] = 4; faceConn[5][0] = nPoly; faceConn[5][1] = nPoly+nn4; faceConn[5][2] = nn2+nn4; faceConn[5][3] = nn2;
788  break;
789  }
