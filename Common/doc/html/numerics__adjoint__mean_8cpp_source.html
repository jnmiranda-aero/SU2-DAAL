470  val_resconv_j[nVar-1] = -(ProjVelocity_j*MeanPsiE + Gamma_Minus_One*phis1);

475  su2double ProjGridVel = 0.0;
476  for (iDim = 0; iDim < nDim; iDim++)
477  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];

479  for (iDim = 0; iDim < nDim; iDim++)

493  for (jDim = 0; jDim < nDim; jDim++)

496  val_Jacobian_jj[iDim+1][nVar-1] = -0.5*Enthalpy_j*Normal[iDim] + 0.5*Gamma_Minus_One*Velocity_j[iDim]*ProjVelocity_j;
497  }
498  val_Jacobian_jj[nVar-1][0] = 0;
499  for (jDim = 0; jDim < nDim; jDim++)
500  val_Jacobian_jj[nVar-1][jDim+1] = -0.5*Gamma_Minus_One*Normal[jDim];
501  val_Jacobian_jj[nVar-1][nVar-1] = -0.5*Gamma*ProjVelocity_j;
502 
503  for (iVar = 0; iVar < nVar; iVar++)
504  for (jVar = 0; jVar < nVar; jVar++)
505  val_Jacobian_ji[iVar][jVar] = val_Jacobian_jj[iVar][jVar];
506 
507  /*--- Jacobian contributions due to grid motion at point j ---*/
508 
509  if (grid_movement) {
510  su2double ProjGridVel = 0.0;
511  for (iDim = 0; iDim < nDim; iDim++)
512  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
513  for (iVar = 0; iVar < nVar; iVar++) {
514  val_Jacobian_jj[iVar][iVar] += 0.5*ProjGridVel;
515  val_Jacobian_ji[iVar][iVar] += 0.5*ProjGridVel;
516  }
517  }
518  }
519

521 
522  for (iVar = 0; iVar < nVar; iVar++) {
523  Diff_Lapl[iVar] = Und_Lapl_i[iVar]-Und_Lapl_j[iVar];
524  Diff_Psi[iVar] = Psi_i[iVar]-Psi_j[iVar];
525  }
526 
527  /*--- Adjustment to projected velocity due to grid motion ---*/
528 
529  if (grid_movement) {
530  su2double ProjGridVel = 0.0;
531  for (iDim = 0; iDim < nDim; iDim++) {
532  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
533  }
534  ProjVelocity_i -= ProjGridVel;
535  ProjVelocity_j += ProjGridVel;
536  }
537 
538  /*--- Compute the spectral radius and stretching factor ---*/
539 
540  Local_Lambda_i = (fabs(ProjVelocity_i)+SoundSpeed_i*Area);
541  Local_Lambda_j = (fabs(ProjVelocity_j)+SoundSpeed_j*Area);
542  MeanLambda = 0.5*(Local_Lambda_i+Local_Lambda_j);
543 
544  Phi_i = pow(Lambda_i/(4.0*MeanLambda), Param_p);
545  Phi_j = pow(Lambda_j/(4.0*MeanLambda), Param_p);
546  StretchingFactor = 4.0*Phi_i*Phi_j/(Phi_i+Phi_j);
547 
548  su2double sc2 = 3.0*(su2double(Neighbor_i)+su2double(Neighbor_j))/(su2double(Neighbor_i)*su2double(Neighbor_j));
549  sc4 = sc2*sc2/4.0;
550  Epsilon_2 = Param_Kappa_2*0.5*(Sensor_i+Sensor_j)*sc2;
551  Epsilon_4 = max(0.0, Param_Kappa_4-Epsilon_2)*sc4;
552 
553  /*--- Compute viscous residual 1st- & 3rd-order dissipation ---*/
554 
555  for (iVar = 0; iVar < nVar; iVar++) {
556  Residual = (Epsilon_2*Diff_Psi[iVar]-Epsilon_4*Diff_Lapl[iVar])*StretchingFactor*MeanLambda;
557  val_resvisc_i[iVar] = -Residual;
558  val_resvisc_j[iVar] = Residual;
559  if (implicit) {
560  val_Jacobian_ii[iVar][iVar] -= Epsilon_2 + su2double(Neighbor_i+1)*Epsilon_4*StretchingFactor*MeanLambda;
561  val_Jacobian_ij[iVar][iVar] += Epsilon_2 + su2double(Neighbor_j+1)*Epsilon_4*StretchingFactor*MeanLambda;
562  val_Jacobian_ji[iVar][iVar] += Epsilon_2 + su2double(Neighbor_i+1)*Epsilon_4*StretchingFactor*MeanLambda;
563  val_Jacobian_jj[iVar][iVar] -= Epsilon_2 + su2double(Neighbor_j+1)*Epsilon_4*StretchingFactor*MeanLambda;
564  }
565  }
