113 void CCentSca_Heat::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i, su2double **val_Jacobian_j, CConfig *config) {
114 
115  AD::StartPreacc();
116  AD::SetPreaccIn(V_i, nDim+1); AD::SetPreaccIn(V_j, nDim+1);
117  AD::SetPreaccIn(Temp_i); AD::SetPreaccIn(Temp_j);
118  AD::SetPreaccIn(Normal, nDim);
119  if (grid_movement) {
120  AD::SetPreaccIn(GridVel_i, nDim); AD::SetPreaccIn(GridVel_j, nDim);
121  }
122 
123  /*--- Primitive variables at point i and j ---*/
124 
125  Pressure_i = V_i[0]; Pressure_j = V_j[0];
126  DensityInc_i = V_i[nDim+1]; DensityInc_j = V_j[nDim+1];
127  BetaInc2_i = V_i[nDim+2]; BetaInc2_j = V_j[nDim+2];
128 
129  /*--- Projected velocities at the current edge ---*/
130 
131  ProjVelocity = 0.0; ProjVelocity_i = 0.0; ProjVelocity_j = 0.0; Area = 0.0;
132  for (iDim = 0; iDim < nDim; iDim++) {
133  ProjVelocity_i += V_i[iDim+1]*Normal[iDim];
134  ProjVelocity_j += V_j[iDim+1]*Normal[iDim];
135  Area += Normal[iDim]*Normal[iDim];
136  }
137  Area = sqrt(Area);
138 
139  /*--- Computing the second order centered scheme part ---*/
140 
141  ProjVelocity = 0.5*(ProjVelocity_i+ProjVelocity_j);
142 
143  val_residual[0] = 0.5*(Temp_i + Temp_j)*ProjVelocity;
144 
145  if (implicit) {
146  val_Jacobian_i[0][0] = 0.5*ProjVelocity;
147  val_Jacobian_j[0][0] = 0.5*ProjVelocity;
148  }
149 
150  /*--- Adding artificial dissipation to stabilize the centered scheme ---*/
151 
152  Diff_Lapl[0] = Und_Lapl_i[0]-Und_Lapl_j[0];
153 
154  SoundSpeed_i = sqrt(ProjVelocity_i*ProjVelocity_i + (BetaInc2_i/DensityInc_i)*Area*Area);
155  SoundSpeed_j = sqrt(ProjVelocity_j*ProjVelocity_j + (BetaInc2_j/DensityInc_j)*Area*Area);
156 
157  Local_Lambda_i = fabs(ProjVelocity_i)+SoundSpeed_i;
158  Local_Lambda_j = fabs(ProjVelocity_j)+SoundSpeed_j;
159  MeanLambda = 0.5*(Local_Lambda_i+Local_Lambda_j);
160 
161  val_residual[0] += -Param_Kappa_4*Diff_Lapl[0]*MeanLambda;
162 
163  if (implicit) {
164  cte_0 = Param_Kappa_4*su2double(Neighbor_i+1)*MeanLambda;
165  cte_1 = Param_Kappa_4*su2double(Neighbor_j+1)*MeanLambda;
166 
167  val_Jacobian_i[0][0] += cte_0;
168  val_Jacobian_j[0][0] -= cte_1;
169  }
170 
171  AD::SetPreaccOut(val_residual[0]);
172  AD::EndPreacc();
173 
174 }

200 void CAvgGrad_Heat::ComputeResidual(su2double *val_residual, su2double **Jacobian_i, su2double **Jacobian_j, CConfig *config) {
201 
202  AD::StartPreacc();
203  AD::SetPreaccIn(Coord_i, nDim); AD::SetPreaccIn(Coord_j, nDim);
204  AD::SetPreaccIn(Normal, nDim);
205  AD::SetPreaccIn(Temp_i); AD::SetPreaccIn(Temp_j);
206  AD::SetPreaccIn(ConsVar_Grad_i[0],nDim); AD::SetPreaccIn(ConsVar_Grad_j[0],nDim);
207  AD::SetPreaccIn(Thermal_Diffusivity_i); AD::SetPreaccIn(Thermal_Conductivity_j);
208 
209  Thermal_Diffusivity_Mean = 0.5*(Thermal_Diffusivity_i + Thermal_Diffusivity_j);
210 
211  /*--- Compute vector going from iPoint to jPoint ---*/
212 
213  dist_ij_2 = 0; proj_vector_ij = 0;
214  for (iDim = 0; iDim < nDim; iDim++) {
215  Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
216  dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
217  proj_vector_ij += Edge_Vector[iDim]*Normal[iDim];
218  }
219  if (dist_ij_2 == 0.0) proj_vector_ij = 0.0;
220  else proj_vector_ij = proj_vector_ij/dist_ij_2;
221 
222  /*--- Mean gradient approximation. Projection of the mean gradient in the direction of the edge ---*/
223  for (iVar = 0; iVar < nVar; iVar++) {
224  Proj_Mean_GradHeatVar_Normal[iVar] = 0.0;
225  Proj_Mean_GradHeatVar_Corrected[iVar] = 0.0;
226  for (iDim = 0; iDim < nDim; iDim++) {
227  Mean_GradHeatVar[iVar][iDim] = 0.5*(ConsVar_Grad_i[iVar][iDim] + ConsVar_Grad_j[iVar][iDim]);
228  Proj_Mean_GradHeatVar_Normal[iVar] += Mean_GradHeatVar[iVar][iDim]*Normal[iDim];
229  }
230  Proj_Mean_GradHeatVar_Corrected[iVar] = Proj_Mean_GradHeatVar_Normal[iVar];
231  }
232 
233  val_residual[0] = Thermal_Diffusivity_Mean*Proj_Mean_GradHeatVar_Corrected[0];
234 
235  /*--- For Jacobians -> Use of TSL approx. to compute derivatives of the gradients ---*/
236  if (implicit) {
237  Jacobian_i[0][0] = -Thermal_Diffusivity_Mean*proj_vector_ij;
238  Jacobian_j[0][0] = Thermal_Diffusivity_Mean*proj_vector_ij;
239  }
240 
241  AD::SetPreaccOut(val_residual, nVar);
242  AD::EndPreacc();
