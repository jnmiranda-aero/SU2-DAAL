42 inline void CBaseMPIWrapper::Error(std::string ErrorMsg, std::string FunctionName){
43 
44  /* Set MinRankError to Rank, as the error message is called on this rank. */
45  MinRankError = Rank;
46  int flag = 0;
47 
48 #if MPI_VERSION >= 3
49  /* Find out whether the error call is collective via MPI_Ibarrier. */
50  Request barrierRequest;
51  MPI_Ibarrier(currentComm, &barrierRequest);
52 
53  /* Try to complete the non-blocking barrier call for a second. */
54  double startTime = MPI_Wtime();
55  while( true ) {
56 
57  MPI_Test(&barrierRequest, &flag, MPI_STATUS_IGNORE);
58  if( flag ) break;
59 
60  double currentTime = MPI_Wtime();
61  if(currentTime > startTime + 1.0) break;
62  }
63 #else
64  /* MPI_Ibarrier function is not supported. Simply wait for one
65  second to give other ranks the opportunity to reach this point. */
66  sleep(1);
67 #endif
68 
69  if( flag ) {
70  /* The barrier is completed and hence the error call is collective.
71  Set MinRankError to 0. */
72  MinRankError = 0;
73  }
74  else {
75  /* The error call is not collective and the minimum rank must be
76  determined by one sided communication. Loop over the lower numbered
77  ranks to check if they participate in the error message. */
78  for(int i=0; i<Rank; ++i) {
79 
80  int MinRankErrorOther;
81  MPI_Win_lock(MPI_LOCK_SHARED, i, 0, winMinRankError);
82  MPI_Get(&MinRankErrorOther, 1, MPI_INT, i, 0, 1, MPI_INT, winMinRankError);
83  MPI_Win_unlock(i, winMinRankError);
84 
85  if(MinRankErrorOther < MinRankError) {
86  MinRankError = MinRankErrorOther;
87  break;
88  }
89  }
90  }
91 
92  /* Check if this rank must write the error message and do so. */
93  if (Rank == MinRankError){
94  std::cout << std::endl << std::endl;
95  std::cout << "Error in \"" << FunctionName << "\": " << std::endl;
96  std::cout << "-------------------------------------------------------------------------" << std::endl;
97  std::cout << ErrorMsg << std::endl;
98  std::cout << "------------------------------ Error Exit -------------------------------" << std::endl;
99  std::cout << std::endl << std::endl;
100  }
101  Abort(currentComm, 0);
102 }

602 inline void CBaseMPIWrapper::CopyData(void *sendbuf, void *recvbuf, int size, Datatype datatype){
603  switch (datatype) {
604  case MPI_DOUBLE:
605  for (int i = 0; i < size; i++){
606  static_cast<su2double*>(recvbuf)[i] = static_cast<su2double*>(sendbuf)[i];
607  }
608  break;
609  case MPI_UNSIGNED_LONG:
610  for (int i = 0; i < size; i++){
611  static_cast<unsigned long*>(recvbuf)[i] = static_cast<unsigned long*>(sendbuf)[i];
612  }
613  break;
