619 void CFluidIteration::SetWind_GustField(CConfig *config_container, CGeometry **geometry_container, CSolver ***solver_container) {
620  // The gust is imposed on the flow field via the grid velocities. This method called the Field Velocity Method is described in the
621  // NASA TM–2012-217771 - Development, Verification and Use of Gust Modeling in the NASA Computational Fluid Dynamics Code FUN3D
622  // the desired gust is prescribed as the negative of the grid velocity.
623 
624  // If a source term is included to account for the gust field, the method is described by Jones et al. as the Split Velocity Method in
625  // Simulation of Airfoil Gust Responses Using Prescribed Velocities.
626  // In this routine the gust derivatives needed for the source term are calculated when applicable.
627  // If the gust derivatives are zero the source term is also zero.
628  // The source term itself is implemented in the class CSourceWindGust
629 
630  if (rank == MASTER_NODE)
631  cout << endl << "Running simulation with a Wind Gust." << endl;
632  unsigned short iDim, nDim = geometry_container[MESH_0]->GetnDim(); //We assume nDim = 2
633  if (nDim != 2) {
634  if (rank == MASTER_NODE) {
635  cout << endl << "WARNING - Wind Gust capability is only verified for 2 dimensional simulations." << endl;
636  }
637  }
638 
639  /*--- Gust Parameters from config ---*/
640  unsigned short Gust_Type = config_container->GetGust_Type();
641  su2double xbegin = config_container->GetGust_Begin_Loc(); // Location at which the gust begins.
642  su2double L = config_container->GetGust_WaveLength(); // Gust size
643  su2double tbegin = config_container->GetGust_Begin_Time(); // Physical time at which the gust begins.
644  su2double gust_amp = config_container->GetGust_Ampl(); // Gust amplitude
645  su2double n = config_container->GetGust_Periods(); // Number of gust periods
646  unsigned short GustDir = config_container->GetGust_Dir(); // Gust direction
647 
648  /*--- Variables needed to compute the gust ---*/
649  unsigned short Kind_Grid_Movement = config_container->GetKind_GridMovement(ZONE_0);
650  unsigned long iPoint;
651  unsigned short iMGlevel, nMGlevel = config_container->GetnMGLevels();
652 
653  su2double x, y, x_gust, dgust_dx, dgust_dy, dgust_dt;
654  su2double *Gust, *GridVel, *NewGridVel, *GustDer;
655 
656  su2double Physical_dt = config_container->GetDelta_UnstTime();
657  unsigned long ExtIter = config_container->GetExtIter();
658  su2double Physical_t = ExtIter*Physical_dt;
659 
660  su2double Uinf = solver_container[MESH_0][FLOW_SOL]->GetVelocity_Inf(0); // Assumption gust moves at infinity velocity
661 
662  Gust = new su2double [nDim];
663  NewGridVel = new su2double [nDim];
664  for (iDim = 0; iDim < nDim; iDim++) {
665  Gust[iDim] = 0.0;
666  NewGridVel[iDim] = 0.0;
667  }
668 
669  GustDer = new su2double [3];
670  for (unsigned short i = 0; i < 3; i++) {
671  GustDer[i] = 0.0;
672  }
673 
674  // Vortex variables
675  unsigned long nVortex = 0;
676  vector<su2double> x0, y0, vort_strenth, r_core; //vortex is positive in clockwise direction.
677  if (Gust_Type == VORTEX) {
678  InitializeVortexDistribution(nVortex, x0, y0, vort_strenth, r_core);
679  }
680 
681  /*--- Check to make sure gust lenght is not zero or negative (vortex gust doesn't use this). ---*/
682  if (L <= 0.0 && Gust_Type != VORTEX) {
683  SU2_MPI::Error("The gust length needs to be positive", CURRENT_FUNCTION);
684  }
685 
686  /*--- Loop over all multigrid levels ---*/
687 
688  for (iMGlevel = 0; iMGlevel <= nMGlevel; iMGlevel++) {
689 
690  /*--- Loop over each node in the volume mesh ---*/
691 
692  for (iPoint = 0; iPoint < geometry_container[iMGlevel]->GetnPoint(); iPoint++) {
693 
694  /*--- Reset the Grid Velocity to zero if there is no grid movement ---*/
695  if (Kind_Grid_Movement == GUST) {
696  for (iDim = 0; iDim < nDim; iDim++)
697  geometry_container[iMGlevel]->node[iPoint]->SetGridVel(iDim, 0.0);
698  }
699 
700  /*--- initialize the gust and derivatives to zero everywhere ---*/
701 
702  for (iDim = 0; iDim < nDim; iDim++) {Gust[iDim]=0.0;}
703  dgust_dx = 0.0; dgust_dy = 0.0; dgust_dt = 0.0;
704 
705  /*--- Begin applying the gust ---*/
706 
707  if (Physical_t >= tbegin) {
708 
709  x = geometry_container[iMGlevel]->node[iPoint]->GetCoord()[0]; // x-location of the node.
710  y = geometry_container[iMGlevel]->node[iPoint]->GetCoord()[1]; // y-location of the node.
711 
712  // Gust coordinate
713  x_gust = (x - xbegin - Uinf*(Physical_t-tbegin))/L;
714 
715  /*--- Calculate the specified gust ---*/
716  switch (Gust_Type) {
717 
718  case TOP_HAT:
719  // Check if we are in the region where the gust is active
720  if (x_gust > 0 && x_gust < n) {
721  Gust[GustDir] = gust_amp;
722  // Still need to put the gust derivatives. Think about this.
723  }
724  break;
725 
726  case SINE:
727  // Check if we are in the region where the gust is active
728  if (x_gust > 0 && x_gust < n) {
729  Gust[GustDir] = gust_amp*(sin(2*PI_NUMBER*x_gust));
730 
731  // Gust derivatives
732  //dgust_dx = gust_amp*2*PI_NUMBER*(cos(2*PI_NUMBER*x_gust))/L;
733  //dgust_dy = 0;
734  //dgust_dt = gust_amp*2*PI_NUMBER*(cos(2*PI_NUMBER*x_gust))*(-Uinf)/L;
735  }
736  break;
737 
738  case ONE_M_COSINE:
739  // Check if we are in the region where the gust is active
740  if (x_gust > 0 && x_gust < n) {
741  Gust[GustDir] = gust_amp*(1-cos(2*PI_NUMBER*x_gust));
742 
743  // Gust derivatives
744  //dgust_dx = gust_amp*2*PI_NUMBER*(sin(2*PI_NUMBER*x_gust))/L;
745  //dgust_dy = 0;
746  //dgust_dt = gust_amp*2*PI_NUMBER*(sin(2*PI_NUMBER*x_gust))*(-Uinf)/L;
747  }
748  break;
749 
750  case EOG:
751  // Check if we are in the region where the gust is active
752  if (x_gust > 0 && x_gust < n) {
753  Gust[GustDir] = -0.37*gust_amp*sin(3*PI_NUMBER*x_gust)*(1-cos(2*PI_NUMBER*x_gust));
754  }
755  break;
756 
757  case VORTEX:
758 
759  /*--- Use vortex distribution ---*/
760  // Algebraic vortex equation.
761  for (unsigned long i=0; i<nVortex; i++) {
762  su2double r2 = pow(x-(x0[i]+Uinf*(Physical_t-tbegin)), 2) + pow(y-y0[i], 2);
763  su2double r = sqrt(r2);
764  su2double v_theta = vort_strenth[i]/(2*PI_NUMBER) * r/(r2+pow(r_core[i],2));
765  Gust[0] = Gust[0] + v_theta*(y-y0[i])/r;
766  Gust[1] = Gust[1] - v_theta*(x-(x0[i]+Uinf*(Physical_t-tbegin)))/r;
767  }
768  break;
769 
770  case NONE: default:
771 
772  /*--- There is no wind gust specified. ---*/
773  if (rank == MASTER_NODE) {
774  cout << "No wind gust specified." << endl;
775  }
776  break;
777 
778  }
779  }
780 
781  /*--- Set the Wind Gust, Wind Gust Derivatives and the Grid Velocities ---*/
782 
783  GustDer[0] = dgust_dx;
784  GustDer[1] = dgust_dy;
785  GustDer[2] = dgust_dt;
786 
787  solver_container[iMGlevel][FLOW_SOL]->node[iPoint]->SetWindGust(Gust);
788  solver_container[iMGlevel][FLOW_SOL]->node[iPoint]->SetWindGustDer(GustDer);
789 
790  GridVel = geometry_container[iMGlevel]->node[iPoint]->GetGridVel();
791 
792  /*--- Store new grid velocity ---*/
793 
794  for (iDim = 0; iDim < nDim; iDim++) {
795  NewGridVel[iDim] = GridVel[iDim] - Gust[iDim];
796  geometry_container[iMGlevel]->node[iPoint]->SetGridVel(iDim, NewGridVel[iDim]);
797  }
798 
799  }
800  }
801 
802  delete [] Gust;
803  delete [] GustDer;
804  delete [] NewGridVel;
805 
806 }
