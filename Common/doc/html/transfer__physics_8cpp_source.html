683 void CTransfer_ConjugateHeatVars::GetDonor_Variable(CSolver *donor_solution, CGeometry *donor_geometry, CConfig *donor_config,
684  unsigned long Marker_Donor, unsigned long Vertex_Donor, unsigned long Point_Donor) {
685 
686  unsigned long iPoint;
687  unsigned long PointNormal;
688  unsigned short nDim, iDim;
689 
690  nDim = donor_geometry->GetnDim();
691 
692  su2double *Coord, *Coord_Normal, *Normal, *Edge_Vector, dist, dist2, Area, Twall = 0.0, Tnormal = 0.0,
693  dTdn, cp_fluid, rho_cp_solid, Prandtl_Lam, laminar_viscosity,
694  thermal_diffusivity, thermal_conductivity, thermal_conductivityND, heat_flux_density, conductivity_over_dist, Temperature_Ref;
695  su2double Gamma = donor_config->GetGamma();
696  su2double Gas_Constant = donor_config->GetGas_ConstantND();
697  su2double Cp = (Gamma / (Gamma - 1.0)) * Gas_Constant;
698  Edge_Vector = new su2double[nDim];
699 
700  /*--- Check whether the current zone is a solid zone or a fluid zone ---*/
701  bool flow = ((donor_config->GetKind_Solver() == NAVIER_STOKES)
702  || (donor_config->GetKind_Solver() == RANS)
703  || (donor_config->GetKind_Solver() == DISC_ADJ_NAVIER_STOKES)
704  || (donor_config->GetKind_Solver() == DISC_ADJ_RANS));
705  bool compressible_flow = (donor_config->GetKind_Regime() == COMPRESSIBLE) && flow;
706  bool incompressible_flow = (donor_config->GetEnergy_Equation()) && flow;
707  bool heat_equation = donor_config->GetKind_Solver() == HEAT_EQUATION_FVM;
708 
709  Temperature_Ref = donor_config->GetTemperature_Ref();
710  Prandtl_Lam = donor_config->GetPrandtl_Lam();
711  laminar_viscosity = donor_config->GetMu_ConstantND(); // TDE check for consistency
712  cp_fluid = donor_config->GetSpecific_Heat_Cp();
713  rho_cp_solid = donor_config->GetSpecific_Heat_Cp_Solid()*donor_config->GetDensity_Solid();
714 
715  PointNormal = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNormal_Neighbor();
716  Coord = donor_geometry->node[Point_Donor]->GetCoord();
717  Coord_Normal = donor_geometry->node[PointNormal]->GetCoord();
718  Normal = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNormal();
719 
720  dist2 = 0.0;
721  Area = 0.0;
722  for (iDim = 0; iDim < nDim; iDim++) {
723  Edge_Vector[iDim] = Coord_Normal[iDim] - Coord[iDim];
724  dist2 += Edge_Vector[iDim]*Edge_Vector[iDim];
725  Area += Normal[iDim]*Normal[iDim];
726  }
727  dist = sqrt(dist2);
728  Area = sqrt(Area);
729 
730  /*--- Retrieve temperature solution (later set is as the first donor variable) and its gradient ---*/
731 
732  dTdn = 0.0;
733 
734  if (compressible_flow) {
735 
736  Twall = donor_solution->node[Point_Donor]->GetPrimitive(0)*Temperature_Ref;
737  Tnormal = donor_solution->node[PointNormal]->GetPrimitive(0)*Temperature_Ref;
738 
739  dTdn = (Twall - Tnormal)/dist;
740  }
741  else if (incompressible_flow) {
742 
743  Twall = donor_solution->node[Point_Donor]->GetTemperature()*Temperature_Ref;
744  Tnormal = donor_solution->node[PointNormal]->GetTemperature()*Temperature_Ref;
745 
746  dTdn = (Twall - Tnormal)/dist;
747  }
748  else if (flow || heat_equation) {
749  Twall = donor_solution->node[Point_Donor]->GetSolution(0)*Temperature_Ref;
750  Tnormal = donor_solution->node[PointNormal]->GetSolution(0)*Temperature_Ref;
751 
752 // for (iDim = 0; iDim < nDim; iDim++) {
753 // dTdn += (Twall - Tnormal)/dist * (Edge_Vector[iDim]/dist) * (Normal[iDim]/Area);
754 // }
755 
756  dTdn = (Twall - Tnormal)/dist;
757  }
758  else {
759  cout << "WARNING: Transfer of conjugate heat variables is called with non-supported donor solver!" << endl;
760  }
761 
762  /*--- Calculate the heat flux density (temperature gradient times thermal conductivity) and set it as second donor variable ---*/
763  if (compressible_flow) {
764 
765  iPoint = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNode();
766 
767  thermal_conductivityND = Cp*(laminar_viscosity/Prandtl_Lam);
768  thermal_conductivity = thermal_conductivityND*donor_config->GetViscosity_Ref();
769 
770  heat_flux_density = thermal_conductivity*dTdn;
771  conductivity_over_dist = thermal_conductivity/dist;
772  }
773  else if (incompressible_flow) {
774 
775  iPoint = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNode();
776 
777  thermal_conductivityND = donor_solution->node[iPoint]->GetThermalConductivity();
778  thermal_conductivity = thermal_conductivityND*donor_config->GetConductivity_Ref();
779 
780  switch (donor_config->GetKind_ConductivityModel()) {
781 
782  case CONSTANT_CONDUCTIVITY:
783  thermal_conductivity = thermal_conductivityND*donor_config->GetConductivity_Ref();
784  break;
785 
786  case CONSTANT_PRANDTL:
787  thermal_conductivity = thermal_conductivityND*donor_config->GetGas_Constant_Ref()*donor_config->GetViscosity_Ref();
788  break;
789  }
790 
791  heat_flux_density = thermal_conductivity*dTdn;
792  conductivity_over_dist = thermal_conductivity/dist;
793  }
794  else if (flow) {
795 
796  iPoint = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNode();
797 
798  thermal_conductivityND = laminar_viscosity/Prandtl_Lam;
799  thermal_conductivity = thermal_conductivityND*donor_config->GetViscosity_Ref()*cp_fluid;
800 
801  heat_flux_density = thermal_conductivity*dTdn;
802  conductivity_over_dist = thermal_conductivity/dist;
803  }
804  else {
805 
806  thermal_diffusivity = donor_config->GetThermalDiffusivity_Solid();
807  heat_flux_density = (thermal_diffusivity*dTdn)*rho_cp_solid;
808  conductivity_over_dist = thermal_diffusivity*rho_cp_solid/dist;
809  }
810 
811  Donor_Variable[0] = Twall;
812  Donor_Variable[1] = heat_flux_density;
813  Donor_Variable[2] = conductivity_over_dist;
814  Donor_Variable[3] = Tnormal;
815 
816  delete [] Edge_Vector;
817 }
