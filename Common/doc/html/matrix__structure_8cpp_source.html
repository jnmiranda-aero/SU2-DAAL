760

762  su2double weight, aux;
763

765 
766  /*--- Copy block matrix, note that the original matrix
767  is modified by the algorithm---*/
768 
769  if (!transposed) {
770  for (iVar = 0; iVar < (short)nVar; iVar++)
771  for (jVar = 0; jVar < (short)nVar; jVar++)

773  } else {

775  for (jVar = 0; jVar < (short)nVar; jVar++)

779

781  rhs[0] /= block[0];
782  }
