48 void CSysSolve::GenerateGivens(su2double & dx, su2double & dy, su2double & s, su2double & c) {
49 
50  if ( (dx == 0.0) && (dy == 0.0) ) {
51  c = 1.0;
52  s = 0.0;
53  }
54  else if ( fabs(dy) > fabs(dx) ) {
55  su2double tmp = dx/dy;
56  dx = sqrt(1.0 + tmp*tmp);
57  s = Sign(1.0/dx, dy);
58  c = tmp*s;
59  }
60  else if ( fabs(dy) <= fabs(dx) ) {
61  su2double tmp = dy/dx;
62  dy = sqrt(1.0 + tmp*tmp);
63  c = Sign(1.0/dy, dx);
64  s = tmp*c;
65  }
66  else {
67  // dx and/or dy must be invalid
68  dx = 0.0;
69  dy = 0.0;
70  c = 1.0;
71  s = 0.0;
72  }
73  dx = fabs(dx*dy);
74  dy = 0.0;
