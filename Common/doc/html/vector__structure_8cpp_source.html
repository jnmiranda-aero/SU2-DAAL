392 su2double dotProd(const CSysVector & u, const CSysVector & v) {
393 
394  /*--- check for consistent sizes ---*/
395  if (u.nElm != v.nElm) {
396  SU2_MPI::Error("Sizes do not match", CURRENT_FUNCTION);
397  }
398 
399  /*--- find local inner product and, if a parallel run, sum over all
400  processors (we use nElemDomain instead of nElem) ---*/
401  su2double loc_prod = 0.0;
402  for (unsigned long i = 0; i < u.nElmDomain; i++)
403  loc_prod += u.vec_val[i]*v.vec_val[i];
404  su2double prod = 0.0;
405 
406 #ifdef HAVE_MPI
407  SU2_MPI::Allreduce(&loc_prod, &prod, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
408 #else
409  prod = loc_prod;
410 #endif
