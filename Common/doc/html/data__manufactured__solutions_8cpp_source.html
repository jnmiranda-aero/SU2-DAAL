62  + u_y *cos(a_uy *PiLInv*y)

66  + v_y *sin(a_vy *PiLInv*y)

70  + P_y *sin(a_Py *PiLInv*y)
71  + P_xy*sin(a_Pxy*PiL2Inv*x*y);
72 
73  /* Compute the conservative variables from the primitive ones.
74  Note that the implementation below is valid for both 2D and 3D. */
75  sol[0] = rho;
76  sol[1] = rho*u;
77  sol[2] = rho*v;
78  sol[3] = 0.0;
