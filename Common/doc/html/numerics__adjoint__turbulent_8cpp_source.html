423  su2double cv1_3 = cv1*cv1*cv1;
424  su2double k = 0.41;
425  su2double k2 = k*k;

459  dfw_g = glim*cw3_6/(g_6+cw3_6);
460  dg_r = 1.0 + cw2*(6.0*pow(r,5.0)-1.0);
461  dr_nuhat = 1.0/(Shat*k2*dist_0_2);
462  dr_Shat = -dr_nuhat*TurbVar_i[0]/Shat;
463 
464  dShat_nuhat = fv2/(k2*dist_0_2);
465  dShat_fv2 = TurbVar_i[0]/(k2*dist_0_2);
466  dfv2_fv1 = Ji_2*one_o_oneplusJifv1*one_o_oneplusJifv1;

471 
472  Bs = dTs_nuhat; // nu_hat term
473  Bs += dTs_Shat*dShat_nuhat; // S_hat term
474  Bs += dTs_fw*dfw_g*dg_r*(dr_nuhat+dr_Shat*dShat_nuhat); // fw terms
475 
476  val_residual[0] = -Bs*TurbPsi_i[0]*Volume;
477 
478  if (implicit)
479  val_Jacobian_i[0][0] = -Bs*Volume;
480 
481  /*---SECOND PART: \partial_nu_hat mu^k F^{vk} cdot \grad Psi ---*/
482  su2double dEddyVisc_nuhat;
483  if (!config->GetFrozen_Visc_Cont())
484  dEddyVisc_nuhat = U_i[0]*fv1*(1.0 + 3.0*cv1_3/(Ji_3+cv1_3));
485  else
486  dEddyVisc_nuhat = 0;
487 
488  for (iDim = 0; iDim < nDim; iDim++) {
489  for (jDim = 0; jDim < nDim; jDim++)
490  tau[iDim][jDim] = PrimVar_Grad_i[iDim+1][jDim] + PrimVar_Grad_i[jDim+1][iDim];
491  tau[iDim][iDim] -= TWO3*div_vel;
492  }
493 
494  su2double Gas_Constant = config->GetGas_ConstantND();
495  su2double Cp = (Gamma/Gamma_Minus_One)*Gas_Constant;
496  su2double tau_gradphi = 0.0, vel_tau_gradpsi5 = 0.0, gradT_gradpsi5 = 0.0;
497 
498  for (iDim = 0; iDim < nDim; iDim++) {
499  gradT_gradpsi5 += PrimVar_Grad_i[0][iDim]*PsiVar_Grad_i[nVar-1][iDim];
500  for (jDim = 0; jDim < nDim; jDim++) {
501  tau_gradphi += tau[iDim][jDim]*PsiVar_Grad_i[iDim+1][jDim];
502  vel_tau_gradpsi5 += Velocity[iDim]*tau[iDim][jDim]*PsiVar_Grad_i[nVar-1][jDim];
503  }
504  }
505  val_residual[0] += (tau_gradphi + vel_tau_gradpsi5 + Cp/Prandtl_Turb*gradT_gradpsi5)*dEddyVisc_nuhat*Volume;
506 
507  }

520 void CSourceConservative_AdjTurb::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i, su2double **val_Jacobian_j, CConfig *config) {
521 
522  /*--- SOURCE term --> \nabla ( \psi_\mu \B7 E^{s} )
523  E^{s} = 2 c_{b2}/\sigma \nabla \hat{nu} ---*/
524 
525  unsigned short iDim;
526  bool implicit = (config->GetKind_TimeIntScheme_AdjTurb() == EULER_IMPLICIT);
527 
528  su2double cb2 = 0.622;
529  su2double sigma = 2./3.;
530  su2double coeff = 2.0*cb2/sigma;
531  su2double E_ij, proj_TurbVar_Grad_i, proj_TurbVar_Grad_j;
532 
533  E_ij = 0.0; proj_TurbVar_Grad_i = 0.0; proj_TurbVar_Grad_j = 0.0;
534  for (iDim = 0; iDim < nDim; iDim++) {
535  proj_TurbVar_Grad_i += coeff*TurbVar_Grad_i[0][iDim]*Normal[iDim];
536  proj_TurbVar_Grad_j += coeff*TurbVar_Grad_j[0][iDim]*Normal[iDim];
537  E_ij += 0.5*(TurbPsi_i[0]*proj_TurbVar_Grad_i + TurbPsi_j[0]*proj_TurbVar_Grad_j);
538  }
539 
540  val_residual[0] = E_ij;
541 
542  if (implicit) {
543  val_Jacobian_i[0][0] = 0.5*proj_TurbVar_Grad_i;
544  val_Jacobian_j[0][0] = 0.5*proj_TurbVar_Grad_j;
545  }
