585  AD::StartPreacc();
586  AD::SetPreaccIn(Gradient_Primitive, nDim+1, nDim);

607  StrainMag += 2.0*pow(0.5*(Gradient_Primitive[2][2] + Gradient_Primitive[3][1]), 2.0);
608  }
609 
610  StrainMag = sqrt(2.0*StrainMag);
611 
612  AD::SetPreaccOut(StrainMag);

619 void CNSVariable::SetRoe_Dissipation_NTS(su2double val_delta,
620  su2double val_const_DES){
621 
622  static const su2double cnu = pow(0.09, 1.5),
623  ch1 = 3.0,
624  ch2 = 1.0,
625  ch3 = 2.0,
626  sigma_max = 1.0;
627 
628  unsigned short iDim;
629  su2double Omega, Omega_2 = 0, Baux, Gaux, Lturb, Kaux, Aaux;
630 
631  AD::StartPreacc();
632  AD::SetPreaccIn(Vorticity, 3);
633  AD::SetPreaccIn(StrainMag);
634  AD::SetPreaccIn(val_delta);
635  AD::SetPreaccIn(val_const_DES);
636  /*--- Density ---*/
637  AD::SetPreaccIn(Solution[0]);
638  /*--- Laminar viscosity --- */
639  AD::SetPreaccIn(Primitive[nDim+5]);
640  /*--- Eddy viscosity ---*/
641  AD::SetPreaccIn(Primitive[nDim+6]);
642 
643  /*--- Central/upwind blending based on:
644  * Zhixiang Xiao, Jian Liu, Jingbo Huang, and Song Fu. "Numerical
645  * Dissipation Effects on Massive Separation Around Tandem Cylinders",
646  * AIAA Journal, Vol. 50, No. 5 (2012), pp. 1119-1136.
647  * https://doi.org/10.2514/1.J051299
648  * ---*/
649 
650  for (iDim = 0; iDim < 3; iDim++){
651  Omega_2 += Vorticity[iDim]*Vorticity[iDim];
652  }
653  Omega = sqrt(Omega_2);
654 
655  Baux = (ch3 * Omega * max(StrainMag, Omega)) /
656  max((pow(StrainMag,2)+Omega_2)*0.5, 1E-20);
657  Gaux = tanh(pow(Baux,4.0));
658 
659  Kaux = max(sqrt((Omega_2 + pow(StrainMag, 2))*0.5), 0.1 * inv_TimeScale);
660 
661  const su2double nu = GetLaminarViscosity()/GetDensity();
662  const su2double nu_t = GetEddyViscosity()/GetDensity();
663  Lturb = sqrt((nu + nu_t)/(cnu*Kaux));
664

674 void CNSVariable::SetRoe_Dissipation_FD(su2double val_wall_dist){
675 
676  /*--- Constants for Roe Dissipation ---*/
677 
678  static const su2double k2 = pow(0.41,2.0);
679 
680  su2double uijuij = 0;
681  unsigned short iDim, jDim;
682 
683  AD::StartPreacc();
684  AD::SetPreaccIn(Gradient_Primitive, nVar, nDim);
685  AD::SetPreaccIn(val_wall_dist);
686  /*--- Eddy viscosity ---*/
687  AD::SetPreaccIn(Primitive[nDim+5]);
688  /*--- Laminar viscosity --- */
689  AD::SetPreaccIn(Primitive[nDim+6]);
690 
691  for(iDim=0;iDim<nDim;++iDim){
692  for(jDim=0;jDim<nDim;++jDim){
693  uijuij+= Gradient_Primitive[1+iDim][jDim]*Gradient_Primitive[1+iDim][jDim];
694  }
695  }
696 
697  uijuij=sqrt(fabs(uijuij));
698  uijuij=max(uijuij,1e-10);
699 
700  const su2double nu = GetLaminarViscosity()/GetDensity();
701  const su2double nu_t = GetEddyViscosity()/GetDensity();
702  const su2double r_d = (nu + nu_t)/(uijuij*k2*pow(val_wall_dist, 2.0));
703 
704  Roe_Dissipation = 1.0-tanh(pow(8.0*r_d,3.0));
705 
706  AD::SetPreaccOut(Roe_Dissipation);
707  AD::EndPreacc();
708
