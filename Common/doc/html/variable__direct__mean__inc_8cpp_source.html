447  unsigned short iDim;
448 
449  AD::StartPreacc();
450  AD::SetPreaccIn(Gradient_Primitive, nDim+1, nDim);
451 
452  Div = 0.0;
453  for (iDim = 0; iDim < nDim; iDim++) {
454  Div += Gradient_Primitive[iDim+1][iDim];
455  }
456 
457  StrainMag = 0.0;
458 
459  /*--- Add diagonal part ---*/

462  StrainMag += pow(Gradient_Primitive[iDim+1][iDim] - 1.0/3.0*Div, 2.0);
463  }
464 
465  /*--- Add off diagonals ---*/
466 
467  StrainMag += 2.0*pow(0.5*(Gradient_Primitive[1][1] + Gradient_Primitive[2][0]), 2.0);
468 
469  if (nDim == 3) {
470  StrainMag += 2.0*pow(0.5*(Gradient_Primitive[1][2] + Gradient_Primitive[3][0]), 2.0);
471  StrainMag += 2.0*pow(0.5*(Gradient_Primitive[2][2] + Gradient_Primitive[3][1]), 2.0);
472  }
473 
474  StrainMag = sqrt(2.0*StrainMag);
475 
476  AD::SetPreaccOut(StrainMag);
