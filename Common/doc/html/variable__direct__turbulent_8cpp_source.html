113  su2double numerator, trace0, trace1, denominator;
114

117  AD::SetPreaccIn(Vorticity, 3);

121  AD::SetPreaccIn(LaminarViscosity);
122 
123  Strain[0][0] = PrimGrad_Flow[1][0];
124  Strain[1][0] = 0.5*(PrimGrad_Flow[2][0] + PrimGrad_Flow[1][1]);
125  Strain[0][1] = 0.5*(PrimGrad_Flow[1][1] + PrimGrad_Flow[2][0]);
126  Strain[1][1] = PrimGrad_Flow[2][1];
127  if (nDim == 3){
128  Strain[0][2] = 0.5*(PrimGrad_Flow[3][0] + PrimGrad_Flow[1][2]);
129  Strain[1][2] = 0.5*(PrimGrad_Flow[3][1] + PrimGrad_Flow[2][2]);
130  Strain[2][0] = 0.5*(PrimGrad_Flow[1][2] + PrimGrad_Flow[3][0]);
131  Strain[2][1] = 0.5*(PrimGrad_Flow[2][2] + PrimGrad_Flow[3][1]);
132  Strain[2][2] = PrimGrad_Flow[3][2];
133  }
134 
135  Omega = sqrt(Vorticity[0]*Vorticity[0] + Vorticity[1]*Vorticity[1]+ Vorticity[2]*Vorticity[2]);
136 
137  StrainDotVort[0] = Strain[0][0]*Vorticity[0]+Strain[0][1]*Vorticity[1]+Strain[0][2]*Vorticity[2];
138  StrainDotVort[1] = Strain[1][0]*Vorticity[0]+Strain[1][1]*Vorticity[1]+Strain[1][2]*Vorticity[2];
139  StrainDotVort[2] = Strain[2][0]*Vorticity[0]+Strain[2][1]*Vorticity[1]+Strain[2][2]*Vorticity[2];
140 
141  numVecVort[0] = StrainDotVort[1]*Vorticity[2] - StrainDotVort[2]*Vorticity[1];
142  numVecVort[1] = StrainDotVort[2]*Vorticity[0] - StrainDotVort[0]*Vorticity[2];
143  numVecVort[2] = StrainDotVort[0]*Vorticity[1] - StrainDotVort[1]*Vorticity[0];
144 
145  numerator = sqrt(6.0) * sqrt(numVecVort[0]*numVecVort[0] + numVecVort[1]*numVecVort[1] + numVecVort[2]*numVecVort[2]);
146  trace0 = 3.0*(pow(Strain[0][0],2.0) + pow(Strain[1][1],2.0) + pow(Strain[2][2],2.0));
147  trace1 = pow(Strain[0][0] + Strain[1][1] + Strain[2][2],2.0);
148  denominator = pow(Omega, 2.0) * sqrt(trace0-trace1);
149 
150  Vortex_Tilting = (numerator/denominator) * max(1.0,0.2*LaminarViscosity/muT);
151 
152  AD::SetPreaccOut(Vortex_Tilting);
153  AD::EndPreacc();

202 void CTurbSSTVariable::SetBlendingFunc(su2double val_viscosity, su2double val_dist, su2double val_density) {
203  unsigned short iDim;
204  su2double arg2, arg2A, arg2B, arg1;
205 
206  AD::StartPreacc();
207  AD::SetPreaccIn(val_viscosity); AD::SetPreaccIn(val_dist);
208  AD::SetPreaccIn(val_density);
209  AD::SetPreaccIn(Solution, nVar);
210  AD::SetPreaccIn(Gradient, nVar, nDim);
211 
212  /*--- Cross diffusion ---*/
213 
214  CDkw = 0.0;
215  for (iDim = 0; iDim < nDim; iDim++)
216  CDkw += Gradient[0][iDim]*Gradient[1][iDim];
217  CDkw *= 2.0*val_density*sigma_om2/Solution[1];
218  CDkw = max(CDkw, pow(10.0, -20.0));
219 
220  /*--- F1 ---*/
221 
222  arg2A = sqrt(Solution[0])/(beta_star*Solution[1]*val_dist+EPS*EPS);
223  arg2B = 500.0*val_viscosity / (val_density*val_dist*val_dist*Solution[1]+EPS*EPS);
224  arg2 = max(arg2A, arg2B);
225  arg1 = min(arg2, 4.0*val_density*sigma_om2*Solution[0] / (CDkw*val_dist*val_dist+EPS*EPS));
226  F1 = tanh(pow(arg1, 4.0));
227 
228  /*--- F2 ---*/
229 
230  arg2 = max(2.0*arg2A, arg2B);
231  F2 = tanh(pow(arg2, 2.0));
232 
233  AD::SetPreaccOut(F1); AD::SetPreaccOut(F2); AD::SetPreaccOut(CDkw);
234  AD::EndPreacc();
