1044 void CFEM_NeoHookean_Comp::Compute_Plane_Stress_Term(CElement *element, CConfig *config) {
1045

1053 
1054  // Maximum number of iterations and tolerance (relative)
1055  nNR = 10;
1056  NRTOL = 1E-25;

1059  j_red = F_Mat[0][0] * F_Mat[1][1] - F_Mat[1][0] * F_Mat[0][1];
1060  // cte: constant term in the NR method
1061  cte = Lambda*log(j_red) - Mu;
1062 
1063  // f(f33) = mu*f33^2 + lambda*ln(f33) + (lambda*ln(j_red)-mu) = 0
1064  // f'(f33) = 2*mu*f33 + lambda/f33
1065

1067  fx = Mu*pow(xk,2.0) + Lambda*log(xk) + cte;
1068  fpx = 2*Mu*xk + (Lambda / xk);
1069  xkm1 = xk - fx / fpx;
1070  if (((xkm1 - xk) / xk) < NRTOL) break;
1071  xk = xkm1;
