809 void CAdjTurbSolver::ImplicitEuler_Iteration(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
810 
811  unsigned short iVar;
812  unsigned long iPoint, total_index;
813  su2double Delta, Vol;
814 
815  /*--- Set maximum residual to zero ---*/
816 
817  for (iVar = 0; iVar < nVar; iVar++) {
818  SetRes_RMS(iVar, 0.0);
819  SetRes_Max(iVar, 0.0, 0);
820  }
821 
822  /*--- Build implicit system ---*/
823 
824  for (iPoint = 0; iPoint < nPoint; iPoint++) {
825 
826  /*--- Read the volume ---*/
827 
828  Vol = geometry->node[iPoint]->GetVolume();
829 
830  /*--- Modify matrix diagonal to assure diagonal dominance ---*/
831 
832  Delta = Vol / (config->GetCFLRedCoeff_AdjTurb()*solver_container[FLOW_SOL]->node[iPoint]->GetDelta_Time());
833 
834  Jacobian.AddVal2Diag(iPoint, Delta);
835 
836  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
837 
838  for (iVar = 0; iVar < nVar; iVar++) {
839  total_index = iPoint*nVar+iVar;
840  LinSysRes[total_index] = -LinSysRes[total_index];
841  LinSysSol[total_index] = 0.0;
842  AddRes_RMS(iVar, LinSysRes[total_index]*LinSysRes[total_index]);
843  AddRes_Max(iVar, fabs(LinSysRes[total_index]), geometry->node[iPoint]->GetGlobalIndex(), geometry->node[iPoint]->GetCoord());
844  }
845 
846  }
847 
848  /*--- Initialize residual and solution at the ghost points ---*/
849 
850  for (iPoint = nPointDomain; iPoint < nPoint; iPoint++) {
851  for (iVar = 0; iVar < nVar; iVar++) {
852  total_index = iPoint*nVar + iVar;
853  LinSysRes[total_index] = 0.0;
854  LinSysSol[total_index] = 0.0;
855  }
856  }
857 
858  /*--- Solve or smooth the linear system ---*/
859 
860  CSysSolve system;
861  system.Solve(Jacobian, LinSysRes, LinSysSol, geometry, config);
862 
863  /*--- Update solution (system written in terms of increments) ---*/
864 
865  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
866  for (iVar = 0; iVar < nVar; iVar++)
867  node[iPoint]->AddSolution(iVar, LinSysSol[iPoint*nVar+iVar]);
868  }
869 
870  /*--- MPI solution ---*/
871 
872  Set_MPI_Solution(geometry, config);
873 
874  /*--- Compute the root mean square residual ---*/
875 
876  SetResidual_RMS(geometry, config);
877 
878 }
