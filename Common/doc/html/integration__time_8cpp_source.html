984  CNumerics ******numerics_container,
985  CConfig **config,
986  unsigned short RunTime_EqSystem,
987  unsigned long Iteration,
988  unsigned short iZone,
989  unsigned short iInst) {
990 
991  unsigned short iMesh, iStep, iLimit = 1;
992  unsigned short SolContainer_Position = config[iZone]->GetContainerPosition(RunTime_EqSystem);
993  unsigned short FinestMesh = config[iZone]->GetFinestMesh();
994 
995  /*--- For now, we assume no geometric multigrid. ---*/
996  iMesh = FinestMesh;
997 
998  /*--- Check if only the Jacobian of the spatial discretization must
999  be computed. If so, call the appropriate function and return. ---*/
1000  if (config[iZone]->GetJacobian_Spatial_Discretization_Only()) {
1001  solver_container[iZone][iInst][iMesh][SolContainer_Position]->ComputeSpatialJacobian(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
1002  numerics_container[iZone][iInst][iMesh][SolContainer_Position],
1003  config[iZone], iMesh, RunTime_EqSystem);
1004  return;
1005  }
1006 
1007  /*--- Determine the number of stages in the time stepping algorithm.
1008  For the Runge-Kutta schemes this is the number of RK stages,
1009  while for ADER-DG this information is not used, because a more
1010  complicated algorithm must be used to facilitate time accurate
1011  local time stepping. Note that we are currently hard-coding
1012  the classical RK4 scheme. ---*/
1013  bool useADER = false;
1014  switch (config[iZone]->GetKind_TimeIntScheme()) {
1015  case RUNGE_KUTTA_EXPLICIT: iLimit = config[iZone]->GetnRKStep(); break;
1016  case CLASSICAL_RK4_EXPLICIT: iLimit = 4; break;
1017  case ADER_DG: iLimit = 1; useADER = true; break;
1018  case EULER_EXPLICIT: case EULER_IMPLICIT: iLimit = 1; break; }
1019 
1020  /*--- In case an unsteady simulation is carried out, it is possible that a
1021  synchronization time step is specified. If so, set the boolean
1022  TimeSynSpecified to true, which leads to an outer loop in the
1023  algorithm below. ---*/
1024  bool TimeSyncSpecified = false;
1025  const su2double TimeSync = config[iZone]->GetDelta_UnstTimeND();
1026  if(config[iZone]->GetUnsteady_Simulation() == TIME_STEPPING &&
1027  config[iZone]->GetUnst_CFL() != 0.0 &&
1028  TimeSync != 0.0) TimeSyncSpecified = true;
1029 
1030  /*--- Outer loop, which is only active when a synchronization time has been
1031  specified for an unsteady simulation. ---*/
1032  bool syncTimeReached = false;
1033  su2double timeEvolved = 0.0;
1034  while( !syncTimeReached ) {
1035 
1036  /* Compute the time step for stability. */
1037  solver_container[iZone][iInst][iMesh][SolContainer_Position]->SetTime_Step(geometry[iZone][iInst][iMesh],
1038  solver_container[iZone][iInst][iMesh],
1039  config[iZone], iMesh, Iteration);
1040  /* Possibly overrule the specified time step when a synchronization time was
1041  specified and determine whether or not the time loop must be continued.
1042  When TimeSyncSpecified is false, the loop is always terminated. */
1043  if( TimeSyncSpecified )
1044  solver_container[iZone][iInst][iMesh][SolContainer_Position]->CheckTimeSynchronization(config[iZone],
1045  TimeSync, timeEvolved,
1046  syncTimeReached);
1047  else
1048  syncTimeReached = true;
1049 
1050  /*--- For ADER in combination with time accurate local time stepping, the
1051  space and time integration are tightly coupled and cannot be treated
1052  segregatedly. Therefore a different function is called for ADER to
1053  carry out the space and time integration. ---*/
1054  if( useADER ) {
1055  solver_container[iZone][iInst][iMesh][SolContainer_Position]->ADER_SpaceTimeIntegration(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
1056  numerics_container[iZone][iInst][iMesh][SolContainer_Position],
1057  config[iZone], iMesh, RunTime_EqSystem);
1058  }
1059  else {
1060 
1061  /*--- Time and space integration can be decoupled. ---*/
1062  for (iStep = 0; iStep < iLimit; iStep++) {
1063 
1064  /*--- Preprocessing ---*/
1065  solver_container[iZone][iInst][iMesh][SolContainer_Position]->Preprocessing(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
1066  config[iZone], iMesh, iStep, RunTime_EqSystem, false);
1067 
1068  /*--- Space integration ---*/
1069  Space_Integration(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
1070  numerics_container[iZone][iInst][iMesh][SolContainer_Position],
1071  config[iZone], iMesh, iStep, RunTime_EqSystem);
1072 
1073  /*--- Time integration, update solution using the old solution plus the solution increment ---*/
1074  Time_Integration(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
1075  config[iZone], iStep, RunTime_EqSystem, Iteration);
1076 
1077  /*--- Postprocessing ---*/
1078  solver_container[iZone][iInst][iMesh][SolContainer_Position]->Postprocessing(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
1079  config[iZone], iMesh);
1080  }
1081  }
1082  }
1083 
1084  /*--- Calculate the inviscid and viscous forces ---*/
1085  solver_container[iZone][iInst][FinestMesh][SolContainer_Position]->Pressure_Forces(geometry[iZone][iInst][iMesh], config[iZone]);
1086 
1087  solver_container[iZone][iInst][FinestMesh][SolContainer_Position]->Friction_Forces(geometry[iZone][iInst][iMesh], config[iZone]);
1088 
1089  /*--- Convergence strategy ---*/
1090 
1091  //Convergence_Monitoring(geometry[iZone][iInst][FinestMesh], config[iZone], Iteration, monitor, FinestMesh);
1092 }
