36 /* Trapezoidal map implementation. Reference:
37  * M. de Berg, O. Cheong M. van Kreveld, M. Overmars,
38  * Computational Geometry, Algorithms and Applications pp. 121-146 (2008)
39  * NOTE: the current implementation is actually the simpler 'slab' approach.
40  */
41 CTrapezoidalMap::CTrapezoidalMap(const su2double* samples_x, const su2double* samples_y, const unsigned long size,
42  vector<std::array<unsigned long, 2> > const& edges,
43  su2vector<vector<unsigned long> > const& val_edge_to_triangle, bool display) {
44  int rank = SU2_MPI::GetRank();
45  su2double startTime = SU2_MPI::Wtime();
46 
47  edge_to_triangle = su2vector<vector<unsigned long> >(val_edge_to_triangle);
48 
49  unique_bands_x.assign(samples_x, samples_x + size);
50 
51  /* sort x_bands and make them unique */
52  sort(unique_bands_x.begin(), unique_bands_x.end());
53 
54  auto iter = unique(unique_bands_x.begin(), unique_bands_x.end());
55 
56  unique_bands_x.resize(distance(unique_bands_x.begin(), iter));
57 
58  edge_limits_x.resize(edges.size(), 2);
59  edge_limits_y.resize(edges.size(), 2);
60 
61  /* store x and y values of each edge in a vector for a slight speed up
62  * as it prevents some uncoalesced accesses */
63  for (unsigned long j = 0; j < edges.size(); j++) {
64  edge_limits_x[j][0] = samples_x[edges[j][0]];
65  edge_limits_x[j][1] = samples_x[edges[j][1]];
66  edge_limits_y[j][0] = samples_y[edges[j][0]];
67  edge_limits_y[j][1] = samples_y[edges[j][1]];
68  }
69 
70  /* number of bands */
71  unsigned long n_bands_x = unique_bands_x.size() - 1;
72  /* band index */
73  unsigned long i_band = 0;
74  /* number of edges */
75  unsigned long n_edges = edges.size();
76  /* edge index */
77  unsigned long i_edge = 0;
78  unsigned long j_edge = 0;
79  /* counter for edges intersects */
80  unsigned long n_intersects = 0;
81  /* lower and upper x value of each band */
82  su2double band_lower_x = 0;
83  su2double band_upper_x = 0;
84 
85  su2double x_0;
86  su2double y_0;
87  su2double dy_edge;
88  su2double dx_edge;
89  su2double x_band_mid;
90 
91  /* y values of all intersecting edges for every band */
92  y_edge_at_band_mid.resize(unique_bands_x.size() - 1);
93 
94  /* loop over bands */
95  while (i_band < n_bands_x) {
96  band_lower_x = unique_bands_x[i_band];
97  band_upper_x = unique_bands_x[i_band + 1];
98  i_edge = 0;
99  n_intersects = 0;
100 
101  /* loop over edges and determine which edges appear in current band */
102  while (i_edge < n_edges) {
103  /* check if edge intersects the band
104  * (vertical edges are automatically discarded) */
105  if (((edge_limits_x[i_edge][0] <= band_lower_x) and (edge_limits_x[i_edge][1] >= band_upper_x)) or
106  ((edge_limits_x[i_edge][1] <= band_lower_x) and (edge_limits_x[i_edge][0] >= band_upper_x))) {
107  y_edge_at_band_mid[i_band].emplace_back(0.0, 0);
108 
109  x_0 = edge_limits_x[i_edge][0];
110  y_0 = edge_limits_y[i_edge][0];
111 
112  dy_edge = edge_limits_y[i_edge][1] - edge_limits_y[i_edge][0];
113  dx_edge = edge_limits_x[i_edge][1] - edge_limits_x[i_edge][0];
114  x_band_mid = (band_lower_x + band_upper_x) / 2.0;
115 
116  y_edge_at_band_mid[i_band][n_intersects].first = y_0 + dy_edge / dx_edge * (x_band_mid - x_0);
117 
118  /* save edge index so it can later be recalled when searching */
119  y_edge_at_band_mid[i_band][n_intersects].second = i_edge;
120 
121  n_intersects++;
122  }
123  i_edge++;
124  }
125 
126  /* sort edges by their y values.
127  * note that these y values are unique (i.e. edges cannot
128  * intersect in a band) */
129  sort(y_edge_at_band_mid[i_band].begin(), y_edge_at_band_mid[i_band].end());
130 
131  i_band++;
132  }
133 
134  su2double stopTime = SU2_MPI::Wtime();
135 
136  /* calculate size of trapezoidal map components */
137  double size_unique_bands = sizeof(su2double) * unique_bands_x.size() / 1e6;
138  double size_edge_limits_x = sizeof(su2double) * edge_limits_x.size() * 2 / 1e6;
139  double size_edge_limits_y = sizeof(su2double) * edge_limits_y.size() * 2 / 1e6;
140 
141  double size_edge_to_triangle = 0;
142  for (i_edge = 0; i_edge < edge_to_triangle.size(); i_edge++)
143  for (j_edge = 0; j_edge < edge_to_triangle[i_edge].size(); j_edge++)
144  size_edge_to_triangle += sizeof(unsigned long) / 1e6;
145 
146  double size_y_edge_at_band_mid = 0;
147  for (unsigned long i_y = 0; i_y < y_edge_at_band_mid.size(); i_y++)
148  for (unsigned long j_y = 0; j_y < y_edge_at_band_mid[i_y].size(); j_y++)
149  size_y_edge_at_band_mid += sizeof(su2double) / 1e6 + sizeof(unsigned long) / 1e6;
150 
151  memory_footprint =
152  size_unique_bands + size_edge_limits_x + size_edge_limits_y + size_edge_to_triangle + size_y_edge_at_band_mid;
153 
154  /* print size of trapezoidal map components to screen */
155  if ((rank == MASTER_NODE) && display) {
156  cout << setfill(' ');
157  cout << "\n" << endl;
158  cout << "+------------------------------------------------------------------+\n";
159  cout << "| Trapezoidal map info |\n";
160  cout << "+------------------------------------------------------------------+" << endl;
161 
162  cout << "| Time to construct trapezoidal map: " << setw(22) << right << stopTime - startTime << " sec"
163  << " |" << endl;
164  cout << "| Size of unique_bands in memory: " << setw(22) << size_unique_bands << " MB "
165  << " |" << endl;
166  cout << "| Size of edge_limits_x in memory: " << setw(22) << size_edge_limits_x << " MB "
167  << " |" << endl;
168  cout << "| Size of edge_limits_y in memory: " << setw(22) << size_edge_limits_y << " MB "
169  << " |" << endl;
170  cout << "| Size of edge_to_triangle in memory: " << setw(22) << size_edge_to_triangle << " MB "
171  << " |" << endl;
172  cout << "| Size of y_edge_at_band_mid in memory: " << setw(22) << size_y_edge_at_band_mid << " MB "
173  << " |" << endl;
174  cout << "| Total: " << setw(22) << memory_footprint << " MB "
175  << " |" << endl;
176  cout << "+------------------------------------------------------------------+" << endl;
177  cout << "\n" << endl;
178  }
179 }

181 unsigned long CTrapezoidalMap::GetTriangle(const su2double val_x, const su2double val_y) {
182  /* find x band in which val_x sits */
183  pair<unsigned long, unsigned long> band = GetBand(val_x);
184 
185  /* within that band, find edges which enclose the (val_x, val_y) point */
186  pair<unsigned long, unsigned long> edges = GetEdges(band, val_x, val_y);
187 
188  /* identify the adjacent triangles using the two edges */
189  std::array<unsigned long, 2> triangles_edge_low;
190  for (unsigned long i = 0; i < edge_to_triangle[edges.first].size(); i++)
191  triangles_edge_low[i] = edge_to_triangle[edges.first][i];
192 
193  std::array<unsigned long, 2> triangles_edge_up;
194  for (unsigned long i = 0; i < edge_to_triangle[edges.second].size(); i++)
195  triangles_edge_up[i] = edge_to_triangle[edges.second][i];
196 
197  sort(triangles_edge_low.begin(), triangles_edge_low.end());
198  sort(triangles_edge_up.begin(), triangles_edge_up.end());
199 
200  /* The intersection of the faces to which upper or lower belongs is the face that both belong to. */
201  vector<unsigned long> triangle;
202  set_intersection(triangles_edge_up.begin(), triangles_edge_up.end(), triangles_edge_low.begin(),
203  triangles_edge_low.end(), std::back_inserter(triangle));
204 
205  /*--- We failed to find an intersection, so take the lower triangle inside the band enclosing the point---*/
206  if (triangle.size() < 1) {
207  triangle.resize(1, triangles_edge_low[0]);
208  }
209 
210  return triangle[0];
211 }

213 pair<unsigned long, unsigned long> CTrapezoidalMap::GetBand(const su2double val_x) {
214  unsigned long i_low = 0;
215  unsigned long i_up = 0;
216  su2double val_x_sample = val_x;
217  /* check if val_x is in x-bounds of the table, if not then project val_x to either x-min or x-max */
218  if (val_x_sample < unique_bands_x.front()) val_x_sample = unique_bands_x.front();
219  if (val_x_sample > unique_bands_x.back()) val_x_sample = unique_bands_x.back();
220 
221  std::pair<std::vector<su2double>::iterator, std::vector<su2double>::iterator> bounds;
222  bounds = std::equal_range(unique_bands_x.begin(), unique_bands_x.end(), val_x_sample);
223 
224  /*--- if upper bound = 0, then use the range [0,1] ---*/
225  i_up = max<unsigned long>(1, bounds.first - unique_bands_x.begin());
226  i_low = i_up - 1;
227 
228  return make_pair(i_low, i_up);
229 }

231 pair<unsigned long, unsigned long> CTrapezoidalMap::GetEdges(pair<unsigned long, unsigned long> val_band,
232  su2double val_x, su2double val_y) const {
233  su2double next_y;
234  su2double y_edge_low;
235  su2double y_edge_up;
236  su2double x_edge_low;
237  su2double x_edge_up;
238 
239  unsigned long i_band_low = val_band.first;
240 
241  unsigned long next_edge;
242 
243  unsigned long j_low = 0;
244  unsigned long j_mid = 0;
245  unsigned long j_up = 0;
246 
247  j_up = y_edge_at_band_mid[i_band_low].size() - 1;
248  j_low = 0;
249 
250  while (j_up - j_low > 1) {
251  j_mid = (j_up + j_low) / 2;
252 
253  // Select the edge associated with the x band (i_band_low)
254  // Search for the RunEdge in the y direction (second value is index of
255  // edge)
256  next_edge = y_edge_at_band_mid[i_band_low][j_mid].second;
257 
258  y_edge_low = edge_limits_y[next_edge][0];
259  y_edge_up = edge_limits_y[next_edge][1];
260  x_edge_low = edge_limits_x[next_edge][0];
261  x_edge_up = edge_limits_x[next_edge][1];
262 
263  // The search variable in j should be interpolated in i as well
264  next_y = y_edge_low + (y_edge_up - y_edge_low) / (x_edge_up - x_edge_low) * (val_x - x_edge_low);
265 
266  if (next_y > val_y) {
267  j_up = j_mid;
268 
269  } else if (next_y < val_y) {
270  j_low = j_mid;
271 
272  } else if (next_y == val_y) {
273  j_low = j_mid;
274  j_up = j_low + 1;
275  break;
276  }
277  }
278 
279  unsigned long edge_low = y_edge_at_band_mid[i_band_low][j_low].second;
280  unsigned long edge_up = y_edge_at_band_mid[i_band_low][j_up].second;
281 
282  return make_pair(edge_low, edge_up);
283 }
