35 CAdjElasticityOutput::CAdjElasticityOutput(CConfig *config, unsigned short nDim) : COutput(config, nDim, false) {
36 
37  /*--- Initialize number of variables ---*/
38  nVar_FEM = nDim;
39 
40  /*--- Set the default history fields if nothing is set in the config file ---*/
41 
42  if (nRequestedHistoryFields == 0){
43  requestedHistoryFields.emplace_back("ITER");
44  requestedHistoryFields.emplace_back("RESIDUALS");
45  requestedHistoryFields.emplace_back("SENSITIVITY");
46  nRequestedHistoryFields = requestedHistoryFields.size();
47  }
48 
49  if (nRequestedScreenFields == 0){
50  if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
51  requestedScreenFields.emplace_back("INNER_ITER");
52  requestedScreenFields.emplace_back("ADJOINT_DISP_X");
53  requestedScreenFields.emplace_back("ADJOINT_DISP_Y");
54  requestedScreenFields.emplace_back("SENS_E_0");
55  requestedScreenFields.emplace_back("SENS_NU_0");
56  nRequestedScreenFields = requestedScreenFields.size();
57  }
58 
59  if (nRequestedVolumeFields == 0){
60  requestedVolumeFields.emplace_back("COORDINATES");
61  requestedVolumeFields.emplace_back("SOLUTION");
62  requestedVolumeFields.emplace_back("SENSITIVITY");
63  nRequestedVolumeFields = requestedVolumeFields.size();
64  }
65 
66  if (find(requestedVolumeFields.begin(), requestedVolumeFields.end(), "SENSITIVITY") == requestedVolumeFields.end()) {
67  requestedVolumeFields.emplace_back("SENSITIVITY");
68  nRequestedVolumeFields++;
69  }
70 
71  stringstream ss;
72  ss << "Zone " << config->GetiZone() << " (Adj. Structure)";
73  multiZoneHeaderString = ss.str();
74 
75  /*--- Set the volume filename --- */
76 
77  volumeFilename = config->GetAdj_FileName();
78 
79  /*--- Set the surface filename --- */
80 
81  surfaceFilename = config->GetSurfAdjCoeff_FileName();
82 
83  /*--- Set the restart filename --- */
84 
85  restartFilename = config->GetRestart_AdjFileName();
86 
87  /*--- Add the obj. function extension --- */
88 
89  restartFilename = config->GetObjFunc_Extension(restartFilename);
90 
91  /*--- Set the default convergence field --- */
92 
93  if (convFields.empty() ) convFields.emplace_back("ADJOINT_DISP_X");
94 
95 }

99 void CAdjElasticityOutput::SetHistoryOutputFields(CConfig *config){
100 
101  /*--- Residuals ---*/
102  AddHistoryOutput("ADJOINT_DISP_X", "rms[Ux_adj]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint of the X displacements.", HistoryFieldType::RESIDUAL);
103  AddHistoryOutput("ADJOINT_DISP_Y", "rms[Uy_adj]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint of the Y displacements.", HistoryFieldType::RESIDUAL);
104  if (nVar_FEM == 3) {
105  AddHistoryOutput("ADJOINT_DISP_Z", "rms[Uz_adj]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint of the Z displacements.", HistoryFieldType::RESIDUAL);
106  }
107 
108  /*--- Sensitivities ---*/
109  for (auto iVar = 0u; iVar < config->GetnElasticityMat(); iVar++) {
110  const auto iVarS = std::to_string(iVar);
111  AddHistoryOutput("SENS_E_" + iVarS, "Sens[E" + iVarS + ']', ScreenOutputFormat::SCIENTIFIC, "SENSITIVITY", "d Objective / d Elasticity modulus");
112  AddHistoryOutput("SENS_NU_" + iVarS, "Sens[Nu" + iVarS + ']', ScreenOutputFormat::SCIENTIFIC, "SENSITIVITY", "d Objective / d Poisson ratio");
113  if (config->GetTime_Domain() && !config->GetPseudoStatic()) {
114  AddHistoryOutput("SENS_RHO_" + iVarS, "Sens[Rho" + iVarS + ']', ScreenOutputFormat::SCIENTIFIC, "SENSITIVITY", "d Objective / d Material density");
115  }
116  if (config->GetDeadLoad()) {
117  AddHistoryOutput("SENS_RHO_DL_" + iVarS, "Sens[RhoDL" + iVarS + ']', ScreenOutputFormat::SCIENTIFIC, "SENSITIVITY", "d Objective / d Dead load density");
118  }
119  }
120  if (config->GetDE_Effects()) {
121  for (auto iVar = 0u; iVar < config->GetnElectric_Field(); iVar++) {
122  const auto iVarS = std::to_string(iVar);
123  AddHistoryOutput("SENS_EFIELD_" + iVarS, "Sens[EField" + iVarS + ']', ScreenOutputFormat::SCIENTIFIC, "SENSITIVITY", "d Objective / d Electric field");
124  }
125  }
126 
127  AddHistoryOutput("LINSOL_ITER", "LinSolIter", ScreenOutputFormat::INTEGER, "LINSOL", "Number of iterations of the linear solver.");
128  AddHistoryOutput("LINSOL_RESIDUAL", "LinSolRes", ScreenOutputFormat::FIXED, "LINSOL", "Residual of the linear solver.");
129 
130  if (multiZone) {
131  AddHistoryOutput("BGS_ADJ_DISP_X", "bgs[A_Ux]", ScreenOutputFormat::FIXED, "BGS_RES", "BGS residual of the adjoint X displacement.", HistoryFieldType::RESIDUAL);
132  AddHistoryOutput("BGS_ADJ_DISP_Y", "bgs[A_Uy]", ScreenOutputFormat::FIXED, "BGS_RES", "BGS residual of the adjoint Y displacement.", HistoryFieldType::RESIDUAL);
133  if (nVar_FEM == 3) {
134  AddHistoryOutput("BGS_ADJ_DISP_Z", "bgs[A_Uz]", ScreenOutputFormat::FIXED, "BGS_RES", "BGS residual of the adjoint Z displacement.", HistoryFieldType::RESIDUAL);
135  }
136  }
137 }

139 inline void CAdjElasticityOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
140 
141  SetHistoryOutputValue("ADJOINT_DISP_X", log10(solver[ADJFEA_SOL]->GetRes_RMS(0)));
142  SetHistoryOutputValue("ADJOINT_DISP_Y", log10(solver[ADJFEA_SOL]->GetRes_RMS(1)));
143  if (nVar_FEM == 3) {
144  SetHistoryOutputValue("ADJOINT_DISP_Z", log10(solver[ADJFEA_SOL]->GetRes_RMS(2)));
145  }
146 
147  for (unsigned short iVar = 0; iVar < config->GetnElasticityMat(); iVar++) {
148  const auto iVarS = std::to_string(iVar);
149  SetHistoryOutputValue("SENS_E_" + iVarS, solver[ADJFEA_SOL]->GetTotal_Sens_E(iVar));
150  SetHistoryOutputValue("SENS_NU_" + iVarS, solver[ADJFEA_SOL]->GetTotal_Sens_Nu(iVar));
151  if (config->GetTime_Domain() && !config->GetPseudoStatic()) {
152  SetHistoryOutputValue("SENS_RHO_" + iVarS, solver[ADJFEA_SOL]->GetTotal_Sens_Rho(iVar));
153  }
154  if (config->GetDeadLoad()) {
155  SetHistoryOutputValue("SENS_RHO_DL_" + iVarS, solver[ADJFEA_SOL]->GetTotal_Sens_Rho_DL(iVar));
156  }
157  }
158  if (config->GetDE_Effects()) {
159  for (auto iVar = 0u; iVar < config->GetnElectric_Field(); iVar++) {
160  const auto iVarS = std::to_string(iVar);
161  SetHistoryOutputValue("SENS_EFIELD_" + iVarS, solver[ADJFEA_SOL]->GetTotal_Sens_EField(iVar));
162  }
163  }
164 
165  SetHistoryOutputValue("LINSOL_ITER", solver[ADJFEA_SOL]->GetIterLinSolver());
166  SetHistoryOutputValue("LINSOL_RESIDUAL", log10(solver[ADJFEA_SOL]->GetResLinSolver()));
167 
168  if (multiZone) {
169  SetHistoryOutputValue("BGS_ADJ_DISP_X", log10(solver[ADJFEA_SOL]->GetRes_BGS(0)));
170  SetHistoryOutputValue("BGS_ADJ_DISP_Y", log10(solver[ADJFEA_SOL]->GetRes_BGS(1)));
171  if (nVar_FEM == 3){
172  SetHistoryOutputValue("BGS_ADJ_DISP_Z", log10(solver[ADJFEA_SOL]->GetRes_BGS(2)));
173  }
174  }
175 
176  ComputeSimpleCustomOutputs(config);
177 }

179 void CAdjElasticityOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
180 
181  CVariable* Node_Struc = solver[ADJFEA_SOL]->GetNodes();
182  CPoint* Node_Geo = geometry->nodes;
183 
184  SetVolumeOutputValue("COORD-X", iPoint, Node_Geo->GetCoord(iPoint, 0));
185  SetVolumeOutputValue("COORD-Y", iPoint, Node_Geo->GetCoord(iPoint, 1));
186  if (nDim == 3)
187  SetVolumeOutputValue("COORD-Z", iPoint, Node_Geo->GetCoord(iPoint, 2));
188 
189  SetVolumeOutputValue("ADJOINT-X", iPoint, Node_Struc->GetSolution(iPoint, 0));
190  SetVolumeOutputValue("ADJOINT-Y", iPoint, Node_Struc->GetSolution(iPoint, 1));
191  if (nVar_FEM == 3)
192  SetVolumeOutputValue("ADJOINT-Z", iPoint, Node_Struc->GetSolution(iPoint, 2));
193 
194  SetVolumeOutputValue("SENSITIVITY-X", iPoint, Node_Struc->GetSensitivity(iPoint, 0));
195  SetVolumeOutputValue("SENSITIVITY-Y", iPoint, Node_Struc->GetSensitivity(iPoint, 1));
196  if (nDim == 3)
197  SetVolumeOutputValue("SENSITIVITY-Z", iPoint, Node_Struc->GetSensitivity(iPoint, 2));
198 
199  if (!config->GetTime_Domain()) return;
200 
201  SetVolumeOutputValue("SENS_DISP-X", iPoint, Node_Struc->GetSolution_time_n(iPoint, 0));
202  SetVolumeOutputValue("SENS_DISP-Y", iPoint, Node_Struc->GetSolution_time_n(iPoint, 1));
203  if (nDim == 3)
204  SetVolumeOutputValue("SENS_DISP-Z", iPoint, Node_Struc->GetSolution_time_n(iPoint, 2));
205 
206  SetVolumeOutputValue("SENS_VEL-X", iPoint, Node_Struc->GetSolution_time_n(iPoint, nDim));
207  SetVolumeOutputValue("SENS_VEL-Y", iPoint, Node_Struc->GetSolution_time_n(iPoint, nDim + 1));
208  if (nDim == 3)
209  SetVolumeOutputValue("SENS_VEL-Z", iPoint, Node_Struc->GetSolution_time_n(iPoint, 5));
210 
211  SetVolumeOutputValue("SENS_ACCEL-X", iPoint, Node_Struc->GetSolution_time_n(iPoint, 2 * nDim));
212  SetVolumeOutputValue("SENS_ACCEL-Y", iPoint, Node_Struc->GetSolution_time_n(iPoint, 2 * nDim + 1));
213  if (nDim == 3)
214  SetVolumeOutputValue("SENS_ACCEL-Z", iPoint, Node_Struc->GetSolution_time_n(iPoint, 8));
215 }

217 void CAdjElasticityOutput::SetVolumeOutputFields(CConfig *config){
218 
219  // Grid coordinates
220  AddVolumeOutput("COORD-X", "x", "COORDINATES", "x-component of the coordinate vector");
221  AddVolumeOutput("COORD-Y", "y", "COORDINATES", "y-component of the coordinate vector");
222  if (nDim == 3)
223  AddVolumeOutput("COORD-Z", "z", "COORDINATES", "z-component of the coordinate vector");
224 
225  /// BEGIN_GROUP: SOLUTION, DESCRIPTION: Adjoint variables of the current objective function.
226  /// DESCRIPTION: Adjoint x-component.
227  AddVolumeOutput("ADJOINT-X", "Adjoint_x", "SOLUTION", "adjoint of displacement in the x direction");
228  /// DESCRIPTION: Adjoint y-component.
229  AddVolumeOutput("ADJOINT-Y", "Adjoint_y", "SOLUTION", "adjoint of displacement in the y direction");
230  if (nVar_FEM == 3)
231  /// DESCRIPTION: Adjoint z-component.
232  AddVolumeOutput("ADJOINT-Z", "Adjoint_z", "SOLUTION", "adjoint of displacement in the z direction");
233  /// END_GROUP
234 
235  /// BEGIN_GROUP: SENSITIVITY, DESCRIPTION: Geometrical sensitivities of the current objective function.
236  /// DESCRIPTION: Sensitivity x-component.
237  AddVolumeOutput("SENSITIVITY-X", "Sensitivity_x", "SENSITIVITY", "geometric sensitivity in the x direction");
238  /// DESCRIPTION: Sensitivity y-component.
239  AddVolumeOutput("SENSITIVITY-Y", "Sensitivity_y", "SENSITIVITY", "geometric sensitivity in the y direction");
240  if (nDim == 3)
241  /// DESCRIPTION: Sensitivity z-component.
242  AddVolumeOutput("SENSITIVITY-Z", "Sensitivity_z", "SENSITIVITY", "geometric sensitivity in the z direction");
243  /// END_GROUP
244 
245  if (!config->GetTime_Domain()) return;
246 
247  /*--- Sensitivities with respect to initial conditions. ---*/
248 
249  AddVolumeOutput("SENS_DISP-X", "SensitivityDispN_x", "SENSITIVITY_N", "sensitivity to the previous x displacement");
250  AddVolumeOutput("SENS_DISP-Y", "SensitivityDispN_y", "SENSITIVITY_N", "sensitivity to the previous y displacement");
251  if (nDim == 3)
252  AddVolumeOutput("SENS_DISP-Z", "SensitivityDispN_z", "SENSITIVITY_N", "sensitivity to the previous z displacement");
253 
254  AddVolumeOutput("SENS_VEL-X", "SensitivityVelN_x", "SENSITIVITY_N", "sensitivity to the previous x velocity");
255  AddVolumeOutput("SENS_VEL-Y", "SensitivityVelN_y", "SENSITIVITY_N", "sensitivity to the previous y velocity");
256  if (nDim == 3)
257  AddVolumeOutput("SENS_VEL-Z", "SensitivityVelN_z", "SENSITIVITY_N", "sensitivity to the previous z velocity");
258 
259  AddVolumeOutput("SENS_ACCEL-X", "SensitivityAccelN_x", "SENSITIVITY_N", "sensitivity to the previous x acceleration");
260  AddVolumeOutput("SENS_ACCEL-Y", "SensitivityAccelN_y", "SENSITIVITY_N", "sensitivity to the previous y acceleration");
261  if (nDim == 3)
262  AddVolumeOutput("SENS_ACCEL-Z", "SensitivityAccelN_z", "SENSITIVITY_N", "sensitivity to the previous z acceleration");
263 
264 }
