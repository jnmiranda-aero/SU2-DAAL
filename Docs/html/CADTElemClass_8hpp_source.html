87 /*!
88 * \brief Function, which determines the element that contains the given coordinate.
89 * \note This simply forwards the call to the implementation function selecting the right
90 * working variables for the current thread.
91 * \param[in] coor Coordinate which the element must contain.
92 * \param[out] markerID Local marker ID of the element containing the coordinate.
93 * \param[out] elemID Local element ID of the element containing the coordinate.
94 * \param[out] rankID Rank on which element containing the coordinate is stored.
95 * \param[out] parCoor Parametric coordinates of coor inside the element,
96 which contains the coordinate.
97 * \param[out] weightsInterpol Interpolation weights of coor inside the element,
98 which contains the coordinate.
99 * \return True if an element is found, false if not.
100 */
101 inline bool DetermineContainingElement(const su2double* coor, unsigned short& markerID, unsigned long& elemID,
102 int& rankID, su2double* parCoor, su2double* weightsInterpol) {
103 const auto iThread = omp_get_thread_num();
104 return DetermineContainingElement_impl(FrontLeaves[iThread], FrontLeavesNew[iThread], coor, markerID, elemID,
105 rankID, parCoor, weightsInterpol);
106 }
107
108 /*!
109 * \brief Function, which determines the nearest element in the ADT for the given coordinate.
110 * \note This simply forwards the call to the implementation function selecting the right
111 * working variables for the current thread.
112 * \param[in] coor Coordinate for which the nearest element in the ADT must be determined.
113 * \param[out] dist Distance to the nearest element in the ADT.
114 * \param[out] markerID Local marker ID of the nearest element in the ADT.
115 * \param[out] elemID Local element ID of the nearest element in the ADT.
116 * \param[out] rankID Rank on which the nearest element in the ADT is stored.
117 */
118 inline void DetermineNearestElement(const su2double* coor, su2double& dist, unsigned short& markerID,
119 unsigned long& elemID, int& rankID) {
120 const auto iThread = omp_get_thread_num();
121 DetermineNearestElement_impl(BBoxTargets[iThread], FrontLeaves[iThread], FrontLeavesNew[iThread], coor, dist,
122 markerID, elemID, rankID);
123 }
