61 /*!
62 * \brief Applies an isothermal condition to a vertex of a marker.
63 */
64 void IsothermalBoundaryCondition(CGeometry *geometry, CSolver *flow_solver, const CConfig *config,
65 unsigned short iMarker, unsigned long iVertex, const su2double& temperature) {
66
67 const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
68 if (!geometry->nodes->GetDomain(iPoint)) return;
69
70 const bool implicit = config->GetKind_TimeIntScheme() == EULER_IMPLICIT;
71 const su2double prandtl_lam = config->GetPrandtl_Lam();
72 const su2double const_diffusivity = config->GetThermalDiffusivity();
73
74 const auto Point_Normal = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
75
76 const auto* Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
77 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
78
79 const auto* Coord_i = geometry->nodes->GetCoord(iPoint);
80 const auto* Coord_j = geometry->nodes->GetCoord(Point_Normal);
81 const su2double dist_ij = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
82
83 const su2double dTdn = -(nodes->GetTemperature(Point_Normal) - temperature) / dist_ij;
84
85 su2double thermal_diffusivity = const_diffusivity;
86 if (flow) {
87 thermal_diffusivity = flow_solver->GetNodes()->GetLaminarViscosity(iPoint) / prandtl_lam;
88 }
89 LinSysRes(iPoint, 0) -= thermal_diffusivity * dTdn * Area;
90
91 if (implicit) {
92 su2double Jacobian_i[] = {-thermal_diffusivity / dist_ij * Area};
93 Jacobian.SubtractBlock2Diag(iPoint, &Jacobian_i);
94 }
95 }

97 /*!
98 * \brief Compute the viscous flux for the scalar equation at a particular edge.
99 * \param[in] iEdge - Edge for which we want to compute the flux
100 * \param[in] geometry - Geometrical definition of the problem.
101 * \param[in] solver_container - Container vector with all the solutions.
102 * \param[in] numerics - Description of the numerical method.
103 * \param[in] config - Definition of the particular problem.
104 * \note Calls a generic implementation after defining a SolverSpecificNumerics object.
105 */
106 inline void Viscous_Residual(unsigned long iEdge, CGeometry* geometry, CSolver** solver_container,
107 CNumerics* numerics, CConfig* config) override {
108 const CVariable* flow_nodes = flow ? solver_container[FLOW_SOL]->GetNodes() : nullptr;
109
110 const su2double const_diffusivity = config->GetThermalDiffusivity();
111 const su2double pr_lam = config->GetPrandtl_Lam();
112 const su2double pr_turb = config->GetPrandtl_Turb();
113
114 su2double thermal_diffusivity_i{}, thermal_diffusivity_j{};
115
116 /*--- Computes the thermal diffusivity to use in the viscous numerics. ---*/
117 auto compute_thermal_diffusivity = [&](unsigned long iPoint, unsigned long jPoint) {
118 if (flow) {
119 thermal_diffusivity_i = flow_nodes->GetLaminarViscosity(iPoint) / pr_lam +
120 flow_nodes->GetEddyViscosity(iPoint) / pr_turb;
121 thermal_diffusivity_j = flow_nodes->GetLaminarViscosity(jPoint) / pr_lam +
122 flow_nodes->GetEddyViscosity(jPoint) / pr_turb;
123 numerics->SetDiffusionCoeff(&thermal_diffusivity_i, &thermal_diffusivity_j);
124 }
125 else {
126 numerics->SetDiffusionCoeff(&const_diffusivity, &const_diffusivity);
127 }
128 };
129 /*--- Compute residual and Jacobians. ---*/
130 Viscous_Residual_impl(compute_thermal_diffusivity, iEdge, geometry, solver_container, numerics, config);
131 }

333 /*!
334 * \brief Compute objective output
335 * \param[in] config - Definition of the problem.
336 * \param[in] solver - Container vector with all the solutions.
337 */
338 void Evaluate_ObjFunc(const CConfig *config, CSolver**) override {
339 const auto weight = config->GetWeight_ObjFunc(0);
340
341 switch (config->GetKind_ObjFunc()) {
342 case TOTAL_HEATFLUX:
343 Total_ComboObj = weight * Total_HeatFlux;
344 break;
345 case AVG_TEMPERATURE:
346 Total_ComboObj = weight * Total_AverageT;
347 break;
348 case CUSTOM_OBJFUNC:
349 Total_ComboObj = weight * Total_Custom_ObjFunc;
350 break;
351 default:
352 Total_ComboObj = 0.0;
353 }
