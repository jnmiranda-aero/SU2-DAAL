32CNSUnitQuadSolution::CNSUnitQuadSolution(unsigned short val_nDim, unsigned short val_nVar, unsigned short val_iMesh,
33 CConfig* config)
34 : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
35 /*--- Write a message that the solution is initialized for the
36 Navier-Stokes case on a unit quad. Note that heat conduction
37 is neglected for this case. ---*/
38
39 if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
40 cout << endl;
41 cout << "Warning: Fluid properties and solution are being " << endl;
42 cout << " initialized for the Navier Stokes equations " << endl;
43 cout << " on a unit quad. Note that heat conduction " << endl;
44 cout << " is neglected for this case!!!" << endl;
45 cout << endl << flush;
46 }
47
48 /*--- Coefficients, needed to determine the solution. ---*/
49 Gm1 = config->GetGamma() - 1.0;
50 flowAngle = config->GetAoA() * PI_NUMBER / 180.0;
51 Viscosity = config->GetMu_ConstantND();
52
53 /*--- Perform some sanity and error checks for this solution here. ---*/
54 if (config->GetTime_Marching() != TIME_MARCHING::STEADY)
55 SU2_MPI::Error("Steady mode must be selected for the NS Unit Quad case", CURRENT_FUNCTION);
56
57 if (Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES &&
58 Kind_Solver != MAIN_SOLVER::RANS && Kind_Solver != MAIN_SOLVER::FEM_EULER &&
59 Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
60 Kind_Solver != MAIN_SOLVER::FEM_LES)
61 SU2_MPI::Error("Compressible flow equations must be selected for the NS Unit Quad case", CURRENT_FUNCTION);
62
63 if ((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) && (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
64 SU2_MPI::Error("Navier Stokes equations must be selected for the NS Unit Quad case", CURRENT_FUNCTION);
65
66 if (config->GetKind_FluidModel() != IDEAL_GAS)
67 SU2_MPI::Error("Ideal gas must be selected for the NS Unit Quad case", CURRENT_FUNCTION);
68
69 if (fabs(Gm1 - 0.5) > 1.e-8) SU2_MPI::Error("Gamma must be 1.5 for the NS Unit Quad case", CURRENT_FUNCTION);
70
71 if (config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
72 SU2_MPI::Error("Constant viscosity must be selected for the NS Unit Quad case", CURRENT_FUNCTION);
73
74 if (config->GetKind_ConductivityModel() != CONDUCTIVITYMODEL::CONSTANT_PRANDTL)
75 SU2_MPI::Error("Constant Prandtl number must be selected for the NS Unit Quad case", CURRENT_FUNCTION);
76
77 if (config->GetPrandtl_Lam() < 1.e+20)
78 SU2_MPI::Error("Laminar Prandtl number must be larger than 1.e+20 for the NS Unit Quad case", CURRENT_FUNCTION);
79}

89void CNSUnitQuadSolution::GetSolution(const su2double* val_coords, const su2double val_t,
90 su2double* val_solution) const {
91 /*--- Compute the flow direction and the coordinates in
92 the rotated frame. ---*/
93 const su2double cosFlowAngle = cos(flowAngle);
94 const su2double sinFlowAngle = sin(flowAngle);
95
96 const su2double xTilde = val_coords[0] * cosFlowAngle - val_coords[1] * sinFlowAngle;
97 const su2double yTilde = val_coords[0] * sinFlowAngle + val_coords[1] * cosFlowAngle;
98
99 /*--- Compute the exact solution for this case. Note that it works
100 both in 2D and 3D. ---*/
101 val_solution[0] = 1.0;
102 val_solution[1] = cosFlowAngle * yTilde * yTilde;
103 val_solution[2] = -sinFlowAngle * yTilde * yTilde;
104 val_solution[3] = 0.0;
105 val_solution[nVar - 1] = (2.0 * Viscosity * xTilde + 10.0) / Gm1 + 0.5 * yTilde * yTilde * yTilde * yTilde;
106}
