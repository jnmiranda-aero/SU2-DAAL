36 CHeatSolver::CHeatSolver(CGeometry *geometry, CConfig *config, unsigned short iMesh)
37  : CScalarSolver<CHeatVariable>(geometry, config, false),
38  flow(config->GetFluidProblem()), heat_equation(config->GetHeatProblem()) {
39 
40  /*--- Dimension of the problem --> temperature is the only conservative variable ---*/
41 
42  nVar = 1;
43  nPrimVar = 1;
44  nPoint = geometry->GetnPoint();
45  nPointDomain = geometry->GetnPointDomain();
46 
47  /*--- Initialize nVarGrad for deallocation ---*/
48 
49  nVarGrad = nVar;
50 
51  /*--- Define geometry constants in the solver structure ---*/
52 
53  nDim = geometry->GetnDim();
54 
55  /*--- Define some structures for locating max residuals ---*/
56 
57  Residual_RMS.resize(nVar,0.0);
58  Residual_Max.resize(nVar,0.0);
59  Point_Max.resize(nVar,0);
60  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
61 
62  /*--- Initialization of the structure of the whole Jacobian ---*/
63 
64  if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (heat equation) MG level: " << iMesh << "." << endl;
65  Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config, ReducerStrategy);
66  LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
67  LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
68  if (ReducerStrategy) EdgeFluxes.Initialize(geometry->GetnEdge(), geometry->GetnEdge(), nVar, nullptr);
69 
70  if (config->GetExtraOutput()) {
71  if (nDim == 2) { nOutputVariables = 13; }
72  else if (nDim == 3) { nOutputVariables = 19; }
73  OutputVariables.Initialize(nPoint, nPointDomain, nOutputVariables, 0.0);
74  OutputHeadingNames = new string[nOutputVariables];
75  }
76 
77  HeatFlux_per_Marker.resize(nMarker, 0.0);
78  AverageT_per_Marker.resize(nMarker, 0.0);
79  Surface_Areas.resize(config->GetnMarker_HeatFlux(), 0.0);
80 
81  Set_Heatflux_Areas(geometry, config);
82 
83  /*--- Set the reference values for temperature ---*/
84 
85  su2double Temperature_FreeStream = config->GetTemperature_FreeStream();
86  su2double Temperature_Ref = 0.0;
87 
88  if (config->GetRef_Inc_NonDim() == DIMENSIONAL) {
89  Temperature_Ref = 1.0;
90  }
91  else if (config->GetRef_Inc_NonDim() == INITIAL_VALUES) {
92  Temperature_Ref = Temperature_FreeStream;
93  }
94  else if (config->GetRef_Inc_NonDim() == REFERENCE_VALUES) {
95  Temperature_Ref = config->GetInc_Temperature_Ref();
96  }
97  config->SetTemperature_Ref(Temperature_Ref);
98  config->SetTemperature_FreeStreamND(Temperature_FreeStream / Temperature_Ref);
99  Solution_Inf[0] = config->GetTemperature_FreeStreamND();
100 
101  /*--- Set the reference values for heat fluxes. If the heat solver runs stand-alone,
102  * thermal conductivity is read directly from config file ---*/
103 
104  if (heat_equation) {
105  su2double rho_cp = config->GetMaterialDensity(0)*config->GetSpecific_Heat_Cp();
106  config->SetThermalDiffusivity(config->GetThermal_Conductivity_Constant() / rho_cp);
107 
108  /*--- Fluxes are computed via thermal diffusivity (not conductivity), so we have to divide by rho*cp ---*/
109  config->SetHeat_Flux_Ref(rho_cp*Temperature_Ref);
110  }
111  else if (flow) {
112  config->SetHeat_Flux_Ref(config->GetViscosity_Ref()*config->GetSpecific_Heat_Cp());
113  }
114 
115  config->SetDelta_UnstTimeND(config->GetDelta_UnstTime() / config->GetTime_Ref());
116 
117  /*--- Store the value of the temperature and the heat flux density at the boundaries,
118  used for communications with donor cells ---*/
119 
120  const unsigned short nConjVariables = 4;
121  AllocVectorOfMatrices(nVertex, nConjVariables, ConjugateVar, Solution_Inf[0]);
122 
123  /*--- Heat flux in all the markers ---*/
124 
125  AllocVectorOfVectors(nVertex, HeatFlux);
126 
127  if (config->GetMultizone_Problem()){
128  /*--- Initialize the BGS residuals. ---*/
129  Residual_BGS.resize(nVar,1.0);
130  Residual_Max_BGS.resize(nVar,1.0);
131  Point_Max_BGS.resize(nVar,0);
132  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
133  }
134 
135  /*--- Initialize the nodes vector. ---*/
136 
137  nodes = new CHeatVariable(Solution_Inf[0], nPoint, nDim, nVar, config);
138 
139  SetBaseClassPointerToNodes();
140 
141  /*--- Communicate and store volume and the number of neighbors for any dual CVs that lie on on periodic markers. ---*/
142  for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic() / 2; iPeriodic++) {
143  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_VOLUME);
144  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_VOLUME);
145  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_NEIGHBORS);
146  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_NEIGHBORS);
147  }
148  /*--- Store if implicit scheme is used. This has implications on the Residual and Jacobian handling for periodic
149  * boundaries ---*/
150  const bool euler_implicit = (config->GetKind_TimeIntScheme_Heat() == EULER_IMPLICIT);
151  SetImplicitPeriodic(euler_implicit);
152 
153  /*--- MPI solution ---*/
154 
155  InitiateComms(geometry, config, MPI_QUANTITIES::SOLUTION);
156  CompleteComms(geometry, config, MPI_QUANTITIES::SOLUTION);
157 
158  /*--- Store the initial CFL number for all grid points. ---*/
159 
160  const su2double CFL = config->GetCFL(MGLevel) * config->GetCFLRedCoeff_Turb();
161  SU2_OMP_FOR_STAT(omp_chunk_size)
162  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
163  nodes->SetLocalCFL(iPoint, CFL);
164  }
165  END_SU2_OMP_FOR
166  Min_CFL_Local = CFL;
167  Max_CFL_Local = CFL;
168  Avg_CFL_Local = CFL;
169 
170  /*--- Add the solver name. ---*/
171 
172  SolverName = "HEAT";
173 }

192 void CHeatSolver::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter,
193  bool val_update_geo) {
194  const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
195 
196  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
197  /*--- Skip coordinates ---*/
198 
199  unsigned short skipVars = nDim;
200 
201  if (flow) {
202  // P, vx, vy (,vz)
203  skipVars += 1 + nDim;
204  }
205 
206  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
207 
208  if (config->GetRead_Binary_Restart()) {
209  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
210  } else {
211  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
212  }
213 
214  /*--- Load data from the restart into correct containers. ---*/
215  unsigned long counter = 0;
216  for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++ ) {
217 
218  /*--- Retrieve local index. If this node from the restart file lives
219  on the current processor, we will load and instantiate the vars. ---*/
220 
221  const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
222 
223  if (iPoint_Local > -1) {
224  /*--- We need to store this point's data, so jump to the correct
225  offset in the buffer of data from the restart file and load it. ---*/
226 
227  const auto index = counter*Restart_Vars[1] + skipVars;
228  for (auto iVar = 0u; iVar < nVar; iVar++) nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index + iVar]);
229 
230  /*--- Increment the overall counter for how many points have been loaded. ---*/
231  counter++;
232  }
233  }
234 
235  /*--- Detect a wrong solution file ---*/
236 
237  if (counter != nPointDomain) {
238  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" does not match with the mesh file!\n") +
239  string("This can be caused by empty lines at the end of the file."), CURRENT_FUNCTION);
240  }
241  }
242  END_SU2_OMP_SAFE_GLOBAL_ACCESS
243 
244  /*--- Communicate the loaded solution on the fine grid before we transfer
245  it down to the coarse levels. We alo call the preprocessing routine
246  on the fine level in order to have all necessary quantities updated,
247  especially if this is a turbulent simulation (eddy viscosity). ---*/
248 
249  solver[MESH_0][HEAT_SOL]->InitiateComms(geometry[MESH_0], config, MPI_QUANTITIES::SOLUTION);
250  solver[MESH_0][HEAT_SOL]->CompleteComms(geometry[MESH_0], config, MPI_QUANTITIES::SOLUTION);
251 
252  solver[MESH_0][HEAT_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER, RUNTIME_HEAT_SYS, false);
253 
254  /*--- Interpolate the solution down to the coarse multigrid levels ---*/
255 
256  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
257  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][HEAT_SOL]->GetNodes()->GetSolution(),
258  *geometry[iMesh], solver[iMesh][HEAT_SOL]->GetNodes()->GetSolution());
259  solver[iMesh][HEAT_SOL]->InitiateComms(geometry[iMesh], config, MPI_QUANTITIES::SOLUTION);
260  solver[iMesh][HEAT_SOL]->CompleteComms(geometry[iMesh], config, MPI_QUANTITIES::SOLUTION);
261 
262  solver[iMesh][HEAT_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_HEAT_SYS, false);
263  }
264 
265  /*--- Delete the class memory that is used to load the restart. ---*/
266 
267  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
268  Restart_Vars = decltype(Restart_Vars){};
269  Restart_Data = decltype(Restart_Data){};
270  }
271  END_SU2_OMP_SAFE_GLOBAL_ACCESS
272 }

281 void CHeatSolver::Viscous_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
282  CConfig *config, unsigned short iMesh, unsigned short iRKStep) {
283  /*--- For fluid problems the viscous residual is included in the convective residual. ---*/
284  if (flow) return;
285  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
286  CNumerics* numerics = numerics_container[VISC_TERM + omp_get_thread_num() * MAX_TERMS];
287 
288  bool pausePreacc = false;
289  if (ReducerStrategy)
290  pausePreacc = AD::PausePreaccumulation();
291  else
292  AD::StartNoSharedReading();
293 
294  for (auto color : EdgeColoring) {
295  SU2_OMP_FOR_DYN(nextMultiple(OMP_MIN_SIZE, color.groupSize))
296  for (auto k = 0ul; k < color.size; ++k) {
297  auto iEdge = color.indices[k];
298  Viscous_Residual(iEdge, geometry, solver_container, numerics, config);
299  }
300  END_SU2_OMP_FOR
301  }
302 
303  /*--- Restore preaccumulation and adjoint evaluation state. ---*/
304  AD::ResumePreaccumulation(pausePreacc);
305  if (!ReducerStrategy) AD::EndNoSharedReading();
306 
307  if (ReducerStrategy) {
308  SumEdgeFluxes(geometry);
309  if (implicit) Jacobian.SetDiagonalAsColumnSum();
310  }

341 
342  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
343 
344  if(!geometry->nodes->GetDomain(iPoint)) continue;

365  }
366 
367  delete[] Local_Surface_Areas;
368 
369  } END_SU2_OMP_SAFE_GLOBAL_ACCESS
370 }

416 void CHeatSolver::BC_Inlet(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
417  CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
418  if (!flow) return;
419 
420  const bool viscous = config->GetViscous();
421  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
422 
423  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
424  for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
425 
426  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
427 
428  if (!geometry->nodes->GetDomain(iPoint)) continue;
429 
430  su2double Normal[MAXNDIM];
431  geometry->vertex[val_marker][iVertex]->GetNormal(Normal);
432  for (auto iDim = 0u; iDim < nDim; iDim++) Normal[iDim] = -Normal[iDim];
433 
434  /*--- Normal vector for this vertex (negate for outward convention) ---*/
435 
436  conv_numerics->SetNormal(Normal);
437 
438  /*--- Retrieve solution at this boundary node ---*/
439 
440  const auto* V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
441 
442  /*--- Retrieve the specified velocity for the inlet. ---*/
443 
444  const auto* V_inlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
445 
446  conv_numerics->SetPrimitive(V_domain, V_inlet);
447 
448  if (dynamic_grid)
449  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
450 
451  const su2double Temp_i = nodes->GetTemperature(iPoint);
452  const su2double Temp_j = V_inlet[prim_idx.Temperature()];
453  conv_numerics->SetScalarVar(&Temp_i, &Temp_j);
454 
455  /*--- Compute the residual using an upwind scheme ---*/
456 
457  auto residual = conv_numerics->ComputeResidual(config);
458 
459  /*--- Update residual value ---*/
460 
461  LinSysRes.AddBlock(iPoint, residual);
462 
463  /*--- Jacobian contribution for implicit integration ---*/
464 
465  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
466 
467  /*--- Viscous contribution ---*/
468 
469  if (viscous) {
470  IsothermalBoundaryCondition(geometry, solver_container[FLOW_SOL], config, val_marker, iVertex, Temp_j);
471  }
472  }
473  END_SU2_OMP_FOR
474 }

709 void CHeatSolver::SetTime_Step(CGeometry *geometry, CSolver **solver_container, CConfig *config,
710  unsigned short iMesh, unsigned long Iteration) {
711 
712  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
713  const bool time_stepping = (config->GetTime_Marching() == TIME_MARCHING::TIME_STEPPING);
714  const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
715  (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
716  const su2double K_v = 0.25;
717 
718  const su2double prandtl_lam = config->GetPrandtl_Lam();
719  const su2double prandtl_turb = config->GetPrandtl_Turb();
720  const su2double constant_thermal_diffusivity = config->GetThermalDiffusivity();
721 
722  const CVariable* flow_nodes = solver_container[FLOW_SOL] ? solver_container[FLOW_SOL]->GetNodes() : nullptr;
723 
724  /*--- Init thread-shared variables to compute min/max values.
725  * Critical sections are used for this instead of reduction
726  * clauses for compatibility with OpenMP 2.0 (Windows...). ---*/
727  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
728  Min_Delta_Time = 1e30;
729  Max_Delta_Time = 0.0;
730  Global_Delta_UnstTimeND = 1e30;
731  } END_SU2_OMP_SAFE_GLOBAL_ACCESS
732 
733  /*--- Loop domain points. ---*/
734 
735  SU2_OMP_FOR_DYN(omp_chunk_size)
736  for (auto iPoint = 0ul; iPoint < nPointDomain; ++iPoint) {
737 
738  /*--- Set maximum eigenvalues to zero. ---*/
739 
740  nodes->SetMax_Lambda_Visc(iPoint, 0.0);
741 
742  /*--- Loop over the neighbors of point i. ---*/
743 
744  for (unsigned short iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); ++iNeigh) {
745  const auto jPoint = geometry->nodes->GetPoint(iPoint,iNeigh);
746  const auto iEdge = geometry->nodes->GetEdge(iPoint,iNeigh);
747  const auto* Normal = geometry->edges->GetNormal(iEdge);
748  const su2double Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
749 
750  if (flow) {
751  const su2double laminar_viscosity = 0.5 * (flow_nodes->GetLaminarViscosity(iPoint) +
752  flow_nodes->GetLaminarViscosity(jPoint));
753  const su2double eddy_viscosity = 0.5 * (flow_nodes->GetEddyViscosity(iPoint) +
754  flow_nodes->GetEddyViscosity(jPoint));
755 
756  const su2double thermal_diffusivity = laminar_viscosity / prandtl_lam + eddy_viscosity / prandtl_turb;
757  nodes->AddMax_Lambda_Visc(iPoint, thermal_diffusivity * Area2);
758  } else {
759  nodes->AddMax_Lambda_Visc(iPoint, constant_thermal_diffusivity * Area2);
760  }
761  }
762  }
763  END_SU2_OMP_FOR
764 
765  /*--- Loop boundary edges ---*/
766 
767  for (unsigned short iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
768  if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
769  (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
770  (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
771 
772  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
773  for (auto iVertex = 0ul; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
774 
775  /*--- Point identification, Normal vector and area ---*/
776 
777  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
778 
779  if (!geometry->nodes->GetDomain(iPoint)) continue;
780 
781  const auto* Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
782  const su2double Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
783 
784  if (flow) {
785  const su2double thermal_diffusivity = flow_nodes->GetLaminarViscosity(iPoint) / prandtl_lam +
786  flow_nodes->GetEddyViscosity(iPoint) / prandtl_turb;
787  nodes->AddMax_Lambda_Visc(iPoint, thermal_diffusivity * Area2);
788  } else {
789  nodes->AddMax_Lambda_Visc(iPoint, constant_thermal_diffusivity * Area2);
790  }
791  }
792  END_SU2_OMP_FOR
793  }
794  }
795 
796  /*--- Each element uses their own speed, steady state simulation. ---*/
797  {
798  /*--- Thread-local variables for min/max reduction. ---*/
799  su2double minDt = 1e30, maxDt = 0.0;
800 
801  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
802  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
803 
804  su2double local_delta_time = 0.0;
805 
806  const su2double Vol = geometry->nodes->GetVolume(iPoint);
807  if (Vol != 0.0) {
808  local_delta_time = nodes->GetLocalCFL(iPoint) * K_v * pow(Vol, 2) / nodes->GetMax_Lambda_Visc(iPoint);
809  if (flow) {
810  switch (config->GetKind_TimeStep_Heat()) {
811  case BYFLOW:
812  case CONVECTIVE:
813  local_delta_time = flow_nodes->GetDelta_Time(iPoint);
814  break;
815  case MINIMUM:
816  local_delta_time = min(local_delta_time, flow_nodes->GetDelta_Time(iPoint));
817  break;
818  case VISCOUS:
819  break;
820  default:
821  break;
822  }
823  }
824  minDt = min(minDt, local_delta_time);
825  maxDt = max(maxDt, local_delta_time);
826  }
827  nodes->SetDelta_Time(iPoint, local_delta_time);
828  }
829  END_SU2_OMP_FOR
830  /*--- Min/max over threads. ---*/
831  SU2_OMP_CRITICAL
832  {
833  Min_Delta_Time = min(Min_Delta_Time, minDt);
834  Max_Delta_Time = max(Max_Delta_Time, maxDt);
835  Global_Delta_Time = Min_Delta_Time;
836  }
837  END_SU2_OMP_CRITICAL
838  }
839 
840  /*--- Compute the min/max dt (in parallel, now over mpi ranks). ---*/
841 
842  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
843  if (config->GetComm_Level() == COMM_FULL) {
844  su2double rbuf_time;
845  SU2_MPI::Allreduce(&Min_Delta_Time, &rbuf_time, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
846  Min_Delta_Time = rbuf_time;
847 
848  SU2_MPI::Allreduce(&Max_Delta_Time, &rbuf_time, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
849  Max_Delta_Time = rbuf_time;
850  }
851  } END_SU2_OMP_SAFE_GLOBAL_ACCESS
852 
853  /*--- For exact time solution use the minimum delta time of the whole mesh. ---*/
854  if (time_stepping) {
855 
856  /*--- If the unsteady CFL is set to zero, it uses the defined unsteady time step,
857  * otherwise it computes the time step based on the unsteady CFL. ---*/
858 
859  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
860  {
861  if (config->GetUnst_CFL() == 0.0) {
862  Global_Delta_Time = config->GetDelta_UnstTime();
863  }
864  else {
865  Global_Delta_Time = Min_Delta_Time;
866  }
867  Max_Delta_Time = Global_Delta_Time;
868 
869  config->SetDelta_UnstTimeND(Global_Delta_Time);
870  }
871  END_SU2_OMP_SAFE_GLOBAL_ACCESS
872 
873  /*--- Sets the regular CFL equal to the unsteady CFL. ---*/
874 
875  SU2_OMP_FOR_STAT(omp_chunk_size)
876  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
877  nodes->SetLocalCFL(iPoint, config->GetUnst_CFL());
878  nodes->SetDelta_Time(iPoint, Global_Delta_Time);
879  }
880  END_SU2_OMP_FOR
881 
882  }
883 
884  /*--- Recompute the unsteady time step for the dual time strategy if the unsteady CFL is diferent from 0.
885  * This is only done once because in dual time the time step cannot be variable. ---*/
886 
887  if (dual_time && (Iteration == config->GetRestart_Iter()) && (config->GetUnst_CFL() != 0.0) && (iMesh == MESH_0)) {
888 
889  /*--- Thread-local variable for reduction. ---*/
890  su2double glbDtND = 1e30;
891 
892  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
893  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
894  glbDtND = min(glbDtND, config->GetUnst_CFL()*Global_Delta_Time / nodes->GetLocalCFL(iPoint));
895  }
896  END_SU2_OMP_FOR
897  SU2_OMP_CRITICAL
898  Global_Delta_UnstTimeND = min(Global_Delta_UnstTimeND, glbDtND);
899  END_SU2_OMP_CRITICAL
900 
901  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
902  {
903  SU2_MPI::Allreduce(&Global_Delta_UnstTimeND, &glbDtND, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
904  Global_Delta_UnstTimeND = glbDtND;
905 
906  config->SetDelta_UnstTimeND(Global_Delta_UnstTimeND);
907  }
908  END_SU2_OMP_SAFE_GLOBAL_ACCESS
909  }
910 
911  /*--- The pseudo local time (explicit integration) cannot be greater than the physical time ---*/
912 
913  if (dual_time && !implicit) {
914  SU2_OMP_FOR_STAT(omp_chunk_size)
915  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
916  su2double dt = min((2.0/3.0)*config->GetDelta_UnstTimeND(), nodes->GetDelta_Time(iPoint));
917  nodes->SetDelta_Time(iPoint, dt);
918  }
919  END_SU2_OMP_FOR
920  }
921 }

1014  }

1017  if (implicit) {
1018  if (first_order) Jacobian.AddVal2Diag(iPoint, Volume_nP1 / TimeStep);
1019  if (second_order) Jacobian.AddVal2Diag(iPoint, (Volume_nP1 * 3.0) / (2.0 * TimeStep));
