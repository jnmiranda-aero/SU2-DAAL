355 /*!
356 * \brief Set the solution of all solvers (adjoint or primal) in a zone.
357 * \param[in] iZone - Index of the zone.
358 * \param[in] adjoint - True to consider adjoint solvers instead of primal.
359 * \param[in] solution - Solution object with interface (iPoint,iVar).
360 * \tparam Old - If true set "old solutions" instead.
361 */
362 template <class Container, bool Old = false>
363 void SetAllSolutions(unsigned short iZone, bool adjoint, const Container& solution) {
364 const auto nPoint = geometry_container[iZone][INST_0][MESH_0]->GetnPoint();
365 for (auto iSol = 0u, offset = 0u; iSol < MAX_SOLS; ++iSol) {
366 auto solver = solver_container[iZone][INST_0][MESH_0][iSol];
367 if (!(solver && (solver->GetAdjoint() == adjoint))) continue;
368 for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint)
369 for (auto iVar = 0ul; iVar < solver->GetnVar(); ++iVar)
370 if (!Old) {
371 solver->GetNodes()->SetSolution(iPoint, iVar, solution(iPoint, offset + iVar));
372 } else {
373 solver->GetNodes()->SetSolution_Old(iPoint, iVar, solution(iPoint, offset + iVar));
374 }
375 offset += solver->GetnVar();
376 }
377 }
378
379 /*!
380 * \brief Set the "old solution" of all solvers (adjoint or primal) in a zone.
381 */
382 template <class Container>
383 void SetAllSolutionsOld(unsigned short iZone, bool adjoint, const Container& solution) {
384 SetAllSolutions<Container, true>(iZone, adjoint, solution);
385 }
386
387 /*!
388 * \brief Get the solution of all solvers (adjoint or primal) in a zone.
389 * \param[in] iZone - Index of the zone.
390 * \param[in] adjoint - True to consider adjoint solvers instead of primal.
391 * \param[out] solution - Solution object with interface (iPoint,iVar).
392 */
393 template <class Container>
394 void GetAllSolutions(unsigned short iZone, bool adjoint, Container& solution) const {
395 const auto nPoint = geometry_container[iZone][INST_0][MESH_0]->GetnPoint();
396 for (auto iSol = 0u, offset = 0u; iSol < MAX_SOLS; ++iSol) {
397 auto solver = solver_container[iZone][INST_0][MESH_0][iSol];
398 if (!(solver && (solver->GetAdjoint() == adjoint))) continue;
399 const auto& sol = solver->GetNodes()->GetSolution();
400 for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint)
401 for (auto iVar = 0ul; iVar < solver->GetnVar(); ++iVar)
402 solution(iPoint, offset + iVar) = SU2_TYPE::GetValue(sol(iPoint, iVar));
403 offset += solver->GetnVar();
404 }
405 }
406
407 /*!
408 * \brief Sum the number of primal or adjoint variables for all solvers in a given zone.
409 * \param[in] iZone - Index of the zone.
410 * \param[in] adjoint - True to consider adjoint solvers instead of primal.
411 * \return Total number of solution variables.
412 */
413 unsigned short GetTotalNumberOfVariables(unsigned short iZone, bool adjoint) const {
414 unsigned short nVar = 0;
415 for (auto iSol = 0u; iSol < MAX_SOLS; iSol++) {
416 auto solver = solver_container[iZone][INST_0][MESH_0][iSol];
417 if (solver && (solver->GetAdjoint() == adjoint)) nVar += solver->GetnVar();
418 }
419 return nVar;
420 }
