32CNEMONumerics::CNEMONumerics(unsigned short val_nDim, unsigned short val_nVar,
33 unsigned short val_nPrimVar,
34 unsigned short val_nPrimVarGrad,
35 const CConfig* config) :
36 CNumerics(val_nDim, val_nVar, config) {
37
38 nSpecies = nVar - nDim - 2;
39 nPrimVar = val_nPrimVar;
40 nPrimVarGrad = val_nPrimVarGrad;
41
42 RHOS_INDEX = 0;
43 T_INDEX = nSpecies;
44 TVE_INDEX = nSpecies+1;
45 VEL_INDEX = nSpecies+2;
46 P_INDEX = nSpecies+nDim+2;
47 RHO_INDEX = nSpecies+nDim+3;
48 H_INDEX = nSpecies+nDim+4;
49 A_INDEX = nSpecies+nDim+5;
50 RHOCVTR_INDEX = nSpecies+nDim+6;
51 RHOCVVE_INDEX = nSpecies+nDim+7;
52 LAM_VISC_INDEX = nSpecies+nDim+8;
53 EDDY_VISC_INDEX = nSpecies+nDim+9;
54
55 /*--- Read from CConfig ---*/
56 implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
57
58 ionization = config->GetIonization();
59 if (ionization) { nHeavy = nSpecies-1; nEl = 1; }
60 else { nHeavy = nSpecies; nEl = 0; }
61
62 /*--- Instatiate the correct fluid model ---*/
63 switch (config->GetKind_FluidModel()) {
64 case MUTATIONPP:
65 #if defined(HAVE_MPP) && !defined(CODI_REVERSE_TYPE) && !defined(CODI_FORWARD_TYPE)
66 fluidmodel = new CMutationTCLib(config, nDim);
67 #else
68 SU2_MPI::Error(string("Mutation++ has not been configured/compiled. Add 1) '-Denable-mpp=true' to your meson string or 2) '-DHAVE_MPP' to the CXX FLAGS of your configure string, and recompile."),
69 CURRENT_FUNCTION);
70 #endif
71 break;
72 case SU2_NONEQ:
73 fluidmodel = new CSU2TCLib(config, nDim, false);
74 break;
75 }
76}

83void CNEMONumerics::GetInviscidProjFlux(const su2double *val_U,
84 const su2double *val_V,
85 const su2double *val_normal,
86 su2double *val_Proj_Flux) {
87
88 const su2double *rhos;
89
90 /*--- Initialize vectors ---*/
91 for (auto iVar = 0ul; iVar < nVar; iVar++)
92 val_Proj_Flux[iVar] = 0.0;
93
94 /*--- Rename for convienience ---*/
95 const su2double rho = val_V[RHO_INDEX];
96 const su2double u = val_V[VEL_INDEX];
97 const su2double v = val_V[VEL_INDEX+1];
98 const su2double w = val_V[VEL_INDEX+2];
99 const su2double P = val_V[P_INDEX];
100 const su2double H = val_V[H_INDEX];
101 const su2double rhoEve = val_U[nSpecies+nDim+1];
102 rhos = &val_V[RHOS_INDEX];
103
104 if (nDim == 2) {
105
106 /*--- iDim = 0 (x-direction) ---*/
107 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
108 val_Proj_Flux[iSpecies] = (rhos[iSpecies]*u) * val_normal[0];
109 val_Proj_Flux[nSpecies] = (rho*u*u + P) * val_normal[0];
110 val_Proj_Flux[nSpecies+1] = (rho*u*v) * val_normal[0];
111 val_Proj_Flux[nSpecies+2] = (rho*u*H) * val_normal[0];
112 val_Proj_Flux[nSpecies+3] = (rhoEve*u) * val_normal[0];
113
114 /*---- iDim = 1 (y-direction) ---*/
115 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
116 val_Proj_Flux[iSpecies] += (rhos[iSpecies]*v) * val_normal[1];
117 val_Proj_Flux[nSpecies] += (rho*v*u) * val_normal[1];
118 val_Proj_Flux[nSpecies+1] += (rho*v*v + P) * val_normal[1];
119 val_Proj_Flux[nSpecies+2] += (rho*v*H) * val_normal[1];
120 val_Proj_Flux[nSpecies+3] += (rhoEve*v) * val_normal[1];
121 }
122 else {
123
124 /*--- iDim = 0 (x-direction) ---*/
125 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
126 val_Proj_Flux[iSpecies] = (rhos[iSpecies]*u) * val_normal[0];
127 val_Proj_Flux[nSpecies] = (rho*u*u + P) * val_normal[0];
128 val_Proj_Flux[nSpecies+1] = (rho*u*v) * val_normal[0];
129 val_Proj_Flux[nSpecies+2] = (rho*u*w) * val_normal[0];
130 val_Proj_Flux[nSpecies+3] = (rho*u*H) * val_normal[0];
131 val_Proj_Flux[nSpecies+4] = (rhoEve*u) * val_normal[0];
132
133 /*--- iDim = 0 (y-direction) ---*/
134 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
135 val_Proj_Flux[iSpecies] += (rhos[iSpecies]*v) * val_normal[1];
136 val_Proj_Flux[nSpecies] += (rho*v*u) * val_normal[1];
137 val_Proj_Flux[nSpecies+1] += (rho*v*v + P) * val_normal[1];
138 val_Proj_Flux[nSpecies+2] += (rho*v*w) * val_normal[1];
139 val_Proj_Flux[nSpecies+3] += (rho*v*H) * val_normal[1];
140 val_Proj_Flux[nSpecies+4] += (rhoEve*v) * val_normal[1];
141
142 /*--- iDim = 0 (z-direction) ---*/
143 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
144 val_Proj_Flux[iSpecies] += (rhos[iSpecies]*w) * val_normal[2];
145 val_Proj_Flux[nSpecies] += (rho*w*u) * val_normal[2];
146 val_Proj_Flux[nSpecies+1] += (rho*w*v) * val_normal[2];
147 val_Proj_Flux[nSpecies+2] += (rho*w*w + P) * val_normal[2];
148 val_Proj_Flux[nSpecies+3] += (rho*w*H) * val_normal[2];
149 val_Proj_Flux[nSpecies+4] += (rhoEve*w) * val_normal[2];
150 }
151}

153void CNEMONumerics::GetInviscidProjJac(const su2double *val_U, const su2double *val_V,
154 const su2double *val_dPdU, const su2double *val_normal,
155 const su2double val_scale,
156 su2double **val_Proj_Jac_Tensor) {
157
158 const su2double *rhos;
159 rhos = &val_V[RHOS_INDEX];
160
161 /*--- Initialize the Jacobian tensor ---*/
162 for (auto iVar = 0ul; iVar < nVar; iVar++)
163 for (auto jVar = 0ul; jVar < nVar; jVar++)
164 val_Proj_Jac_Tensor[iVar][jVar] = 0.0;
165
166 /*--- Rename for convenience ---*/
167 const su2double rho = val_V[RHO_INDEX];
168 const su2double H = val_V[H_INDEX];
169 const su2double rhoEve = val_U[nSpecies+nDim+1];
170
171 su2double u[MAXNDIM];
172 for (auto iDim = 0ul; iDim < nDim; iDim++)
173 u[iDim] = val_V[VEL_INDEX+iDim];
174
175 /*--- Calculate projected velocity ---*/
176 su2double proj_vel = GeometryToolbox::DotProduct(nDim, u, val_normal);
177
178 /*--- Species density rows ---*/
179 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
180 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++) {
181 val_Proj_Jac_Tensor[iSpecies][jSpecies] += -(rhos[iSpecies]/rho) * proj_vel;
182 }
183 val_Proj_Jac_Tensor[iSpecies][iSpecies] += proj_vel;
184 for (auto iDim = 0ul; iDim < nDim; iDim++) {
185 val_Proj_Jac_Tensor[iSpecies][nSpecies+iDim] += (rhos[iSpecies]/rho) * val_normal[iDim];
186 val_Proj_Jac_Tensor[nSpecies+iDim][iSpecies] += val_dPdU[iSpecies]*val_normal[iDim] - proj_vel*u[iDim];
187 }
188 val_Proj_Jac_Tensor[nSpecies+nDim][iSpecies] += (val_dPdU[iSpecies]-H) * proj_vel;
189 val_Proj_Jac_Tensor[nSpecies+nDim+1][iSpecies] += -proj_vel * rhoEve/rho;
190 }
191
192 /*--- Momentum rows ---*/
193 for (auto iDim = 0ul; iDim < nDim; iDim++) {
194 for (auto jDim = 0ul; jDim < nDim; jDim++) {
195 val_Proj_Jac_Tensor[nSpecies+iDim][nSpecies+jDim] += val_dPdU[nSpecies+jDim]*val_normal[iDim] + u[iDim]*val_normal[jDim];
196 }
197 val_Proj_Jac_Tensor[nSpecies+iDim][nSpecies+iDim] += proj_vel;
198 val_Proj_Jac_Tensor[nSpecies+iDim][nSpecies+nDim] += val_dPdU[nSpecies+nDim]*val_normal[iDim];
199 val_Proj_Jac_Tensor[nSpecies+iDim][nSpecies+nDim+1] += val_dPdU[nSpecies+nDim+1]*val_normal[iDim];
200 }
201
202 /*--- Total energy row ---*/
203 for (auto iDim = 0ul; iDim < nDim; iDim++)
204 val_Proj_Jac_Tensor[nSpecies+nDim][nSpecies+iDim] += val_dPdU[nSpecies+iDim]*proj_vel + H*val_normal[iDim];
205 val_Proj_Jac_Tensor[nSpecies+nDim][nSpecies+nDim] += (1+val_dPdU[nSpecies+nDim])*proj_vel;
206 val_Proj_Jac_Tensor[nSpecies+nDim][nSpecies+nDim+1] += val_dPdU[nSpecies+nDim+1] *proj_vel;
207
208 /*--- Vib.-el. energy row ---*/
209 for (auto iDim = 0ul; iDim < nDim; iDim++)
210 val_Proj_Jac_Tensor[nSpecies+nDim+1][nSpecies+iDim] = rhoEve/rho*val_normal[iDim];
211 val_Proj_Jac_Tensor[nSpecies+nDim+1][nSpecies+nDim+1] = proj_vel;
212
213 for (auto iVar = 0ul; iVar < nVar; iVar++)
214 for (auto jVar = 0ul; jVar < nVar; jVar++)
215 val_Proj_Jac_Tensor[iVar][jVar] = val_scale * val_Proj_Jac_Tensor[iVar][jVar];
216}

218void CNEMONumerics::GetViscousProjFlux(const su2double *val_primvar,
219 const su2double* const* val_gradprimvar,
220 su2double *val_eve,
221 const su2double *val_normal,
222 const su2double *val_diffusioncoeff,
223 su2double val_lam_viscosity,
224 su2double val_eddy_viscosity,
225 su2double val_therm_conductivity,
226 su2double val_therm_conductivity_ve,
227 const CConfig *config) {
228
229 // Requires a slightly non-standard primitive vector:
230 // Assumes - V = [Y1, ... , Yn, T, Tve, ... ]
231 // and gradient GV = [GY1, ... , GYn, GT, GTve, ... ]
232 // rather than the standard V = [r1, ... , rn, T, Tve, ... ]
233
234 su2activematrix Flux_Tensor(nVar,nDim);
235
236 /*--- Initialize ---*/
237 for (auto iVar = 0ul; iVar < nVar; iVar++) {
238 Proj_Flux_Tensor[iVar] = 0.0;
239 for (auto iDim = 0ul; iDim < nDim; iDim++)
240 Flux_Tensor[iVar][iDim] = 0.0;
241 }
242
243 /*--- Rename variables for convenience ---*/
244 const auto& Ds = val_diffusioncoeff;
245 const su2double mu = val_lam_viscosity+val_eddy_viscosity;
246 const su2double ktr = val_therm_conductivity;
247 const su2double kve = val_therm_conductivity_ve;
248 const su2double rho = val_primvar[RHO_INDEX];
249 const su2double T = val_primvar[T_INDEX];
250 const su2double Tve = val_primvar[TVE_INDEX];
251 const auto& V = val_primvar;
252 const auto& GV = val_gradprimvar;
253 const auto& hs = fluidmodel->ComputeSpeciesEnthalpy(T, Tve, val_eve);
254
255 /*--- Pre-compute mixture quantities ---*/ //TODO
256 su2double Vector[MAXNDIM] = {0.0};
257 for (auto iDim = 0ul; iDim < nDim; iDim++) {
258 for (auto iSpecies = 0ul; iSpecies < nHeavy; iSpecies++) {
259 Vector[iDim] += rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][iDim];
260 }
261 }
262
263 /*--- Compute the viscous stress tensor ---*/
264 ComputeStressTensor(nDim,tau,val_gradprimvar+VEL_INDEX, mu);
265
266 /*--- Populate entries in the viscous flux vector ---*/
267 for (auto iDim = 0ul; iDim < nDim; iDim++) {
268
269 /*--- Species diffusion velocity ---*/
270 for (auto iSpecies = 0ul; iSpecies < nHeavy; iSpecies++) {
271 Flux_Tensor[iSpecies][iDim] = rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][iDim]
272 - V[RHOS_INDEX+iSpecies]*Vector[iDim];
273 }
274
275 /*--- Shear-stress/momentum related terms ---*/
276 Flux_Tensor[nSpecies+nDim][iDim] = 0.0;
277 for (auto jDim = 0ul; jDim < nDim; jDim++) {
278 Flux_Tensor[nSpecies+jDim][iDim] = tau[iDim][jDim];
279 Flux_Tensor[nSpecies+nDim][iDim] += tau[iDim][jDim]*val_primvar[VEL_INDEX+jDim];
280 }
281
282 /*--- Diffusion terms ---*/
283 for (auto iSpecies = 0ul; iSpecies < nHeavy; iSpecies++) {
284 Flux_Tensor[nSpecies+nDim][iDim] += Flux_Tensor[iSpecies][iDim] * hs[iSpecies];
285 Flux_Tensor[nSpecies+nDim+1][iDim] += Flux_Tensor[iSpecies][iDim] * val_eve[iSpecies];
286 }
287
288 /*--- Heat transfer terms ---*/
289 Flux_Tensor[nSpecies+nDim][iDim] += ktr*GV[T_INDEX][iDim] + kve*GV[TVE_INDEX][iDim];
290 Flux_Tensor[nSpecies+nDim+1][iDim] += kve*GV[TVE_INDEX][iDim];
291 }
292
293 for (auto iVar = 0ul; iVar < nVar; iVar++) {
294 for (auto iDim = 0ul; iDim < nDim; iDim++) {
295 Proj_Flux_Tensor[iVar] += Flux_Tensor[iVar][iDim]*val_normal[iDim];
296 }
297 }
298}

355void CNEMONumerics::GetPMatrix(const su2double *U, const su2double *V, const su2double *val_dPdU,
356 const su2double *val_normal, const su2double *l, const su2double *m,
357 su2double **val_p_tensor) const {
358
359 /*--- Initialize the P matrix to zero ---*/
360 for (auto iVar = 0ul; iVar < nVar; iVar++)
361 for (auto jVar = 0ul; jVar < nVar; jVar++)
362 val_p_tensor[iVar][jVar] = 0.0;
363
364 /*--- Pre-compute useful quantities ---*/
365 su2double sqvel = 0.0;
366 const su2double rho = V[RHO_INDEX];
367 const su2double eve = U[nSpecies+nDim+1]/rho;
368 su2double vU, vV, vW;
369 vU = vV = vW = 0.0;
370 for (auto iDim = 0ul; iDim < nDim; iDim++) {
371 vU += V[VEL_INDEX+iDim] * val_normal[iDim];
372 vV += V[VEL_INDEX+iDim] * l[iDim];
373 vW += V[VEL_INDEX+iDim] * m[iDim];
374 sqvel += V[VEL_INDEX+iDim] * V[VEL_INDEX+iDim];
375 }
376 const su2double a = V[A_INDEX];
377 const su2double a2 = V[A_INDEX]*V[A_INDEX];
378
379 if(nDim == 2) {
380 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
381 val_p_tensor[iSpecies][iSpecies] += 1.0/a2;
382 val_p_tensor[iSpecies][nSpecies] += 0.0;
383 val_p_tensor[iSpecies][nSpecies+1] += V[RHOS_INDEX+iSpecies] / (2.0*rho*a2);
384 val_p_tensor[iSpecies][nSpecies+2] += V[RHOS_INDEX+iSpecies] / (2.0*rho*a2);
385 val_p_tensor[iSpecies][nSpecies+3] += 0.0;
386
387 val_p_tensor[nSpecies][iSpecies] += V[VEL_INDEX] / a2;
388 val_p_tensor[nSpecies+1][iSpecies] += V[VEL_INDEX+1] / a2;
389 val_p_tensor[nSpecies+2][iSpecies] += (val_dPdU[nSpecies+nDim]*sqvel-val_dPdU[iSpecies]) /
390 (val_dPdU[nSpecies+nDim]*a2);
391 val_p_tensor[nSpecies+3][iSpecies] += 0.0;
392 }
393
394 for (auto iDim = 0ul; iDim < nDim; iDim++){
395 val_p_tensor[nSpecies+iDim][nSpecies] += l[iDim];
396 val_p_tensor[nSpecies+iDim][nSpecies+1] += (V[VEL_INDEX+iDim]+a*val_normal[iDim]) / (2.0*a2);
397 val_p_tensor[nSpecies+iDim][nSpecies+2] += (V[VEL_INDEX+iDim]-a*val_normal[iDim]) / (2.0*a2);
398 val_p_tensor[nSpecies+iDim][nSpecies+3] += 0.0;
399 }
400
401 val_p_tensor[nSpecies+2][nSpecies] += vV;
402 val_p_tensor[nSpecies+2][nSpecies+1] += ((V[H_INDEX])+a*vU) / (2.0*a2);
403 val_p_tensor[nSpecies+2][nSpecies+2] += ((V[H_INDEX])-a*vU) / (2.0*a2);
404 val_p_tensor[nSpecies+2][nSpecies+3] += -val_dPdU[nSpecies+nDim+1] / (val_dPdU[nSpecies+nDim]*a2);
405
406 val_p_tensor[nSpecies+3][nSpecies] += 0.0;
407 val_p_tensor[nSpecies+3][nSpecies+1] += eve / (2.0*a2);
408 val_p_tensor[nSpecies+3][nSpecies+2] += eve / (2.0*a2);
409 val_p_tensor[nSpecies+3][nSpecies+3] += 1.0 / a2;
410
411 } else {
412
413 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
414 val_p_tensor[iSpecies][iSpecies] = 1.0/a2;
415 val_p_tensor[iSpecies][nSpecies] = 0.0;
416 val_p_tensor[iSpecies][nSpecies+1] = 0.0;
417 val_p_tensor[iSpecies][nSpecies+2] = V[RHOS_INDEX+iSpecies] / (2.0*rho*a2);
418 val_p_tensor[iSpecies][nSpecies+3] = V[RHOS_INDEX+iSpecies] / (2.0*rho*a2);
419 val_p_tensor[iSpecies][nSpecies+4] = 0.0;
420
421 val_p_tensor[nSpecies][iSpecies] = V[VEL_INDEX] / a2;
422 val_p_tensor[nSpecies+1][iSpecies] = V[VEL_INDEX+1] / a2;
423 val_p_tensor[nSpecies+2][iSpecies] = V[VEL_INDEX+2] / a2;
424 val_p_tensor[nSpecies+3][iSpecies] = (val_dPdU[nSpecies+3]*sqvel-val_dPdU[iSpecies])
425 / (val_dPdU[nSpecies+3]*a2);
426 val_p_tensor[nSpecies+4][iSpecies] = 0.0;
427 }
428
429 for (auto iDim = 0ul; iDim < nDim; iDim++){
430 val_p_tensor[nSpecies+iDim][nSpecies] = l[iDim];
431 val_p_tensor[nSpecies+iDim][nSpecies+1] = m[iDim];
432 val_p_tensor[nSpecies+iDim][nSpecies+2] = (V[VEL_INDEX+iDim]+a*val_normal[iDim]) / (2.0*a2);
433 val_p_tensor[nSpecies+iDim][nSpecies+3] = (V[VEL_INDEX+iDim]-a*val_normal[iDim]) / (2.0*a2);
434 val_p_tensor[nSpecies+iDim][nSpecies+4] = 0.0;
435 }
436
437 val_p_tensor[nSpecies+3][nSpecies] = vV;
438 val_p_tensor[nSpecies+3][nSpecies+1] = vW;
439 val_p_tensor[nSpecies+3][nSpecies+2] = ((V[H_INDEX])+a*vU) / (2.0*a2);
440 val_p_tensor[nSpecies+3][nSpecies+3] = ((V[H_INDEX])-a*vU) / (2.0*a2);
441 val_p_tensor[nSpecies+3][nSpecies+4] = -val_dPdU[nSpecies+nDim+1] / (val_dPdU[nSpecies+nDim]*a2);
442
443 val_p_tensor[nSpecies+4][nSpecies] = 0.0;
444 val_p_tensor[nSpecies+4][nSpecies+1] = 0.0;
445 val_p_tensor[nSpecies+4][nSpecies+2] = eve / (2.0*a2);
446 val_p_tensor[nSpecies+4][nSpecies+3] = eve / (2.0*a2);
447 val_p_tensor[nSpecies+4][nSpecies+4] = 1.0 / a2;

451void CNEMONumerics::GetPMatrix_inv(const su2double *U, const su2double *V, const su2double *val_dPdU,
452 const su2double *val_normal, const su2double *l, const su2double *m,
453 su2double **val_invp_tensor) const {
454
455 for (auto iVar = 0ul; iVar < nVar; iVar++)
456 for (auto jVar = 0ul; jVar < nVar; jVar++)
457 val_invp_tensor[iVar][jVar] = 0.0;
458
459 /*--- Pre-compute useful quantities ---*/
460 const su2double rho = V[RHO_INDEX];
461 const su2double eve = U[nSpecies+nDim+1]/rho;
462 su2double vU, vV, vW;
463 vU = vV = vW = 0.0;
464 for (auto iDim = 0ul; iDim < nDim; iDim++) {
465 vU += V[VEL_INDEX+iDim] * val_normal[iDim];
466 vV += V[VEL_INDEX+iDim] * l[iDim];
467 vW += V[VEL_INDEX+iDim] * m[iDim];
468 }
469 const su2double a = V[A_INDEX];
470 const su2double a2 = V[A_INDEX]*V[A_INDEX];
471
472 if (nDim == 2) {
473
474 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
475 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++) {
476 val_invp_tensor[iSpecies][jSpecies] += -(V[RHOS_INDEX+iSpecies]/rho) * val_dPdU[jSpecies];
477 }
478 val_invp_tensor[iSpecies][iSpecies] += a2;
479 val_invp_tensor[iSpecies][nSpecies] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX] * (V[RHOS_INDEX+iSpecies]/rho);
480 val_invp_tensor[iSpecies][nSpecies+1] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX+1] * (V[RHOS_INDEX+iSpecies]/rho);
481 val_invp_tensor[iSpecies][nSpecies+2] += -val_dPdU[nSpecies+nDim] * (V[RHOS_INDEX+iSpecies]/rho);
482 val_invp_tensor[iSpecies][nSpecies+3] += -val_dPdU[nSpecies+nDim+1] * (V[RHOS_INDEX+iSpecies]/rho);
483
484 val_invp_tensor[nSpecies][iSpecies] += -vV;
485 val_invp_tensor[nSpecies+1][iSpecies] += val_dPdU[iSpecies] - vU*a;
486 val_invp_tensor[nSpecies+2][iSpecies] += val_dPdU[iSpecies] + vU*a;
487 val_invp_tensor[nSpecies+3][iSpecies] += -eve * val_dPdU[iSpecies];
488 }
489
490 val_invp_tensor[nSpecies][nSpecies] += l[0];
491 val_invp_tensor[nSpecies][nSpecies+1] += l[1];
492 val_invp_tensor[nSpecies][nSpecies+2] += 0.0;
493 val_invp_tensor[nSpecies][nSpecies+3] += 0.0;
494
495 val_invp_tensor[nSpecies+1][nSpecies] += a*val_normal[0] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX];
496 val_invp_tensor[nSpecies+1][nSpecies+1] += a*val_normal[1] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX+1];
497 val_invp_tensor[nSpecies+1][nSpecies+2] += val_dPdU[nSpecies+nDim];
498 val_invp_tensor[nSpecies+1][nSpecies+3] += val_dPdU[nSpecies+nDim+1];
499
500 val_invp_tensor[nSpecies+2][nSpecies] += -a*val_normal[0] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX];
501 val_invp_tensor[nSpecies+2][nSpecies+1] += -a*val_normal[1] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX+1];
502 val_invp_tensor[nSpecies+2][nSpecies+2] += val_dPdU[nSpecies+nDim];
503 val_invp_tensor[nSpecies+2][nSpecies+3] += val_dPdU[nSpecies+nDim+1];
504
505 val_invp_tensor[nSpecies+3][nSpecies] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX] * eve;
506 val_invp_tensor[nSpecies+3][nSpecies+1] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX+1] * eve;
507 val_invp_tensor[nSpecies+3][nSpecies+2] += -val_dPdU[nSpecies+nDim] * eve;
508 val_invp_tensor[nSpecies+3][nSpecies+3] += a2 - val_dPdU[nSpecies+nDim+1]*eve;
509
510 } else {
511
512 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
513 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++) {
514 val_invp_tensor[iSpecies][jSpecies] += -(V[RHOS_INDEX+iSpecies]/rho) * val_dPdU[jSpecies];
515 }
516 val_invp_tensor[iSpecies][iSpecies] += a2;
517 val_invp_tensor[iSpecies][nSpecies] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX] * (V[RHOS_INDEX+iSpecies]/rho);
518 val_invp_tensor[iSpecies][nSpecies+1] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX+1] * (V[RHOS_INDEX+iSpecies]/rho);
519 val_invp_tensor[iSpecies][nSpecies+2] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX+2] * (V[RHOS_INDEX+iSpecies]/rho);
520 val_invp_tensor[iSpecies][nSpecies+3] += -val_dPdU[nSpecies+nDim] * (V[RHOS_INDEX+iSpecies]/rho);
521 val_invp_tensor[iSpecies][nSpecies+4] += -val_dPdU[nSpecies+nDim+1] * (V[RHOS_INDEX+iSpecies]/rho);
522
523 val_invp_tensor[nSpecies][iSpecies] += -vV;
524 val_invp_tensor[nSpecies+1][iSpecies] += -vW;
525 val_invp_tensor[nSpecies+2][iSpecies] += val_dPdU[iSpecies] - vU*a;
526 val_invp_tensor[nSpecies+3][iSpecies] += val_dPdU[iSpecies] + vU*a;
527 val_invp_tensor[nSpecies+4][iSpecies] += -eve * val_dPdU[iSpecies];
528 }
529
530 val_invp_tensor[nSpecies][nSpecies] += l[0];
531 val_invp_tensor[nSpecies][nSpecies+1] += l[1];
532 val_invp_tensor[nSpecies][nSpecies+2] += l[2];
533 val_invp_tensor[nSpecies][nSpecies+3] += 0.0;
534 val_invp_tensor[nSpecies][nSpecies+4] += 0.0;
535
536 val_invp_tensor[nSpecies+1][nSpecies] += m[0];
537 val_invp_tensor[nSpecies+1][nSpecies+1] += m[1];
538 val_invp_tensor[nSpecies+1][nSpecies+2] += m[2];
539 val_invp_tensor[nSpecies+1][nSpecies+3] += 0.0;
540 val_invp_tensor[nSpecies+1][nSpecies+4] += 0.0;
541
542 val_invp_tensor[nSpecies+2][nSpecies] += a*val_normal[0] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX];
543 val_invp_tensor[nSpecies+2][nSpecies+1] += a*val_normal[1] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX+1];
544 val_invp_tensor[nSpecies+2][nSpecies+2] += a*val_normal[2] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX+2];
545 val_invp_tensor[nSpecies+2][nSpecies+3] += val_dPdU[nSpecies+nDim];
546 val_invp_tensor[nSpecies+2][nSpecies+4] += val_dPdU[nSpecies+nDim+1];
547
548 val_invp_tensor[nSpecies+3][nSpecies] += -a*val_normal[0] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX];
549 val_invp_tensor[nSpecies+3][nSpecies+1] += -a*val_normal[1] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX+1];
550 val_invp_tensor[nSpecies+3][nSpecies+2] += -a*val_normal[2] - val_dPdU[nSpecies+nDim]*V[VEL_INDEX+2];
551 val_invp_tensor[nSpecies+3][nSpecies+3] += val_dPdU[nSpecies+nDim];
552 val_invp_tensor[nSpecies+3][nSpecies+4] += val_dPdU[nSpecies+nDim+1];
553
554 val_invp_tensor[nSpecies+4][nSpecies] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX] * eve;
555 val_invp_tensor[nSpecies+4][nSpecies+1] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX+1] * eve;
556 val_invp_tensor[nSpecies+4][nSpecies+2] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX+2] * eve;
557 val_invp_tensor[nSpecies+4][nSpecies+3] += -val_dPdU[nSpecies+nDim] * eve;
558 val_invp_tensor[nSpecies+4][nSpecies+4] += a2 - val_dPdU[nSpecies+nDim+1]*eve;
559 }
