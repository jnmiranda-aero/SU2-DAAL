69CNumerics::ResidualType<> CSource_NEMO::ComputeChemistry(const CConfig *config) {
70
71 /*--- Nonequilibrium chemistry ---*/
72 vector<su2double> rhos;
73 rhos.resize(nSpecies,0.0);
74
75 /*--- Initialize residual and Jacobian arrays ---*/
76 for (auto iVar = 0ul; iVar < nVar; iVar++)
77 residual[iVar] = 0.0;
78
79 if (implicit)
80 for (auto iVar = 0ul; iVar < nVar; iVar++)
81 for (auto jVar = 0ul; jVar < nVar; jVar++)
82 jacobian[iVar][jVar] = 0.0;
83
84 /*--- Rename for convenience ---*/
85 su2double T = V_i[T_INDEX];
86 su2double Tve = V_i[TVE_INDEX];
87 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
88 rhos[iSpecies]=V_i[RHOS_INDEX+iSpecies];
89
90 /*--- Set mixture state ---*/
91 fluidmodel->SetTDStateRhosTTv(rhos, T, Tve);
92
93 /*---Compute Prodcution/destruction terms ---*/
94 const auto& ws = fluidmodel->ComputeNetProductionRates(implicit, V_i, eve_i, Cvve_i,
95 dTdU_i, dTvedU_i, jacobian);
96
97 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++){
98 residual[iSpecies] = ws[iSpecies] * Volume;}
99
100 if (implicit) {
101 for (auto iVar = 0ul; iVar<nVar; iVar++) {
102 for (auto jVar = 0ul; jVar<nVar; jVar++) {
103 jacobian[iVar][jVar] *= Volume;
104 }
105 }
106 }
107
108 return ResidualType<>(residual, jacobian, nullptr);

112CNumerics::ResidualType<> CSource_NEMO::ComputeVibRelaxation(const CConfig *config) {
113
114 /*--- Trans.-rot. & vibrational energy exchange via inelastic collisions ---*/
115 // Note: Electronic energy not implemented
116 // Note: Landau-Teller formulation
117 // Note: Millikan & White relaxation time (requires P in Atm.)
118 // Note: Park limiting cross section
119 const su2double res_min = -1E6;
120 const su2double res_max = 1E6;
121
122 vector<su2double> rhos;
123 rhos.resize(nSpecies,0.0);
124
125 /*--- Initialize residual and Jacobian arrays ---*/
126 for (auto iVar = 0ul; iVar < nVar; iVar++) {
127 residual[iVar] = 0.0;
128 }
129 if (implicit) {
130 for (auto iVar = 0ul; iVar < nVar; iVar++)
131 for (auto jVar = 0ul; jVar < nVar; jVar++)
132 jacobian[iVar][jVar] = 0.0;
133 }
134
135 /*--- Rename for convenience ---*/
136 const su2double T = V_i[T_INDEX];
137 const su2double Tve = V_i[TVE_INDEX];
138 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
139 rhos[iSpecies]=V_i[RHOS_INDEX+iSpecies];
140
141 /*--- Set fluid state ---*/
142 fluidmodel->SetTDStateRhosTTv(rhos, T, Tve);
143
144 /*--- Compute residual and jacobians ---*/
145 const su2double VTterm = fluidmodel -> ComputeEveSourceTerm();
146 if (implicit) {
147 fluidmodel->GetEveSourceTermJacobian(V_i, eve_i, Cvve_i, dTdU_i,
148 dTvedU_i, jacobian);
149 }
150
151 residual[nSpecies+nDim+1] = VTterm * Volume;
152
153 if (implicit) {
154 for (auto iVar = 0ul; iVar<nVar; iVar++) {
155 for (auto jVar = 0ul; jVar<nVar; jVar++) {
156 jacobian[iVar][jVar] *= Volume;
157 }
158 }
159 }
160
161 /*--- Relax/limit vt transfer ---*/
162 if (config->GetVTTransferResidualLimiting()) {
163 if (residual[nSpecies+nDim+1]>res_max) residual[nSpecies+nDim+1]=res_max;
164 if (residual[nSpecies+nDim+1]<res_min) residual[nSpecies+nDim+1]=res_min;
165 }
166
167 return ResidualType<>(residual, jacobian, nullptr);
168}

170CNumerics::ResidualType<> CSource_NEMO::ComputeAxisymmetric(const CConfig *config) {
171
172 /*--- Rename for convenience ---*/
173 const auto Ds = Diffusion_Coeff_i;
174 const auto GV = PrimVar_Grad_i;
175 const su2double ktr = Thermal_Conductivity_i;
176 const su2double kve = Thermal_Conductivity_ve_i;
177 const su2double rho = V_i[RHO_INDEX];
178 const su2double RuSI = UNIVERSAL_GAS_CONSTANT;
179 const su2double Ru = 1000.0*RuSI;
180 const su2double rhou = U_i[nSpecies];
181 const su2double rhov = U_i[nSpecies+1];
182 const su2double H = V_i[H_INDEX];
183 const su2double rhoEve = U_i[nVar-1];
184 const auto& Ms = fluidmodel->GetSpeciesMolarMass();
185 const auto& hs = fluidmodel->ComputeSpeciesEnthalpy(V_i[T_INDEX], V_i[TVE_INDEX], eve_i );
186
187 const bool viscous = config->GetViscous();
188 const bool rans = (config->GetKind_Turb_Model() != TURB_MODEL::NONE);
189
190 /*--- Initialize residual and Jacobian arrays ---*/
191 for (auto iVar = 0ul; iVar < nVar; iVar++) residual[iVar] = 0.0;
192
193 /*--- Calculate inverse of y coordinate ---*/
194 su2double yinv = 0.0;
195 if (Coord_i[1]!= 0.0) yinv = 1.0/Coord_i[1];
196 else yinv = 0.0;
197
198 /*--- Rename mass flux for convenience ---*/
199 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
200 Y[iSpecies] = V_i[RHOS_INDEX+iSpecies] / rho;
201
202 /*--- Compute residual for inviscid axisym flow---*/
203 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
204 residual[iSpecies] = yinv*rhov*Y[iSpecies]*Volume;
205 residual[nSpecies] = yinv*rhov*rhou/rho*Volume;
206 residual[nSpecies+1] = yinv*rhov*rhov/rho*Volume;
207 residual[nSpecies+2] = yinv*rhov*H*Volume;
208 residual[nSpecies+3] = yinv*rhov*rhoEve/rho*Volume;
209
210 /*---Compute Jacobian for inviscid axisym flow ---*/
211 if (implicit) {
212
213 /*--- Initialize matrices ---*/
214 for (auto iVar = 0ul; iVar < nVar; iVar++)
215 for (auto jVar = 0ul; jVar < nVar; jVar++)
216 jacobian[iVar][jVar] = 0.0;
217
218 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
219 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++)
220 dYdr[iSpecies][jSpecies] = 0.0;
221
222 /*--- Calculate additional quantities ---*/
223 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
224 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++) {
225 dYdr[iSpecies][jSpecies] += -1/rho*Y[iSpecies];
226 }
227 dYdr[iSpecies][iSpecies] += 1/rho;
228 }
229
230 /*--- Populate Jacobian ---*/
231
232 // Species density
233 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
234 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++) {
235 jacobian[iSpecies][jSpecies] = dYdr[iSpecies][jSpecies]*rhov;
236 }
237 jacobian[iSpecies][nSpecies+1] = Y[iSpecies];
238 }
239
240 // X-momentum
241 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
242 jacobian[nSpecies][iSpecies] = -rhou*rhov/(rho*rho);
243 jacobian[nSpecies][nSpecies] = rhov/rho;
244 jacobian[nSpecies][nSpecies+1] = rhou/rho;
245
246 // Y-momentum
247 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
248 jacobian[nSpecies+1][iSpecies] = -rhov*rhov/(rho*rho);
249 jacobian[nSpecies+1][nSpecies+1] = 2*rhov/rho;
250
251 // Energy
252 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
253 jacobian[nSpecies+nDim][iSpecies] = -H*rhov/rho + dPdU_i[iSpecies]*rhov/rho;
254 jacobian[nSpecies+nDim][nSpecies] = dPdU_i[nSpecies]*rhov/rho;
255 jacobian[nSpecies+nDim][nSpecies+1] = H + dPdU_i[nSpecies+1]*rhov/rho;
256 jacobian[nSpecies+nDim][nSpecies+nDim] = (1+dPdU_i[nSpecies+nDim])*rhov/rho;
257 jacobian[nSpecies+nDim][nSpecies+nDim+1] = dPdU_i[nSpecies+nDim+1]*rhov/rho;
258
259 // Vib-el energy
260 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
261 jacobian[nSpecies+nDim+1][iSpecies] = -rhoEve*rhov/(rho*rho);
262 jacobian[nSpecies+nDim+1][nSpecies+1] = rhoEve/rho;
263 jacobian[nSpecies+nDim+1][nSpecies+nDim+1] = rhov/rho;
264
265 for (auto iVar = 0ul; iVar < nVar; iVar++)
266 for (auto jVar = 0ul; jVar < nVar; jVar++)
267 jacobian[iVar][jVar] *= yinv*Volume;
268 }
269
270 /*--- Compute residual for viscous portion of axisym flow ---*/
271 if (viscous) {
272 if (!rans){ turb_ke_i = 0.0; }
273
274 su2double Vector = 0.0;
275 for (auto iSpecies = 0ul; iSpecies < nHeavy; iSpecies++)
276 Vector += rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][1];
277
278 su2double Mass = 0.0;
279 for (auto iSpecies=0ul; iSpecies<nSpecies; iSpecies++)
280 Mass += V_i[iSpecies]/rho*Ms[iSpecies];
281
282 const su2double heat_capacity_cp_i = V_i[RHOCVTR_INDEX]/rho + Ru/Mass;
283 const su2double total_viscosity_i = Laminar_Viscosity_i + Eddy_Viscosity_i;
284 const su2double total_conductivity_i = ktr + kve + heat_capacity_cp_i*Eddy_Viscosity_i/Prandtl_Turb;
285 const su2double u = V_i[VEL_INDEX];
286 const su2double v = V_i[VEL_INDEX+1];
287 const su2double qy_t = -total_conductivity_i*GV[T_INDEX][1];
288 const su2double qy_ve = -kve*GV[TVE_INDEX][1];
289
290 /*--- Enthalpy and vib-el energy transport due to y-direction diffusion---*/
291 su2double sumJhs_y, sumJeve_y;
292 sumJhs_y = sumJeve_y = 0.0;
293 for (auto iSpecies = 0ul; iSpecies < nHeavy; iSpecies++) {
294 sumJhs_y += -(rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][1] - V_i[RHOS_INDEX+iSpecies]*Vector) * hs[iSpecies];
295 sumJeve_y += -(rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][1] - V_i[RHOS_INDEX+iSpecies]*Vector) * eve_i[iSpecies];
296 }
297
298 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
299 residual[iSpecies] -= 0.0;
300 residual[nSpecies] -= Volume*(yinv*total_viscosity_i*(GV[nSpecies+2][1]+GV[nSpecies+3][0])
301 -TWO3*AuxVar_Grad_i[0][0]);
302 residual[nSpecies+1] -= Volume*(yinv*total_viscosity_i*2*(GV[nSpecies+3][1]-v*yinv)
303 -TWO3*AuxVar_Grad_i[0][1]);
304 residual[nSpecies+2] -= Volume*(yinv*(-sumJhs_y + total_viscosity_i*(u*(GV[nSpecies+3][0]+GV[nSpecies+2][1])
305 +v*TWO3*(2*GV[nSpecies+3][1]-GV[nSpecies+2][0]
306 -v*yinv+rho*turb_ke_i))-qy_t)
307 -TWO3*(AuxVar_Grad_i[1][1]+AuxVar_Grad_i[2][0]));
308 residual[nSpecies+3] -= Volume*(yinv*(-sumJeve_y -qy_ve));
309 }
310
311 return ResidualType<>(residual, jacobian, nullptr);
312}
