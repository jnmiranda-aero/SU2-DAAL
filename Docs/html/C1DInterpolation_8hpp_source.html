35 /*!
36  * \brief Base class for 1D interpolation.
37  * \ingroup LookUpInterp
38  */
39 class C1DInterpolation {
40  protected:
41  std::vector<su2double> x, y; /*!< \brief Data points. */
42 
43  /*!
44  * \brief Find containing interval.
45  * \param[in] xi - Point for which interpolation is required.
46  * \return The start index of the interval, or the size if the coordinate is out of bounds.
47  */
48  inline size_t lower_bound(su2double xi) const {
49  if (xi <= x.front() || xi >= x.back()) return x.size();
50 
51  size_t lb = 0, ub = x.size() - 1;
52 
53  while (ub - lb > 1) {
54  size_t mid = (lb + ub) / 2;
55  auto& change = (xi < x[mid]) ? ub : lb;
56  change = mid;
57  }
58  return lb;
59  }
60 
61  public:
62  /*!
63  * \brief Virtual destructor of the C1DInterpolation class.
64  */
65  virtual ~C1DInterpolation() = default;
66 
67  /*!
68  * \brief Virtual method for setting the coefficients of the respective spline.
69  * \param[in] X - the x values.
70  * \param[in] Data - the f(x) values.
71  */
72  virtual void SetSpline(const std::vector<su2double>& X, const std::vector<su2double>& Data) {
73  assert(X.size() == Data.size());
74  x = X;
75  y = Data;
76  }
77 
78  /*!
79  * \brief Evaluate the value of the spline at a point.
80  */
81  virtual su2double EvaluateSpline(su2double Point_Interp) const = 0;
82  inline su2double operator()(su2double Point_Interp) const { return EvaluateSpline(Point_Interp); }
83 };
