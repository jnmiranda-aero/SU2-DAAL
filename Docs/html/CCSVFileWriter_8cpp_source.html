37 void CCSVFileWriter::WriteData(string val_filename){
38 
39  /*--- Routine to write the surface CSV files (ASCII). We
40  assume here that, as an ASCII file, it is safer to merge the
41  surface data onto the master rank for writing for 2 reasons:
42  (a) as a surface file, the amount of data should be much less
43  than the volume solution, and (b) writing ASCII files in parallel
44  requires serializing the IO calls with barriers, which ruins
45  the performance at moderate to high rank counts. ---*/
46 
47  unsigned short iVar;
48 
49  int iProcessor, nProcessor = size;
50 
51  unsigned long iPoint, index;
52  unsigned long Buffer_Send_nVertex[1], *Buffer_Recv_nVertex = nullptr;
53  unsigned long nLocalVertex_Surface = 0, MaxLocalVertex_Surface = 0;
54 
55  const vector<string> fieldNames = dataSorter->GetFieldNames();
56 
57  ofstream Surf_file;
58  Surf_file.precision(15);
59 
60  /*--- Find the max number of surface vertices among all
61  partitions so we can set up buffers. The master node will handle
62  the writing of the CSV file after gathering all of the data. ---*/
63 
64  nLocalVertex_Surface = dataSorter->GetnPoints();
65  Buffer_Send_nVertex[0] = nLocalVertex_Surface;
66  if (rank == MASTER_NODE) Buffer_Recv_nVertex = new unsigned long[nProcessor];
67 
68  /*--- Communicate the number of local vertices on each partition
69  to the master node with collective calls. ---*/
70 
71  SU2_MPI::Allreduce(&nLocalVertex_Surface, &MaxLocalVertex_Surface, 1,
72  MPI_UNSIGNED_LONG, MPI_MAX, SU2_MPI::GetComm());
73 
74  SU2_MPI::Gather(&Buffer_Send_nVertex, 1, MPI_UNSIGNED_LONG,
75  Buffer_Recv_nVertex, 1, MPI_UNSIGNED_LONG,
76  MASTER_NODE, SU2_MPI::GetComm());
77 
78  /*--- Allocate buffers for send/recv of the data and global IDs. ---*/
79 
80  auto *bufD_Send = new su2double[MaxLocalVertex_Surface*fieldNames.size()]();
81  su2double *bufD_Recv = nullptr;
82 
83  auto *bufL_Send = new unsigned long [MaxLocalVertex_Surface]();
84  unsigned long *bufL_Recv = nullptr;
85 
86  /*--- Load send buffers with the local data on this rank. ---*/
87 
88  index = 0;
89  for (iPoint = 0; iPoint < nLocalVertex_Surface; iPoint++) {
90 
91  /*--- Global index values. ---*/
92 
93  bufL_Send[iPoint] = dataSorter->GetGlobalIndex(iPoint);
94 
95  /*--- Solution data. ---*/
96 
97  for (iVar = 0; iVar < fieldNames.size(); iVar++){
98  bufD_Send[index] = dataSorter->GetData(iVar, iPoint);
99  index++;
100  }
101 
102  }
103 
104  /*--- Only the master rank allocates buffers for the recv. ---*/
105 
106  if (rank == MASTER_NODE) {
107  bufD_Recv = new su2double[nProcessor*MaxLocalVertex_Surface*fieldNames.size()]();
108  bufL_Recv = new unsigned long[nProcessor*MaxLocalVertex_Surface];
109  }
110 
111  /*--- Collective comms of the solution data and global IDs. ---*/
112 
113  SU2_MPI::Gather(bufD_Send, (int)MaxLocalVertex_Surface*fieldNames.size(), MPI_DOUBLE,
114  bufD_Recv, (int)MaxLocalVertex_Surface*fieldNames.size(), MPI_DOUBLE, MASTER_NODE, SU2_MPI::GetComm());
115 
116  SU2_MPI::Gather(bufL_Send, (int)MaxLocalVertex_Surface, MPI_UNSIGNED_LONG,
117  bufL_Recv, (int)MaxLocalVertex_Surface, MPI_UNSIGNED_LONG, MASTER_NODE, SU2_MPI::GetComm());
118 
119  /*--- The master rank alone writes the surface CSV file. ---*/
120 
121  if (rank == MASTER_NODE) {
122 
123  /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
124  val_filename.append(fileExt);
125 
126  /*--- Open the CSV file and write the header with variable names. ---*/
127  Surf_file.open(val_filename.c_str(), ios::out);
128  Surf_file << "\"Point\",";
129  for (iVar = 0; iVar < fieldNames.size()-1; iVar++) {
130  Surf_file << "\"" << fieldNames[iVar] << "\",";
131  }
132  Surf_file << "\"" << fieldNames[fieldNames.size()-1] << "\"" << endl;
133 
134  /*--- Loop through all of the collected data and write each node's values ---*/
135 
136  for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
137  for (iPoint = 0; iPoint < Buffer_Recv_nVertex[iProcessor]; iPoint++) {
138 
139  /*--- Current index position for global index access. ---*/
140 
141  index = iProcessor*MaxLocalVertex_Surface + iPoint;
142 
143  /*--- Write global index values. ---*/
144 
145  Surf_file << bufL_Recv[index] << ", ";
146 
147  /*--- Reset index for solution data access. ---*/
148 
149  index = (iProcessor*MaxLocalVertex_Surface*fieldNames.size() +
150  iPoint*fieldNames.size());
151 
152  /*--- Write the solution data for each field variable. ---*/
153 
154  for (iVar = 0; iVar < fieldNames.size(); iVar++){
155  Surf_file << scientific << bufD_Recv[index + iVar];
156  if (iVar != fieldNames.size() -1) Surf_file << ", ";
157  }
158  Surf_file << endl;
159 
160  }
161  }
162 
163  /*--- Close the file. ---*/
164 
165  Surf_file.close();
166 
167  }
168 
169  /*--- Free temporary memory. ---*/
170 
171  if (rank == MASTER_NODE) {
172  delete [] bufL_Recv;
173  delete [] bufD_Recv;
174  delete [] Buffer_Recv_nVertex;
175  }
176  delete [] bufL_Send;
177  delete [] bufD_Send;
178 }
