116void CLookUpTable::LoadTableRaw(const string& var_file_name_lut) {
117 CFileReaderLUT file_reader;
118
119 if (rank == MASTER_NODE) cout << "Loading lookup table, filename = " << var_file_name_lut << " ..." << endl;
120
121 file_reader.ReadRawLUT(var_file_name_lut);
122 table_dim = file_reader.GetTableDim();
123 n_table_levels = file_reader.GetNLevels();
124
125 n_points.resize(n_table_levels);
126 n_triangles.resize(n_table_levels);
127 n_hull_points.resize(n_table_levels);
128 table_data.resize(n_table_levels);
129 hull.resize(n_table_levels);
130 triangles.resize(n_table_levels);
131 interp_mat_inv_x_y.resize(n_table_levels);
132 edges.resize(n_table_levels);
133 edge_to_triangle.resize(n_table_levels);
134
135 for (unsigned long i_level = 0; i_level < n_table_levels; i_level++) {
136 n_points[i_level] = file_reader.GetNPoints(i_level);
137 n_triangles[i_level] = file_reader.GetNTriangles(i_level);
138 n_hull_points[i_level] = file_reader.GetNHullPoints(i_level);
139 table_data[i_level] = file_reader.GetTableData(i_level);
140 triangles[i_level] = file_reader.GetTriangles(i_level);
141 hull[i_level] = file_reader.GetHull(i_level);
142 memory_footprint_data += n_points[i_level] * sizeof(su2double);
143 }
144 memory_footprint_data /= 1e6;
145
146 n_variables = file_reader.GetNVariables();
147 version_lut = file_reader.GetVersionLUT();
148 version_reader = file_reader.GetVersionReader();
149 names_var = file_reader.GetNamesVar();
150
151 if (table_dim == 3) {
152 z_values_levels.resize(n_table_levels);
153 for (unsigned long i_level = 0; i_level < n_table_levels; i_level++) {
154 z_values_levels[i_level] = file_reader.GetTableLevel(i_level);
155 }
156 }
157 if (rank == MASTER_NODE) cout << " done." << endl;
158}

160void CLookUpTable::FindTableLimits(const string& name_cv1, const string& name_cv2) {
161 int idx_CV1 = GetIndexOfVar(name_cv1);
162 int idx_CV2 = GetIndexOfVar(name_cv2);
163 limits_table_x.resize(n_table_levels);
164 limits_table_y.resize(n_table_levels);
165
166 /* we find the lowest and highest value of y and x in the table */
167 for (auto i_level = 0u; i_level < n_table_levels; i_level++) {
168 limits_table_y[i_level] =
169 minmax_element(table_data[i_level][idx_CV2], table_data[i_level][idx_CV2] + table_data[i_level].cols());
170 limits_table_x[i_level] =
171 minmax_element(table_data[i_level][idx_CV1], table_data[i_level][idx_CV1] + table_data[i_level].cols());
172 }
173
174 if (table_dim == 3) {
175 limits_table_z = minmax_element(z_values_levels.data(), z_values_levels.data() + z_values_levels.size());
176 }
177}

252void CLookUpTable::IdentifyUniqueEdges() {
253 for (auto i_level = 0u; i_level < n_table_levels; i_level++) {
254 /* loop through elements and store the vector of element IDs (neighbors)
255 for each of the points in the table */
256
257 vector<vector<unsigned long> > neighborElemsOfPoint;
258
259 neighborElemsOfPoint.resize(n_points[i_level]);
260 for (unsigned long iElem = 0; iElem < n_triangles[i_level]; iElem++) {
261 /* loop over 3 points per triangle */
262 for (unsigned long iPoint = 0; iPoint < N_POINTS_TRIANGLE; iPoint++) {
263 /* get the global ID of the current point */
264 const unsigned long GlobalIndex = triangles[i_level][iElem][iPoint];
265
266 /* add the current element ID to the neighbor list for this point */
267 neighborElemsOfPoint[GlobalIndex].push_back(iElem);
268 }
269 }
270
271 /* remove duplicates from the neighboring element lists*/
272 vector<unsigned long>::iterator vecIt;
273 for (unsigned long iPoint = 0; iPoint < n_points[i_level]; iPoint++) {
274 /* sort neighboring elements for each point */
275 sort(neighborElemsOfPoint[iPoint].begin(), neighborElemsOfPoint[iPoint].end());
276
277 /* uniquify list of neighboring elements */
278 vecIt = unique(neighborElemsOfPoint[iPoint].begin(), neighborElemsOfPoint[iPoint].end());
279
280 /* adjust size of vector */
281 neighborElemsOfPoint[iPoint].resize(vecIt - neighborElemsOfPoint[iPoint].begin());
282 }
283
284 /* loop through all neighboring elements of each point and store
285 the point IDs that are neighboring points */
286 vector<vector<unsigned long> > neighborPointsOfPoint;
287 neighborPointsOfPoint.resize(n_points[i_level]);
288 for (unsigned long iPoint = 0; iPoint < n_points[i_level]; iPoint++) {
289 for (unsigned long iElem = 0; iElem < neighborElemsOfPoint[iPoint].size(); iElem++) {
290 /* loop over element points */
291 for (unsigned long jPoint = 0; jPoint < N_POINTS_TRIANGLE; jPoint++) {
292 /* get the global ID of the current point */
293 const unsigned long GlobalIndex = triangles[i_level][neighborElemsOfPoint[iPoint][iElem]][jPoint];
294
295 /* add the current element ID to the neighbor list for this point */
296 if (GlobalIndex != iPoint) neighborPointsOfPoint[iPoint].push_back(GlobalIndex);
297 }
298 }
299 }
300
301 /* remove duplicates from the neighboring points lists */
302 for (unsigned long iPoint = 0; iPoint < n_points[i_level]; iPoint++) {
303 /* sort neighboring points for each point */
304 sort(neighborPointsOfPoint[iPoint].begin(), neighborPointsOfPoint[iPoint].end());
305
306 /* uniquify list of neighboring elements */
307 vecIt = unique(neighborPointsOfPoint[iPoint].begin(), neighborPointsOfPoint[iPoint].end());
308
309 /* adjust size of vector */
310 neighborPointsOfPoint[iPoint].resize(vecIt - neighborPointsOfPoint[iPoint].begin());
311 }
312
313 /* loop through our point neighbors and fill the vector of the unique
314 point pairs making up each edge in the grid. We impose a requirement
315 that the smaller global index is in the first position and the larger
316 is in the second to make for a unique set, so there's no need to
317 remove duplicates. */
318 for (unsigned long iPoint = 0; iPoint < n_points[i_level]; iPoint++) {
319 for (unsigned long jPoint = 0; jPoint < neighborPointsOfPoint[iPoint].size(); jPoint++) {
320 /* Store the neighbor index more clearly. */
321 const unsigned long GlobalIndex = neighborPointsOfPoint[iPoint][jPoint];
322
323 /* Store the edge so that the lower index of the pair is always first. */
324 if (iPoint < GlobalIndex) {
325 std::array<unsigned long, 2> edge;
326 edge[0] = iPoint;
327 edge[1] = GlobalIndex;
328 edges[i_level].push_back(edge);
329 }
330 }
331 }
332
333 /* Loop over our edges data structure. For the first point in each
334 pair, loop through the neighboring elements and store the two
335 elements that contain the second point in the edge ('left' and 'right' of the edge). */
336 edge_to_triangle[i_level].resize(edges[i_level].size());
337 for (unsigned long iEdge = 0; iEdge < edges[i_level].size(); iEdge++) {
338 /* Store the two points of the edge more clearly. */
339 const unsigned long iPoint = edges[i_level][iEdge][0];
340 const unsigned long jPoint = edges[i_level][iEdge][1];
341
342 /* Loop over all neighboring elements to iPoint. */
343 for (unsigned long iElem = 0; iElem < neighborElemsOfPoint[iPoint].size(); iElem++) {
344 /* loop over 3 points per triangle */
345 for (unsigned long kPoint = 0; kPoint < N_POINTS_TRIANGLE; kPoint++) {
346 /* Get the global ID of the current point. */
347 const unsigned long GlobalIndex = triangles[i_level][neighborElemsOfPoint[iPoint][iElem]][kPoint];
348
349 /* Add the current element ID to the neighbor list for this point. */
350 if (GlobalIndex == jPoint) edge_to_triangle[i_level][iEdge].push_back(neighborElemsOfPoint[iPoint][iElem]);
351 }
352 }
353 }
354 }

357void CLookUpTable::ComputeInterpCoeffs() {
358 for (auto i_level = 0ul; i_level < n_table_levels; i_level++) {
359 /* build KD tree for y, x space */
360
361 std::array<unsigned long, 3> next_triangle;
362
363 const su2double* val_CV1 = GetDataP(name_CV1, i_level);
364 const su2double* val_CV2 = GetDataP(name_CV2, i_level);
365
366 /* calculate weights for each triangle (basically a distance function) and
367 * build inverse interpolation matrices */
368 interp_mat_inv_x_y[i_level].resize(n_triangles[i_level]);
369 for (unsigned long i_triangle = 0; i_triangle < n_triangles[i_level]; i_triangle++) {
370 for (int p = 0; p < 3; p++) {
371 next_triangle[p] = triangles[i_level][i_triangle][p];
372 }
373
374 su2activematrix x_interp_mat_inv(3, 3);
375 GetInterpMatInv(val_CV1, val_CV2, next_triangle, x_interp_mat_inv);
376 interp_mat_inv_x_y[i_level][i_triangle] = x_interp_mat_inv;
377 }
378 }
379}

381void CLookUpTable::GetInterpMatInv(const su2double* vec_x, const su2double* vec_y,
382 std::array<unsigned long, 3>& point_ids, su2activematrix& interp_mat_inv) {
383 const unsigned int M = 3;
384 CSquareMatrixCM global_M(3);
385
386 /* setup LHM matrix for the interpolation */
387 for (unsigned int i_point = 0; i_point < M; i_point++) {
388 su2double x = vec_x[point_ids[i_point]];
389 su2double y = vec_y[point_ids[i_point]];
390
391 global_M(i_point, 0) = SU2_TYPE::GetValue(1.0);
392 global_M(i_point, 1) = SU2_TYPE::GetValue(x);
393 global_M(i_point, 2) = SU2_TYPE::GetValue(y);
394 }
395
396 global_M.Invert();
397 global_M.Transpose();
398
399 for (unsigned int i = 0; i < M; i++) {
400 for (unsigned int j = 0; j < M; j++) {
401 interp_mat_inv[i][j] = global_M(i, j);
402 }
403 }

406std::pair<unsigned long, unsigned long> CLookUpTable::FindInclusionLevels(const su2double val_CV3) {
407 /*--- Find the table levels with constant z-values directly below and above the query value val_CV3 ---*/
408
409 /* Check if val_CV3 lies outside table bounds */
410 if (val_CV3 >= *limits_table_z.second) {
411 return std::make_pair(n_table_levels - 1, n_table_levels - 1);
412 } else if (val_CV3 <= *limits_table_z.first) {
413 return std::make_pair(0, 0);
414 } else {
415 std::pair<unsigned long, unsigned long> inclusion_levels;
416 /* Loop over table levels to find the levels directly above and below the query value */
417 for (auto i_level = 0ul; i_level < n_table_levels - 1; i_level++) {
418 if ((val_CV3 >= z_values_levels[i_level]) && (val_CV3 < z_values_levels[i_level + 1])) {
419 inclusion_levels = std::make_pair(i_level, i_level + 1);
420 }
421 }
422 return inclusion_levels;
423 }

426unsigned long CLookUpTable::LookUp_XYZ(const std::string& val_name_var, su2double* val_var, su2double val_CV1,
427 su2double val_CV2, su2double val_CV3) {
428 /*--- Perform quasi-3D interpolation for a single variable named val_name_var on a query point
429 with coordinates val_CV1, val_CV2, and val_CV3 ---*/
430
431 /* 1: Find table levels directly above and below the query point (the levels that sandwhich val_CV3) */
432 std::pair<unsigned long, unsigned long> inclusion_levels = FindInclusionLevels(val_CV3);
433 bool within_z_limits = (inclusion_levels.first != inclusion_levels.second);
434
435 if (within_z_limits) {
436 /* 2: Determine val_CV1 and val_CV2 for the inclusion table levels. */
437 std::array<std::array<su2double, 2>, 2> lower_upper_CV1_2 =
438 ComputeNormalizedXY(inclusion_levels, val_CV1, val_CV2, val_CV3);
439 su2double val_CV1_lower = lower_upper_CV1_2[0][0], val_CV2_lower = lower_upper_CV1_2[0][1],
440 val_CV1_upper = lower_upper_CV1_2[1][0], val_CV2_upper = lower_upper_CV1_2[1][1];
441
442 /* 3: Perform 2D interpolations on upper and lower inclusion levels */
443 unsigned long lower_level = inclusion_levels.first;
444 unsigned long upper_level = inclusion_levels.second;
445
446 su2double val_var_lower, val_var_upper;
447 unsigned long exit_code_lower = LookUp_XY(val_name_var, &val_var_lower, val_CV1_lower, val_CV2_lower, lower_level);
448 unsigned long exit_code_upper = LookUp_XY(val_name_var, &val_var_upper, val_CV1_upper, val_CV2_upper, upper_level);
449
450 /* 4: Perform linear interpolation along the z-direction using the x-y interpolation results
451 from upper and lower trapezoidal maps */
452 Linear_Interpolation(val_CV3, lower_level, upper_level, val_var_lower, val_var_upper, val_var);
453
454 return max(exit_code_lower, exit_code_upper);
455 } else {
456 /* Perform single, 2D interpolation when val_CV3 lies outside table bounds */
457 unsigned long bound_level = inclusion_levels.first;
458 LookUp_XY(val_name_var, val_var, val_CV1, val_CV2, bound_level);
459 return 1;
460 }

501void CLookUpTable::Linear_Interpolation(const su2double val_CV3, const unsigned long lower_level,
502 const unsigned long upper_level, su2double& lower_value, su2double& upper_value,
503 su2double*& var_val) const {
504 /* Perform linear interpolation along the z-direction of the table for a single variable */
505
506 /* Retrieve constant z-values of inclusion levels */
507 su2double val_z_lower = z_values_levels[lower_level];
508 su2double val_z_upper = z_values_levels[upper_level];
509
510 /* Compute linear interpolation coefficients */
511 su2double factor_upper = (val_CV3 - val_z_lower) / (val_z_upper - val_z_lower);
512 su2double factor_lower = (val_z_upper - val_CV3) / (val_z_upper - val_z_lower);
513
514 /* Perform linear interpolation */
515 *var_val = lower_value * factor_lower + upper_value * factor_upper;

518void CLookUpTable::Linear_Interpolation(const su2double val_CV3, const unsigned long lower_level,
519 const unsigned long upper_level, std::vector<su2double>& lower_values,
520 std::vector<su2double>& upper_values, std::vector<su2double>& var_vals) const {
521 /* Perform linear interpolation along the z-direction of the table for multiple variables */
522
523 /* Retrieve constant z-values of inclusion levels */
524 su2double val_z_lower = z_values_levels[lower_level];
525 su2double val_z_upper = z_values_levels[upper_level];
526
527 /* Compute linear interpolation coefficients */
528 su2double factor_upper = (val_CV3 - val_z_lower) / (val_z_upper - val_z_lower);
529 su2double factor_lower = (val_z_upper - val_CV3) / (val_z_upper - val_z_lower);
530
531 /* Perform linear interpolation */
532 for (size_t iVar = 0; iVar < var_vals.size(); iVar++) {
533 var_vals[iVar] = lower_values[iVar] * factor_lower + upper_values[iVar] * factor_upper;
534 }

537std::array<std::array<su2double, 2>, 2> CLookUpTable::ComputeNormalizedXY(
538 std::pair<unsigned long, unsigned long>& inclusion_levels, const su2double val_CV1, const su2double val_CV2,
539 const su2double val_CV3) {
540 /* Determine the values of the controlling variables of the upper and lower table levels corresponding to the
541 normalized query point coordinates. */
542
543 unsigned long lower_level = inclusion_levels.first;
544 unsigned long upper_level = inclusion_levels.second;
545
546 /* Get the upper and lower limits of the first and second controlling variables on the inclusion levels. */
547 su2double x_min_lower = *limits_table_x[lower_level].first, x_max_lower = *limits_table_x[lower_level].second;
548 su2double x_min_upper = *limits_table_x[upper_level].first, x_max_upper = *limits_table_x[upper_level].second;
549 su2double y_min_lower = *limits_table_y[lower_level].first, y_max_lower = *limits_table_y[lower_level].second;
550 su2double y_min_upper = *limits_table_y[upper_level].first, y_max_upper = *limits_table_y[upper_level].second;
551
552 /* Interpolate the table limits to the query point coordinates. */
553 su2double x_min_q = x_min_lower + (x_min_upper - x_min_lower) * (val_CV3 - z_values_levels[lower_level]) /
554 (z_values_levels[upper_level] - z_values_levels[lower_level] + EPS);
555 su2double x_max_q = x_max_lower + (x_max_upper - x_max_lower) * (val_CV3 - z_values_levels[lower_level]) /
556 (z_values_levels[upper_level] - z_values_levels[lower_level] + EPS);
557 su2double y_min_q = y_min_lower + (y_min_upper - y_min_lower) * (val_CV3 - z_values_levels[lower_level]) /
558 (z_values_levels[upper_level] - z_values_levels[lower_level] + EPS);
559 su2double y_max_q = y_max_lower + (y_max_upper - y_max_lower) * (val_CV3 - z_values_levels[lower_level]) /
560 (z_values_levels[upper_level] - z_values_levels[lower_level] + EPS);
561
562 /* Compute the normalized first and second controlling variable values. */
563 su2double x_norm_q = (val_CV1 - x_min_q) / (x_max_q - x_min_q);
564 su2double y_norm_q = (val_CV2 - y_min_q) / (y_max_q - y_min_q);
565
566 /* De-normalize the normalized coordinates for the upper and lower table levels. */
567 su2double val_CV1_upper = x_min_upper + (x_max_upper - x_min_upper) * x_norm_q;
568 su2double val_CV2_upper = y_min_upper + (y_max_upper - y_min_upper) * y_norm_q;
569 su2double val_CV1_lower = x_min_lower + (x_max_lower - x_min_lower) * x_norm_q;
570 su2double val_CV2_lower = y_min_lower + (y_max_lower - y_min_lower) * y_norm_q;
571
572 /* Store coordinates in output */
573 std::array<std::array<su2double, 2>, 2> lower_upper_CVs;
574 lower_upper_CVs[0][0] = val_CV1_lower;
575 lower_upper_CVs[0][1] = val_CV2_lower;
576 lower_upper_CVs[1][0] = val_CV1_upper;
577 lower_upper_CVs[1][1] = val_CV2_upper;
578
579 return lower_upper_CVs;
580}

582unsigned long CLookUpTable::LookUp_XY(const string& val_name_var, su2double* val_var, su2double val_CV1,
583 su2double val_CV2, unsigned long i_level) {
584 unsigned long exit_code = 1;
585
586 if (noSource(val_name_var)) {
587 *val_var = 0.0;
588 exit_code = 0;
589 return exit_code;
590 }
591
592 /* check if x and y value is in table range */
593 if ((val_CV1 >= *limits_table_x[i_level].first && val_CV1 <= *limits_table_x[i_level].second) &&
594 (val_CV2 >= *limits_table_y[i_level].first && val_CV2 <= *limits_table_y[i_level].second)) {
595 /* find the triangle that holds the (x, y) point */
596 unsigned long id_triangle = trap_map_x_y[i_level].GetTriangle(val_CV1, val_CV2);
597
598 if (IsInTriangle(val_CV1, val_CV2, id_triangle, i_level)) {
599 /* get interpolation coefficients for point on triangle */
600 std::array<su2double, 3> interp_coeffs{0};
601 GetInterpCoeffs(val_CV1, val_CV2, interp_mat_inv_x_y[i_level][id_triangle], interp_coeffs);
602
603 /* first, copy the single triangle from the large triangle list*/
604 std::array<unsigned long, 3> triangle{0};
605 for (int p = 0; p < 3; p++) triangle[p] = triangles[i_level][id_triangle][p];
606
607 *val_var = Interpolate(GetDataP(val_name_var, i_level), triangle, interp_coeffs);
608 exit_code = 0;
609 }
610 }
611 if (exit_code == 1) InterpolateToNearestNeighbors(val_CV1, val_CV2, val_name_var, val_var, i_level);
612
613 return exit_code;
614}

629unsigned long CLookUpTable::LookUp_XY(const vector<string>& val_names_var, vector<su2double*>& val_vars,
630 su2double val_CV1, su2double val_CV2, unsigned long i_level) {
631 unsigned long exit_code = 1;
632 unsigned long id_triangle = 0;
633 std::array<su2double, 3> interp_coeffs{0};
634 std::array<unsigned long, 3> triangle{0};
635
636 /* check if x value is in table x-dimension range
637 * and if y is in table y-dimension table range */
638 if ((val_CV1 >= *limits_table_x[i_level].first && val_CV1 <= *limits_table_x[i_level].second) &&
639 (val_CV2 >= *limits_table_y[i_level].first && val_CV2 <= *limits_table_y[i_level].second)) {
640 /* if so, try to find the triangle that holds the (prog, enth) point */
641 id_triangle = trap_map_x_y[i_level].GetTriangle(val_CV1, val_CV2);
642
643 /* check if point is inside a triangle (if table domain is non-rectangular,
644 * the previous range check might be true but the point could still be outside of the domain) */
645 if (IsInTriangle(val_CV1, val_CV2, id_triangle, i_level)) {
646 /* if so, get interpolation coefficients for point in the triangle */
647 GetInterpCoeffs(val_CV1, val_CV2, interp_mat_inv_x_y[i_level][id_triangle], interp_coeffs);
648
649 /* exit_code 0 means point was in triangle */
650 exit_code = 0;
651 }
652 }
653
654 /* loop over variable names and interpolate / get values */
655 for (long unsigned int i_var = 0; i_var < val_names_var.size(); ++i_var) {
656 if (noSource(val_names_var[i_var])) {
657 *val_vars[i_var] = 0.0;
658 exit_code = 0;
659 } else {
660 if (exit_code == 0) {
661 /* first, copy the single triangle from the large triangle list*/
662 for (int p = 0; p < 3; p++) triangle[p] = triangles[i_level][id_triangle][p];
663 *val_vars[i_var] = Interpolate(GetDataP(val_names_var[i_var], i_level), triangle, interp_coeffs);
664 } else {
665 InterpolateToNearestNeighbors(val_CV1, val_CV2, val_names_var[i_var], val_vars[i_var], i_level);
666 }
667 }
668 }
669
670 return exit_code;
671}

673void CLookUpTable::GetInterpCoeffs(su2double val_CV1, su2double val_CV2, su2activematrix& interp_mat_inv,
674 std::array<su2double, 3>& interp_coeffs) {
675 std::array<su2double, 3> query_vector = {1, val_CV1, val_CV2};
676
677 su2double d;
678 for (int i = 0; i < 3; i++) {
679 d = 0;
680 for (int j = 0; j < 3; j++) {
681 d = d + interp_mat_inv[i][j] * query_vector[j];
682 }
683 interp_coeffs[i] = d;
684 }

700unsigned long CLookUpTable::FindNearestNeighborOnHull(su2double val_CV1, su2double val_CV2, unsigned long i_level) {
701 su2double min_distance = 1.e99;
702 su2double next_distance = 1.e99;
703 su2double next_x_norm;
704 su2double next_y_norm;
705 unsigned long neighbor_id = 0;
706
707 const su2double* x_table = GetDataP(name_CV1, i_level);
708 const su2double* y_table = GetDataP(name_CV2, i_level);
709
710 su2double norm_coeff_x = 1. / (limits_table_x[i_level].second - limits_table_x[i_level].first);
711 su2double norm_coeff_y = 1. / (limits_table_y[i_level].second - limits_table_y[i_level].first);
712 su2double val_CV1_norm = val_CV1 / (limits_table_x[i_level].second - limits_table_x[i_level].first);
713 su2double val_CV2_norm = val_CV2 / (limits_table_y[i_level].second - limits_table_y[i_level].first);
714
715 for (unsigned long i_point = 0; i_point < n_hull_points[i_level]; ++i_point) {
716 next_x_norm = x_table[hull[i_level][i_point]] * norm_coeff_x;
717 next_y_norm = y_table[hull[i_level][i_point]] * norm_coeff_y;
718
719 next_distance = sqrt(pow(val_CV1_norm - next_x_norm, 2) + pow(val_CV2_norm - next_y_norm, 2));
720
721 if (next_distance < min_distance) {
722 min_distance = next_distance;
723 neighbor_id = hull[i_level][i_point];
724 }
725 }
726 return neighbor_id;
727}

729void CLookUpTable::InterpolateToNearestNeighbors(const su2double val_CV1, const su2double val_CV2,
730 const std::vector<std::string>& names_var,
731 std::vector<su2double*>& var_vals, const unsigned long i_level) {
732 /* Interpolate data using distance-weighted averaging on the two nearest table nodes. */
733
734 su2double min_distance = 1e99, second_distance = 1e99;
735 su2double norm_coeff_x = 1. / (*limits_table_x[i_level].second - *limits_table_x[i_level].first);
736 su2double norm_coeff_y = 1. / (*limits_table_y[i_level].second - *limits_table_y[i_level].first);
737 su2double val_CV1_norm = val_CV1 / (*limits_table_x[i_level].second - *limits_table_x[i_level].first);
738 su2double val_CV2_norm = val_CV2 / (*limits_table_y[i_level].second - *limits_table_y[i_level].first);
739
740 const su2double* x_table = GetDataP(name_CV1, i_level);
741 const su2double* y_table = GetDataP(name_CV2, i_level);
742 unsigned long i_nearest = 0, i_second_nearest = 0;
743
744 for (unsigned long i_point = 0; i_point < n_hull_points[i_level]; ++i_point) {
745 su2double next_x_norm = x_table[hull[i_level][i_point]] * norm_coeff_x;
746 su2double next_y_norm = y_table[hull[i_level][i_point]] * norm_coeff_y;
747
748 su2double next_distance = pow(val_CV1_norm - next_x_norm, 2) + pow(val_CV2_norm - next_y_norm, 2);
749
750 /* Check if distance to node is lower than current nearest or second nearest node. */
751 if (next_distance < min_distance) {
752 second_distance = min_distance;
753 min_distance = next_distance;
754
755 i_second_nearest = i_nearest;
756 i_nearest = hull[i_level][i_point];
757 } else if ((next_distance > min_distance) && (next_distance < second_distance)) {
758 i_second_nearest = hull[i_level][i_point];
759
760 second_distance = next_distance;
761 }
762 }
763
764 min_distance = sqrt(min_distance);
765 second_distance = sqrt(second_distance);
766
767 /* Interpolate data using distance-weighted averaging */
768 su2double delimiter = (1.0 / min_distance) + (1.0 / second_distance);
769 for (auto iVar = 0u; iVar < var_vals.size(); iVar++) {
770 su2double data_nearest = GetDataP(names_var[iVar], i_level)[i_nearest],
771 data_second_nearest = GetDataP(names_var[iVar], i_level)[i_second_nearest];
772 *var_vals[iVar] = (data_nearest * (1.0 / min_distance) + data_second_nearest * (1.0 / second_distance)) / delimiter;
773 }
774}

776void CLookUpTable::InterpolateToNearestNeighbors(const su2double val_CV1, const su2double val_CV2,
777 const std::string& name_var, su2double* var_vals,
778 const unsigned long i_level) {
779 std::vector<std::string> names_var = {name_var};
780 std::vector<su2double*> val_names_var = {var_vals};
781 InterpolateToNearestNeighbors(val_CV1, val_CV2, names_var, val_names_var, i_level);
782}
