168  limits_table_x.resize(n_table_levels);

257 void CLookUpTable::IdentifyUniqueEdges() {
258  for (auto i_level = 0u; i_level < n_table_levels; i_level++) {
259  /* loop through elements and store the vector of element IDs (neighbors)
260  for each of the points in the table */
261 
262  vector<vector<unsigned long> > neighborElemsOfPoint;
263 
264  neighborElemsOfPoint.resize(n_points[i_level]);
265  for (auto iElem = 0u; iElem < n_triangles[i_level]; iElem++) {
266  /* loop over 3 points per triangle */
267  for (auto iPoint = 0u; iPoint < N_POINTS_TRIANGLE; iPoint++) {
268  /* get the global ID of the current point */
269  const auto GlobalIndex = triangles[i_level][iElem][iPoint];
270 
271  /* add the current element ID to the neighbor list for this point */
272  neighborElemsOfPoint[GlobalIndex].push_back(iElem);
273  }
274  }
275 
276  /* remove duplicates from the neighboring element lists*/
277  vector<unsigned long>::iterator vecIt;
278  for (auto iPoint = 0u; iPoint < n_points[i_level]; iPoint++) {
279  /* sort neighboring elements for each point */
280  sort(neighborElemsOfPoint[iPoint].begin(), neighborElemsOfPoint[iPoint].end());
281 
282  /* uniquify list of neighboring elements */
283  vecIt = unique(neighborElemsOfPoint[iPoint].begin(), neighborElemsOfPoint[iPoint].end());
284 
285  /* adjust size of vector */
286  neighborElemsOfPoint[iPoint].resize(vecIt - neighborElemsOfPoint[iPoint].begin());
287  }
288 
289  /* loop through all neighboring elements of each point and store
290  the point IDs that are neighboring points */
291  vector<vector<unsigned long> > neighborPointsOfPoint;
292  neighborPointsOfPoint.resize(n_points[i_level]);
293  for (auto iPoint = 0u; iPoint < n_points[i_level]; iPoint++) {
294  for (auto iElem = 0u; iElem < neighborElemsOfPoint[iPoint].size(); iElem++) {
295  /* loop over element points */
296  for (auto jPoint = 0u; jPoint < N_POINTS_TRIANGLE; jPoint++) {
297  /* get the global ID of the current point */
298  const auto GlobalIndex = triangles[i_level][neighborElemsOfPoint[iPoint][iElem]][jPoint];
299 
300  /* add the current element ID to the neighbor list for this point */
301  if (GlobalIndex != iPoint) neighborPointsOfPoint[iPoint].push_back(GlobalIndex);
302  }
303  }
304  }
305 
306  /* remove duplicates from the neighboring points lists */
307  for (auto iPoint = 0u; iPoint < n_points[i_level]; iPoint++) {
308  /* sort neighboring points for each point */
309  sort(neighborPointsOfPoint[iPoint].begin(), neighborPointsOfPoint[iPoint].end());
310 
311  /* uniquify list of neighboring elements */
312  vecIt = unique(neighborPointsOfPoint[iPoint].begin(), neighborPointsOfPoint[iPoint].end());
313 
314  /* adjust size of vector */
315  neighborPointsOfPoint[iPoint].resize(vecIt - neighborPointsOfPoint[iPoint].begin());
316  }
317 
318  /* loop through our point neighbors and fill the vector of the unique
319  point pairs making up each edge in the grid. We impose a requirement
320  that the smaller global index is in the first position and the larger
321  is in the second to make for a unique set, so there's no need to
322  remove duplicates. */
323  for (auto iPoint = 0u; iPoint < n_points[i_level]; iPoint++) {
324  for (auto jPoint = 0u; jPoint < neighborPointsOfPoint[iPoint].size(); jPoint++) {
325  /* Store the neighbor index more clearly. */
326  const auto GlobalIndex = neighborPointsOfPoint[iPoint][jPoint];
327 
328  /* Store the edge so that the lower index of the pair is always first. */
329  if (iPoint < GlobalIndex) {
330  std::array<unsigned long, 2> edge;
331  edge[0] = iPoint;
332  edge[1] = GlobalIndex;
333  edges[i_level].push_back(edge);
334  }
335  }
336  }
337 
338  /* Loop over our edges data structure. For the first point in each
339  pair, loop through the neighboring elements and store the two
340  elements that contain the second point in the edge ('left' and 'right' of the edge). */
341  edge_to_triangle[i_level].resize(edges[i_level].size());
342  for (auto iEdge = 0u; iEdge < edges[i_level].size(); iEdge++) {
343  /* Store the two points of the edge more clearly. */
344  const auto iPoint = edges[i_level][iEdge][0];
345  const auto jPoint = edges[i_level][iEdge][1];
346 
347  /* Loop over all neighboring elements to iPoint. */
348  for (auto iElem = 0u; iElem < neighborElemsOfPoint[iPoint].size(); iElem++) {
349  /* loop over 3 points per triangle */
350  for (auto kPoint = 0u; kPoint < N_POINTS_TRIANGLE; kPoint++) {
351  /* Get the global ID of the current point. */
352  const auto GlobalIndex = triangles[i_level][neighborElemsOfPoint[iPoint][iElem]][kPoint];
353 
354  /* Add the current element ID to the neighbor list for this point. */
355  if (GlobalIndex == jPoint) edge_to_triangle[i_level][iEdge].push_back(neighborElemsOfPoint[iPoint][iElem]);
356  }
357  }
358  }
359  }
360 }

362 void CLookUpTable::ComputeInterpCoeffs() {
363  for (auto i_level = 0ul; i_level < n_table_levels; i_level++) {
364  /* build KD tree for y, x space */
365 
366  std::array<unsigned long, 3> next_triangle;
367 
368  const su2double* val_CV1 = table_data[i_level][idx_CV1];
369  const su2double* val_CV2 = table_data[i_level][idx_CV2];
370 
371  /* calculate weights for each triangle (basically a distance function) and
372  * build inverse interpolation matrices */
373  interp_mat_inv_x_y[i_level].resize(n_triangles[i_level]);
374  for (auto i_triangle = 0u; i_triangle < n_triangles[i_level]; i_triangle++) {
375  for (auto p = 0u; p < N_POINTS_TRIANGLE; p++) {
376  next_triangle[p] = triangles[i_level][i_triangle][p];
377  }
378 
379  su2activematrix x_interp_mat_inv(N_POINTS_TRIANGLE, N_POINTS_TRIANGLE);
380  GetInterpMatInv(val_CV1, val_CV2, next_triangle, x_interp_mat_inv);
381  interp_mat_inv_x_y[i_level][i_triangle] = x_interp_mat_inv;
382  }
383  }

386 void CLookUpTable::GetInterpMatInv(const su2double* vec_x, const su2double* vec_y,
387  std::array<unsigned long, 3>& point_ids, su2activematrix& interp_mat_inv) {
388  CSquareMatrixCM global_M(N_POINTS_TRIANGLE);
389 
390  /* setup LHM matrix for the interpolation */
391  for (auto i_vertex = 0u; i_vertex < N_POINTS_TRIANGLE; i_vertex++) {
392  su2double x = vec_x[point_ids[i_vertex]];
393  su2double y = vec_y[point_ids[i_vertex]];
394 
395  global_M(i_vertex, 0) = SU2_TYPE::GetValue(1.0);
396  global_M(i_vertex, 1) = SU2_TYPE::GetValue(x);
397  global_M(i_vertex, 2) = SU2_TYPE::GetValue(y);
398  }
399 
400  global_M.Invert();
401  global_M.Transpose();
402 
403  for (auto i = 0u; i < N_POINTS_TRIANGLE; i++) {
404  for (auto j = 0u; j < N_POINTS_TRIANGLE; j++) {
405  interp_mat_inv[i][j] = global_M(i, j);
406  }
407  }

410 std::pair<unsigned long, unsigned long> CLookUpTable::FindInclusionLevels(const su2double val_CV3) {
411  /*--- Find the table levels with constant z-values directly below and above the query value val_CV3 ---*/
412 
413  su2double val_z = val_CV3;
414  unsigned long i_up{0}, i_low{0};
415  /* Check if val_CV3 lies outside table bounds */
416  if (val_z < z_values_levels.front()) {
417  val_z = z_values_levels.front();
418  return make_pair(i_low, i_up);
419  }
420  if (val_z > z_values_levels.back()) {
421  val_z = z_values_levels.back();
422  i_low = n_table_levels - 1;
423  i_up = n_table_levels - 1;
424  return make_pair(i_low, i_up);
425  }
426  std::pair<std::vector<su2double>::iterator, std::vector<su2double>::iterator> bounds;
427  bounds = std::equal_range(z_values_levels.begin(), z_values_levels.end(), val_z);
428 
429  /*--- if upper bound = 0, then use the range [0,1] ---*/
430  i_up = max<unsigned long>(1, bounds.first - z_values_levels.begin());
431  i_low = i_up - 1;
432  return make_pair(i_low, i_up);
433 }

452 bool CLookUpTable::LookUp_XYZ(const std::vector<unsigned long>& idx_var, std::vector<su2double>& val_vars,
453  const su2double val_CV1, const su2double val_CV2, const su2double val_CV3) {
454  /*--- Perform quasi-3D interpolation for a vector of variables with names val_names_var
455  on a query point with coordinates val_CV1, val_CV2, and val_CV3 ---*/
456 
457  /* 1: Find table levels directly above and below the query point (the levels that sandwhich val_CV3) */
458  std::pair<unsigned long, unsigned long> inclusion_levels = FindInclusionLevels(val_CV3);
459  bool within_z_limits = (inclusion_levels.first != inclusion_levels.second);
460 
461  if (within_z_limits) {
462  /* 2: Determine val_CV1 and val_CV2 for the inclusion table levels. */
463  std::array<std::array<su2double, 2>, 2> lower_upper_CV1_2 =
464  ComputeNormalizedXY(inclusion_levels, val_CV1, val_CV2, val_CV3);
465  su2double val_CV1_lower = lower_upper_CV1_2[0][0], val_CV2_lower = lower_upper_CV1_2[0][1],
466  val_CV1_upper = lower_upper_CV1_2[1][0], val_CV2_upper = lower_upper_CV1_2[1][1];
467 
468  /* 3: Perform 2D interpolations on upper and lower inclusion levels */
469  unsigned long lower_level = inclusion_levels.first;
470  unsigned long upper_level = inclusion_levels.second;
471 
472  std::vector<su2double> val_vars_lower, val_vars_upper;
473  val_vars_lower.resize(val_vars.size());
474  val_vars_upper.resize(val_vars.size());
475  auto inside_lower = LookUp_XY(idx_var, val_vars_lower, val_CV1_lower, val_CV2_lower, lower_level);
476  auto inside_upper = LookUp_XY(idx_var, val_vars_upper, val_CV1_upper, val_CV2_upper, upper_level);
477 
478  /* 4: Perform linear interpolation along the z-direction using the x-y interpolation results
479  from upper and lower trapezoidal maps */
480  Linear_Interpolation(val_CV3, lower_level, upper_level, val_vars_lower, val_vars_upper, val_vars);
481 
482  return (inside_upper && inside_lower);
483  } else {
484  /* Perform single, 2D interpolation when val_CV3 lies outside table bounds */
485  unsigned long bound_level = inclusion_levels.first;
486  LookUp_XY(idx_var, val_vars, val_CV1, val_CV2, bound_level);
487  return false;
488  }
489 }

527 std::array<std::array<su2double, 2>, 2> CLookUpTable::ComputeNormalizedXY(
528  std::pair<unsigned long, unsigned long>& inclusion_levels, const su2double val_CV1, const su2double val_CV2,
529  const su2double val_CV3) {
530  /* Determine the values of the controlling variables of the upper and lower table levels corresponding to the
531  normalized query point coordinates. */
532 
533  unsigned long lower_level = inclusion_levels.first;
534  unsigned long upper_level = inclusion_levels.second;
535 
536  /* Get the upper and lower limits of the first and second controlling variables on the inclusion levels. */
537  su2double x_min_lower = *limits_table_x[lower_level].first, x_max_lower = *limits_table_x[lower_level].second;
538  su2double x_min_upper = *limits_table_x[upper_level].first, x_max_upper = *limits_table_x[upper_level].second;
539  su2double y_min_lower = *limits_table_y[lower_level].first, y_max_lower = *limits_table_y[lower_level].second;
540  su2double y_min_upper = *limits_table_y[upper_level].first, y_max_upper = *limits_table_y[upper_level].second;
541 
542  /* Interpolate the table limits to the query point coordinates. */
543  su2double x_min_q = x_min_lower + (x_min_upper - x_min_lower) * (val_CV3 - z_values_levels[lower_level]) /
544  (z_values_levels[upper_level] - z_values_levels[lower_level] + EPS);
545  su2double x_max_q = x_max_lower + (x_max_upper - x_max_lower) * (val_CV3 - z_values_levels[lower_level]) /
546  (z_values_levels[upper_level] - z_values_levels[lower_level] + EPS);
547  su2double y_min_q = y_min_lower + (y_min_upper - y_min_lower) * (val_CV3 - z_values_levels[lower_level]) /
548  (z_values_levels[upper_level] - z_values_levels[lower_level] + EPS);
549  su2double y_max_q = y_max_lower + (y_max_upper - y_max_lower) * (val_CV3 - z_values_levels[lower_level]) /
550  (z_values_levels[upper_level] - z_values_levels[lower_level] + EPS);
551 
552  /* Compute the normalized first and second controlling variable values. */
553  su2double x_norm_q = (val_CV1 - x_min_q) / (x_max_q - x_min_q);
554  su2double y_norm_q = (val_CV2 - y_min_q) / (y_max_q - y_min_q);
555 
556  /* De-normalize the normalized coordinates for the upper and lower table levels. */
557  su2double val_CV1_upper = x_min_upper + (x_max_upper - x_min_upper) * x_norm_q;
558  su2double val_CV2_upper = y_min_upper + (y_max_upper - y_min_upper) * y_norm_q;
559  su2double val_CV1_lower = x_min_lower + (x_max_lower - x_min_lower) * x_norm_q;
560  su2double val_CV2_lower = y_min_lower + (y_max_lower - y_min_lower) * y_norm_q;
561 
562  /* Store coordinates in output */
563  std::array<std::array<su2double, 2>, 2> lower_upper_CVs;
564  lower_upper_CVs[0][0] = val_CV1_lower;
565  lower_upper_CVs[0][1] = val_CV2_lower;
566  lower_upper_CVs[1][0] = val_CV1_upper;
567  lower_upper_CVs[1][1] = val_CV2_upper;
568 
569  return lower_upper_CVs;
570 }

661 std::pair<unsigned long, unsigned long> CLookUpTable::FindNearestNeighbors(const su2double val_CV1,
662  const su2double val_CV2,
663  const unsigned long i_level) {
664  su2double min_distance = 1e99, second_distance = 1e99;
665  su2double norm_coeff_x = 1. / (*limits_table_x[i_level].second - *limits_table_x[i_level].first);
666  su2double norm_coeff_y = 1. / (*limits_table_y[i_level].second - *limits_table_y[i_level].first);
667  su2double val_CV1_norm = val_CV1 / (*limits_table_x[i_level].second - *limits_table_x[i_level].first);
668  su2double val_CV2_norm = val_CV2 / (*limits_table_y[i_level].second - *limits_table_y[i_level].first);
669 
670  const su2double* x_table = table_data[i_level][idx_CV1];
671  const su2double* y_table = table_data[i_level][idx_CV2];
672  unsigned long i_nearest = 0, i_second_nearest = 0;
673 
674  for (unsigned long i_point = 0; i_point < n_hull_points[i_level]; ++i_point) {
675  su2double next_x_norm = x_table[hull[i_level][i_point]] * norm_coeff_x;
676  su2double next_y_norm = y_table[hull[i_level][i_point]] * norm_coeff_y;
677 
678  su2double next_distance = pow(val_CV1_norm - next_x_norm, 2) + pow(val_CV2_norm - next_y_norm, 2);
679 
680  /* Check if distance to node is lower than current nearest or second nearest node. */
681  if (next_distance < min_distance) {
682  second_distance = min_distance;
683  min_distance = next_distance;
684 
685  i_second_nearest = i_nearest;
686  i_nearest = hull[i_level][i_point];
687  } else if ((next_distance > min_distance) && (next_distance < second_distance)) {
688  i_second_nearest = hull[i_level][i_point];
689 
690  second_distance = next_distance;
691  }
692  }
693  return make_pair(i_nearest, i_second_nearest);
694 }

696 unsigned long CLookUpTable::FindNearestNeighborOnHull(su2double val_CV1, su2double val_CV2, unsigned long i_level) {
697  su2double min_distance = 1.e99;
698  su2double next_distance = 1.e99;
699  unsigned long neighbor_id = 0;
700 
701  const su2double* x_table = table_data[i_level][idx_CV1];
702  const su2double* y_table = table_data[i_level][idx_CV2];
703 
704  su2double norm_coeff_x = 1. / (limits_table_x[i_level].second - limits_table_x[i_level].first);
705  su2double norm_coeff_y = 1. / (limits_table_y[i_level].second - limits_table_y[i_level].first);
706  su2double val_CV1_norm = val_CV1 / (limits_table_x[i_level].second - limits_table_x[i_level].first);
707  su2double val_CV2_norm = val_CV2 / (limits_table_y[i_level].second - limits_table_y[i_level].first);
708 
709  for (auto i_point = 0u; i_point < n_hull_points[i_level]; ++i_point) {
710  su2double next_x_norm = x_table[hull[i_level][i_point]] * norm_coeff_x;
711  su2double next_y_norm = y_table[hull[i_level][i_point]] * norm_coeff_y;
712 
713  next_distance = sqrt(pow(val_CV1_norm - next_x_norm, 2) + pow(val_CV2_norm - next_y_norm, 2));
714 
715  if (next_distance < min_distance) {
716  min_distance = next_distance;
717  neighbor_id = hull[i_level][i_point];
718  }
719  }
720  return neighbor_id;
721 }

723 void CLookUpTable::InterpolateToNearestNeighbors(const su2double val_CV1, const su2double val_CV2,
724  const std::vector<unsigned long>& idx_var,
725  std::vector<su2double>& var_vals, const unsigned long i_level) {
726  /* Interpolate data using distance-weighted averaging on the two nearest table nodes. */
727 
728  su2double norm_coeff_x = 1. / (*limits_table_x[i_level].second - *limits_table_x[i_level].first);
729  su2double norm_coeff_y = 1. / (*limits_table_y[i_level].second - *limits_table_y[i_level].first);
730  su2double val_CV1_norm = val_CV1 / (*limits_table_x[i_level].second - *limits_table_x[i_level].first);
731  su2double val_CV2_norm = val_CV2 / (*limits_table_y[i_level].second - *limits_table_y[i_level].first);
732 
733  std::pair<unsigned long, unsigned long> nearest_IDs = FindNearestNeighbors(val_CV1, val_CV2, i_level);
734  unsigned long i_nearest = nearest_IDs.first, i_second_nearest = nearest_IDs.second;
735  su2double x_nearest = table_data[i_level][idx_CV1][nearest_IDs.first],
736  x_second_nearest = table_data[i_level][idx_CV1][nearest_IDs.second],
737  y_nearest = table_data[i_level][idx_CV2][nearest_IDs.first],
738  y_second_nearest = table_data[i_level][idx_CV2][nearest_IDs.second];
739  su2double min_distance =
740  pow(val_CV1_norm - x_nearest * norm_coeff_x, 2) + pow(val_CV2_norm - y_nearest * norm_coeff_y, 2);
741  su2double second_distance =
742  pow(val_CV1_norm - x_second_nearest * norm_coeff_x, 2) + pow(val_CV2_norm - y_second_nearest * norm_coeff_y, 2);
743 
744  min_distance = sqrt(min_distance);
745  second_distance = sqrt(second_distance);
746 
747  /* Interpolate data using distance-weighted averaging */
748  su2double delimiter = (1.0 / min_distance) + (1.0 / second_distance);
749  for (auto iVar = 0u; iVar < var_vals.size(); iVar++) {
750  if (idx_var[iVar] == idx_null) {
751  var_vals[iVar] = 0;
752  } else {
753  su2double data_nearest = table_data[i_level][idx_var[iVar]][i_nearest],
754  data_second_nearest = table_data[i_level][idx_var[iVar]][i_second_nearest];
755  var_vals[iVar] =
756  (data_nearest * (1.0 / min_distance) + data_second_nearest * (1.0 / second_distance)) / delimiter;
757  }
758  }
759 }
