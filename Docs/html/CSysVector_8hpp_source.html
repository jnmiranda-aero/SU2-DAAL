36/*!
37 * \brief OpenMP worksharing construct used in CSysVector for loops.
38 * \note The loop will only run in parallel if methods are called from a
39 * parallel region (if not the results will still be correct).
40 * Static schedule to reduce overhead, chunk size determined at initialization.
41 * "nowait" clause is safe when calling CSysVector methods after each other
42 * as the loop size is the same. Methods of other classes that operate on a
43 * CSysVector and do not have the same work scheduling must use a
44 * SU2_OMP_BARRIER before using the vector.
45 */
46#ifdef HAVE_OMP
47#ifdef HAVE_OMP_SIMD
48#define CSYSVEC_PARFOR SU2_OMP_FOR_(simd schedule(static, omp_chunk_size) SU2_NOWAIT)
49#else
50#define CSYSVEC_PARFOR SU2_OMP_FOR_(schedule(static, omp_chunk_size) SU2_NOWAIT)
51#endif
52#define END_CSYSVEC_PARFOR END_SU2_OMP_FOR
53#else
54#define CSYSVEC_PARFOR SU2_OMP_SIMD
55#define END_CSYSVEC_PARFOR
56#endif

87 /*!
88 * \brief Helper to unpack (transpose) a SIMD input block.
89 */
90 template <size_t N, size_t nVar, class VecTypeSIMD, class F>
91 FORCEINLINE static void UnpackBlock(const VecTypeSIMD& in, simd::Array<F, N> mask, ScalarType out[][nVar]) {
92 static_assert(VecTypeSIMD::StaticSize, "This method requires static size vectors.");
93 for (size_t i = 0; i < nVar; ++i) {
94 SU2_OMP_SIMD_IF_NOT_AD
95 for (size_t k = 0; k < N; ++k) out[k][i] = mask[k] * in[i][k];
96 }

178 /*!
179 * \brief Set our values (resizing if required) by copying from other, the derivative information is lost.
180 * \param[in] other - source CSysVector
181 */
182 template <class T>
183 void PassiveCopy(const CSysVector<T>& other) {
184 /*--- This is a method and not the overload of an operator to make sure who
185 * calls it knows the consequence to the derivative information (lost) ---*/
186
187 /*--- check if self-assignment, otherwise perform deep copy ---*/
188 if ((const void*)this == (const void*)&other) return;
189
190 SU2_OMP_SAFE_GLOBAL_ACCESS(
191 Initialize(other.GetNBlk(), other.GetNBlkDomain(), other.GetNVar(), nullptr, true, false);)
192
193 CSYSVEC_PARFOR
194 for (auto i = 0ul; i < nElm; i++) vec_val[i] = SU2_TYPE::GetValue(other[i]);
195 END_CSYSVEC_PARFOR
196 }

290 /*!
291 * \brief Dot product between "this" and an expression.
292 * \param[in] expr - Expression.
293 * \return Result of dot product
294 */
295 template <class T>
296 ScalarType dot(const VecExpr::CVecExpr<T, ScalarType>& expr) const {
297 static ScalarType dotRes;
298 /*--- All threads get the same "view" of the vectors and shared variable. ---*/
299 SU2_OMP_SAFE_GLOBAL_ACCESS(dotRes = 0.0;)
300
301 /*--- Local dot product for each thread. ---*/
302 ScalarType sum = 0.0;
303
304 CSYSVEC_PARFOR
305 for (auto i = 0ul; i < nElmDomain; ++i) {
306 sum += vec_val[i] * expr.derived()[i];
307 }
308 END_CSYSVEC_PARFOR
309
310 /*--- Update shared variable with "our" partial sum. ---*/
311 atomicAdd(sum, dotRes);
312
313#ifdef HAVE_MPI
314 /*--- Reduce across all mpi ranks, only master thread communicates. ---*/
315 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
316 sum = dotRes;
317 const auto mpi_type = (sizeof(ScalarType) < sizeof(double)) ? MPI_FLOAT : MPI_DOUBLE;
318 SelectMPIWrapper<ScalarType>::W::Allreduce(&sum, &dotRes, 1, mpi_type, MPI_SUM, SU2_MPI::GetComm());
319 }
320 END_SU2_OMP_SAFE_GLOBAL_ACCESS
321#else
322 /*--- Make view of result consistent across threads. ---*/
323 SU2_OMP_BARRIER
324#endif
325
326 return dotRes;
327 }

357 /*!
358 * \brief Set "block" to the vector.
359 * \note Template param Overwrite can be set to false to update existing values.
360 * \param[in] iPoint - index of the point where set the residual.
361 * \param[in] block - Value to set to the residual.
362 * \param[in] alpha - Scale factor (axpy-type operation).
363 */
364 template <class VectorType, bool Overwrite = true>
365 FORCEINLINE void SetBlock(unsigned long iPoint, const VectorType& block, ScalarType alpha = 1) {
366 if (Overwrite) {
367 for (auto i = 0ul; i < nVar; ++i) vec_val[iPoint * nVar + i] = alpha * block[i];
368 } else {
369 for (auto i = 0ul; i < nVar; ++i) vec_val[iPoint * nVar + i] += alpha * block[i];
370 }
371 }

399 /*!
400 * \brief Vectorized version of SetBlock, sets multiple iPoint's.
401 * \param[in] iPoint - SIMD integer, the positions to update.
402 * \param[in] vector - Vector of SIMD scalars.
403 * \param[in] mask - Optional scale factor (axpy type operation).
404 * \note Nothing is updated if the mask is 0.
405 */
406 template <size_t N, class T, class VecTypeSIMD, class F = ScalarType>
407 FORCEINLINE void SetBlock(simd::Array<T, N> iPoint, const VecTypeSIMD& vector, simd::Array<F, N> mask = 1) {
408 /*--- "Transpose" and scale input vector. ---*/
409 constexpr size_t nVar = VecTypeSIMD::StaticSize;
410 assert(nVar == this->nVar);
411 ScalarType vec[N][nVar];
412 UnpackBlock(vector, mask, vec);
413
414 /*--- Update one by one skipping if mask is 0. ---*/
415 for (size_t k = 0; k < N; ++k) {
416 if (mask[k] == 0) continue;
417 SU2_OMP_SIMD
418 for (size_t i = 0; i < nVar; ++i) vec_val[iPoint[k] * nVar + i] = vec[k][i];
419 }
420 }
421
422 /*!
423 * \brief Vectorized version of UpdateBlocks, updates multiple i/jPoint's.
424 * \note See SIMD overload of SetBlock.
425 */
426 template <size_t N, class T, class VecTypeSIMD, class F = ScalarType>
427 FORCEINLINE void UpdateBlocks(simd::Array<T, N> iPoint, simd::Array<T, N> jPoint, const VecTypeSIMD& vector,
428 simd::Array<F, N> mask = 1) {
429 /*--- "Transpose" and scale input vector. ---*/
430 constexpr size_t nVar = VecTypeSIMD::StaticSize;
431 assert(nVar == this->nVar);
432 ScalarType vec[N][nVar];
433 UnpackBlock(vector, mask, vec);
434
435 /*--- Update one by one skipping if mask is 0. ---*/
436 for (size_t k = 0; k < N; ++k) {
437 if (mask[k] == 0) continue;
438 SU2_OMP_SIMD
439 for (size_t i = 0; i < nVar; ++i) {
440 vec_val[iPoint[k] * nVar + i] += vec[k][i];
441 vec_val[jPoint[k] * nVar + i] -= vec[k][i];
442 }
443 }
444 }
