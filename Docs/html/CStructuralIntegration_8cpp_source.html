34 CNumerics ******numerics_container, CConfig **config,
35 unsigned short RunTime_EqSystem, unsigned short iZone,

37

41
42
43 solver->Preprocessing(geometry[iZone][iInst][MESH_0],

69void CStructuralIntegration::Space_Integration_FEM(CGeometry *geometry, CSolver **solver_container,
70 CNumerics **numerics, CConfig *config,
71 unsigned short RunTime_EqSystem) {
72 const bool first_iter = (config->GetInnerIter() == 0);
73 const bool linear_analysis = (config->GetGeometricConditions() == STRUCT_DEFORMATION::SMALL);
74 const auto IterativeScheme = config->GetKind_SpaceIteScheme_FEA();
75
76 unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
77 CSolver* solver = solver_container[MainSolver];
78
79 /*--- Mass Matrix was computed during preprocessing, see notes therein. ---*/
80
81 if (linear_analysis) {
82 /*--- If the analysis is linear, only a the constitutive term of the stiffness matrix has to be computed. ---*/
83 solver->Compute_StiffMatrix(geometry, numerics, config);
84 }
85 else {
86 /*--- If the analysis is nonlinear the stress terms also need to be computed. ---*/
87 /*--- For full Newton-Raphson the stiffness matrix and the nodal term are updated every time. ---*/
88 if (IterativeScheme == STRUCT_SPACE_ITE::NEWTON) {
89 solver->Compute_StiffMatrix_NodalStressRes(geometry, numerics, config);
90 }
91
92 /*--- If the method is modified Newton-Raphson, the stiffness matrix is only computed once at the beginning
93 * of the time step, then only the Nodal Stress Term has to be computed on each iteration. ---*/
94 if (IterativeScheme == STRUCT_SPACE_ITE::MOD_NEWTON) {
95 if (first_iter)
96 solver->Compute_StiffMatrix_NodalStressRes(geometry, numerics, config);
97 else
98 solver->Compute_NodalStressRes(geometry, numerics, config);
99 }
100 }
101
102 /*--- Apply the NATURAL BOUNDARY CONDITIONS (loads). ---*/
103 /*--- If there are FSI loads, they have to be previously applied at other level involving both zones. ---*/
104
105 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
106 switch (config->GetMarker_All_KindBC(iMarker)) {
107
108 case LOAD_DIR_BOUNDARY:
109 solver->BC_Dir_Load(geometry, config, iMarker);
110 break;
111
112 case LOAD_SINE_BOUNDARY:
113 solver->BC_Sine_Load(geometry, config, iMarker);
114 break;
115
116 case LOAD_BOUNDARY:
117 solver->BC_Normal_Load(geometry, config, iMarker);
118 break;
119
120 case DAMPER_BOUNDARY:
121 solver->BC_Damper(geometry, config, iMarker);
122 break;
123 }
124 }
125
126}

128void CStructuralIntegration::Time_Integration_FEM(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics,
129 CConfig *config, unsigned short RunTime_EqSystem) {
130 unsigned short iMarker;
131
132 unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
133
134 /*--- Set the Jacobian according to the different time integration methods ---*/
135
136 switch (config->GetKind_TimeIntScheme_FEA()) {
137 case (STRUCT_TIME_INT::NEWMARK_IMPLICIT):
138 solver_container[MainSolver]->ImplicitNewmark_Iteration(geometry, numerics, config);
139 break;
140 case (STRUCT_TIME_INT::GENERALIZED_ALPHA):
141 solver_container[MainSolver]->GeneralizedAlpha_Iteration(geometry, numerics, config);
142 break;
143 }
144
145 /*--- Apply ESSENTIAL BOUNDARY CONDITIONS ---*/
146
147 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
148 switch (config->GetMarker_All_KindBC(iMarker)) {
149 case CLAMPED_BOUNDARY:
150 solver_container[MainSolver]->BC_Clamped(geometry, config, iMarker);
151 break;
152 case SYMMETRY_PLANE:
153 solver_container[MainSolver]->BC_Sym_Plane(geometry, config, iMarker);
154 break;
155 case DISP_DIR_BOUNDARY:
156 solver_container[MainSolver]->BC_DispDir(geometry, config, iMarker);
157 break;
158 }
159 }
160
161 /*--- Solve linear system ---*/
162
163 solver_container[MainSolver]->Solve_System(geometry, config);
164
165 /*--- Update solution ---*/
166
167 switch (config->GetKind_TimeIntScheme_FEA()) {
168 case (STRUCT_TIME_INT::NEWMARK_IMPLICIT):
169 solver_container[MainSolver]->ImplicitNewmark_Update(geometry, config);
170 break;
171 case (STRUCT_TIME_INT::GENERALIZED_ALPHA):
172 solver_container[MainSolver]->GeneralizedAlpha_UpdateDisp(geometry, config);
173 break;
174 }
175
176 /*--- Reinforce ESSENTIAL BOUNDARY CONDITIONS: avoids accumulation of numerical error ---*/
177
178 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
179 switch (config->GetMarker_All_KindBC(iMarker)) {
180 case CLAMPED_BOUNDARY:
181 solver_container[MainSolver]->BC_Clamped_Post(geometry, config, iMarker);
182 break;
183 }
184 }

188void CStructuralIntegration::SetDualTime_Solver(const CGeometry *geometry, CSolver *solver, const CConfig *config, unsigned short iMesh) {
189
190 const bool fsi = config->GetFSI_Simulation();
191
192 /*--- Update the solution according to the integration scheme used ---*/
193
194 switch (config->GetKind_TimeIntScheme_FEA()) {
195 case (STRUCT_TIME_INT::NEWMARK_IMPLICIT):
196 if (fsi && config->GetRelaxation()) solver->ImplicitNewmark_Relaxation(geometry, config);
197 break;
198 case (STRUCT_TIME_INT::GENERALIZED_ALPHA):
199 solver->GeneralizedAlpha_UpdateSolution(geometry, config);
200 solver->GeneralizedAlpha_UpdateLoads(geometry, config);
201 break;
202 }
203
204 /*--- Store the solution at t+1 as solution at t, both for the local points and for the halo points ---*/
205
206 solver->GetNodes()->Set_Solution_time_n();
207
208 /*--- If FSI problem, save the last Aitken relaxation parameter of the previous time step ---*/
209
210 if (fsi) solver->SetWAitken_Dyn_tn1();
