118 void CMMSNSUnitQuadSolution::GetSolution(const su2double* val_coords, const su2double val_t,
119  su2double* val_solution) const {
120  /* Easier storage of the x- and y-coordinates. */
121  const su2double x = val_coords[0];
122  const su2double y = val_coords[1];
123 
124  /* Determine the solution for the density, velocity
125  components and pressure. */
126  const su2double LInv = 1.0 / L;
127  const su2double PiLInv = PI_NUMBER * LInv;
128  const su2double PiL2Inv = PiLInv * LInv;
129 
130  const su2double rho = rho_0 + rho_x * sin(a_rhox * PiLInv * x) + rho_y * cos(a_rhoy * PiLInv * y) +
131  rho_xy * cos(a_rhoxy * PiL2Inv * x * y);
132 
133  const su2double u =
134  u_0 + u_x * sin(a_ux * PiLInv * x) + u_y * cos(a_uy * PiLInv * y) + u_xy * cos(a_uxy * PiL2Inv * x * y);
135 
136  const su2double v =
137  v_0 + v_x * cos(a_vx * PiLInv * x) + v_y * sin(a_vy * PiLInv * y) + v_xy * cos(a_vxy * PiL2Inv * x * y);
138 
139  const su2double p =
140  P_0 + P_x * cos(a_Px * PiLInv * x) + P_y * sin(a_Py * PiLInv * y) + P_xy * sin(a_Pxy * PiL2Inv * x * y);
141 
142  /* Compute the conservative variables from the primitive ones.
143  Note that the implementation below is valid for both 2D and 3D. */
144  val_solution[0] = rho;
145  val_solution[1] = rho * u;
146  val_solution[2] = rho * v;
147  val_solution[3] = 0.0;
148  val_solution[nDim + 1] = p / (Gamma - 1.0) + 0.5 * rho * (u * u + v * v);
149 }
