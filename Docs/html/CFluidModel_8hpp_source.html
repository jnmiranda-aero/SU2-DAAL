172  inline virtual su2double GetThermalConductivity() {
173  ThermalConductivity->SetConductivity(Temperature, Density, Mu, Mu_Turb, Cp, dmudrho_T, dmudT_rho);
174  Kt = ThermalConductivity->GetConductivity();
175  dktdrho_T = ThermalConductivity->Getdktdrho_T();
176  dktdT_rho = ThermalConductivity->GetdktdT_rho();
