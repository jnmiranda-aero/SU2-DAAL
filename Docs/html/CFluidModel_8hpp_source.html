155 inline virtual su2double GetLaminarViscosity() {
156 LaminarViscosity->SetViscosity(Temperature, Density);
157 Mu = LaminarViscosity->GetViscosity();
158 dmudrho_T = LaminarViscosity->Getdmudrho_T();
159 dmudT_rho = LaminarViscosity->GetdmudT_rho();
160 return Mu;
161 }

163 /*!
164 * \brief Get fluid thermal conductivity.
165 */
166 inline virtual su2double GetThermalConductivity() {
167 ThermalConductivity->SetConductivity(Temperature, Density, Mu, Mu_Turb, Cp, dmudrho_T, dmudT_rho);
168 Kt = ThermalConductivity->GetConductivity();
169 dktdrho_T = ThermalConductivity->Getdktdrho_T();
170 dktdT_rho = ThermalConductivity->GetdktdT_rho();
171 return Kt;
172 }
