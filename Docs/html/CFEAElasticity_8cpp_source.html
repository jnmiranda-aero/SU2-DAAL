32CFEAElasticity::CFEAElasticity(unsigned short val_nDim, unsigned short val_nVar,
33 const CConfig *config) : CNumerics() {
34
35 nDim = val_nDim;
36 nVar = val_nVar;
37
38 bool body_forces = config->GetDeadLoad(); // Body forces (dead loads).
39 bool pseudo_static = config->GetPseudoStatic();
40
41 unsigned short iVar;
42
43 /*--- Initialize vector structures for multiple material definition ---*/
44 const auto nProp = config->GetnElasticityMat();
45
46 E_i = new su2double[nProp];
47 for (iVar = 0; iVar < nProp; iVar++)
48 E_i[iVar] = config->GetElasticyMod(iVar);
49
50 Nu_i = new su2double[nProp];
51 for (iVar = 0; iVar < nProp; iVar++)
52 Nu_i[iVar] = config->GetPoissonRatio(iVar);
53
54 Rho_s_i = new su2double[nProp]; // For inertial effects
55 Rho_s_DL_i = new su2double[nProp]; // For dead loads
56
57 for (iVar = 0; iVar < nProp; iVar++) {
58 Rho_s_DL_i[iVar] = config->GetMaterialDensity(iVar);
59 Rho_s_i[iVar] = pseudo_static ? 0.0 : config->GetMaterialDensity(iVar);
60 }
61
62 // Initialization
63 E = E_i[0];
64 Nu = Nu_i[0];
65 Rho_s = Rho_s_i[0];
66 Rho_s_DL = Rho_s_DL_i[0];
67
68 Compute_Lame_Parameters();
69
70 // Auxiliary vector for body forces (dead load)
71 FAux_Dead_Load = nullptr;
72 if (body_forces) FAux_Dead_Load = new su2double [nDim];
73
74 plane_stress = (config->GetElas2D_Formulation() == STRUCT_2DFORM::PLANE_STRESS);
75
76 KAux_ab = new su2double* [nDim];
77 for (iVar = 0; iVar < nDim; iVar++) {
78 KAux_ab[iVar] = new su2double[nDim];
79 }
80
81 unsigned short nStrain = (nDim==2) ? DIM_STRAIN_2D : DIM_STRAIN_3D;
82 unsigned short nNodes = (nDim==2) ? NNODES_2D : NNODES_3D;
83
84 Ba_Mat = new su2double* [nStrain];
85 Bb_Mat = new su2double* [nStrain];
86 D_Mat = new su2double* [nStrain];
87 Ni_Vec = new su2double [nNodes];
88 GradNi_Ref_Mat = new su2double* [nNodes];
89 GradNi_Curr_Mat = new su2double* [nNodes];
90 for (iVar = 0; iVar < nStrain; iVar++) {
91 Ba_Mat[iVar] = new su2double[nDim];
92 Bb_Mat[iVar] = new su2double[nDim];
93 D_Mat[iVar] = new su2double[nStrain];
94 }
95 for (iVar = 0; iVar < nNodes; iVar++) {
96 GradNi_Ref_Mat[iVar] = new su2double[nDim];
97 GradNi_Curr_Mat[iVar] = new su2double[nDim];
98 }
99
100 DV_Val = nullptr;
101 n_DV = 0;
102 switch (config->GetDV_FEA()) {
103 case YOUNG_MODULUS:
104 case POISSON_RATIO:
105 case DENSITY_VAL:
106 case DEAD_WEIGHT:
107 case ELECTRIC_FIELD:
108 ReadDV(config);
109 for (unsigned short iDV = 0; iDV < n_DV; iDV++)
110 if ((config->GetDirectDiff() == D_YOUNG) ||
111 (config->GetDirectDiff() == D_POISSON) ||
112 (config->GetDirectDiff() == D_RHO) ||
113 (config->GetDirectDiff() == D_RHO_DL) ||
114 (config->GetDirectDiff() == D_EFIELD)){
115 SU2_TYPE::SetDerivative(DV_Val[config->GetnID_DV()],1.0);
116 }
117 break;
118
119 default:
120 switch (config->GetDirectDiff()){
121 case D_YOUNG:
122 SU2_TYPE::SetDerivative(E_i[config->GetnID_DV()],1.0);
123 break;
124 case D_POISSON:
125 SU2_TYPE::SetDerivative(Nu_i[config->GetnID_DV()],1.0);
126 break;
127 case D_RHO:
128 SU2_TYPE::SetDerivative(Rho_s_i[config->GetnID_DV()],1.0);
129 break;
130 case D_RHO_DL:
131 SU2_TYPE::SetDerivative(Rho_s_DL_i[config->GetnID_DV()],1.0);
132 break;
133 }
134
135 }
136}

177void CFEAElasticity::Compute_Mass_Matrix(CElement *element, const CConfig *config) {
178
179 /*--- Initialize values for the material model considered ---*/
180 SetElement_Properties(element, config);
181 /*-----------------------------------------------------------*/
182
183 unsigned short iGauss, nGauss;
184 unsigned short iNode, jNode, nNode;
185
186 su2double Weight, Jac_X, val_Mab;
187
188 /*--- Register pre-accumulation inputs, density and reference coords. ---*/
189 AD::StartPreacc();
190 AD::SetPreaccIn(Rho_s);
191 element->SetPreaccIn_Coords(false);
192
193 element->ClearElement(); /*--- Restarts the element: avoids adding over previous results in other elements --*/
194 element->ComputeGrad_Linear(); /*--- Need to compute the gradients to obtain the Jacobian ---*/
195
196 nNode = element->GetnNodes();
197 nGauss = element->GetnGaussPoints();
198
199 for (iGauss = 0; iGauss < nGauss; iGauss++) {
200
201 Weight = element->GetWeight(iGauss);
202 Jac_X = element->GetJ_X(iGauss); /*--- The mass matrix is computed in the reference configuration ---*/
203
204 /*--- Retrieve the values of the shape functions for each node ---*/
205 /*--- This avoids repeated operations ---*/
206 for (iNode = 0; iNode < nNode; iNode++) {
207 Ni_Vec[iNode] = element->GetNi(iNode,iGauss);
208 }
209
210 for (iNode = 0; iNode < nNode; iNode++) {
211
212 /*--- Assumming symmetry ---*/
213 for (jNode = iNode; jNode < nNode; jNode++) {
214
215 val_Mab = Weight * Ni_Vec[iNode] * Ni_Vec[jNode] * Jac_X * Rho_s;
216
217 element->Add_Mab(iNode, jNode, val_Mab);
218 /*--- Symmetric terms --*/
219 if (iNode != jNode) {
220 element->Add_Mab(jNode, iNode, val_Mab);
221 }
222
223 }
224
225 }
226
227 }
228
229 /*--- Register the mass matrix as preaccumulation output. ---*/
230 element->SetPreaccOut_Mab();
231 AD::EndPreacc();
232
233}

236void CFEAElasticity::Compute_Dead_Load(CElement *element, const CConfig *config) {
237
238 /*--- Initialize values for the material model considered ---*/
239 SetElement_Properties(element, config);
240 /*-----------------------------------------------------------*/
241
242 /*--- Register pre-accumulation inputs, density and reference coords. ---*/
243 AD::StartPreacc();
244 AD::SetPreaccIn(Rho_s_DL);
245 element->SetPreaccIn_Coords(false);
246
247 unsigned short iGauss, nGauss;
248 unsigned short iNode, iDim, nNode;
249
250 su2double Weight, Jac_X;
251
252 /* -- Gravity directionality:
253 * -- For 2D problems, we assume the direction for gravity is -y
254 * -- For 3D problems, we assume the direction for gravity is -z
255 */
256 su2double g_force[3] = {0.0,0.0,0.0};
257
258 if (nDim == 2) g_force[1] = -1*STANDARD_GRAVITY;
259 else if (nDim == 3) g_force[2] = -1*STANDARD_GRAVITY;
260
261 element->ClearElement(); /*--- Restart the element to avoid adding over previous results. --*/
262 element->ComputeGrad_Linear(); /*--- Need to compute the gradients to obtain the Jacobian. ---*/
263
264 nNode = element->GetnNodes();
265 nGauss = element->GetnGaussPoints();
266
267 for (iGauss = 0; iGauss < nGauss; iGauss++) {
268
269 Weight = element->GetWeight(iGauss);
270 Jac_X = element->GetJ_X(iGauss); /*--- The dead load is computed in the reference configuration ---*/
271
272 /*--- Retrieve the values of the shape functions for each node ---*/
273 /*--- This avoids repeated operations ---*/
274 for (iNode = 0; iNode < nNode; iNode++) {
275 Ni_Vec[iNode] = element->GetNi(iNode,iGauss);
276 }
277
278 for (iNode = 0; iNode < nNode; iNode++) {
279
280 for (iDim = 0; iDim < nDim; iDim++) {
281 FAux_Dead_Load[iDim] = Weight * Ni_Vec[iNode] * Jac_X * Rho_s_DL * g_force[iDim];
282 }
283
284 element->Add_FDL_a(iNode, FAux_Dead_Load);
285
286 }
287
288 }
289
290 /*--- Register the dead load as preaccumulation output. ---*/
291 element->SetPreaccOut_FDL_a();
292 AD::EndPreacc();
293
294}

297void CFEAElasticity::SetElement_Properties(const CElement *element, const CConfig *config) {
298
299 /*--- These variables are set as preaccumulation inputs in Compute_Tangent_Matrix and
300 Compute_NodalStress_Term, if you add variables here be sure to register them in those routines too. ---*/
301 E = E_i[element->Get_iProp()];
302 Nu = Nu_i[element->Get_iProp()];
303 Rho_s = Rho_s_i[element->Get_iProp()];
304 Rho_s_DL = Rho_s_DL_i[element->Get_iProp()];
305
306 switch (config->GetDV_FEA()) {
307 case YOUNG_MODULUS:
308 E = DV_Val[element->Get_iDV()] * E;
309 break;
310 case POISSON_RATIO:
311 Nu = DV_Val[element->Get_iDV()] * Nu;
312 break;
313 case DENSITY_VAL:
314 Rho_s = DV_Val[element->Get_iDV()] * Rho_s;
315 break;
316 case DEAD_WEIGHT:
317 Rho_s_DL = DV_Val[element->Get_iDV()] * Rho_s_DL;
318 break;
319 }
320
321 Compute_Lame_Parameters();
322
323}

326void CFEAElasticity::ReadDV(const CConfig *config) {
327
328 int rank = SU2_MPI::GetRank();
329 bool master_node = false;
330 SU2_OMP_MASTER
331 master_node = (rank == MASTER_NODE);
332 END_SU2_OMP_MASTER
333
334 unsigned long index;
335
336 string filename;
337 ifstream properties_file;
338
339 /*--- Choose the filename of the design variable ---*/
340
341 string input_name;
342
343 switch (config->GetDV_FEA()) {
344 case YOUNG_MODULUS:
345 input_name = "dv_young.opt";
346 break;
347 case POISSON_RATIO:
348 input_name = "dv_poisson.opt";
349 break;
350 case DENSITY_VAL:
351 case DEAD_WEIGHT:
352 input_name = "dv_density.opt";
353 break;
354 case ELECTRIC_FIELD:
355 input_name = "dv_efield.opt";
356 break;
357 default:
358 input_name = "dv.opt";
359 break;
360 }
361
362 filename = input_name;
363
364 if (master_node) cout << "Filename: " << filename << "." << endl;
365
366 properties_file.open(filename.data(), ios::in);
367
368 /*--- In case there is no file, all elements get the same property (0) ---*/
369
370 if (properties_file.fail()) {
371
372 if (master_node)
373 cout << "There is no design variable file." << endl;
374
375 n_DV = 1;
376 DV_Val = new su2double[n_DV];
377 for (unsigned short iDV = 0; iDV < n_DV; iDV++)
378 DV_Val[iDV] = 1.0;
379
380 }
381 else{
382
383 string text_line;
384
385 /*--- First pass: determine number of design variables ---*/
386
387 unsigned short iDV = 0;
388
389 /*--- Skip the first line: it is the header ---*/
390
391 getline (properties_file, text_line);
392
393 while (getline (properties_file, text_line)) iDV++;
394
395 /*--- Close the restart file ---*/
396
397 properties_file.close();
398
399 n_DV = iDV;
400 DV_Val = new su2double[n_DV];
401
402 /*--- Reopen the file (TODO: improve this) ---*/
403
404 properties_file.open(filename.data(), ios::in);
405
406 /*--- Skip the first line: it is the header ---*/
407
408 getline (properties_file, text_line);
409
410 iDV = 0;
411 while (getline (properties_file, text_line)) {
412
413 istringstream point_line(text_line);
414
415 point_line >> index >> DV_Val[iDV];
416
417 iDV++;
418
419 }
420
421 /*--- Close the restart file ---*/
422
423 properties_file.close();
424
425 }
