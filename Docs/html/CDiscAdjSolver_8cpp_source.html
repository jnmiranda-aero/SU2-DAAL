32 CDiscAdjSolver::CDiscAdjSolver(CGeometry *geometry, CConfig *config, CSolver *direct_sol,
33  unsigned short Kind_Solver, unsigned short iMesh) : CSolver() {
34 
35  /*-- Store some information about direct solver ---*/
36 
37  KindDirect_Solver = Kind_Solver;
38  direct_solver = direct_sol;
39 
40  adjoint = true;
41 
42  nVar = direct_solver->GetnVar();
43  nDim = geometry->GetnDim();
44 
45  nMarker = config->GetnMarker_All();
46  nPoint = geometry->GetnPoint();
47  nPointDomain = geometry->GetnPointDomain();
48 
49  omp_chunk_size = computeStaticChunkSize(nPoint, omp_get_max_threads(), OMP_MAX_SIZE);
50 
51  /*--- Define some auxiliary vectors related to the residual ---*/
52 
53  Residual_RMS.resize(nVar,1.0);
54  Residual_Max.resize(nVar,1.0);
55  Point_Max.resize(nVar,0);
56  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
57 
58  /*--- Define some auxiliary vectors related to the residual for problems with a BGS strategy---*/
59 
60  if (config->GetMultizone_Residual()){
61 
62  Residual_BGS.resize(nVar,1.0);
63  Residual_Max_BGS.resize(nVar,1.0);
64  Point_Max_BGS.resize(nVar,0);
65  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
66  }
67 
68  /*--- Sensitivity definition and coefficient in all the markers ---*/
69 
70  CSensitivity.resize(nMarker);
71  for (auto iMarker = 0ul; iMarker < nMarker; iMarker++) {
72  const auto nVertex = geometry->nVertex[iMarker];
73  CSensitivity[iMarker].resize(nVertex, 0.0);
74  }
75 
76  Sens_Geo.resize(config->GetnMarker_Monitoring(), 0.0);
77 
78  /*--- Initialize the discrete adjoint solution to zero everywhere. ---*/
79 
80  if (nVar > MAXNVAR) {
81  SU2_MPI::Error("Oops! The CDiscAdjSolver static array sizes are not large enough.",CURRENT_FUNCTION);
82  }
83 
84  vector<su2double> Solution(nVar,1e-16);
85  nodes = new CDiscAdjVariable(Solution.data(), nPoint, nDim, nVar, config);
86  SetBaseClassPointerToNodes();
87 
88  /*--- Allocate extra solution variables, if any are in use. ---*/
89 
90  const auto nVarExtra = direct_solver->RegisterSolutionExtra(true, config);
91  nodes->AllocateAdjointSolutionExtra(nVarExtra);
92 
93  switch(KindDirect_Solver){
94  case RUNTIME_FLOW_SYS:
95  SolverName = "ADJ.FLOW";
96  break;
97  case RUNTIME_HEAT_SYS:
98  SolverName = "ADJ.HEAT";
99  break;
100  case RUNTIME_TURB_SYS:
101  SolverName = "ADJ.TURB";
102  break;
103  case RUNTIME_SPECIES_SYS:
104  SolverName = "ADJ.SPECIES";
105  break;
106  case RUNTIME_RADIATION_SYS:
107  SolverName = "ADJ.RAD";
108  break;
109  default:
110  SolverName = "ADJ.SOL";
111  break;
112  }
113 }

174 void CDiscAdjSolver::RegisterVariables(CGeometry *geometry, CConfig *config, bool reset) {
175 
176  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
177 
178  /*--- Register farfield values as input ---*/
179 
180  if((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) && (KindDirect_Solver == RUNTIME_FLOW_SYS && !config->GetBoolTurbomachinery())) {
181 
182  su2double Velocity_Ref = config->GetVelocity_Ref();
183  Alpha = config->GetAoA()*PI_NUMBER/180.0;
184  Beta = config->GetAoS()*PI_NUMBER/180.0;
185  Mach = config->GetMach();
186  Pressure = config->GetPressure_FreeStreamND();
187  Temperature = config->GetTemperature_FreeStreamND();
188 
189  su2double SoundSpeed = 0.0;
190 
191  if (nDim == 2) { SoundSpeed = config->GetVelocity_FreeStreamND()[0]*Velocity_Ref/(cos(Alpha)*Mach); }
192  if (nDim == 3) { SoundSpeed = config->GetVelocity_FreeStreamND()[0]*Velocity_Ref/(cos(Alpha)*cos(Beta)*Mach); }
193 
194  if (!reset) {
195  AD::RegisterInput(Mach);
196  AD::RegisterInput(Alpha);
197  AD::RegisterInput(Temperature);
198  AD::RegisterInput(Pressure);
199  }
200 
201  /*--- Recompute the free stream velocity ---*/
202 
203  if (nDim == 2) {
204  config->GetVelocity_FreeStreamND()[0] = cos(Alpha)*Mach*SoundSpeed/Velocity_Ref;
205  config->GetVelocity_FreeStreamND()[1] = sin(Alpha)*Mach*SoundSpeed/Velocity_Ref;
206  }
207  if (nDim == 3) {
208  config->GetVelocity_FreeStreamND()[0] = cos(Alpha)*cos(Beta)*Mach*SoundSpeed/Velocity_Ref;
209  config->GetVelocity_FreeStreamND()[1] = sin(Beta)*Mach*SoundSpeed/Velocity_Ref;
210  config->GetVelocity_FreeStreamND()[2] = sin(Alpha)*cos(Beta)*Mach*SoundSpeed/Velocity_Ref;
211  }
212 
213  config->SetTemperature_FreeStreamND(Temperature);
214  direct_solver->SetTemperature_Inf(Temperature);
215  config->SetPressure_FreeStreamND(Pressure);
216  direct_solver->SetPressure_Inf(Pressure);
217 
218  }
219 
220  if ((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) && (KindDirect_Solver == RUNTIME_FLOW_SYS) && config->GetBoolTurbomachinery()){
221 
222  BPressure = config->GetPressureOut_BC();
223  Temperature = config->GetTotalTemperatureIn_BC();
224 
225  if (!reset){
226  AD::RegisterInput(BPressure);
227  AD::RegisterInput(Temperature);
228  }
229 
230  config->SetPressureOut_BC(BPressure);
231  config->SetTotalTemperatureIn_BC(Temperature);
232  }
233 
234  /*--- Register incompressible initialization values as input ---*/
235 
236  if ((config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
237  ((KindDirect_Solver == RUNTIME_FLOW_SYS &&
238  (!config->GetBoolTurbomachinery())))) {
239 
240  /*--- Access the velocity (or pressure) and temperature at the
241  inlet BC and the back pressure at the outlet. Note that we are
242  assuming that have internal flow, which will be true for the
243  majority of cases. External flows with far-field BCs will report
244  zero for these sensitivities. ---*/
245 
246  ModVel = config->GetIncInlet_BC();
247  BPressure = config->GetIncPressureOut_BC();
248  Temperature = config->GetIncTemperature_BC();
249 
250  /*--- Register the variables for AD. ---*/
251 
252  if (!reset) {
253  AD::RegisterInput(ModVel);
254  AD::RegisterInput(BPressure);
255  AD::RegisterInput(Temperature);
256  }
257 
258  /*--- Set the BC values in the config class. ---*/
259 
260  config->SetIncInlet_BC(ModVel);
261  config->SetIncPressureOut_BC(BPressure);
262  config->SetIncTemperature_BC(Temperature);
263 
264  }
265 
266  /*--- Register incompressible radiation values as input ---*/
267 
268  if ((config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
269  ((KindDirect_Solver == RUNTIME_RADIATION_SYS &&
270  (!config->GetBoolTurbomachinery())))) {
271 
272  /*--- Access the nondimensional freestream temperature. ---*/
273 
274  TemperatureRad = config->GetTemperature_FreeStreamND();
275 
276  /*--- Register the variables for AD. ---*/
277 
278  if (!reset) {
279  AD::RegisterInput(TemperatureRad);
280  }
281 
282  /*--- Set the temperature at infinity in the direct solver class. ---*/
283 
284  direct_solver->SetTemperature_Inf(TemperatureRad);
285 
286  }
287 
288  /*--- Here it is possible to register other variables as input that influence the flow solution
289  * and thereby also the objective function. The adjoint values (i.e. the derivatives) can be
290  * extracted in the ExtractAdjointVariables routine. ---*/
291 
292  }
293  END_SU2_OMP_SAFE_GLOBAL_ACCESS
294 }

305 void CDiscAdjSolver::ExtractAdjoint_Solution(CGeometry *geometry, CConfig *config, bool CrossTerm) {
306 
307  const bool time_n1_needed = config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND;
308  const bool time_n_needed = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) || time_n1_needed;
309 
310  const su2double relax = (config->GetInnerIter()==0) ? 1.0 : config->GetRelaxation_Factor_Adjoint();
311 
312  /*--- Thread-local residual variables. ---*/
313  su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
314  unsigned long idxMax[MAXNVAR] = {0};
315 
316  /*--- Set the old solution and compute residuals. ---*/
317 
318  if (!config->GetMultizone_Problem()) nodes->Set_OldSolution();
319 
320  AD::BeginUseAdjoints();
321 
322  SU2_OMP_FOR_STAT(omp_chunk_size)
323  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
324 
325  /*--- Extract the adjoint solution ---*/
326 
327  su2double Solution[MAXNVAR] = {0.0};
328  direct_solver->GetNodes()->GetAdjointSolution(iPoint,Solution);
329 
330  /*--- Relax and store the adjoint solution, compute the residuals. ---*/
331 
332  for (auto iVar = 0u; iVar < nVar; iVar++) {
333  su2double residual = Solution[iVar]-nodes->GetSolution_Old(iPoint,iVar);
334  nodes->AddSolution(iPoint, iVar, relax*residual);
335 
336  if (iPoint < nPointDomain) {
337  /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
338  ResidualReductions_PerThread(iPoint,iVar,residual,resRMS,resMax,idxMax);
339  }
340  }
341  }
342  END_SU2_OMP_FOR
343 
344  AD::EndUseAdjoints();
345 
346  direct_solver->ExtractAdjoint_SolutionExtra(nodes->GetSolutionExtra(), config);
347 
348  /*--- Residuals and time_n terms are not needed when evaluating multizone cross terms. ---*/
349  if (CrossTerm) return;
350 
351  /*--- "Add" residuals from all threads to global residual variables. ---*/
352  ResidualReductions_FromAllThreads(geometry, config, resRMS,resMax,idxMax);
353 
354  SU2_OMP_MASTER {
355  SetIterLinSolver(direct_solver->System.GetIterations());
356  SetResLinSolver(direct_solver->System.GetResidual());
357  }
358  END_SU2_OMP_MASTER
359 
360  /*--- Extract and store the adjoint of the primal solution at time n ---*/
361  if (time_n_needed) {
362  AD::BeginUseAdjoints();
363 
364  SU2_OMP_FOR_STAT(omp_chunk_size)
365  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
366  su2double Solution[MAXNVAR] = {0.0};
367  direct_solver->GetNodes()->GetAdjointSolution_time_n(iPoint,Solution);
368  nodes->Set_Solution_time_n(iPoint,Solution);
369  }
370  END_SU2_OMP_FOR
371 
372  AD::EndUseAdjoints();
373  }
374 
375  /*--- Extract and store the adjoint of the primal solution at time n-1 ---*/
376  if (time_n1_needed) {
377  AD::BeginUseAdjoints();
378 
379  SU2_OMP_FOR_STAT(omp_chunk_size)
380  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
381  su2double Solution[MAXNVAR] = {0.0};
382  direct_solver->GetNodes()->GetAdjointSolution_time_n1(iPoint,Solution);
383  nodes->Set_Solution_time_n1(iPoint,Solution);
384  }
385  END_SU2_OMP_FOR
386 
387  AD::EndUseAdjoints();
388  }
389

392 void CDiscAdjSolver::ExtractAdjoint_Variables(CGeometry *geometry, CConfig *config) {
393 
394  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
395 
396  /*--- Extract the adjoint values of the farfield values ---*/
397 
398  if ((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) && (KindDirect_Solver == RUNTIME_FLOW_SYS) && !config->GetBoolTurbomachinery()) {
399  su2double Local_Sens_Press, Local_Sens_Temp, Local_Sens_AoA, Local_Sens_Mach;
400 
401  Local_Sens_Mach = SU2_TYPE::GetDerivative(Mach);
402  Local_Sens_AoA = SU2_TYPE::GetDerivative(Alpha);
403  Local_Sens_Temp = SU2_TYPE::GetDerivative(Temperature);
404  Local_Sens_Press = SU2_TYPE::GetDerivative(Pressure);
405 
406  SU2_MPI::Allreduce(&Local_Sens_Mach, &Total_Sens_Mach, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
407  SU2_MPI::Allreduce(&Local_Sens_AoA, &Total_Sens_AoA, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
408  SU2_MPI::Allreduce(&Local_Sens_Temp, &Total_Sens_Temp, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
409  SU2_MPI::Allreduce(&Local_Sens_Press, &Total_Sens_Press, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
410  }
411 
412  if ((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) && (KindDirect_Solver == RUNTIME_FLOW_SYS) && config->GetBoolTurbomachinery()){
413  su2double Local_Sens_BPress, Local_Sens_Temperature;
414 
415  Local_Sens_BPress = SU2_TYPE::GetDerivative(BPressure);
416  Local_Sens_Temperature = SU2_TYPE::GetDerivative(Temperature);
417 
418  SU2_MPI::Allreduce(&Local_Sens_BPress, &Total_Sens_BPress, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
419  SU2_MPI::Allreduce(&Local_Sens_Temperature, &Total_Sens_Temp, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
420  }
421 
422  if ((config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
423  (KindDirect_Solver == RUNTIME_FLOW_SYS &&
424  (!config->GetBoolTurbomachinery()))) {
425 
426  su2double Local_Sens_ModVel, Local_Sens_BPress, Local_Sens_Temp;
427 
428  Local_Sens_ModVel = SU2_TYPE::GetDerivative(ModVel);
429  Local_Sens_BPress = SU2_TYPE::GetDerivative(BPressure);
430  Local_Sens_Temp = SU2_TYPE::GetDerivative(Temperature);
431 
432  SU2_MPI::Allreduce(&Local_Sens_ModVel, &Total_Sens_ModVel, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
433  SU2_MPI::Allreduce(&Local_Sens_BPress, &Total_Sens_BPress, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
434  SU2_MPI::Allreduce(&Local_Sens_Temp, &Total_Sens_Temp, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
435  }
436 
437  if ((config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
438  (KindDirect_Solver == RUNTIME_RADIATION_SYS &&
439  (!config->GetBoolTurbomachinery()))) {
440 
441  su2double Local_Sens_Temp_Rad;
442  Local_Sens_Temp_Rad = SU2_TYPE::GetDerivative(TemperatureRad);
443 
444  SU2_MPI::Allreduce(&Local_Sens_Temp_Rad, &Total_Sens_Temp_Rad, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
445 
446  /*--- Store it in the Total_Sens_Temp container so it's accessible without the need of a new method ---*/
447  Total_Sens_Temp = Total_Sens_Temp_Rad;
448 
449  }
450 
451  /*--- Extract here the adjoint values of everything else that is registered as input in RegisterInput. ---*/
452 
453  }
454  END_SU2_OMP_SAFE_GLOBAL_ACCESS
455 }

457 void CDiscAdjSolver::SetAdjoint_Output(CGeometry *geometry, CConfig *config) {
458 
459  const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST ||
460  config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
461  const bool multizone = config->GetMultizone_Problem();
462 
463  /*--- Local container to manipulate the adjoint solution. ---*/
464  su2double Solution[MAXNVAR] = {0.0};
465 
466  SU2_OMP_FOR_STAT(omp_chunk_size)
467  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
468 
469  /*--- Get and store the adjoint solution of a point. ---*/
470  for (auto iVar = 0u; iVar < nVar; iVar++) {
471  Solution[iVar] = nodes->GetSolution(iPoint,iVar);
472  }
473 
474  /*--- Add dual time contributions to the adjoint solution. Two terms stored for DT-2nd-order. ---*/
475  if (dual_time && !multizone) {
476  for (auto iVar = 0u; iVar < nVar; iVar++) {
477  Solution[iVar] += nodes->GetDual_Time_Derivative(iPoint,iVar);
478  }
479  }
480 
481  /*--- Set the adjoint values of the primal solution. ---*/
482 
483  direct_solver->GetNodes()->SetAdjointSolution(iPoint,Solution);
484  }
485  END_SU2_OMP_FOR
486 
487  direct_solver->SetAdjoint_SolutionExtra(nodes->GetSolutionExtra(), config);
488 }

490 void CDiscAdjSolver::SetSensitivity(CGeometry *geometry, CConfig *config, CSolver*) {
491 
492  AD::BeginUseAdjoints();
493 
494  SU2_OMP_PARALLEL {
495 
496  const bool time_stepping = (config->GetTime_Marching() != TIME_MARCHING::STEADY);
497  const su2double eps = config->GetAdjSharp_LimiterCoeff()*config->GetRefElemLength();
498 
499  SU2_OMP_FOR_STAT(omp_chunk_size)
500  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
501 
502  auto Coord = geometry->nodes->GetCoord(iPoint);
503 
504  for (auto iDim = 0u; iDim < nDim; iDim++) {
505 
506  su2double Sensitivity = geometry->nodes->GetAdjointSolution(iPoint, iDim);
507  AD::ResetInput(Coord[iDim]);
508 
509  /*--- If sharp edge, set the sensitivity to 0 on that region ---*/
510 
511  if (config->GetSens_Remove_Sharp() && geometry->nodes->GetSharpEdge_Distance(iPoint) < eps) {
512  Sensitivity = 0.0;
513  }
514  if (!time_stepping) {
515  nodes->SetSensitivity(iPoint,iDim, Sensitivity);
516  } else {
517  nodes->SetSensitivity(iPoint,iDim, nodes->GetSensitivity(iPoint,iDim) + Sensitivity);
518  }
519  }
520  }
521  END_SU2_OMP_FOR
522 
523  SetSurface_Sensitivity(geometry, config);
524 
525  }
526  END_SU2_OMP_PARALLEL
527 
528  AD::EndUseAdjoints();
529 }

531 void CDiscAdjSolver::SetSurface_Sensitivity(CGeometry *geometry, CConfig *config) {
532 
533  SU2_OMP_MASTER
534  for (auto& x : Sens_Geo) x = 0.0;
535  END_SU2_OMP_MASTER
536 
537  /*--- Loop over boundary markers to select those for Euler walls and NS walls ---*/
538 
539  for (auto iMarker = 0ul; iMarker < nMarker; iMarker++) {
540 
541  if (!config->GetSolid_Wall(iMarker)) continue;
542 
543  su2double Sens = 0.0;
544 
545  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
546  for (auto iVertex = 0ul; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
547 
548  /*--- Projection of the gradient calculated with AD onto the normal vector of the surface ---*/
549 
550  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
551  const auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
552 
553  su2double Sens_Vertex = 0.0;
554  for (auto iDim = 0u; iDim < nDim; iDim++) {
555  Sens_Vertex += Normal[iDim] * nodes->GetSensitivity(iPoint,iDim);
556  }
557  Sens_Vertex /= GeometryToolbox::Norm(nDim, Normal);
558 
559  CSensitivity[iMarker][iVertex] = -Sens_Vertex;
560  Sens += pow(Sens_Vertex,2);
561  }
562  END_SU2_OMP_FOR
563 
564  if (config->GetMarker_All_Monitoring(iMarker) == NO) continue;
565 
566  /*--- Compute sensitivity for each surface point ---*/
567 
568  const auto Marker_Tag = config->GetMarker_All_TagBound(iMarker);
569 
570  for (size_t iMarker_Mon = 0; iMarker_Mon < Sens_Geo.size(); iMarker_Mon++) {
571  if (Marker_Tag == config->GetMarker_Monitoring_TagBound(iMarker_Mon)) {
572  atomicAdd(Sens, Sens_Geo[iMarker_Mon]);
573  break;
574  }
575  }
576  }
577

616 void CDiscAdjSolver::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter, bool val_update_geo) {
617 
618  /*--- Restart the solution from file information ---*/
619 
620  auto filename = config->GetSolution_AdjFileName();
621  auto restart_filename = config->GetObjFunc_Extension(filename);
622  restart_filename = config->GetFilename(restart_filename, "", val_iter);
623 
624  const bool rans = (config->GetKind_Turb_Model() != TURB_MODEL::NONE);
625 
626  /*--- Skip coordinates ---*/
627  unsigned short skipVars = geometry[MESH_0]->GetnDim();
628 
629  /*--- Skip flow adjoint variables ---*/
630  if (KindDirect_Solver == RUNTIME_TURB_SYS) {
631  skipVars += nDim + 2;
632  }
633 
634  if (KindDirect_Solver == RUNTIME_SPECIES_SYS) {
635  // Skip the number of Flow Vars and Turb Vars to get to the adjoint species vars
636  skipVars += nDim + 2;
637  if (rans) skipVars += solver[MESH_0][TURB_SOL]->GetnVar();
638  }
639 
640  /*--- Skip flow adjoint and turbulent variables ---*/
641  if (KindDirect_Solver == RUNTIME_RADIATION_SYS) {
642  skipVars += nDim + 2;
643  if (rans) skipVars += solver[MESH_0][TURB_SOL]->GetnVar();
644  }
645 
646  BasicLoadRestart(geometry[MESH_0], config, restart_filename, skipVars);
647 
648  /*--- Interpolate solution on coarse grids ---*/
649 
650  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
651  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][ADJFLOW_SOL]->GetNodes()->GetSolution(),
652  *geometry[iMesh], solver[iMesh][ADJFLOW_SOL]->GetNodes()->GetSolution());
653  }
