206 * \brief Integer division rounding up.

208inline constexpr size_t roundUpDiv(size_t numerator, size_t denominator) {
209 return (numerator + denominator - 1) / denominator;
210}

219/*!
220 * \brief Compute a chunk size based on totalWork and number of threads such that
221 * all threads get the same number of chunks (with limited size).
222 * \param[in] totalWork - e.g. total number of loop iterations.
223 * \param[in] numThreads - Number of threads that will share the work.
224 * \param[in] maxChunkSize - Upper bound for chunk size.
225 * \return The chunkSize.
226 */
227inline size_t computeStaticChunkSize(size_t totalWork, size_t numThreads, size_t maxChunkSize) {
228 if (!totalWork) return maxChunkSize;
229 size_t workPerThread = roundUpDiv(totalWork, numThreads);
230 size_t chunksPerThread = roundUpDiv(workPerThread, maxChunkSize);
231 return roundUpDiv(workPerThread, chunksPerThread);
232}

234/*!
235 * \brief Copy data from one array-like object to another in parallel.
236 * \param[in] size - Number of elements.
237 * \param[in] src - Source array.
238 * \param[in] dst - Destination array.
239 */
240template <class T, class U>
241void parallelCopy(size_t size, const T* src, U* dst) {
242 SU2_OMP_FOR_STAT(2048)
243 for (size_t i = 0; i < size; ++i) dst[i] = src[i];
244 END_SU2_OMP_FOR
245}
