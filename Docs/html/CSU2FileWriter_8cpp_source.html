35void CSU2FileWriter::WriteData(string val_filename){
36
37 ofstream restart_file;
38 const vector<string> fieldNames = dataSorter->GetFieldNames();
39 /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
40 val_filename.append(fileExt);
41
42 /*--- Set a timer for the file writing. ---*/
43
44 startTime = SU2_MPI::Wtime();
45
46 /*--- Only the FIRST node writes the header (it does not matter if that is the master). ---*/
47
48 if (rank == 0) {
49 restart_file.open(val_filename);
50 restart_file << "\"PointID\"";
51 for (auto& field : fieldNames) restart_file << ",\"" << field << "\"";
52 restart_file << "\n";
53 restart_file.close();
54 }
55
56 /*--- Serialize the writes to the restart file. ---*/
57
58 for (int iProcessor = 0; iProcessor < size; iProcessor++) {
59 if (rank == iProcessor) {
60 restart_file.open(val_filename, ios::app);
61 restart_file.precision(15);
62
63 for (auto iPoint = 0ul; iPoint < dataSorter->GetnPoints(); iPoint++) {
64
65 /*--- Write global index of the current point. ---*/
66
67 restart_file << dataSorter->GetGlobalIndex(iPoint);
68
69 /*--- Loop over the variables and write the values to file. ---*/
70
71 for (size_t iVar = 0; iVar < fieldNames.size(); iVar++)
72 restart_file << ", " << scientific << dataSorter->GetData(iVar, iPoint);
73 restart_file << "\n";
74 }
75
76 restart_file.close();
77 }
78
79 /*--- Wait for iProcessor to finish and close the file. ---*/
80
81 SU2_MPI::Barrier(SU2_MPI::GetComm());
82 }
83
84 /*--- Compute and store the write time. ---*/
85
86 stopTime = SU2_MPI::Wtime();
87
88 usedTime = stopTime-startTime;
89
90 /*--- Determine the file size ---*/
91
92 fileSize = DetermineFilesize(val_filename);
93
94 /*--- Compute and store the bandwidth ---*/
95
96 bandwidth = fileSize/(1.0e6)/usedTime;
97}
