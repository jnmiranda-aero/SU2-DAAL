113 /*!
114 * \brief Get the element container index and number of nodes of a given VTK type.
115 * \param[in] VTK_Type - Type of element.
116 * \param[out] EL_KIND - Element container index.
117 * \param[out] nNodes - Number of nodes.
118 */
119 static void GetElemKindAndNumNodes(unsigned short VTK_Type, int& EL_KIND, unsigned short& nNodes) {
120 switch (VTK_Type) {
121 case TRIANGLE: nNodes = 3; EL_KIND = EL_TRIA; break;
122 case QUADRILATERAL: nNodes = 4; EL_KIND = EL_QUAD; break;
123 case TETRAHEDRON: nNodes = 4; EL_KIND = EL_TETRA; break;
124 case PYRAMID: nNodes = 5; EL_KIND = EL_PYRAM; break;
125 case PRISM: nNodes = 6; EL_KIND = EL_PRISM; break;
126 case HEXAHEDRON: nNodes = 8; EL_KIND = EL_HEXA; break;
127 default: assert(false); nNodes = 0; EL_KIND = -(1<<30); break;
128 }

542 /*!
543 * \brief Compute the objective function.
544 * \param[in] config - Definition of the problem.
545 * \param[in] solver - Container vector with all the solutions.
546 */
547 void Evaluate_ObjFunc(const CConfig *config, CSolver**) final {
548 Total_ComboObj = 0.0;
549 switch (config->GetKind_ObjFunc()) {
550 case REFERENCE_GEOMETRY:
551 Total_ComboObj = GetTotal_OFRefGeom();
552 break;
553 case REFERENCE_NODE:
554 Total_ComboObj = GetTotal_OFRefNode();
555 break;
556 case TOPOL_COMPLIANCE:
557 Total_ComboObj = GetTotal_OFCompliance();
558 break;
559 case VOLUME_FRACTION:
560 Total_ComboObj = GetTotal_OFVolFrac();
561 break;
562 case TOPOL_DISCRETENESS:
563 Total_ComboObj = GetTotal_OFDiscreteness();
564 break;
565 case STRESS_PENALTY:
566 Total_ComboObj = GetTotal_OFStressPenalty();
567 break;
568 case CUSTOM_OBJFUNC:
569 Total_ComboObj = Total_Custom_ObjFunc;
570 break;
571 }
