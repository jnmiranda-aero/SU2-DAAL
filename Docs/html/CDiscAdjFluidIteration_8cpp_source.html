31 void CDiscAdjFluidIteration::Preprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
32  CSolver***** solver, CNumerics****** numerics, CConfig** config,
33  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
34  CFreeFormDefBox*** FFDBox, unsigned short iZone, unsigned short iInst) {
35  StartTime = SU2_MPI::Wtime();
36 
37  const auto TimeIter = config[iZone]->GetTimeIter();
38  const bool dual_time_1st = (config[iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
39  const bool dual_time_2nd = (config[iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
40  const bool dual_time = (dual_time_1st || dual_time_2nd);
41  const bool grid_IsMoving = config[iZone]->GetGrid_Movement();
42  const bool species = config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE;
43  const bool heat = config[iZone]->GetWeakly_Coupled_Heat();
44  const bool radiation = config[iZone]->AddRadiation();
45 
46  auto solvers0 = solver[iZone][iInst][MESH_0];
47  auto geometries = geometry[iZone][iInst];
48 
49  unsigned long nSolvers = 1;
50  std::array<int, 5> solversToProcess{{FLOW_SOL}};
51  if (turbulent) solversToProcess[nSolvers++] = TURB_SOL;
52  if (species) solversToProcess[nSolvers++] = SPECIES_SOL;
53  if (heat) solversToProcess[nSolvers++] = HEAT_SOL;
54  if (radiation) solversToProcess[nSolvers++] = RAD_SOL;
55 
56  /*--- For the unsteady adjoint, load direct solutions from restart files. ---*/
57 
58  if (config[iZone]->GetTime_Marching() != TIME_MARCHING::STEADY) {
59  const int Direct_Iter = static_cast<int>(config[iZone]->GetUnst_AdjointIter()) -
60  static_cast<int>(TimeIter) - 2 + dual_time;
61 
62  /*--- For dual-time stepping we want to load the already converged solution at previous timesteps.
63  * In general we only load one file and shift the previously loaded solutions, on the first we
64  * load one or two more (depending on dual time order). ---*/
65 
66  if (dual_time_2nd) {
67  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 2);
68  } else if (dual_time_1st) {
69  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 1);
70  }
71 
72  if (TimeIter == 0) {
73  /*--- Push solution back one level. ---*/
74 
75  if (dual_time) {
76  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
77  for (auto iSol = 0ul; iSol < nSolvers; ++iSol) {
78  solver[iZone][iInst][iMesh][solversToProcess[iSol]]->GetNodes()->Set_Solution_time_n();
79  }
80  if (grid_IsMoving) {
81  geometries[iMesh]->nodes->SetCoord_n();
82  }
83  if (config[iZone]->GetDynamic_Grid()) {
84  geometries[iMesh]->nodes->SetVolume_n();
85  }
86  }
87  }
88 
89  /*--- If required load another time step. Push the previous time step to n-1 and the
90  loaded time step to n. ---*/
91 
92  if (dual_time_2nd) {
93  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 1);
94 
95  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
96  for (auto iSol = 0ul; iSol < nSolvers; ++iSol) {
97  solver[iZone][iInst][iMesh][solversToProcess[iSol]]->GetNodes()->Set_Solution_time_n1();
98  solver[iZone][iInst][iMesh][solversToProcess[iSol]]->GetNodes()->Set_Solution_time_n();
99  }
100  if (grid_IsMoving) {
101  geometries[iMesh]->nodes->SetCoord_n1();
102  geometries[iMesh]->nodes->SetCoord_n();
103  }
104  if (config[iZone]->GetDynamic_Grid()) {
105  geometries[iMesh]->nodes->SetVolume_nM1();
106  geometries[iMesh]->nodes->SetVolume_n();
107  }
108  }
109  }
110 
111  /*--- Load solution timestep n ---*/
112 
113  LoadUnsteady_Solution(geometry, solver, config, iZone, iInst, Direct_Iter);
114 
115  if (config[iZone]->GetDeform_Mesh()) {
116  solvers0[MESH_SOL]->LoadRestart(geometries, solver[iZone][iInst], config[iZone], Direct_Iter, true);
117  }
118 
119  }
120  if ((TimeIter > 0) && dual_time) {
121  /*--- Here the primal solutions (only working variables) are loaded and put in the correct order
122  into containers. For ALE the mesh coordinates have to be put into the correct containers as well,
123  i.e. follow the same logic for the solution. Afterwards the GridVelocity is computed based on
124  the Coordinates. ---*/
125 
126  /*--- Temporarily store the loaded volumes into old containers. ---*/
127  if (config[iZone]->GetDynamic_Grid()) {
128  for (auto iMesh=0; iMesh<=config[iZone]->GetnMGLevels();iMesh++) {
129  geometries[iMesh]->nodes->SetVolume_Old();
130  geometries[iMesh]->nodes->SetVolume_n_Old();
131  geometries[iMesh]->nodes->SetVolume_nM1_Old();
132  }
133  }
134 
135  /*--- Load mesh solver. ---*/
136  if (config[iZone]->GetDeform_Mesh()) {
137  solvers0[MESH_SOL]->LoadRestart(geometries, solver[iZone][iInst], config[iZone], Direct_Iter, true);
138  }
139 
140  /*--- Set volumes into correct containers ---*/
141  if (config[iZone]->GetDynamic_Grid()) {
142  for (auto iMesh = 0; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
143  /*--- If negative iteration number, set default. ---*/
144  if (Direct_Iter - 1 - dual_time_2nd < 0) {
145  for(auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
146  geometries[iMesh]->nodes->SetVolume(iPoint, 0.0);
147  }
148  }
149 
150  /*--- Set currently loaded volume to Volume_nM1 ---*/
151  geometries[iMesh]->nodes->SetVolume_n();
152  geometries[iMesh]->nodes->SetVolume_nM1();
153 
154  /*--- Set Volume_n and Volume from old containers ---*/
155  geometries[iMesh]->nodes->SetVolume_n_from_OldnM1();
156  geometries[iMesh]->nodes->SetVolume_from_Oldn();
157  }
158  }
159 
160  /*--- Temporarily store the loaded solution and coordinates in the "Old" arrays. ---*/
161 
162  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
163  for (auto iSol = 0ul; iSol < nSolvers; ++iSol) {
164  solver[iZone][iInst][iMesh][solversToProcess[iSol]]->Set_OldSolution();
165  }
166  if (grid_IsMoving) {
167  geometries[iMesh]->nodes->SetCoord_Old();
168  }
169  }
170 
171  /*--- Move timestep n to current solution. ---*/
172 
173  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
174  for (auto iSol = 0ul; iSol < nSolvers; ++iSol) {
175  auto* s = solver[iZone][iInst][iMesh][solversToProcess[iSol]];
176  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
177  s->GetNodes()->SetSolution(iPoint, s->GetNodes()->GetSolution_time_n(iPoint));
178  }
179  }
180  if (grid_IsMoving) {
181  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
182  geometries[iMesh]->nodes->SetCoord(iPoint, geometries[iMesh]->nodes->GetCoord_n(iPoint));
183  }
184  }
185  }
186 
187  /*--- Finally, place the loaded solution in the correct place (n or n-1 depending on order). ---*/
188 
189  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
190  for (auto iSol = 0ul; iSol < nSolvers; ++iSol) {
191  auto* s = solver[iZone][iInst][iMesh][solversToProcess[iSol]];
192  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
193  if (dual_time_2nd) {
194  /*--- If required also move timestep n-1 to timestep n. ---*/
195  s->GetNodes()->Set_Solution_time_n(iPoint, s->GetNodes()->GetSolution_time_n1(iPoint));
196  s->GetNodes()->Set_Solution_time_n1(iPoint, s->GetNodes()->GetSolution_Old(iPoint));
197  } else {
198  s->GetNodes()->Set_Solution_time_n(iPoint, s->GetNodes()->GetSolution_Old(iPoint));
199  }
200  }
201  }
202  if (grid_IsMoving) {
203  for (auto iPoint = 0ul; iPoint < geometries[iMesh]->GetnPoint(); iPoint++) {
204  if (dual_time_2nd) {
205  geometries[iMesh]->nodes->SetCoord_n(iPoint, geometries[iMesh]->nodes->GetCoord_n1(iPoint));
206  geometries[iMesh]->nodes->SetCoord_n1(iPoint, geometries[iMesh]->nodes->GetCoord_Old(iPoint));
207  } else {
208  geometries[iMesh]->nodes->SetCoord_n(iPoint, geometries[iMesh]->nodes->GetCoord_Old(iPoint));
209  }
210  }
211  }
212  }
213 
214  } // else if TimeIter > 0
215 
216  /*--- Compute & set Grid Velocity via finite differences of the Coordinates. ---*/
217  if (grid_IsMoving) {
218  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++)
219  geometries[iMesh]->SetGridVelocity(config[iZone]);
220  }
221 
222  } // if unsteady
223 
224  SU2_OMP_PARALLEL_(if(solvers0[ADJFLOW_SOL]->GetHasHybridParallel())) {
225 
226  /*--- Store flow solution also in the adjoint solver in order to be able to reset it later ---*/
227 
228  if (TimeIter == 0 || dual_time) {
229  auto SetSolutionDirect = [&](CSolver** solvers, int adj, int primal, unsigned long nPoint) {
230  SU2_OMP_FOR_STAT(1024)
231  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++)
232  solvers[adj]->GetNodes()->SetSolution_Direct(iPoint, solvers[primal]->GetNodes()->GetSolution(iPoint));
233  END_SU2_OMP_FOR
234  };
235 
236  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
237  auto solvers = solver[iZone][iInst][iMesh];
238  SetSolutionDirect(solvers, ADJFLOW_SOL, FLOW_SOL, geometries[iMesh]->GetnPoint());
239  }
240  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
241  SetSolutionDirect(solvers0, ADJTURB_SOL, TURB_SOL, geometries[MESH_0]->GetnPoint());
242  }
243  if (species) {
244  SetSolutionDirect(solvers0, ADJSPECIES_SOL, SPECIES_SOL, geometries[MESH_0]->GetnPoint());
245  }
246  if (heat) {
247  SetSolutionDirect(solvers0, ADJHEAT_SOL, HEAT_SOL, geometries[MESH_0]->GetnPoint());
248  }
249  if (radiation) {
250  SetSolutionDirect(solvers0, ADJRAD_SOL, RAD_SOL, geometries[MESH_0]->GetnPoint());
251  }
252  }
253 
254  solvers0[ADJFLOW_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
255  MESH_0, 0, RUNTIME_ADJFLOW_SYS, false);
256 
257  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
258  solvers0[ADJTURB_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
259  MESH_0, 0, RUNTIME_ADJTURB_SYS, false);
260  }
261  if (species) {
262  solvers0[ADJSPECIES_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
263  MESH_0, 0, RUNTIME_ADJSPECIES_SYS, false);
264  }
265  if (heat) {
266  solvers0[ADJHEAT_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
267  MESH_0, 0, RUNTIME_ADJHEAT_SYS, false);
268  }
269  if (radiation) {
270  solvers0[ADJRAD_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone],
271  MESH_0, 0, RUNTIME_ADJRAD_SYS, false);
272  }
273 
274  }
275  END_SU2_OMP_PARALLEL
276 }

278 void CDiscAdjFluidIteration::LoadUnsteady_Solution(CGeometry**** geometry, CSolver***** solver, CConfig** config,
279  unsigned short iZone, unsigned short iInst, int DirectIter) {
280 
281  auto solvers = solver[iZone][iInst];
282  auto geometries = geometry[iZone][iInst];
283  const bool species = config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE;
284 
285  if (DirectIter >= 0) {
286  if (rank == MASTER_NODE)
287  cout << " Loading flow solution from direct iteration " << DirectIter << " for zone " << iZone << "." << endl;
288 
289  solvers[MESH_0][FLOW_SOL]->LoadRestart(geometries, solvers, config[iZone], DirectIter, true);
290 
291  if (turbulent) {
292  solvers[MESH_0][TURB_SOL]->LoadRestart(geometries, solvers, config[iZone], DirectIter, false);
293  }
294  if (species) {
295  solvers[MESH_0][SPECIES_SOL]->LoadRestart(geometries, solvers, config[iZone], DirectIter, false);
296  }
297  if (config[iZone]->GetWeakly_Coupled_Heat()) {
298  solvers[MESH_0][HEAT_SOL]->LoadRestart(geometries, solvers, config[iZone], DirectIter, false);
299  }
300  } else {
301  /*--- If there is no solution file we set the freestream condition ---*/
302  if (rank == MASTER_NODE)
303  cout << " Setting freestream conditions at direct iteration " << DirectIter << " for zone " << iZone << "." << endl;
304 
305  for (auto iMesh = 0u; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
306  solvers[iMesh][FLOW_SOL]->SetFreeStream_Solution(config[iZone]);
307  solvers[iMesh][FLOW_SOL]->Preprocessing(geometries[iMesh], solvers[iMesh], config[iZone], iMesh,
308  DirectIter, RUNTIME_FLOW_SYS, false);
309  if (turbulent) {
310  solvers[iMesh][TURB_SOL]->SetFreeStream_Solution(config[iZone]);
311  solvers[iMesh][TURB_SOL]->Postprocessing(geometries[iMesh], solvers[iMesh], config[iZone], iMesh);
312  }
313  if (species) {
314  solvers[iMesh][SPECIES_SOL]->SetFreeStream_Solution(config[iZone]);
315  solvers[iMesh][SPECIES_SOL]->Postprocessing(geometries[iMesh], solvers[iMesh], config[iZone], iMesh);
316  }
317  if (config[iZone]->GetWeakly_Coupled_Heat()) {
318  solvers[iMesh][HEAT_SOL]->SetFreeStream_Solution(config[iZone]);
319  solvers[iMesh][HEAT_SOL]->Postprocessing(geometries[iMesh], solvers[iMesh], config[iZone], iMesh);
320  }
321  }
322  }

325 void CDiscAdjFluidIteration::IterateDiscAdj(CGeometry**** geometry, CSolver***** solver, CConfig** config,
326  unsigned short iZone, unsigned short iInst, bool CrossTerm) {
327  auto solvers0 = solver[iZone][iInst][MESH_0];
328  auto geometry0 = geometry[iZone][iInst][MESH_0];
329 
330  SU2_OMP_PARALLEL_(if(solvers0[ADJFLOW_SOL]->GetHasHybridParallel())) {
331 
332  /*--- Extract the adjoints of the conservative input variables and store them for the next iteration ---*/
333 
334  if (config[iZone]->GetFluidProblem()) {
335  solvers0[ADJFLOW_SOL]->ExtractAdjoint_Solution(geometry0, config[iZone], CrossTerm);
336 
337  solvers0[ADJFLOW_SOL]->ExtractAdjoint_Variables(geometry0, config[iZone]);
338  }
339  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
340  solvers0[ADJTURB_SOL]->ExtractAdjoint_Solution(geometry0, config[iZone], CrossTerm);
341  }
342  if (config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
343  solvers0[ADJSPECIES_SOL]->ExtractAdjoint_Solution(geometry0, config[iZone], CrossTerm);
344  }
345  if (config[iZone]->GetWeakly_Coupled_Heat()) {
346  solvers0[ADJHEAT_SOL]->ExtractAdjoint_Solution(geometry0, config[iZone], CrossTerm);
347  }
348  if (config[iZone]->AddRadiation()) {
349  solvers0[ADJRAD_SOL]->ExtractAdjoint_Solution(geometry0, config[iZone], CrossTerm);
350 
351  solvers0[ADJRAD_SOL]->ExtractAdjoint_Variables(geometry0, config[iZone]);
352  }
353 
354  }
355  END_SU2_OMP_PARALLEL

358 void CDiscAdjFluidIteration::InitializeAdjoint(CSolver***** solver, CGeometry**** geometry, CConfig** config,
359  unsigned short iZone, unsigned short iInst) {
360  auto solvers0 = solver[iZone][iInst][MESH_0];
361  auto geometry0 = geometry[iZone][iInst][MESH_0];
362 
363  AD::ResizeAdjoints();
364  AD::BeginUseAdjoints();
365 
366  SU2_OMP_PARALLEL_(if(solvers0[ADJFLOW_SOL]->GetHasHybridParallel())) {
367 
368  /*--- Initialize the adjoints the conservative variables ---*/
369 
370  if (config[iZone]->GetFluidProblem()) {
371  solvers0[ADJFLOW_SOL]->SetAdjoint_Output(geometry0, config[iZone]);
372  }
373 
374  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
375  solvers0[ADJTURB_SOL]->SetAdjoint_Output(geometry0, config[iZone]);
376  }
377 
378  if (config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
379  solvers0[ADJSPECIES_SOL]->SetAdjoint_Output(geometry0, config[iZone]);
380  }
381 
382  if (config[iZone]->GetWeakly_Coupled_Heat()) {
383  solvers0[ADJHEAT_SOL]->SetAdjoint_Output(geometry0, config[iZone]);
384  }
385 
386  if (config[iZone]->AddRadiation()) {
387  solvers0[ADJRAD_SOL]->SetAdjoint_Output(geometry0, config[iZone]);
388  }
389 
390  if (config[iZone]->GetFluidProblem() && !config[iZone]->GetMultizone_Problem()) {
391  solvers0[FLOW_SOL]->SetVertexTractionsAdjoint(geometry0, config[iZone]);
392  }
393 
394  }
395  END_SU2_OMP_PARALLEL
396 
397  AD::EndUseAdjoints();
398 }

400 void CDiscAdjFluidIteration::RegisterInput(CSolver***** solver, CGeometry**** geometry, CConfig** config,
401  unsigned short iZone, unsigned short iInst, RECORDING kind_recording) {
402  auto solvers0 = solver[iZone][iInst][MESH_0];
403  auto geometry0 = geometry[iZone][iInst][MESH_0];
404 
405  SU2_OMP_PARALLEL_(if(solvers0[ADJFLOW_SOL]->GetHasHybridParallel())) {
406 
407  if (kind_recording == RECORDING::SOLUTION_VARIABLES || kind_recording == RECORDING::SOLUTION_AND_MESH) {
408  /*--- Register flow and turbulent variables as input ---*/
409 
410  if (config[iZone]->GetFluidProblem()) {
411  solvers0[ADJFLOW_SOL]->RegisterSolution(geometry0, config[iZone]);
412 
413  solvers0[ADJFLOW_SOL]->RegisterVariables(geometry0, config[iZone]);
414  }
415 
416  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
417  solvers0[ADJTURB_SOL]->RegisterSolution(geometry0, config[iZone]);
418  }
419  if (config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
420  solvers0[ADJSPECIES_SOL]->RegisterSolution(geometry0, config[iZone]);
421  }
422  if (config[iZone]->GetWeakly_Coupled_Heat()) {
423  solvers0[ADJHEAT_SOL]->RegisterSolution(geometry0, config[iZone]);
424  }
425  if (config[iZone]->AddRadiation()) {
426  solvers0[ADJRAD_SOL]->RegisterSolution(geometry0, config[iZone]);
427 
428  solvers0[ADJRAD_SOL]->RegisterVariables(geometry0, config[iZone]);
429  }
430  }
431 
432  if (kind_recording == RECORDING::MESH_COORDS || kind_recording == RECORDING::SOLUTION_AND_MESH) {
433  /*--- Register node coordinates as input ---*/
434 
435  geometry0->RegisterCoordinates();
436  }
437 
438  if (kind_recording == RECORDING::MESH_DEFORM) {
439  /*--- Undeformed mesh coordinates ---*/
440  solvers0[ADJMESH_SOL]->RegisterSolution(geometry0, config[iZone]);
441 
442  /*--- Boundary displacements ---*/
443  solvers0[ADJMESH_SOL]->RegisterVariables(geometry0, config[iZone]);
444  }
445  }
446  END_SU2_OMP_PARALLEL
447 }

449 void CDiscAdjFluidIteration::SetDependencies(CSolver***** solver, CGeometry**** geometry, CNumerics****** numerics,
450  CConfig** config, unsigned short iZone, unsigned short iInst,
451  RECORDING kind_recording) {
452  auto solvers0 = solver[iZone][iInst][MESH_0];
453  auto geometry0 = geometry[iZone][iInst][MESH_0];
454 
455  if ((kind_recording == RECORDING::MESH_COORDS) ||
456  (kind_recording == RECORDING::CLEAR_INDICES) ||
457  (kind_recording == RECORDING::SOLUTION_AND_MESH)) {
458  /*--- Update geometry to get the influence on other geometry variables (normals, volume etc) ---*/
459 
460  SU2_OMP_PARALLEL
461  CGeometry::UpdateGeometry(geometry[iZone][iInst], config[iZone]);
462  END_SU2_OMP_PARALLEL
463 
464  CGeometry::ComputeWallDistance(config, geometry);
465  }
466 
467  SU2_OMP_PARALLEL_(if(solvers0[ADJFLOW_SOL]->GetHasHybridParallel())) {
468 
469  /*--- Compute coupling between flow, turbulent and species equations ---*/
470  solvers0[FLOW_SOL]->Preprocessing(geometry0, solvers0, config[iZone], MESH_0, NO_RK_ITER, RUNTIME_FLOW_SYS, true);
471  solvers0[FLOW_SOL]->InitiateComms(geometry0, config[iZone], MPI_QUANTITIES::SOLUTION);
472  solvers0[FLOW_SOL]->CompleteComms(geometry0, config[iZone], MPI_QUANTITIES::SOLUTION);
473 
474  if (config[iZone]->GetBoolTurbomachinery()) {
475  solvers0[FLOW_SOL]->TurboAverageProcess(solvers0, geometry0, config[iZone], INFLOW);
476  solvers0[FLOW_SOL]->TurboAverageProcess(solvers0, geometry0, config[iZone], OUTFLOW);
477  }
478  if (turbulent && !config[iZone]->GetFrozen_Visc_Disc()) {
479  solvers0[TURB_SOL]->Postprocessing(geometry0, solvers0,
480  config[iZone], MESH_0);
481  solvers0[TURB_SOL]->InitiateComms(geometry0, config[iZone], MPI_QUANTITIES::SOLUTION);
482  solvers0[TURB_SOL]->CompleteComms(geometry0, config[iZone], MPI_QUANTITIES::SOLUTION);
483  }
484  if (config[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
485  solvers0[SPECIES_SOL]->Preprocessing(geometry0, solvers0, config[iZone], MESH_0, NO_RK_ITER, RUNTIME_FLOW_SYS, true);
486  solvers0[SPECIES_SOL]->InitiateComms(geometry0, config[iZone], MPI_QUANTITIES::SOLUTION);
487  solvers0[SPECIES_SOL]->CompleteComms(geometry0, config[iZone], MPI_QUANTITIES::SOLUTION);
488  }
489  if (config[iZone]->GetWeakly_Coupled_Heat()) {
