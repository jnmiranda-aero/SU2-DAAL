63  /*!
64  * \brief Less than operator. Needed for the sorting of the candidates.
65  * \param[in] other Object to which the current object must be compared.
66  */
67  inline bool operator<(const CBBoxTargetClass& other) const {
68  /* Make sure that the bounding boxes with the smallest possible distances
69  are stored first. */
70  if (possibleMinDist2 < other.possibleMinDist2) return true;
71  if (possibleMinDist2 > other.possibleMinDist2) return false;
72  if (guaranteedMinDist2 < other.guaranteedMinDist2) return true;
73  if (guaranteedMinDist2 > other.guaranteedMinDist2) return false;
74  if (boundingBoxID < other.boundingBoxID) return true;
75  return false;
76  }
