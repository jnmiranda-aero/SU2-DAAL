64 /*!

68 inline void SetTurboNormal(const su2double* val_normal) {
69 unsigned short iDim;
70 for (iDim = 0; iDim < nDim; iDim++) TurboNormal[iDim] = val_normal[iDim];

89 for (unsigned short iDim = 0; iDim < nDim; iDim++) val_normal[iDim] = TurboNormal[iDim];
