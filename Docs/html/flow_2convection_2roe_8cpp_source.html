95 CNumerics::ResidualType<> CUpwRoeBase_Flow::ComputeResidual(const CConfig* config) {
96 
97  implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
98 
99  unsigned short iVar, jVar, iDim;
100  su2double ProjGridVel = 0.0, Energy_i, Energy_j;
101 
102  AD::StartPreacc();
103  AD::SetPreaccIn(V_i, nDim+4); AD::SetPreaccIn(V_j, nDim+4); AD::SetPreaccIn(Normal, nDim);
104  if (dynamic_grid) {
105  AD::SetPreaccIn(GridVel_i, nDim); AD::SetPreaccIn(GridVel_j, nDim);
106  }
107  if (roe_low_dissipation){
108  AD::SetPreaccIn(Sensor_i); AD::SetPreaccIn(Sensor_j);
109  AD::SetPreaccIn(Dissipation_i); AD::SetPreaccIn(Dissipation_j);
110  }
111 
112  /*--- Face area (norm of the normal vector) and unit normal ---*/
113 
114  Area = GeometryToolbox::Norm(nDim, Normal);
115 
116  for (iDim = 0; iDim < nDim; iDim++)
117  UnitNormal[iDim] = Normal[iDim]/Area;
118 
119  /*--- Primitive variables at point i ---*/
120 
121  for (iDim = 0; iDim < nDim; iDim++)
122  Velocity_i[iDim] = V_i[iDim+1];
123  Pressure_i = V_i[nDim+1];
124  Density_i = V_i[nDim+2];
125  Enthalpy_i = V_i[nDim+3];
126  Energy_i = Enthalpy_i - Pressure_i/Density_i;
127 
128  /*--- Primitive variables at point j ---*/
129 
130  for (iDim = 0; iDim < nDim; iDim++)
131  Velocity_j[iDim] = V_j[iDim+1];
132  Pressure_j = V_j[nDim+1];
133  Density_j = V_j[nDim+2];
134  Enthalpy_j = V_j[nDim+3];
135  Energy_j = Enthalpy_j - Pressure_j/Density_j;
136 
137  /*--- Compute variables that are common to the derived schemes ---*/
138 
139  /*--- Roe-averaged variables at interface between i & j ---*/
140 
141  su2double R = sqrt(fabs(Density_j/Density_i));
142  RoeDensity = R*Density_i;
143  su2double sq_vel = 0.0;
144  for (iDim = 0; iDim < nDim; iDim++) {
145  RoeVelocity[iDim] = (R*Velocity_j[iDim]+Velocity_i[iDim])/(R+1);
146  sq_vel += RoeVelocity[iDim]*RoeVelocity[iDim];
147  }
148  RoeEnthalpy = (R*Enthalpy_j+Enthalpy_i)/(R+1);
149  RoeSoundSpeed2 = (Gamma-1)*(RoeEnthalpy-0.5*sq_vel);
150 
151  /*--- Negative RoeSoundSpeed^2, the jump variables is too large, clear fluxes and exit. ---*/
152 
153  if (RoeSoundSpeed2 <= 0.0) {
154  for (iVar = 0; iVar < nVar; iVar++) {
155  Flux[iVar] = 0.0;
156  if (implicit){
157  for (jVar = 0; jVar < nVar; jVar++) {
158  Jacobian_i[iVar][jVar] = 0.0;
159  Jacobian_j[iVar][jVar] = 0.0;
160  }
161  }
162  }
163  AD::SetPreaccOut(Flux, nVar);
164  AD::EndPreacc();
165 
166  return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
167  }
168 
169  RoeSoundSpeed = sqrt(RoeSoundSpeed2);
170 
171  /*--- P tensor ---*/
172 
173  GetPMatrix(&RoeDensity, RoeVelocity, &RoeSoundSpeed, UnitNormal, P_Tensor);
174 
175  /*--- Projected velocity adjusted for mesh motion ---*/
176 
177  ProjVelocity = 0.0;
178  for (iDim = 0; iDim < nDim; iDim++)
179  ProjVelocity += RoeVelocity[iDim]*UnitNormal[iDim];
180 
181  if (dynamic_grid) {
182  for (iDim = 0; iDim < nDim; iDim++)
183  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*UnitNormal[iDim];
184  ProjVelocity -= ProjGridVel;
185  }
186 
187  /*--- Flow eigenvalues ---*/
188 
189  for (iDim = 0; iDim < nDim; iDim++)
190  Lambda[iDim] = ProjVelocity;
191 
192  Lambda[nVar-2] = ProjVelocity + RoeSoundSpeed;
193  Lambda[nVar-1] = ProjVelocity - RoeSoundSpeed;
194 
195  /*--- Apply Mavriplis' entropy correction to eigenvalues ---*/
196 
197  su2double MaxLambda = fabs(ProjVelocity) + RoeSoundSpeed;
198 
199  for (iVar = 0; iVar < nVar; iVar++)
200  Lambda[iVar] = max(fabs(Lambda[iVar]), config->GetEntropyFix_Coeff()*MaxLambda);
201 
202  /*--- Reconstruct conservative variables ---*/
203 
204  Conservatives_i[0] = Density_i;
205  Conservatives_j[0] = Density_j;
206 
207  for (iDim = 0; iDim < nDim; iDim++) {
208  Conservatives_i[iDim+1] = Density_i*Velocity_i[iDim];
209  Conservatives_j[iDim+1] = Density_j*Velocity_j[iDim];
210  }
211  Conservatives_i[nDim+1] = Density_i*Energy_i;
212  Conservatives_j[nDim+1] = Density_j*Energy_j;
213 
214  /*--- Compute left and right fluxes ---*/
215 
216  GetInviscidProjFlux(&Density_i, Velocity_i, &Pressure_i, &Enthalpy_i, Normal, ProjFlux_i);
217  GetInviscidProjFlux(&Density_j, Velocity_j, &Pressure_j, &Enthalpy_j, Normal, ProjFlux_j);
218 
219  /*--- Initialize residual (flux) and Jacobians ---*/
220 
221  for (iVar = 0; iVar < nVar; iVar++)
222  Flux[iVar] = 0.5*(ProjFlux_i[iVar]+ProjFlux_j[iVar]);
223 
224  if (implicit) {
225  GetInviscidProjJac(Velocity_i, &Energy_i, Normal, 0.5, Jacobian_i);
226  GetInviscidProjJac(Velocity_j, &Energy_j, Normal, 0.5, Jacobian_j);
227  }
228 
229  /*--- Finalize in children class ---*/
230 
231  FinalizeResidual(Flux, Jacobian_i, Jacobian_j, config);
232 
233  /*--- Correct for grid motion ---*/
234 
235  if (dynamic_grid) {
236  for (iVar = 0; iVar < nVar; iVar++) {
237  Flux[iVar] -= ProjGridVel*Area * 0.5*(Conservatives_i[iVar]+Conservatives_j[iVar]);
238 
239  if (implicit) {
240  Jacobian_i[iVar][iVar] -= 0.5*ProjGridVel*Area;
241  Jacobian_j[iVar][iVar] -= 0.5*ProjGridVel*Area;
242  }
243  }
244  }
245 
246  AD::SetPreaccOut(Flux, nVar);
247  AD::EndPreacc();
248 
249  return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
250 
251 }

256 void CUpwRoe_Flow::FinalizeResidual(su2double *val_residual, su2double **val_Jacobian_i,
257  su2double **val_Jacobian_j, const CConfig* config) {
258 
259  unsigned short iVar, jVar, kVar;
260 
261  /*--- Compute inverse P tensor ---*/
262  GetPMatrix_inv(&RoeDensity, RoeVelocity, &RoeSoundSpeed, UnitNormal, invP_Tensor);
263 
264  /*--- Diference between conservative variables at jPoint and iPoint ---*/
265  for (iVar = 0; iVar < nVar; iVar++)
266  Diff_U[iVar] = Conservatives_j[iVar]-Conservatives_i[iVar];
267 
268  /*--- Low dissipation formulation ---*/
269  if (roe_low_dissipation)
270  Dissipation_ij = GetRoe_Dissipation(Dissipation_i, Dissipation_j, Sensor_i, Sensor_j, config);
271  else
272  Dissipation_ij = 1.0;
273 
274  /*--- Standard Roe "dissipation" ---*/
275 
276  for (iVar = 0; iVar < nVar; iVar++) {
277  for (jVar = 0; jVar < nVar; jVar++) {
278  /*--- Compute |Proj_ModJac_Tensor| = P x |Lambda| x inverse P ---*/
279  su2double Proj_ModJac_Tensor_ij = 0.0;
280  for (kVar = 0; kVar < nVar; kVar++)
281  Proj_ModJac_Tensor_ij += P_Tensor[iVar][kVar]*Lambda[kVar]*invP_Tensor[kVar][jVar];
282 
283  /*--- Update residual and Jacobians ---*/
284  val_residual[iVar] -= (1.0-kappa)*Proj_ModJac_Tensor_ij*Diff_U[jVar]*Area*Dissipation_ij;
285 
286  if(implicit){
287  val_Jacobian_i[iVar][jVar] += (1.0-kappa)*Proj_ModJac_Tensor_ij*Area;
288  val_Jacobian_j[iVar][jVar] -= (1.0-kappa)*Proj_ModJac_Tensor_ij*Area;
289  }
290  }
291  }

298 void CUpwL2Roe_Flow::FinalizeResidual(su2double *val_residual, su2double **val_Jacobian_i,
299  su2double **val_Jacobian_j, const CConfig* config) {
300 
301  /*--- L2Roe: a low dissipation version of Roe's approximate Riemann solver for low Mach numbers. IJNMF 2015 ---*/
302 
303  unsigned short iVar, jVar, kVar, iDim;
304 
305  /*--- Clamped Mach number ---*/
306 
307  su2double M_i = 0.0, M_j = 0.0;
308  for (iDim = 0; iDim < nDim; iDim++) {
309  M_i += Velocity_i[iDim]*Velocity_i[iDim];
310  M_j += Velocity_j[iDim]*Velocity_j[iDim];
311  }
312  M_i = sqrt(M_i / fabs(Pressure_i*Gamma/Density_i));
313  M_j = sqrt(M_j / fabs(Pressure_j*Gamma/Density_j));
314 
315  su2double zeta = max(0.05,min(max(M_i,M_j),1.0));
316 
317  /*--- Compute wave amplitudes (characteristics) ---*/
318 
319  su2double proj_delta_vel = 0.0, delta_vel[3] = {0.0};
320  for (iDim = 0; iDim < nDim; iDim++) {
321  delta_vel[iDim] = Velocity_j[iDim] - Velocity_i[iDim];
322  proj_delta_vel += delta_vel[iDim]*UnitNormal[iDim];
323  }
324  proj_delta_vel *= zeta;
325  su2double delta_p = Pressure_j - Pressure_i;
326  su2double delta_rho = Density_j - Density_i;
327 
328  su2double delta_wave[5] = {0.0};
329  if (nDim == 2) {
330  delta_wave[0] = delta_rho - delta_p/RoeSoundSpeed2;
331  delta_wave[1] = (UnitNormal[1]*delta_vel[0]-UnitNormal[0]*delta_vel[1])*zeta;
332  delta_wave[2] = proj_delta_vel + delta_p/(RoeDensity*RoeSoundSpeed);
333  delta_wave[3] = -proj_delta_vel + delta_p/(RoeDensity*RoeSoundSpeed);
334  } else {
335  delta_wave[0] = delta_rho - delta_p/RoeSoundSpeed2;
336  delta_wave[1] = (UnitNormal[0]*delta_vel[2]-UnitNormal[2]*delta_vel[0])*zeta;
337  delta_wave[2] = (UnitNormal[1]*delta_vel[0]-UnitNormal[0]*delta_vel[1])*zeta;
338  delta_wave[3] = proj_delta_vel + delta_p/(RoeDensity*RoeSoundSpeed);
339  delta_wave[4] = -proj_delta_vel + delta_p/(RoeDensity*RoeSoundSpeed);
340  }
341 
342  /*--- Update residual ---*/
343 
344  for (iVar = 0; iVar < nVar; iVar++)
345  for (kVar = 0; kVar < nVar; kVar++)
346  val_residual[iVar] -= (1.0-kappa)*Lambda[kVar]*delta_wave[kVar]*P_Tensor[iVar][kVar]*Area;
347 
348  if (!implicit) return;
349 
350  /*--- If implicit use the Jacobians of the standard Roe scheme as an approximation ---*/
351 
352  GetPMatrix_inv(&RoeDensity, RoeVelocity, &RoeSoundSpeed, UnitNormal, invP_Tensor);
353 
354  for (iVar = 0; iVar < nVar; iVar++) {
355  for (jVar = 0; jVar < nVar; jVar++) {
356  /*--- Compute |Proj_ModJac_Tensor| = P x |Lambda| x inverse P ---*/
357  su2double Proj_ModJac_Tensor_ij = 0.0;
358  for (kVar = 0; kVar < nVar; kVar++)
359  Proj_ModJac_Tensor_ij += P_Tensor[iVar][kVar]*Lambda[kVar]*invP_Tensor[kVar][jVar];
360 
361  val_Jacobian_i[iVar][jVar] += (1.0-kappa)*Proj_ModJac_Tensor_ij*Area;
362  val_Jacobian_j[iVar][jVar] -= (1.0-kappa)*Proj_ModJac_Tensor_ij*Area;
363  }
364  }
365 
366 }

520  /*-- Unit Normal ---*/
521 
522  for (iDim = 0; iDim < nDim; iDim++)
523  UnitNormal[iDim] = Normal[iDim]/Area;
524 
525  /*--- Primitive variables at point i ---*/
526 
527  for (iDim = 0; iDim < nDim; iDim++)
528  Velocity_i[iDim] = V_i[iDim+1];
529  Pressure_i = V_i[nDim+1];
530  Density_i = V_i[nDim+2];
531  Enthalpy_i = V_i[nDim+3];
532  Energy_i = Enthalpy_i - Pressure_i/Density_i;
533  SoundSpeed_i = sqrt(fabs(Pressure_i*Gamma/Density_i));
534 
535  /*--- Primitive variables at point j ---*/
536 
537  for (iDim = 0; iDim < nDim; iDim++)
538  Velocity_j[iDim] = V_j[iDim+1];
539  Pressure_j = V_j[nDim+1];
540  Density_j = V_j[nDim+2];
541  Enthalpy_j = V_j[nDim+3];
542  Energy_j = Enthalpy_j - Pressure_j/Density_j;
543  SoundSpeed_j = sqrt(fabs(Pressure_j*Gamma/Density_j));
544 
545  /*--- Recompute conservative variables ---*/
546 
547  U_i[0] = Density_i; U_j[0] = Density_j;
548  for (iDim = 0; iDim < nDim; iDim++) {
549  U_i[iDim+1] = Density_i*Velocity_i[iDim]; U_j[iDim+1] = Density_j*Velocity_j[iDim];
550  }
551  U_i[nDim+1] = Density_i*Energy_i; U_j[nDim+1] = Density_j*Energy_j;
552 
553  /*--- Roe-averaged variables at interface between i & j ---*/
554 
555  R = sqrt(fabs(Density_j/Density_i));
556  RoeDensity = R*Density_i;
557  sq_vel = 0.0;
558  for (iDim = 0; iDim < nDim; iDim++) {
559  RoeVelocity[iDim] = (R*Velocity_j[iDim]+Velocity_i[iDim])/(R+1);
560  sq_vel += RoeVelocity[iDim]*RoeVelocity[iDim];
561  }
562  RoeEnthalpy = (R*Enthalpy_j+Enthalpy_i)/(R+1);
563  RoeSoundSpeed = sqrt(fabs((Gamma-1)*(RoeEnthalpy-0.5*sq_vel)));
564  RoePressure = RoeDensity/Gamma*RoeSoundSpeed*RoeSoundSpeed;
565 
566  /*--- Compute ProjFlux_i ---*/
567  GetInviscidProjFlux(&Density_i, Velocity_i, &Pressure_i, &Enthalpy_i, Normal, ProjFlux_i);
568

576  ProjVelocity_j += Velocity_j[iDim]*UnitNormal[iDim];
577  }

623

628  /*--- Jacobians of the inviscid flux, scaled by
629  0.5 because Flux ~ 0.5*(fc_i+fc_j)*Normal ---*/
630  GetInviscidProjJac(Velocity_i, &Energy_i, Normal, 0.5, Jacobian_i);
631  GetInviscidProjJac(Velocity_j, &Energy_j, Normal, 0.5, Jacobian_j);
632  }
633 
634  for (iVar = 0; iVar < nVar; iVar ++) {
635  for (jVar = 0; jVar < nVar; jVar ++) {
636  Matrix[iVar][jVar] = 0.0;
637  for (kVar = 0; kVar < nVar; kVar++)
638  Matrix[iVar][jVar] += absPeJac[iVar][kVar]*R_Tensor[kVar][jVar];
639  }
640  }
641 
642  for (iVar = 0; iVar < nVar; iVar ++) {
643  for (jVar = 0; jVar < nVar; jVar ++) {
644  Art_Visc[iVar][jVar] = 0.0;
645  for (kVar = 0; kVar < nVar; kVar++)
646  Art_Visc[iVar][jVar] += invRinvPe[iVar][kVar]*Matrix[kVar][jVar];
647  }
648  }
649 
650  /*--- Roe's Flux approximation ---*/
651  for (iVar = 0; iVar < nVar; iVar++) {
652  Flux[iVar] = 0.5*(ProjFlux_i[iVar]+ProjFlux_j[iVar]);
653  for (jVar = 0; jVar < nVar; jVar++) {
654  Flux[iVar] -= 0.5*Art_Visc[iVar][jVar]*Diff_U[jVar];
655  if (implicit) {
656  Jacobian_i[iVar][jVar] += 0.5*Art_Visc[iVar][jVar];
657  Jacobian_j[iVar][jVar] -= 0.5*Art_Visc[iVar][jVar];
658  }
659  }
660  }
661 
662  /*--- Contributions due to mesh motion---*/
663  if (dynamic_grid) {
664  ProjVelocity = 0.0;
665  for (iDim = 0; iDim < nDim; iDim++)
666  ProjVelocity += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*UnitNormal[iDim];
667  for (iVar = 0; iVar < nVar; iVar++) {
668  Flux[iVar] -= ProjVelocity * 0.5*(U_i[iVar]+U_j[iVar]);
669  /*--- Implicit terms ---*/
670  if (implicit) {
671  Jacobian_i[iVar][iVar] -= 0.5*ProjVelocity;
672  Jacobian_j[iVar][iVar] -= 0.5*ProjVelocity;
673  }
674  }
675  }
676 
677  return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
