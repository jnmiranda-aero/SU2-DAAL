38void CTecplotFileWriter::WriteData(string val_filename){
39
40 /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
41 val_filename.append(fileExt);
42
43 if (!dataSorter->GetConnectivitySorted()){
44 SU2_MPI::Error("Connectivity must be sorted.", CURRENT_FUNCTION);
45 }
46
47 const vector<string> fieldNames = dataSorter->GetFieldNames();
48
49 unsigned short iVar;
50
51 unsigned long iPoint, iElem;
52
53 int iProcessor;
54
55 ofstream Tecplot_File;
56
57 fileSize = 0.0;
58
59 /*--- Set a timer for the file writing. ---*/
60
61 startTime = SU2_MPI::Wtime();
62
63 /*--- Reduce the total number of each element. ---*/
64
65 unsigned long nParallel_Line = dataSorter->GetnElem(LINE),
66 nParallel_Tria = dataSorter->GetnElem(TRIANGLE),
67 nParallel_Quad = dataSorter->GetnElem(QUADRILATERAL),
68 nParallel_Tetr = dataSorter->GetnElem(TETRAHEDRON),
69 nParallel_Hexa = dataSorter->GetnElem(HEXAHEDRON),
70 nParallel_Pris = dataSorter->GetnElem(PRISM),
71 nParallel_Pyra = dataSorter->GetnElem(PYRAMID);
72
73 unsigned long nTot_Line = dataSorter->GetnElemGlobal(LINE),
74 nTot_Tria = dataSorter->GetnElemGlobal(TRIANGLE),
75 nTot_Quad = dataSorter->GetnElemGlobal(QUADRILATERAL),
76 nTot_Tetr = dataSorter->GetnElemGlobal(TETRAHEDRON),
77 nTot_Hexa = dataSorter->GetnElemGlobal(HEXAHEDRON),
78 nTot_Pris = dataSorter->GetnElemGlobal(PRISM),
79 nTot_Pyra = dataSorter->GetnElemGlobal(PYRAMID);
80
81 /*--- Open Tecplot ASCII file and write the header. ---*/
82
83 if (rank == MASTER_NODE) {
84 Tecplot_File.open(val_filename.c_str(), ios::out);
85 Tecplot_File.precision(6);
86 Tecplot_File << "TITLE = \"Visualization of the solution\"" << endl;
87
88 Tecplot_File << "VARIABLES = ";
89 for (iVar = 0; iVar < fieldNames.size()-1; iVar++) {
90 Tecplot_File << "\"" << fieldNames[iVar] << "\",";
91 }
92 Tecplot_File << "\"" << fieldNames[fieldNames.size()-1] << "\"" << endl;
93
94 /*--- Write the header ---*/
95
96 Tecplot_File << "ZONE ";
97
98 if (timeStep > 0.0){
99 Tecplot_File << "STRANDID="<<SU2_TYPE::Int(timeIter+1)<<", SOLUTIONTIME="<< timeIter*timeStep <<", ";
100 }
101
102 Tecplot_File << "NODES= "<< dataSorter->GetnPointsGlobal() <<", ELEMENTS= "<< dataSorter->GetnElemGlobal();
103
104 if (dataSorter->GetnDim() == 3){
105 if ((nTot_Quad > 0 || nTot_Tria > 0) && (nTot_Hexa + nTot_Pris + nTot_Pyra + nTot_Tetr == 0)){
106 Tecplot_File << ", DATAPACKING=POINT, ZONETYPE=FEQUADRILATERAL" << endl;
107 }
108 else {
109 Tecplot_File <<", DATAPACKING=POINT, ZONETYPE=FEBRICK"<< endl;
110 }
111 }
112 else {
113 if (nTot_Line > 0 && (nTot_Tria + nTot_Quad == 0)){
114 Tecplot_File << ", DATAPACKING=POINT, ZONETYPE=FELINESEG"<< endl;
115 }
116 else{
117 Tecplot_File << ", DATAPACKING=POINT, ZONETYPE=FEQUADRILATERAL"<< endl;
118 }
119 }
120 Tecplot_File.close();
121 }
122
123#ifdef HAVE_MPI
124 SU2_MPI::Barrier(SU2_MPI::GetComm());
125#endif
126
127 /*--- Each processor opens the file. ---*/
128
129 Tecplot_File.open(val_filename.c_str(), ios::out | ios::app);
130
131 /*--- Write surface and volumetric solution data. ---*/
132
133 for (iProcessor = 0; iProcessor < size; iProcessor++) {
134 if (rank == iProcessor) {
135
136 /*--- Write the node data from this proc ---*/
137
138
139 for (iPoint = 0; iPoint < dataSorter->GetnPoints(); iPoint++) {
140 for (iVar = 0; iVar < fieldNames.size(); iVar++)
141 Tecplot_File << scientific << dataSorter->GetData(iVar, iPoint) << "\t";
142 Tecplot_File << endl;
143 }
144 }
145
146 Tecplot_File.flush();
147#ifdef HAVE_MPI
148 SU2_MPI::Barrier(SU2_MPI::GetComm());
149#endif
150 }
151
152
153 /*--- Write connectivity data. ---*/
154
155 for (iProcessor = 0; iProcessor < size; iProcessor++) {
156 if (rank == iProcessor) {
157
158 for (iElem = 0; iElem < nParallel_Line; iElem++) {
159 Tecplot_File << dataSorter->GetElemConnectivity(LINE, iElem, 0) << "\t";
160 Tecplot_File << dataSorter->GetElemConnectivity(LINE, iElem, 1)<< "\n";
161 }
162
163
164 for (iElem = 0; iElem < nParallel_Tria; iElem++) {
165 Tecplot_File << dataSorter->GetElemConnectivity(TRIANGLE, iElem, 0) << "\t";
166 Tecplot_File << dataSorter->GetElemConnectivity(TRIANGLE, iElem, 1) << "\t";
167 Tecplot_File << dataSorter->GetElemConnectivity(TRIANGLE, iElem, 2) << "\t";
168 Tecplot_File << dataSorter->GetElemConnectivity(TRIANGLE, iElem, 2) << "\n";
169 }
170
171 for (iElem = 0; iElem < nParallel_Quad; iElem++) {
172 Tecplot_File << dataSorter->GetElemConnectivity(QUADRILATERAL, iElem, 0) << "\t";
173 Tecplot_File << dataSorter->GetElemConnectivity(QUADRILATERAL, iElem, 1) << "\t";
174 Tecplot_File << dataSorter->GetElemConnectivity(QUADRILATERAL, iElem, 2) << "\t";
175 Tecplot_File << dataSorter->GetElemConnectivity(QUADRILATERAL, iElem, 3) << "\n";
176 }
177
178 for (iElem = 0; iElem < nParallel_Tetr; iElem++) {
179 Tecplot_File << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 0) << "\t" << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 1) << "\t";
180 Tecplot_File << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 2) << "\t" << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 2) << "\t";
181 Tecplot_File << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 3) << "\t" << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 3) << "\t";
182 Tecplot_File << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 3) << "\t" << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 3) << "\n";
183 }
184
185 for (iElem = 0; iElem < nParallel_Hexa; iElem++) {
186 Tecplot_File << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 0) << "\t" << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 1) << "\t";
187 Tecplot_File << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 2) << "\t" << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 3) << "\t";
188 Tecplot_File << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 4) << "\t" << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 5) << "\t";
189 Tecplot_File << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 6) << "\t" << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 7) << "\n";
190 }
191
192 for (iElem = 0; iElem < nParallel_Pris; iElem++) {
193 Tecplot_File << dataSorter->GetElemConnectivity(PRISM, iElem, 0) << "\t" << dataSorter->GetElemConnectivity(PRISM, iElem, 1) << "\t";
194 Tecplot_File << dataSorter->GetElemConnectivity(PRISM, iElem, 1) << "\t" << dataSorter->GetElemConnectivity(PRISM, iElem, 2) << "\t";
195 Tecplot_File << dataSorter->GetElemConnectivity(PRISM, iElem, 3) << "\t" << dataSorter->GetElemConnectivity(PRISM, iElem, 4) << "\t";
196 Tecplot_File << dataSorter->GetElemConnectivity(PRISM, iElem, 4) << "\t" << dataSorter->GetElemConnectivity(PRISM, iElem, 5) << "\n";
197 }
198
199 for (iElem = 0; iElem < nParallel_Pyra; iElem++) {
200 Tecplot_File << dataSorter->GetElemConnectivity(PYRAMID, iElem, 0) << "\t" << dataSorter->GetElemConnectivity(PYRAMID, iElem, 1) << "\t";
201 Tecplot_File << dataSorter->GetElemConnectivity(PYRAMID, iElem, 2) << "\t" << dataSorter->GetElemConnectivity(PYRAMID, iElem, 3) << "\t";
202 Tecplot_File << dataSorter->GetElemConnectivity(PYRAMID, iElem, 4) << "\t" << dataSorter->GetElemConnectivity(PYRAMID, iElem, 4) << "\t";
203 Tecplot_File << dataSorter->GetElemConnectivity(PYRAMID, iElem, 4) << "\t" << dataSorter->GetElemConnectivity(PYRAMID, iElem, 4) << "\n";
204 }
205
206
207 }
208 Tecplot_File.flush();
209#ifdef HAVE_MPI
210 SU2_MPI::Barrier(SU2_MPI::GetComm());
211#endif
212 }
213
214 Tecplot_File.close();
215
216 /*--- Compute and store the write time. ---*/
217
218 stopTime = SU2_MPI::Wtime();
219
220 usedTime = stopTime-startTime;
221
222 fileSize = DetermineFilesize(val_filename);
223
224 /*--- Compute and store the bandwidth ---*/
225
226 bandwidth = fileSize/(1.0e6)/usedTime;
227}
