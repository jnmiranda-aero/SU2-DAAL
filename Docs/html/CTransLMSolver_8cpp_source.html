41 CTransLMSolver::CTransLMSolver(CGeometry *geometry, CConfig *config, unsigned short iMesh)
42  : CTurbSolver(geometry, config, true) {
43  unsigned long iPoint;
44  ifstream restart_file;
45  string text_line;
46 
47  bool multizone = config->GetMultizone_Problem();
48 
49  /*--- Dimension of the problem --> 2 Transport equations (intermittency, Reth) ---*/
50  nVar = 2;
51  nPrimVar = 2;
52  nPoint = geometry->GetnPoint();
53  nPointDomain = geometry->GetnPointDomain();
54 
55  /*--- Initialize nVarGrad for deallocation ---*/
56 
57  nVarGrad = nVar;
58 
59  /*--- Define geometry constants in the solver structure ---*/
60 
61  nDim = geometry->GetnDim();
62 
63  /*--- Define variables needed for transition from config file */
64  options = config->GetLMParsedOptions();
65  TransCorrelations.SetOptions(options);
66  TurbFamily = TurbModelFamily(config->GetKind_Turb_Model());
67 
68  /*--- Single grid simulation ---*/
69 
70  if (iMesh == MESH_0) {
71 
72  /*--- Define some auxiliary vector related with the residual ---*/
73 
74  Residual_RMS.resize(nVar,0.0);
75  Residual_Max.resize(nVar,0.0);
76  Point_Max.resize(nVar,0);
77  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
78 
79  /*--- Initialization of the structure of the whole Jacobian ---*/
80 
81  if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (LM transition model)." << endl;
82  Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config, ReducerStrategy);
83  LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
84  LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
85  System.SetxIsZero(true);
86 
87  if (ReducerStrategy)
88  EdgeFluxes.Initialize(geometry->GetnEdge(), geometry->GetnEdge(), nVar, nullptr);
89 
90  /*--- Initialize the BGS residuals in multizone problems. ---*/
91  if (multizone){
92  Residual_BGS.resize(nVar,0.0);
93  Residual_Max_BGS.resize(nVar,0.0);
94  Point_Max_BGS.resize(nVar,0);
95  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
96  }
97 
98  }
99 
100  /*--- Initialize lower and upper limits---*/
101  lowerlimit[0] = 1.0e-4;
102  upperlimit[0] = 5.0;
103 
104  lowerlimit[1] = 1.0e-4;
105  upperlimit[1] = 1.0e15;
106 
107  /*--- Far-field flow state quantities and initialization. ---*/
108  const su2double Intensity = config->GetTurbulenceIntensity_FreeStream()*100.0;
109 
110  const su2double Intermittency_Inf = 1.0;
111  su2double ReThetaT_Inf = 100.0;
112 
113  /*--- Momentum thickness Reynolds number, initialized from freestream turbulent intensity*/
114  if (Intensity <= 1.3) {
115  if(Intensity >=0.027) {
116  ReThetaT_Inf = (1173.51-589.428*Intensity+0.2196/(Intensity*Intensity));
117  }
118  else {
119  ReThetaT_Inf = (1173.51-589.428*Intensity+0.2196/(0.27*0.27));
120  }
121  }
122  else if(Intensity>1.3) {
123  ReThetaT_Inf = 331.5*pow(Intensity-0.5658,-0.671);
124  }
125 
126  Solution_Inf[0] = Intermittency_Inf;
127  Solution_Inf[1] = ReThetaT_Inf;
128 
129  /*--- Initialize the solution to the far-field state everywhere. ---*/
130  nodes = new CTransLMVariable(Intermittency_Inf, ReThetaT_Inf, 1.0, 1.0, nPoint, nDim, nVar, config);
131  SetBaseClassPointerToNodes();
132 
133  /*--- MPI solution ---*/
134 
135  InitiateComms(geometry, config, MPI_QUANTITIES::SOLUTION);
136  CompleteComms(geometry, config, MPI_QUANTITIES::SOLUTION);
137 
138  /*--- Initializate quantities for SlidingMesh Interface ---*/
139 
140  SlidingState.resize(nMarker);
141  SlidingStateNodes.resize(nMarker);
142 
143  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
144  if (config->GetMarker_All_KindBC(iMarker) == FLUID_INTERFACE) {
145  SlidingState[iMarker].resize(nVertex[iMarker], nPrimVar+1) = nullptr;
146  SlidingStateNodes[iMarker].resize(nVertex[iMarker],0);
147  }
148  }
149 
150  /*-- Allocation of inlets has to happen in derived classes (not CTurbSolver),
151  due to arbitrary number of turbulence variables ---*/
152 
153  Inlet_TurbVars.resize(nMarker);
154  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
155  Inlet_TurbVars[iMarker].resize(nVertex[iMarker],nVar);
156  for (unsigned long iVertex = 0; iVertex < nVertex[iMarker]; ++iVertex) {
157  Inlet_TurbVars[iMarker](iVertex,0) = Intermittency_Inf;
158  Inlet_TurbVars[iMarker](iVertex,1) = ReThetaT_Inf;
159  }
160  }
161 
162  const su2double CFL = config->GetCFL(MGLevel)*config->GetCFLRedCoeff_Turb();
163  for (iPoint = 0; iPoint < nPoint; iPoint++) {
164  nodes->SetLocalCFL(iPoint, CFL);
165  }
166  Min_CFL_Local = CFL;
167  Max_CFL_Local = CFL;
168  Avg_CFL_Local = CFL;
169 
170  /*--- Add the solver name. ---*/
171  SolverName = "LM model";
172 
173 }

183 void CTransLMSolver::Postprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh) {
184 
185  /*--- Compute LM model gradients. ---*/
186 
187  if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
188  SetSolution_Gradient_GG(geometry, config, -1);
189  }
190  if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
191  SetSolution_Gradient_LS(geometry, config, -1);
192  }
193 
194  AD::StartNoSharedReading();
195  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
196  auto* turbNodes = su2staticcast_p<CTurbVariable*>(solver_container[TURB_SOL]->GetNodes());
197 
198  SU2_OMP_FOR_STAT(omp_chunk_size)
199  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint ++) {
200 
201  // Here the nodes already have the new solution, thus I have to compute everything from scratch
202 
203  const su2double Gamma = config->GetGamma();
204  const su2double Gas_Constant = config->GetGas_Constant();
205  const su2double* vel_inf = config->GetVelocity_FreeStream();
206  const su2double Vel_inf_Mag = sqrt(vel_inf[0]*vel_inf[0] + vel_inf[1]*vel_inf[1] + vel_inf[2]*vel_inf[2]);
207  const su2double Density_inf = config->GetDensity_FreeStream();
208  const su2double Pressure_inf = config->GetPressure_FreeStream();
209  const su2double Temperature_inf = config->GetTemperature_FreeStream();
210 
211  const su2double rho = flowNodes->GetDensity(iPoint);
212  // const su2double Pressure_i = flowNodes->GetPressure(iPoint);
213  const su2double Pressure_i = flowNodes->GetPressure(iPoint);
214  const su2double mu = flowNodes->GetLaminarViscosity(iPoint);
215  const su2double muT = turbNodes->GetmuT(iPoint);
216  const su2double dist = geometry->nodes->GetWall_Distance(iPoint);
217  su2double VorticityMag = GeometryToolbox::Norm(3, flowNodes->GetVorticity(iPoint));
218  su2double StrainMag = flowNodes->GetStrainMag(iPoint);
219  VorticityMag = max(VorticityMag, 1e-12);
220  StrainMag = max(StrainMag, 1e-12); // safety against division by zero
221  const su2double Intermittency = nodes->GetSolution(iPoint,0);
222  const su2double Re_t = nodes->GetSolution(iPoint,1);
223  const su2double Re_v = rho*dist*dist*StrainMag/mu;
224  const su2double vel_u = flowNodes->GetVelocity(iPoint, 0);
225  const su2double vel_v = flowNodes->GetVelocity(iPoint, 1);
226  const su2double vel_w = (nDim ==3) ? flowNodes->GetVelocity(iPoint, 2) : 0.0;
227  const su2double VelocityMag = sqrt(vel_u*vel_u + vel_v*vel_v + vel_w*vel_w);
228  su2double omega = 0.0;
229  su2double k = 0.0;
230 
231  su2double Velocity_BLEdge = 0.1*sqrt(VelocityMag * VelocityMag + ((2 * Gamma) / (Gamma - 1)) * (1 - pow(Pressure_i / Pressure_inf, (1 - (1 / Gamma)))) * (Pressure_i / Density_inf)); //remove 0.1
232  su2double Speed_of_Sound_BLEdge = sqrt((Gamma * Gas_Constant * Temperature_inf) * pow(Pressure_i / Pressure_inf, ((1 - Gamma) / Gamma)));
233 
234  // Calculate Mach Number at Boundary Layer Edge
235  su2double Mach_BLEdge = Velocity_BLEdge / Speed_of_Sound_BLEdge;
236 
237  // Compute Compressibility Correction Factor
238  su2double C_Me = 1.0 - 0.06124 * Mach_BLEdge + 0.2402 * pow(Mach_BLEdge, 2) - 0.00346 * pow(Mach_BLEdge, 3);
239 
240  if(TurbFamily == TURB_FAMILY::KW){
241  omega = turbNodes->GetSolution(iPoint,1);
242  k = turbNodes->GetSolution(iPoint,0);
243  }
244  su2double Tu = 1.0;
245  if(TurbFamily == TURB_FAMILY::KW)
246  Tu = max(100.0*sqrt( 2.0 * k / 3.0 ) / VelocityMag,0.027);
247  if(TurbFamily == TURB_FAMILY::SA)
248  Tu = config->GetTurbulenceIntensity_FreeStream()*100;
249 
250  // const su2double Re_theta_t = nodes->GetSolution(iPoint,1);
251  // const su2double Corr_Rec = TransCorrelations.ReThetaC_Correlations(Tu, Re_theta_t);
252 
253  su2double Corr_Rec = TransCorrelations.ReThetaC_Correlations(Tu, Re_t);
254 
255  // Apply the compressibility correction
256  su2double Corr_Rec_Corrected = Corr_Rec / C_Me;
257 
258  su2double R_t = 1.0;
259  if(TurbFamily == TURB_FAMILY::KW)
260  R_t = rho*k/ mu/ omega;
261  if(TurbFamily == TURB_FAMILY::SA)
262  R_t = muT/ mu;
263 
264  const su2double f_reattach = exp(-pow(R_t/20,4));
265  su2double f_wake = 0.0;
266  if(TurbFamily == TURB_FAMILY::KW){
267  const su2double re_omega = rho*omega*dist*dist/mu;
268  f_wake = exp(-pow(re_omega/(1.0e+05),2));
269  }
270  if(TurbFamily == TURB_FAMILY::SA)
271  f_wake = 1.0;
272 
273  const su2double theta_bl = Re_t*mu / rho /VelocityMag;
274  const su2double delta_bl = 7.5*theta_bl;
275  const su2double delta = 50.0*VorticityMag*dist/VelocityMag*delta_bl + 1e-20;
276  const su2double var1 = (Intermittency-1.0/50.0)/(1.0-1.0/50.0);
277  const su2double var2 = 1.0 - pow(var1,2.0);
278  const su2double f_theta = min(max(f_wake*exp(-pow(dist/delta, 4)), var2), 1.0);
279  su2double Intermittency_Sep = 2.0*max(0.0, Re_v/(3.235*Corr_Rec)-1.0)*f_reattach;
280  Intermittency_Sep = min(Intermittency_Sep,2.0)*f_theta;
281  Intermittency_Sep = min(max(0.0, Intermittency_Sep), 2.0);
282 
283  nodes -> SetIntermittencySep(iPoint, Intermittency_Sep);
284  nodes -> SetIntermittencyEff(iPoint, Intermittency_Sep);
285 
286  }
287  END_SU2_OMP_FOR
288 
289  AD::EndNoSharedReading();
290 }

306 void CTransLMSolver::Source_Residual(CGeometry *geometry, CSolver **solver_container,
307  CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
308 
309  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
310 
311  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
312  //auto* turbNodes = su2staticcast_p<CFlowVariable*>(solver_container[TURB_SOL]->GetNodes());
313  CVariable* turbNodes = solver_container[TURB_SOL]->GetNodes();
314 
315  /*--- Pick one numerics object per thread. ---*/
316  auto* numerics = numerics_container[SOURCE_FIRST_TERM + omp_get_thread_num()*MAX_TERMS];
317 
318  /*--- Loop over all points. ---*/
319 
320  AD::StartNoSharedReading();
321 
322  SU2_OMP_FOR_DYN(omp_chunk_size)
323  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
324 
325 
326 
327  /*--- Conservative variables w/o reconstruction ---*/
328 
329  numerics->SetPrimitive(flowNodes->GetPrimitive(iPoint), nullptr);
330 
331  /*--- Gradient of the primitive and conservative variables ---*/
332 
333  numerics->SetPrimVarGradient(flowNodes->GetGradient_Primitive(iPoint), nullptr);
334 
335  /*--- Turbulent variables w/o reconstruction, and its gradient ---*/
336  /*--- ScalarVar & ScalarVarGradient : Turbulence model solution(k&w) ---*/
337 
338  numerics->SetScalarVar(turbNodes->GetSolution(iPoint), nullptr);
339  numerics->SetScalarVarGradient(turbNodes->GetGradient(iPoint), nullptr);
340 
341  /*--- Transition variables w/o reconstruction, and its gradient ---*/
342 
343  numerics->SetTransVar(nodes->GetSolution(iPoint), nullptr);
344  numerics->SetTransVarGradient(nodes->GetGradient(iPoint), nullptr);
345 
346  /*--- Set volume ---*/
347 
348  numerics->SetVolume(geometry->nodes->GetVolume(iPoint));
349 
350  /*--- Set distance to the surface ---*/
351 
352  numerics->SetDistance(geometry->nodes->GetWall_Distance(iPoint), 0.0);
353 
354  /*--- Set vorticity and strain rate magnitude ---*/
355 
356  numerics->SetVorticity(flowNodes->GetVorticity(iPoint), nullptr);
357 
358  numerics->SetStrainMag(flowNodes->GetStrainMag(iPoint), 0.0);
359 
360  /*--- Set coordinate (for debugging) ---*/
361  numerics->SetCoord(geometry->nodes->GetCoord(iPoint), nullptr);
362 
363  if (options.LM2015) {
364  /*--- Set local grid length (for LM2015)*/
365  numerics->SetLocalGridLength(geometry->nodes->GetMaxLength(iPoint));
366  }
367 
368  /*--- Compute the source term ---*/
369 
370  auto residual = numerics->ComputeResidual(config);
371 
372  /*--- Subtract residual and the Jacobian ---*/
373 
374  LinSysRes.SubtractBlock(iPoint, residual);
375  if (implicit) Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
376 
377  }
378  END_SU2_OMP_FOR
379 
380  AD::EndNoSharedReading();
381 
382 }

388 void CTransLMSolver::BC_HeatFlux_Wall(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
389  CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
390 
391  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
392 
393  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
394  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
395 
396  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
397 
398  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
399 
400  if (geometry->nodes->GetDomain(iPoint)) {
401 
402  /*--- Allocate the value at the infinity ---*/
403 
404  auto V_infty = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
405 
406  /*--- Retrieve solution at the farfield boundary node ---*/
407 
408  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
409 
410  conv_numerics->SetPrimitive(V_domain, V_infty);
411 
412  /*--- Set turbulent variable at the wall, and at infinity ---*/
413 
414  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Solution_Inf);
415 
416  /*--- Set Normal (it is necessary to change the sign) ---*/
417  /*--- It's mean wall normal zero flux. */
418 
419  su2double Normal[MAXNDIM] = {0.0};
420  for (auto iDim = 0u; iDim < nDim; iDim++)
421  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
422  conv_numerics->SetNormal(Normal);
423 
424  /*--- Grid Movement ---*/
425 
426  if (dynamic_grid)
427  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint),
428  geometry->nodes->GetGridVel(iPoint));
429 
430  /*--- Compute residuals and Jacobians ---*/
431 
432  auto residual = conv_numerics->ComputeResidual(config);
433 
434  /*--- Add residuals and Jacobians ---*/
435 
436  LinSysRes.AddBlock(iPoint, residual);
437  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
438  }
439  }
440  END_SU2_OMP_FOR
441

451 void CTransLMSolver::BC_Inlet(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics, CNumerics *visc_numerics, CConfig *config,
452  unsigned short val_marker) {
453  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
454 
455  /*--- Loop over all the vertices on this boundary marker ---*/
456 
457  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
458  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
459 
460  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
461 
462  /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
463 
464  if (geometry->nodes->GetDomain(iPoint)) {
465 
466  /*--- Normal vector for this vertex (negate for outward convention) ---*/
467 
468  su2double Normal[MAXNDIM] = {0.0};
469  for (auto iDim = 0u; iDim < nDim; iDim++)
470  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
471  conv_numerics->SetNormal(Normal);
472 
473  /*--- Allocate the value at the inlet ---*/
474 
475  auto V_inlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
476 
477  /*--- Retrieve solution at the farfield boundary node ---*/
478 
479  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
480 
481  /*--- Set various quantities in the solver class ---*/
482 
483  conv_numerics->SetPrimitive(V_domain, V_inlet);
484 
485  /*--- Non-dimensionalize Inlet_TurbVars if Inlet-Files are used. ---*/
486  su2double Inlet_Vars[MAXNVAR];
487  Inlet_Vars[0] = Inlet_TurbVars[val_marker][iVertex][0];
488  Inlet_Vars[1] = Inlet_TurbVars[val_marker][iVertex][1];
489  if (config->GetInlet_Profile_From_File()) {
490  Inlet_Vars[0] /= pow(config->GetVelocity_Ref(), 2);
491  Inlet_Vars[1] *= config->GetViscosity_Ref() / (config->GetDensity_Ref() * pow(config->GetVelocity_Ref(), 2));
492  }
493 
494  /*--- Set the LM variable states. ---*/
495  /*--- Load the inlet transition LM model variables (uniform by default). ---*/
496 
497  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Inlet_Vars);
498 
499  /*--- Set various other quantities in the solver class ---*/
500 
501  if (dynamic_grid)
502  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint),
503  geometry->nodes->GetGridVel(iPoint));
504 
505  /*--- Compute the residual using an upwind scheme ---*/
506 
507  auto residual = conv_numerics->ComputeResidual(config);
508  LinSysRes.AddBlock(iPoint, residual);
509 
510  /*--- Jacobian contribution for implicit integration ---*/
511 
512  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
513  }
514  }
515  END_SU2_OMP_FOR
516 
517 }

524 void CTransLMSolver::LoadRestart(CGeometry** geometry, CSolver*** solver, CConfig* config, int val_iter,
525  bool val_update_geo) {
526 
527  const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
528 
529  /*--- To make this routine safe to call in parallel most of it can only be executed by one thread. ---*/
530  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
531  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
532 
533  if (config->GetRead_Binary_Restart()) {
534  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
535  } else {
536  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
537  }
538 
539  /*--- Skip flow variables ---*/
540 
541  unsigned short skipVars = nDim + solver[MESH_0][FLOW_SOL]->GetnVar() + solver[MESH_0][TURB_SOL] ->GetnVar();
542 
543  /*--- Adjust the number of solution variables in the incompressible
544  restart. We always carry a space in nVar for the energy equation in the
545  mean flow solver, but we only write it to the restart if it is active.
546  Therefore, we must reduce skipVars here if energy is inactive so that
547  the turbulent variables are read correctly. ---*/
548 
549  const bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
550  const bool energy = config->GetEnergy_Equation();
551  const bool weakly_coupled_heat = config->GetWeakly_Coupled_Heat();
552 
553  if (incompressible && ((!energy) && (!weakly_coupled_heat))) skipVars--;
554 
555  /*--- Load data from the restart into correct containers. ---*/
556 
557  unsigned long counter = 0;
558  for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
559  /*--- Retrieve local index. If this node from the restart file lives
560  on the current processor, we will load and instantiate the vars. ---*/
561 
562  const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
563 
564  if (iPoint_Local > -1) {
565  /*--- We need to store this point's data, so jump to the correct
566  offset in the buffer of data from the restart file and load it. ---*/
567 
568  const auto index = counter * Restart_Vars[1] + skipVars;
569  for (auto iVar = 0u; iVar < nVar; iVar++) nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index + iVar]);
570  nodes ->SetIntermittencySep(iPoint_Local, Restart_Data[index + 2]);
571  nodes ->SetIntermittencyEff(iPoint_Local, Restart_Data[index + 3]);
572 
573  /*--- Increment the overall counter for how many points have been loaded. ---*/
574  counter++;
575  }
576  }
577 
578  /*--- Detect a wrong solution file ---*/
579 
580  if (counter != nPointDomain) {
581  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" does not match with the mesh file!\n") +
582  string("This can be caused by empty lines at the end of the file."),
583  CURRENT_FUNCTION);
584  }
585 
586  } // end SU2_OMP_MASTER, pre and postprocessing are thread-safe.
587  END_SU2_OMP_SAFE_GLOBAL_ACCESS
588 
589  /*--- MPI solution and compute the eddy viscosity ---*/
590 
591  solver[MESH_0][TRANS_SOL]->InitiateComms(geometry[MESH_0], config, MPI_QUANTITIES::SOLUTION);
592  solver[MESH_0][TRANS_SOL]->CompleteComms(geometry[MESH_0], config, MPI_QUANTITIES::SOLUTION);
593 
594  /*--- For turbulent+species simulations the solver Pre-/Postprocessing is done by the species solver. ---*/
595  if (config->GetKind_Species_Model() == SPECIES_MODEL::NONE) {
596  solver[MESH_0][FLOW_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
597  RUNTIME_FLOW_SYS, false);
598  solver[MESH_0][TURB_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
599  solver[MESH_0][TRANS_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
600  }
601 
602  /*--- Interpolate the solution down to the coarse multigrid levels ---*/
603 
604  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
605 
606  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][TRANS_SOL]->GetNodes()->GetSolution(),
607  *geometry[iMesh], solver[iMesh][TRANS_SOL]->GetNodes()->GetSolution());
608  solver[iMesh][TRANS_SOL]->InitiateComms(geometry[iMesh], config, MPI_QUANTITIES::SOLUTION);
609  solver[iMesh][TRANS_SOL]->CompleteComms(geometry[iMesh], config, MPI_QUANTITIES::SOLUTION);
610 
611  if (config->GetKind_Species_Model() == SPECIES_MODEL::NONE) {
612  solver[iMesh][FLOW_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_FLOW_SYS,
613  false);
614  solver[iMesh][TRANS_SOL]->Postprocessing(geometry[iMesh], solver[iMesh], config, iMesh);
615  }
616  }
617 
618  /*--- Go back to single threaded execution. ---*/
619  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
620  /*--- Delete the class memory that is used to load the restart. ---*/
621 
622  Restart_Vars.clear();
623  Restart_Data.clear();
624  }
625  END_SU2_OMP_SAFE_GLOBAL_ACCESS
626 
627 }
