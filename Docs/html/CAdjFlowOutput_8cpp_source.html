40void CAdjFlowOutput::AddHistoryOutputFields_AdjScalarRMS_RES(const CConfig* config) {
41 if (!frozen_visc) {
42 switch (TurbModelFamily(turb_model)) {
43 case TURB_FAMILY::SA:
44 /// DESCRIPTION: Root-mean square residual of the adjoint nu tilde.
45 AddHistoryOutput("RMS_ADJ_NU_TILDE", "rms[A_nu]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint nu tilde.", HistoryFieldType::RESIDUAL);
46 break;
47 case TURB_FAMILY::KW:
48 /// DESCRIPTION: Root-mean square residual of the adjoint kinetic energy.
49 AddHistoryOutput("RMS_ADJ_TKE", "rms[A_k]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint kinetic energy.", HistoryFieldType::RESIDUAL);
50 /// DESCRIPTION: Root-mean square residual of the adjoint dissipation.
51 AddHistoryOutput("RMS_ADJ_DISSIPATION", "rms[A_w]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint dissipation.", HistoryFieldType::RESIDUAL);
52 break;
53 case TURB_FAMILY::NONE:
54 break;
55 }
56 }
57
58 if (config->GetKind_Species_Model() == SPECIES_MODEL::SPECIES_TRANSPORT) {
59 for (unsigned short iVar = 0; iVar < config->GetnSpecies(); iVar++) {
60 AddHistoryOutput("RMS_ADJ_SPECIES_" + std::to_string(iVar), "rms[A_rho*Y_" + std::to_string(iVar) + "]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint transported species.", HistoryFieldType::RESIDUAL);
61 }
62 }
63
64 if (config->GetKind_Species_Model() == SPECIES_MODEL::FLAMELET) {
65 for (auto iCV=0u; iCV < config->GetNControlVars(); iCV++) {
66 const auto& cv_name = config->GetControllingVariableName(iCV);
67 AddHistoryOutput("RMS_ADJ_"+cv_name, "rms[" + cv_name + "]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint " + cv_name, HistoryFieldType::RESIDUAL);
68 }
69 for (unsigned short i_scalar = 0; i_scalar < config->GetNUserScalars(); i_scalar++) {
70 const auto& scalar_name = config->GetUserScalarName(i_scalar);
71 AddHistoryOutput("RMS_ADJ_" + scalar_name, "rms[" + scalar_name + "]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint of " + scalar_name + " .", HistoryFieldType::RESIDUAL);
72 }
73 }

169void CAdjFlowOutput::LoadHistoryDataAdjScalar(const CConfig* config, const CSolver* const* solver) {
170 const auto adjturb_solver = solver[ADJTURB_SOL];
171 const auto adjspecies_solver = solver[ADJSPECIES_SOL];
172
173 if (!frozen_visc) {
174 switch (TurbModelFamily(turb_model)) {
175 case TURB_FAMILY::SA:
176 SetHistoryOutputValue("RMS_ADJ_NU_TILDE", log10(adjturb_solver->GetRes_RMS(0)));
177 SetHistoryOutputValue("MAX_ADJ_NU_TILDE", log10(adjturb_solver->GetRes_Max(0)));
178 if (multiZone) {
179 SetHistoryOutputValue("BGS_ADJ_NU_TILDE", log10(adjturb_solver->GetRes_BGS(0)));
180 }
181 break;
182 case TURB_FAMILY::KW:
183 SetHistoryOutputValue("RMS_ADJ_TKE", log10(adjturb_solver->GetRes_RMS(0)));
184 SetHistoryOutputValue("RMS_ADJ_DISSIPATION", log10(adjturb_solver->GetRes_RMS(1)));
185 SetHistoryOutputValue("MAX_ADJ_TKE", log10(adjturb_solver->GetRes_Max(0)));
186 SetHistoryOutputValue("MAX_ADJ_DISSIPATION", log10(adjturb_solver->GetRes_Max(1)));
187 if (multiZone) {
188 SetHistoryOutputValue("BGS_ADJ_TKE", log10(adjturb_solver->GetRes_BGS(0)));
189 SetHistoryOutputValue("BGS_ADJ_DISSIPATION", log10(adjturb_solver->GetRes_BGS(1)));
190 }
191 break;
192 case TURB_FAMILY::NONE:
193 break;
194 }
195
196 if (turb_model != TURB_MODEL::NONE) {
197 SetHistoryOutputValue("LINSOL_ITER_TURB", adjturb_solver->GetIterLinSolver());
198 SetHistoryOutputValue("LINSOL_RESIDUAL_TURB", log10(adjturb_solver->GetResLinSolver()));
199 }
200 }
201
202 if (config->GetKind_Species_Model() == SPECIES_MODEL::SPECIES_TRANSPORT) {
203 for (unsigned short iVar = 0; iVar < config->GetnSpecies(); iVar++) {
204 SetHistoryOutputValue("RMS_ADJ_SPECIES_" + std::to_string(iVar), log10(adjspecies_solver->GetRes_RMS(iVar)));
205 SetHistoryOutputValue("MAX_ADJ_SPECIES_" + std::to_string(iVar), log10(adjspecies_solver->GetRes_Max(iVar)));
206 if (multiZone) {
207 SetHistoryOutputValue("BGS_ADJ_SPECIES_" + std::to_string(iVar), log10(adjspecies_solver->GetRes_BGS(iVar)));
208 }
209 }
210
211 SetHistoryOutputValue("LINSOL_ITER_SPECIES", adjspecies_solver->GetIterLinSolver());
212 SetHistoryOutputValue("LINSOL_RESIDUAL_SPECIES", log10(adjspecies_solver->GetResLinSolver()));
213 }
214
215 if (config->GetKind_Species_Model() == SPECIES_MODEL::FLAMELET) {
216 for (auto iCV=0u; iCV < config->GetNControlVars(); iCV++) {
217 const auto& cv_name = config->GetControllingVariableName(iCV);
218 SetHistoryOutputValue("RMS_ADJ_"+cv_name, log10(adjspecies_solver->GetRes_RMS(iCV)));
219 SetHistoryOutputValue("MAX_ADJ_"+cv_name, log10(adjspecies_solver->GetRes_Max(iCV)));
220 if (multiZone)
221 SetHistoryOutputValue("BGS_ADJ_" + cv_name, log10(adjspecies_solver->GetRes_BGS(iCV)));
222 }
223
224 for (unsigned short i_scalar = 0; i_scalar < config->GetNUserScalars(); i_scalar++) {
225 const auto& scalar_name = config->GetUserScalarName(i_scalar);
226 SetHistoryOutputValue("RMS_ADJ_" + scalar_name, log10(adjspecies_solver->GetRes_RMS(2 + i_scalar)));
227 SetHistoryOutputValue("MAX_ADJ_" + scalar_name, log10(adjspecies_solver->GetRes_Max(2 + i_scalar)));
228 if (multiZone) {
229 SetHistoryOutputValue("BGS_ADJ_" + scalar_name, log10(adjspecies_solver->GetRes_BGS(2 + i_scalar)));
230 }
231 }
232
233 SetHistoryOutputValue("LINSOL_ITER_FLAMELET", adjspecies_solver->GetIterLinSolver());
234 SetHistoryOutputValue("LINSOL_RESIDUAL_FLAMELET", log10(adjspecies_solver->GetResLinSolver()));
235 }
236
237 ComputeSimpleCustomOutputs(config);
238}

318void CAdjFlowOutput::LoadVolumeDataAdjScalar(const CConfig* config, const CSolver* const* solver,
319 const unsigned long iPoint) {
320 const auto Node_AdjTurb =
321 ((turb_model != TURB_MODEL::NONE) && !frozen_visc) ? solver[ADJTURB_SOL]->GetNodes() : nullptr;
322 const auto Node_AdjSpecies =
323 (config->GetKind_Species_Model() != SPECIES_MODEL::NONE) ? solver[ADJSPECIES_SOL]->GetNodes() : nullptr;
324
325 if (!frozen_visc) {
326 switch (TurbModelFamily(turb_model)) {
327 case TURB_FAMILY::SA:
328 SetVolumeOutputValue("ADJ_NU_TILDE", iPoint, Node_AdjTurb->GetSolution(iPoint, 0));
329 SetVolumeOutputValue("RES_ADJ_NU_TILDE", iPoint,
330 Node_AdjTurb->GetSolution(iPoint, 0) - Node_AdjTurb->GetSolution_Old(iPoint, 0));
331 break;
332 case TURB_FAMILY::KW:
333 SetVolumeOutputValue("ADJ_TKE", iPoint, Node_AdjTurb->GetSolution(iPoint, 0));
334 SetVolumeOutputValue("ADJ_DISSIPATION", iPoint, Node_AdjTurb->GetSolution(iPoint, 1));
335 SetVolumeOutputValue("RES_ADJ_TKE", iPoint,
336 Node_AdjTurb->GetSolution(iPoint, 0) - Node_AdjTurb->GetSolution_Old(iPoint, 0));
337 SetVolumeOutputValue("RES_ADJ_DISSIPATION", iPoint,
338 Node_AdjTurb->GetSolution(iPoint, 1) - Node_AdjTurb->GetSolution_Old(iPoint, 1));
339 break;
340 case TURB_FAMILY::NONE:
341 break;
342 }
343 }
344
345 if (config->GetKind_Species_Model() == SPECIES_MODEL::SPECIES_TRANSPORT) {
346 for (unsigned short iVar = 0; iVar < config->GetnSpecies(); iVar++) {
347 SetVolumeOutputValue("ADJ_SPECIES_" + std::to_string(iVar), iPoint, Node_AdjSpecies->GetSolution(iPoint, iVar));
348 SetVolumeOutputValue("RES_ADJ_SPECIES_" + std::to_string(iVar), iPoint,
349 Node_AdjSpecies->GetSolution(iPoint, iVar) - Node_AdjSpecies->GetSolution_Old(iPoint, iVar));
350 }
351 }
352
353 if (config->GetKind_Species_Model() == SPECIES_MODEL::FLAMELET) {
354
355 for (auto iCV=0u; iCV < config->GetNControlVars(); iCV++) {
356 const auto& cv_name = config->GetControllingVariableName(iCV);
357 SetVolumeOutputValue("ADJ_" + cv_name, iPoint, Node_AdjSpecies->GetSolution(iPoint, iCV));
358 SetVolumeOutputValue("RES_ADJ_" + cv_name, iPoint, Node_AdjSpecies->GetSolution(iPoint, iCV) - Node_AdjSpecies->GetSolution_Old(iPoint, iCV));
359 }
360 for (unsigned short i_scalar = 0; i_scalar < config->GetNUserScalars(); i_scalar++) {
361 const auto& scalar_name = config->GetUserScalarName(i_scalar);
362 SetVolumeOutputValue("ADJ_" + scalar_name, iPoint, Node_AdjSpecies->GetSolution(iPoint, config->GetNControlVars() + i_scalar));
363 SetVolumeOutputValue("RES_ADJ_" + scalar_name, iPoint, Node_AdjSpecies->GetSolution(iPoint, config->GetNControlVars() + i_scalar) - Node_AdjSpecies->GetSolution_Old(iPoint, config->GetNControlVars() + i_scalar));
364 }
365 }
