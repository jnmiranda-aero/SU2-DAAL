37CDeformationDriver::CDeformationDriver(char* confFile, SU2_Comm MPICommunicator)
38 : CDriverBase(confFile, 1, MPICommunicator) {
39 /*--- Preprocessing of the config files. ---*/
40
41 PreprocessInput();
42
43 /*--- Set up a timer for performance benchmarking. ---*/
44
45 StartTime = SU2_MPI::Wtime();
46
47 /*--- Preprocessing of the geometry for all zones. ---*/
48
49 InitializeGeometry();
50
51 /*--- Preprocessing of the output for all zones. ---*/
52
53 PreprocessOutput();
54
55 if (driver_config->GetDeform_Mesh()) {
56 /*--- Preprocessing of the mesh solver for all zones. ---*/
57
58 InitializeSolver();
59
60 /*--- Preprocessing of the mesh solver for all zones. ---*/
61
62 InitializeNumerics();
63 }
64
65 /*--- Preprocessing time is reported now, but not included in the next compute portion. ---*/
66
67 StopTime = SU2_MPI::Wtime();
68
69 /*--- Compute the total time for performance benchmarking. ---*/
70
71 UsedTime = StopTime - StartTime;
72 UsedTimePreproc = UsedTime;
73 UsedTimeCompute = 0.0;
74}

76void CDeformationDriver::PreprocessInput() {
77 /*--- Initialize a char to store the zone filename. ---*/
78
79 char zone_file_name[MAX_STRING_SIZE];
80
81 /*--- Initialize the configuration of the driver. ---*/
82
83 driver_config = new CConfig(config_file_name, SU2_COMPONENT::SU2_DEF);
84
85 nZone = driver_config->GetnZone();
86
87 /*--- Initialize containers. --- */
88
89 InitializeContainers();
90
91 /*--- Loop over all zones to initialize the various classes. In most
92 cases, nZone is equal to one. This represents the solution of a partial
93 differential equation on a single block, unstructured mesh. ---*/
94
95 for (iZone = 0; iZone < nZone; iZone++) {
96 /*--- Definition of the configuration option class for all zones. In this
97 constructor, the input configuration file is parsed and all options are
98 read and stored. ---*/
99
100 if (driver_config->GetnConfigFiles() > 0) {
101 strcpy(zone_file_name, driver_config->GetConfigFilename(iZone).c_str());
102 config_container[iZone] = new CConfig(driver_config, zone_file_name, SU2_COMPONENT::SU2_DEF, iZone, nZone, true);
103 } else {
104 config_container[iZone] =
105 new CConfig(driver_config, config_file_name, SU2_COMPONENT::SU2_DEF, iZone, nZone, true);
106 }
107
108 config_container[iZone]->SetMPICommunicator(SU2_MPI::GetComm());
109 }
110
111 /*--- Set the multi-zone part of the problem. ---*/
112
113 if (driver_config->GetMultizone_Problem()) {
114 for (iZone = 0; iZone < nZone; iZone++) {
115 /*--- Set the interface markers for multi-zone. ---*/
116
117 config_container[iZone]->SetMultizone(driver_config, config_container);
118 }
119 }
120
121 /*--- Keep a reference to the main (ZONE 0) config. ---*/
122
123 main_config = config_container[ZONE_0];
124}

126void CDeformationDriver::InitializeGeometry() {
127 for (iZone = 0; iZone < nZone; iZone++) {
128 /*--- Definition of the geometry class to store the primal grid in the partitioning process. ---*/
129
130 CGeometry* geometry_aux = nullptr;
131
132 /*--- All ranks process the grid and call ParMETIS for partitioning. ---*/
133
134 geometry_aux = new CPhysicalGeometry(config_container[iZone], iZone, nZone);
135
136 /*--- Color the initial grid and set the send-receive domains (ParMETIS). ---*/
137
138 geometry_aux->SetColorGrid_Parallel(config_container[iZone]);
139
140 /*--- Build the grid data structures using the ParMETIS coloring. ---*/
141
142 unsigned short nInst_Zone = nInst[iZone];
143 unsigned short nMesh = 1;
144
145 geometry_container[iZone] = new CGeometry**[nInst_Zone]();
146 geometry_container[iZone][INST_0] = new CGeometry*[nMesh]();
147 geometry_container[iZone][INST_0][MESH_0] = new CPhysicalGeometry(geometry_aux, config_container[iZone]);
148
149 /*--- Deallocate the memory of geometry_aux. ---*/
150
151 delete geometry_aux;
152
153 /*--- Add the Send/Receive boundaries. ---*/
154
155 geometry_container[iZone][INST_0][MESH_0]->SetSendReceive(config_container[iZone]);
156
157 /*--- Add the Send/Receive boundaries. ---*/
158
159 geometry_container[iZone][INST_0][MESH_0]->SetBoundaries(config_container[iZone]);
160
161 /*--- Computational grid preprocessing. ---*/
162
163 if (rank == MASTER_NODE)
164 cout << "\n----------------------- Preprocessing computations ----------------------" << endl;
165
166 /*--- Compute elements surrounding points, points surrounding points. ---*/
167
168 if (rank == MASTER_NODE) cout << "Setting local point connectivity." << endl;
169 geometry_container[iZone][INST_0][MESH_0]->SetPoint_Connectivity();
170
171 /*--- Check the orientation before computing geometrical quantities. ---*/
172
173 geometry_container[iZone][INST_0][MESH_0]->SetBoundVolume();
174 if (config_container[iZone]->GetReorientElements()) {
175 if (rank == MASTER_NODE) cout << "Checking the numerical grid orientation of the interior elements." << endl;
176 geometry_container[iZone][INST_0][MESH_0]->Check_IntElem_Orientation(config_container[iZone]);
177 geometry_container[iZone][INST_0][MESH_0]->Check_BoundElem_Orientation(config_container[iZone]);
178 }
179
180 /*--- Create the edge structure. ---*/
181
182 if (rank == MASTER_NODE) cout << "Identify edges and vertices." << endl;
183 geometry_container[iZone][INST_0][MESH_0]->SetEdges();
184 geometry_container[iZone][INST_0][MESH_0]->SetVertex(config_container[iZone]);
185
186 if (config_container[iZone]->GetDesign_Variable(0) != NO_DEFORMATION) {
187 /*--- Create the dual control volume structures. ---*/
188
189 if (rank == MASTER_NODE) cout << "Setting the bound control volume structure." << endl;
190 geometry_container[iZone][INST_0][MESH_0]->SetControlVolume(config_container[iZone], ALLOCATE);
191 geometry_container[iZone][INST_0][MESH_0]->SetBoundControlVolume(config_container[iZone], ALLOCATE);
192 }
193
194 /*--- Create the point-to-point MPI communication structures. ---*/
195
196 geometry_container[iZone][INST_0][MESH_0]->PreprocessP2PComms(geometry_container[iZone][INST_0][MESH_0],
197 config_container[iZone]);
198 }
199
200 /*--- Get the number of dimensions. ---*/
201
202 nDim = geometry_container[ZONE_0][INST_0][MESH_0]->GetnDim();
203
204 /*--- Keep a reference to the main (ZONE_0, INST_0, MESH_0) geometry. ---*/
205
206 main_geometry = geometry_container[ZONE_0][INST_0][MESH_0];
207}

290void CDeformationDriver::DeformMesh() {
291 if (!driver_config->GetDeform_Mesh()) return DeformLegacy();
292
293 for (iZone = 0; iZone < nZone; iZone++) {
294 /*--- Set the stiffness of each element mesh into the mesh numerics. ---*/
295
296 solver_container[iZone][INST_0][MESH_0][MESH_SOL]->SetMesh_Stiffness(
297 numerics_container[iZone][INST_0][MESH_0][MESH_SOL], config_container[iZone]);
298
299 /*--- Deform the volume grid around the new boundary locations. ---*/
300 /*--- Force the number of levels to be 0 because in this driver we do not build MG levels. ---*/
301 const auto nMGLevels = config_container[iZone]->GetnMGLevels();
302 config_container[iZone]->SetMGLevels(0);
303 solver_container[iZone][INST_0][MESH_0][MESH_SOL]->DeformMesh(geometry_container[iZone][INST_0],
304 numerics_container[iZone][INST_0][MESH_0][MESH_SOL],
305 config_container[iZone]);
306 config_container[iZone]->SetMGLevels(nMGLevels);
307 }
308}

310void CDeformationDriver::DeformLegacy() {
311 for (iZone = 0; iZone < nZone; iZone++) {
312 if (config_container[iZone]->GetDesign_Variable(0) != NO_DEFORMATION) {
313 unsigned short nInst_Zone = nInst[iZone];
314
315 /*--- Definition of the Class for grid movement. ---*/
316
317 grid_movement[iZone] = new CVolumetricMovement*[nInst_Zone]();
318 grid_movement[iZone][INST_0] =
319 new CVolumetricMovement(geometry_container[iZone][INST_0][MESH_0], config_container[iZone]);
320
321 /*--- Save original coordinates to be reused in convexity checking procedure. ---*/
322
323 auto OriginalCoordinates = geometry_container[iZone][INST_0][MESH_0]->nodes->GetCoord();
324
325 /*--- First check for volumetric grid deformation/transformations. ---*/
326
327 if (config_container[iZone]->GetDesign_Variable(0) == SCALE_GRID) {
328 if (rank == MASTER_NODE)
329 cout << endl
330 << "--------------------- Volumetric grid scaling (ZONE " << iZone << ") ------------------" << endl;
331 grid_movement[iZone][INST_0]->SetVolume_Scaling(geometry_container[iZone][INST_0][MESH_0],
332 config_container[iZone], false);
333
334 } else if (config_container[iZone]->GetDesign_Variable(0) == TRANSLATE_GRID) {
335 if (rank == MASTER_NODE)
336 cout << endl
337 << "------------------- Volumetric grid translation (ZONE " << iZone << ") ----------------" << endl;
338 grid_movement[iZone][INST_0]->SetVolume_Translation(geometry_container[iZone][INST_0][MESH_0],
339 config_container[iZone], false);
340
341 } else if (config_container[iZone]->GetDesign_Variable(0) == ROTATE_GRID) {
342 if (rank == MASTER_NODE)
343 cout << endl
344 << "--------------------- Volumetric grid rotation (ZONE " << iZone << ") -----------------" << endl;
345 grid_movement[iZone][INST_0]->SetVolume_Rotation(geometry_container[iZone][INST_0][MESH_0],
346 config_container[iZone], false);
347
348 } else {
349 /*--- If no volume-type deformations are requested, then this is a
350 * surface-based deformation or FFD set up. ---*/
351
352 if (rank == MASTER_NODE)
353 cout << endl
354 << "--------------------- Surface grid deformation (ZONE " << iZone << ") -----------------" << endl;
355
356 /*--- Definition and initialization of the surface deformation class. ---*/
357
358 surface_movement[iZone] = new CSurfaceMovement();
359 haveSurfaceDeformation = true;
360
361 /*--- Copy coordinates to the surface structure. ---*/
362
363 surface_movement[iZone]->CopyBoundary(geometry_container[iZone][INST_0][MESH_0], config_container[iZone]);
364
365 /*--- Surface grid deformation. ---*/
366
367 if (rank == MASTER_NODE) cout << "Performing the deformation of the surface grid." << endl;
368 auto TotalDeformation = surface_movement[iZone]->SetSurface_Deformation(
369 geometry_container[iZone][INST_0][MESH_0], config_container[iZone]);
370
371 if (config_container[iZone]->GetDesign_Variable(0) != FFD_SETTING) {
372 if (rank == MASTER_NODE)
373 cout << endl
374 << "------------------- Volumetric grid deformation (ZONE " << iZone << ") ----------------" << endl;
375
376 if (rank == MASTER_NODE) cout << "Performing the deformation of the volumetric grid." << endl;
377 grid_movement[iZone][INST_0]->SetVolume_Deformation(geometry_container[iZone][INST_0][MESH_0],
378 config_container[iZone], false);
379
380 /*--- Get parameters for convexity check. ---*/
381 bool ConvexityCheck;
382 unsigned short ConvexityCheck_MaxIter, ConvexityCheck_MaxDepth;
383
384 tie(ConvexityCheck, ConvexityCheck_MaxIter, ConvexityCheck_MaxDepth) =
385 config_container[iZone]->GetConvexityCheck();
386
387 /*--- Recursively change deformations if there are non-convex elements. ---*/
388
389 if (ConvexityCheck && geometry_container[iZone][INST_0][MESH_0]->GetnNonconvexElements() > 0) {
390 if (rank == MASTER_NODE) {
391 cout << "Non-convex elements present after deformation." << endl;
392 cout << "Recursively lowering deformation magnitude." << endl;
393 }
394
395 /*--- Load initial deformation values. ---*/
396
397 auto InitialDeformation = TotalDeformation;
398
399 unsigned short ConvexityCheckIter, RecursionDepth = 0;
400 su2double DeformationFactor = 1.0, DeformationDifference = 1.0;
401 for (ConvexityCheckIter = 1; ConvexityCheckIter <= ConvexityCheck_MaxIter; ConvexityCheckIter++) {
402 /*--- Recursively change deformation magnitude (decrease for non-convex elements, increase otherwise).
403 * ---*/
404
405 DeformationDifference /= 2.0;
406
407 if (geometry_container[iZone][INST_0][MESH_0]->GetnNonconvexElements() > 0) {
408 DeformationFactor -= DeformationDifference;
409 } else {
410 RecursionDepth += 1;
411
412 if (RecursionDepth == ConvexityCheck_MaxDepth) {
413 if (rank == MASTER_NODE) {
414 cout << "Maximum recursion depth reached." << endl;
415 cout << "Remaining amount of original deformation: ";
416 cout << DeformationFactor * 100.0 << " percent. " << endl;
417 }
418 break;
419 }
420
421 DeformationFactor += DeformationDifference;
422 }
423
424 /*--- Load mesh to start every iteration with an undeformed grid. ---*/
425
426 for (auto iPoint = 0ul; iPoint < OriginalCoordinates.rows(); iPoint++) {
427 for (auto iDim = 0ul; iDim < OriginalCoordinates.cols(); iDim++) {
428 geometry_container[iZone][INST_0][MESH_0]->nodes->SetCoord(iPoint, iDim,
429 OriginalCoordinates(iPoint, iDim));
430 }
431 }
432
433 /*--- Set deformation magnitude as percentage of initial deformation. ---*/
434
435 for (auto iDV = 0u; iDV < driver_config->GetnDV(); iDV++) {
436 for (auto iDV_Value = 0u; iDV_Value < driver_config->GetnDV_Value(iDV); iDV_Value++) {
437 config_container[iZone]->SetDV_Value(iDV, iDV_Value,
438 InitialDeformation[iDV][iDV_Value] * DeformationFactor);
439 }
440 }
441
442 /*--- Surface grid deformation. ---*/
443
444 if (rank == MASTER_NODE) cout << "Performing the deformation of the surface grid." << endl;
445
446 TotalDeformation = surface_movement[iZone]->SetSurface_Deformation(
447 geometry_container[iZone][INST_0][MESH_0], config_container[iZone]);
448
449 if (rank == MASTER_NODE)
450 cout << endl
451 << "------------------- Volumetric grid deformation (ZONE " << iZone << ") ----------------"
452 << endl;
453
454 if (rank == MASTER_NODE) cout << "Performing the deformation of the volumetric grid." << endl;
455 grid_movement[iZone][INST_0]->SetVolume_Deformation(geometry_container[iZone][INST_0][MESH_0],
456 config_container[iZone], false);
457
458 if (rank == MASTER_NODE) {
459 cout << "Number of non-convex elements for iteration " << ConvexityCheckIter << ": ";
460 cout << geometry_container[iZone][INST_0][MESH_0]->GetnNonconvexElements() << endl;
461 cout << "Remaining amount of original deformation: ";
462 cout << DeformationFactor * 100.0 << " percent. " << endl;
463 }
464 }
465 }
466 }
467 }
468 }
469 }

472void CDeformationDriver::OutputFiles() {
473 /*--- Output deformed grid for visualization, if requested (surface and volumetric), in parallel
474 requires to move all the data to the master node. ---*/
475
476 if (rank == MASTER_NODE)
477 cout << endl << "----------------------- Write deformed grid files -----------------------" << endl;
478
479 for (iZone = 0; iZone < nZone; iZone++) {
480 /*--- Compute Mesh Quality if requested. Necessary geometry preprocessing re-done beforehand. ---*/
481
482 if (config_container[iZone]->GetWrt_MeshQuality() && !driver_config->GetStructuralProblem()) {
483 if (rank == MASTER_NODE) cout << "Recompute geometry properties necessary to evaluate mesh quality statistics.\n";
484
485 geometry_container[iZone][INST_0][MESH_0]->SetPoint_Connectivity();
486 geometry_container[iZone][INST_0][MESH_0]->SetBoundVolume();
487 geometry_container[iZone][INST_0][MESH_0]->SetEdges();
488 geometry_container[iZone][INST_0][MESH_0]->SetVertex(config_container[iZone]);
489 geometry_container[iZone][INST_0][MESH_0]->SetControlVolume(config_container[iZone], ALLOCATE);
490 geometry_container[iZone][INST_0][MESH_0]->SetBoundControlVolume(config_container[iZone], ALLOCATE);
491
492 if (rank == MASTER_NODE) cout << "Computing mesh quality statistics for the dual control volumes.\n";
493 geometry_container[iZone][INST_0][MESH_0]->ComputeMeshQualityStatistics(config_container[iZone]);
494 }
495
496 /*--- Load the data. --- */
497
498 output_container[iZone]->LoadData(geometry_container[iZone][INST_0][MESH_0], config_container[iZone], nullptr);
499
500 output_container[iZone]->WriteToFile(config_container[iZone], geometry_container[iZone][INST_0][MESH_0],
501 OUTPUT_TYPE::MESH, driver_config->GetMesh_Out_FileName());
502
503 /*--- Set the file names for the visualization files. ---*/
504
505 output_container[iZone]->SetVolumeFilename("volume_deformed");
506 output_container[iZone]->SetSurfaceFilename("surface_deformed");
507
508 for (unsigned short iFile = 0; iFile < config_container[iZone]->GetnVolumeOutputFiles(); iFile++) {
509 auto FileFormat = config_container[iZone]->GetVolumeOutputFiles();
510 if (FileFormat[iFile] != OUTPUT_TYPE::RESTART_ASCII && FileFormat[iFile] != OUTPUT_TYPE::RESTART_BINARY &&
511 FileFormat[iFile] != OUTPUT_TYPE::CSV)
512 output_container[iZone]->WriteToFile(config_container[iZone], geometry_container[iZone][INST_0][MESH_0],
513 FileFormat[iFile]);
514 }
515 }
516
517 if (!driver_config->GetDeform_Mesh()) {
518 if ((config_container[ZONE_0]->GetDesign_Variable(0) != NO_DEFORMATION) &&
519 (config_container[ZONE_0]->GetDesign_Variable(0) != SCALE_GRID) &&
520 (config_container[ZONE_0]->GetDesign_Variable(0) != TRANSLATE_GRID) &&
521 (config_container[ZONE_0]->GetDesign_Variable(0) != ROTATE_GRID)) {
522 /*--- Write the free form deformation boxes after deformation if defined. ---*/
523 if (!haveSurfaceDeformation) {
524 if (rank == MASTER_NODE) cout << "No FFD information available." << endl;
525 } else {
526 if (rank == MASTER_NODE) cout << "Adding any FFD information to the SU2 file." << endl;
527
528 surface_movement[ZONE_0]->WriteFFDInfo(surface_movement, geometry_container, config_container);
529 }
530 }
531 }

534void CDeformationDriver::Finalize() {
535 if (rank == MASTER_NODE) cout << "\n------------------------- Finalize Solver -------------------------" << endl;
536
537 if (driver_config->GetDeform_Mesh()) {
538 for (iZone = 0; iZone < nZone; iZone++) {
539 if (numerics_container[iZone] != nullptr) {
540 for (int thread = 0; thread < omp_get_max_threads(); thread++) {
541 const int iTerm = FEA_TERM + thread * MAX_TERMS;
542 delete numerics_container[iZone][INST_0][MESH_0][MESH_SOL][iTerm];
543 }
544 delete[] numerics_container[iZone][INST_0][MESH_0][MESH_SOL];
545 delete[] numerics_container[iZone][INST_0][MESH_0];
546 delete[] numerics_container[iZone][INST_0];
547 }
548 /*--- The remaining levels in the container are deleted in CommonFinalize ---*/
549 }
550
551 for (iZone = 0; iZone < nZone; iZone++) {
552 if (solver_container[iZone] != nullptr) {
553 delete solver_container[iZone][INST_0][MESH_0][MESH_SOL];
554 delete[] solver_container[iZone][INST_0][MESH_0];
555 delete[] solver_container[iZone][INST_0];
556 }
557 /*--- The remaining levels in the container are deleted in CommonFinalize ---*/
558 }
559 }
560
561 CommonFinalize();
562
563 /*--- Exit the solver cleanly. ---*/
564
565 if (rank == MASTER_NODE)
566 cout << "\n------------------------- Exit Success (SU2_DEF) ------------------------" << endl << endl;
