30 /* Function, which determines the colors for the vertices of the given graph. */
31 void CGraphColoringStructure::GraphVertexColoring(CConfig* config, const vector<unsigned long>& nVerticesPerRank,
32  const vector<vector<unsigned long> >& entriesVertices,
33  int& nGlobalColors, vector<int>& colorLocalVertices) {
34  /* Determine the number of ranks and the current rank. */
35  int nRank = 1;
36  int myRank = 0;
37 
38 #ifdef HAVE_MPI
39  SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &myRank);
40  SU2_MPI::Comm_size(SU2_MPI::GetComm(), &nRank);
41 #endif
42 
43  /*--- Determine the algorithm to use for the graph coloring. ---*/
44  switch (config->GetKind_Matrix_Coloring()) {
45  case GREEDY_COLORING: {
46  /* Greedy algorithm, which is implemented sequentially.
47  Make a distinction between the master rank and the other ranks. */
48  if (myRank == 0) {
49  /*--------------------------------------------------------------------*/
50  /* Master node, which does all the work. */
51  /* Step 1: Create the global vector for the graph by gathering all the*/
52  /* data from the other ranks. This is done, because the greedy*/
53  /* algorithm used here is a sequential algorithm. Moreover the*/
54  /* algorithm requires the neighbors of neighbors, which would */
55  /* require a rather cumbersome communication step anyway. */
56  /**************************************************************************/
57 
58  /* Define the global vector and copy my data in it. */
59  vector<vector<unsigned long> > entriesVert(nVerticesPerRank[nRank], vector<unsigned long>(0));
60 
61  for (unsigned long i = nVerticesPerRank[0]; i < nVerticesPerRank[1]; ++i) entriesVert[i] = entriesVertices[i];
62 
63  /* Receive the data from the other ranks. Only needed in parallel mode. */
64 #ifdef HAVE_MPI
65  for (int rank = 1; rank < nRank; ++rank) {
66  /* Determine the size of the message to be received. */
67  SU2_MPI::Status status;
68  SU2_MPI::Probe(rank, rank, SU2_MPI::GetComm(), &status);
69 
70  int sizeMess;
71  SU2_MPI::Get_count(&status, MPI_UNSIGNED_LONG, &sizeMess);
72 
73  /* Allocate the memory for the receive buffer and receive the message. */
74  vector<unsigned long> recvBuf(sizeMess);
75  SU2_MPI::Recv(recvBuf.data(), sizeMess, MPI_UNSIGNED_LONG, rank, rank, SU2_MPI::GetComm(), &status);
76 
77  /* Store the data just received in the global vector for the graph. */
78  unsigned long ii = 0;
79  for (unsigned long i = nVerticesPerRank[rank]; i < nVerticesPerRank[rank + 1]; ++i) {
80  const unsigned long nEntries = recvBuf[ii++];
81 
82  entriesVert[i].resize(nEntries);
83  for (unsigned long j = 0; j < nEntries; ++j, ++ii) entriesVert[i][j] = recvBuf[ii];
84  }
85  }
86 #endif
87 
88  /**********************************************************************/
89  /* Step 2: The greedy algorithm to determine the vertex colors. */
90  /**********************************************************************/
91 
92  /* Allocate the memory to store the color of the vertices. */
93  vector<int> colorVertices(nVerticesPerRank[nRank]);
94 
95  /* Allocate and reserve the memory for vectors that are used
96  in the greedy algorithm. */
97  vector<unsigned long> flagColorStored(nVerticesPerRank[nRank], nVerticesPerRank[nRank]);
98  vector<int> colorNeighbors;
99  colorNeighbors.reserve(1000);
100 
101  /* Loop over the vertices of the graph. */
102  for (unsigned long i = 0; i < nVerticesPerRank[nRank]; ++i) {
103  /* Make sure that colorNeighbors is empty. */
104  colorNeighbors.resize(0);
105 
106  /* Loop over the entries of this vertex. */
107  for (unsigned long j = 0; j < entriesVert[i].size(); ++j) {
108  const unsigned long jj = entriesVert[i][j];
109 
110  /* Add the color of jj if jj is less than i and if its color
111  is not stored yet. */
112  if (jj < i) {
113  const int cJJ = colorVertices[jj];
114  if (flagColorStored[cJJ] != i) {
115  flagColorStored[cJJ] = i;
116  colorNeighbors.push_back(cJJ);
117  }
118  }
119 
120  /* Loop over the entries of vertex jj. */
121  for (unsigned long k = 0; k < entriesVert[jj].size(); ++k) {
122  const unsigned long kk = entriesVert[jj][k];
123 
124  /* Add the color of kk if kk is less than i and if its color
125  is not stored yet. */
126  if (kk < i) {
127  const int cKK = colorVertices[kk];
128  if (flagColorStored[cKK] != i) {
129  flagColorStored[cKK] = i;
130  colorNeighbors.push_back(cKK);
131  }
132  }
133  }
134  }
135 
136  /* Sort colorNeighbors in increasing order. */
137  sort(colorNeighbors.begin(), colorNeighbors.end());
138 
139  /* Determine the color of this vertex. */
140  int cVert;
141  for (cVert = 0; cVert < (int)colorNeighbors.size(); ++cVert) {
142  if (cVert != colorNeighbors[cVert]) break;
143  }
144 
145  colorVertices[i] = cVert;
146  }
147 
148  /* Check if the coloring is done correctly. */
149  flagColorStored.assign(nVerticesPerRank[nRank], nVerticesPerRank[nRank]);
150  for (unsigned long i = 0; i < entriesVert.size(); ++i) {
151  for (unsigned long j = 0; j < entriesVert[i].size(); ++j) {
152  const int cJJ = colorVertices[entriesVert[i][j]];
153  if (flagColorStored[cJJ] == i) {
154  cout << "In function GraphVertexColoring: Color " << cJJ
155  << " appears more than once in the stencil of vertex " << i << "." << endl;
156  exit(1);
157  }
158  flagColorStored[cJJ] = i;
159  }
160  }
161 
162  /**********************************************************************/
163  /* Step 3: Store the coloring information in the local vectors again. */
164  /**********************************************************************/
165 
166  /* Store the data of the root rank. */
167  colorLocalVertices.resize(nVerticesPerRank[1]);
168  memcpy(colorLocalVertices.data(), colorVertices.data(), nVerticesPerRank[1] * sizeof(int));
169 
170  /* Send the color of the vertices to the other ranks. Only in parallel
171  mode. Use blocking sends, because deadlock cannot occur. */
172 #ifdef HAVE_MPI
173  for (int rank = 1; rank < nRank; ++rank) {
174  int* sendBuf = colorVertices.data() + nVerticesPerRank[rank];
175  unsigned long sizeMess = nVerticesPerRank[rank + 1] - nVerticesPerRank[rank];
176  SU2_MPI::Send(sendBuf, sizeMess, MPI_INT, rank, rank + 1, SU2_MPI::GetComm());
177  }
178 #endif
179  }
180 #ifdef HAVE_MPI
181  else {
182  /* Not the master node. Communicate the data of my graph to the master
183  node. First copy the data in a buffer. */
184  vector<unsigned long> sendBuf;
185  for (unsigned long i = 0; i < entriesVertices.size(); ++i) {
186  sendBuf.push_back(entriesVertices[i].size());
187  sendBuf.insert(sendBuf.end(), entriesVertices[i].begin(), entriesVertices[i].end());
188  }
189 
190  /* Send the data to the master node. A blocking send can be used,
191  because there is no danger of deadlock here. */
192  SU2_MPI::Send(sendBuf.data(), sendBuf.size(), MPI_UNSIGNED_LONG, 0, myRank, SU2_MPI::GetComm());
193 
194  /* Receive the data for the colors of my locally owned DOFs. */
195  unsigned long nLocalVert = entriesVertices.size();
196  colorLocalVertices.resize(nLocalVert);
197 
198  SU2_MPI::Status status;
199  SU2_MPI::Recv(colorLocalVertices.data(), nLocalVert, MPI_INT, 0, myRank + 1, SU2_MPI::GetComm(), &status);
200  }
201 #endif
202  break;
203  }
204 
205  /*-----------------------------------------------------------------------*/
206 
207  case NATURAL_COLORING: {
208  /* Natural coloring, i.e. every DOF gets its own color. This is very
209  inefficient and should only be used for debugging. */
210  unsigned long nLocalVert = entriesVertices.size();
211  colorLocalVertices.resize(nLocalVert);
212 
213  for (unsigned long i = 0; i < nLocalVert; ++i) colorLocalVertices[i] = (int)(i + nVerticesPerRank[myRank]);
214 
215  break;
216  }
217  }
218 
219  /* Determine the total (global) number of colors. */
220  int nLocalColors = 0;
221  for (unsigned long i = 0; i < colorLocalVertices.size(); ++i) nLocalColors = max(nLocalColors, colorLocalVertices[i]);
222  ++nLocalColors;
223 
224 #ifdef HAVE_MPI
225  SU2_MPI::Allreduce(&nLocalColors, &nGlobalColors, 1, MPI_INT, MPI_MAX, SU2_MPI::GetComm());
226 #else
227  nGlobalColors = nLocalColors;
228 #endif
229 }
