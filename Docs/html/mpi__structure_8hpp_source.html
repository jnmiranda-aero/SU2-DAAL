313  static inline AMPI_Datatype convertDatatype(MPI_Datatype datatype) {
314  if (datatype == MPI_DOUBLE) {
315  return AMPI_ADOUBLE;
316  } else if (datatype == MPI_SHORT) {
317  return AMPI_SHORT;
318  } else if (datatype == MPI_UNSIGNED_SHORT) {
319  return AMPI_UNSIGNED_SHORT;
320  } else if (datatype == MPI_LONG) {
321  return AMPI_LONG;
322  } else if (datatype == MPI_UNSIGNED_LONG) {
323  return AMPI_UNSIGNED_LONG;
324  } else if (datatype == MPI_SHORT) {
325  return AMPI_SHORT;
326  } else if (datatype == MPI_CHAR) {
327  return AMPI_CHAR;
328  } else if (datatype == MPI_INT) {
329  return AMPI_INT;
330  } else {
331  Error("Conversion not implemented", CURRENT_FUNCTION);
332  return AMPI_DOUBLE;
333  }
334  }

336  static inline AMPI_Op convertOp(MPI_Op op) {
