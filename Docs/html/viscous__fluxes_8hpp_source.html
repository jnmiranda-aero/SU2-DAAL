35/*!
36 * \class CNoViscousFlux
37 * \ingroup ViscDiscr
38 * \brief Numerics classes that accept a compile-time decorator should use this
39 * class template as a "do-nothing" decorator and as a link to the interface when
40 * they are not being decorated.
41 * Compile-time decoration works by specifying the base class as a template parameter.
42 * Then the class being decorated should call a method of its base class to add some
43 * contribution to the flux/source and Jacobians just before writting the results to
44 * CSysVector and CSysMatrix. The mechanism can be used to chain any number of classes
45 * at compile-time, but its main purpose is to combine convective and viscous fluxes
46 * in the most (nearly) efficient way.
47 */
48template<size_t NDIM>
49class CNoViscousFlux : public CNumericsSIMD {
50protected:
51 static constexpr size_t nDim = NDIM;
52 static constexpr size_t nPrimVar = 0;
53
54 template<class... Ts>
55 CNoViscousFlux(Ts&...) {}
56
57 /*!
58 * \brief Empty method, real decorators should take as arguments whatever
59 * the decorated class can pass them to avoid expensive data accesses.
60 */
61 template<class... Ts>
62 void viscousTerms(Ts&...) const {}
63};

114 /*!
115 * \brief Add viscous contributions to flux and jacobians.
116 */
117 template<class PrimVarType, size_t nVar>
118 FORCEINLINE void viscousTerms(Int iEdge,
119 Int iPoint,
120 Int jPoint,
121 const PrimVarType& avgV,
122 const CPair<PrimVarType>& V,
123 const CVariable& solution_,
124 const VectorDbl<nDim>& vector_ij,
125 const CGeometry& geometry,
126 const CConfig& config,
127 Double area,
128 const VectorDbl<nDim>& unitNormal,
129 bool implicit,
130 VectorDbl<nVar>& flux,
131 MatrixDbl<nVar>& jac_i,
132 MatrixDbl<nVar>& jac_j) const {
133
134 static_assert(PrimVarType::nVar <= Derived::nPrimVar,"");
135
136 /*--- Pointer on which to call the "compile-time virtual" methods. ---*/
137
138 const auto derived = static_cast<const Derived*>(this);
139
140 const auto& solution = static_cast<const CNSVariable&>(solution_);
141 const auto& gradient = solution.GetGradient_Primitive();
142
143 /*--- Compute distance and handle zero without "ifs" by making it large. ---*/
144
145 auto dist2_ij = squaredNorm(vector_ij);
146 Double mask = dist2_ij < EPS*EPS;
147 dist2_ij += mask / (EPS*EPS);
148
149 /*--- Compute the corrected mean gradient. ---*/
150
151 auto avgGrad = averageGradient<nPrimVarGrad,nDim>(iPoint, jPoint, gradient);
152 if(correct) correctGradient(V, vector_ij, dist2_ij, avgGrad);
153
154 /*--- Stress and heat flux tensors. ---*/
155
156 auto tau = stressTensor(avgV.laminarVisc() + (uq? Double(0.0) : avgV.eddyVisc()), avgGrad);
157 if(useSA_QCR) addQCR(avgGrad, tau);
158 if(uq) {
159 Double turb_ke = 0.5*(gatherVariables(iPoint, turbVars->GetSolution()) +
160 gatherVariables(jPoint, turbVars->GetSolution()));
161 addPerturbedRSM(avgV, avgGrad, turb_ke, tau,
162 uq_eigval_comp, uq_permute, uq_delta_b, uq_urlx);
163 }
164
165 if(wallFun) addTauWall(iPoint, jPoint, solution.GetTau_Wall(), unitNormal, tau);
166
167 Double cond = derived->thermalConductivity(avgV);
168 VectorDbl<nDim> heatFlux;
169 for (size_t iDim = 0; iDim < nDim; ++iDim) {
170 heatFlux(iDim) = cond * avgGrad(0,iDim);
171 }
172
173 /*--- Projected flux. ---*/
174
175 auto viscFlux = viscousFlux<nVar>(avgV, tau, heatFlux, unitNormal);
176 for (size_t iVar = 0; iVar < nVar; ++iVar) {
177 viscFlux(iVar) *= area;
178 flux(iVar) -= viscFlux(iVar);
179 }
180
181 if (!implicit) return;
182
183 /*--- Flux Jacobians. ---*/
184
185 Double dist_ij = sqrt(dist2_ij);
186 auto dtau = stressTensorJacobian<nVar>(avgV, unitNormal, dist_ij);
187
188 /*--- Energy flux Jacobian. ---*/
189 auto dEdU = derived->energyJacobian(avgV, dtau, cond, area, dist_ij, iPoint, jPoint, solution);
190
191 /*--- Update momentum and energy terms ("symmetric" part). ---*/
192 for (size_t iDim = 0; iDim < nDim; ++iDim) {
193 for (size_t iVar = 0; iVar < nVar; ++iVar) {
194 jac_i(iDim+1,iVar) -= area * dtau(iDim,iVar);
195 jac_j(iDim+1,iVar) += area * dtau(iDim,iVar);
196 }
197 }
198 for (size_t iVar = 0; iVar < nVar; ++iVar) {
199 jac_i(nDim+1,iVar) += dEdU(iVar);
200 jac_j(nDim+1,iVar) -= dEdU(iVar);
201 }
202 /*--- "Non-symmetric" energy terms. ---*/
203 Double proj = dot<nDim>(&viscFlux(1), avgV.velocity());
204 Double halfOnRho = 0.5/avgV.density();
205 jac_i(nDim+1,0) += halfOnRho * proj;
206 jac_j(nDim+1,0) += halfOnRho * proj;
207 for (size_t iDim = 0; iDim < nDim; ++iDim) {
208 jac_i(nDim+1,iDim+1) -= halfOnRho * viscFlux(iDim+1);
209 jac_j(nDim+1,iDim+1) -= halfOnRho * viscFlux(iDim+1);
210 }
211 }
212
213 /*!
214 * \overload Average primitives if not provided yet.
215 */
216 template<class PrimVarType, class... Ts>
217 FORCEINLINE void viscousTerms(Int iEdge,
218 Int iPoint,
219 Int jPoint,
220 const CPair<PrimVarType>& V,
221 Ts&... args) const {
222 PrimVarType avgV;
223 for (size_t iVar = 0; iVar < PrimVarType::nVar; ++iVar) {
224 avgV.all(iVar) = 0.5 * (V.i.all(iVar) + V.j.all(iVar));
225 }
226
227 /*--- Continue calculation. ---*/
228 viscousTerms(iEdge, iPoint, jPoint, avgV, V, args...);
229 }
230
231 /*!
232 * \overload Compute the i-j vector if not provided yet.
233 */
234 template<class PrimVarType, class... Ts>
235 FORCEINLINE void viscousTerms(Int iEdge,
236 Int iPoint,
237 Int jPoint,
238 const PrimVarType& avgV,
239 const CPair<PrimVarType>& V,
240 const CVariable& solution_,
241 const CGeometry& geometry,
242 Ts&... args) const {
243
244 const auto vector_ij = distanceVector<nDim>(iPoint, jPoint, geometry.nodes->GetCoord());
245
246 /*--- Continue calculation. ---*/
247 viscousTerms(iEdge, iPoint, jPoint, avgV, V, solution_, vector_ij, geometry, args...);
248 }

256template<size_t NDIM>
257class CCompressibleViscousFlux : public CCompressibleViscousFluxBase<NDIM, CCompressibleViscousFlux<NDIM> > {
258public:
259 static constexpr size_t nPrimVar = NDIM+7;
260 using Base = CCompressibleViscousFluxBase<NDIM, CCompressibleViscousFlux<NDIM> >;
261 using Base::gamma;
262 using Base::gasConst;
263 using Base::prandtlLam;
264 using Base::prandtlTurb;
265 using Base::cp;

267 /*!
268 * \brief Constructor, initialize constants and booleans.
269 */
270 template<class... Ts>
271 CCompressibleViscousFlux(Ts&... args) : Base(args...) {}
272
273 /*!
274 * \brief Compute the thermal conductivity.
275 */
276 template<class PrimitiveType>
277 FORCEINLINE Double thermalConductivity(const PrimitiveType& V) const {
278 return cp * (V.laminarVisc()/prandtlLam + V.eddyVisc()/prandtlTurb);
279 }
280
281 /*!
282 * \brief Compute Jacobian of the energy flux, except the part due to the work of viscous forces.
283 */
284 template<size_t nVar, size_t nDim, class PrimitiveType, class... Ts>
285 FORCEINLINE VectorDbl<nVar> energyJacobian(const PrimitiveType& V,
286 const MatrixDbl<nDim,nVar>& dtau,
287 Double thermalCond,
288 Double area,
289 Double dist_ij,
290 Ts&... args) const {
291 Double vel2 = 0.5 * squaredNorm<nDim>(V.velocity());
292 Double phi = (gamma-1) / V.density();
293 Double RdTdrho = phi*vel2 - V.pressure() / pow(V.density(),2);
294 Double condOnRd = thermalCond / (gasConst * dist_ij);
295 Double contraction = 0.0;
296 for (size_t iDim = 0; iDim < nDim; ++iDim) {
297 contraction += dtau(iDim,0) * V.velocity(iDim);
298 }
299 VectorDbl<nVar> dEdU;
300 dEdU(0) = area * (condOnRd * RdTdrho - contraction);
301 for (size_t iDim = 0; iDim < nDim; ++iDim) {
302 dEdU(iDim+1) = area * (dtau(iDim,0) - condOnRd*phi*V.velocity(iDim));
303 }
304 dEdU(nDim+1) = area * condOnRd * phi;
305
306 return dEdU;
307 }
308};

315template<size_t NDIM>
316class CGeneralCompressibleViscousFlux : public CCompressibleViscousFluxBase<NDIM, CGeneralCompressibleViscousFlux<NDIM> > {
317public:
318 static constexpr size_t nPrimVar = NDIM+9;
319 static constexpr size_t nSecVar = 4;
320 using Base = CCompressibleViscousFluxBase<NDIM, CGeneralCompressibleViscousFlux<NDIM> >;
321 using Base::prandtlTurb;
322
323 /*!
324 * \brief Constructor, initialize constants and booleans.
325 */
326 template<class... Ts>
327 CGeneralCompressibleViscousFlux(Ts&... args) : Base(args...) {}
328
329 /*!
330 * \brief Compute the thermal conductivity.
331 */
332 template<class PrimitiveType>
333 FORCEINLINE Double thermalConductivity(const PrimitiveType& V) const {
334 return V.thermalCond() + V.cp()*V.eddyVisc()/prandtlTurb;
335 }
336
337 /*!
338 * \brief Compute Jacobian of the energy flux, except the part due to the work of viscous forces.
339 */
340 template<size_t nVar, size_t nDim, class PrimitiveType, class VariableType>
341 FORCEINLINE VectorDbl<nVar> energyJacobian(const PrimitiveType& V,
342 const MatrixDbl<nDim,nVar>& dtau,
343 Double thermalCond,
344 Double area,
345 Double dist_ij,
346 Int iPoint,
347 Int jPoint,
348 const VariableType& solution) const {
349 Double vel2 = squaredNorm<nDim>(V.velocity());
350 Double contraction = 0.0;
351 for (size_t iDim = 0; iDim < nDim; ++iDim) {
352 contraction += dtau(iDim,0) * V.velocity(iDim);
353 }
354
355 auto secVar_i = gatherVariables<nSecVar>(iPoint, solution.GetSecondary());
356 auto secVar_j = gatherVariables<nSecVar>(jPoint, solution.GetSecondary());
357
358 Double dTdrho_e = 0.5 * (secVar_i(2) + secVar_j(2));
359 Double dTde_rho = 0.5 * (secVar_i(3) + secVar_j(3)) / V.density();
360
361 Double condOnDist = thermalCond / dist_ij;
362 Double dTdrho = dTdrho_e + dTde_rho*(vel2-V.enthalpy()+V.pressure()/V.density());
363
364 VectorDbl<nVar> dEdU;
365 dEdU(0) = area * (condOnDist * dTdrho - contraction);
366 for (size_t iDim = 0; iDim < nDim; ++iDim) {
367 dEdU(iDim+1) = area * (dtau(iDim,0) - condOnDist*dTde_rho*V.velocity(iDim));
368 }
369 dEdU(nDim+1) = area * condOnDist * dTde_rho;
370
371 return dEdU;
372 }
