90  /*!
91  * \brief Invert a square matrix.
92  * \param[in] M - Size.
93  * \param[in,out] mat - Matrix, and inverse on exit.
94  */
95  template <class Mat>
96  static void inverse(const int M, Mat& mat) {
97  using Scalar = typename Mat::Scalar;
98 
99  /*--- Copy the data from A into the augmented matrix and initialize mat with the identity. ---*/
100  Mat aug = mat;
101  mat = Scalar(0);
102  for (int j = 0; j < M; ++j) mat(j, j) = 1;
103 
104  /*--- Outer loop of the Gauss-Jordan elimination. ---*/
105  for (int j = 0; j < M; ++j) {
106  /*--- Find the pivot in the current column. ---*/
107  int jj = j;
108  Scalar valMax = fabs(aug(j, j));
109  for (int i = j + 1; i < M; ++i) {
110  Scalar val = fabs(aug(i, j));
111  if (val > valMax) {
112  jj = i;
113  valMax = val;
114  }
115  }
116 
117  /*--- Swap the rows j and jj, if needed. ---*/
118  if (jj > j) {
119  for (int k = j; k < M; ++k) std::swap(aug(j, k), aug(jj, k));
120  for (int k = 0; k < M; ++k) std::swap(mat(j, k), mat(jj, k));
121  }
122 
123  /*--- Performing row operations to form required identity
124  matrix out of the input matrix. ---*/
125  for (int i = 0; i < M; ++i) {
126  if (i == j) continue;
127  valMax = aug(i, j) / aug(j, j);
128  for (int k = j; k < M; ++k) aug(i, k) -= valMax * aug(j, k);
129  for (int k = 0; k < M; ++k) mat(i, k) -= valMax * mat(j, k);
130  }
131 
132  valMax = 1.0 / aug(j, j);
133  for (int k = j; k < M; ++k) aug(j, k) *= valMax;
134  for (int k = 0; k < M; ++k) mat(j, k) *= valMax;
135  }
136  }

139  * \brief tred2
140  * Author:
141  *
142  * Original FORTRAN77 version by Smith, Boyle, Dongarra, Garbow, Ikebe,
143  * Klema, Moler.
144  * C++ version by Aashwin Mishra and Jayant Mukhopadhaya.
145  *
146  * Reference:
147  *
148  * Martin, Reinsch, Wilkinson,
149  * TRED2,
150  * Numerische Mathematik,
151  * Volume 11, pages 181-195, 1968.
152  *
153  * James Wilkinson, Christian Reinsch,
154  * Handbook for Automatic Computation,
155  * Volume II, Linear Algebra, Part 2,
156  * Springer, 1971,
157  * ISBN: 0387054146,
158  * LC: QA251.W67.
159  *
160  * Brian Smith, James Boyle, Jack Dongarra, Burton Garbow,
161  * Yasuhiko Ikebe, Virginia Klema, Cleve Moler,
162  * Matrix Eigensystem Routines, EISPACK Guide,
163  * Lecture Notes in Computer Science, Volume 6,
164  * Springer Verlag, 1976,
165  * ISBN13: 978-3540075462,
166  * LC: QA193.M37
167  *
168  * \param[in,out] V: matrix that needs to be decomposed
169  * \param[in,out] d: holds eigenvalues
170  * \param[in,out] e: work vector
171  * \param[in] n: order of matrix V
172  */
173  template <class Mat, class Vec, class W>
174  static void tred2(Mat& V, Vec& d, W& e, int n) {
175  using Scalar = typename std::decay<decltype(e[0])>::type;
176 
177  int i, j, k;
178 
179  for (j = 0; j < n; j++) {
180  d[j] = V[n - 1][j];
181  }
182 
183  /* Householder reduction to tridiagonal form. */
184 
185  for (i = n - 1; i > 0; i--) {
186  /* Scale to avoid under/overflow. */
187 
188  Scalar scale = 0.0;
189  Scalar h = 0.0;
190  for (k = 0; k < i; k++) {
191  scale = scale + fabs(d[k]);
192  }
193  if (scale == 0.0) {
194  e[i] = d[i - 1];
195  for (j = 0; j < i; j++) {
196  d[j] = V[i - 1][j];
197  V[i][j] = 0.0;
198  V[j][i] = 0.0;
199  }
200  } else {
201  /* Generate Householder vector. */
202 
203  for (k = 0; k < i; k++) {
204  d[k] /= scale;
205  h += d[k] * d[k];
206  }
207  Scalar f = d[i - 1];
208  Scalar g = sqrt(h);
209  if (f > 0) {
210  g = -g;
211  }
212  e[i] = scale * g;
213  h = h - f * g;
214  d[i - 1] = f - g;
215  for (j = 0; j < i; j++) {
216  e[j] = 0.0;
217  }
218 
219  /* Apply similarity transformation to remaining columns. */
220 
221  for (j = 0; j < i; j++) {
222  f = d[j];
223  V[j][i] = f;
224  g = e[j] + V[j][j] * f;
225  for (k = j + 1; k <= i - 1; k++) {
226  g += V[k][j] * d[k];
227  e[k] += V[k][j] * f;
228  }
229  e[j] = g;
230  }
231  f = 0.0;
232  for (j = 0; j < i; j++) {
233  e[j] /= h;
234  f += e[j] * d[j];
235  }
236  Scalar hh = f / (h + h);
237  for (j = 0; j < i; j++) {
238  e[j] -= hh * d[j];
239  }
240  for (j = 0; j < i; j++) {
241  f = d[j];
242  g = e[j];
243  for (k = j; k <= i - 1; k++) {
244  V[k][j] -= (f * e[k] + g * d[k]);
245  }
246  d[j] = V[i - 1][j];
247  V[i][j] = 0.0;
248  }
249  }
250  d[i] = h;
251  }
252 
253  /* Accumulate transformations. */
254 
255  for (i = 0; i < n - 1; i++) {
256  V[n - 1][i] = V[i][i];
257  V[i][i] = 1.0;
258  Scalar h = d[i + 1];
259  if (h != 0.0) {
260  for (k = 0; k <= i; k++) {
261  d[k] = V[k][i + 1] / h;
262  }
263  for (j = 0; j <= i; j++) {
264  Scalar g = 0.0;
265  for (k = 0; k <= i; k++) {
266  g += V[k][i + 1] * V[k][j];
267  }
268  for (k = 0; k <= i; k++) {
269  V[k][j] -= g * d[k];
270  }
271  }
272  }
273  for (k = 0; k <= i; k++) {
274  V[k][i + 1] = 0.0;
275  }
276  }
277  for (j = 0; j < n; j++) {
278  d[j] = V[n - 1][j];
279  V[n - 1][j] = 0.0;
280  }
281  V[n - 1][n - 1] = 1.0;
282  e[0] = 0.0;
283  }

286  * \brief tql2
287  * Author:
288  *
289  * Original FORTRAN77 version by Smith, Boyle, Dongarra, Garbow, Ikebe,
290  * Klema, Moler.
291  * C++ version by Aashwin Mishra and Jayant Mukhopadhaya.
292  *
293  * Reference:
294  *
295  * Bowdler, Martin, Reinsch, Wilkinson,
296  * TQL2,
297  * Numerische Mathematik,
298  * Volume 11, pages 293-306, 1968.
299  *
300  * James Wilkinson, Christian Reinsch,
301  * Handbook for Automatic Computation,
302  * Volume II, Linear Algebra, Part 2,
303  * Springer, 1971,
304  * ISBN: 0387054146,
305  * LC: QA251.W67.
306  *
307  * Brian Smith, James Boyle, Jack Dongarra, Burton Garbow,
308  * Yasuhiko Ikebe, Virginia Klema, Cleve Moler,
309  * Matrix Eigensystem Routines, EISPACK Guide,
310  * Lecture Notes in Computer Science, Volume 6,
311  * Springer Verlag, 1976,
312  * ISBN13: 978-3540075462,
313  * LC: QA193.M37
314  *
315  * \param[in,out] V: matrix that will hold the eigenvectors
316  * \param[in,out] d: array that will hold the ordered eigenvalues
317  * \param[in,out] e: work vector
318  * \param[in] n: order of matrix V
319  */
320  template <class Mat, class Vec, class W>
321  static void tql2(Mat& V, Vec& d, W& e, int n) {
322  using Scalar = typename std::decay<decltype(e[0])>::type;
323 
324  int i, j, k, l;
325  for (i = 1; i < n; i++) {
326  e[i - 1] = e[i];
327  }
328  e[n - 1] = 0.0;
329 
330  Scalar f = 0.0;
331  Scalar tst1 = 0.0;
332  Scalar eps = pow(2.0, -52.0);
333  for (l = 0; l < n; l++) {
334  /* Find small subdiagonal element */
335 
336  tst1 = max(tst1, (fabs(d[l]) + fabs(e[l])));
337  int m = l;
338  while (m < n) {
339  if (fabs(e[m]) <= eps * tst1) {
340  break;
341  }
342  m++;
343  }
344 
345  /* If m == l, d[l] is an eigenvalue, */
346  /* otherwise, iterate. */
347 
348  if (m > l) {
349  int iter = 0;
350  do {
351  iter = iter + 1; /* (Could check iteration count here.) */
352 
353  /* Compute implicit shift */
354 
355  Scalar g = d[l];
356  Scalar p = (d[l + 1] - g) / (2.0 * e[l]);
357  Scalar r = sqrt(p * p + 1.0);
358  if (p < 0) {
359  r = -r;
360  }
361  d[l] = e[l] / (p + r);
362  d[l + 1] = e[l] * (p + r);
363  Scalar dl1 = d[l + 1];
364  Scalar h = g - d[l];
365  for (i = l + 2; i < n; i++) {
366  d[i] -= h;
367  }
368  f = f + h;
369 
370  /* Implicit QL transformation. */
371 
372  p = d[m];
373  Scalar c = 1.0;
374  Scalar c2 = c;
375  Scalar c3 = c;
376  Scalar el1 = e[l + 1];
377  Scalar s = 0.0;
378  Scalar s2 = 0.0;
379  for (i = m - 1; i >= l; i--) {
380  c3 = c2;
381  c2 = c;
382  s2 = s;
383  g = c * e[i];
384  h = c * p;
385  r = sqrt(p * p + e[i] * e[i]);
386  e[i + 1] = s * r;
387  s = e[i] / r;
388  c = p / r;
389  p = c * d[i] - s * g;
390  d[i + 1] = h + s * (c * g + s * d[i]);
391 
392  /* Accumulate transformation. */
393 
394  for (k = 0; k < n; k++) {
395  h = V[k][i + 1];
396  V[k][i + 1] = s * V[k][i] + c * h;
397  V[k][i] = c * V[k][i] - s * h;
398  }
399  }
400  p = -s * s2 * c3 * el1 * e[l] / dl1;
401  e[l] = s * p;
402  d[l] = c * p;
403 
404  /* Check for convergence. */
405 
406  } while (fabs(e[l]) > eps * tst1);
407  }
408  d[l] = d[l] + f;
409  e[l] = 0.0;
410  }
411 
412  /* Sort eigenvalues and corresponding vectors. */
413 
414  for (i = 0; i < n - 1; i++) {
415  k = i;
416  Scalar p = d[i];
417  for (j = i + 1; j < n; j++) {
418  if (d[j] < p) {
419  k = j;
420  p = d[j];
421  }
422  }
423  if (k != i) {
424  d[k] = d[i];
425  d[i] = p;
426  for (j = 0; j < n; j++) {
427  p = V[j][i];
428  V[j][i] = V[j][k];
429  V[j][k] = p;
430  }
431  }
432  }
433  }

472  /*!
473  * \brief Algorithm to solve a linear system with a tridiagonal matrix.
474  * \param[in] lower - lower diagonal
475  * \param[in] main - main diagonal
476  * \param[in,out] upper - upper diagonal (modified on exit)
477  * \param[in,out] rhs - right hand side on entry, solution on exit
478  * \note Same size for all vectors. Use row index for lower and upper vector (e.g. lower[0] does not matter).
479  */
480  template <class Vec, class Scalar = su2double>
481  static void tdma(const Vec& lower, const Vec& main, Vec& upper, Vec& rhs) {
482  const int N = main.size();
483 
484  upper[0] /= main[0];
485  rhs[0] /= main[0];
486 
487  for (int i = 1; i < N; i++) {
488  const Scalar denom = 1.0 / (main[i] - lower[i] * upper[i - 1]);
489  upper[i] *= denom;
490  rhs[i] = (rhs[i] - lower[i] * rhs[i - 1]) * denom;
491  }
492 
493  for (int i = N - 2; i >= 0; i--) rhs[i] -= upper[i] * rhs[i + 1];
494  }
