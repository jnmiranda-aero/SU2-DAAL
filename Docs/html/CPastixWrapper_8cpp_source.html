39 template <class ScalarType>
40 void CPastixWrapper<ScalarType>::Initialize(CGeometry* geometry, const CConfig* config) {
41  using namespace PaStiX;
42 
43  if (isinitialized) return; // only need to do this once
44 
45  unsigned long nVar = matrix.nVar, nPoint = matrix.nPoint, nPointDomain = matrix.nPointDomain;
46  const unsigned long *row_ptr = matrix.rowptr, *col_ind = matrix.colidx;
47 
48  unsigned long iPoint, offset = 0, nNonZero = row_ptr[nPointDomain];
49 
50  /*--- Allocate ---*/
51 
52  nCols = pastix_int_t(nPointDomain);
53  colptr.resize(nPointDomain + 1);
54  rowidx.clear();
55  rowidx.reserve(nNonZero);
56  values.resize(nNonZero * nVar * nVar);
57  loc2glb.resize(nPointDomain);
58  perm.resize(nPointDomain);
59  workvec.resize(nPointDomain * nVar);
60 
61  /*--- Set default parameter values ---*/
62 
63  pastix_int_t incomplete = iparm[IPARM_INCOMPLETE];
64 
65  iparm[IPARM_MODIFY_PARAMETER] = API_NO;
66  Run();
67 
68  /*--- Customize important parameters ---*/
69 
70  switch (verb) {
71  case 1:
72  iparm[IPARM_VERBOSE] = API_VERBOSE_NO;
73  break;
74  case 2:
75  iparm[IPARM_VERBOSE] = API_VERBOSE_YES;
76  break;
77  default:
78  iparm[IPARM_VERBOSE] = API_VERBOSE_NOT;
79  break;
80  }
81  iparm[IPARM_DOF_NBR] = pastix_int_t(nVar);
82  iparm[IPARM_MATRIX_VERIFICATION] = API_NO;
83  iparm[IPARM_FREE_CSCPASTIX] = API_CSC_FREE;
84  iparm[IPARM_CSCD_CORRECT] = API_NO;
85  iparm[IPARM_RHSD_CHECK] = API_NO;
86  iparm[IPARM_ORDERING] = API_ORDER_PTSCOTCH;
87  iparm[IPARM_INCOMPLETE] = incomplete;
88  iparm[IPARM_LEVEL_OF_FILL] = pastix_int_t(config->GetPastixFillLvl());
89  iparm[IPARM_THREAD_NBR] = omp_get_max_threads();
90 #if defined(HAVE_MPI) && defined(HAVE_OMP)
91  int comm_mode = MPI_THREAD_SINGLE;
92  MPI_Query_thread(&comm_mode);
93  if (comm_mode == MPI_THREAD_MULTIPLE)
94  iparm[IPARM_THREAD_COMM_MODE] = API_THREAD_MULTIPLE;
95  else
96  iparm[IPARM_THREAD_COMM_MODE] = API_THREAD_FUNNELED;
97 #endif
98 
99  /*--- Prepare sparsity structure ---*/
100 
101  /*--- We need it in global coordinates, i.e. shifted according to the position
102  of the current rank in the linear partitioning space, and "unpacked" halo part.
103  The latter forces us to re-sort the column indices of rows with halo points, which
104  in turn requires blocks to be swapped accordingly. Moreover we need "pointer" and
105  indices in Fortran-style numbering (start at 1), effectively the matrix is copied.
106  Here we prepare the pointer and index part, and map the required swaps. ---*/
107 
108  /*--- 1 - Determine position in the linear partitioning ---*/
109 
110 #ifdef HAVE_MPI
111  vector<unsigned long> domain_sizes(mpi_size);
112  MPI_Allgather(&nPointDomain, 1, MPI_UNSIGNED_LONG, domain_sizes.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
113  for (int i = 0; i < mpi_rank; ++i) offset += domain_sizes[i];
114 #endif
115 
116  iota(loc2glb.begin(), loc2glb.end(), offset + 1);
117 
118  /*--- 2 - Communicate global indices of halo points to then renumber
119  column indices from local to global when unpacking halos. ---*/
120 
121  vector<pastix_int_t> map(nPoint - nPointDomain, 0);
122 
123 #ifdef HAVE_MPI
124  for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
125  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
126  unsigned short MarkerS = iMarker, MarkerR = iMarker + 1;
127 
128  int sender = config->GetMarker_All_SendRecv(MarkerS) - 1;
129  int recver = abs(config->GetMarker_All_SendRecv(MarkerR)) - 1;
130 
131  unsigned long nVertexS = geometry->nVertex[MarkerS];
132  unsigned long nVertexR = geometry->nVertex[MarkerR];
133 
134  /*--- Allocate Send/Receive buffers ---*/
135  vector<unsigned long> Buffer_Recv(nVertexR), Buffer_Send(nVertexS);
136 
137  /*--- Prepare data to send ---*/
138  for (unsigned long iVertex = 0; iVertex < nVertexS; iVertex++)
139  Buffer_Send[iVertex] = geometry->vertex[MarkerS][iVertex]->GetNode() + offset;
140 
141  /*--- Send and Receive data ---*/
142  MPI_Sendrecv(Buffer_Send.data(), nVertexS, MPI_UNSIGNED_LONG, sender, 0, Buffer_Recv.data(), nVertexR,
143  MPI_UNSIGNED_LONG, recver, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
144 
145  /*--- Store received data---*/
146  for (unsigned long iVertex = 0; iVertex < nVertexR; iVertex++)
147  map[geometry->vertex[MarkerR][iVertex]->GetNode() - nPointDomain] = Buffer_Recv[iVertex];
148  }
149  }
150 #endif
151 
152  /*--- 3 - Copy, map the sparsity, and put it in Fortran numbering ---*/
153 
154  for (iPoint = 0; iPoint < nPointDomain; ++iPoint) {
155  colptr[iPoint] = pastix_int_t(row_ptr[iPoint] + 1);
156 
157  unsigned long begin = row_ptr[iPoint], end = row_ptr[iPoint + 1], j;
158 
159  /*--- If last point of row is halo ---*/
160  bool sort_required = (col_ind[end - 1] >= nPointDomain);
161 
162  if (sort_required) {
163  unsigned long nnz_row = end - begin;
164 
165  sort_rows.push_back(iPoint);
166  sort_order.push_back(vector<unsigned long>(nnz_row));
167 
168  /*--- Sort mapped indices ("first") and keep track of source ("second")
169  for when we later need to swap blocks for these rows. ---*/
170 
171  vector<pair<pastix_int_t, unsigned long> > aux(nnz_row);
172 
173  for (j = begin; j < end; ++j) {
174  if (col_ind[j] < nPointDomain)
175  aux[j - begin].first = pastix_int_t(offset + col_ind[j] + 1);
176  else
177  aux[j - begin].first = pastix_int_t(map[col_ind[j] - nPointDomain] + 1);
178  aux[j - begin].second = j;
179  }
180  sort(aux.begin(), aux.end());
181 
182  for (j = 0; j < nnz_row; ++j) {
183  rowidx.push_back(aux[j].first);
184  sort_order.back()[j] = aux[j].second;
185  }
186  } else {
187  /*--- These are all internal, no need to go through map. ---*/
188  for (j = begin; j < end; ++j) rowidx.push_back(pastix_int_t(offset + col_ind[j] + 1));
189  }
190  }
191  colptr[nPointDomain] = pastix_int_t(nNonZero + 1);
192 
193  if (rowidx.size() != nNonZero) SU2_MPI::Error("Error during preparation of PaStiX data", CURRENT_FUNCTION);
194 
195  /*--- 4 - Perform ordering, symbolic factorization, and analysis steps ---*/
196 
197  if (mpi_rank == MASTER_NODE && verb > 0) cout << endl;
198 
199  iparm[IPARM_START_TASK] = API_TASK_ORDERING;
200  iparm[IPARM_END_TASK] = API_TASK_ANALYSE;
201  Run();
202 
203  if (mpi_rank == MASTER_NODE && verb > 0)
204  cout << " +--------------------------------------------------------------------+" << endl;
205 
206  isinitialized = true;
207 }

209 template <class ScalarType>
210 void CPastixWrapper<ScalarType>::Factorize(CGeometry* geometry, const CConfig* config, unsigned short kind_fact) {
211  using namespace PaStiX;
212 
213  /*--- Detect a possible change of settings between direct and adjoint that requires a reset ---*/
214  if (isinitialized)
215  if ((kind_fact == PASTIX_ILU) != (iparm[IPARM_INCOMPLETE] == API_YES)) {
216  Clean();
217  isinitialized = false;
218  iter = 0;
219  }
220 
221  verb = config->GetPastixVerbLvl();
222  iparm[IPARM_INCOMPLETE] = (kind_fact == PASTIX_ILU);
223 
224  Initialize(geometry, config);
225 
226  /*--- Set some options that affect "compute" and could (one day) change during run ---*/
227 
228  switch (verb) {
229  case 1:
230  iparm[IPARM_VERBOSE] = API_VERBOSE_NO;
231  break;
232  case 2:
233  iparm[IPARM_VERBOSE] = API_VERBOSE_YES;
234  break;
235  default:
236  iparm[IPARM_VERBOSE] = API_VERBOSE_NOT;
237  break;
238  }
239 
240  if (kind_fact == PASTIX_LDLT || kind_fact == PASTIX_LDLT_P)
241  iparm[IPARM_TRANSPOSE_SOLVE] = API_NO; // symmetric so no need for slower transp. solve
242  else
243  iparm[IPARM_TRANSPOSE_SOLVE] = API_YES; // negated due to CSR to CSC copy
244 
245  /*--- Is factorizing needed on this iteration? ---*/
246 
247  bool factorize = false;
248  if (config->GetPastixFactFreq() != 0) factorize = (iter % config->GetPastixFactFreq() == 0);
249 
250  iter++;
251 
252  if (isfactorized && !factorize) return; // No
253 
254  /*--- Yes ---*/
255 
256  if (mpi_rank == MASTER_NODE && verb > 0) {
257  cout << endl;
258  cout << " +--------------------------------------------------------------------+" << endl;
259  cout << " + PaStiX : Parallel Sparse matriX package +" << endl;
260  cout << " +--------------------------------------------------------------------+" << endl;
261  }
262 
263  unsigned long i, j, k, iRow, begin, target, source, szBlk = matrix.nVar * matrix.nVar, nNonZero = values.size();
264 
265  /*--- Copy matrix values and swap blocks as required ---*/
266 
267  for (i = 0; i < nNonZero; ++i) values[i] = SU2_TYPE::GetValue(matrix.values[i]);
268 
269  for (i = 0; i < sort_rows.size(); ++i) {
270  iRow = sort_rows[i];
271  begin = matrix.rowptr[iRow];
272 
273  for (j = 0; j < sort_order[i].size(); ++j) {
274  target = (begin + j) * szBlk;
275  source = sort_order[i][j] * szBlk;
276 
277  for (k = 0; k < szBlk; ++k) values[target + k] = SU2_TYPE::GetValue(matrix.values[source + k]);
278  }
279  }
280 
281  /*--- Set factorization options ---*/
282 
283  switch (kind_fact) {
284  case PASTIX_LDLT:
285  case PASTIX_LDLT_P:
286  iparm[IPARM_SYM] = API_SYM_YES;
287  iparm[IPARM_FACTORIZATION] = API_FACT_LDLT;
288  break;
289  case PASTIX_LU:
290  case PASTIX_LU_P:
291  case PASTIX_ILU:
292  iparm[IPARM_SYM] = API_SYM_NO;
293  iparm[IPARM_FACTORIZATION] = API_FACT_LU;
294  break;
295  default:
296  SU2_MPI::Error("Unknown type of PaStiX factorization.", CURRENT_FUNCTION);
297  break;
298  }
299 
300  /*--- Compute factorization ---*/
301 
302  iparm[IPARM_START_TASK] = API_TASK_NUMFACT;
303  iparm[IPARM_END_TASK] = API_TASK_NUMFACT;
304  Run();
305 
306  if (mpi_rank == MASTER_NODE && verb > 0)
307  cout << " +--------------------------------------------------------------------+" << endl << endl;
308 
309  isfactorized = true;
310 }
