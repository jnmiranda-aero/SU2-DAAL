168  void AddError_Max(unsigned short val_var, su2double val_error, unsigned long val_point, const su2double* val_coord) {
169  if (val_error > Error_Max[val_var]) {
170  Error_Max[val_var] = val_error;
171  Error_Point_Max[val_var] = val_point;

173  }
