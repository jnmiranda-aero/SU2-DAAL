35CFluidFlamelet::CFluidFlamelet(CConfig* config, su2double value_pressure_operating) : CFluidModel() {
36 rank = SU2_MPI::GetRank();
37 Kind_DataDriven_Method = config->GetKind_DataDriven_Method();
38
39 /* -- number of auxiliary species transport equations, e.g. 1=CO, 2=NOx --- */
40 n_user_scalars = config->GetNUserScalars();
41 n_control_vars = config->GetNControlVars();
42 include_mixture_fraction = (n_control_vars == 3);
43 n_scalars = config->GetNScalars();
44
45 if (rank == MASTER_NODE) {
46 cout << "Number of scalars: " << n_scalars << endl;
47 cout << "Number of user scalars: " << n_user_scalars << endl;
48 cout << "Number of control variables: " << n_control_vars << endl;
49 }
50
51 scalars_vector.resize(n_scalars);
52
53 table_scalar_names.resize(n_scalars);
54 for (auto iCV = 0u; iCV < n_control_vars; iCV++) table_scalar_names[iCV] = config->GetControllingVariableName(iCV);
55
56 /*--- auxiliary species transport equations---*/
57 for (size_t i_aux = 0; i_aux < n_user_scalars; i_aux++) {
58 table_scalar_names[n_control_vars + i_aux] = config->GetUserScalarName(i_aux);
59 }
60
61 controlling_variable_names.resize(n_control_vars);
62 for (auto iCV = 0u; iCV < n_control_vars; iCV++)
63 controlling_variable_names[iCV] = config->GetControllingVariableName(iCV);
64
65 passive_specie_names.resize(n_user_scalars);
66 for (auto i_aux = 0u; i_aux < n_user_scalars; i_aux++) passive_specie_names[i_aux] = config->GetUserScalarName(i_aux);
67
68 switch (Kind_DataDriven_Method) {
69 case ENUM_DATADRIVEN_METHOD::LUT:
70 if (rank == MASTER_NODE) {
71 cout << "*****************************************" << endl;
72 cout << "*** initializing the lookup table ***" << endl;
73 cout << "*****************************************" << endl;
74 }
75 look_up_table = new CLookUpTable(config->GetDataDriven_FileNames()[0], table_scalar_names[I_PROGVAR],
76 table_scalar_names[I_ENTH]);
77 break;
78 default:
79 if (rank == MASTER_NODE) {
80 cout << "***********************************************" << endl;
81 cout << "*** initializing the multi-layer perceptron ***" << endl;
82 cout << "***********************************************" << endl;
83 }
84#ifdef USE_MLPCPP
85 lookup_mlp = new MLPToolbox::CLookUp_ANN(config->GetNDataDriven_Files(), config->GetDataDriven_FileNames());
86 if ((rank == MASTER_NODE)) lookup_mlp->DisplayNetworkInfo();
87#else
88 SU2_MPI::Error("SU2 was not compiled with MLPCpp enabled (-Denable-mlpcpp=true).", CURRENT_FUNCTION);
89#endif
90 break;
91 }
92
93 Pressure = value_pressure_operating;
94
95 PreprocessLookUp(config);
96}

116void CFluidFlamelet::SetTDState_T(su2double val_temperature, const su2double* val_scalars) {
117 for (auto iVar = 0u; iVar < n_scalars; iVar++) scalars_vector[iVar] = val_scalars[iVar];
118
119 /*--- Add all quantities and their names to the look up vectors. ---*/
120 EvaluateDataSet(scalars_vector, FLAMELET_LOOKUP_OPS::TD, val_vars_TD);
121
122 Temperature = val_vars_TD[LOOKUP_TD::TEMPERATURE];
123 Cp = val_vars_TD[LOOKUP_TD::HEATCAPACITY];
124 Mu = val_vars_TD[LOOKUP_TD::VISCOSITY];
125 Kt = val_vars_TD[LOOKUP_TD::CONDUCTIVITY];
126 mass_diffusivity = val_vars_TD[LOOKUP_TD::DIFFUSIONCOEFFICIENT];
127 switch (density_model) {
128 case INC_DENSITYMODEL::FLAMELET:
129 Density = val_vars_TD[LOOKUP_TD::MOLARWEIGHT];
130 molar_weight = Pressure / (Density * UNIVERSAL_GAS_CONSTANT * Temperature);
131 break;
132 case INC_DENSITYMODEL::VARIABLE:
133 molar_weight = val_vars_TD[LOOKUP_TD::MOLARWEIGHT];
134 Density = (molar_weight / 1000) * Pressure / (UNIVERSAL_GAS_CONSTANT * Temperature);
135 break;
136 default:
137 break;
138 }
139 /*--- Compute Cv from Cp and molar weight of the mixture (ideal gas). ---*/
140 Cv = Cp - UNIVERSAL_GAS_CONSTANT / molar_weight;
141}

143void CFluidFlamelet::PreprocessLookUp(CConfig* config) {
144 density_model = config->GetKind_DensityModel();
145 /*--- Thermodynamic state variables and names. ---*/
146 varnames_TD.resize(LOOKUP_TD::SIZE);
147 val_vars_TD.resize(LOOKUP_TD::SIZE);
148
149 /*--- The string in varnames_TD as it appears in the LUT file. ---*/
150 varnames_TD[LOOKUP_TD::TEMPERATURE] = "Temperature";
151 varnames_TD[LOOKUP_TD::HEATCAPACITY] = "Cp";
152 varnames_TD[LOOKUP_TD::VISCOSITY] = "ViscosityDyn";
153 varnames_TD[LOOKUP_TD::CONDUCTIVITY] = "Conductivity";
154 varnames_TD[LOOKUP_TD::DIFFUSIONCOEFFICIENT] = "DiffusionCoefficient";
155
156 /*--- In case of FLAMELET density model, the density is directly interpolated from the manifold.---*/
157 switch (density_model) {
158 case INC_DENSITYMODEL::FLAMELET:
159 varnames_TD[LOOKUP_TD::MOLARWEIGHT] = "Density";
160 break;
161 case INC_DENSITYMODEL::VARIABLE:
162 varnames_TD[LOOKUP_TD::MOLARWEIGHT] = "MolarWeightMix";
163 break;
164 default:
165 break;
166 }
167 /*--- Scalar source term variables and names. ---*/
168 size_t n_sources = n_control_vars + 2 * n_user_scalars;
169 varnames_Sources.resize(n_sources);
170 val_vars_Sources.resize(n_sources);
171 for (auto iCV = 0u; iCV < n_control_vars; iCV++)
172 varnames_Sources[iCV] = config->GetControllingVariableSourceName(iCV);
173 /*--- No source term for enthalpy ---*/
174
175 /*--- For the auxiliary equations, we use a positive (production) and a negative (consumption) term:
176 S_tot = S_PROD + S_CONS * Y ---*/
177
178 for (size_t i_aux = 0; i_aux < n_user_scalars; i_aux++) {
179 /*--- Order of the source terms: S_prod_1, S_cons_1, S_prod_2, S_cons_2, ...---*/
180 varnames_Sources[n_control_vars + 2 * i_aux] = config->GetUserSourceName(2 * i_aux);
181 varnames_Sources[n_control_vars + 2 * i_aux + 1] = config->GetUserSourceName(2 * i_aux + 1);
182 }
183
184 /*--- Passive look-up terms ---*/
185 size_t n_lookups = config->GetNLookups();
186 varnames_LookUp.resize(n_lookups);
187 val_vars_LookUp.resize(n_lookups);
188 for (auto iLookup = 0u; iLookup < n_lookups; iLookup++) varnames_LookUp[iLookup] = config->GetLookupName(iLookup);
189
190 if (Kind_DataDriven_Method == ENUM_DATADRIVEN_METHOD::MLP) {
191#ifdef USE_MLPCPP
192 iomap_TD = new MLPToolbox::CIOMap(controlling_variable_names, varnames_TD);
193 iomap_Sources = new MLPToolbox::CIOMap(controlling_variable_names, varnames_Sources);
194 iomap_LookUp = new MLPToolbox::CIOMap(controlling_variable_names, varnames_LookUp);
195 lookup_mlp->PairVariableswithMLPs(*iomap_TD);
196 lookup_mlp->PairVariableswithMLPs(*iomap_Sources);
197 lookup_mlp->PairVariableswithMLPs(*iomap_LookUp);
198#endif
199 }
200}

202unsigned long CFluidFlamelet::EvaluateDataSet(const vector<su2double>& input_scalar, unsigned short lookup_type,
203 vector<su2double>& output_refs) {
204 su2double val_enth = input_scalar[I_ENTH];
205 su2double val_prog = input_scalar[I_PROGVAR];
206 su2double val_mixfrac = include_mixture_fraction ? input_scalar[I_MIXFRAC] : 0.0;
207 vector<string> varnames;
208 vector<su2double> val_vars;
209 vector<su2double*> refs_vars;
210 switch (lookup_type) {
211 case FLAMELET_LOOKUP_OPS::TD:
212 varnames = varnames_TD;
213#ifdef USE_MLPCPP
214 iomap_Current = iomap_TD;
215#endif
216 break;
217 case FLAMELET_LOOKUP_OPS::SOURCES:
218 varnames = varnames_Sources;
219#ifdef USE_MLPCPP
220 iomap_Current = iomap_Sources;
221#endif
222 break;
223 case FLAMELET_LOOKUP_OPS::LOOKUP:
224 varnames = varnames_LookUp;
225#ifdef USE_MLPCPP
226 iomap_Current = iomap_LookUp;
227#endif
228 break;
229 default:
230 break;
231 }
232 if (output_refs.size() != varnames.size())
233 SU2_MPI::Error(string("Output vector size incompatible with manifold lookup operation."), CURRENT_FUNCTION);
234
235 /*--- Add all quantities and their names to the look up vectors. ---*/
236 switch (Kind_DataDriven_Method) {
237 case ENUM_DATADRIVEN_METHOD::LUT:
238 if (include_mixture_fraction) {
239 extrapolation = look_up_table->LookUp_XYZ(varnames, output_refs, val_prog, val_enth, val_mixfrac);
240 } else {
241 extrapolation = look_up_table->LookUp_XY(varnames, output_refs, val_prog, val_enth);
242 }
243 break;
244 case ENUM_DATADRIVEN_METHOD::MLP:
245 refs_vars.resize(output_refs.size());
246 for (auto iVar = 0u; iVar < output_refs.size(); iVar++) refs_vars[iVar] = &output_refs[iVar];
247#ifdef USE_MLPCPP
248 extrapolation = lookup_mlp->PredictANN(iomap_Current, input_scalar, refs_vars);
249#endif
250 break;
251 default:
252 break;
253 }
254
255 return extrapolation;
256}
