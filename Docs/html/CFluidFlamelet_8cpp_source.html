37 CFluidFlamelet::CFluidFlamelet(CConfig* config, su2double value_pressure_operating) : CFluidModel() {
38  rank = SU2_MPI::GetRank();
39  Kind_DataDriven_Method = config->GetKind_DataDriven_Method();
40 
41  /* -- number of auxiliary species transport equations, e.g. 1=CO, 2=NOx --- */
42  n_user_scalars = config->GetNUserScalars();
43  n_control_vars = config->GetNControlVars();
44  include_mixture_fraction = (n_control_vars == 3);
45  n_scalars = config->GetNScalars();
46 
47  if (rank == MASTER_NODE) {
48  cout << "Number of scalars: " << n_scalars << endl;
49  cout << "Number of user scalars: " << n_user_scalars << endl;
50  cout << "Number of control variables: " << n_control_vars << endl;
51  }
52  scalars_vector.resize(n_scalars);
53 
54  table_scalar_names.resize(n_scalars);
55  for (auto iCV = 0u; iCV < n_control_vars; iCV++) table_scalar_names[iCV] = config->GetControllingVariableName(iCV);
56 
57  /*--- auxiliary species transport equations---*/
58  for (auto i_aux = 0u; i_aux < n_user_scalars; i_aux++) {
59  table_scalar_names[n_control_vars + i_aux] = config->GetUserScalarName(i_aux);
60  }
61 
62  controlling_variable_names.resize(n_control_vars);
63  for (auto iCV = 0u; iCV < n_control_vars; iCV++)
64  controlling_variable_names[iCV] = config->GetControllingVariableName(iCV);
65 
66  passive_specie_names.resize(n_user_scalars);
67  for (auto i_aux = 0u; i_aux < n_user_scalars; i_aux++) passive_specie_names[i_aux] = config->GetUserScalarName(i_aux);
68 
69  switch (Kind_DataDriven_Method) {
70  case ENUM_DATADRIVEN_METHOD::LUT:
71  if (rank == MASTER_NODE) {
72  cout << "*****************************************" << endl;
73  cout << "*** initializing the lookup table ***" << endl;
74  cout << "*****************************************" << endl;
75  }
76  look_up_table = new CLookUpTable(config->GetDataDriven_FileNames()[0], table_scalar_names[I_PROGVAR],
77  table_scalar_names[I_ENTH]);
78  break;
79  default:
80  if (rank == MASTER_NODE) {
81  cout << "***********************************************" << endl;
82  cout << "*** initializing the multi-layer perceptron ***" << endl;
83  cout << "***********************************************" << endl;
84  }
85 #ifdef USE_MLPCPP
86  lookup_mlp = new MLPToolbox::CLookUp_ANN(config->GetNDataDriven_Files(), config->GetDataDriven_FileNames());
87  if ((rank == MASTER_NODE)) lookup_mlp->DisplayNetworkInfo();
88 #else
89  SU2_MPI::Error("SU2 was not compiled with MLPCpp enabled (-Denable-mlpcpp=true).", CURRENT_FUNCTION);
90 #endif
91  break;
92  }
93 
94  Pressure = value_pressure_operating;
95 
96  PreprocessLookUp(config);
97 
98  if (rank == MASTER_NODE) {
99  cout << "Preferential diffusion: " << (preferential_diffusion ? "Enabled" : "Disabled") << endl;
100  }
101 }

117 void CFluidFlamelet::SetTDState_T(su2double val_temperature, const su2double* val_scalars) {
118  for (auto iVar = 0u; iVar < n_scalars; iVar++) scalars_vector[iVar] = val_scalars[iVar];
119 
120  /*--- Add all quantities and their names to the look up vectors. ---*/
121  EvaluateDataSet(scalars_vector, FLAMELET_LOOKUP_OPS::THERMO, val_vars_TD);
122 
123  Temperature = val_vars_TD[LOOKUP_TD::TEMPERATURE];
124  Cp = val_vars_TD[LOOKUP_TD::HEATCAPACITY];
125  Mu = val_vars_TD[LOOKUP_TD::VISCOSITY];
126  Kt = val_vars_TD[LOOKUP_TD::CONDUCTIVITY];
127  mass_diffusivity = val_vars_TD[LOOKUP_TD::DIFFUSIONCOEFFICIENT];
128  switch (density_model) {
129  case INC_DENSITYMODEL::FLAMELET:
130  Density = val_vars_TD[LOOKUP_TD::MOLARWEIGHT];
131  molar_weight = Pressure / (Density * UNIVERSAL_GAS_CONSTANT * Temperature);
132  break;
133  case INC_DENSITYMODEL::VARIABLE:
134  molar_weight = val_vars_TD[LOOKUP_TD::MOLARWEIGHT];
135  Density = (molar_weight / 1000) * Pressure / (UNIVERSAL_GAS_CONSTANT * Temperature);
136  break;
137  default:
138  break;
139  }
140  /*--- Compute Cv from Cp and molar weight of the mixture (ideal gas). ---*/
141  Cv = Cp - UNIVERSAL_GAS_CONSTANT / molar_weight;
142 }

144 void CFluidFlamelet::PreprocessLookUp(CConfig* config) {
145  density_model = config->GetKind_DensityModel();
146  /*--- Thermodynamic state variables and names. ---*/
147  varnames_TD.resize(LOOKUP_TD::SIZE);
148  val_vars_TD.resize(LOOKUP_TD::SIZE);
149 
150  /*--- The string in varnames_TD as it appears in the LUT file. ---*/
151  varnames_TD[LOOKUP_TD::TEMPERATURE] = "Temperature";
152  varnames_TD[LOOKUP_TD::HEATCAPACITY] = "Cp";
153  varnames_TD[LOOKUP_TD::VISCOSITY] = "ViscosityDyn";
154  varnames_TD[LOOKUP_TD::CONDUCTIVITY] = "Conductivity";
155  varnames_TD[LOOKUP_TD::DIFFUSIONCOEFFICIENT] = "DiffusionCoefficient";
156 
157  /*--- In case of FLAMELET density model, the density is directly interpolated from the manifold.---*/
158  switch (density_model) {
159  case INC_DENSITYMODEL::FLAMELET:
160  varnames_TD[LOOKUP_TD::MOLARWEIGHT] = "Density";
161  break;
162  case INC_DENSITYMODEL::VARIABLE:
163  varnames_TD[LOOKUP_TD::MOLARWEIGHT] = "MolarWeightMix";
164  break;
165  default:
166  break;
167  }
168  /*--- Scalar source term variables and names. ---*/
169  size_t n_sources = n_control_vars + 2 * n_user_scalars;
170  varnames_Sources.resize(n_sources);
171  val_vars_Sources.resize(n_sources);
172  for (auto iCV = 0u; iCV < n_control_vars; iCV++)
173  varnames_Sources[iCV] = config->GetControllingVariableSourceName(iCV);
174  /*--- No source term for enthalpy ---*/
175 
176  /*--- For the auxiliary equations, we use a positive (production) and a negative (consumption) term:
177  S_tot = S_PROD + S_CONS * Y ---*/
178 
179  for (size_t i_aux = 0; i_aux < n_user_scalars; i_aux++) {
180  /*--- Order of the source terms: S_prod_1, S_cons_1, S_prod_2, S_cons_2, ...---*/
181  varnames_Sources[n_control_vars + 2 * i_aux] = config->GetUserSourceName(2 * i_aux);
182  varnames_Sources[n_control_vars + 2 * i_aux + 1] = config->GetUserSourceName(2 * i_aux + 1);
183  }
184 
185  /*--- Passive look-up terms ---*/
186  size_t n_lookups = config->GetNLookups();
187  if (n_lookups == 0) {
188  varnames_LookUp.resize(1);
189  val_vars_LookUp.resize(1);
190  varnames_LookUp[0] = "NULL";
191  } else {
192  varnames_LookUp.resize(n_lookups);
193  val_vars_LookUp.resize(n_lookups);
194  for (auto iLookup = 0u; iLookup < n_lookups; iLookup++) varnames_LookUp[iLookup] = config->GetLookupName(iLookup);
195  }
196 
197  /*--- Preferential diffusion scalars ---*/
198  varnames_PD.resize(FLAMELET_PREF_DIFF_SCALARS::N_BETA_TERMS);
199  val_vars_PD.resize(FLAMELET_PREF_DIFF_SCALARS::N_BETA_TERMS);
200 
201  varnames_PD[FLAMELET_PREF_DIFF_SCALARS::I_BETA_PROGVAR] = "Beta_ProgVar";
202  varnames_PD[FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH_THERMAL] = "Beta_Enth_Thermal";
203  varnames_PD[FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH] = "Beta_Enth";
204  varnames_PD[FLAMELET_PREF_DIFF_SCALARS::I_BETA_MIXFRAC] = "Beta_MixFrac";
205 
206  val_vars_PD[FLAMELET_PREF_DIFF_SCALARS::I_BETA_PROGVAR] = beta_progvar;
207  val_vars_PD[FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH_THERMAL] = beta_enth_thermal;
208  val_vars_PD[FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH] = beta_enth;
209  val_vars_PD[FLAMELET_PREF_DIFF_SCALARS::I_BETA_MIXFRAC] = beta_mixfrac;
210 
211  preferential_diffusion = config->GetPreferentialDiffusion();
212  switch (Kind_DataDriven_Method) {
213  case ENUM_DATADRIVEN_METHOD::LUT:
214  preferential_diffusion = look_up_table->CheckForVariables(varnames_PD);
215  break;
216  case ENUM_DATADRIVEN_METHOD::MLP:
217 #ifdef USE_MLPCPP
218  n_betas = 0;
219  for (auto iMLP = 0u; iMLP < config->GetNDataDriven_Files(); iMLP++) {
220  auto outputMap = lookup_mlp->FindVariableIndices(iMLP, varnames_PD, false);
221  n_betas += outputMap.size();
222  }
223  preferential_diffusion = (n_betas == varnames_PD.size());
224 #endif
225  break;
226  default:
227  break;
228  }
229 
230  if (!preferential_diffusion && config->GetPreferentialDiffusion())
231  SU2_MPI::Error("Preferential diffusion scalars not included in flamelet manifold.", CURRENT_FUNCTION);
232 
233  if (Kind_DataDriven_Method == ENUM_DATADRIVEN_METHOD::MLP) {
234 #ifdef USE_MLPCPP
235  iomap_TD = new MLPToolbox::CIOMap(controlling_variable_names, varnames_TD);
236  iomap_Sources = new MLPToolbox::CIOMap(controlling_variable_names, varnames_Sources);
237  iomap_LookUp = new MLPToolbox::CIOMap(controlling_variable_names, varnames_LookUp);
238  lookup_mlp->PairVariableswithMLPs(*iomap_TD);
239  lookup_mlp->PairVariableswithMLPs(*iomap_Sources);
240  lookup_mlp->PairVariableswithMLPs(*iomap_LookUp);
241  if (preferential_diffusion) {
242  iomap_PD = new MLPToolbox::CIOMap(controlling_variable_names, varnames_PD);
243  lookup_mlp->PairVariableswithMLPs(*iomap_PD);
244  }
245 #endif
246  } else {
247  for (auto iVar=0u; iVar < varnames_TD.size(); iVar++) {
248  LUT_idx_TD.push_back(look_up_table->GetIndexOfVar(varnames_TD[iVar]));
249  }
250  for (auto iVar=0u; iVar < varnames_Sources.size(); iVar++) {
251  unsigned long LUT_idx;
252  if (noSource(varnames_Sources[iVar])) {
253  LUT_idx = look_up_table->GetNullIndex();
254  } else {
255  LUT_idx = look_up_table->GetIndexOfVar(varnames_Sources[iVar]);
256  }
257  LUT_idx_Sources.push_back(LUT_idx);
258  }
259  for (auto iVar=0u; iVar < varnames_LookUp.size(); iVar++) {
260  unsigned long LUT_idx;
261  if (noSource(varnames_LookUp[iVar]))
262  LUT_idx = look_up_table->GetNullIndex();
263  else
264  LUT_idx = look_up_table->GetIndexOfVar(varnames_LookUp[iVar]);
265  LUT_idx_LookUp.push_back(LUT_idx);
266  }
267  if (preferential_diffusion) {
268  for (auto iVar=0u; iVar < varnames_PD.size(); iVar++) {
269  LUT_idx_PD.push_back(look_up_table->GetIndexOfVar(varnames_PD[iVar]));
270  }
271  }
272  }
273 }

275 unsigned long CFluidFlamelet::EvaluateDataSet(const vector<su2double>& input_scalar, unsigned short lookup_type,
276  vector<su2double>& output_refs) {
277  AD::StartPreacc();
278  for (auto iVar = 0u; iVar < input_scalar.size(); iVar++) AD::SetPreaccIn(input_scalar[iVar]);
279 
280  su2double val_enth = input_scalar[I_ENTH];
281  su2double val_prog = input_scalar[I_PROGVAR];
282  su2double val_mixfrac = include_mixture_fraction ? input_scalar[I_MIXFRAC] : 0.0;
283  vector<su2double> val_vars;
284  vector<su2double*> refs_vars;
285  vector<unsigned long> LUT_idx;
286  switch (lookup_type) {
287  case FLAMELET_LOOKUP_OPS::THERMO:
288  LUT_idx = LUT_idx_TD;
289 #ifdef USE_MLPCPP
290  iomap_Current = iomap_TD;
291 #endif
292  break;
293  case FLAMELET_LOOKUP_OPS::PREFDIF:
294  LUT_idx = LUT_idx_PD;
295 #ifdef USE_MLPCPP
296  iomap_Current = iomap_PD;
297 #endif
298  break;
299  case FLAMELET_LOOKUP_OPS::SOURCES:
300  LUT_idx = LUT_idx_Sources;
301 #ifdef USE_MLPCPP
302  iomap_Current = iomap_Sources;
303 #endif
304  break;
305  case FLAMELET_LOOKUP_OPS::LOOKUP:
306  LUT_idx = LUT_idx_LookUp;
307 #ifdef USE_MLPCPP
308  iomap_Current = iomap_LookUp;
309 #endif
310  break;
311  default:
312  break;
313  }
314 
315 
316  /*--- Add all quantities and their names to the look up vectors. ---*/
317  bool inside;
318  switch (Kind_DataDriven_Method) {
319  case ENUM_DATADRIVEN_METHOD::LUT:
320  if (output_refs.size() != LUT_idx.size())
321  SU2_MPI::Error(string("Output vector size incompatible with manifold lookup operation."), CURRENT_FUNCTION);
322  if (include_mixture_fraction) {
323  inside = look_up_table->LookUp_XYZ(LUT_idx, output_refs, val_prog, val_enth, val_mixfrac);
324  } else {
325  inside = look_up_table->LookUp_XY(LUT_idx, output_refs, val_prog, val_enth);
326  }
327  if (inside) extrapolation = 0;
328  else extrapolation = 1;
329  break;
330  case ENUM_DATADRIVEN_METHOD::MLP:
331  refs_vars.resize(output_refs.size());
332  for (auto iVar = 0u; iVar < output_refs.size(); iVar++) refs_vars[iVar] = &output_refs[iVar];
333 #ifdef USE_MLPCPP
334  extrapolation = lookup_mlp->PredictANN(iomap_Current, input_scalar, refs_vars);
335 #endif
336  break;
337  default:
338  break;
339  }
340  for (auto iVar = 0u; iVar < output_refs.size(); iVar++) AD::SetPreaccOut(output_refs[iVar]);
341  AD::EndPreacc();
342  return extrapolation;
343 }
