35 template <class ScalarType>
36 FORCEINLINE ScalarType* CSysMatrix<ScalarType>::GetBlock_ILUMatrix(unsigned long block_i, unsigned long block_j) {
37  /*--- The position of the diagonal block is known which allows halving the search space. ---*/
38  const auto end = (block_j < block_i) ? dia_ptr_ilu[block_i] : row_ptr_ilu[block_i + 1];
39  for (auto index = (block_j < block_i) ? row_ptr_ilu[block_i] : dia_ptr_ilu[block_i]; index < end; ++index)
40  if (col_ind_ilu[index] == block_j) return &ILU_matrix[index * nVar * nVar];
41  return nullptr;
42 }

54 template <class T, bool alpha, bool beta, bool transp>
55 FORCEINLINE void gemv_impl(unsigned long n, unsigned long m, const T* a, const T* b, T* c) {
56  /*---
57  This is a templated version of GEMV with the constants as boolean
58  template parameters so that they can be optimized away at compilation.
59  This is still the traditional "row dot vector" method.
60  ---*/
61  if (!transp) {
62  for (auto i = 0ul; i < n; i++) {
63  if (!beta) c[i] = 0.0;
64  for (auto j = 0ul; j < m; j++) c[i] += (alpha ? 1 : -1) * a[i * m + j] * b[j];
65  }
66  } else {
67  if (!beta)
68  for (auto j = 0ul; j < m; j++) c[j] = 0.0;
69  for (auto i = 0ul; i < n; i++)
70  for (auto j = 0ul; j < m; j++) c[j] += (alpha ? 1 : -1) * a[i * n + j] * b[i];
71  }
72 }

74 template <class T>
75 FORCEINLINE void gemm_impl(unsigned long n, const T* a, const T* b, T* c) {
76  /*--- Same deal as for GEMV but here only the type is templated. ---*/
77  unsigned long i, j, k;
78  for (i = 0; i < n; i++) {
79  for (j = 0; j < n; j++) {
80  c[i * n + j] = 0.0;
81  for (k = 0; k < n; k++) c[i * n + j] += a[i * n + k] * b[k * n + j];
82  }
83  }
84 }
