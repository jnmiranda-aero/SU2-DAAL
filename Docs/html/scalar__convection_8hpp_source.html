33/*!
34 * \class CUpwScalar
35 * \brief Template class for scalar upwind fluxes between nodes i and j.
36 * \details This class serves as a template for the scalar upwinding residual
37 * classes. The general structure of a scalar upwinding calculation is the
38 * same for many different models, which leads to a lot of repeated code.
39 * By using the template design pattern, these sections of repeated code are
40 * moved to this shared base class, and the specifics of each model
41 * are implemented by derived classes. In order to add a new residual
42 * calculation for a convection residual, extend this class and implement
43 * the pure virtual functions with model-specific behavior.
44 * \ingroup ConvDiscr
45 * \author C. Pederson, A. Bueno., and A. Campos.
46 */
47template <class FlowIndices>
48class CUpwScalar : public CNumerics {
49 protected:
50 enum : unsigned short {MAXNVAR = 8};
51
52 const FlowIndices idx; /*!< \brief Object to manage the access to the flow primitives. */
53 su2double a0 = 0.0; /*!< \brief The maximum of the face-normal velocity and 0. */
54 su2double a1 = 0.0; /*!< \brief The minimum of the face-normal velocity and 0. */
55 su2double Flux[MAXNVAR]; /*!< \brief Final result, diffusive flux/residual. */
56 su2double* Jacobian_i[MAXNVAR]; /*!< \brief Flux Jacobian w.r.t. node i. */
57 su2double* Jacobian_j[MAXNVAR]; /*!< \brief Flux Jacobian w.r.t. node j. */
58 su2double JacobianBuffer[2*MAXNVAR*MAXNVAR]; /*!< \brief Static storage for the two Jacobians. */
59
60 const bool incompressible = false, dynamic_grid = false;
61
62 /*!
63 * \brief A pure virtual function. Derived classes must use it to register the additional
64 * variables they use as preaccumulation inputs, e.g. the density for SST.
65 */
66 virtual void ExtraADPreaccIn() = 0;
67
68 /*!
69 * \brief Model-specific steps in the ComputeResidual method, derived classes
70 * compute the Flux and its Jacobians via this method.
71 * \param[in] config - Definition of the particular problem.
72 */
73 virtual void FinishResidualCalc(const CConfig* config) = 0;
74
75 public:
76 /*!
77 * \brief Constructor of the class.
78 * \param[in] ndim - Number of dimensions of the problem.
79 * \param[in] nvar - Number of variables of the problem.
80 * \param[in] config - Definition of the particular problem.
81 */
82 CUpwScalar(unsigned short ndim, unsigned short nvar, const CConfig* config)
83 : CNumerics(ndim, nvar, config),
84 idx(ndim, config->GetnSpecies()),
85 incompressible(config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE),
86 dynamic_grid(config->GetDynamic_Grid()) {
87 if (nVar > MAXNVAR) {
88 SU2_MPI::Error("Static arrays are too small.", CURRENT_FUNCTION);
89 }
90 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
91 Jacobian_i[iVar] = &JacobianBuffer[iVar * nVar];
92 Jacobian_j[iVar] = &JacobianBuffer[iVar * nVar + MAXNVAR * MAXNVAR];
93 }
94
95 /*--- Initialize the JacobianBuffer to zero. ---*/
96 for (unsigned short iVar = 0; iVar < 2*MAXNVAR*MAXNVAR; iVar++) {
97 JacobianBuffer[iVar] = 0.0;
98 }
99 }
100
101 /*!
102 * \brief Compute the scalar upwind flux between two nodes i and j.
103 * \param[in] config - Definition of the particular problem.
104 * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
105 */
106 CNumerics::ResidualType<> ComputeResidual(const CConfig* config) final {
107 AD::StartPreacc();
108 AD::SetPreaccIn(Normal, nDim);
109 AD::SetPreaccIn(ScalarVar_i, nVar);
110 AD::SetPreaccIn(ScalarVar_j, nVar);
111 if (dynamic_grid) {
112 AD::SetPreaccIn(GridVel_i, nDim);
113 AD::SetPreaccIn(GridVel_j, nDim);
114 }
115 AD::SetPreaccIn(&V_i[idx.Velocity()], nDim);
116 AD::SetPreaccIn(&V_j[idx.Velocity()], nDim);
117 AD::SetPreaccIn(V_i[idx.Density()]);
118 AD::SetPreaccIn(V_j[idx.Density()]);
119 AD::SetPreaccIn(MassFlux);
120
121 ExtraADPreaccIn();
122
123 if (bounded_scalar) {
124 a0 = fmax(0.0, MassFlux) / V_i[idx.Density()];
125 a1 = fmin(0.0, MassFlux) / V_j[idx.Density()];
126 } else {
127 su2double q_ij = 0.0;
128 if (dynamic_grid) {
129 for (unsigned short iDim = 0; iDim < nDim; iDim++) {
130 su2double Velocity_i = V_i[iDim + idx.Velocity()] - GridVel_i[iDim];
131 su2double Velocity_j = V_j[iDim + idx.Velocity()] - GridVel_j[iDim];
132 q_ij += 0.5 * (Velocity_i + Velocity_j) * Normal[iDim];
133 }
134 } else {
135 for (unsigned short iDim = 0; iDim < nDim; iDim++) {
136 q_ij += 0.5 * (V_i[iDim + idx.Velocity()] + V_j[iDim + idx.Velocity()]) * Normal[iDim];
137 }
138 }
139 a0 = fmax(0.0, q_ij);
140 a1 = fmin(0.0, q_ij);
141 }
142
143 FinishResidualCalc(config);
144
145 AD::SetPreaccOut(Flux, nVar);
146 AD::EndPreacc();
147
148 return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
149 }
150};
