30CMultizoneOutput::CMultizoneOutput(const CConfig* driver_config, const CConfig* const* config, unsigned short nDim) :
31 COutput(driver_config, nDim, false) {
32
33 unsigned short iZone = 0;
34
35 nZone = driver_config->GetnZone();
36
37 bgs_res_name = "BGS_RES";
38
39 write_zone = false;
40
41 /*--- If the zone output is disabled for every zone ---*/
42 for (iZone = 0; iZone < nZone; iZone++){
43 write_zone = config[iZone]->GetWrt_ZoneConv();
44 }
45
46 if (nRequestedHistoryFields == 0){
47 requestedHistoryFields.emplace_back("ITER");
48 if (config[ZONE_0]->GetTime_Domain()) requestedHistoryFields.emplace_back("CUR_TIME");
49 for (iZone = 0; iZone < nZone; iZone++){
50 requestedHistoryFields.emplace_back(bgs_res_name + "[" + PrintingToolbox::to_string(iZone) + "]");
51 requestedHistoryFields.emplace_back("AVG_RES[" + PrintingToolbox::to_string(iZone) + "]");
52 }
53 nRequestedHistoryFields = requestedHistoryFields.size();
54 }
55
56 if (nRequestedScreenFields == 0){
57 if (config[ZONE_0]->GetTime_Domain()) requestedScreenFields.emplace_back("TIME_ITER");
58 requestedScreenFields.emplace_back("OUTER_ITER");
59 for (iZone = 0; iZone < nZone; iZone++){
60 requestedScreenFields.emplace_back("AVG_" + bgs_res_name + "[" + PrintingToolbox::to_string(iZone) + "]");
61 }
62 nRequestedScreenFields = requestedScreenFields.size();
63 }
64
65 multiZoneHeaderString = "Multizone Summary";
66
67 historyFilename = driver_config->GetCaseName();
68
69 /*--- Add the correct file extension depending on the file format ---*/
70
71 string hist_ext = ".csv";
72 if (driver_config->GetTabular_FileFormat() == TAB_OUTPUT::TAB_TECPLOT) hist_ext = ".dat";
73
74 historyFilename += hist_ext;
75
76 /*--- Set the default convergence field --- */
77
78 if (convFields.empty()) convFields.emplace_back("AVG_BGS_RES[0]");
79
80}

82void CMultizoneOutput::LoadMultizoneHistoryData(const COutput* const* output, const CConfig* const* config) {
83
84 string nameMultizone, zoneIndex;
85 su2double comboValue = 0;
86
87 for (unsigned short iZone = 0; iZone < nZone; iZone++) {
88 zoneIndex = "[" + PrintingToolbox::to_string(iZone) + "]";
89
90 /*--- For all the variables per solver ---*/
91 for (const auto& item : output[iZone]->GetHistoryFields()) {
92 const auto& name = item.first;
93 if (name != "TIME_ITER" && name != "OUTER_ITER") {
94 nameMultizone = name + zoneIndex;
95 SetHistoryOutputValue(nameMultizone, item.second.value);
96 }
97 if (name == "COMBO") {
98 comboValue += item.second.value;
99 }
100 }
101
102 /*-- Load the PerSurface values.- ---*/
103 for (const auto& item : output[iZone]->GetHistoryPerSurfaceFields()) {
104 const auto& name = item.first;
105 nameMultizone = name + zoneIndex;
106
107 unsigned short iMarker = 0;
108 for (const auto& field : item.second) {
109 SetHistoryOutputPerSurfaceValue(nameMultizone, field.value, iMarker++);
110 }
111 }
112 }
113 SetHistoryOutputValue("COMBO", comboValue);
114}

116void CMultizoneOutput::SetMultizoneHistoryOutputFields(const COutput* const* output, const CConfig* const* config) {
117
118 string name, header, group, zoneIndex;
119
120 /*--- Set the fields ---*/
121 for (unsigned short iZone = 0; iZone < nZone; iZone++) {
122
123 const auto& ZoneHistoryFields = output[iZone]->GetHistoryFields();
124 zoneIndex = "[" + PrintingToolbox::to_string(iZone) + "]";
125
126 /*--- For all the variables per solver ---*/
127 for (const auto& nameSinglezone : output[iZone]->GetHistoryOutputList()) {
128
129 if (nameSinglezone != "TIME_ITER" && nameSinglezone != "OUTER_ITER") {
130
131 const auto& field = ZoneHistoryFields.at(nameSinglezone);
132
133 name = nameSinglezone + zoneIndex;
134 header = field.fieldName + zoneIndex;
135 group = field.outputGroup + zoneIndex;
136
137 AddHistoryOutput(name, header, field.screenFormat, group, "", field.fieldType );
138 }
139 }
140
141 /*--- Prepare Marker lists that are passed to 'AddHistoryOutputPerSurface'. ---*/
142 vector<string> Marker_Analyze;
143 for (unsigned short iMarker_Analyze = 0; iMarker_Analyze < config[iZone]->GetnMarker_Analyze(); iMarker_Analyze++) {
144 Marker_Analyze.push_back(config[iZone]->GetMarker_Analyze_TagBound(iMarker_Analyze));
145 }
146
147 vector<string> Marker_Monitoring;
148 for (unsigned short iMarker_Monitoring = 0; iMarker_Monitoring < config[iZone]->GetnMarker_Monitoring(); iMarker_Monitoring++) {
149 Marker_Monitoring.push_back(config[iZone]->GetMarker_Monitoring_TagBound(iMarker_Monitoring));
150 }
151
152 /*--- Add the PerSurface outputs. ---*/
153 const auto& ZoneHistoryPerSurfaceFields = output[iZone]->GetHistoryPerSurfaceFields();
154
155 for (const auto& nameSinglezone : output[iZone]->GetHistoryOutputPerSurfaceList()) {
156
157 const auto& field = ZoneHistoryPerSurfaceFields.at(nameSinglezone);
158
159 name = nameSinglezone + zoneIndex;
160
161 /*--- Remove the unnecessary Marker name from the fieldName, i.e. "Avg_Massflow(inlet)"->"Avg_Massflow". ---*/
162 /*--- Note that index zero in 'field[0]' refers to a specific Marker. Some attributes remain constant over the markers
163 like the first part of the name, the screenFormat and the fieldType. ---*/
164 string baseheader;
165 const auto pos = field[0].fieldName.find('(');
166 if (pos != std::string::npos)
167 baseheader = field[0].fieldName.substr(0, pos);
168 else
169 SU2_MPI::Error("Cannot process PerSurface *_SURF history output: " + baseheader, CURRENT_FUNCTION);
170
171 header = baseheader + zoneIndex;
172 /*--- Attach zone-index to the group after determining which group it is. ---*/
173 group = field[0].outputGroup;
174
175 /*--- Determine whether Maker_Analyze/Monitoring has to be used. ---*/
176 auto* Marker = &Marker_Monitoring;
177 if ((group == "FLOW_COEFF_SURF") || (group == "SPECIES_COEFF_SURF") )
178 Marker = &Marker_Analyze;
179 else if (group != "AERO_COEFF_SURF" && group != "HEAT_SURF")
180 SU2_MPI::Error("Per Surface output group unknown: " + group, CURRENT_FUNCTION);
181
182 group += zoneIndex;
183
184 AddHistoryOutputPerSurface(name, header, field[0].screenFormat, group, *Marker, field[0].fieldType );
185 }
186 }
187 AddHistoryOutput("COMBO", "ComboObj", ScreenOutputFormat::SCIENTIFIC, "COMBO", "Combined obj. function value.", HistoryFieldType::COEFFICIENT);
188}

190bool CMultizoneOutput::WriteScreenHeader(const CConfig *config) {
191
192 /*--- Print header if the outer iteration is zero or zonal convergence is printed ---*/
193
194 return curOuterIter == 0 || write_zone;
195
196 /*--- Always print header if it is forced ---*/
197
198 if (headerNeeded){
199 headerNeeded = false;
200 return true;
201 }
202
203 return false;
204}

206bool CMultizoneOutput::WriteScreenOutput(const CConfig *config) {
207
208 unsigned long ScreenWrt_Freq_Outer = config->GetScreen_Wrt_Freq(1);
209 unsigned long ScreenWrt_Freq_Time = config->GetScreen_Wrt_Freq(0);
210
211 /*--- Check if screen output should be written --- */
212
213 if (!PrintOutput(curTimeIter, ScreenWrt_Freq_Time)&&
214 !(curTimeIter == config->GetnTime_Iter() - 1)){
215
216 return false;
217
218 }
219
220 if (convergence) {return true;}
221
222 if (!PrintOutput(curOuterIter, ScreenWrt_Freq_Outer) &&
223 !(curOuterIter == config->GetnOuter_Iter() - 1)){
224
225 return false;
226
227 }
228
229 return true;
230}
