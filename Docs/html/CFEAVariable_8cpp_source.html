30CFEAVariable::CFEAVariable(const su2double *val_fea, unsigned long npoint, unsigned long ndim,
31 unsigned long nvar, CConfig *config) :
32 CVariable(npoint, ndim, config->GetTime_Domain()? 3*nvar : nvar, config) {
33
34 /*--- In time domain CVariable::nVar is mult. by 3 ^^^ (for vel. and accel.)
35 * and the original value then restored (below). ---*/
36 nVar = nvar;
37 /*--- This simplifies the discrete adjoint of this solver, as it allows an abstract
38 * treatment of the "state" (disp. vel. accel.) whose details are only important
39 * for the primal solver. In time domain the primal "believes" it has nVar variables,
40 * which it uses for linear solvers, and then handles velocity and acceleration
41 * explicitly (for time integration). Whereas the discrete adjoint "thinks" the
42 * primal solution has 3*nVar variables. This is a little different from simply
43 * giving names to parts of the solution, it requires the methods of CVariable that
44 * deal with adjoints to deduce "nVar" from the container, rather than relying on
45 * the nVar member (which is manipulated above, so that CVariable::SetSolution, etc.
46 * still work as expected for the primal solver). ---*/
47
48 const bool dynamic_analysis = config->GetTime_Domain();
49 const bool body_forces = config->GetDeadLoad();
50 const bool prestretch_fem = config->GetPrestretch(); // Structure is prestretched
51 const bool discrete_adjoint = config->GetDiscrete_Adjoint();
52 const bool refgeom = config->GetRefGeom(); // Reference geometry needs to be stored
53 const bool multizone = config->GetMultizone_Problem();
54 const bool fsi_analysis = config->GetFSI_Simulation() || multizone;
55
56 VonMises_Stress.resize(nPoint) = su2double(0.0);
57
58 if (nDim==2) Stress.resize(nPoint,3);
59 else Stress.resize(nPoint,6);
60
61 /*--- Initialization of variables ---*/
62 for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint)
63 for (unsigned long iVar = 0; iVar < nVar; iVar++)
64 Solution(iPoint,iVar) = val_fea[iVar];
65
66 if (fsi_analysis) {
67 Solution_Pred.resize(nPoint,nVar);
68 Solution_Pred_Old.resize(nPoint,nVar);
69
70 for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint)
71 for (unsigned long iVar = 0; iVar < nVar; iVar++)
72 Solution_Pred(iPoint,iVar) = Solution_Pred_Old(iPoint,iVar) = val_fea[iVar];
73 }
74
75 if (dynamic_analysis) {
76 if (fsi_analysis) Solution_Vel_Pred.resize(nPoint,nVar);
77
78 for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
79 for (unsigned long iVar = 0; iVar < nVar; iVar++) {
80 Solution_Vel_time_n(iPoint,iVar) = Solution_Vel(iPoint,iVar) = val_fea[iVar+nVar];
81 Solution_Accel_time_n(iPoint,iVar) = Solution_Accel(iPoint,iVar) = val_fea[iVar+2*nVar];
82 if (fsi_analysis) Solution_Vel_Pred(iPoint,iVar) = val_fea[iVar+nVar];
83 }
84 }
85 }
86
87 if (discrete_adjoint && fsi_analysis) Solution_Old = Solution;
88
89 /*--- Body residual ---*/
90 if (body_forces) Residual_Ext_Body.resize(nPoint,nVar) = su2double(0.0);
91
92 if (refgeom) Reference_Geometry.resize(nPoint,nVar);
93
94 if (prestretch_fem) Prestretch.resize(nPoint,nVar);
95
96 if (multizone) Set_BGSSolution_k();
97
98 if (config->GetTopology_Optimization()) {
99 nAuxVar = 1;
100 AuxVar.resize(nPoint);
101 }
102}
