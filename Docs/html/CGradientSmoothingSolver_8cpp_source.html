214 void CGradientSmoothingSolver::ApplyGradientSmoothingSurface(CGeometry* geometry, CNumerics* numerics,
215  const CConfig* config) {
216  /*--- Set vector and sparse matrix to 0 ---*/
217  LinSysSol.SetValZero();
218  LinSysRes.SetValZero();
219  Jacobian.SetValZero();
220  std::fill(visited.begin(), visited.end(), false);
221 
222  /*--- Loop over all DV markers to compute the stiffness matrix for the smoothing operator. ---*/
223  for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
224  if (config->GetMarker_All_DV(iMarker) == YES) {
225  /*--- Compute the stiffness matrix for the smoothing operator. ---*/
226  Compute_Surface_StiffMatrix(geometry, numerics, config, iMarker);
227 
228  Compute_Surface_Residual(geometry, config, iMarker);
229 
230  if (config->GetDirichletSurfaceBound()) {
231  BC_Surface_Dirichlet(geometry, config, iMarker);
232  }
233  }
234  }
235 
236  /*--- Set the matrix to identity if the current mpi rank holds no part of the DV marker. ---*/
237  Complete_Surface_StiffMatrix(geometry);
238 
239  /*--- Solve the system and write the result back. ---*/
240  Solve_Linear_System(geometry, config);
241 
242  WriteSensitivity(geometry, config);
243 }

245 void CGradientSmoothingSolver::ApplyGradientSmoothingDV(CGeometry* geometry, CNumerics* numerics,
246  CSurfaceMovement* surface_movement,
247  CVolumetricMovement* grid_movement, CConfig* config,
248  su2double** Gradient) {
249  unsigned nDVtotal=config->GetnDV_Total();
250  unsigned column, row;
251  unsigned long iPoint;
252  unsigned short iDim;
253  vector<su2double> seedvector(nDVtotal, 0.0);
254  vector<su2double> x(nDVtotal, 0.0);
255  hessian.Initialize(nDVtotal);
256 
257  /*--- Reset the Jacobian to 0 ---*/
258  Jacobian.SetValZero();
259 
260  /*--- Record the parameterization on the AD tape. ---*/
261  if (rank == MASTER_NODE) cout << " calculate the original gradient" << endl;
262  RecordParameterizationJacobian(geometry, surface_movement, activeCoord, config);
263 
264  /*--- Calculate the original gradient. ---*/
265  CalculateOriginalGradient(geometry, grid_movement, config, Gradient);
266 
267  /*--- Compute the full Sobolev Hessian approximation column by column. ---*/
268  if (rank == MASTER_NODE) cout << " computing the system matrix line by line" << endl;
269 
270  auto mat_vec = GetStiffnessMatrixVectorProduct<su2matvecscalar>(geometry, numerics, config);
271 
272  for (column=0; column<nDVtotal; column++) {
273 
274  if (rank == MASTER_NODE) cout << " working in column " << column << endl;
275 
276  /*--- Create a seeding vector and reset auxilliary vectors for the surface case. ---*/
277  std::fill(seedvector.begin(), seedvector.end(), 0.0);
278  seedvector[column] = 1.0;
279 
280  helperVecIn.SetValZero();
281  helperVecOut.SetValZero();
282  helperVecAux.SetValZero();
283 
284  /*--- Forward mode evaluation, i.e., matrix vector produt with the parameterization Jacobian.---*/
285  ProjectDVtoMesh(geometry, seedvector, helperVecIn, activeCoord, config);
286 
287  /*--- Matrix vector product with the Laplace-Beltrami stiffness matrix. ---*/
288  if (config->GetSmoothOnSurface()) {
289 
290  CSysMatrixComms::Initiate(helperVecIn, geometry, config, MPI_QUANTITIES::SOLUTION_MATRIX);
291  CSysMatrixComms::Complete(helperVecIn, geometry, config, MPI_QUANTITIES::SOLUTION_MATRIX);
292 
293  mat_vec(helperVecIn, helperVecAux);
294 
295  /*--- Delete entries from halo cells, otherwise we get errors for parallel computations ---*/
296  for (iPoint = 0; iPoint < geometry->GetnPointDomain(); iPoint++) {
297  for (iDim = 0; iDim < nDim; iDim++) {
298  helperVecOut(iPoint,iDim) = helperVecAux(iPoint,iDim);
299  }
300  }
301 
302  } else {
303 
304  /*--- Forward evaluation of the mesh deformation ---*/
305  CGradientSmoothingSolverDetails::WriteVectorToGeometry<su2matvecscalar>(geometry, helperVecIn);
306  grid_movement->SetVolume_Deformation(geometry, config, false, true, true);
307  CGradientSmoothingSolverDetails::ReadVectorToGeometry<su2matvecscalar>(geometry, helperVecIn);
308 
309  CSysMatrixComms::Initiate(helperVecIn, geometry, config, MPI_QUANTITIES::SOLUTION_MATRIX);
310  CSysMatrixComms::Complete(helperVecIn, geometry, config, MPI_QUANTITIES::SOLUTION_MATRIX);
311 
312  mat_vec(helperVecIn, helperVecAux);
313 
314  /*--- Only consider values inside the domain for the output of the matrix vector product ---*/
315  for (iPoint = 0; iPoint < geometry->GetnPointDomain(); iPoint++) {
316  for (iDim = 0; iDim < nDim; iDim++) {
317  helperVecOut(iPoint,iDim) = helperVecAux(iPoint,iDim);
318  }
319  }
320 
321  /*--- Forward evaluation of the mesh deformation ---*/
322  CGradientSmoothingSolverDetails::WriteVectorToGeometry<su2matvecscalar>(geometry, helperVecOut);
323  grid_movement->SetVolume_Deformation(geometry, config, false, true, false);
324  CGradientSmoothingSolverDetails::ReadVectorToGeometry<su2matvecscalar>(geometry, helperVecOut);
325 
326  }
327 
328  /*--- Reverse mode evaluation, i.e., matrix vector produt with the transposed parameterization Jacobian.---*/
329  ProjectMeshToDV(geometry, helperVecOut, seedvector, activeCoord, config);
330 
331  /*--- Extract the projected direction ---*/
332  for (row=0; row<nDVtotal; row++) {
333  hessian(row,column) = SU2_TYPE::GetValue(seedvector[row]);
334  }
335  }
336 
337  /*--- Output the complete system matrix. ---*/
338  if (rank == MASTER_NODE) {
339  /*--- For multizone append zone number to filename. ---*/
340  string hess_filename = config->GetObjFunc_Hess_FileName();
341  if (config->GetMultizone_Problem()) {
342  hess_filename = config->GetMultizone_FileName(hess_filename, config->GetiZone(), ".dat");
343  }
344  ofstream SysMatrix(hess_filename);
345  SysMatrix.precision(config->GetOutput_Precision());
346 
347  for (row=0; row<nDVtotal; row++) {
348  for (column=0; column<nDVtotal; column++) {
349  SysMatrix << hessian(row,column);
350  if (column!=nDVtotal-1) SysMatrix << ", ";
351  }
352  if (row!=nDVtotal-1) SysMatrix << '\n';
353  }
354  SysMatrix.close();
355  }
356 
357  /*--- Calculate and output the treated gradient. ---*/
358  hessian.Invert();
359  hessian.MatVecMult(deltaP.begin(), x.begin());
360  deltaP = x;
361 
362  OutputDVGradient(config);
363 }

365 void CGradientSmoothingSolver::Compute_StiffMatrix(CGeometry* geometry, CNumerics* numerics, const CConfig* config) {
366  unsigned long iElem, iNode;
367  unsigned int iDim, nNodes = 0, NelNodes, jNode;
368  std::array<unsigned long, MAXNNODE_3D> indexNode;
369  su2double val_Coord, HiHj = 0.0;
370  int EL_KIND = 0;
371  su2activematrix DHiDHj, Jacobian_block;
372 
373  Jacobian_block.resize(nDim, nDim) = su2double(0.0);
374 
375  /*--- Loops over all the elements ---*/
376 
377  for (iElem = 0; iElem < geometry->GetnElem(); iElem++) {
378 
379  GetElemKindAndNumNodes(geometry->elem[iElem]->GetVTK_Type(), EL_KIND, nNodes);
380 
381  for (iNode = 0; iNode < nNodes; iNode++) {
382 
383  indexNode[iNode] = geometry->elem[iElem]->GetNode(iNode);
384 
385  for (iDim = 0; iDim < nDim; iDim++) {
386  val_Coord = Get_ValCoord(geometry, indexNode[iNode], iDim);
387  element_container[GRAD_TERM][EL_KIND]->SetRef_Coord(iNode, iDim, val_Coord);
388  }
389 
390  }
391 
392  /*--- compute the contributions of the single elements inside the numerics container ---*/
393 
394  numerics->Compute_Tangent_Matrix(element_container[GRAD_TERM][EL_KIND], config);
395 
396  NelNodes = element_container[GRAD_TERM][EL_KIND]->GetnNodes();
397 
398  /*--- for all nodes add the contribution to the system Jacobian ---*/
399 
400  for (iNode = 0; iNode < NelNodes; iNode++) {
401 
402  for (jNode = 0; jNode < NelNodes; jNode++) {
403 
404  DHiDHj = element_container[GRAD_TERM][EL_KIND]->Get_DHiDHj(iNode, jNode);
405  HiHj = element_container[GRAD_TERM][EL_KIND]->Get_HiHj(iNode, jNode);
406 
407  if (config->GetSmoothSepDim()) {
408  Jacobian_block[0][0] = DHiDHj[GetCurrentDim()][GetCurrentDim()] + HiHj;
409  Jacobian.AddBlock(indexNode[iNode], indexNode[jNode], Jacobian_block);
410 
411  } else {
412  for (iDim = 0; iDim < nDim; iDim++) {
413  Jacobian_block[iDim][iDim] = DHiDHj[iDim][iDim] + HiHj;
414  }
415  Jacobian.AddBlock(indexNode[iNode], indexNode[jNode], Jacobian_block);
416  }
417  }
418  }
419  }

423 void CGradientSmoothingSolver::Compute_Surface_StiffMatrix(CGeometry* geometry, CNumerics* numerics,
424  const CConfig* config, unsigned long val_marker,
425  unsigned int nSurfDim) {
426  unsigned long iElem, iPoint, iVertex, iDim, iSurfDim;
427  unsigned int iNode, jNode, nNodes = 0, NelNodes;
428  std::array<unsigned long, MAXNNODE_2D> indexNode;
429  std::array<unsigned long, MAXNNODE_2D> indexVertex;
430  int EL_KIND = 0;
431  su2double val_Coord, HiHj = 0.0;
432  su2activematrix DHiDHj, Jacobian_block;
433 
434  Jacobian_block.resize(nDim, nDim) = su2double(0.0);
435 
436  /*--- Loops over all the elements ---*/
437  for (iElem = 0; iElem < geometry->GetnElem_Bound(val_marker); iElem++) {
438 
439  /*--- Identify the kind of boundary element ---*/
440  GetElemKindAndNumNodes(geometry->bound[val_marker][iElem]->GetVTK_Type(), EL_KIND, nNodes);
441 
442  /*--- Retrieve the boundary reference and current coordinates ---*/
443 
444  for (iNode = 0; iNode < nNodes; iNode++) {
445  indexNode[iNode] = geometry->bound[val_marker][iElem]->GetNode(iNode);
446 
447  for (iDim = 0; iDim < nDim; iDim++) {
448  val_Coord = Get_ValCoord(geometry, indexNode[iNode], iDim);
449  element_container[GRAD_TERM][EL_KIND]->SetRef_Coord(iNode, iDim, val_Coord);
450  }
451  }
452 
453  /*--- We need the indices of the vertices, which are "Dual Grid Info" ---*/
454  for (iVertex = 0; iVertex < geometry->nVertex[val_marker]; iVertex++) {
455  iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
456  for (iNode = 0; iNode < nNodes; iNode++) {
457  if (iPoint == indexNode[iNode]) indexVertex[iNode] = iVertex;
458  }
459  }
460 
461  /*--- compute the contributions of the single elements inside the numerics container ---*/
462  numerics->Compute_Tangent_Matrix(element_container[GRAD_TERM][EL_KIND], config);
463 
464  NelNodes = element_container[GRAD_TERM][EL_KIND]->GetnNodes();
465 
466  /*--- for all nodes add the contribution to the system Jacobian ---*/
467 
468  for (iNode = 0; iNode < NelNodes; iNode++) {
469  for (jNode = 0; jNode < NelNodes; jNode++) {
470 
471  DHiDHj = element_container[GRAD_TERM][EL_KIND]->Get_DHiDHj(iNode, jNode);
472  HiHj = element_container[GRAD_TERM][EL_KIND]->Get_HiHj(iNode, jNode);
473 
474  /*--- Get the calculated stiffness blocks for the current element and add them to the block. ---*/
475  for (iSurfDim=0; iSurfDim<nSurfDim; iSurfDim++) {
476  Jacobian_block[iSurfDim][iSurfDim] = DHiDHj[0][0] + HiHj;
477  }
478  Jacobian.AddBlock(indexNode[iNode], indexNode[jNode], Jacobian_block);
479 
480  /*--- Store if we set values for a given node already ---*/
481  visited[indexNode[iNode]]=true;
482  visited[indexNode[jNode]]=true;
483 
484  }
485  }

489 void CGradientSmoothingSolver::Compute_Residual(CGeometry* geometry, const CConfig* config) {
490  unsigned long iElem;
491  unsigned int iDim, iNode, nNodes = 0;
492  int EL_KIND = 0;
493  std::array<unsigned long, MAXNNODE_3D> indexNode;
494  std::array<su2double, MAXNDIM> Residual;
495  su2double Weight, Jac_X, val_Coord;
496 
497  for (iElem = 0; iElem < geometry->GetnElem(); iElem++) {
498 
499  GetElemKindAndNumNodes(geometry->elem[iElem]->GetVTK_Type(), EL_KIND, nNodes);
500 
501  for (iNode = 0; iNode < nNodes; iNode++) {
502 
503  indexNode[iNode] = geometry->elem[iElem]->GetNode(iNode);
504 
505  for (iDim = 0; iDim < nDim; iDim++) {
506  val_Coord = Get_ValCoord(geometry, indexNode[iNode], iDim);
507  element_container[GRAD_TERM][EL_KIND]->SetRef_Coord(iNode, iDim, val_Coord);
508  }
509 
510  }
511 
512  element_container[GRAD_TERM][EL_KIND]->ClearElement(); /*--- Restarts the element: avoids adding over previous results in other elements --*/
513  element_container[GRAD_TERM][EL_KIND]->ComputeGrad_Linear();
514  unsigned int nGauss = element_container[GRAD_TERM][EL_KIND]->GetnGaussPoints();
515 
516  for (unsigned int iGauss = 0; iGauss < nGauss; iGauss++) {
517 
518  for (iNode = 0; iNode < nNodes; iNode++) {
519  indexNode[iNode] = geometry->elem[iElem]->GetNode(iNode);
520  }
521 
522  Weight = element_container[GRAD_TERM][EL_KIND]->GetWeight(iGauss);
523  Jac_X = element_container[GRAD_TERM][EL_KIND]->GetJ_X(iGauss);
524 
525  for (unsigned int iNode = 0; iNode < nNodes; iNode++) {
526  if (config->GetSmoothSepDim()) {
527  Residual[GetCurrentDim()] +=
528  Weight * Jac_X * element_container[GRAD_TERM][EL_KIND]->GetNi(iNode, iGauss) * nodes->GetSensitivity(indexNode[iNode], GetCurrentDim());
529  LinSysRes.AddBlock(indexNode[iNode], &Residual[GetCurrentDim()]);
530  } else {
531  for (iDim = 0; iDim < nDim; iDim++) {
532  Residual[iDim] += Weight * Jac_X * element_container[GRAD_TERM][EL_KIND]->GetNi(iNode, iGauss) * nodes->GetSensitivity(indexNode[iNode], iDim);
533  }
534  LinSysRes.AddBlock(indexNode[iNode], Residual);
535  }
536 
537  for (iDim = 0; iDim < nDim; iDim++) {
538  Residual[iDim] = 0;
539  }
540 
541  }
542  }
543  }
