86CNumerics::ResidualType<> CCentLaxInc_Flow::ComputeResidual(const CConfig* config) {
87
88 implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
89
90 su2double U_i[5] = {0.0}, U_j[5] = {0.0};
91 su2double ProjGridVel = 0.0, ProjVelocity = 0.0;
92
93 /*--- Primitive variables at point i and j ---*/
94
95 Pressure_i = V_i[0]; Pressure_j = V_j[0];
96 Temperature_i = V_i[nDim+1]; Temperature_j = V_j[nDim+1];
97 DensityInc_i = V_i[nDim+2]; DensityInc_j = V_j[nDim+2];
98 BetaInc2_i = V_i[nDim+3]; BetaInc2_j = V_j[nDim+3];
99 Cp_i = V_i[nDim+7]; Cp_j = V_j[nDim+7];
100 Enthalpy_i = Cp_i*Temperature_i; Enthalpy_j = Cp_j*Temperature_j;
101
102 Area = 0.0;
103 sq_vel_i = 0.0; sq_vel_j = 0.0; ProjVelocity_i = 0.0; ProjVelocity_j = 0.0;
104 for (iDim = 0; iDim < nDim; iDim++) {
105 Velocity_i[iDim] = V_i[iDim+1];
106 Velocity_j[iDim] = V_j[iDim+1];
107 MeanVelocity[iDim] = 0.5*(Velocity_i[iDim]+Velocity_j[iDim]);
108 sq_vel_i += 0.5*Velocity_i[iDim]*Velocity_i[iDim];
109 sq_vel_j += 0.5*Velocity_j[iDim]*Velocity_j[iDim];
110 ProjVelocity_i += Velocity_i[iDim]*Normal[iDim];
111 ProjVelocity_j += Velocity_j[iDim]*Normal[iDim];
112 Area += Normal[iDim]*Normal[iDim];
113 }
114 Area = sqrt(Area);
115
116 /*--- Compute mean values of the variables ---*/
117
118 MeanDensity = 0.5*(DensityInc_i + DensityInc_j);
119 MeanPressure = 0.5*(Pressure_i + Pressure_j);
120 MeanBetaInc2 = 0.5*(BetaInc2_i + BetaInc2_j);
121 MeanEnthalpy = 0.5*(Enthalpy_i + Enthalpy_j);
122 MeanCp = 0.5*(Cp_i + Cp_j);
123 MeanTemperature = 0.5*(Temperature_i + Temperature_j);
124
125 /*--- We need the derivative of the equation of state to build the
126 preconditioning matrix. For now, the only option is the ideal gas
127 law, but in the future, dRhodT should be in the fluid model. ---*/
128
129 MeandRhodT = 0.0;
130 if (variable_density) {
131 MeandRhodT = -MeanDensity/MeanTemperature;
132 }
133
134 /*--- Get projected flux tensor ---*/
135
136 GetInviscidIncProjFlux(&MeanDensity, MeanVelocity, &MeanPressure, &MeanBetaInc2, &MeanEnthalpy, Normal, ProjFlux);
137
138 /*--- Jacobians of the inviscid flux ---*/
139
140 if (implicit) {
141 GetInviscidIncProjJac(&MeanDensity, MeanVelocity, &MeanBetaInc2, &MeanCp, &MeanTemperature, &MeandRhodT, Normal, 0.5, Jacobian_i);
142 for (iVar = 0; iVar < nVar; iVar++) {
143 for (jVar = 0; jVar < nVar; jVar++) {
144 Jacobian_j[iVar][jVar] = Jacobian_i[iVar][jVar];
145 }
146 }
147 }
148
149 /*--- Corrections due to grid motion ---*/
150 if (dynamic_grid) {
151
152 /*--- Recompute conservative variables ---*/
153
154 U_i[0] = DensityInc_i; U_j[0] = DensityInc_j;
155 for (iDim = 0; iDim < nDim; iDim++) {
156 U_i[iDim+1] = DensityInc_i*Velocity_i[iDim]; U_j[iDim+1] = DensityInc_j*Velocity_j[iDim];
157 }
158 U_i[nDim+1] = DensityInc_i*Enthalpy_i; U_j[nDim+1] = DensityInc_j*Enthalpy_j;
159
160 for (iDim = 0; iDim < nDim; iDim++)
161 ProjVelocity += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
162
163 /*--- Residual contributions ---*/
164 for (iVar = 0; iVar < nVar; iVar++) {
165 ProjFlux[iVar] -= ProjVelocity * 0.5*(U_i[iVar]+U_j[iVar]);
166
167 /*--- Jacobian contributions ---*/
168 /*--- Implicit terms ---*/
169 if (implicit) {
170 for (iDim = 0; iDim < nDim; iDim++){
171 Jacobian_i[iDim+1][iDim+1] -= 0.5*ProjVelocity*DensityInc_i;
172 Jacobian_j[iDim+1][iDim+1] -= 0.5*ProjVelocity*DensityInc_j;
173 }
174 Jacobian_i[nDim+1][nDim+1] -= 0.5*ProjVelocity*DensityInc_i*Cp_i;
175 Jacobian_j[nDim+1][nDim+1] -= 0.5*ProjVelocity*DensityInc_j*Cp_j;
176 }
177 }
178 }
179
180 /*--- Computes differences btw. conservative variables ---*/
181
182 for (iVar = 0; iVar < nVar; iVar++)
183 Diff_V[iVar] = V_i[iVar]-V_j[iVar];
184
185 /*--- Build the preconditioning matrix using mean values ---*/
186
187 GetPreconditioner(&MeanDensity, MeanVelocity, &MeanBetaInc2, &MeanCp, &MeanTemperature, &MeandRhodT, Precon);
188
189 /*--- Compute the local espectral radius of the preconditioned system
190 and the stretching factor. ---*/
191
192 SoundSpeed_i = sqrt(BetaInc2_i*Area*Area);
193 SoundSpeed_j = sqrt(BetaInc2_j*Area*Area);
194
195 /*--- Projected velocity adjustment due to mesh motion ---*/
196
197 if (dynamic_grid) {
198 ProjGridVel = 0.0;
199 for (iDim = 0; iDim < nDim; iDim++) {
200 ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
201 }
202 ProjVelocity_i -= ProjGridVel;
203 ProjVelocity_j -= ProjGridVel;
204 }
205
206 Local_Lambda_i = fabs(ProjVelocity_i)+SoundSpeed_i;
207 Local_Lambda_j = fabs(ProjVelocity_j)+SoundSpeed_j;
208
209 MeanLambda = 0.5*(Local_Lambda_i+Local_Lambda_j);
210
211 Phi_i = pow(Lambda_i/(4.0*MeanLambda), Param_p);
212 Phi_j = pow(Lambda_j/(4.0*MeanLambda), Param_p);
213
214 StretchingFactor = 4.0*Phi_i*Phi_j/(Phi_i+Phi_j);
215
216 sc0 = 3.0*(su2double(Neighbor_i)+su2double(Neighbor_j))/(su2double(Neighbor_i)*su2double(Neighbor_j));
217 Epsilon_0 = Param_Kappa_0*sc0*su2double(nDim)/3.0;
218
219 /*--- Compute viscous part of the residual ---*/
220
221 for (iVar = 0; iVar < nVar; iVar++) {
222 for (jVar = 0; jVar < nVar; jVar++) {
223 ProjFlux[iVar] += Precon[iVar][jVar]*Epsilon_0*Diff_V[jVar]*StretchingFactor*MeanLambda;
224 if (implicit) {
225 Jacobian_i[iVar][jVar] += fix_factor*Precon[iVar][jVar]*Epsilon_0*StretchingFactor*MeanLambda;
226 Jacobian_j[iVar][jVar] -= fix_factor*Precon[iVar][jVar]*Epsilon_0*StretchingFactor*MeanLambda;
227 }
228 }
229 }
230
231 /*--- Remove energy contributions if we aren't solving the energy equation. ---*/
232
233 if (!energy) {
234 ProjFlux[nDim+1] = 0.0;
235 if (implicit) {
236 for (iVar = 0; iVar < nVar; iVar++) {
237 Jacobian_i[iVar][nDim+1] = 0.0;
238 Jacobian_j[iVar][nDim+1] = 0.0;
239
240 Jacobian_i[nDim+1][iVar] = 0.0;
241 Jacobian_j[nDim+1][iVar] = 0.0;
242 }
243 }
244 }
245
246 return ResidualType<>(ProjFlux, Jacobian_i, Jacobian_j);
247
248}

395 Jacobian_i[iDim+1][iDim+1] -= 0.5*ProjVelocity*DensityInc_i;
396 Jacobian_j[iDim+1][iDim+1] -= 0.5*ProjVelocity*DensityInc_j;

398 Jacobian_i[nDim+1][nDim+1] -= 0.5*ProjVelocity*DensityInc_i*Cp_i;
399 Jacobian_j[nDim+1][nDim+1] -= 0.5*ProjVelocity*DensityInc_j*Cp_j;
400 }
401 }
402 }
403
404 /*--- Computes differences between Laplacians and conservative variables ---*/
405
406 for (iVar = 0; iVar < nVar; iVar++) {
407 Diff_Lapl[iVar] = Und_Lapl_i[iVar]-Und_Lapl_j[iVar];
408 Diff_V[iVar] = V_i[iVar]-V_j[iVar];
409 }
410
411 /*--- Build the preconditioning matrix using mean values ---*/
412
413 GetPreconditioner(&MeanDensity, MeanVelocity, &MeanBetaInc2, &MeanCp, &MeanTemperature, &MeandRhodT, Precon);
414
415 /*--- Compute the local spectral radius of the preconditioned system
416 and the stretching factor. ---*/
417
418 /*--- Projected velocity adjustment due to mesh motion ---*/
419
420 if (dynamic_grid) {
421 ProjGridVel = 0.0;
422 for (iDim = 0; iDim < nDim; iDim++) {
423 ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
424 }
425 ProjVelocity_i -= ProjGridVel;
426 ProjVelocity_j -= ProjGridVel;

436
437 Phi_i = pow(Lambda_i/(4.0*MeanLambda), Param_p);
438 Phi_j = pow(Lambda_j/(4.0*MeanLambda), Param_p);
439
440 StretchingFactor = 4.0*Phi_i*Phi_j/(Phi_i+Phi_j);
441
442 sc2 = 3.0*(su2double(Neighbor_i)+su2double(Neighbor_j))/(su2double(Neighbor_i)*su2double(Neighbor_j));
443 sc4 = sc2*sc2/4.0;
444
445 Epsilon_2 = Param_Kappa_2*0.5*(Sensor_i+Sensor_j)*sc2;
446 Epsilon_4 = max(0.0, Param_Kappa_4-Epsilon_2)*sc4;
447
448 /*--- Compute viscous part of the residual ---*/
449
450 for (iVar = 0; iVar < nVar; iVar++) {
451 for (jVar = 0; jVar < nVar; jVar++) {
452 ProjFlux[iVar] += Precon[iVar][jVar]*(Epsilon_2*Diff_V[jVar] - Epsilon_4*Diff_Lapl[jVar])*StretchingFactor*MeanLambda;
453 if (implicit) {
454 Jacobian_i[iVar][jVar] += fix_factor*Precon[iVar][jVar]*(Epsilon_2 + Epsilon_4*su2double(Neighbor_i+1))*StretchingFactor*MeanLambda;
455 Jacobian_j[iVar][jVar] -= fix_factor*Precon[iVar][jVar]*(Epsilon_2 + Epsilon_4*su2double(Neighbor_j+1))*StretchingFactor*MeanLambda;
456 }
457 }
458 }
459
460 /*--- Remove energy contributions if not solving the energy equation. ---*/
461
462 if (!energy) {
463 ProjFlux[nDim+1] = 0.0;
464 if (implicit) {
465 for (iVar = 0; iVar < nVar; iVar++) {
466 Jacobian_i[iVar][nDim+1] = 0.0;
467 Jacobian_j[iVar][nDim+1] = 0.0;
468
469 Jacobian_i[nDim+1][iVar] = 0.0;
470 Jacobian_j[nDim+1][iVar] = 0.0;
471 }
472 }
473 }
474
