37void CDiscAdjFEAIteration::Preprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
38 CSolver***** solver, CNumerics****** numerics, CConfig** config,
39 CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
40 CFreeFormDefBox*** FFDBox, unsigned short iZone, unsigned short iInst) {
41 auto solvers0 = solver[iZone][iInst][MESH_0];
42 auto geometry0 = geometry[iZone][iInst][MESH_0];
43 auto dirNodes = solvers0[FEA_SOL]->GetNodes();
44 auto adjNodes = solvers0[ADJFEA_SOL]->GetNodes();
45
46 auto StoreDirectSolution = [&]() {
47 for (auto iPoint = 0ul; iPoint < geometry0->GetnPoint(); iPoint++) {
48 adjNodes->SetSolution_Direct(iPoint, dirNodes->GetSolution(iPoint));
49 }
50 };
51
52 /*--- For the dynamic adjoint, load direct solutions from restart files.
53 * For steady, store the direct solution to be able to reset it later. ---*/
54
55 if (config[iZone]->GetTime_Domain()) {
56 const int TimeIter = config[iZone]->GetTimeIter();
57 const int Direct_Iter = SU2_TYPE::Int(config[iZone]->GetUnst_AdjointIter()) - TimeIter - 1;
58
59 /*--- We need the already converged solution at timesteps n and n-1. On the first
60 * adjoint time step we load both, then n-1 becomes "n" and we only load n-2. ---*/
61
62 if (TimeIter > 0) {
63 /*--- Save n-1 to become n. ---*/
64 for (auto iPoint = 0ul; iPoint < geometry0->GetnPoint(); iPoint++) {
65 adjNodes->SetSolution_Direct(iPoint, dirNodes->GetSolution_time_n(iPoint));
66 }
67 }
68
69 /*--- Load solution at timestep n-1 and push back to correct array. ---*/
70
71 LoadDynamic_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 1);
72 dirNodes->Set_Solution_time_n();
73
74 /*--- Load or set solution at timestep n. ---*/
75
76 if (TimeIter == 0) {
77 LoadDynamic_Solution(geometry, solver, config, iZone, iInst, Direct_Iter);
78 StoreDirectSolution();
79 } else {
80 /*--- Set n-1 as n. ---*/
81 for (auto iPoint = 0ul; iPoint < geometry0->GetnPoint(); iPoint++)
82 for (auto iVar = 0u; iVar < solvers0[ADJFEA_SOL]->GetnVar(); iVar++)
83 dirNodes->SetSolution(iPoint, iVar, adjNodes->GetSolution_Direct(iPoint)[iVar]);
84 }
85
86 } else {
87 /*--- Steady. ---*/
88 StoreDirectSolution();
89 }
90
91 solvers0[ADJFEA_SOL]->Preprocessing(geometry0, solvers0, config[iZone], MESH_0, 0, RUNTIME_ADJFEA_SYS, false);
92
93}

95void CDiscAdjFEAIteration::LoadDynamic_Solution(CGeometry**** geometry, CSolver***** solver, CConfig** config,
96 unsigned short iZone, unsigned short iInst,
97 int val_DirectIter) {
98 /*--- Set to false to prevent updating Solution_time_n when loading primal solutions of unsteady cases. ---*/
99 const bool update_geo = false;
100 auto*** solvers = solver[iZone][iInst];
101
102 if (val_DirectIter >= 0) {
103 if (rank == MASTER_NODE && iZone == ZONE_0)
104 cout << " Loading FEA solution from direct iteration " << val_DirectIter << ".\n";
105 solvers[MESH_0][FEA_SOL]->LoadRestart(geometry[iZone][iInst], solvers, config[iZone], val_DirectIter, update_geo);
106 } else {
107 /*--- If there is no solution file we set the initial conditions. ---*/
108 if (rank == MASTER_NODE && iZone == ZONE_0)
109 cout << " Setting static conditions at direct iteration " << val_DirectIter << ".\n";
110 solvers[MESH_0][FEA_SOL]->SetInitialCondition(geometry[iZone][iInst], solvers, config[iZone], val_DirectIter);
111 }

146void CDiscAdjFEAIteration::SetDependencies(CSolver***** solver, CGeometry**** geometry, CNumerics****** numerics,
147 CConfig** config, unsigned short iZone, unsigned short iInst,
148 RECORDING kind_recording) {
149 auto dir_solver = solver[iZone][iInst][MESH_0][FEA_SOL];
150 auto adj_solver = solver[iZone][iInst][MESH_0][ADJFEA_SOL];
151 auto structural_geometry = geometry[iZone][iInst][MESH_0];
152 auto structural_numerics = numerics[iZone][iInst][MESH_0][FEA_SOL];
153
154 /*--- Some numerics are only instanciated under these conditions ---*/
155 const bool fsi = config[iZone]->GetFSI_Simulation() || config[iZone]->GetMultizone_Problem();
156 const bool nonlinear = config[iZone]->GetGeometricConditions() == STRUCT_DEFORMATION::LARGE;
157 const bool de_effects = config[iZone]->GetDE_Effects() && nonlinear;
158 const bool element_based = dir_solver->IsElementBased() && nonlinear;
159
160 SU2_OMP_PARALLEL
161 {
162
163 const int thread = omp_get_thread_num();
164 const int offset = thread*MAX_TERMS;
165 const int fea_term = FEA_TERM+offset;
166 const int mat_nhcomp = MAT_NHCOMP+offset;
167 const int mat_idealde = MAT_IDEALDE+offset;
168 const int mat_knowles = MAT_KNOWLES+offset;
169 const int de_term = DE_TERM+offset;
170
171 for (unsigned short iProp = 0; iProp < config[iZone]->GetnElasticityMat(); iProp++) {
172 su2double E = adj_solver->GetVal_Young(iProp);
173 su2double nu = adj_solver->GetVal_Poisson(iProp);
174 su2double rho = adj_solver->GetVal_Rho(iProp);
175 su2double rhoDL = adj_solver->GetVal_Rho_DL(iProp);
176
177 /*--- Add dependencies for E and Nu ---*/
178
179 structural_numerics[fea_term]->SetMaterial_Properties(iProp, E, nu);
180
181 /*--- Add dependencies for Rho and Rho_DL ---*/
182
183 structural_numerics[fea_term]->SetMaterial_Density(iProp, rho, rhoDL);
184
185 /*--- Add dependencies for element-based simulations. ---*/
186
187 if (element_based) {
188 /*--- Neo Hookean Compressible ---*/
189 structural_numerics[mat_nhcomp]->SetMaterial_Properties(iProp, E, nu);
190 structural_numerics[mat_nhcomp]->SetMaterial_Density(iProp, rho, rhoDL);
191
192 /*--- Ideal DE ---*/
193 structural_numerics[mat_idealde]->SetMaterial_Properties(iProp, E, nu);
194 structural_numerics[mat_idealde]->SetMaterial_Density(iProp, rho, rhoDL);
195
196 /*--- Knowles ---*/
197 structural_numerics[mat_knowles]->SetMaterial_Properties(iProp, E, nu);
198 structural_numerics[mat_knowles]->SetMaterial_Density(iProp, rho, rhoDL);
199 }
200 }
201
202 if (de_effects) {
203 for (unsigned short iEField = 0; iEField < adj_solver->GetnEField(); iEField++) {
204 structural_numerics[fea_term]->Set_ElectricField(iEField, adj_solver->GetVal_EField(iEField));
205 structural_numerics[de_term]->Set_ElectricField(iEField, adj_solver->GetVal_EField(iEField));
206 }
207 }
208
209 /*--- Add dependencies for element-based simulations. ---*/
210
211 switch (config[iZone]->GetDV_FEA()) {
212 case YOUNG_MODULUS:
213 case POISSON_RATIO:
214 case DENSITY_VAL:
215 case DEAD_WEIGHT:
216 case ELECTRIC_FIELD:
217
218 for (unsigned short iDV = 0; iDV < adj_solver->GetnDVFEA(); iDV++) {
219 su2double dvfea = adj_solver->GetVal_DVFEA(iDV);
220
221 structural_numerics[fea_term]->Set_DV_Val(iDV, dvfea);
222
223 if (de_effects) structural_numerics[de_term]->Set_DV_Val(iDV, dvfea);
224
225 if (element_based) {
226 structural_numerics[mat_nhcomp]->Set_DV_Val(iDV, dvfea);
227 structural_numerics[mat_idealde]->Set_DV_Val(iDV, dvfea);
228 structural_numerics[mat_knowles]->Set_DV_Val(iDV, dvfea);
229 }
230 }
231 break;
232 }
233
234 /*--- MPI dependencies. ---*/
235
236 dir_solver->InitiateComms(structural_geometry, config[iZone], SOLUTION_FEA);
237 dir_solver->CompleteComms(structural_geometry, config[iZone], SOLUTION_FEA);
238
239 structural_geometry->InitiateComms(structural_geometry, config[iZone], COORDINATES);
240 structural_geometry->CompleteComms(structural_geometry, config[iZone], COORDINATES);
241
242 }
243 END_SU2_OMP_PARALLEL
244
245 /*--- FSI specific dependencies. ---*/
246 if (fsi) {
247 /*--- Set relation between solution and predicted displacements, which are the transferred ones. ---*/
248 dir_solver->PredictStruct_Displacement(structural_geometry, config[iZone]);
249 }
250
251 /*--- Topology optimization dependencies. ---*/
252
253 /*--- We only differentiate wrt to this variable in the adjoint secondary recording. ---*/
254 if (config[iZone]->GetTopology_Optimization() && (kind_recording == RECORDING::MESH_COORDS)) {
255 /*--- The filter may require the volumes of the elements. ---*/
256 structural_geometry->SetElemVolume();
257 /// TODO: Ideally there would be a way to capture this dependency without the `static_cast`, but
258 /// making it a virtual method of CSolver does not feel "right" as its purpose could be confused.
259 static_cast<CFEASolver*>(dir_solver)->FilterElementDensities(structural_geometry, config[iZone]);
260 }
261
262}

294void CDiscAdjFEAIteration::Postprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
295 CSolver***** solver, CNumerics****** numerics, CConfig** config,
296 CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
297 CFreeFormDefBox*** FFDBox, unsigned short iZone, unsigned short iInst) {
298 auto solvers0 = solver[iZone][iInst][MESH_0];
299
300 // TEST: for implementation of python framework in standalone structural problems
301 if (config[iZone]->GetAdvanced_FEAElementBased() && (rank == MASTER_NODE)) {
302 /*--- Header of the temporary output file ---*/
303 ofstream myfile_res;
304 bool outputDVFEA = true;
305
306 switch (config[iZone]->GetDV_FEA()) {
307 case YOUNG_MODULUS:
308 myfile_res.open("grad_young.opt");
309 break;
310 case POISSON_RATIO:
311 myfile_res.open("grad_poisson.opt");
312 break;
313 case DENSITY_VAL:
314 case DEAD_WEIGHT:
315 myfile_res.open("grad_density.opt");
316 break;
317 case ELECTRIC_FIELD:
318 myfile_res.open("grad_efield.opt");
319 break;
320 default:
321 outputDVFEA = false;
322 break;
323 }
324
325 if (outputDVFEA) {
326 const auto nDV = solvers0[ADJFEA_SOL]->GetnDVFEA();
327 myfile_res << "INDEX\tGRAD\n";
328 myfile_res.precision(15);
329
330 for (auto iDV = 0u; iDV < nDV; iDV++) {
331 myfile_res << iDV << "\t";
332 myfile_res << scientific << solvers0[ADJFEA_SOL]->GetTotal_Sens_DVFEA(iDV) << "\n";
333 }
334 }
335 }
