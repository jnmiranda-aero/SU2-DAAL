31CTETRA1::CTETRA1() : CElementWithKnownSizes<NGAUSS, NNODE, NDIM>() {
32 /*--- Gauss coordinates and weights ---*/
33
34 GaussCoord[0][0] = 0.25;
35 GaussCoord[0][1] = 0.25;
36 GaussCoord[0][2] = 0.25;
37 GaussWeight(0) = 1.0 / 6.0;
38
39 /*--- Store the values of the shape functions and their derivatives ---*/
40
41 unsigned short iGauss;
42 su2double Xi, Eta, Zeta, val_Ni;
43
44 for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
45 Xi = GaussCoord[iGauss][0];
46 Eta = GaussCoord[iGauss][1];
47 Zeta = GaussCoord[iGauss][2];
48
49 val_Ni = Xi;
50 GaussPoint[iGauss].SetNi(val_Ni, 0);
51 val_Ni = Eta;
52 GaussPoint[iGauss].SetNi(val_Ni, 1);
53 val_Ni = 1.0 - Xi - Eta - Zeta;
54 GaussPoint[iGauss].SetNi(val_Ni, 2);
55 val_Ni = Zeta;
56 GaussPoint[iGauss].SetNi(val_Ni, 3);
57
58 /*--- dN/d xi, dN/d eta, dN/d zeta ---*/
59
60 dNiXj[iGauss][0][0] = 1.0;
61 dNiXj[iGauss][0][1] = 0.0;
62 dNiXj[iGauss][0][2] = 0.0;
63 dNiXj[iGauss][1][0] = 0.0;
64 dNiXj[iGauss][1][1] = 1.0;
65 dNiXj[iGauss][1][2] = 0.0;
66 dNiXj[iGauss][2][0] = -1.0;
67 dNiXj[iGauss][2][1] = -1.0;
68 dNiXj[iGauss][2][2] = -1.0;
69 dNiXj[iGauss][3][0] = 0.0;
70 dNiXj[iGauss][3][1] = 0.0;
71 dNiXj[iGauss][3][2] = 1.0;
72 }
73
74 /*--- Shape functions evaluated at the nodes for extrapolation of the stresses at the Gaussian Points ---*/
75 /*--- The stress is constant at a TETRA1 element ---*/
76
77 NodalExtrap[0][0] = 1.0;
78 NodalExtrap[1][0] = 1.0;
79 NodalExtrap[2][0] = 1.0;
80 NodalExtrap[3][0] = 1.0;
81}

83su2double CTETRA1::ComputeVolume(const FrameType mode) const {
84 unsigned short iDim;
85 su2double r1[3] = {0.0, 0.0, 0.0}, r2[3] = {0.0, 0.0, 0.0}, r3[3] = {0.0, 0.0, 0.0},
86 CrossProduct[3] = {0.0, 0.0, 0.0};
87 su2double Volume = 0.0;
88
89 /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
90 for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed)---*/
91 const su2activematrix& Coord = (mode == REFERENCE) ? RefCoord : CurrentCoord;
92
93 for (iDim = 0; iDim < NDIM; iDim++) {
94 r1[iDim] = Coord[1][iDim] - Coord[0][iDim];
95 r2[iDim] = Coord[2][iDim] - Coord[0][iDim];
96 r3[iDim] = Coord[3][iDim] - Coord[0][iDim];
97 }
98
99 GeometryToolbox::CrossProduct(r1, r2, CrossProduct);
100 Volume = fabs(GeometryToolbox::DotProduct(3, CrossProduct, r3)) / 6.0;
101
102 return Volume;
103}
