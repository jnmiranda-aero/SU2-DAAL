35CIntegration** CIntegrationFactory::CreateIntegrationContainer(MAIN_SOLVER kindMainSolver,
36 const CSolver* const* solver_container){
37
38 auto **integration = new CIntegration* [MAX_SOLS]();
39
40 for (unsigned int iSol = 0; iSol < MAX_SOLS; iSol++){
41 if (solver_container[iSol] != nullptr){
42 const SolverMetaData &solverInfo = CSolverFactory::GetSolverMeta(solver_container[iSol]);
43 integration[iSol] = CreateIntegration(solverInfo.integrationType);
44 }
45 }
46
47 return integration;
48}

50CIntegration* CIntegrationFactory::CreateIntegration(INTEGRATION_TYPE integrationType){
51
52 CIntegration *integration = nullptr;
53
54 switch(integrationType){
55 case INTEGRATION_TYPE::DEFAULT:
56 integration = new CIntegration();
57 break;
58 case INTEGRATION_TYPE::SINGLEGRID:
59 integration = new CSingleGridIntegration();
60 break;
61 case INTEGRATION_TYPE::MULTIGRID:
62 integration = new CMultiGridIntegration();
63 break;
64 case INTEGRATION_TYPE::NEWTON:
65 integration = new CNewtonIntegration();
66 break;
67 case INTEGRATION_TYPE::STRUCTURAL:
68 integration = new CStructuralIntegration();
69 break;
70 case INTEGRATION_TYPE::FEM_DG:
71 integration = new CFEM_DG_Integration();
72 break;
73 case INTEGRATION_TYPE::NONE:
74 integration = nullptr;
75 break;
76 }
77
78 return integration;
