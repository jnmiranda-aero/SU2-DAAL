32CRinglebSolution::CRinglebSolution(unsigned short val_nDim, unsigned short val_nVar, unsigned short val_iMesh,
33 CConfig* config)
34 : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
35 /*--- Write a message that the solution is initialized for the
36 Ringleb test case. ---*/
37 if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
38 cout << endl;
39 cout << "Warning: Fluid properties and solution are being " << endl;
40 cout << " initialized for the Ringleb case!!!" << endl;
41 cout << endl << flush;
42 }
43
44 /*--- Useful coefficients in which Gamma is present. ---*/
45 Gamma = config->GetGamma();
46 Gm1 = Gamma - 1.0;
47 tovGm1 = 2.0 / Gm1;
48 tGamOvGm1 = Gamma * tovGm1;
49
50 /*--- Perform some sanity and error checks for this solution here. ---*/
51 if (config->GetTime_Marching() != TIME_MARCHING::STEADY)
52 SU2_MPI::Error("Steady mode must be selected for the Ringleb case", CURRENT_FUNCTION);
53
54 if (Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES &&
55 Kind_Solver != MAIN_SOLVER::RANS && Kind_Solver != MAIN_SOLVER::FEM_EULER &&
56 Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
57 Kind_Solver != MAIN_SOLVER::FEM_LES)
58 SU2_MPI::Error("Compressible flow equations must be selected for the Ringleb case", CURRENT_FUNCTION);
59
60 if ((Kind_Solver != MAIN_SOLVER::EULER) && (Kind_Solver != MAIN_SOLVER::FEM_EULER))
61 SU2_MPI::Error("Euler equations must be selected for the Ringleb case", CURRENT_FUNCTION);
62
63 if ((config->GetKind_FluidModel() != STANDARD_AIR) && (config->GetKind_FluidModel() != IDEAL_GAS))
64 SU2_MPI::Error("Standard air or ideal gas must be selected for the Ringleb case", CURRENT_FUNCTION);
65}

77void CRinglebSolution::GetSolution(const su2double* val_coords, const su2double val_t, su2double* val_solution) const {
78 /* Easier storage of the coordinates and abbreviate y*y. */
79 const su2double x = val_coords[0], y = val_coords[1];
80 const su2double y2 = y * y;
81
82 /* Initial guess for q (velocity magnitude) and k (streamline parameter). */
83 su2double k = 1.2;
84 su2double q = 1.0;
85
86 /* Newton algorithm to solve for the variables q and k for the given x and y. */
87 const int iterMax = 500;
88 su2double duMaxPrev = 10.0;
89
90 int iter;
91 for (iter = 0; iter < iterMax; ++iter) {
92 /* Compute the speed of sound, the density, the parameter JJ
93 and its derivatives w.r.t. q. */
94 const su2double a = sqrt(1.0 - 0.5 * Gm1 * q * q);
95 const su2double rho = pow(a, tovGm1);
96 const su2double JJ =
97 1.0 / a + 1.0 / (3.0 * a * a * a) + 1.0 / (5.0 * a * a * a * a * a) - 0.5 * log((1.0 + a) / (1.0 - a));
98
99 const su2double dadq = -0.5 * Gm1 * q / a;
100 const su2double drhodq = 2.0 * rho * dadq / (Gm1 * a);
101 const su2double dJJdq = dadq / (pow(a, 6) * (a * a - 1.0));
102
103 /* Determine the values of the nonlinear equations to solve
104 and its corresponding Jacobian matrix. */
105 const su2double y2c = (k * k - q * q) / (k * k * k * k * rho * rho * q * q);
106 const su2double f[] = {(2.0 / (k * k) - 1.0 / (q * q)) / (2.0 * rho) - 0.5 * JJ - x, y2c - y2};
107 su2double Jac[2][2];
108 Jac[0][0] = -(1.0 / (k * k) - 0.50 / (q * q)) * drhodq / (rho * rho) + 1.0 / (rho * q * q * q) - 0.5 * dJJdq;
109 Jac[0][1] = -2.0 / (rho * k * k * k);
110 Jac[1][0] = -2.0 / (k * k * rho * rho * q * q * q) - 2.0 * y2c * drhodq / rho;
111 Jac[1][1] = (4.0 * q * q - 2.0 * k * k) / (k * k * k * k * k * rho * rho * q * q);
112
113 /* Determine the update dU. */
114 const su2double det = Jac[0][0] * Jac[1][1] - Jac[0][1] * Jac[1][0];
115 const su2double dU[] = {(f[0] * Jac[1][1] - f[1] * Jac[0][1]) / det, (f[1] * Jac[0][0] - f[0] * Jac[1][0]) / det};
116
117 /* Determine the underrelaxation coefficient alp. */
118 const su2double dUMax = max(fabs(dU[0]), fabs(dU[1]));
119 su2double alp = 1.0;
120 if (dUMax > 1.0)
121 alp = 0.04;
122 else if (dUMax > 0.1)
123 alp = 0.2;
124
125 /* Update q and k. */
126 q -= alp * dU[0];
127 k -= alp * dU[1];
128
129 /* Convergence check, which is independent of the precision used. */
130 if ((dUMax < 1.e-3) && (dUMax >= duMaxPrev)) break;
131 duMaxPrev = dUMax;
132 }
133
134 /* Check if the Newton algorithm actually converged. */
135 if (iter == iterMax) SU2_MPI::Error("Newton algorithm did not converge", CURRENT_FUNCTION);
136
137 /* Compute the speed of sound, density and pressure. */
138 const su2double a = sqrt(1.0 - 0.5 * Gm1 * q * q);
139 const su2double rho = pow(a, tovGm1);
140 const su2double p = pow(a, tGamOvGm1) / Gamma;
141
142 /* Determine the derivative of x w.r.t. q and ydxdq. */
143 const su2double dadq = -0.5 * Gm1 * q / a;
144 const su2double drhodq = 2.0 * rho * dadq / (Gm1 * a);
145 const su2double dJJdq = dadq / (pow(a, 6) * (a * a - 1.0));
146
147 const su2double dxdq =
148 -(1.0 / (k * k) - 0.5 / (q * q)) * drhodq / (rho * rho) + 1.0 / (rho * q * q * q) - 0.5 * dJJdq;
149 const su2double ydxdq = y * dxdq;
150
151 /* Determine the derivative of 1/2 y2 w.r.t. q, which is ydydq. The reason is
152 that ydydq is always well defined, while dydq is singular for y = 0. */
153 const su2double ydydq =
154 drhodq * (q * q - k * k) / (k * k * k * k * rho * rho * rho * q * q) - 1.0 / (k * k * rho * rho * q * q * q);
155
156 /* Determine the direction of the streamline. */
157 const su2double vecLen = sqrt(ydxdq * ydxdq + ydydq * ydydq);
158
159 su2double velDir[] = {ydxdq / vecLen, ydydq / vecLen};
160 if (velDir[1] > 0.0) {
161 velDir[0] = -velDir[0];
162 velDir[1] = -velDir[1];
163 }
164
165 /* Compute the conservative variables. Note that both 2D and 3D
166 cases are treated correctly. */
167 val_solution[0] = rho;
168 val_solution[1] = rho * q * velDir[0];
169 val_solution[2] = rho * q * velDir[1];
170 val_solution[3] = 0.0;
171 val_solution[nVar - 1] = p / Gm1 + 0.5 * rho * q * q;
172}
