53 void CTurbSAVariable::SetVortex_Tilting(unsigned long iPoint, CMatrixView<const su2double> PrimGrad_Flow,
54  const su2double* Vorticity, su2double LaminarViscosity) {
55 
56  su2double Strain[3][3] = {{0,0,0}, {0,0,0}, {0,0,0}}, Omega, StrainDotVort[3], numVecVort[3];
57  su2double numerator, trace0, trace1, denominator;
58 
59  AD::StartPreacc();
60  AD::SetPreaccIn(PrimGrad_Flow, nDim+1, nDim);
61  AD::SetPreaccIn(Vorticity, 3);
62  /*--- Eddy viscosity ---*/
63  AD::SetPreaccIn(muT(iPoint));
64  /*--- Laminar viscosity --- */
65  AD::SetPreaccIn(LaminarViscosity);
66 
67  Strain[0][0] = PrimGrad_Flow[1][0];
68  Strain[1][0] = 0.5*(PrimGrad_Flow[2][0] + PrimGrad_Flow[1][1]);
69  Strain[0][1] = 0.5*(PrimGrad_Flow[1][1] + PrimGrad_Flow[2][0]);
70  Strain[1][1] = PrimGrad_Flow[2][1];
71  if (nDim == 3){
72  Strain[0][2] = 0.5*(PrimGrad_Flow[3][0] + PrimGrad_Flow[1][2]);
73  Strain[1][2] = 0.5*(PrimGrad_Flow[3][1] + PrimGrad_Flow[2][2]);
74  Strain[2][0] = 0.5*(PrimGrad_Flow[1][2] + PrimGrad_Flow[3][0]);
75  Strain[2][1] = 0.5*(PrimGrad_Flow[2][2] + PrimGrad_Flow[3][1]);
76  Strain[2][2] = PrimGrad_Flow[3][2];
77  }
78 
79  Omega = sqrt(Vorticity[0]*Vorticity[0] + Vorticity[1]*Vorticity[1]+ Vorticity[2]*Vorticity[2]);
80 
81  StrainDotVort[0] = Strain[0][0]*Vorticity[0]+Strain[0][1]*Vorticity[1]+Strain[0][2]*Vorticity[2];
82  StrainDotVort[1] = Strain[1][0]*Vorticity[0]+Strain[1][1]*Vorticity[1]+Strain[1][2]*Vorticity[2];
83  StrainDotVort[2] = Strain[2][0]*Vorticity[0]+Strain[2][1]*Vorticity[1]+Strain[2][2]*Vorticity[2];
84 
85  numVecVort[0] = StrainDotVort[1]*Vorticity[2] - StrainDotVort[2]*Vorticity[1];
86  numVecVort[1] = StrainDotVort[2]*Vorticity[0] - StrainDotVort[0]*Vorticity[2];
87  numVecVort[2] = StrainDotVort[0]*Vorticity[1] - StrainDotVort[1]*Vorticity[0];
88 
89  numerator = sqrt(6.0) * sqrt(numVecVort[0]*numVecVort[0] + numVecVort[1]*numVecVort[1] + numVecVort[2]*numVecVort[2]);
90  trace0 = 3.0*(pow(Strain[0][0],2.0) + pow(Strain[1][1],2.0) + pow(Strain[2][2],2.0));
91  trace1 = pow(Strain[0][0] + Strain[1][1] + Strain[2][2],2.0);
92  denominator = pow(Omega, 2.0) * sqrt(trace0-trace1);
93 
94  Vortex_Tilting(iPoint) = (numerator/denominator) * max(1.0,0.2*LaminarViscosity/muT(iPoint));
95 
96  AD::SetPreaccOut(Vortex_Tilting(iPoint));
97  AD::EndPreacc();
98 }
