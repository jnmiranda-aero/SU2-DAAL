43 void CFlowTractionInterface::Preprocess(const CConfig *flow_config, const CConfig *struct_config,
44  CGeometry *struct_geometry, CSolver *struct_solution) {
45 
46  /*--- Clear the tractions only on the markers involved in interface, fluid tractions
47  * on other markers can be specified via e.g. the python wrapper. ---*/
48 
49  for (auto iMarkerInt = 0u; iMarkerInt < struct_config->GetMarker_n_ZoneInterface()/2; iMarkerInt++) {
50  const auto markDonor = flow_config->FindInterfaceMarker(iMarkerInt);
51  const auto markTarget = struct_config->FindInterfaceMarker(iMarkerInt);
52 
53  if(!CInterpolator::CheckInterfaceBoundary(markDonor, markTarget)) continue;
54  if (markTarget < 0) continue;
55 
56  for (auto iVertex = 0ul; iVertex < struct_geometry->GetnVertex(markTarget); iVertex++) {
57  const auto iPoint = struct_geometry->vertex[markTarget][iVertex]->GetNode();
58  if (!struct_geometry->nodes->GetDomain(iPoint)) continue;
59  su2double zeros[3] = {};
60  struct_solution->GetNodes()->Set_FlowTraction(iPoint, zeros);
61  }
62  }
63 
64  /*--- Compute the constant factor to dimensionalize pressure and shear stress. ---*/
65  const su2double *Velocity_ND, *Velocity_Real;
66  su2double Density_ND, Density_Real, Velocity2_Real, Velocity2_ND;
67 
68  Velocity_Real = flow_config->GetVelocity_FreeStream();
69  Density_Real = flow_config->GetDensity_FreeStream();
70 
71  Velocity_ND = flow_config->GetVelocity_FreeStreamND();
72  Density_ND = flow_config->GetDensity_FreeStreamND();
73 
74  Velocity2_Real = 0.0;
75  Velocity2_ND = 0.0;
76  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
77  Velocity2_Real += Velocity_Real[iVar]*Velocity_Real[iVar];
78  Velocity2_ND += Velocity_ND[iVar]*Velocity_ND[iVar];
79  }
80 
81  Physical_Constants[0] = Density_Real * Velocity2_Real / ( Density_ND * Velocity2_ND );
82 }

84 void CFlowTractionInterface::ComputeVertexAreas(const CConfig *config,
85  CGeometry *geometry,
86  CSolver *solution) {
87  constexpr size_t MAXNDIM = 3;
88  const auto nodes = solution->GetNodes();
89  const auto nDim = nVar;
90 
91  /*--- Compute current area associated with each vertex. ---*/
92 
93  vertexArea.clear();
94  unordered_set<short> processedMarkers({-1});
95 
96  for (auto iMarkerInt = 0; iMarkerInt < config->GetMarker_n_ZoneInterface()/2; ++iMarkerInt) {
97  /*--- Find the marker index associated with the pair. ---*/
98  const auto iMarker = config->FindInterfaceMarker(iMarkerInt);
99  /*--- The current mpi rank may not have this marker, or it may have been processed already. ---*/
100  if (processedMarkers.count(iMarker) > 0) continue;
101  processedMarkers.insert(iMarker);
102 
103  for (auto iElem = 0u; iElem < geometry->GetnElem_Bound(iMarker); ++iElem) {
104  /*--- Define the boundary element. ---*/
105  unsigned long nodeList[N_POINTS_MAXIMUM] = {0};
106  su2double coords[N_POINTS_MAXIMUM][MAXNDIM] = {{0.0}};
107  bool quad = geometry->bound[iMarker][iElem]->GetVTK_Type() == QUADRILATERAL;
108  auto nNode = quad? 4u : nDim;
109 
110  for (auto iNode = 0u; iNode < nNode; ++iNode) {
111  nodeList[iNode] = geometry->bound[iMarker][iElem]->GetNode(iNode);
112  for (auto iDim = 0u; iDim < nDim; ++iDim)
113  coords[iNode][iDim] = geometry->nodes->GetCoord(nodeList[iNode], iDim)+
114  nodes->GetSolution(nodeList[iNode],iDim);
115  }
116 
117  /*--- Compute the area contribution to each node. ---*/
118  su2double normal[MAXNDIM] = {0.0};
119 
120  switch (nNode) {
121  case 2: GeometryToolbox::LineNormal(coords, normal); break;
122  case 3: GeometryToolbox::TriangleNormal(coords, normal); break;
123  case 4: GeometryToolbox::QuadrilateralNormal(coords, normal); break;
124  }
125 
126  su2double area = GeometryToolbox::Norm(MAXNDIM, normal) / nNode;
127 
128  /*--- Update area of nodes. ---*/
129  for (auto iNode = 0u; iNode < nNode; ++iNode) {
130  auto iPoint = nodeList[iNode];
131  if (vertexArea.count(iPoint) == 0)
132  vertexArea[iPoint] = area;
133  else
134  vertexArea[iPoint] += area;
135  }
136  }
137  }
138 }

140 void CFlowTractionInterface::GetPhysical_Constants(CSolver *flow_solution, CSolver *struct_solution,
141  CGeometry *flow_geometry, CGeometry *struct_geometry,
142  const CConfig *flow_config, const CConfig *struct_config) {
143 
144  Preprocess(flow_config, struct_config, struct_geometry, struct_solution);
145 
146  if (!conservative) ComputeVertexAreas(struct_config, struct_geometry, struct_solution);
147 
148  /*--- Apply a ramp to the transfer of the fluid loads ---*/
149 
150  su2double ModAmpl = 0.0;
151  su2double CurrentTime = struct_config->GetCurrent_UnstTime();
152 
153  bool Ramp_Load = struct_config->GetRamp_Load();
154  su2double Ramp_Time = struct_config->GetRamp_Time();
155 
156  ModAmpl = struct_solution->Compute_LoadCoefficient(CurrentTime, Ramp_Time, struct_config);
157 
158  Physical_Constants[1] = ModAmpl;
159 
160  /*--- For static FSI, we cannot apply the ramp like this ---*/
161  if ((!flow_config->GetTime_Domain())){
162  Physical_Constants[1] = 1.0;
163  if (Ramp_Load){
164  CurrentTime = static_cast<su2double>(struct_config->GetOuterIter());
165  Ramp_Time = static_cast<su2double>(struct_config->GetnIterFSI_Ramp() - 1);
166 
167  ModAmpl = struct_solution->Compute_LoadCoefficient(CurrentTime, Ramp_Time, struct_config);
168  Physical_Constants[1] = ModAmpl;
169  }
170  }
171 
172  /*--- Store the force coefficient ---*/
173  struct_solution->SetForceCoeff(Physical_Constants[1]);
174 
175 }

177 void CFlowTractionInterface::GetDonor_Variable(CSolver *flow_solution, CGeometry *flow_geometry,
178  const CConfig *flow_config, unsigned long Marker_Flow,
179  unsigned long Vertex_Flow, unsigned long Point_Struct) {
180 
181  const auto Point_Flow = flow_geometry->vertex[Marker_Flow][Vertex_Flow]->GetNode();
182 
183  // Get the normal at the vertex: this normal goes inside the fluid domain.
184  const su2double* Normal_Flow = flow_geometry->vertex[Marker_Flow][Vertex_Flow]->GetNormal();
185 
186  // If we do not want integrated tractions, i.e. forces, we will need to divide by area.
187  su2double oneOnArea = 1.0;
188  if (!conservative) oneOnArea = 1.0 / GeometryToolbox::Norm(nVar, Normal_Flow);
189 
190  // Retrieve the values of pressure
191 
192  CVariable* flow_nodes = flow_solution->GetNodes();
193  su2double Pinf = flow_solution->GetPressure_Inf();
194  su2double Pn = flow_nodes->GetPressure(Point_Flow);
195 
196  // Calculate tn in the fluid nodes for the inviscid term --> Units of force (non-dimensional).
197 
198  for (auto iVar = 0u; iVar < nVar; iVar++)
199  Donor_Variable[iVar] = -(Pn-Pinf)*Normal_Flow[iVar];
200 
201  // Calculate tn in the fluid nodes for the viscous term
202 
203  if (flow_config->GetViscous()) {
204 
205  su2double Viscosity = flow_nodes->GetLaminarViscosity(Point_Flow);
206 
207  su2double tau[3][3];
208  CNumerics::ComputeStressTensor(nVar, tau, flow_nodes->GetVelocityGradient(Point_Flow), Viscosity);
209  for (auto iVar = 0u; iVar < nVar; iVar++) {
210  for (auto jVar = 0u; jVar < nVar; jVar++) {
211  // Viscous component in the tn vector --> Units of force (non-dimensional).
212  Donor_Variable[iVar] += tau[iVar][jVar] * Normal_Flow[jVar];
213  }
214  }
215 
216  }
217 
218  // Redimensionalize and take into account ramp transfer of the loads
219  for (auto iVar = 0u; iVar < nVar; iVar++) {
220  Donor_Variable[iVar] *= Physical_Constants[0] * Physical_Constants[1] * oneOnArea;
221  }
222 
223 }

225 void CFlowTractionInterface::SetTarget_Variable(CSolver *fea_solution, CGeometry *fea_geometry,
226  const CConfig *fea_config, unsigned long Marker_Struct,
227  unsigned long Vertex_Struct, unsigned long Point_Struct) {
228 
229  /*--- Add the Flow traction, if consistent interpolation is in use the traction needs to be integrated. ---*/
230  if (!conservative) {
231  assert(vertexArea.count(Point_Struct));
232  const auto area = vertexArea.at(Point_Struct);
233  for (auto iVar = 0u; iVar < nVar; ++iVar) Target_Variable[iVar] *= area;
234  }
235 
236  fea_solution->GetNodes()->Add_FlowTraction(Point_Struct, Target_Variable);
237 }
