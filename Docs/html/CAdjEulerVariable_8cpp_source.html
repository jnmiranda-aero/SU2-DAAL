31CAdjEulerVariable::CAdjEulerVariable(su2double psirho, const su2double *phi, su2double psie, unsigned long npoint, unsigned long ndim,
32 unsigned long nvar, CConfig *config) : CVariable(npoint, ndim, nvar, config),
33 Gradient_Reconstruction(config->GetReconstructionGradientRequired() ? Gradient_Aux : Gradient) {
34
35 bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
36 (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
37
38 /*--- Allocate residual structures ---*/
39 Res_TruncError.resize(nPoint,nVar) = su2double(0.0);
40
41 /*--- Only for residual smoothing (multigrid) ---*/
42 for (unsigned long iMesh = 0; iMesh <= config->GetnMGLevels(); iMesh++) {
43 if (config->GetMG_CorrecSmooth(iMesh) > 0) {
44 Residual_Sum.resize(nPoint,nVar);
45 Residual_Old.resize(nPoint,nVar);
46 break;
47 }
48 }
49
50 Gradient.resize(nPoint,nVar,nDim,0.0);
51
52 if (config->GetReconstructionGradientRequired()) {
53 Gradient_Aux.resize(nPoint,nVar,nDim,0.0);
54 }
55
56 if (config->GetLeastSquaresRequired()) {
57 Rmatrix.resize(nPoint,nDim,nDim,0.0);
58 }
59
60 /*--- Allocate undivided laplacian (centered) and limiter (upwind)---*/
61 if (config->GetKind_ConvNumScheme_AdjFlow() == SPACE_CENTERED)
62 Undivided_Laplacian.resize(nPoint,nVar);
63
64 /*--- Always allocate the slope limiter,
65 and the auxiliar variables (check the logic - JST with 2nd order Turb model - ) ---*/
66 Limiter.resize(nPoint,nVar) = su2double(0.0);
67 Solution_Max.resize(nPoint,nVar) = su2double(0.0);
68 Solution_Min.resize(nPoint,nVar) = su2double(0.0);
69
70 /*--- Solution initialization ---*/
71 su2double val_solution[5] = {psirho, phi[0], phi[1], psie, psie};
72 if(nDim==3) val_solution[3] = phi[2];
73
74 for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint)
75 for (unsigned long iVar = 0; iVar < nVar; iVar++)
76 Solution(iPoint,iVar) = val_solution[iVar];
77
78 Solution_Old = Solution;
79
80 /*--- Allocate and initializate solution for dual time strategy ---*/
81 if (dual_time) {
82 Solution_time_n = Solution;
83 Solution_time_n1 = Solution;
84 }
85
86 /*--- Allocate auxiliar vector for sensitivity computation ---*/
87 nAuxVar = 1;
88 AuxVar.resize(nPoint, nAuxVar) = su2double(0.0);
89 Grad_AuxVar.resize(nPoint, nAuxVar, nDim);
90
91 /*--- Allocate and initializate projection vector for wall boundary condition ---*/
92 ForceProj_Vector.resize(nPoint,nDim) = su2double(0.0);
93
94 /*--- Allocate space for the harmonic balance source terms ---*/
95 if (config->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE)
96 HB_Source.resize(nPoint,nVar) = su2double(0.0);
97
98 if (config->GetMultizone_Problem())
99 Set_BGSSolution_k();
100
101 Sensor.resize(nPoint);
102
103 /* Non-physical point (first-order) initialization. */
104 Non_Physical.resize(nPoint) = false;
105 Non_Physical_Counter.resize(nPoint) = 0;
106
107}

109bool CAdjEulerVariable::SetPrimVar(unsigned long iPoint, su2double SharpEdge_Distance, bool check, CConfig *config) {
110
111 bool RightVol = true;
112
113 su2double adj_limit = config->GetAdjointLimit();
114
115 bool check_dens = (fabs(Solution(iPoint,0)) > adj_limit);
116
117 /*--- Check that the adjoint solution is bounded ---*/
118
119 if (check_dens) {
120
121 /*--- Copy the old solution ---*/
122
123 for (unsigned long iVar = 0; iVar < nVar; iVar++)
124 Solution(iPoint,iVar) = Solution_Old(iPoint,iVar);
125
126 RightVol = false;
127
128 }
129
130 return RightVol;
