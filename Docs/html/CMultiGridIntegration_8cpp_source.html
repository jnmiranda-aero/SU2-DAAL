34void CMultiGridIntegration::MultiGrid_Iteration(CGeometry ****geometry,
35 CSolver *****solver_container,
36 CNumerics ******numerics_container,
37 CConfig **config,
38 unsigned short RunTime_EqSystem,
39 unsigned short iZone,
40 unsigned short iInst) {
41
42 bool direct;
43 switch (config[iZone]->GetKind_Solver()) {
44 case MAIN_SOLVER::EULER:
45 case MAIN_SOLVER::NAVIER_STOKES:
46 case MAIN_SOLVER::NEMO_EULER:
47 case MAIN_SOLVER::NEMO_NAVIER_STOKES:
48 case MAIN_SOLVER::RANS:
49 case MAIN_SOLVER::FEM_EULER:
50 case MAIN_SOLVER::FEM_NAVIER_STOKES:
51 case MAIN_SOLVER::FEM_RANS:
52 case MAIN_SOLVER::FEM_LES:
53 case MAIN_SOLVER::DISC_ADJ_EULER:
54 case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES:
55 case MAIN_SOLVER::DISC_ADJ_FEM_EULER:
56 case MAIN_SOLVER::DISC_ADJ_FEM_NS:
57 case MAIN_SOLVER::DISC_ADJ_RANS:
58 direct = true;
59 break;
60 default:
61 direct = false;
62 break;
63 }
64
65 const unsigned short Solver_Position = config[iZone]->GetContainerPosition(RunTime_EqSystem);
66
67 /*--- Start an OpenMP parallel region covering the entire MG iteration, if the solver supports it. ---*/
68
69 SU2_OMP_PARALLEL_(if(solver_container[iZone][iInst][MESH_0][Solver_Position]->GetHasHybridParallel()))
70 {
71
72 su2double monitor = 1.0;
73 bool FullMG = false;
74
75 unsigned short RecursiveParam = config[iZone]->GetMGCycle();
76
77 if (config[iZone]->GetMGCycle() == FULLMG_CYCLE) {
78 RecursiveParam = V_CYCLE;
79 FullMG = true;
80 }
81
82 /*--- Full multigrid strategy and start up with fine grid only works with the direct problem ---*/
83
84 unsigned short FinestMesh = config[iZone]->GetFinestMesh();
85
86 /// TODO: This was always false.
87 const bool Convergence_FullMG = false;
88
89 if (!config[iZone]->GetRestart() && FullMG && direct && ( Convergence_FullMG && (FinestMesh != MESH_0 ))) {
90
91 SetProlongated_Solution(RunTime_EqSystem,
92 solver_container[iZone][iInst][FinestMesh-1][Solver_Position],
93 solver_container[iZone][iInst][FinestMesh][Solver_Position],
94 geometry[iZone][iInst][FinestMesh-1],
95 geometry[iZone][iInst][FinestMesh],
96 config[iZone]);
97
98 SU2_OMP_SAFE_GLOBAL_ACCESS(config[iZone]->SubtractFinestMesh();)
99 }
100
101 /*--- Set the current finest grid (full multigrid strategy) ---*/
102
103 FinestMesh = config[iZone]->GetFinestMesh();
104
105 /*--- Perform the Full Approximation Scheme multigrid ---*/
106
107 MultiGrid_Cycle(geometry, solver_container, numerics_container, config,
108 FinestMesh, RecursiveParam, RunTime_EqSystem, iZone, iInst);
109
110
111 /*--- Computes primitive variables and gradients in the finest mesh (useful for the next solver (turbulence) and output ---*/
112
113 solver_container[iZone][iInst][MESH_0][Solver_Position]->Preprocessing(geometry[iZone][iInst][MESH_0],
114 solver_container[iZone][iInst][MESH_0],
115 config[iZone], MESH_0, NO_RK_ITER,
116 RunTime_EqSystem, true);
117
118 /*--- Compute non-dimensional parameters and the convergence monitor ---*/
119
120 NonDimensional_Parameters(geometry[iZone][iInst], solver_container[iZone][iInst],
121 numerics_container[iZone][iInst], config[iZone],
122 FinestMesh, RunTime_EqSystem, &monitor);
123
124 }
125 END_SU2_OMP_PARALLEL
126
127}

129void CMultiGridIntegration::MultiGrid_Cycle(CGeometry ****geometry,
130 CSolver *****solver_container,
131 CNumerics ******numerics_container,
132 CConfig **config_container,
133 unsigned short iMesh,
134 unsigned short RecursiveParam,
135 unsigned short RunTime_EqSystem,
136 unsigned short iZone,
137 unsigned short iInst) {
138
139 CConfig* config = config_container[iZone];
140
141 const unsigned short Solver_Position = config->GetContainerPosition(RunTime_EqSystem);
142 const bool classical_rk4 = (config->GetKind_TimeIntScheme() == CLASSICAL_RK4_EXPLICIT);
143 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
144
145 /*--- Shorter names to refer to fine grid entities. ---*/
146
147 CGeometry* geometry_fine = geometry[iZone][iInst][iMesh];
148 CSolver** solver_container_fine = solver_container[iZone][iInst][iMesh];
149 CSolver* solver_fine = solver_container_fine[Solver_Position];
150 CNumerics** numerics_fine = numerics_container[iZone][iInst][iMesh][Solver_Position];
151
152 /*--- Number of RK steps. ---*/
153
154 unsigned short iRKLimit = 1;
155
156 switch (config->GetKind_TimeIntScheme()) {
157 case RUNGE_KUTTA_EXPLICIT:
158 iRKLimit = config->GetnRKStep();
159 break;
160 case CLASSICAL_RK4_EXPLICIT:
161 iRKLimit = 4;
162 break;
163 case EULER_EXPLICIT:
164 case EULER_IMPLICIT:
165 iRKLimit = 1;
166 break;
167 }
168
169 /*--- Do a presmoothing on the grid iMesh to be restricted to the grid iMesh+1 ---*/
170
171 for (unsigned short iPreSmooth = 0; iPreSmooth < config->GetMG_PreSmooth(iMesh); iPreSmooth++) {
172
173 /*--- Time and space integration ---*/
174
175 for (unsigned short iRKStep = 0; iRKStep < iRKLimit; iRKStep++) {
176
177 /*--- Send-Receive boundary conditions, and preprocessing ---*/
178
179 solver_fine->Preprocessing(geometry_fine, solver_container_fine, config, iMesh, iRKStep, RunTime_EqSystem, false);
180
181
182 if (iRKStep == 0) {
183
184 /*--- Set the old solution ---*/
185
186 solver_fine->Set_OldSolution();
187
188 if (classical_rk4) solver_fine->Set_NewSolution();
189
190 /*--- Compute time step, max eigenvalue, and integration scheme (steady and unsteady problems) ---*/
191
192 solver_fine->SetTime_Step(geometry_fine, solver_container_fine, config, iMesh, config->GetTimeIter());
193
194 /*--- Restrict the solution and gradient for the adjoint problem ---*/
195
196 Adjoint_Setup(geometry, solver_container, config_container, RunTime_EqSystem, config->GetTimeIter(), iZone);
197
198 }
199
200 /*--- Space integration ---*/
201
202 Space_Integration(geometry_fine, solver_container_fine, numerics_fine, config, iMesh, iRKStep, RunTime_EqSystem);
203
204 /*--- Time integration, update solution using the old solution plus the solution increment ---*/
205
206 Time_Integration(geometry_fine, solver_container_fine, config, iRKStep, RunTime_EqSystem);
207
208 /*--- Send-Receive boundary conditions, and postprocessing ---*/
209
210 solver_fine->Postprocessing(geometry_fine, solver_container_fine, config, iMesh);
211
212 }
213
214 }
215
216 /*--- Compute Forcing Term $P_(k+1) = I^(k+1)_k(P_k+F_k(u_k))-F_(k+1)(I^(k+1)_k u_k)$ and update solution for multigrid ---*/
217
218 if ( iMesh < config->GetnMGLevels() ) {
219
220 /*--- Shorter names to refer to coarse grid entities. ---*/
221
222 CGeometry* geometry_coarse = geometry[iZone][iInst][iMesh+1];
223 CSolver** solver_container_coarse = solver_container[iZone][iInst][iMesh+1];
224 CSolver* solver_coarse = solver_container_coarse[Solver_Position];
225 CNumerics** numerics_coarse = numerics_container[iZone][iInst][iMesh+1][Solver_Position];
226
227 /*--- Temporarily disable implicit integration, for what follows we do not need the Jacobian. ---*/
228
229 if (implicit) {
230 SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetKind_TimeIntScheme(EULER_EXPLICIT);)
231 }
232
233 /*--- Compute $r_k = P_k + F_k(u_k)$ ---*/
234
235 solver_fine->Preprocessing(geometry_fine, solver_container_fine, config, iMesh, NO_RK_ITER, RunTime_EqSystem, false);
236
237 Space_Integration(geometry_fine, solver_container_fine, numerics_fine, config, iMesh, NO_RK_ITER, RunTime_EqSystem);
238
239 SetResidual_Term(geometry_fine, solver_fine);
240
241 /*--- Compute $r_(k+1) = F_(k+1)(I^(k+1)_k u_k)$ ---*/
242
243 SetRestricted_Solution(RunTime_EqSystem, solver_fine, solver_coarse, geometry_fine, geometry_coarse, config);
244
245 solver_coarse->Preprocessing(geometry_coarse, solver_container_coarse, config, iMesh+1, NO_RK_ITER, RunTime_EqSystem, false);
246
247 Space_Integration(geometry_coarse, solver_container_coarse, numerics_coarse, config, iMesh+1, NO_RK_ITER, RunTime_EqSystem);
248
249 /*--- Compute $P_(k+1) = I^(k+1)_k(r_k) - r_(k+1) ---*/
250
251 SetForcing_Term(solver_fine, solver_coarse, geometry_fine, geometry_coarse, config, iMesh+1);
252
253 /*--- Restore the time integration settings. ---*/
254
255 if (implicit) {
256 SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetKind_TimeIntScheme(EULER_IMPLICIT);)
257 }
258
259 /*--- Recursive call to MultiGrid_Cycle (this routine). ---*/
260
261 for (unsigned short imu = 0; imu <= RecursiveParam; imu++) {
262
263 unsigned short nextRecurseParam = RecursiveParam;
264 if (iMesh == config->GetnMGLevels()-2)
265 nextRecurseParam = 0;
266
267 MultiGrid_Cycle(geometry, solver_container, numerics_container, config_container,
268 iMesh+1, nextRecurseParam, RunTime_EqSystem, iZone, iInst);
269 }
270
271 /*--- Compute prolongated solution, and smooth the correction $u^(new)_k = u_k + Smooth(I^k_(k+1)(u_(k+1)-I^(k+1)_k u_k))$ ---*/
272
273 GetProlongated_Correction(RunTime_EqSystem, solver_fine, solver_coarse, geometry_fine, geometry_coarse, config);
274
275 SmoothProlongated_Correction(RunTime_EqSystem, solver_fine, geometry_fine, config->GetMG_CorrecSmooth(iMesh), 1.25, config);
276
277 SetProlongated_Correction(solver_fine, geometry_fine, config, iMesh);
278
279
280 /*--- Solution post-smoothing in the prolongated grid. ---*/
281
282 for (unsigned short iPostSmooth = 0; iPostSmooth < config->GetMG_PostSmooth(iMesh); iPostSmooth++) {
283
284 for (unsigned short iRKStep = 0; iRKStep < iRKLimit; iRKStep++) {
285
286 solver_fine->Preprocessing(geometry_fine, solver_container_fine, config, iMesh, iRKStep, RunTime_EqSystem, false);
287
288 if (iRKStep == 0) {
289 solver_fine->Set_OldSolution();
290
291 if (classical_rk4) solver_fine->Set_NewSolution();
292
293 solver_fine->SetTime_Step(geometry_fine, solver_container_fine, config, iMesh, config->GetTimeIter());
294 }
295
296 Space_Integration(geometry_fine, solver_container_fine, numerics_fine, config, iMesh, iRKStep, RunTime_EqSystem);
297
298 Time_Integration(geometry_fine, solver_container_fine, config, iRKStep, RunTime_EqSystem);
299
300 solver_fine->Postprocessing(geometry_fine, solver_container_fine, config, iMesh);
301
302 }
303 }
304 }
305
306}

308void CMultiGridIntegration::GetProlongated_Correction(unsigned short RunTime_EqSystem, CSolver *sol_fine, CSolver *sol_coarse,
309 CGeometry *geo_fine, CGeometry *geo_coarse, CConfig *config) {
310 unsigned long Point_Fine, Point_Coarse, iVertex;
311 unsigned short iMarker, iChildren, iVar;
312 su2double Area_Parent, Area_Children;
313 const su2double *Solution_Fine = nullptr, *Solution_Coarse = nullptr;
314
315 const unsigned short nVar = sol_coarse->GetnVar();
316
317 auto *Solution = new su2double[nVar];
318
319 SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
320 for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
321
322 Area_Parent = geo_coarse->nodes->GetVolume(Point_Coarse);
323
324 for (iVar = 0; iVar < nVar; iVar++) Solution[iVar] = 0.0;
325
326 for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
327 Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
328 Area_Children = geo_fine->nodes->GetVolume(Point_Fine);
329 Solution_Fine = sol_fine->GetNodes()->GetSolution(Point_Fine);
330 for (iVar = 0; iVar < nVar; iVar++)
331 Solution[iVar] -= Solution_Fine[iVar]*Area_Children/Area_Parent;
332 }
333
334 Solution_Coarse = sol_coarse->GetNodes()->GetSolution(Point_Coarse);
335
336 for (iVar = 0; iVar < nVar; iVar++)
337 Solution[iVar] += Solution_Coarse[iVar];
338
339 for (iVar = 0; iVar < nVar; iVar++)
340 sol_coarse->GetNodes()->SetSolution_Old(Point_Coarse,Solution);
341 }
342 END_SU2_OMP_FOR
343
344 delete [] Solution;
345
346 /*--- Remove any contributions from no-slip walls. ---*/
347
348 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
349 if (config->GetViscous_Wall(iMarker)) {
350
351 SU2_OMP_FOR_STAT(32)
352 for (iVertex = 0; iVertex < geo_coarse->nVertex[iMarker]; iVertex++) {
353
354 Point_Coarse = geo_coarse->vertex[iMarker][iVertex]->GetNode();
355
356 /*--- For dirichlet boundary condtions, set the correction to zero.
357 Note that Solution_Old stores the correction not the actual value ---*/
358
359 su2double zero[3] = {0.0};
360 sol_coarse->GetNodes()->SetVelocity_Old(Point_Coarse, zero);
361
362 }
363 END_SU2_OMP_FOR
364 }
365 }
366
367 /*--- MPI the set solution old ---*/
368
369 sol_coarse->InitiateComms(geo_coarse, config, SOLUTION_OLD);
370 sol_coarse->CompleteComms(geo_coarse, config, SOLUTION_OLD);
371
372 SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
373 for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
374 for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
375 Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
376 sol_fine->LinSysRes.SetBlock(Point_Fine, sol_coarse->GetNodes()->GetSolution_Old(Point_Coarse));
377 }
378 }
379 END_SU2_OMP_FOR
380
381}

383void CMultiGridIntegration::SmoothProlongated_Correction(unsigned short RunTime_EqSystem, CSolver *solver, CGeometry *geometry,
384 unsigned short val_nSmooth, su2double val_smooth_coeff, CConfig *config) {
385
386 /*--- Check if there is work to do. ---*/
387 if (val_nSmooth == 0) return;
388
389 const su2double *Residual_Old, *Residual_Sum, *Residual_j;
390 unsigned short iVar, iSmooth, iMarker, iNeigh;
391 unsigned long iPoint, jPoint, iVertex;
392
393 const unsigned short nVar = solver->GetnVar();
394
395 SU2_OMP_FOR_STAT(roundUpDiv(geometry->GetnPoint(), omp_get_num_threads()))
396 for (iPoint = 0; iPoint < geometry->GetnPoint(); iPoint++) {
397 Residual_Old = solver->LinSysRes.GetBlock(iPoint);
398 solver->GetNodes()->SetResidual_Old(iPoint,Residual_Old);
399 }
400 END_SU2_OMP_FOR
401
402 /*--- Jacobi iterations. ---*/
403
404 for (iSmooth = 0; iSmooth < val_nSmooth; iSmooth++) {
405
406 /*--- Loop over all mesh points (sum the residuals of direct neighbors). ---*/
407
408 SU2_OMP_FOR_STAT(roundUpDiv(geometry->GetnPoint(), omp_get_num_threads()))
409 for (iPoint = 0; iPoint < geometry->GetnPoint(); ++iPoint) {
410
411 solver->GetNodes()->SetResidualSumZero(iPoint);
412
413 for (iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); ++iNeigh) {
414 jPoint = geometry->nodes->GetPoint(iPoint, iNeigh);
415 Residual_j = solver->LinSysRes.GetBlock(jPoint);
416 solver->GetNodes()->AddResidual_Sum(iPoint, Residual_j);
417 }
418
419 }
420 END_SU2_OMP_FOR
421
422 /*--- Loop over all mesh points (update residuals with the neighbor averages). ---*/
423
424 SU2_OMP_FOR_STAT(roundUpDiv(geometry->GetnPoint(), omp_get_num_threads()))
425 for (iPoint = 0; iPoint < geometry->GetnPoint(); ++iPoint) {
426
427 su2double factor = 1.0/(1.0+val_smooth_coeff*su2double(geometry->nodes->GetnPoint(iPoint)));
428
429 Residual_Sum = solver->GetNodes()->GetResidual_Sum(iPoint);
430 Residual_Old = solver->GetNodes()->GetResidual_Old(iPoint);
431
432 for (iVar = 0; iVar < nVar; iVar++)
433 solver->LinSysRes(iPoint,iVar) = (Residual_Old[iVar] + val_smooth_coeff*Residual_Sum[iVar])*factor;
434 }
435 END_SU2_OMP_FOR
436
437 /*--- Restore original residuals (without average) at boundary points. ---*/
438
439 for (iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
440 if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
441 (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
442 (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
443
444 SU2_OMP_FOR_STAT(32)
445 for (iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
446 iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
447 Residual_Old = solver->GetNodes()->GetResidual_Old(iPoint);
448 solver->LinSysRes.SetBlock(iPoint, Residual_Old);
449 }
450 END_SU2_OMP_FOR
451 }
452 }
453
454 }

458void CMultiGridIntegration::SetProlongated_Correction(CSolver *sol_fine, CGeometry *geo_fine,
459 CConfig *config, unsigned short iMesh) {
460 unsigned long Point_Fine;
461 unsigned short iVar;
462 su2double *Solution_Fine, *Residual_Fine;
463
464 const unsigned short nVar = sol_fine->GetnVar();
465 const su2double factor = config->GetDamp_Correc_Prolong();
466
467 SU2_OMP_FOR_STAT(roundUpDiv(geo_fine->GetnPointDomain(), omp_get_num_threads()))
468 for (Point_Fine = 0; Point_Fine < geo_fine->GetnPointDomain(); Point_Fine++) {
469 Residual_Fine = sol_fine->LinSysRes.GetBlock(Point_Fine);
470 Solution_Fine = sol_fine->GetNodes()->GetSolution(Point_Fine);
471 for (iVar = 0; iVar < nVar; iVar++) {
472 /*--- Prevent a fine grid divergence due to a coarse grid divergence ---*/
473 if (Residual_Fine[iVar] != Residual_Fine[iVar])
474 Residual_Fine[iVar] = 0.0;
475 Solution_Fine[iVar] += factor*Residual_Fine[iVar];
476 }
477 }
478 END_SU2_OMP_FOR
479
480 /*--- MPI the new interpolated solution ---*/
481
482 sol_fine->InitiateComms(geo_fine, config, SOLUTION);
483 sol_fine->CompleteComms(geo_fine, config, SOLUTION);
484
485}

502void CMultiGridIntegration::SetForcing_Term(CSolver *sol_fine, CSolver *sol_coarse, CGeometry *geo_fine,
503 CGeometry *geo_coarse, CConfig *config, unsigned short iMesh) {
504
505 unsigned long Point_Fine, Point_Coarse, iVertex;
506 unsigned short iMarker, iVar, iChildren;
507 const su2double *Residual_Fine;
508
509 const unsigned short nVar = sol_coarse->GetnVar();
510 su2double factor = config->GetDamp_Res_Restric();
511
512 auto *Residual = new su2double[nVar];
513
514 SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
515 for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
516
517 sol_coarse->GetNodes()->SetRes_TruncErrorZero(Point_Coarse);
518
519 for (iVar = 0; iVar < nVar; iVar++) Residual[iVar] = 0.0;
520
521 for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
522 Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
523 Residual_Fine = sol_fine->LinSysRes.GetBlock(Point_Fine);
524 for (iVar = 0; iVar < nVar; iVar++)
525 Residual[iVar] += factor*Residual_Fine[iVar];
526 }
527 sol_coarse->GetNodes()->AddRes_TruncError(Point_Coarse, Residual);
528 }
529 END_SU2_OMP_FOR
530
531 delete [] Residual;
532
533 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
534 if (config->GetViscous_Wall(iMarker)) {
535 SU2_OMP_FOR_STAT(32)
536 for (iVertex = 0; iVertex < geo_coarse->nVertex[iMarker]; iVertex++) {
537 Point_Coarse = geo_coarse->vertex[iMarker][iVertex]->GetNode();
538 sol_coarse->GetNodes()->SetVel_ResTruncError_Zero(Point_Coarse);
539 }
540 END_SU2_OMP_FOR
541 }
542 }
543
544 SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
545 for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
546 sol_coarse->GetNodes()->SubtractRes_TruncError(Point_Coarse, sol_coarse->LinSysRes.GetBlock(Point_Coarse));
547 }
548 END_SU2_OMP_FOR
549
550}

563void CMultiGridIntegration::SetRestricted_Solution(unsigned short RunTime_EqSystem, CSolver *sol_fine, CSolver *sol_coarse,
564 CGeometry *geo_fine, CGeometry *geo_coarse, CConfig *config) {
565
566 const unsigned short Solver_Position = config->GetContainerPosition(RunTime_EqSystem);
567 const bool grid_movement = config->GetGrid_Movement();
568
569 /*--- Compute coarse solution from fine solution ---*/
570
571 CSolver::MultigridRestriction(*geo_fine, sol_fine->GetNodes()->GetSolution(),
572 *geo_coarse, sol_coarse->GetNodes()->GetSolution());
573
574 /*--- Update the solution at the no-slip walls ---*/
575
576 for (auto iMarker = 0u; iMarker < config->GetnMarker_All(); iMarker++) {
577 if (config->GetViscous_Wall(iMarker)) {
578
579 SU2_OMP_FOR_STAT(32)
580 for (auto iVertex = 0ul; iVertex < geo_coarse->nVertex[iMarker]; iVertex++) {
581
582 const auto Point_Coarse = geo_coarse->vertex[iMarker][iVertex]->GetNode();
583
584 if (Solver_Position == FLOW_SOL) {
585
586 /*--- At moving walls, set the solution based on the new density and wall velocity ---*/
587
588 if (grid_movement) {
589 const auto* Grid_Vel = geo_coarse->nodes->GetGridVel(Point_Coarse);
590 sol_coarse->GetNodes()->SetVelSolutionVector(Point_Coarse, Grid_Vel);
591 }
592 else {
593 /*--- For stationary no-slip walls, set the velocity to zero. ---*/
594 su2double zero[3] = {0.0};
595 sol_coarse->GetNodes()->SetVelSolutionVector(Point_Coarse, zero);
596 }
597
598 }
599
600 if (Solver_Position == ADJFLOW_SOL) {
601 sol_coarse->GetNodes()->SetVelSolutionDVector(Point_Coarse);
602 }
603
604 }
605 END_SU2_OMP_FOR
606 }
607 }
608
609 /*--- MPI the new interpolated solution ---*/
610
611 sol_coarse->InitiateComms(geo_coarse, config, SOLUTION);
612 sol_coarse->CompleteComms(geo_coarse, config, SOLUTION);
613
614}

616void CMultiGridIntegration::SetRestricted_Gradient(unsigned short RunTime_EqSystem, CSolver *sol_fine, CSolver *sol_coarse,
617 CGeometry *geo_fine, CGeometry *geo_coarse, CConfig *config) {
618 unsigned long Point_Fine, Point_Coarse;
619 unsigned short iVar, iDim, iChildren;
620 su2double Area_Parent, Area_Children;
621
622 const unsigned short nDim = geo_coarse->GetnDim();
623 const unsigned short nVar = sol_coarse->GetnVar();
624
625 auto **Gradient = new su2double* [nVar];
626 for (iVar = 0; iVar < nVar; iVar++)
627 Gradient[iVar] = new su2double [nDim];
628
629 SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPoint(), omp_get_num_threads()))
630 for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPoint(); Point_Coarse++) {
631 Area_Parent = geo_coarse->nodes->GetVolume(Point_Coarse);
632
633 for (iVar = 0; iVar < nVar; iVar++)
634 for (iDim = 0; iDim < nDim; iDim++)
635 Gradient[iVar][iDim] = 0.0;
636
637 for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
638 Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
639 Area_Children = geo_fine->nodes->GetVolume(Point_Fine);
640 auto Gradient_fine = sol_fine->GetNodes()->GetGradient(Point_Fine);
641
642 for (iVar = 0; iVar < nVar; iVar++)
643 for (iDim = 0; iDim < nDim; iDim++)
644 Gradient[iVar][iDim] += Gradient_fine[iVar][iDim]*Area_Children/Area_Parent;
645 }
646 sol_coarse->GetNodes()->SetGradient(Point_Coarse,Gradient);
647 }
648 END_SU2_OMP_FOR
649
650 for (iVar = 0; iVar < nVar; iVar++)
651 delete [] Gradient[iVar];
652 delete [] Gradient;
653
654}

656void CMultiGridIntegration::NonDimensional_Parameters(CGeometry **geometry, CSolver ***solver_container,
657 CNumerics ****numerics_container, CConfig *config,
658 unsigned short FinestMesh, unsigned short RunTime_EqSystem,
659 su2double *monitor) {
660 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS

662
663 case RUNTIME_FLOW_SYS:
664
665 /*--- Calculate the inviscid and viscous forces ---*/

668 solver_container[FinestMesh][FLOW_SOL]->Momentum_Forces(geometry[FinestMesh], config);
669 solver_container[FinestMesh][FLOW_SOL]->Friction_Forces(geometry[FinestMesh], config);
670

673
674 /*--- Average quantities at the inflow and outflow boundaries ---*/
675
676 solver_container[FinestMesh][FLOW_SOL]->TurboAverageProcess(solver_container[FinestMesh], geometry[FinestMesh],config,INFLOW);
677 solver_container[FinestMesh][FLOW_SOL]->TurboAverageProcess(solver_container[FinestMesh], geometry[FinestMesh], config, OUTFLOW);
678
679 /*--- Gather Inflow and Outflow quantities on the Master Node to compute performance ---*/
680
681 solver_container[FinestMesh][FLOW_SOL]->GatherInOutAverageValues(config, geometry[FinestMesh]);
682
683 }
684
685 break;
686
687 case RUNTIME_ADJFLOW_SYS:
688
689 /*--- Calculate the inviscid and viscous sensitivities ---*/
690
691 solver_container[FinestMesh][ADJFLOW_SOL]->Inviscid_Sensitivity(geometry[FinestMesh], solver_container[FinestMesh],
692 numerics_container[FinestMesh][ADJFLOW_SOL][CONV_BOUND_TERM], config);
693
694 solver_container[FinestMesh][ADJFLOW_SOL]->Viscous_Sensitivity(geometry[FinestMesh], solver_container[FinestMesh],
695 numerics_container[FinestMesh][ADJFLOW_SOL][CONV_BOUND_TERM], config);
696
697 /*--- Smooth the inviscid and viscous sensitivities ---*/
698
699 if (config->GetKind_SensSmooth() != NONE)
700 solver_container[FinestMesh][ADJFLOW_SOL]->Smooth_Sensitivity(geometry[FinestMesh], solver_container[FinestMesh],
701 numerics_container[FinestMesh][ADJFLOW_SOL][CONV_BOUND_TERM], config);
702 break;
703 }
704 END_SU2_OMP_SAFE_GLOBAL_ACCESS
