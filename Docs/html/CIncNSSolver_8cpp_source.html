63void CIncNSSolver::Preprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh,
64 unsigned short iRKStep, unsigned short RunTime_EqSystem, bool Output) {
65
66 const auto InnerIter = config->GetInnerIter();
67 const bool muscl = config->GetMUSCL_Flow() && (iMesh == MESH_0);
68 const bool center = (config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED);
69 const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE) && (InnerIter <= config->GetLimiterIter());
70 const bool van_albada = (config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE);
71 const bool wall_functions = config->GetWall_Functions();
72
73 /*--- Common preprocessing steps (implemented by CEulerSolver) ---*/
74
75 CommonPreprocessing(geometry, solver_container, config, iMesh, iRKStep, RunTime_EqSystem, Output);
76
77 /*--- Compute gradient for MUSCL reconstruction ---*/
78
79 if (config->GetReconstructionGradientRequired() && muscl && !center) {
80 switch (config->GetKind_Gradient_Method_Recon()) {
81 case GREEN_GAUSS:
82 SetPrimitive_Gradient_GG(geometry, config, true); break;
83 case LEAST_SQUARES:
84 case WEIGHTED_LEAST_SQUARES:
85 SetPrimitive_Gradient_LS(geometry, config, true); break;
86 default: break;
87 }
88 }
89
90 /*--- Compute gradient of the primitive variables ---*/
91
92 if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
93 SetPrimitive_Gradient_GG(geometry, config);
94 }
95 else if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
96 SetPrimitive_Gradient_LS(geometry, config);
97 }
98
99 /*--- Compute the limiters ---*/
100
101 if (muscl && !center && limiter && !van_albada && !Output) {
102 SetPrimitive_Limiter(geometry, config);
103 }
104
105 ComputeVorticityAndStrainMag(*config, geometry, iMesh);
106
107 /*--- Compute the TauWall from the wall functions ---*/
108
109 if (wall_functions) {
110 SU2_OMP_SAFE_GLOBAL_ACCESS(SetTau_Wall_WF(geometry, solver_container, config);)
111 // nijso: we have to set this as well??
112 // seteddyviscfirstpoint
113 }
114
115 /*--- Compute recovered pressure and temperature for streamwise periodic flow ---*/
116 if (config->GetKind_Streamwise_Periodic() != ENUM_STREAMWISE_PERIODIC::NONE)
117 Compute_Streamwise_Periodic_Recovered_Values(config, geometry, iMesh);
118}

120void CIncNSSolver::GetStreamwise_Periodic_Properties(const CGeometry *geometry,
121 CConfig *config,
122 const unsigned short iMesh) {
123
124 /*---------------------------------------------------------------------------------------------*/
125 // 1. Evaluate massflow, area avg density & Temperature and Area at streamwise periodic outlet.
126 // 2. Update delta_p is target massflow is chosen.
127 // 3. Loop Heatflux markers and integrate heat across the boundary. Only if energy equation is on.
128 /*---------------------------------------------------------------------------------------------*/
129
130 /*-------------------------------------------------------------------------------------------------*/
131 /*--- 1. Evaluate Massflow [kg/s], area-averaged density [kg/m^3] and Area [m^2] at the ---*/
132 /*--- (there can be only one) streamwise periodic outlet/donor marker. Massflow is obviously ---*/
133 /*--- needed for prescribed massflow but also for the additional source and heatflux ---*/
134 /*--- boundary terms of the energy equation. Area and the avg-density are used for the ---*/
135 /*--- Pressure-Drop update in case of a prescribed massflow. ---*/
136 /*-------------------------------------------------------------------------------------------------*/
137
138 su2double Area_Local = 0.0,
139 MassFlow_Local = 0.0,
140 Average_Density_Local = 0.0,
141 Temperature_Local = 0.0;
142
143 for (auto iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
144
145 /*--- Only "outlet"/donor periodic marker. GetnMarker_Periodic counts from 1: First all the inlet marker from
146 the periodic marker pairs and then the all the outlets. I.e. in order to get the outlet of the first pair
147 we need to divide the number of periodic markers by 2 and add 1 (because count starts at 1). ---*/
148 if (config->GetMarker_All_KindBC(iMarker) == PERIODIC_BOUNDARY &&
149 config->GetMarker_All_PerBound(iMarker) == (config->GetnMarker_Periodic()/2 + 1)) {
150
151 for (auto iVertex = 0ul; iVertex < geometry->nVertex[iMarker]; iVertex++) {
152
153 auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
154
155 if (geometry->nodes->GetDomain(iPoint)) {
156
157 /*--- A = dot_prod(n_A*n_A), with n_A beeing the area-normal. ---*/
158
159 const auto AreaNormal = geometry->vertex[iMarker][iVertex]->GetNormal();
160
161 const su2double FaceArea = GeometryToolbox::Norm(nDim, AreaNormal);
162
163 /*--- m_dot = dot_prod(n*v) * A * rho, with n beeing unit normal. ---*/
164 MassFlow_Local += nodes->GetProjVel(iPoint, AreaNormal) * nodes->GetDensity(iPoint);
165
166 Area_Local += FaceArea;
167
168 Average_Density_Local += FaceArea * nodes->GetDensity(iPoint);
169
170 /*--- Due to periodicity, temperatures are equal one the inlet(1) and outlet(2) ---*/
171 Temperature_Local += FaceArea * nodes->GetTemperature(iPoint);
172
173 } // if domain
174 } // loop vertices
175 } // loop periodic boundaries
176 } // loop MarkerAll
177
178 // MPI Communication: Sum Area, Sum rho*A & T*A and divide by AreaGlobal, sum massflow
179 su2double Area_Global(0), Average_Density_Global(0), MassFlow_Global(0), Temperature_Global(0);
180 SU2_MPI::Allreduce(&Area_Local, &Area_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
181 SU2_MPI::Allreduce(&Average_Density_Local, &Average_Density_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
182 SU2_MPI::Allreduce(&MassFlow_Local, &MassFlow_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
183 SU2_MPI::Allreduce(&Temperature_Local, &Temperature_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
184
185 Average_Density_Global /= Area_Global;
186 Temperature_Global /= Area_Global;
187
188 /*--- Set solver variables ---*/
189 SPvals.Streamwise_Periodic_MassFlow = MassFlow_Global;
190 SPvals.Streamwise_Periodic_InletTemperature = Temperature_Global;
191 SPvals.Streamwise_Periodic_BoundaryArea = Area_Global;
192 SPvals.Streamwise_Periodic_AvgDensity = Average_Density_Global;
193
194 if (config->GetEnergy_Equation()) {
195 /*---------------------------------------------------------------------------------------------*/
196 /*--- 3. Compute the integrated Heatflow [W] for the energy equation source term, heatflux ---*/
197 /*--- boundary term and recovered Temperature. The computation is not completely clear. ---*/
198 /*--- Here the Heatflux from all Boundary markers in the config-file is used. ---*/
199 /*---------------------------------------------------------------------------------------------*/
200
201 su2double HeatFlow_Local = 0.0, HeatFlow_Global = 0.0;
202
203 /*--- Loop over all heatflux Markers ---*/
204 for (auto iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
205
206 if (config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX) {
207
208 /*--- Identify the boundary by string name and retrive heatflux from config ---*/
209 const auto Marker_StringTag = config->GetMarker_All_TagBound(iMarker);
210 const su2double Wall_HeatFlux = config->GetWall_HeatFlux(Marker_StringTag);
211
212 for (auto iVertex = 0ul; iVertex < geometry->nVertex[iMarker]; iVertex++) {
213
214 auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
215
216 if (!geometry->nodes->GetDomain(iPoint)) continue;
217
218 const auto AreaNormal = geometry->vertex[iMarker][iVertex]->GetNormal();
219
220 const su2double FaceArea = GeometryToolbox::Norm(nDim, AreaNormal);
221
222 HeatFlow_Local += FaceArea * (-1.0) * Wall_HeatFlux/config->GetHeat_Flux_Ref();
223 } // loop Vertices
224 } // loop Heatflux marker
225 } // loop AllMarker
226
227 /*--- MPI Communication sum up integrated Heatflux from all processes ---*/
228 SU2_MPI::Allreduce(&HeatFlow_Local, &HeatFlow_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
229
230 /*--- Set the solver variable Integrated Heatflux ---*/
231 SPvals.Streamwise_Periodic_IntegratedHeatFlow = HeatFlow_Global;
232 } // if energy
233}

236void CIncNSSolver::Compute_Streamwise_Periodic_Recovered_Values(CConfig *config, const CGeometry *geometry,
237 const unsigned short iMesh) {
238
239 const bool energy = (config->GetEnergy_Equation() && config->GetStreamwise_Periodic_Temperature());
240 const auto InnerIter = config->GetInnerIter();
241
242 /*--- Reference node on inlet periodic marker to compute relative distance along periodic translation vector. ---*/
243 const auto ReferenceNode = geometry->GetStreamwise_Periodic_RefNode();
244
245 /*--- Compute square of the distance between the 2 periodic surfaces. ---*/
246 const su2double norm2_translation = GeometryToolbox::SquaredNorm(nDim, config->GetPeriodic_Translation(0));
247
248 /*--- Compute recoverd pressure and temperature for all points ---*/
249 SU2_OMP_FOR_STAT(omp_chunk_size)
250 for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
251
252 /*--- First, compute helping terms based on relative distance (0,l) between periodic markers ---*/
253 su2double dot_product = 0.0;
254 for (unsigned short iDim = 0; iDim < nDim; iDim++)
255 dot_product += fabs( (geometry->nodes->GetCoord(iPoint,iDim) - ReferenceNode[iDim]) * config->GetPeriodic_Translation(0)[iDim]);
256
257 /*--- Second, substract/add correction from reduced pressure/temperature to get recoverd pressure/temperature ---*/
258 const su2double Pressure_Recovered = nodes->GetPressure(iPoint) - SPvals.Streamwise_Periodic_PressureDrop /
259 norm2_translation * dot_product;
260 nodes->SetStreamwise_Periodic_RecoveredPressure(iPoint, Pressure_Recovered);
261
262 /*--- InnerIter > 0 as otherwise MassFlow in the denominator would be zero ---*/
263 if (energy && InnerIter > 0) {
264 su2double Temperature_Recovered = nodes->GetTemperature(iPoint);
265 Temperature_Recovered += SPvals.Streamwise_Periodic_IntegratedHeatFlow /
266 (SPvals.Streamwise_Periodic_MassFlow * nodes->GetSpecificHeatCp(iPoint) * norm2_translation) * dot_product;
267 nodes->SetStreamwise_Periodic_RecoveredTemperature(iPoint, Temperature_Recovered);
268 }
269 } // for iPoint
270 END_SU2_OMP_FOR
271
272 /*--- Compute the integrated Heatflux Q into the domain, and massflow over periodic markers ---*/
273 SU2_OMP_SAFE_GLOBAL_ACCESS(GetStreamwise_Periodic_Properties(geometry, config, iMesh);)
274}

282unsigned long CIncNSSolver::SetPrimitive_Variables(CSolver **solver_container, const CConfig *config) {
283
284 unsigned long iPoint, nonPhysicalPoints = 0;
285 su2double eddy_visc = 0.0, turb_ke = 0.0, DES_LengthScale = 0.0;
286 const su2double* scalar = nullptr;
287 const TURB_MODEL turb_model = config->GetKind_Turb_Model();
288 const SPECIES_MODEL species_model = config->GetKind_Species_Model();
289
290 bool tkeNeeded = (turb_model == TURB_MODEL::SST);
291
292 AD::StartNoSharedReading();
293
294 SU2_OMP_FOR_STAT(omp_chunk_size)
295 for (iPoint = 0; iPoint < nPoint; iPoint++) {
296
297 /*--- Retrieve the value of the kinetic energy (if needed) ---*/
298
299 if (turb_model != TURB_MODEL::NONE && solver_container[TURB_SOL] != nullptr) {
300 eddy_visc = solver_container[TURB_SOL]->GetNodes()->GetmuT(iPoint);
301 if (tkeNeeded) turb_ke = solver_container[TURB_SOL]->GetNodes()->GetSolution(iPoint,0);
302
303 if (config->GetKind_HybridRANSLES() != NO_HYBRIDRANSLES){
304 DES_LengthScale = solver_container[TURB_SOL]->GetNodes()->GetDES_LengthScale(iPoint);
305 }
306 }
307
308 /*--- Retrieve scalar values (if needed) ---*/
309 if (species_model != SPECIES_MODEL::NONE && solver_container[SPECIES_SOL] != nullptr) {
310 scalar = solver_container[SPECIES_SOL]->GetNodes()->GetSolution(iPoint);
311 }
312
313 /*--- Incompressible flow, primitive variables --- */
314
315 bool physical = static_cast<CIncNSVariable*>(nodes)->SetPrimVar(iPoint,eddy_visc, turb_ke, GetFluidModel(), scalar);
316
317 /* Check for non-realizable states for reporting. */
318
319 if (!physical) nonPhysicalPoints++;
320
321 /*--- Set the DES length scale ---*/
322
323 nodes->SetDES_LengthScale(iPoint,DES_LengthScale);
324
325 }
326 END_SU2_OMP_FOR
327
328 AD::EndNoSharedReading();
329
330 return nonPhysicalPoints;
331

334void CIncNSSolver::BC_Wall_Generic(const CGeometry *geometry, const CConfig *config,
335 unsigned short val_marker, unsigned short kind_boundary) {
336
337 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
338 const bool energy = config->GetEnergy_Equation();
339 const bool py_custom = config->GetMarker_All_PyCustom(val_marker);
340
341 /*--- Variables for streamwise periodicity ---*/
342 const bool streamwise_periodic = (config->GetKind_Streamwise_Periodic() != ENUM_STREAMWISE_PERIODIC::NONE);
343 const bool streamwise_periodic_temperature = config->GetStreamwise_Periodic_Temperature();
344 su2double Cp, thermal_conductivity, dot_product, scalar_factor;
345
346 /*--- Identify the boundary by string name ---*/
347
348 const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
349
350 /*--- Get the specified wall heat flux, temperature or heat transfer coefficient from config ---*/
351
352 su2double Wall_HeatFlux = 0.0, Twall = 0.0, Tinfinity = 0.0, Transfer_Coefficient = 0.0;
353
354 switch (kind_boundary) {
355 case HEAT_FLUX:
356 Wall_HeatFlux = config->GetWall_HeatFlux(Marker_Tag) / config->GetHeat_Flux_Ref();
357 if (config->GetIntegrated_HeatFlux()) {
358 Wall_HeatFlux /= geometry->GetSurfaceArea(config, val_marker);
359 }
360 break;
361 case ISOTHERMAL:
362 Twall = config->GetIsothermal_Temperature(Marker_Tag) / config->GetTemperature_Ref();
363 break;
364 case HEAT_TRANSFER:
365 Transfer_Coefficient = config->GetWall_HeatTransfer_Coefficient(Marker_Tag) * config->GetTemperature_Ref() /
366 config->GetHeat_Flux_Ref();
367 Tinfinity = config->GetWall_HeatTransfer_Temperature(Marker_Tag) / config->GetTemperature_Ref();
368 break;
369 default:
370 SU2_MPI::Error("Unknown type of boundary condition.", CURRENT_FUNCTION);
371 break;
372 }
373
374 /*--- Get wall function treatment from config. ---*/
375
376 //const auto Wall_Function = config->GetWallFunction_Treatment(Marker_Tag);
377 // nijso: we do not have a special treatment yet for heated walls
378 // the wall function model is written for heat flux, we have to implement isothermal wall conditions
379 //if (Wall_Function != WALL_FUNCTIONS::NONE)
380 // SU2_MPI::Error("Wall function treatment not implemented yet.", CURRENT_FUNCTION);
381
382 /*--- Loop over all of the vertices on this boundary marker ---*/
383
384 SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
385 for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
386 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
387
388 /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
389
390 if (!geometry->nodes->GetDomain(iPoint)) continue;
391
392 /*--- Compute dual-grid area and boundary normal ---*/
393
394 const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
395
396 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
397
398 /*--- Impose the value of the velocity as a strong boundary
399 condition (Dirichlet). Fix the velocity and remove any
400 contribution to the residual at this node. ---*/
401
402 if (dynamic_grid) {
403 nodes->SetVelocity_Old(iPoint, geometry->nodes->GetGridVel(iPoint));
404 } else {
405 su2double zero[MAXNDIM] = {0.0};
406 nodes->SetVelocity_Old(iPoint, zero);
407 }
408
409 for (unsigned short iDim = 0; iDim < nDim; iDim++)
410 LinSysRes(iPoint, iDim+1) = 0.0;
411 nodes->SetVel_ResTruncError_Zero(iPoint);
412
413 /*--- Enforce the no-slip boundary condition in a strong way by
414 modifying the velocity-rows of the Jacobian (1 on the diagonal). ---*/
415
416 if (implicit) {
417 for (unsigned short iVar = 1; iVar <= nDim; iVar++)
418 Jacobian.DeleteValsRowi(iPoint*nVar+iVar);
419 }
420
421 if (!energy) continue;
422
423 switch(kind_boundary) {
424 case HEAT_FLUX:
425
426 /*--- Apply a weak boundary condition for the energy equation.
427 Compute the residual due to the prescribed heat flux. ---*/
428
429 if (py_custom) {
430 Wall_HeatFlux = geometry->GetCustomBoundaryHeatFlux(val_marker, iVertex) / config->GetHeat_Flux_Ref();
431 }
432 LinSysRes(iPoint, nDim+1) -= Wall_HeatFlux*Area;
433
434 /*--- With streamwise periodic flow and heatflux walls an additional term is introduced in the boundary formulation ---*/
435 if (streamwise_periodic && streamwise_periodic_temperature) {
436
437 Cp = nodes->GetSpecificHeatCp(iPoint);
438 thermal_conductivity = nodes->GetThermalConductivity(iPoint);
439
440 /*--- Scalar factor of the residual contribution ---*/
441 const su2double norm2_translation = GeometryToolbox::SquaredNorm(nDim, config->GetPeriodic_Translation(0));
442 scalar_factor = SPvals.Streamwise_Periodic_IntegratedHeatFlow*thermal_conductivity / (SPvals.Streamwise_Periodic_MassFlow * Cp * norm2_translation);
443
444 /*--- Dot product ---*/
445 dot_product = GeometryToolbox::DotProduct(nDim, config->GetPeriodic_Translation(0), Normal);
446
447 LinSysRes(iPoint, nDim+1) += scalar_factor*dot_product;
448 } // if streamwise_periodic
449 break;
450
451 case HEAT_TRANSFER:
452 Twall = nodes->GetTemperature(iPoint);
453 Wall_HeatFlux = Transfer_Coefficient * (Tinfinity - Twall);
454
455 /*--- Apply a weak boundary condition for the energy equation.
456 Compute the residual due to the prescribed temperature and transfer coefficient.
457 Note that for the Heat_Transfer wall, basically a heatflux wall that depends on the local Temperature is applied. ---*/
458 LinSysRes(iPoint, nDim+1) -= Wall_HeatFlux*Area;
459
460 if (implicit) {
461 Jacobian.AddVal2Diag(iPoint, nDim+1, Transfer_Coefficient*Area);
462 }
463 break;
464
465 case ISOTHERMAL:
466 if (py_custom) {
467 Twall = geometry->GetCustomBoundaryTemperature(val_marker, iVertex) / config->GetTemperature_Ref();
468 }
469 const auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
470
471 /*--- Get coordinates of i & nearest normal and compute distance ---*/
472
473 const auto Coord_i = geometry->nodes->GetCoord(iPoint);
474 const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);
475 su2double Edge_Vector[MAXNDIM];
476 GeometryToolbox::Distance(nDim, Coord_j, Coord_i, Edge_Vector);
477 su2double dist_ij_2 = GeometryToolbox::SquaredNorm(nDim, Edge_Vector);
478 su2double dist_ij = sqrt(dist_ij_2);
479
480 /*--- Compute the normal gradient in temperature using Twall ---*/
481
482 su2double dTdn = -(nodes->GetTemperature(Point_Normal) - Twall)/dist_ij;
483
484 /*--- Get thermal conductivity ---*/
485
486 su2double thermal_conductivity = nodes->GetThermalConductivity(iPoint);
487
488 /*--- Apply a weak boundary condition for the energy equation.
489 Compute the residual due to the prescribed heat flux. ---*/
490
491 LinSysRes(iPoint, nDim+1) -= thermal_conductivity*dTdn*Area;
492
493 /*--- Jacobian contribution for temperature equation. ---*/
494
495 if (implicit) {
496 su2double proj_vector_ij = 0.0;
497 if (dist_ij_2 > 0.0)
498 proj_vector_ij = GeometryToolbox::DotProduct(nDim, Edge_Vector, Normal) / dist_ij_2;
499 Jacobian.AddVal2Diag(iPoint, nDim+1, thermal_conductivity*proj_vector_ij);
500 }
501 break;
502 } // switch
503 }
504 END_SU2_OMP_FOR
505}

524void CIncNSSolver::BC_ConjugateHeat_Interface(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
525 CConfig *config, unsigned short val_marker) {
526
527 const su2double Temperature_Ref = config->GetTemperature_Ref();
528 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
529 const bool energy = config->GetEnergy_Equation();
530
531 /*--- Identify the boundary ---*/
532
533 const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
534
535 /*--- Retrieve the specified wall function treatment.---*/
536
537 if (config->GetWallFunction_Treatment(Marker_Tag) != WALL_FUNCTIONS::NONE) {
538 SU2_MPI::Error("Wall function treatment not implemented yet.", CURRENT_FUNCTION);
539 }
540
541 /*--- Loop over boundary points ---*/
542
543 SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
544 for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
545
546 auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
547
548 if (!geometry->nodes->GetDomain(iPoint)) continue;
549
550 /*--- Impose the value of the velocity as a strong boundary
551 condition (Dirichlet). Fix the velocity and remove any
552 contribution to the residual at this node. ---*/
553
554 if (dynamic_grid) {
555 nodes->SetVelocity_Old(iPoint, geometry->nodes->GetGridVel(iPoint));
556 } else {
557 su2double zero[MAXNDIM] = {0.0};
558 nodes->SetVelocity_Old(iPoint, zero);
559 }
560
561 for (unsigned short iDim = 0; iDim < nDim; iDim++)
562 LinSysRes(iPoint, iDim+1) = 0.0;
563 nodes->SetVel_ResTruncError_Zero(iPoint);
564
565 /*--- Enforce the no-slip boundary condition in a strong way by
566 modifying the velocity-rows of the Jacobian (1 on the diagonal). ---*/
567
568 if (implicit) {
569 for (unsigned short iVar = 1; iVar <= nDim; iVar++)
570 Jacobian.DeleteValsRowi(iPoint*nVar+iVar);
571 if (energy) Jacobian.DeleteValsRowi(iPoint*nVar+nDim+1);
572 }
573
574 if (!energy) continue;
575
576 su2double Tconjugate = GetConjugateHeatVariable(val_marker, iVertex, 0) / Temperature_Ref;

580 (config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {

583
584 const auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();

587
588 const auto Coord_i = geometry->nodes->GetCoord(iPoint);
589 const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);
590 su2double dist_ij = GeometryToolbox::Distance(nDim, Coord_j, Coord_i);

597 su2double HF_FactorConjugate = GetConjugateHeatVariable(val_marker, iVertex, 2);

611

613

615 nodes->SetSolution_Old(iPoint, nDim+1, Twall);
616 nodes->SetEnergy_ResTruncError_Zero(iPoint);

621void CIncNSSolver::SetTau_Wall_WF(CGeometry *geometry, CSolver **solver_container, const CConfig *config) {
622 /*--- The wall function implemented herein is based on Nichols and Nelson, AIAA J. v32 n6 2004. ---*/
623
624 unsigned long notConvergedCounter = 0; /*--- Counts the number of wall cells that are not converged ---*/
625 unsigned long smallYPlusCounter = 0; /*--- Counts the number of wall cells where y+ < 5 ---*/
626
627 const auto max_iter = config->GetwallModel_MaxIter();
628 const su2double relax = config->GetwallModel_RelFac();
629
630 /*--- Typical constants from boundary layer theory ---*/
631
632 const su2double kappa = config->GetwallModel_Kappa();
633 const su2double B = config->GetwallModel_B();
634
635 for (auto iMarker = 0u; iMarker < config->GetnMarker_All(); iMarker++) {
636
637 if (!config->GetViscous_Wall(iMarker)) continue;
638
639 /*--- Identify the boundary by string name ---*/
640
641 const auto Marker_Tag = config->GetMarker_All_TagBound(iMarker);
642
643 /*--- Jump to another BC if it is not wall function ---*/
644
645 if (config->GetWallFunction_Treatment(Marker_Tag) != WALL_FUNCTIONS::STANDARD_FUNCTION)
646 continue;
647
648 /*--- Loop over all of the vertices on this boundary marker ---*/
649
650 SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
651 for (auto iVertex = 0u; iVertex < geometry->nVertex[iMarker]; iVertex++) {
652
653 const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
654 const auto Point_Normal = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
655 /*--- On the finest mesh compute also on halo nodes to avoid communication of tau wall. ---*/
656 if ((!geometry->nodes->GetDomain(iPoint)) && !(MGLevel==MESH_0)) continue;
657
658 /*--- Get coordinates of the current vertex and nearest normal point ---*/
659
660 const auto Coord = geometry->nodes->GetCoord(iPoint);
661 const auto Coord_Normal = geometry->nodes->GetCoord(Point_Normal);
662
663 /*--- Compute dual-grid area and boundary normal ---*/
664
665 const auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
666
667 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
668
669 su2double UnitNormal[MAXNDIM] = {0.0};
670 for (auto iDim = 0u; iDim < nDim; iDim++)
671 UnitNormal[iDim] = -Normal[iDim]/Area;
672
673 /*--- Get the velocity, pressure, and temperature at the nearest
674 (normal) interior point. ---*/
675
676 su2double Vel[MAXNDIM] = {0.0};
677 for (auto iDim = 0u; iDim < nDim; iDim++)
678 Vel[iDim] = nodes->GetVelocity(Point_Normal,iDim);
679
680 /*--- Compute the wall-parallel velocity at first point off the wall ---*/
681
682 const su2double VelNormal = GeometryToolbox::DotProduct(int(MAXNDIM), Vel, UnitNormal);
683
684 su2double VelTang[MAXNDIM] = {0.0};
685 for (auto iDim = 0u; iDim < nDim; iDim++)
686 VelTang[iDim] = Vel[iDim] - VelNormal*UnitNormal[iDim];
687
688 const su2double VelTangMod = GeometryToolbox::Norm(int(MAXNDIM), VelTang);
689
690 /*--- Compute normal distance of the interior point from the wall ---*/
691
692 su2double WallDist[MAXNDIM] = {0.0};
693 GeometryToolbox::Distance(nDim, Coord, Coord_Normal, WallDist);
694
695 su2double WallDistMod = GeometryToolbox::Norm(int(MAXNDIM), WallDist);
696
697 su2double Density_Wall = nodes->GetDensity(iPoint);
698
699 /*--- Compute the shear stress at the wall in the regular fashion
700 * by using the stress tensor on the surface ---*/
701
702 su2double tau[MAXNDIM][MAXNDIM] = {{0.0}};
703 const su2double Lam_Visc_Wall = nodes->GetLaminarViscosity(iPoint);
704 su2double Eddy_Visc_Wall = nodes->GetEddyViscosity(iPoint);
705
706 CNumerics::ComputeStressTensor(nDim, tau, nodes->GetVelocityGradient(iPoint), Lam_Visc_Wall);
707
708 su2double TauTangent[MAXNDIM] = {0.0};
709 GeometryToolbox::TangentProjection(nDim, tau, UnitNormal, TauTangent);
710
711 const su2double WallShearStress = GeometryToolbox::Norm(int(MAXNDIM), TauTangent);
712
713 /*--- Calculate the quantities from boundary layer theory and
714 * iteratively solve for a new wall shear stress. Use the current wall
715 * shear stress as a starting guess for the wall function. ---*/
716
717 unsigned long counter = 0;
718 su2double diff = 1.0;
719 su2double U_Tau = max(1.0e-6,sqrt(WallShearStress/Density_Wall));
720 /*--- Use minimum y+ as defined in the config, in case the routine below for computing y+ does not converge ---*/
721 su2double Y_Plus = 0.99*config->GetwallModel_MinYPlus();
722
723 const su2double Y_Plus_Start = Density_Wall * U_Tau * WallDistMod / Lam_Visc_Wall;
724
725 /*--- Automatic switch off when y+ < "limit" according to Nichols & Nelson (2004) ---*/
726
727 if (Y_Plus_Start < config->GetwallModel_MinYPlus()) {
728 smallYPlusCounter++;
729 continue;
730 }
731
732 /*--- Convergence criterium for the Newton solver, note that 1e-10 is too large ---*/
733 const su2double tol = 1e-12;
734 while (fabs(diff) > tol) {
735
736 /*--- Friction velocity and u+ ---*/
737
738 const su2double U_Plus = VelTangMod / U_Tau;
739
740 /*--- Y+ defined by White & Christoph ---*/
741
742 const su2double kUp = kappa * U_Plus;
743
744 // incompressible adiabatic result
745 const su2double Y_Plus_White = exp(kUp) * exp(-kappa * B);
746
747 /*--- Spalding's universal form for the BL velocity with the
748 * outer velocity form of White & Christoph above. ---*/
749 Y_Plus = U_Plus + Y_Plus_White + (exp(-kappa * B)* (1.0 - kUp - 0.5 * kUp * kUp - kUp * kUp * kUp / 6.0));
750
751 /*--- incompressible formulation ---*/
752 Eddy_Visc_Wall = Lam_Visc_Wall * kappa*exp(-kappa*B) * (exp(kUp) -1.0 - kUp - kUp * kUp / 2.0);
753
754 Eddy_Visc_Wall = max(1.0e-6, Eddy_Visc_Wall);
755
756 /* --- Define function for Newton method to zero --- */
757
758 diff = (Density_Wall * U_Tau * WallDistMod / Lam_Visc_Wall) - Y_Plus;
759
760 /* --- Gradient of function defined above wrt U_Tau --- */
761
762 const su2double dyp_dup = 1.0 + exp(-kappa * B) * (kappa * exp(kUp) - kappa - kUp - 0.5 * kUp * kUp);
763 const su2double dup_dutau = - U_Plus / U_Tau;
764 const su2double grad_diff = Density_Wall * WallDistMod / Lam_Visc_Wall - dyp_dup * dup_dutau;
765
766 /* --- Newton Step --- */
767
768 U_Tau = U_Tau - relax*(diff / grad_diff);
769
770 counter++;
771 if (counter > max_iter) {
772 notConvergedCounter++;
773 // use some safe values for convergence
774 Y_Plus = 30.0;
775 Eddy_Visc_Wall = 1.0;
776 U_Tau = 1.0;
777 break;
778 }
779 }
780
781 /*--- Calculate an updated value for the wall shear stress
782 * using the y+ value, the definition of y+, and the definition of
783 * the friction velocity. ---*/
784 YPlus[iMarker][iVertex] = Y_Plus;
785 EddyViscWall[iMarker][iVertex] = Eddy_Visc_Wall;
786 UTau[iMarker][iVertex] = U_Tau;
787
788 const su2double Tau_Wall = (1.0/Density_Wall)*pow(Y_Plus*Lam_Visc_Wall/WallDistMod,2.0);
789
790 /*--- Store this value for the wall shear stress at the node. ---*/
791 nodes->SetTau_Wall(iPoint, Tau_Wall);
792
793 }
794 END_SU2_OMP_FOR
795 }
796
797 if (config->GetComm_Level() == COMM_FULL) {
798 static unsigned long globalCounter1, globalCounter2;
799
800 ompMasterAssignBarrier(globalCounter1,0, globalCounter2,0);
801
802 SU2_OMP_ATOMIC
803 globalCounter1 += notConvergedCounter;
804
805 SU2_OMP_ATOMIC
806 globalCounter2 += smallYPlusCounter;
807
808 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
809 SU2_MPI::Allreduce(&globalCounter1, &notConvergedCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
810 SU2_MPI::Allreduce(&globalCounter2, &smallYPlusCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
811
812 if (rank == MASTER_NODE) {
813 if (notConvergedCounter)
814 cout << "Warning: Computation of wall coefficients (y+) did not converge in "
815 << notConvergedCounter << " points." << endl;
816
817 if (smallYPlusCounter)
818 cout << "Warning: y+ < " << config->GetwallModel_MinYPlus() << " in " << smallYPlusCounter
819 << " points, for which the wall model is not active." << endl;
820 }
821 }
822 END_SU2_OMP_SAFE_GLOBAL_ACCESS
823 }
