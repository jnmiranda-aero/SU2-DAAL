139 void CDiscAdjDeformationDriver::Geometrical_Preprocessing() {
140  /*--- Loop over all zones to initialize the various classes. In most
141  * cases, nZone is equal to one. This represents the solution of a partial
142  * differential equation on a single block, unstructured mesh. ---*/
143 
144  unsigned short nMesh = 1;
145 
146  for (iZone = 0; iZone < nZone; iZone++) {
147  /*--- Determine if the FEM solver is used, which decides the type of geometry classes that are instantiated. ---*/
148 
149  const bool fem_solver = config_container[iZone]->GetFEMSolver();
150 
151  /*--- Read the number of instances for each zone. ---*/
152 
153  nInst[iZone] = config_container[iZone]->GetnTimeInstances();
154 
155  geometry_container[iZone] = new CGeometry**[nInst[iZone]];
156 
157  for (iInst = 0; iInst < nInst[iZone]; iInst++) {
158  /*--- Definition of the geometry class to store the primal grid in the partitioning process. ---*/
159 
160  CGeometry* geometry_aux = nullptr;
161 
162  /*--- All ranks process the grid and call ParMETIS for partitioning. ---*/
163 
164  geometry_aux = new CPhysicalGeometry(config_container[iZone], iZone, nZone);
165 
166  /*--- Color the initial grid and set the send-receive domains (ParMETIS). ---*/
167 
168  if (fem_solver) {
169  geometry_aux->SetColorFEMGrid_Parallel(config_container[iZone]);
170  } else {
171  geometry_aux->SetColorGrid_Parallel(config_container[iZone]);
172  }
173 
174  /*--- Build the grid data structures using the ParMETIS coloring. ---*/
175 
176  if (fem_solver) {
177  switch (config_container[iZone]->GetKind_FEM_Flow()) {
178  case DG:
179  geometry_container[iZone][iInst] = new CGeometry*[nMesh];
180  geometry_container[iZone][iInst][MESH_0] = new CMeshFEM_DG(geometry_aux, config_container[iZone]);
181  break;
182  }
183  } else {
184  geometry_container[iZone][iInst] = new CGeometry*[nMesh];
185  geometry_container[iZone][iInst][MESH_0] = new CPhysicalGeometry(geometry_aux, config_container[iZone]);
186  }
187 
188  /*--- Deallocate the memory of geometry_aux. ---*/
189 
190  delete geometry_aux;
191 
192  /*--- Add the Send/Receive boundaries. ---*/
193 
194  geometry_container[iZone][iInst][MESH_0]->SetSendReceive(config_container[iZone]);
195 
196  /*--- Add the Send/Receive boundaries. ---*/
197 
198  geometry_container[iZone][iInst][MESH_0]->SetBoundaries(config_container[iZone]);
199 
200  /*--- Create the vertex structure (required for MPI). ---*/
201 
202  if (rank == MASTER_NODE) cout << "Identify vertices." << endl;
203  geometry_container[iZone][iInst][MESH_0]->SetVertex(config_container[iZone]);
204 
205  /*--- Store the global to local mapping after preprocessing. ---*/
206 
207  if (rank == MASTER_NODE) cout << "Storing a mapping from global to local point index." << endl;
208  geometry_container[iZone][iInst][MESH_0]->SetGlobal_to_Local_Point();
209 
210  /*--- Test for a fem solver, because some more work must be done. ---*/
211 
212  if (fem_solver) {
213  /*--- Carry out a dynamic cast to CMeshFEM_DG, such that it is not needed to
214  * define all virtual functions in the base class CGeometry. ---*/
215 
216  auto* DGMesh = dynamic_cast<CMeshFEM_DG*>(geometry_container[iZone][iInst][MESH_0]);
217 
218  /*--- Determine the standard elements for the volume elements. ---*/
219 
220  if (rank == MASTER_NODE) cout << "Creating standard volume elements." << endl;
221  DGMesh->CreateStandardVolumeElements(config_container[iZone]);
222 
223  /*--- Create the face information needed to compute the contour integral
224  * for the elements in the Discontinuous Galerkin formulation. ---*/
225 
226  if (rank == MASTER_NODE) cout << "Creating face information." << endl;
227  DGMesh->CreateFaces(config_container[iZone]);
228  }
229  }
230 
231  if (rank == MASTER_NODE)
232  cout << "\n----------------------- Preprocessing computations ----------------------" << endl;
233 
234  /*--- Compute elements surrounding points, points surrounding points. ---*/
235 
236  if (rank == MASTER_NODE) cout << "Setting local point connectivity." << endl;
237  geometry_container[iZone][INST_0][MESH_0]->SetPoint_Connectivity();
238 
239  /*--- Check the orientation before computing geometrical quantities. ---*/
240 
241  geometry_container[iZone][INST_0][MESH_0]->SetBoundVolume();
242  if (config_container[iZone]->GetReorientElements()) {
243  if (rank == MASTER_NODE) cout << "Checking the numerical grid orientation of the elements." << endl;
244  geometry_container[iZone][INST_0][MESH_0]->Check_IntElem_Orientation(config_container[iZone]);
245  geometry_container[iZone][INST_0][MESH_0]->Check_BoundElem_Orientation(config_container[iZone]);
246  }
247 
248  /*--- Create the edge structure. ---*/
249 
250  if (rank == MASTER_NODE) cout << "Identify edges and vertices." << endl;
251  geometry_container[iZone][INST_0][MESH_0]->SetEdges();
252  geometry_container[iZone][INST_0][MESH_0]->SetVertex(config_container[iZone]);
253 
254  /*--- Create the dual control volume structures. ---*/
255 
256  if (rank == MASTER_NODE) cout << "Setting the bound control volume structure." << endl;
257  geometry_container[iZone][INST_0][MESH_0]->SetBoundControlVolume(config_container[ZONE_0], ALLOCATE);
258 
259  /*--- Store the global to local mapping after preprocessing. ---*/
260 
261  if (rank == MASTER_NODE) cout << "Storing a mapping from global to local point index." << endl;
262  geometry_container[iZone][INST_0][MESH_0]->SetGlobal_to_Local_Point();
263 
264  /*--- Create the point-to-point MPI communication structures. ---*/
265 
266  geometry_container[iZone][INST_0][MESH_0]->PreprocessP2PComms(geometry_container[iZone][INST_0][MESH_0],
267  config_container[iZone]);
268  }
269 
270  /*--- Keep a reference to the main (ZONE_0, INST_0, MESH_0) geometry. ---*/
271 
272  main_geometry = geometry_container[ZONE_0][INST_0][MESH_0];
273 }

301 void CDiscAdjDeformationDriver::Run() {
302  for (iZone = 0; iZone < nZone; iZone++) {
303  if (!config_container[iZone]->GetDiscrete_Adjoint()) {
304  continue;
305  }
306  if (rank == MASTER_NODE)
307  cout << "\n---------------------- Mesh sensitivity computation ---------------------" << endl;
308  if (config_container[iZone]->GetDiscrete_Adjoint() && config_container[iZone]->GetSmoothGradient() &&
309  config_container[iZone]->GetSobMode() == ENUM_SOBOLEV_MODUS::MESH_LEVEL) {
310  DerivativeTreatment_MeshSensitivity(geometry_container[iZone][INST_0][MESH_0], config_container[iZone],
311  grid_movement[iZone][INST_0]);
312  } else {
313  grid_movement[iZone][INST_0]->SetVolume_Deformation(geometry_container[iZone][INST_0][MESH_0],
314  config_container[iZone], false, true);
315  }
316  }
317 
318  if (config_container[ZONE_0]->GetDiscrete_Adjoint()) {
319  if (rank == MASTER_NODE)
320  cout << "\n------------------------ Mesh sensitivity Output ------------------------" << endl;
321  SetSensitivity_Files(geometry_container, config_container, nZone);
322  }
323 
324  ofstream Gradient_file;
325  Gradient_file.precision(driver_config->OptionIsSet("OUTPUT_PRECISION") ? driver_config->GetOutput_Precision() : 6);
326 
327  /*--- For multi-zone computations the gradient contributions are summed up and written into one file. ---*/
328 
329  for (iZone = 0; iZone < nZone; iZone++) {
330  if ((config_container[iZone]->GetDesign_Variable(0) != NONE) &&
331  (config_container[iZone]->GetDesign_Variable(0) != SURFACE_FILE)) {
332  if (rank == MASTER_NODE)
333  cout << "\n---------- Start gradient evaluation using sensitivity information ----------" << endl;
334 
335  /*--- Definition of the Class for surface deformation. ---*/
336 
337  surface_movement[iZone] = new CSurfaceMovement();
338 
339  /*--- Copy coordinates to the surface structure. ---*/
340 
341  surface_movement[iZone]->CopyBoundary(geometry_container[iZone][INST_0][MESH_0], config_container[iZone]);
342 
343  /*--- If AD mode is enabled we can use it to compute the projection, otherwise we use finite differences. ---*/
344 
345  if (config_container[iZone]->GetAD_Mode()) {
346  if (config_container[iZone]->GetSmoothGradient()) {
347  DerivativeTreatment_Gradient(geometry_container[iZone][INST_0][MESH_0], config_container[iZone],
348  grid_movement[iZone][INST_0], surface_movement[iZone], Gradient);
349  } else {
350  SetProjection_AD(geometry_container[iZone][INST_0][MESH_0], config_container[iZone], surface_movement[iZone],
351  Gradient);
352  }
353  } else {
354  SetProjection_FD(geometry_container[iZone][INST_0][MESH_0], config_container[iZone], surface_movement[iZone],
355  Gradient);
356  }
357  }
358  }
359 
360  /*--- Write the gradient to a file. ---*/
361 
362  if (rank == MASTER_NODE) Gradient_file.open(config_container[ZONE_0]->GetObjFunc_Grad_FileName().c_str(), ios::out);
363 
364  /*--- Print gradients to screen and writes to file. ---*/
365 
366  OutputGradient(Gradient, config_container[ZONE_0], Gradient_file);
367 }

385 void CDiscAdjDeformationDriver::SetProjection_FD(CGeometry* geometry, CConfig* config,
386  CSurfaceMovement* surface_movement, su2double** Gradient) {
387  unsigned short iDV, nDV, iFFDBox, nDV_Value, iMarker, iDim;
388  unsigned long iVertex, iPoint;
389  su2double delta_eps, my_Gradient, localGradient, *Normal, dS, *VarCoord, Sensitivity, dalpha[3], deps[3], dalpha_deps;
390  bool *UpdatePoint, MoveSurface, Local_MoveSurface;
391  CFreeFormDefBox** FFDBox;
392 
393  int rank = SU2_MPI::GetRank();
394 
395  nDV = config->GetnDV();
396 
397  /*--- Boolean controlling points to be updated. ---*/
398 
399  UpdatePoint = new bool[geometry->GetnPoint()];
400 
401  /*--- Definition of the FFD deformation class. ---*/
402 
403  unsigned short nFFDBox = MAX_NUMBER_FFD;
404  FFDBox = new CFreeFormDefBox*[nFFDBox];
405  for (iFFDBox = 0; iFFDBox < MAX_NUMBER_FFD; iFFDBox++) FFDBox[iFFDBox] = nullptr;
406 
407  for (iDV = 0; iDV < nDV; iDV++) {
408  nDV_Value = config->GetnDV_Value(iDV);
409  if (nDV_Value != 1) {
410  SU2_MPI::Error(
411  "The projection using finite differences currently only supports a fixed direction of movement for FFD "
412  "points.",
413  CURRENT_FUNCTION);
414  }
415  }
416 
417  /*--- Continuous adjoint gradient computation. ---*/
418 
419  if (rank == MASTER_NODE) cout << "Evaluate functional gradient using Finite Differences." << endl;
420 
421  for (iDV = 0; iDV < nDV; iDV++) {
422  MoveSurface = true;
423  Local_MoveSurface = true;
424 
425  /*--- Free form deformation based. ---*/
426 
427  if ((config->GetDesign_Variable(iDV) == FFD_CONTROL_POINT_2D) ||
428  (config->GetDesign_Variable(iDV) == FFD_CAMBER_2D) || (config->GetDesign_Variable(iDV) == FFD_THICKNESS_2D) ||
429  (config->GetDesign_Variable(iDV) == FFD_CONTROL_POINT) || (config->GetDesign_Variable(iDV) == FFD_NACELLE) ||
430  (config->GetDesign_Variable(iDV) == FFD_GULL) || (config->GetDesign_Variable(iDV) == FFD_TWIST) ||
431  (config->GetDesign_Variable(iDV) == FFD_ROTATION) || (config->GetDesign_Variable(iDV) == FFD_CAMBER) ||
432  (config->GetDesign_Variable(iDV) == FFD_THICKNESS) ||
433  (config->GetDesign_Variable(iDV) == FFD_ANGLE_OF_ATTACK)) {
434  /*--- Read the FFD information in the first iteration. ---*/
435 
436  if (iDV == 0) {
437  if (rank == MASTER_NODE) cout << "Read the FFD information from mesh file." << endl;
438 
439  /*--- Read the FFD information from the grid file. ---*/
440 
441  surface_movement->ReadFFDInfo(geometry, config, FFDBox, config->GetMesh_FileName());
442 
443  /*--- If the FFDBox was not defined in the input file. ---*/
444 
445  if (!surface_movement->GetFFDBoxDefinition()) {
446  SU2_MPI::Error("The input grid doesn't have the entire FFD information!", CURRENT_FUNCTION);
447  }
448 
449  for (iFFDBox = 0; iFFDBox < surface_movement->GetnFFDBox(); iFFDBox++) {
450  if (rank == MASTER_NODE) cout << "Checking FFD box dimension." << endl;
451  surface_movement->CheckFFDDimension(geometry, config, FFDBox[iFFDBox], iFFDBox);
452 
453  if (rank == MASTER_NODE) cout << "Check the FFD box intersections with the solid surfaces." << endl;
454  surface_movement->CheckFFDIntersections(geometry, config, FFDBox[iFFDBox], iFFDBox);
455  }
456 
457  if (rank == MASTER_NODE)
458  cout << "-------------------------------------------------------------------------" << endl;
459  }
460 
461  if (rank == MASTER_NODE) {
462  cout << endl << "Design variable number " << iDV << "." << endl;
463  cout << "Performing 3D deformation of the surface." << endl;
464  }
465 
466  /*--- Apply the control point change. ---*/
467 
468  MoveSurface = false;
469 
470  for (iFFDBox = 0; iFFDBox < surface_movement->GetnFFDBox(); iFFDBox++) {
471  /*--- Reset FFD box. ---*/
472 
473  switch (config->GetDesign_Variable(iDV)) {
474  case FFD_CONTROL_POINT_2D:
475  Local_MoveSurface =
476  surface_movement->SetFFDCPChange_2D(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
477  break;
478  case FFD_CAMBER_2D:
479  Local_MoveSurface = surface_movement->SetFFDCamber_2D(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
480  break;
481  case FFD_THICKNESS_2D:
482  Local_MoveSurface =
483  surface_movement->SetFFDThickness_2D(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
484  break;
485  case FFD_CONTROL_POINT:
486  Local_MoveSurface = surface_movement->SetFFDCPChange(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
487  break;
488  case FFD_NACELLE:
489  Local_MoveSurface = surface_movement->SetFFDNacelle(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
490  break;
491  case FFD_GULL:
492  Local_MoveSurface = surface_movement->SetFFDGull(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
493  break;
494  case FFD_TWIST:
495  Local_MoveSurface = surface_movement->SetFFDTwist(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
496  break;
497  case FFD_ROTATION:
498  Local_MoveSurface = surface_movement->SetFFDRotation(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
499  break;
500  case FFD_CAMBER:
501  Local_MoveSurface = surface_movement->SetFFDCamber(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
502  break;
503  case FFD_THICKNESS:
504  Local_MoveSurface = surface_movement->SetFFDThickness(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
505  break;
506  case FFD_CONTROL_SURFACE:
507  Local_MoveSurface =
508  surface_movement->SetFFDControl_Surface(geometry, config, FFDBox[iFFDBox], FFDBox, iDV, true);
509  break;
510  case FFD_ANGLE_OF_ATTACK:
511  Gradient[iDV][0] = config->GetAoA_Sens();
512  break;
513  }
514 
515  /*--- Recompute cartesian coordinates using the new control points position. ---*/
516 
517  if (Local_MoveSurface) {
518  MoveSurface = true;
519  surface_movement->SetCartesianCoord(geometry, config, FFDBox[iFFDBox], iFFDBox, true);
520  }
521  }
522  }
523 
524  /*--- Hicks-Henne design variable. ---*/
525 
526  else if (config->GetDesign_Variable(iDV) == HICKS_HENNE) {
527  surface_movement->SetHicksHenne(geometry, config, iDV, true);
528  }
529 
530  /*--- Surface bump design variable. ---*/
531 
532  else if (config->GetDesign_Variable(iDV) == SURFACE_BUMP) {
533  surface_movement->SetSurface_Bump(geometry, config, iDV, true);
534  }
535 
536  /*--- Kulfan (CST) design variable. ---*/
537 
538  else if (config->GetDesign_Variable(iDV) == CST) {
539  surface_movement->SetCST(geometry, config, iDV, true);
540  }
541 
542  /*--- Displacement design variable. ---*/
543 
544  else if (config->GetDesign_Variable(iDV) == TRANSLATION) {
545  surface_movement->SetTranslation(geometry, config, iDV, true);
546  }
547 
548  /*--- Angle of Attack design variable. ---*/
549 
550  else if (config->GetDesign_Variable(iDV) == ANGLE_OF_ATTACK) {
551  Gradient[iDV][0] = config->GetAoA_Sens();
552  }
553 
554  /*--- Scale design variable. ---*/
555 
556  else if (config->GetDesign_Variable(iDV) == SCALE) {
557  surface_movement->SetScale(geometry, config, iDV, true);
558  }
559 
560  /*--- Rotation design variable. ---*/
561 
562  else if (config->GetDesign_Variable(iDV) == ROTATION) {
563  surface_movement->SetRotation(geometry, config, iDV, true);
564  }
565 
566  /*--- NACA_4Digits design variable. ---*/
567

587  (config->GetDesign_Variable(iDV) != FFD_ANGLE_OF_ATTACK)) {

648  CSurfaceMovement* surface_movement, su2double** Gradient) {

650  unsigned short iDV_Value = 0, iMarker, nMarker, iDim, nDim, iDV, nDV;

654

657  nDV = config->GetnDV();
658 
659  /*--- Discrete adjoint gradient computation. ---*/
660 
661  if (rank == MASTER_NODE)
662  cout << endl
663  << "Evaluate functional gradient using Algorithmic Differentiation (ZONE " << config->GetiZone() << ")."
664  << endl;
665 
666  /*--- Start recording of operations. ---*/
667 
668  AD::StartRecording();
669 
670  /*--- Register design variables as input and set them to zero
671  * (since we want to have the derivative at alpha = 0, i.e. for the current design). ---*/
672 
673  for (iDV = 0; iDV < nDV; iDV++) {
674  for (iDV_Value = 0; iDV_Value < config->GetnDV_Value(iDV); iDV_Value++) {
675  config->SetDV_Value(iDV, iDV_Value, 0.0);
676 
677  AD::RegisterInput(config->GetDV_Value(iDV, iDV_Value));
678  }
679  }
680

682 
683  surface_movement->SetSurface_Deformation(geometry, config);
684 
685  /*--- Stop the recording. --- */
686 
687  AD::StopRecording();
688 
689  /*--- Create a structure to identify points that have been already visited.
690  * We need that to make sure to set the sensitivity of surface points only once.
691  * Markers share points, so we would visit them more than once in the loop over the markers below). ---*/
692 
693  vector<bool> visited(geometry->GetnPoint(), false);
694 
695  /*--- Initialize the derivatives of the output of the surface deformation routine
696  * with the discrete adjoints from the CFD solution. ---*/
697 
698  for (iMarker = 0; iMarker < nMarker; iMarker++) {
699  if (config->GetMarker_All_DV(iMarker) == YES) {
700  nVertex = geometry->nVertex[iMarker];
701  for (iVertex = 0; iVertex < nVertex; iVertex++) {
702  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
703  if (!visited[iPoint]) {
704  VarCoord = geometry->vertex[iMarker][iVertex]->GetVarCoord();
705  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
706 
707  Area = 0.0;
708  for (iDim = 0; iDim < nDim; iDim++) {
709  Area += Normal[iDim] * Normal[iDim];
710  }
711  Area = sqrt(Area);
712 
713  for (iDim = 0; iDim < nDim; iDim++) {
714  if (config->GetDiscrete_Adjoint()) {
715  Sensitivity = geometry->GetSensitivity(iPoint, iDim);
716  } else {
717  Sensitivity = -Normal[iDim] * geometry->vertex[iMarker][iVertex]->GetAuxVar() / Area;
718  }
719  SU2_TYPE::SetDerivative(VarCoord[iDim], SU2_TYPE::GetValue(Sensitivity));
720  }
721  visited[iPoint] = true;
722  }
723  }
724  }
725  }
726 
727  /*--- Compute derivatives and extract gradient. ---*/
728 
729  AD::ComputeAdjoint();
730 
731  for (iDV = 0; iDV < nDV; iDV++) {
732  for (iDV_Value = 0; iDV_Value < config->GetnDV_Value(iDV); iDV_Value++) {
733  my_Gradient = SU2_TYPE::GetDerivative(config->GetDV_Value(iDV, iDV_Value));
734 
735  SU2_MPI::Allreduce(&my_Gradient, &localGradient, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
736 
737  /*--- Angle of Attack design variable (this is different, the value comes form the input file). ---*/
738 
739  if ((config->GetDesign_Variable(iDV) == ANGLE_OF_ATTACK) ||
740  (config->GetDesign_Variable(iDV) == FFD_ANGLE_OF_ATTACK)) {
741  Gradient[iDV][iDV_Value] = config->GetAoA_Sens();
742  }
743 
744  Gradient[iDV][iDV_Value] += localGradient;
745  }
746  }
747 
748  AD::Reset();

797  unsigned short val_nZone) {

806 
807  for (iZone = 0; iZone < val_nZone; iZone++) {

809  nDim = geometry[iZone][INST_0][MESH_0]->GetnDim();
810  nMarker = config[iZone]->GetnMarker_All();
