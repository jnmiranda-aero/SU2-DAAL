33/*!
34 * \class CIncEulerVariable
35 * \brief Class for defining the variables of the incompressible Euler solver.
36 * \note Primitive variables (P, vx, vy, vz, T, rho, beta, lamMu, EddyMu, Kt_eff, Cp, Cv)
37 * \note Gradients of primitives (P, vx, vy, vz, T, rho, beta)
38 * \ingroup Euler_Equations

43 static constexpr size_t MAXNVAR = 12;

45 template <class IndexType>
46 struct CIndices {
47 const IndexType nDim;
48 CIndices(IndexType ndim, IndexType) : nDim(ndim) {}
49 inline IndexType NDim() const { return nDim; }
50 inline IndexType NSpecies() const { return 0; }
51 inline IndexType Pressure() const { return 0; }
52 inline IndexType Velocity() const { return 1; }
53 inline IndexType Temperature() const { return nDim+1; }
54 inline IndexType Density() const { return nDim+2; }
55 inline IndexType Beta() const { return nDim+3; }
56 inline IndexType SoundSpeed() const { return Beta(); }
57 inline IndexType LaminarViscosity() const { return nDim+4; }
58 inline IndexType EddyViscosity() const { return nDim+5; }
59 inline IndexType ThermalConductivity() const { return nDim+6; }
60 inline IndexType CpTotal() const { return nDim+7; }
61 inline IndexType CvTotal() const { return nDim+8; }
62
63 /*--- For compatible interface with NEMO. ---*/
64 inline IndexType SpeciesDensities() const { return std::numeric_limits<IndexType>::max(); }
65 inline IndexType Temperature_ve() const { return std::numeric_limits<IndexType>::max(); }
66 inline IndexType Enthalpy() const { return std::numeric_limits<IndexType>::max(); }
67 };

88 /*!

94 /*!
95 * \brief Set the value of the density for the incompressible flows.
96 * \param[in] iPoint - Point index.
97 */
98 inline bool SetDensity(unsigned long iPoint, su2double val_density) final {
99 Primitive(iPoint, indices.Density()) = val_density;
100 return val_density <= 0.0;

102
103 /*!
104 * \brief Set the value of the density for the incompressible flows.
105 * \param[in] iPoint - Point index.
106 */
107 inline void SetVelocity(unsigned long iPoint) final {
108 Velocity2(iPoint) = 0.0;
109 for (unsigned long iDim = 0; iDim < nDim; iDim++) {
110 Primitive(iPoint, iDim+indices.Velocity()) = Solution(iPoint,iDim+1);
111 Velocity2(iPoint) += pow(Primitive(iPoint, iDim+indices.Velocity()), 2);
112 }
113 }

173 /*!
174 * \brief Get the projected velocity in a unitary vector direction (compressible solver).
175 * \param[in] val_vector - Direction of projection.
176 * \return Value of the projected velocity.
177 */
178 inline su2double GetProjVel(unsigned long iPoint, const su2double *val_vector) const final {
179 su2double ProjVel = 0.0;
180 for (unsigned long iDim = 0; iDim < nDim; iDim++)
181 ProjVel += Primitive(iPoint, iDim+indices.Velocity())*val_vector[iDim];
182 return ProjVel;
