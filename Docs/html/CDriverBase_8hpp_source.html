357 /*!
358 * \brief Get the displacements currently imposed of a marker vertex.
359 * \param[in] iMarker - Marker index.
360 * \param[in] iVertex - Marker vertex index.
361 * \return Node displacements (nDim).
362 */
363 inline vector<passivedouble> GetMarkerDisplacement(unsigned short iMarker, unsigned long iVertex) const {
364 vector<passivedouble> disp(GetNumberDimensions(), 0.0);
365 const auto iPoint = GetMarkerNode(iMarker, iVertex);
366 auto* nodes = GetSolverAndCheckMarker(MESH_SOL)->GetNodes();
367
368 for (auto iDim = 0u; iDim < GetNumberDimensions(); ++iDim) {
369 disp[iDim] = SU2_TYPE::GetValue(nodes->GetBound_Disp(iPoint, iDim));
370 }
371 return disp;
372 }

617 /*!
618 * \brief Set the adjoint of the flow tractions of the flow solver.
619 * \note This can be the input of the flow solver in an adjoint FSI setting.
620 * \param[in] iMarker - Marker identifier.
621 * \param[in] iVertex - Vertex identifier.
622 * \param[in] adjointLoad - Vector of adjoint loads.
623 */
624 inline void SetMarkerCustomFlowLoadAdjoint(unsigned short iMarker, unsigned long iVertex,
625 vector<passivedouble> adjointLoad) {
