583  /*!
584  * \brief Sets the nodal force for the structural solver at a vertex of a marker.
585  * \note This can be the input of the FEA solver in an FSI setting.
586  * \param[in] iMarker - Marker identifier.
587  * \param[in] iVertex - Vertex identifier.
588  * \param[in] force - Force vector.
589  */
590  inline void SetMarkerCustomFEALoad(unsigned short iMarker, unsigned long iVertex, std::vector<passivedouble> force) {
591  auto* solver = GetSolverAndCheckMarker(FEA_SOL, iMarker);
