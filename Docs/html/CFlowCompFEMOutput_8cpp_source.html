158void CFlowCompFEMOutput::LoadVolumeDataFEM(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iElem, unsigned long index, unsigned short dof){
159
160 unsigned short iDim;
161
162 unsigned short nVar = solver[FLOW_SOL]->GetnVar();
163
164 /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
165 geometrical information for the FEM DG solver. ---*/
166
167 auto *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
168
169 CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
170
171 /*--- Get a pointer to the fluid model class from the DG-FEM solver
172 so that we can access the states below. ---*/
173
174 CFluidModel *DGFluidModel = solver[FLOW_SOL]->GetFluidModel();
175
176 /* Set the pointers for the solution for this element. */
177
178 const unsigned long offset = nVar*volElem[iElem].offsetDOFsSolLocal;
179 su2double *solDOFs = solver[FLOW_SOL]->GetVecSolDOFs() + offset;
180
181 /*--- Get the conservative variables for this particular DOF. ---*/
182
183 const su2double *U = solDOFs+dof*nVar;
184
185 /*--- Load the coordinate values of the solution DOFs. ---*/
186
187 const su2double *coor = volElem[iElem].coorSolDOFs.data() + dof*nDim;
188
189 /*--- Prepare the primitive states. ---*/
190
191 const su2double DensityInv = 1.0/U[0];
192 su2double vel[3], Velocity2 = 0.0;
193 for(iDim=0; iDim<nDim; ++iDim) {
194 vel[iDim] = U[iDim+1]*DensityInv;
195 Velocity2 += vel[iDim]*vel[iDim];
196 }
197 su2double StaticEnergy = U[nDim+1]*DensityInv - 0.5*Velocity2;
198 DGFluidModel->SetTDState_rhoe(U[0], StaticEnergy);
199
200
201 SetVolumeOutputValue("COORD-X", index, coor[0]);
202 SetVolumeOutputValue("COORD-Y", index, coor[1]);
203 if (nDim == 3)
204 SetVolumeOutputValue("COORD-Z", index, coor[2]);
205 SetVolumeOutputValue("DENSITY", index, U[0]);
206 SetVolumeOutputValue("MOMENTUM-X", index, U[1]);
207 SetVolumeOutputValue("MOMENTUM-Y", index, U[2]);
208 if (nDim == 3){
209 SetVolumeOutputValue("MOMENTUM-Z", index, U[3]);
210 SetVolumeOutputValue("ENERGY", index, U[4]);
211 } else {
212 SetVolumeOutputValue("ENERGY", index, U[3]);
213 }
214
215 SetVolumeOutputValue("PRESSURE", index, DGFluidModel->GetPressure());
216 SetVolumeOutputValue("TEMPERATURE", index, DGFluidModel->GetTemperature());
217 SetVolumeOutputValue("MACH", index, sqrt(Velocity2)/DGFluidModel->GetSoundSpeed());
218 SetVolumeOutputValue("PRESSURE_COEFF", index, DGFluidModel->GetCp());
219
220 if (config->GetKind_Solver() == MAIN_SOLVER::FEM_NAVIER_STOKES){
221 SetVolumeOutputValue("LAMINAR_VISCOSITY", index, DGFluidModel->GetLaminarViscosity());
222 }
223 if ((config->GetKind_Solver() == MAIN_SOLVER::FEM_LES) && (config->GetKind_SGS_Model() != TURB_SGS_MODEL::IMPLICIT_LES)){
224 // todo: Export Eddy instead of Laminar viscosity
225 SetVolumeOutputValue("EDDY_VISCOSITY", index, DGFluidModel->GetLaminarViscosity());
226 }
227}

229void CFlowCompFEMOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
230
231 CSolver* flow_solver = solver[FLOW_SOL];
232
233 SetHistoryOutputValue("RMS_DENSITY", log10(flow_solver->GetRes_RMS(0)));
234 SetHistoryOutputValue("RMS_MOMENTUM-X", log10(flow_solver->GetRes_RMS(1)));
235 SetHistoryOutputValue("RMS_MOMENTUM-Y", log10(flow_solver->GetRes_RMS(2)));
236 if (nDim == 2)
237 SetHistoryOutputValue("RMS_ENERGY", log10(flow_solver->GetRes_RMS(3)));
238 else {
239 SetHistoryOutputValue("RMS_MOMENTUM-Z", log10(flow_solver->GetRes_RMS(3)));
240 SetHistoryOutputValue("RMS_ENERGY", log10(flow_solver->GetRes_RMS(4)));
241 }
242
243
244 SetHistoryOutputValue("MAX_DENSITY", log10(flow_solver->GetRes_Max(0)));
245 SetHistoryOutputValue("MAX_MOMENTUM-X", log10(flow_solver->GetRes_Max(1)));
246 SetHistoryOutputValue("MAX_MOMENTUM-Y", log10(flow_solver->GetRes_Max(2)));
247 if (nDim == 2)
248 SetHistoryOutputValue("MAX_ENERGY", log10(flow_solver->GetRes_Max(3)));
249 else {
250 SetHistoryOutputValue("MAX_MOMENTUM-Z", log10(flow_solver->GetRes_Max(3)));
251 SetHistoryOutputValue("MAX_ENERGY", log10(flow_solver->GetRes_Max(4)));
252 }
253
254 SetHistoryOutputValue("AOA", config->GetAoA());
255 SetHistoryOutputValue("CFL_NUMBER", config->GetCFL(MESH_0));
256
257 if (config->GetnMarker_Analyze() > 0) {
258 SU2_MPI::Error("SetAnalyzeSurface is not implemented for FEM-DG solver.", CURRENT_FUNCTION);
259 }
260
261 /*--- Set aeroydnamic coefficients --- */
262
263 SetAerodynamicCoefficients(config, flow_solver);
264
265 ComputeSimpleCustomOutputs(config);
266}
