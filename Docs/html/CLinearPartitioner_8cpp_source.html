31void CLinearPartitioner::Initialize(unsigned long global_count, unsigned long offset, bool isDisjoint) {
32 /*--- Store MPI size ---*/
33
34 size = SU2_MPI::GetSize();
35
36 /*--- Resize the vectors containing our linear partitioning. ---*/
37
38 firstIndex.resize(size);
39 lastIndex.resize(size);
40 sizeOnRank.resize(size);
41 cumulativeSizeBeforeRank.resize(size + 1);
42
43 /*--- Compute the number of points that will be on each processor.
44 This is a linear partitioning with the addition of a simple load
45 balancing for any remainder points. ---*/
46
47 unsigned long quotient = 0;
48 if (global_count >= (unsigned long)size) quotient = global_count / size;
49
50 int remainder = int(global_count % size);
51 for (int ii = 0; ii < size; ii++) {
52 sizeOnRank[ii] = quotient + int(ii < remainder);
53 }
54
55 /*--- Store the local number of nodes on each proc in the linear
56 partitioning, the beginning/end index, and the linear partitioning
57 within an array in cumulative storage format. ---*/
58
59 unsigned long adjust = 0;
60 if (isDisjoint) adjust = 1;
61
62 firstIndex[0] = offset;
63 lastIndex[0] = firstIndex[0] + sizeOnRank[0] - adjust;
64 cumulativeSizeBeforeRank[0] = 0;
65 for (int iProc = 1; iProc < size; iProc++) {
66 firstIndex[iProc] = lastIndex[iProc - 1] + adjust;
67 lastIndex[iProc] = firstIndex[iProc] + sizeOnRank[iProc] - adjust;
68 cumulativeSizeBeforeRank[iProc] = (cumulativeSizeBeforeRank[iProc - 1] + sizeOnRank[iProc - 1]);
69 }
70 cumulativeSizeBeforeRank[size] = global_count;
71}

73unsigned long CLinearPartitioner::GetRankContainingIndex(unsigned long index) const {
74 /*--- Initial guess ---*/
75
76 unsigned long iProcessor = min<unsigned long>(index / sizeOnRank[0], size - 1);
77
78 /*--- Move up or down until we find the processor. ---*/
79
80 if (index >= cumulativeSizeBeforeRank[iProcessor])
81 while (index >= cumulativeSizeBeforeRank[iProcessor + 1]) iProcessor++;
82 else
83 while (index < cumulativeSizeBeforeRank[iProcessor]) iProcessor--;
84
85 return iProcessor;
86}
