34CMultizoneDriver::CMultizoneDriver(char* confFile, unsigned short val_nZone, SU2_Comm MPICommunicator) :
35 CDriver(confFile, val_nZone, MPICommunicator, false) {
36
37 /*--- Initialize the counter for TimeIter ---*/
38 TimeIter = 0;
39
40 /*--- Structure for multizone convergence ---*/
41 init_res = new su2double*[nZone];
42 residual = new su2double*[nZone];
43 residual_rel = new su2double*[nZone];
44 nVarZone = new unsigned short[nZone];
45
46 for (iZone = 0; iZone < nZone; iZone++){
47 nVarZone[iZone] = 0;
48 /*--- Account for all the solvers ---*/
49 for (unsigned short iSol = 0; iSol < MAX_SOLS; iSol++){
50 if (solver_container[iZone][INST_0][MESH_0][iSol] != nullptr)
51 nVarZone[iZone] += solver_container[iZone][INST_0][MESH_0][iSol]->GetnVar();
52 }
53 init_res[iZone] = new su2double[nVarZone[iZone]];
54 residual[iZone] = new su2double[nVarZone[iZone]];
55 residual_rel[iZone] = new su2double[nVarZone[iZone]];
56 /*--- Initialize the residual containers to 0.0 ---*/
57 for (unsigned short iVar = 0; iVar < nVarZone[iZone]; iVar++){
58 init_res[iZone][iVar] = 0.0;
59 residual[iZone][iVar] = 0.0;
60 residual_rel[iZone][iVar] = 0.0;
61 }
62 }
63
64 /*----------------------------------------------------*/
65 /*------ Determine the properties of the problem -----*/
66 /*----------------------------------------------------*/
67
68 bool structural_zone = false;
69 bool fluid_zone = false;
70 bool heat_zone = false;
71
72 /*--- If there is at least a fluid and a structural zone ---*/
73 for (iZone = 0; iZone < nZone; iZone++){
74 switch (config_container[iZone]->GetKind_Solver()) {
75 case MAIN_SOLVER::EULER: case MAIN_SOLVER::NAVIER_STOKES: case MAIN_SOLVER::RANS:
76 case MAIN_SOLVER::NEMO_EULER: case MAIN_SOLVER::NEMO_NAVIER_STOKES:
77 case MAIN_SOLVER::INC_EULER: case MAIN_SOLVER::INC_NAVIER_STOKES: case MAIN_SOLVER::INC_RANS:
78 fluid_zone = true;
79 break;
80 case MAIN_SOLVER::FEM_ELASTICITY:
81 structural_zone = true;
82 break;
83 case MAIN_SOLVER::HEAT_EQUATION:
84 heat_zone = true;
85 break;
86 default:
87 break;
88 }
89 }
90
91 fsi = false; cht = false;
92 /*--- If the problem has FSI properties ---*/
93 if (fluid_zone && structural_zone) fsi = true;
94 /*--- If the problem has CHT properties ---*/
95 if (fluid_zone && heat_zone) cht = true;
96
97 /*----------------------------------------------------*/
98 /*- Define if a prefixed motion is imposed in a zone -*/
99 /*----------------------------------------------------*/
100
101 prefixed_motion = new bool[nZone];
102 for (iZone = 0; iZone < nZone; iZone++){
103 switch (config_container[iZone]->GetKind_GridMovement()){
104 case RIGID_MOTION:
105 prefixed_motion[iZone] = true; break;
106 default:
107 prefixed_motion[iZone] = false; break;
108 }
109 if (config_container[iZone]->GetSurface_Movement(AEROELASTIC) ||
110 config_container[iZone]->GetSurface_Movement(AEROELASTIC_RIGID_MOTION) ||
111 config_container[iZone]->GetSurface_Movement(DEFORMING) ||
112 config_container[iZone]->GetSurface_Movement(EXTERNAL) ||
113 config_container[iZone]->GetSurface_Movement(EXTERNAL_ROTATION)){
114 prefixed_motion[iZone] = true;
115 }
116 }
117
118}

137void CMultizoneDriver::StartSolver() {
138
139 /*--- Find out the minimum of all references times and then set each zone to this (same) value.
140 To ensure that all zones run synchronously in time, be it a dimensional or non-dimensionalized one. ---*/
141
142 su2double Time_Ref = std::numeric_limits<su2double>::max();
143 for (iZone = 0; iZone < nZone; iZone++)
144 Time_Ref = min(Time_Ref, config_container[iZone]->GetTime_Ref());
145
146 for (iZone = 0; iZone < nZone; iZone++) {
147
148 config_container[iZone]->SetTime_Ref(Time_Ref);
149
150 /*--- Recompute some values as the reference time might has changed in iZone ---*/
151
152 config_container[iZone]->SetDelta_UnstTimeND(config_container[iZone]->GetDelta_UnstTime() / Time_Ref);
153 config_container[iZone]->SetTotal_UnstTimeND(config_container[iZone]->GetTotal_UnstTime() / Time_Ref);
154 }
155
156 StartTime = SU2_MPI::Wtime();
157
158 driver_config->Set_StartTime(StartTime);
159
160 /*--- Main external loop of the solver. Runs for the number of time steps required. ---*/
161
162 if (rank == MASTER_NODE){
163 cout << "\n------------------------------ Begin Solver -----------------------------" << endl;
164 }
165
166 if (rank == MASTER_NODE){
167 cout << "\nSimulation Run using the Multizone Driver" << endl;
168 if (driver_config->GetTime_Domain())
169 cout << "The simulation will run until time step " << driver_config->GetnTime_Iter() - driver_config->GetRestart_Iter() << "." << endl;
170 }
171
172 /*--- Set the initial time iteration to the restart iteration. ---*/
173 if (driver_config->GetRestart() && driver_config->GetTime_Domain())
174 TimeIter = driver_config->GetRestart_Iter();
175
176 /*--- Run the problem until the number of time iterations required is reached. ---*/
177 while ( TimeIter < driver_config->GetnTime_Iter() ) {
178
179 /*--- Perform some preprocessing before starting the time-step simulation. ---*/
180
181 Preprocess(TimeIter);
182
183 /*--- Run a block iteration of the multizone problem. ---*/
184 Run();
185
186 /*--- Update the solution for dual time stepping strategy ---*/
187
188 Update();
189
190 /*--- Monitor the computations after each iteration. ---*/
191
192 StopCalc = Monitor(TimeIter);
193
194 /*--- Output the solution in files. ---*/
195
196 Output(TimeIter);
197
198 /*--- If the convergence criteria has been met, terminate the simulation. ---*/
199
200 if (StopCalc) break;
201
202 TimeIter++;
203
204 }

208void CMultizoneDriver::Preprocess(unsigned long TimeIter) {
209
210 /*--- Set the current time iteration in the config ---*/
211 driver_config->SetTimeIter(TimeIter);
212
213 for (iZone = 0; iZone < nZone; iZone++){
214
215 /*--- Set the value of the external iteration to TimeIter. -------------------------------------*/
216 /*--- TODO: This should be generalised for an homogeneous criteria throughout the code. --------*/
217 config_container[iZone]->SetTimeIter(TimeIter);
218
219
220 /*--- Store the current physical time in the config container, as
221 this can be used for verification / MMS. This should also be more
222 general once the drivers are more stable. ---*/
223
224 if (driver_config->GetTime_Domain()) {
225 config_container[iZone]->SetPhysicalTime(static_cast<su2double>(TimeIter)*config_container[iZone]->GetDelta_UnstTimeND());
226 }
227 else {
228 config_container[iZone]->SetPhysicalTime(0.0);
229 }
230
231 /*--- Set the initial condition for EULER/N-S/RANS ---------------------------------------------*/
232 /*--- For FSI, the initial conditions are set, after the mesh has been moved. --------------------------------------*/
233 if (!fsi && config_container[iZone]->GetFluidProblem()) {
234 solver_container[iZone][INST_0][MESH_0][FLOW_SOL]->SetInitialCondition(geometry_container[iZone][INST_0],
235 solver_container[iZone][INST_0],
236 config_container[iZone], TimeIter);
237 }
238 else if (!fsi && config_container[iZone]->GetHeatProblem()) {
239 /*--- Set the initial condition for HEAT equation ---------------------------------------------*/
240 solver_container[iZone][INST_0][MESH_0][HEAT_SOL]->SetInitialCondition(geometry_container[iZone][INST_0],
241 solver_container[iZone][INST_0],
242 config_container[iZone], TimeIter);
243 }
244 if (!fsi && (config_container[iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE)) {
245 /*--- Set the initial condition for SPECIES solver (species or flamelet) ----------------------*/
246 solver_container[iZone][INST_0][MESH_0][SPECIES_SOL]->SetInitialCondition(geometry_container[ZONE_0][INST_0],
247 solver_container[ZONE_0][INST_0],
248 config_container[ZONE_0], TimeIter);
249 }
250
251
252 }
253
254 SU2_MPI::Barrier(SU2_MPI::GetComm());
255
256 /*--- Run a predictor step ---*/
257 for (iZone = 0; iZone < nZone; iZone++) {
258 if (config_container[iZone]->GetPredictor())
259 iteration_container[iZone][INST_0]->Predictor(output_container[iZone], integration_container, geometry_container,
260 solver_container, numerics_container, config_container, surface_movement,
261 grid_movement, FFDBox, iZone, INST_0);
262 }
263
264 /*--- Perform a dynamic mesh update if required. ---*/
265
266 DynamicMeshUpdate(TimeIter);
267
268 /*--- Updating zone interface communication patterns for unsteady problems with pre-fixed motion in the config file ---*/
269 if (driver_config->GetTime_Domain()) {
270 for (iZone = 0; iZone < nZone; iZone++) {
271 for (unsigned short jZone = 0; jZone < nZone; jZone++){
272 if(jZone != iZone && interpolator_container[iZone][jZone] != nullptr && prefixed_motion[iZone])
273 interpolator_container[iZone][jZone]->SetTransferCoeff(config_container);
274 }
275 }
276 }

280void CMultizoneDriver::RunGaussSeidel() {
281
282 unsigned short UpdateMesh;
283 bool DeformMesh = false;
284
285 for (iZone = 0; iZone < nZone; iZone++) {
286 config_container[iZone]->SetOuterIter(0ul);
287 }
288
289 /*--- Loop over the number of outer iterations ---*/
290 for (auto iOuter_Iter = 0ul; iOuter_Iter < driver_config->GetnOuter_Iter(); iOuter_Iter++) {
291
292 /*--- Loop over the number of zones (IZONE) ---*/
293 for (iZone = 0; iZone < nZone; iZone++) {
294
295 /*--- In principle, the mesh does not need to be updated ---*/
296 UpdateMesh = 0;
297
298 /*--- Set the OuterIter ---*/
299 config_container[iZone]->SetOuterIter(iOuter_Iter);
300 config_container[iZone]->Set_StartTime(SU2_MPI::Wtime());
301 driver_config->SetOuterIter(iOuter_Iter);
302
303 /*--- Transfer from all the remaining zones ---*/
304 for (auto jZone = 0u; jZone < nZone; jZone++){
305 /*--- The target zone is iZone ---*/
306 if (jZone != iZone){
307 DeformMesh = TransferData(jZone, iZone);
308 if (DeformMesh) UpdateMesh+=1;
309 }
310 }
311 /*--- If a mesh update is required due to the transfer of data ---*/
312 if (UpdateMesh > 0) DynamicMeshUpdate(iZone, TimeIter);
313
314 /*--- Iterate the zone as a block, either to convergence or to a max number of iterations ---*/
315 iteration_container[iZone][INST_0]->Solve(output_container[iZone], integration_container, geometry_container,
316 solver_container, numerics_container, config_container,
317 surface_movement, grid_movement, FFDBox, iZone, INST_0);
318
319 /*--- A corrector step can help preventing numerical instabilities ---*/
320 Corrector(iZone);
321
322 }
323
324 if (OuterConvergence(iOuter_Iter)) break;
325
326 }

330void CMultizoneDriver::RunJacobi() {
331
332 unsigned short UpdateMesh;
333 bool DeformMesh = false;
334
335 for (iZone = 0; iZone < nZone; iZone++) {
336 config_container[iZone]->SetOuterIter(0ul);
337 }
338
339 /*--- Loop over the number of outer iterations ---*/
340 for (auto iOuter_Iter = 0ul; iOuter_Iter < driver_config->GetnOuter_Iter(); iOuter_Iter++){
341
342
343 /*--- Transfer from all zones ---*/
344 for (iZone = 0; iZone < nZone; iZone++){
345
346 /*--- In principle, the mesh does not need to be updated ---*/
347 UpdateMesh = 0;
348
349 /*--- Set the OuterIter ---*/
350 config_container[iZone]->SetOuterIter(iOuter_Iter);
351 driver_config->SetOuterIter(iOuter_Iter);
352
353 /*--- Transfer from all the remaining zones ---*/
354 for (auto jZone = 0u; jZone < nZone; jZone++){
355 /*--- The target zone is iZone ---*/
356 if (jZone != iZone && interface_container[iZone][jZone] != nullptr){
357 DeformMesh = TransferData(jZone, iZone);
358 if (DeformMesh) UpdateMesh+=1;
359 }
360 }
361 /*--- If a mesh update is required due to the transfer of data ---*/
362 if (UpdateMesh > 0) DynamicMeshUpdate(iZone, TimeIter);
363
364 }
365
366 /*--- Loop over the number of zones (IZONE) ---*/
367 for (iZone = 0; iZone < nZone; iZone++) {
368
369 /*--- Set the OuterIter ---*/
370 config_container[iZone]->SetOuterIter(iOuter_Iter);
371 config_container[iZone]->Set_StartTime(SU2_MPI::Wtime());
372 driver_config->SetOuterIter(iOuter_Iter);
373
374 /*--- Iterate the zone as a block, either to convergence or to a max number of iterations ---*/
375 iteration_container[iZone][INST_0]->Solve(output_container[iZone], integration_container, geometry_container,
376 solver_container, numerics_container, config_container,
377 surface_movement, grid_movement, FFDBox, iZone, INST_0);
378
379 /*--- A corrector step can help preventing numerical instabilities ---*/
380 Corrector(iZone);
381
382 }
383
384 if (OuterConvergence(iOuter_Iter)) break;
385

398bool CMultizoneDriver::OuterConvergence(unsigned long OuterIter) {
399
400 /*--- Update the residual for the all the zones. ---*/
401
402 for (iZone = 0; iZone < nZone; iZone++) {
403
404 /*--- Account for all the solvers in this zone. ---*/
405
406 auto solvers = solver_container[iZone][INST_0][MESH_0];
407
408 for (unsigned short iSol = 0; iSol < MAX_SOLS; iSol++){
409 if (solvers[iSol] != nullptr) {
410 solvers[iSol]->ComputeResidual_Multizone(geometry_container[iZone][INST_0][MESH_0], config_container[iZone]);
411 solvers[iSol]->GetNodes()->Set_BGSSolution_k();
412 }
413 }
414
415 /*--- Make sure that everything is loaded into the output container. ---*/
416
417 output_container[iZone]->SetHistoryOutput(geometry_container[iZone][INST_0][MESH_0], solvers, config_container[iZone]);
418
419 }
420
421 /*--- Print out the convergence data to screen and history file. ---*/
422
423 driver_output->SetMultizoneHistoryOutput(output_container, config_container, driver_config,
424 driver_config->GetTimeIter(), driver_config->GetOuterIter());
425
426 return driver_output->GetConvergence();
427

430void CMultizoneDriver::Update() {
431
432 /*--- For enabling a consistent restart, we need to update the mesh with the interface information that introduces displacements --*/
433 /*--- Loop over the number of zones (IZONE) ---*/
434 for (iZone = 0; iZone < nZone; iZone++){
435
436 unsigned short UpdateMesh = 0;
437
438 /*--- Transfer from all the remaining zones (JZONE != IZONE)---*/
439 for (auto jZone = 0u; jZone < nZone; jZone++){
440 /*--- The target zone is iZone ---*/
441 if (jZone != iZone){
442 UpdateMesh += TransferData(jZone, iZone);
443 }
444 }
445 /*--- If a mesh update is required due to the transfer of data ---*/
446 if (UpdateMesh > 0) DynamicMeshUpdate(iZone, TimeIter);
447
448 iteration_container[iZone][INST_0]->Update(output_container[iZone], integration_container, geometry_container,
449 solver_container, numerics_container, config_container,
450 surface_movement, grid_movement, FFDBox, iZone, INST_0);
451 }

488void CMultizoneDriver::DynamicMeshUpdate(unsigned long TimeIter) {
489
490 bool AnyDeformMesh = false;
491
492 for (iZone = 0; iZone < nZone; iZone++) {
493 const auto harmonic_balance = (config_container[iZone]->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE);
494 /*--- Dynamic mesh update ---*/
495 if ((config_container[iZone]->GetGrid_Movement()) && (!harmonic_balance) && (!fsi)) {
496 iteration_container[iZone][INST_0]->SetGrid_Movement(geometry_container[iZone][INST_0],surface_movement[iZone],
497 grid_movement[iZone][INST_0], solver_container[iZone][INST_0],
498 config_container[iZone], 0, TimeIter);
499 AnyDeformMesh = true;
500 }
501 }
502 /*--- Update the wall distances if the mesh was deformed. ---*/
503 if (AnyDeformMesh) {
504 CGeometry::ComputeWallDistance(config_container, geometry_container);
505 }

529 CGeometry::ComputeWallDistance(config_container, geometry_container);

533bool CMultizoneDriver::TransferData(unsigned short donorZone, unsigned short targetZone) {
534
535 bool UpdateMesh = false;
536
537 /*--- Select the transfer method according to the magnitudes being transferred ---*/
538
539 auto BroadcastData = [&](int donorSol, int targetSol) {
540 interface_container[donorZone][targetZone]->BroadcastData(
541 *interpolator_container[donorZone][targetZone].get(),
542 solver_container[donorZone][INST_0][MESH_0][donorSol],
543 solver_container[targetZone][INST_0][MESH_0][targetSol],
544 geometry_container[donorZone][INST_0][MESH_0],
545 geometry_container[targetZone][INST_0][MESH_0],
546 config_container[donorZone],
547 config_container[targetZone]);
548 };
549
550 switch (interface_types[donorZone][targetZone]) {
551
552 case SLIDING_INTERFACE:
553 BroadcastData(FLOW_SOL, FLOW_SOL);
554
555 /*--- Additional transfer for turbulence variables. ---*/
556 if (config_container[targetZone]->GetKind_Solver() == MAIN_SOLVER::RANS ||
557 config_container[targetZone]->GetKind_Solver() == MAIN_SOLVER::INC_RANS) {
558 BroadcastData(TURB_SOL, TURB_SOL);
559 }
560
561 /*--- Additional transfer for species variables. ---*/
562 if (config_container[targetZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
563 BroadcastData(SPECIES_SOL, SPECIES_SOL);
564 }
565 break;
566 case CONJUGATE_HEAT_FS:
567 BroadcastData(FLOW_SOL, HEAT_SOL);
568 break;
569 case CONJUGATE_HEAT_WEAKLY_FS:
570 BroadcastData(HEAT_SOL, HEAT_SOL);
571 break;
572 case CONJUGATE_HEAT_SF:
573 BroadcastData(HEAT_SOL, FLOW_SOL);
574 break;
575 case CONJUGATE_HEAT_WEAKLY_SF:
576 BroadcastData(HEAT_SOL, HEAT_SOL);
577 break;
578 case BOUNDARY_DISPLACEMENTS:
579 BroadcastData(FEA_SOL, MESH_SOL);
580 UpdateMesh = true;
581 break;
582 case FLOW_TRACTION:
583 BroadcastData(FLOW_SOL, FEA_SOL);
584 break;
585 case MIXING_PLANE:
586 {
587 const auto nMarkerInt = config_container[donorZone]->GetnMarker_MixingPlaneInterface() / 2;
588
589 /*--- Transfer the average value from the donorZone to the targetZone ---*/
590 for (auto iMarkerInt = 1; iMarkerInt <= nMarkerInt; iMarkerInt++) {
591 interface_container[donorZone][targetZone]->AllgatherAverage(solver_container[donorZone][INST_0][MESH_0][FLOW_SOL],solver_container[targetZone][INST_0][MESH_0][FLOW_SOL],
592 geometry_container[donorZone][INST_0][MESH_0],geometry_container[targetZone][INST_0][MESH_0],
593 config_container[donorZone], config_container[targetZone], iMarkerInt );
594 }
595
596 for (donorZone = 0; donorZone < nZone; donorZone++) {
597 if (interface_types[donorZone][targetZone]==MIXING_PLANE) {
598 interface_container[donorZone][targetZone]->GatherAverageValues(solver_container[donorZone][INST_0][MESH_0][FLOW_SOL],solver_container[targetZone][INST_0][MESH_0][FLOW_SOL], donorZone);
599 geometry_container[targetZone][INST_0][MESH_0]->SetAvgTurboGeoValues(config_container[iZone],geometry_container[iZone][INST_0][MESH_0], iZone);
600 }
601 }
602
603 break;
604 }
605 case NO_TRANSFER:
606 case ZONES_ARE_EQUAL:
607 case NO_COMMON_INTERFACE:
608 break;
609 default:
610 if(rank == MASTER_NODE)
611 cout << "WARNING: One of the intended interface transfer routines is not "
612 << "known to the chosen driver and has not been executed." << endl;
613 break;
614 }
615
616 return UpdateMesh;
617}

629bool CMultizoneDriver::Monitor(unsigned long TimeIter) {
630
631 /*--- Check whether the inner solver has converged --- */
632
633 if (driver_config->GetTime_Domain() == NO) {
634
635 const auto OuterIter = driver_config->GetOuterIter();
636 const auto nOuterIter = driver_config->GetnOuter_Iter();
637
638 const auto InnerConvergence = driver_output->GetConvergence();
639 const bool MaxIterationsReached = (OuterIter+1 >= nOuterIter);
640
641 if ((MaxIterationsReached || InnerConvergence) && (rank == MASTER_NODE)) {
642 cout << "\n----------------------------- Solver Exit -------------------------------" << endl;
643 if (InnerConvergence) cout << "All convergence criteria satisfied." << endl;
644 else cout << "\nMaximum number of iterations reached (OUTER_ITER = " << OuterIter+1 << ") before convergence." << endl;
645 driver_output->PrintConvergenceSummary();
646 cout << "-------------------------------------------------------------------------" << endl;
647 }
648
649 return (MaxIterationsReached || InnerConvergence);
650 }
651 // i.e. unsteady simulation
652
653 /*--- Check whether the outer time integration has reached the final time. ---*/
654 const auto TimeConvergence = GetTimeConvergence();
655
656 const auto nTimeIter = driver_config->GetnTime_Iter();
657 const auto MaxTime = driver_config->GetMax_Time();
658 const auto CurTime = driver_output->GetHistoryFieldValue("CUR_TIME");
659
660 const bool FinalTimeReached = (CurTime >= MaxTime);
661 const bool MaxIterationsReached = (TimeIter+1 >= nTimeIter);
662
663 if ((TimeConvergence || FinalTimeReached || MaxIterationsReached) && (rank == MASTER_NODE)){
664 cout << "\n----------------------------- Solver Exit -------------------------------";
665 if (TimeConvergence) cout << "\nAll windowed time-averaged convergence criteria are fullfilled." << endl;
666 if (FinalTimeReached) cout << "\nMaximum time reached (MAX_TIME = " << MaxTime << "s)." << endl;
667 else cout << "\nMaximum number of time iterations reached (TIME_ITER = " << nTimeIter << ")." << endl;
668 cout << "-------------------------------------------------------------------------" << endl;
669 }
670
671 return (FinalTimeReached || MaxIterationsReached);
672}
