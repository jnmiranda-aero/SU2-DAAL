56/*!
57 * \class AccessorImpl
58 * \brief Base accessor class and version of template for both sizes known at compile time.
59 *
60 * The actual container inherits from this class, this is to reduce the number of
61 * methods that need to be redefined with each size specialization.
62 */
63template <typename Index_t, class Scalar_t, StorageType Store, size_t AlignSize, size_t StaticRows, size_t StaticCols>
64class AccessorImpl {
65 static_assert(!(StaticRows == 1 && Store == StorageType::ColumnMajor), "Row vector should have row-major storage.");
66 static_assert(!(StaticCols == 1 && Store == StorageType::RowMajor),
67 "Column vector should have column-major storage.");
68
69 protected:
70 /*!
71 * For static specializations AlignSize will force the alignment
72 * specification of the entire class, not just the data.
73 */
74 alignas(AlignSize) Scalar_t m_data[StaticRows * StaticCols];
75
76 /*!
77 * Static size specializations use this do-nothing allocation macro.
78 */
79#define DUMMY_ALLOCATOR \
80 void m_allocate(size_t sz, Index_t rows, Index_t cols) noexcept {} \
81 void m_destroy() noexcept {}
82
83 /*!
84 * Dynamic size specializations use this one, EXTRA is used to set some
85 * runtime internal value that depend on the number of rows/columns.
86 * What values need setting depends on the specialization as not all have
87 * members for e.g. number of rows and cols (static size optimization).
88 * Because aligned allocation is used, "placement new" is used after to
89 * default construct the elements of non-trivial type. Such types also
90 * need to be destructed explicitly before freeing the memory.
91 */
92#define REAL_ALLOCATOR(EXTRA) \
93 static_assert(MemoryAllocation::is_power_of_two(AlignSize), "AlignSize is not a power of two."); \
94 \
95 void m_allocate(size_t sz, Index_t rows, Index_t cols) noexcept { \
96 EXTRA; \
97 m_data = MemoryAllocation::aligned_alloc<Scalar_t>(AlignSize, sz); \
98 if (!std::is_trivial<Scalar_t>::value) \
99 for (size_t i = 0; i < size(); ++i) new (m_data + i) Scalar_t(); \
100 } \
101 \
102 void m_destroy() noexcept { \
103 if (!std::is_trivial<Scalar_t>::value) \
104 for (size_t i = 0; i < size(); ++i) m_data[i].~Scalar_t(); \
105 MemoryAllocation::aligned_free<Scalar_t>(m_data); \
106 }
107
108 DUMMY_ALLOCATOR
109
110 public:
111 /*!
112 * Dynamic types need to manage internal data as the derived class would
113 * not compile if it tried to set m_data to null on static specializations.
114 * Move construct/assign are enabled by transferring ownership of data
115 * pointer, the rvalue is left in the empty state.
116 * The default ctor needs to "INIT" some fields. The move ctor/assign need
117 * to "MOVE" those fields, i.e. copy and set "other" appropriately.
118 */
119#define CUSTOM_CTOR_AND_DTOR_BASE(INIT, MOVE) \
120 AccessorImpl() noexcept : m_data(nullptr) { INIT; } \
121 \
122 AccessorImpl(AccessorImpl&& other) noexcept { \
123 MOVE; \
124 m_data = other.m_data; \
125 other.m_data = nullptr; \
126 } \
127 \
128 AccessorImpl& operator=(AccessorImpl&& other) noexcept { \
129 m_destroy(); \
130 MOVE; \
131 m_data = other.m_data; \
132 other.m_data = nullptr; \
133 return *this; \
134 } \
135 \
136 ~AccessorImpl() noexcept { m_destroy(); }
137
138 /*!
139 * Shorthand for when specialization has only one more member than m_data.
140 */
141#define CUSTOM_CTOR_AND_DTOR(X) CUSTOM_CTOR_AND_DTOR_BASE(X = 0, X = other.X; other.X = 0)
142
143 /*!
144 * Universal accessors return a raw pointer to the data.
145 */
146#define UNIV_ACCESSORS \
147 bool empty() const noexcept { return size() == 0; } \
148 Scalar_t* data() noexcept { return m_data; } \
149 const Scalar_t* data() const noexcept { return m_data; } \
150 Scalar_t* begin() noexcept { return data(); } \
151 const Scalar_t* begin() const noexcept { return data(); } \
152 Scalar_t* end() noexcept { return data() + size(); } \
153 const Scalar_t* end() const noexcept { return data() + size(); }
154
155 /*!
156 * Operator (,) gives pointwise access, operator [] returns a pointer to the
157 * first element of the row/column of a row/column-major matrix respectively.
158 */
159#define MATRIX_ACCESSORS(M, N) \
160 UNIV_ACCESSORS \
161 Index_t rows() const noexcept { return M; } \
162 Index_t cols() const noexcept { return N; } \
163 size_t size() const noexcept { return M * N; } \
164 \
165 const Scalar_t& operator()(const Index_t i, const Index_t j) const noexcept { \
166 assert(i >= 0 && i < M && j >= 0 && j < N); \
167 return m_data[(Store == StorageType::RowMajor) ? i * N + j : i + j * M]; \
168 } \
169 \
170 Scalar_t& operator()(const Index_t i, const Index_t j) noexcept { \
171 const AccessorImpl& const_this = *this; \
172 return const_cast<Scalar_t&>(const_this(i, j)); \
173 } \
174 \
175 const Scalar_t* operator[](const Index_t k) const noexcept { \
176 if (Store == StorageType::RowMajor) { \
177 assert(k >= 0 && k < M); \
178 return &m_data[k * N]; \
179 } else { \
180 assert(k >= 0 && k < N); \
181 return &m_data[k * M]; \
182 } \
183 } \
184 \
185 Scalar_t* operator[](const Index_t k) noexcept { \
186 const AccessorImpl& const_this = *this; \
187 return const_cast<Scalar_t*>(const_this[k]); \
188 }
189
190 /*!
191 * Vectors provide both [] and () with the same behavior.
192 */
193#define VECTOR_ACCESSORS(M, ROWMAJOR) \
194 UNIV_ACCESSORS \
195 Index_t rows() const noexcept { return ROWMAJOR ? 1 : M; } \
196 Index_t cols() const noexcept { return ROWMAJOR ? M : 1; } \
197 size_t size() const noexcept { return M; } \
198 \
199 Scalar_t& operator()(const Index_t i) noexcept { \
200 assert(i >= 0 && i < M); \
201 return m_data[i]; \
202 } \
203 \
204 const Scalar_t& operator()(const Index_t i) const noexcept { \
205 assert(i >= 0 && i < M); \
206 return m_data[i]; \
207 } \
208 \
209 Scalar_t& operator[](const Index_t i) noexcept { \
210 assert(i >= 0 && i < M); \
211 return m_data[i]; \
212 } \
213 \
214 const Scalar_t& operator[](const Index_t i) const noexcept { \
215 assert(i >= 0 && i < M); \
216 return m_data[i]; \
217 }
218
219 MATRIX_ACCESSORS(StaticRows, StaticCols)
220};

350/*!
351 * \class C2DContainer
352 * \brief A templated matrix/vector-like object.
353 *
354 * See notes about MATRIX_ACCESSORS and VECTOR_ACCESSORS for how to access data.
355 * For how to construct/resize/initialize see methods below.
356 *
357 * Template parameters:
358 *
359 * \param Index_t - The data type (built-in) for indices, signed and unsigned allowed.
360 * \param Scalar_t - The stored data type, anything that can be default constructed.
361 * \param Store - Mode to map 1D to 2D, row-major or column-major.
362 * \param AlignSize - Desired alignment for the data in bytes, 0 means default.
363 * \param StaticRows - Number of rows at compile time, for dynamic (sized at runtime) use "DynamicSize".
364 * \param StaticCols - As above for columns.
365 *
366 * \note All accesses to data are range checked via assertions, for
367 * release compile with -DNDEBUG to avoid the associated overhead.
368 *
369 */
370template <typename Index_t, class Scalar_t, StorageType Store, size_t AlignSize, size_t StaticRows, size_t StaticCols>
371class C2DContainer
372 : public container_details::AccessorImpl<Index_t, Scalar_t, Store, AlignSize, StaticRows, StaticCols> {
373 static_assert(std::is_integral<Index_t>::value, "");
374
375 private:
376 using Base = container_details::AccessorImpl<Index_t, Scalar_t, Store, AlignSize, StaticRows, StaticCols>;
377 using Base::m_allocate;
378 using Base::m_data;
379 using Base::m_destroy;
380
381 public:
382 using Base::cols;
383 using Base::rows;
384 using Base::size;
385 using Index = Index_t;
386 using Scalar = Scalar_t;
387 static constexpr StorageType Storage = Store;
388 static constexpr bool IsVector = (StaticRows == 1) || (StaticCols == 1);
389 static constexpr bool IsRowMajor = (Store == StorageType::RowMajor);
390 static constexpr bool IsColumnMajor = (Store == StorageType::ColumnMajor);
391 static constexpr size_t StaticSize = StaticRows * StaticCols;

445 private:
446 /*!
447 * \brief Logic to resize data according to arguments, a non DynamicSize cannot be changed.
448 */
449 size_t m_resize(Index_t rows, Index_t cols) noexcept {
450 /*--- fully static, no allocation needed ---*/
451 if (StaticSize != DynamicSize) return StaticSize;
452
453 /*--- dynamic row vector, swap size specification ---*/
454 if (StaticRows == 1 && IsVector) {
455 cols = rows;
456 rows = 1;
457 }
458
459 /*--- assert a static size is not being asked to change ---*/
460 if (StaticRows != DynamicSize) assert(rows == StaticRows && "A static size was asked to change.");
461 if (StaticCols != DynamicSize) assert(cols == StaticCols && "A static size was asked to change.");
462
463 /*--- "rectify" sizes before continuing as asserts are usually dissabled ---*/
464 rows = (StaticRows != DynamicSize) ? StaticRows : rows;
465 cols = (StaticCols != DynamicSize) ? StaticCols : cols;
466
467 /*--- number of requested elements ---*/
468 size_t reqSize = rows * cols;
469
470 /*--- compare with current dimensions to determine if deallocation
471 is needed, also makes the container safe against self assignment
472 no need to check for 0 size as the allocators handle that ---*/
473 if (rows == this->rows() && cols == this->cols()) return reqSize;
474
475 m_destroy();
476
477 /*--- request actual allocation to base class as it needs specialization ---*/
478 size_t bytes = reqSize * sizeof(Scalar_t);
479 m_allocate(bytes, rows, cols);
480
481 return reqSize;
482 }

564 /*!
565 * \brief Return copy of data in a static size container.
566 * \param[in] row - Row of the matrix.
567 * \param[in] start - Starting column to copy the data (amount determined by container size).
568 */
569 template <class StaticContainer>
570 FORCEINLINE StaticContainer get(Index_t row, Index_t start = 0) const noexcept {
571 constexpr size_t Size = StaticContainer::StaticSize;
572 static_assert(Size, "This method requires a static output type.");
573 assert(Size <= cols() - start);
574 StaticContainer ret;
575 SU2_OMP_SIMD
576 for (size_t i = 0; i < Size; ++i)
577 ret.data()[i] = m_data[IsRowMajor ? row * cols() + i + start : row + (i + start) * rows()];
578 return ret;
579 }
580
581 /*!
582 * \brief Return copy of data in a static size container, SIMD version.
583 */
584 template <class StaticContainer, class T, size_t N>
585 FORCEINLINE StaticContainer get(simd::Array<T, N> row, Index_t start = 0) const noexcept {
586 constexpr size_t Size = StaticContainer::StaticSize;
587 static_assert(Size, "This method requires a static output type.");
588 assert(Size <= cols() - start);
589 StaticContainer ret;
590 for (size_t k = 0; k < N; ++k) {
591 SU2_OMP_SIMD_IF_NOT_AD
592 for (size_t i = 0; i < Size; ++i)
593 ret.data()[i][k] = m_data[IsRowMajor ? row[k] * cols() + i + start : row[k] + (i + start) * rows()];
594 }
595 return ret;
596 }
