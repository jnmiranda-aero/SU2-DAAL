47void CBaseMPIWrapper::Error(const std::string& ErrorMsg, const std::string& FunctionName) {
48 /* Set MinRankError to Rank, as the error message is called on this rank. */
49 MinRankError = Rank;
50 int flag = 0;
51
52#if MPI_VERSION >= 3
53 /* Find out whether the error call is collective via MPI_Ibarrier. */
54 Request barrierRequest;
55 MPI_Ibarrier(currentComm, &barrierRequest);
56
57 /* Try to complete the non-blocking barrier call for a second. */
58 double startTime = SU2_MPI::Wtime();
59 while (true) {
60 MPI_Test(&barrierRequest, &flag, MPI_STATUS_IGNORE);
61 if (flag) break;
62
63 double currentTime = SU2_MPI::Wtime();
64 if (currentTime > startTime + 1.0) break;
65 }
66#else
67 /* MPI_Ibarrier function is not supported. Simply wait for one
68 second to give other ranks the opportunity to reach this point. */
69#ifdef _MSC_VER
70 _sleep(1);
71#else
72 sleep(1);
73#endif
74#endif
75
76 if (flag) {
77 /* The barrier is completed and hence the error call is collective.
78 Set MinRankError to 0. */
79 MinRankError = 0;
80 } else {
81 /* The error call is not collective and the minimum rank must be
82 determined by one sided communication. Loop over the lower numbered
83 ranks to check if they participate in the error message. */
84 for (int i = 0; i < Rank; ++i) {
85 int MinRankErrorOther;
86 MPI_Win_lock(MPI_LOCK_SHARED, i, 0, winMinRankError);
87 MPI_Get(&MinRankErrorOther, 1, MPI_INT, i, 0, 1, MPI_INT, winMinRankError);
88 MPI_Win_unlock(i, winMinRankError);
89
90 if (MinRankErrorOther < MinRankError) {
91 MinRankError = MinRankErrorOther;
92 break;
93 }
94 }
95 }
96
97 /* Check if this rank must write the error message and do so. */
98 if (Rank == MinRankError) {
99 std::cout << std::endl << std::endl;
100 std::cout << "Error in \"" << FunctionName << "\": " << std::endl;
101 std::cout << "-------------------------------------------------------------------------" << std::endl;
102 std::cout << ErrorMsg << std::endl;
103 std::cout << "------------------------------ Error Exit -------------------------------" << std::endl;
104 std::cout << std::endl << std::endl;
105 }
106 Abort(currentComm, EXIT_FAILURE);
107}

109void CBaseMPIWrapper::CopyData(const void* sendbuf, void* recvbuf, int size, Datatype datatype, int recvshift,
110 int sendshift) {
111 if (datatype == MPI_DOUBLE) {
112 for (int i = 0; i < size; i++) {
113 static_cast<su2double*>(recvbuf)[i + recvshift] = static_cast<const su2double*>(sendbuf)[i + sendshift];
114 }
115 } else {
116 int scalarsize;
117 MPI_Type_size(datatype, &scalarsize);
118 const char* src = static_cast<const char*>(sendbuf) + sendshift * static_cast<size_t>(scalarsize);
119 char* dest = static_cast<char*>(recvbuf) + recvshift * static_cast<size_t>(scalarsize);
120 std::memcpy(static_cast<void*>(dest), static_cast<const void*>(src), size * static_cast<size_t>(scalarsize));
121 }
122}

137void CBaseMPIWrapper::CopyData(const void* sendbuf, void* recvbuf, int size, Datatype datatype, int recvshift,
138 int sendshift) {
139 switch (datatype) {
140 case MPI_DOUBLE:
141 for (int i = 0; i < size; i++) {
142 static_cast<su2double*>(recvbuf)[i + recvshift] = static_cast<const su2double*>(sendbuf)[i + sendshift];
143 }
144 break;
145 case MPI_UNSIGNED_LONG:
146 for (int i = 0; i < size; i++) {
147 static_cast<unsigned long*>(recvbuf)[i + recvshift] = static_cast<const unsigned long*>(sendbuf)[i + sendshift];
148 }
149 break;
150 case MPI_LONG:
151 for (int i = 0; i < size; i++) {
152 static_cast<long*>(recvbuf)[i + recvshift] = static_cast<const long*>(sendbuf)[i + sendshift];
153 }
154 break;
155 case MPI_UNSIGNED_SHORT:
156 for (int i = 0; i < size; i++) {
157 static_cast<unsigned short*>(recvbuf)[i + recvshift] =
158 static_cast<const unsigned short*>(sendbuf)[i + sendshift];
159 }
160 break;
161 case MPI_CHAR:
162 for (int i = 0; i < size; i++) {
163 static_cast<char*>(recvbuf)[i + recvshift] = static_cast<const char*>(sendbuf)[i + sendshift];
164 }
165 break;
166 case MPI_SHORT:
167 for (int i = 0; i < size; i++) {
168 static_cast<short*>(recvbuf)[i + recvshift] = static_cast<const short*>(sendbuf)[i + sendshift];
169 }
170 break;
171 case MPI_INT:
172 for (int i = 0; i < size; i++) {
173 static_cast<int*>(recvbuf)[i + recvshift] = static_cast<const int*>(sendbuf)[i + sendshift];
174 }
175 break;
176 default:
177 Error("Unknown type", CURRENT_FUNCTION);
178 break;
179 };
