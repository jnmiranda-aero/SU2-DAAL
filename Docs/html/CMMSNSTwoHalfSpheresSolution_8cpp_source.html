32CMMSNSTwoHalfSpheresSolution::CMMSNSTwoHalfSpheresSolution(unsigned short val_nDim, unsigned short val_nVar,
33 unsigned short val_iMesh, CConfig* config)
34 : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
35 /*--- Write a message that the solution is initialized for the manufactured
36 solution for the Navier-Stokes equations between two half spheres. ---*/
37 if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
38 cout << endl;
39 cout << "Warning: Fluid properties and solution are being " << endl;
40 cout << " initialized for the manufactured solution " << endl;
41 cout << " of the Navier-Stokes equations between two half spheres!!!" << endl;
42 cout << endl << flush;
43 }
44
45 /*--- Coefficients, needed to determine the solution. ---*/
46 const su2double Prandtl = config->GetPrandtl_Lam();
47
48 RGas = config->GetGas_Constant();
49 Gamma = config->GetGamma();
50 Viscosity = config->GetMu_Constant();
51 Conductivity = Viscosity * Gamma * RGas / (Prandtl * (Gamma - 1.0));
52
53 /*--- Initialize TWall to the default value of 300 K (in case the outer wall
54 is not modelled as an isothermal wall) and try to retrieve the wall
55 temperature from the boundary conditions. ---*/
56 TWall = 300.0;
57 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
58 if (config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL) {
59 const string Marker_Tag = config->GetMarker_All_TagBound(iMarker);
60 TWall = config->GetIsothermal_Temperature(Marker_Tag);
61 }
62 }
63
64 /*--- Get the reference values for pressure, density and velocity. ---*/
65 Pressure_Ref = config->GetPressure_Ref();
66 Density_Ref = config->GetDensity_Ref();
67 Velocity_Ref = config->GetVelocity_Ref();
68
69 /*--- The constants for the density and velocities. ---*/
70 rho_0 = 1.25;
71 u_0 = 135.78;
72 v_0 = -67.61;
73 w_0 = 82.75;
74
75 /*--- The constants for the temperature solution. ---*/
76 a_T1 = 1.05;
77 a_T2 = -0.85;
78
79 /*--- Perform some sanity and error checks for this solution here. ---*/
80 if (nDim != 3) SU2_MPI::Error("Grid must be 3D for the MMS NS Two Half Spheres case", CURRENT_FUNCTION);
81
82 if (config->GetTime_Marching() != TIME_MARCHING::STEADY)
83 SU2_MPI::Error("Steady mode must be selected for the MMS NS Two Half Spheres case", CURRENT_FUNCTION);
84
85 if (Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES &&
86 Kind_Solver != MAIN_SOLVER::RANS && Kind_Solver != MAIN_SOLVER::FEM_EULER &&
87 Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
88 Kind_Solver != MAIN_SOLVER::FEM_LES)
89 SU2_MPI::Error("Compressible flow equations must be selected for the MMS NS Two Half Spheres case",
90 CURRENT_FUNCTION);
91
92 if ((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) && (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
93 SU2_MPI::Error("Navier Stokes equations must be selected for the MMS NS Two Half Spheres case", CURRENT_FUNCTION);
94
95 if ((config->GetKind_FluidModel() != STANDARD_AIR) && (config->GetKind_FluidModel() != IDEAL_GAS))
96 SU2_MPI::Error("Standard air or ideal gas must be selected for the MMS NS Two Half Spheres case", CURRENT_FUNCTION);
97
98 if (config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
99 SU2_MPI::Error("Sutherland must be selected for viscosity for the MMS NS Two Half Spheres case", CURRENT_FUNCTION);
100
101 if (config->GetKind_ConductivityModel() != CONDUCTIVITYMODEL::CONSTANT_PRANDTL)
102 SU2_MPI::Error("Constant Prandtl number must be selected for the MMS NS Two Half Spheres case", CURRENT_FUNCTION);
103}

113void CMMSNSTwoHalfSpheresSolution::GetSolution(const su2double* val_coords, const su2double val_t,
114 su2double* val_solution) const {
115 /* Easier storage of the x-, y- and z-coordinates. */
116 const su2double x = val_coords[0];
117 const su2double y = val_coords[1];
118 const su2double z = val_coords[2];
119
120 /* Determine the dimensional solution for the temperature. */
121 const su2double Pi = PI_NUMBER;
122 const su2double r = sqrt(x * x + y * y + z * z);
123 const su2double fact = (r - 1.0) * (r - 1.0) / (a_T1 + a_T2);
124
125 su2double T = 0.25 * TWall * (3.0 + fact * (a_T1 * cos(Pi * (r - 2.0)) + a_T2 * cos(Pi * (r - 2.0) * 2.0)));
126
127 /* Determine the dimensional solution for the velocities. */
128 su2double u = u_0 * (r - 1.0) * (2.0 - r) * 4.0;
129 su2double v = v_0 * (r - 1.0) * (2.0 - r) * 4.0;
130 su2double w = w_0 * (r - 1.0) * (2.0 - r) * 4.0;
131
132 /* Compute the pressure from the density and temperature. */
133 su2double rho = rho_0;
134 su2double p = rho * RGas * T;
135
136 /* Determine the non-dimensional solution. */
137 rho /= Density_Ref;
138 p /= Pressure_Ref;
139 u /= Velocity_Ref;
140 v /= Velocity_Ref;
141 w /= Velocity_Ref;
142
143 /* Determine the non-dimensional conserved variables. */
144 val_solution[0] = rho;
145 val_solution[1] = rho * u;
146 val_solution[2] = rho * v;
147 val_solution[3] = rho * w;
148 val_solution[4] = p / (Gamma - 1.0) + 0.5 * rho * (u * u + v * v + w * w);
149}
