48public:
49 /*!
50 * \brief Constructor of the class.
51 * \param[in] psirho - Value of the adjoint density (initialization value).
52 * \param[in] phi - Value of the adjoint velocity (initialization value).
53 * \param[in] psie - Value of the adjoint energy (initialization value).
54 * \param[in] npoint - Number of points/nodes/vertices in the domain.
55 * \param[in] ndim - Number of dimensions of the problem.

72 /*!
73 * \brief Set the value of the adjoint velocity.
74 * \param[in] val_phi - Value of the adjoint velocity.

76 inline void SetPhi_Old(unsigned long iPoint, const su2double *val_phi) final {
77 for (unsigned long iDim = 0; iDim < nDim; iDim++) Solution_Old(iPoint,iDim+1)=val_phi[iDim];

79
80 /*!

85 for (unsigned long iDim = 0; iDim < nDim; iDim++) ForceProj_Vector(iPoint,iDim) = val_ForceProj_Vector[iDim];
86 }
87
88 /*!
89 * \brief Set the value of the objective function source.
90 * \param[in] val_ObjFuncSource - Pointer to the objective function source.
91 */
92 inline void SetObjFuncSource(unsigned long iPoint, const su2double *val_ObjFuncSource) final {
93 for (unsigned long iVar = 0; iVar < nVar; iVar++) ObjFuncSource(iPoint,iVar) = val_ObjFuncSource[iVar];
94 }
95
96 /*!
97 * \brief Set the velocity vector from the old solution.
98 * \param[in] val_velocity - Pointer to the velocity.
99 */
100 inline void SetVelocity_Old(unsigned long iPoint, const su2double *val_velocity) final {
101 for (unsigned long iDim = 0; iDim < nDim; iDim++)
102 Solution_Old(iPoint,iDim+1) = val_velocity[iDim]*Solution(iPoint,0);
103 }
104
105 /*!
106 * \brief Set the momentum part of the truncation error to zero.
107 * \param[in] iPoint - Point index.
108 */
109 inline void SetVel_ResTruncError_Zero(unsigned long iPoint) final {
110 for (unsigned long iDim = 0; iDim < nDim; iDim++) Res_TruncError(iPoint,iDim+1) = 0.0;
111 }
112
113 /*!
114 * \brief Specify a vector to set the velocity components of the solution. Multiplied by density for compressible cases.
115 * \param[in] iPoint - Point index.
116 * \param[in] val_vector - Pointer to the vector.
117 */
118 inline void SetVelSolutionVector(unsigned long iPoint, const su2double *val_vector) final {
119 for (unsigned long iDim = 0; iDim < nDim; iDim++) Solution(iPoint, iDim+1) = GetDensity(iPoint) * val_vector[iDim];
120 }
121

127
128 /*!
129 * \brief Get the value of the objective function source.

133
134 /*!
135 * \brief Set the harmonic balance source term.

137 * \param[in] val_solution - Value of the harmonic balance source term. for the index <i>iVar</i>.
138 */
139 inline void SetHarmonicBalance_Source(unsigned long iPoint, unsigned long iVar, su2double val_source) final {
140 HB_Source(iPoint,iVar) = val_source;
141 }
142
143 /*!
144 * \brief Get the harmonic balance source term.
145 * \param[in] iVar - Index of the variable.
146 * \return Value of the harmonic balance source term for the index <i>iVar</i>.
147 */
148 inline su2double GetHarmonicBalance_Source(unsigned long iPoint, unsigned long iVar) const final {
149 return HB_Source(iPoint,iVar);
150 }
151
152 /*!
153 * \brief Get the array of the reconstruction variables gradient at a node.
154 * \param[in] iPoint - Index of the current node.

156 */
157 inline CMatrixView<su2double> GetGradient_Reconstruction(unsigned long iPoint) final {
158 return Gradient_Reconstruction[iPoint];
159 }
160
161 /*!
162 * \brief Get the reconstruction gradient for variables at all points.

164 */
165 inline CVectorOfMatrix& GetGradient_Reconstruction() final { return Gradient_Reconstruction; }
166 inline const CVectorOfMatrix& GetGradient_Reconstruction() const final { return Gradient_Reconstruction; }
