34CBaselineSolver_FEM::CBaselineSolver_FEM(CGeometry *geometry, CConfig *config) {
35
36 /*--- Define geometry constants in the solver structure ---*/
37
38 nDim = geometry->GetnDim();
39
40 /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
41 geometrical information for the FEM DG solver. If necessary, it is
42 possible to increase nMatchingFacesWithHaloElem a bit, such that
43 the computation of the external faces may be more efficient when
44 using multiple threads. ---*/
45
46 auto *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
47
48 nVolElemTot = DGGeometry->GetNVolElemTot();
49 nVolElemOwned = DGGeometry->GetNVolElemOwned();
50 volElem = DGGeometry->GetVolElem();
51
52 /*--- Routines to access the number of variables and string names. ---*/
53
54 SetOutputVariables(geometry, config);
55
56 /*--- Determine the total number of DOFs stored on this rank and allocate the memory
57 to store the conservative variables. ---*/
58 nDOFsLocOwned = 0;
59 for(unsigned long i=0; i<nVolElemOwned; ++i) nDOFsLocOwned += volElem[i].nDOFsSol;
60
61 nDOFsLocTot = nDOFsLocOwned;
62 for(unsigned long i=nVolElemOwned; i<nVolElemTot; ++i) nDOFsLocTot += volElem[i].nDOFsSol;
63
64 VecSolDOFs.resize(nVar*nDOFsLocTot);
65
66 /*--- Determine the global number of DOFs. ---*/
67#ifdef HAVE_MPI
68 SU2_MPI::Allreduce(&nDOFsLocOwned, &nDOFsGlobal, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
69#else
70 nDOFsGlobal = nDOFsLocOwned;
71#endif
72
73 /*--- Store the number of DOFs in the geometry class in case of restart. ---*/
74 geometry->SetnPointDomain(nDOFsLocOwned);
75 geometry->SetGlobal_nPointDomain(nDOFsGlobal);
76
77 /*--- Initialize the solution to zero. ---*/
78
79 unsigned long ii = 0;
80 for(unsigned long i=0; i<nDOFsLocTot; ++i) {
81 for(unsigned short j=0; j<nVar; ++j, ++ii) {
82 VecSolDOFs[ii] = 0.0;
83 }
84 }

88void CBaselineSolver_FEM::SetOutputVariables(CGeometry *geometry, CConfig *config) {
89
90 /*--- Open the restart file and extract the nVar and field names. ---*/
91
92 string Tag, text_line, AdjExt, UnstExt;
93 unsigned long TimeIter = config->GetTimeIter();
94
95 ifstream restart_file;
96 string filename;
97
98 /*--- Retrieve filename from config ---*/
99
100 filename = config->GetSolution_FileName();
101
102 /*--- Unsteady problems require an iteration number to be appended. ---*/
103
104 if (config->GetTime_Domain()) {
105 filename = config->GetUnsteady_FileName(filename, SU2_TYPE::Int(TimeIter), ".dat");
106 }
107
108 /*--- Read only the number of variables in the restart file. ---*/
109
110 if (config->GetRead_Binary_Restart()) {
111
112 int nVar_Buf = 5;
113 int var_buf[5];
114
115#ifndef HAVE_MPI
116
117 /*--- Serial binary input. ---*/
118
119 FILE *fhw;
120 fhw = fopen(filename.c_str(),"rb");
121 size_t ret;
122
123 /*--- Error check for opening the file. ---*/
124
125 if (!fhw)
126 SU2_MPI::Error(string("Unable to open SU2 restart file ") + filename,
127 CURRENT_FUNCTION);
128
129 /*--- First, read the number of variables and points. ---*/
130
131 ret = fread(var_buf, sizeof(int), nVar_Buf, fhw);
132 if (ret != (unsigned long)nVar_Buf) {
133 SU2_MPI::Error("Error reading restart file.", CURRENT_FUNCTION);
134 }
135
136 /*--- Check that this is an SU2 binary file. SU2 binary files
137 have the hex representation of "SU2" as the first int in the file. ---*/
138
139 if (var_buf[0] != 535532)
140 SU2_MPI::Error(string("File ") + filename + string(" is not a binary SU2 restart file.\n") +
141 string("SU2 reads/writes binary restart files by default.\n") +
142 string("Note that backward compatibility for ASCII restart files is\n") +
143 string("possible with the READ_BINARY_RESTART option."), CURRENT_FUNCTION);
144
145 /*--- Close the file. ---*/
146
147 fclose(fhw);
148
149#else
150
151 /*--- Parallel binary input using MPI I/O. ---*/
152
153 MPI_File fhw;
154 int ierr;
155
156 /*--- All ranks open the file using MPI. ---*/
157
158 char fname[100];
159 strcpy(fname, filename.c_str());
160 ierr = MPI_File_open(SU2_MPI::GetComm(), fname, MPI_MODE_RDONLY, MPI_INFO_NULL, &fhw);
161
162 /*--- Error check opening the file. ---*/
163
164 if (ierr)
165 SU2_MPI::Error(string("Unable to open SU2 restart file ") + filename,
166 CURRENT_FUNCTION);
167
168 /*--- First, read the number of variables and points (i.e., cols and rows),
169 which we will need in order to read the file later. Also, read the
170 variable string names here. Only the master rank reads the header. ---*/
171
172 if (rank == MASTER_NODE)
173 MPI_File_read(fhw, var_buf, nVar_Buf, MPI_INT, MPI_STATUS_IGNORE);
174
175 /*--- Broadcast the number of variables to all procs and store more clearly. ---*/
176
177 SU2_MPI::Bcast(var_buf, nVar_Buf, MPI_INT, MASTER_NODE, SU2_MPI::GetComm());
178
179 /*--- Check that this is an SU2 binary file. SU2 binary files
180 have the hex representation of "SU2" as the first int in the file. ---*/
181
182 if (var_buf[0] != 535532)
183 SU2_MPI::Error(string("File ") + filename + string(" is not a binary SU2 restart file.\n") +
184 string("SU2 reads/writes binary restart files by default.\n") +
185 string("Note that backward compatibility for ASCII restart files is\n") +
186 string("possible with the READ_BINARY_RESTART option."), CURRENT_FUNCTION);
187
188 /*--- All ranks close the file after writing. ---*/
189
190 MPI_File_close(&fhw);
191
192#endif
193
194 /*--- Set the number of variables, one per field in the
195 restart file (without including the PointID) ---*/
196
197 nVar = var_buf[1];
198
199 } else {
200
201 /*--- First, check that this is not a binary restart file. ---*/
202
203 int magic_number;
204
205#ifndef HAVE_MPI
206
207 /*--- Serial binary input. ---*/
208
209 FILE *fhw;
210 fhw = fopen(filename.c_str(), "rb");
211 size_t ret;
212
213 /*--- Error check for opening the file. ---*/
214
215 if (!fhw)
216 SU2_MPI::Error(string("Unable to open SU2 restart file ") + filename,
217 CURRENT_FUNCTION);
218
219 /*--- Attempt to read the first int, which should be our magic number. ---*/
220
221 ret = fread(&magic_number, sizeof(int), 1, fhw);
222 if (ret != 1) {
223 SU2_MPI::Error("Error reading restart file.", CURRENT_FUNCTION);
224 }
225
226 /*--- Check that this is an SU2 binary file. SU2 binary files
227 have the hex representation of "SU2" as the first int in the file. ---*/
228
229 if (magic_number == 535532)
230 SU2_MPI::Error(string("File ") + filename + string(" is a binary SU2 restart file, expected ASCII.\n") +
231 string("SU2 reads/writes binary restart files by default.\n") +
232 string("Note that backward compatibility for ASCII restart files is\n") +
233 string("possible with the READ_BINARY_RESTART option."), CURRENT_FUNCTION);
234 fclose(fhw);
235
236#else
237
238 /*--- Parallel binary input using MPI I/O. ---*/
239
240 MPI_File fhw;
241 int ierr;
242
243 /*--- All ranks open the file using MPI. ---*/
244
245 char fname[100];
246 strcpy(fname, filename.c_str());
247 ierr = MPI_File_open(SU2_MPI::GetComm(), fname, MPI_MODE_RDONLY, MPI_INFO_NULL, &fhw);
248
249 /*--- Error check opening the file. ---*/
250
251 if (ierr)
252 SU2_MPI::Error(string("Unable to open SU2 restart file ") + filename,
253 CURRENT_FUNCTION);
254
255 /*--- Have the master attempt to read the magic number. ---*/
256
257 if (rank == MASTER_NODE)
258 MPI_File_read(fhw, &magic_number, 1, MPI_INT, MPI_STATUS_IGNORE);
259
260 /*--- Broadcast the number of variables to all procs and store clearly. ---*/
261
262 SU2_MPI::Bcast(&magic_number, 1, MPI_INT, MASTER_NODE, SU2_MPI::GetComm());
263
264 /*--- Check that this is an SU2 binary file. SU2 binary files
265 have the hex representation of "SU2" as the first int in the file. ---*/
266
267 if (magic_number == 535532)
268 SU2_MPI::Error(string("File ") + filename + string(" is a binary SU2 restart file, expected ASCII.\n") +
269 string("SU2 reads/writes binary restart files by default.\n") +
270 string("Note that backward compatibility for ASCII restart files is\n") +
271 string("possible with the READ_BINARY_RESTART option."), CURRENT_FUNCTION);
272
273 MPI_File_close(&fhw);
274
275#endif
276
277 /*--- Open the restart file ---*/
278
279 restart_file.open(filename.data(), ios::in);
280
281 /*--- In case there is no restart file ---*/
282
283 if (restart_file.fail())
284 SU2_MPI::Error(string("SU2 solution file ") + filename + string(" not found"), CURRENT_FUNCTION);
285
286 /*--- Identify the number of fields (and names) in the restart file ---*/
287
288 getline (restart_file, text_line);
289
290 stringstream ss(text_line);
291 while (ss >> Tag) {
292 config->fields.push_back(Tag);
293 if (ss.peek() == ',') ss.ignore();
294 }
295
296 /*--- Close the file (the solution date is read later). ---*/
297
298 restart_file.close();
299
300 /*--- Set the number of variables, one per field in the
301 restart file (without including the PointID) ---*/
302
303 nVar = config->fields.size() - 1;
304
305 /*--- Clear the fields vector since we'll read it again. ---*/
306
307 config->fields.clear();
308
309 }

313void CBaselineSolver_FEM::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter, bool val_update_geo) {
314
315 /*--- Restart the solution from file information ---*/
316 unsigned short iVar;
317 unsigned long index;
318
319 string restart_filename = config->GetSolution_FileName();
320
321 if (config->GetTime_Domain()) {
322 restart_filename = config->GetUnsteady_FileName(restart_filename, SU2_TYPE::Int(val_iter), "");
323 }
324
325 int counter = 0;
326 long iPoint_Local = 0; unsigned long iPoint_Global = 0;
327 unsigned short rbuf_NotMatching = 0;
328 unsigned long nDOF_Read = 0;
329
330 /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
331
332 if (config->GetRead_Binary_Restart()) {
333 Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
334 } else {
335 Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
336 }
337
338 /*--- Load data from the restart into correct containers. ---*/
339
340 counter = 0;
341 for (iPoint_Global = 0; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
342
343 /*--- Retrieve local index. If this node from the restart file lives
344 on the current processor, we will load and instantiate the vars. ---*/
345
346 iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
347
348 if (iPoint_Local > -1) {
349
350 /*--- We need to store this point's data, so jump to the correct
351 offset in the buffer of data from the restart file and load it. ---*/
352
353 index = counter*Restart_Vars[1];
354 for (iVar = 0; iVar < nVar; iVar++) {
355 VecSolDOFs[nVar*iPoint_Local+iVar] = Restart_Data[index+iVar];
356 }
357 /*--- Update the local counter nDOF_Read. ---*/
358 ++nDOF_Read;
359
360 /*--- Increment the overall counter for how many points have been loaded. ---*/
361 counter++;
362 }
363
364 }
365
366 /*--- Detect a wrong solution file ---*/
367 if(nDOF_Read < nDOFsLocOwned) rbuf_NotMatching = 1;
368
369#ifdef HAVE_MPI
370 unsigned short sbuf_NotMatching = rbuf_NotMatching;
371 SU2_MPI::Allreduce(&sbuf_NotMatching, &rbuf_NotMatching, 1, MPI_UNSIGNED_SHORT, MPI_MAX, SU2_MPI::GetComm());
372#endif
373
374 if (rbuf_NotMatching != 0)
375 SU2_MPI::Error(string("The solution file ") + restart_filename +
376 string(" doesn't match with the mesh file!\n") +
377 string("It could be empty lines at the end of the file."),
378 CURRENT_FUNCTION);
379
380 /*--- Delete the class memory that is used to load the restart. ---*/
381
382 Restart_Vars = decltype(Restart_Vars){};
383 Restart_Data = decltype(Restart_Data){};
384}
