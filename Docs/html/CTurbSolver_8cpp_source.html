103 void CTurbSolver::LoadRestart(CGeometry** geometry, CSolver*** solver, CConfig* config, int val_iter,
104  bool val_update_geo) {
105  /*--- Restart the solution from file information ---*/
106 
107  const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
108 
109  /*--- To make this routine safe to call in parallel most of it can only be executed by one thread. ---*/
110  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
111  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
112 
113  if (config->GetRead_Binary_Restart()) {
114  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
115  } else {
116  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
117  }
118 
119  /*--- Skip flow variables ---*/
120 
121  unsigned short skipVars = nDim + solver[MESH_0][FLOW_SOL]->GetnVar();
122 
123  /*--- Adjust the number of solution variables in the incompressible
124  restart. We always carry a space in nVar for the energy equation in the
125  mean flow solver, but we only write it to the restart if it is active.
126  Therefore, we must reduce skipVars here if energy is inactive so that
127  the turbulent variables are read correctly. ---*/
128 
129  const bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
130  const bool energy = config->GetEnergy_Equation();
131  const bool weakly_coupled_heat = config->GetWeakly_Coupled_Heat();
132  const bool flamelet = (config->GetKind_FluidModel() == FLUID_FLAMELET);
133 
134  if (incompressible && ((!energy) && (!weakly_coupled_heat) && (!flamelet))) skipVars--;
135 
136  /*--- Load data from the restart into correct containers. ---*/
137 
138  unsigned long counter = 0;
139  for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
140  /*--- Retrieve local index. If this node from the restart file lives
141  on the current processor, we will load and instantiate the vars. ---*/
142 
143  const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
144 
145  if (iPoint_Local > -1) {
146  /*--- We need to store this point's data, so jump to the correct
147  offset in the buffer of data from the restart file and load it. ---*/
148 
149  const auto index = counter * Restart_Vars[1] + skipVars;
150  for (auto iVar = 0u; iVar < nVar; iVar++) nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index + iVar]);
151 
152  /*--- Increment the overall counter for how many points have been loaded. ---*/
153  counter++;
154  }
155  }
156 
157  /*--- Detect a wrong solution file ---*/
158 
159  if (counter != nPointDomain) {
160  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" does not match with the mesh file!\n") +
161  string("This can be caused by empty lines at the end of the file."),
162  CURRENT_FUNCTION);
163  }
164 
165  } // end safe global access, pre and postprocessing are thread-safe.
166  END_SU2_OMP_SAFE_GLOBAL_ACCESS
167 
168  /*--- MPI solution and compute the eddy viscosity ---*/
169 
170  solver[MESH_0][TURB_SOL]->InitiateComms(geometry[MESH_0], config, MPI_QUANTITIES::SOLUTION);
171  solver[MESH_0][TURB_SOL]->CompleteComms(geometry[MESH_0], config, MPI_QUANTITIES::SOLUTION);
172 
173  /*--- For turbulent+species simulations the solver Pre-/Postprocessing is done by the species solver. ---*/
174  if (config->GetKind_Species_Model() == SPECIES_MODEL::NONE && config->GetKind_Trans_Model() == TURB_TRANS_MODEL::NONE) {
175  solver[MESH_0][FLOW_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
176  RUNTIME_FLOW_SYS, false);
177  solver[MESH_0][TURB_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
178  }
179 
180  /*--- Interpolate the solution down to the coarse multigrid levels ---*/
181 
182  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
183  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][TURB_SOL]->GetNodes()->GetSolution(),
184  *geometry[iMesh], solver[iMesh][TURB_SOL]->GetNodes()->GetSolution());
185  solver[iMesh][TURB_SOL]->InitiateComms(geometry[iMesh], config, MPI_QUANTITIES::SOLUTION);
186  solver[iMesh][TURB_SOL]->CompleteComms(geometry[iMesh], config, MPI_QUANTITIES::SOLUTION);
187 
188  if (config->GetKind_Species_Model() == SPECIES_MODEL::NONE) {
189  solver[iMesh][FLOW_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_FLOW_SYS,
190  false);
191  solver[iMesh][TURB_SOL]->Postprocessing(geometry[iMesh], solver[iMesh], config, iMesh);
192  }
193  }
194 
195  /*--- Go back to single threaded execution. ---*/
196  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
197  /*--- Delete the class memory that is used to load the restart. ---*/
198 
199  Restart_Vars = decltype(Restart_Vars){};
200  Restart_Data = decltype(Restart_Data){};
201  }
202  END_SU2_OMP_SAFE_GLOBAL_ACCESS
203 }

205 void CTurbSolver::Impose_Fixed_Values(const CGeometry *geometry, const CConfig *config){
206  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
207 
208  /*--- Check whether turbulence quantities are fixed to far-field values on a half-plane. ---*/
209  if(config->GetTurb_Fixed_Values()){
210 
211  /*--- Form normalized far-field velocity ---*/
212  const su2double* velocity_inf = config->GetVelocity_FreeStreamND();
213  su2double velmag_inf = GeometryToolbox::Norm(nDim, velocity_inf);
214  if(velmag_inf==0)
215  SU2_MPI::Error("Far-field velocity is zero, cannot fix turbulence quantities to inflow values.", CURRENT_FUNCTION);
216  su2double unit_velocity_inf[MAXNDIM];
217  for(unsigned short iDim=0; iDim<nDim; iDim++)
218  unit_velocity_inf[iDim] = velocity_inf[iDim] / velmag_inf;
219 
220  SU2_OMP_FOR_DYN(omp_chunk_size)
221  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
222  if( GeometryToolbox::DotProduct(nDim, geometry->nodes->GetCoord(iPoint), unit_velocity_inf)
223  < config->GetTurb_Fixed_Values_MaxScalarProd() ) {
224  /*--- Set the solution values and zero the residual ---*/
225  nodes->SetSolution_Old(iPoint, Solution_Inf);
226  nodes->SetSolution(iPoint, Solution_Inf);
227  LinSysRes.SetBlock_Zero(iPoint);
228  if (implicit) {
229  /*--- Change rows of the Jacobian (includes 1 in the diagonal) ---*/
230  for(unsigned long iVar=0; iVar<nVar; iVar++)
231  Jacobian.DeleteValsRowi(iPoint*nVar+iVar);
232  }
233  }
234  }
235  END_SU2_OMP_FOR
236  }
