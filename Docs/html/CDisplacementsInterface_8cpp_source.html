38void CDisplacementsInterface::GetDonor_Variable(CSolver *struct_solution, CGeometry *struct_geometry,
39 const CConfig *struct_config, unsigned long Marker_Struct,
40 unsigned long Vertex_Struct, unsigned long Point_Struct) {
41 /*--- The displacements come from the predicted solution, but they are no longer incremental ---*/
42 auto DisplacementDonor = struct_solution->GetNodes()->GetSolution_Pred(Point_Struct);
43
44 for (auto iVar = 0u; iVar < nVar; iVar++)
45 Donor_Variable[iVar] = DisplacementDonor[iVar];
46
47 if (struct_config->GetTime_Domain()) {
48 auto VelocityDonor = struct_solution->GetNodes()->GetSolution_Vel_Pred(Point_Struct);
49 for (auto iVar = nVar/2; iVar < nVar; iVar++)//Assuming dynamic interface always has nVar = 2*nDim, 2D: 4, 3D: 6
50 Donor_Variable[iVar] = VelocityDonor[iVar-nVar/2];
51 }
52}

54void CDisplacementsInterface::SetTarget_Variable(CSolver *mesh_solver, CGeometry *flow_geometry,
55 const CConfig *flow_config, unsigned long Marker_Flow,
56 unsigned long Vertex_Flow, unsigned long Point_Mesh) {
57
58 if (!flow_config->GetTime_Domain()) {
59 /*--- Impose the boundary displacements ---*/
60 mesh_solver->GetNodes()->SetBound_Disp(Point_Mesh,Target_Variable);
61 } else {
62 /*--- Impose the boundary displacements ---*/
63 for (auto iVar = 0u; iVar < nVar/2; iVar++)
64 mesh_solver->GetNodes()->SetBound_Disp(Point_Mesh,iVar,Target_Variable[iVar]);
65
66 /*--- Impose the boundary velocities ---*/
67 for (auto iVar = nVar/2; iVar < nVar; iVar++)
68 mesh_solver->GetNodes()->SetBound_Vel(Point_Mesh,iVar-nVar/2,Target_Variable[iVar]);
69 }
