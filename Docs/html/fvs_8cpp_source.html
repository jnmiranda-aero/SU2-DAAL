97 CNumerics::ResidualType<> CUpwMSW_Flow::ComputeResidual(const CConfig* config) {
98 
99  implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
100 
101  unsigned short iDim, iVar, jVar, kVar;
102  su2double rho_i, rho_j, P_i, P_j, H_i, H_j;
103  su2double ProjVel_i, ProjVel_j, ProjVelst_i, ProjVelst_j;
104  su2double sqvel_i, sqvel_j;
105  su2double alpha, w, dp, onemw;
106  su2double Proj_ModJac_Tensor_i, Proj_ModJac_Tensor_j;
107 
108  /*--- Set parameters in the numerical method ---*/
109  alpha = 6.0;
110 
111  /*--- Calculate supporting geometry parameters ---*/
112 
113  Area = GeometryToolbox::Norm(nDim, Normal);
114 
115  for (iDim = 0; iDim < nDim; iDim++)
116  UnitNormal[iDim] = Normal[iDim]/Area;
117 
118  /*--- Initialize flux & Jacobian vectors ---*/
119 
120  for (iVar = 0; iVar < nVar; iVar++) {
121  Fc_i[iVar] = 0.0;
122  Fc_j[iVar] = 0.0;
123  }
124  if (implicit) {
125  for (iVar = 0; iVar < nVar; iVar++) {
126  for (jVar = 0; jVar < nVar; jVar++) {
127  Jacobian_i[iVar][jVar] = 0.0;
128  Jacobian_j[iVar][jVar] = 0.0;
129  }
130  }
131  }
132 
133  /*--- Load variables from nodes i & j ---*/
134 
135  for (iDim = 0; iDim < nDim; iDim++) {
136  u_i[iDim] = V_i[iDim+1];
137  u_j[iDim] = V_j[iDim+1];
138  }
139  P_i = V_i[nDim+1];
140  P_j = V_j[nDim+1];
141  rho_i = V_i[nDim+2];
142  rho_j = V_j[nDim+2];
143  H_i = V_i[nDim+3];
144  H_j = V_j[nDim+3];
145 
146  /*--- Recompute conservatives ---*/
147 
148  su2double U_i[5] = {0.0}, U_j[5] = {0.0};
149 
150  U_i[0] = rho_i; U_j[0] = rho_j;
151  for (iDim = 0; iDim < nDim; iDim++) {
152  U_i[iDim+1] = rho_i*u_i[iDim];
153  U_j[iDim+1] = rho_j*u_j[iDim];
154  }
155  U_i[nDim+1] = rho_i*H_i - P_i;
156  U_j[nDim+1] = rho_j*H_j - P_j;
157 
158  /*--- Calculate supporting quantities ---*/
159 
160  sqvel_i = 0.0; sqvel_j = 0.0;
161  ProjVel_i = 0.0; ProjVel_j = 0.0;
162  for (iDim = 0; iDim < nDim; iDim++) {
163  sqvel_i += u_i[iDim]*u_i[iDim];
164  sqvel_j += u_j[iDim]*u_j[iDim];
165  ProjVel_i += u_i[iDim]*UnitNormal[iDim];
166  ProjVel_j += u_j[iDim]*UnitNormal[iDim];
167  }
168 
169  /*--- Calculate the state weighting function ---*/
170 
171  dp = fabs(P_j-P_i) / min(P_j, P_i);
172  w = 0.5 * (1.0/(pow(alpha*dp,2.0) +1.0));
173  onemw = 1.0 - w;
174 
175  /*--- Calculate weighted state vector (*) for i & j ---*/
176 
177  for (iVar = 0; iVar < nDim+5; iVar++) {
178  Vst_i[iVar] = onemw*V_i[iVar] + w*V_j[iVar];
179  Vst_j[iVar] = onemw*V_j[iVar] + w*V_i[iVar];
180  }
181  ProjVelst_i = onemw*ProjVel_i + w*ProjVel_j;
182  ProjVelst_j = onemw*ProjVel_j + w*ProjVel_i;
183 
184  for (iDim = 0; iDim < nDim; iDim++) {
185  Velst_i[iDim] = Vst_i[iDim+1];
186  Velst_j[iDim] = Vst_j[iDim+1];
187  }
188 
189  /*--- Flow eigenvalues at i (Lambda+) ---*/
190 
191  for (iDim = 0; iDim < nDim; iDim++) {
192  Lambda_i[iDim] = 0.5*(ProjVelst_i + fabs(ProjVelst_i));
193  }
194 
195  Lambda_i[nDim] = 0.5*( ProjVelst_i + Vst_i[nDim+4] + fabs(ProjVelst_i + Vst_i[nDim+4]) );
196  Lambda_i[nDim+1] = 0.5*( ProjVelst_i - Vst_i[nDim+4] + fabs(ProjVelst_i - Vst_i[nDim+4]) );
197 
198  /*--- Compute projected P, invP, and Lambda ---*/
199 
200  GetPMatrix(&Vst_i[nDim+2], Velst_i, &Vst_i[nDim+4], UnitNormal, P_Tensor);
201  GetPMatrix_inv(&Vst_i[nDim+2], Velst_i, &Vst_i[nDim+4], UnitNormal, invP_Tensor);
202 
203  /*--- Projected flux (f+) at i ---*/
204 
205  for (iVar = 0; iVar < nVar; iVar++) {
206  for (jVar = 0; jVar < nVar; jVar++) {
207  Proj_ModJac_Tensor_i = 0.0;
208 
209  /*--- Compute Proj_ModJac_Tensor = P x Lambda+ x inverse P ---*/
210 
211  for (kVar = 0; kVar < nVar; kVar++)
212  Proj_ModJac_Tensor_i += P_Tensor[iVar][kVar]*Lambda_i[kVar]*invP_Tensor[kVar][jVar];
213  Fc_i[iVar] += Proj_ModJac_Tensor_i*U_i[jVar]*Area;
214  if (implicit)
215  Jacobian_i[iVar][jVar] += Proj_ModJac_Tensor_i*Area;
216  }
217  }
218 
219  /*--- Flow eigenvalues at j (Lambda-) ---*/
220 
221  for (iDim = 0; iDim < nDim; iDim++) {
222  Lambda_j[iDim] = 0.5*(ProjVelst_j - fabs(ProjVelst_j));
223  }
224  Lambda_j[nDim] = 0.5*( ProjVelst_j + Vst_j[nDim+4] - fabs(ProjVelst_j + Vst_j[nDim+4]) );
225  Lambda_j[nDim+1] = 0.5*( ProjVelst_j - Vst_j[nDim+4] - fabs(ProjVelst_j - Vst_j[nDim+4]) );
226 
227  /*--- Compute projected P, invP, and Lambda ---*/
228 
229  GetPMatrix(&Vst_j[nDim+2], Velst_j, &Vst_j[nDim+4], UnitNormal, P_Tensor);
230  GetPMatrix_inv(&Vst_j[nDim+2], Velst_j, &Vst_j[nDim+4], UnitNormal, invP_Tensor);
231 
232  /*--- Projected flux (f-) ---*/
233 
234  for (iVar = 0; iVar < nVar; iVar++) {
235  for (jVar = 0; jVar < nVar; jVar++) {
236  Proj_ModJac_Tensor_j = 0.0;
237  /*--- Compute Proj_ModJac_Tensor = P x Lambda- x inverse P ---*/
238  for (kVar = 0; kVar < nVar; kVar++)
239  Proj_ModJac_Tensor_j += P_Tensor[iVar][kVar]*Lambda_j[kVar]*invP_Tensor[kVar][jVar];
240  Fc_j[iVar] += Proj_ModJac_Tensor_j*U_j[jVar]*Area;
241  if (implicit)
242  Jacobian_j[iVar][jVar] += Proj_ModJac_Tensor_j*Area;
243  }
244  }
245 
246  /*--- Flux splitting, use the i flux as final output. ---*/
247 
248  for (iVar = 0; iVar < nVar; iVar++) {
249  Fc_i[iVar] += Fc_j[iVar];
250  }
251 
252  return ResidualType<>(Fc_i, Jacobian_i, Jacobian_j);
253 
254 }
