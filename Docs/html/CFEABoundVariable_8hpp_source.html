33 /*!
34  * \class CFEABoundVariable
35  * \brief Class that adds storage of boundary variables (tractions) to CFEAVariable.
36  * \note Member variables are allocated only for points marked as "vertex" i.e. on a boundary.
37  * A map is constructed so that variables can be referenced by iPoint instead of iVertex.
38  * \ingroup Structural Finite Element Analysis Variables
39  * \author R. Sanchez.
40  * \version 8.1.0 "Harrier"
41  */

111  /*!

113  */
114  inline void Set_FlowTraction(unsigned long iPoint, const su2double *val_flowTraction) override {
115  if (!fsi_analysis) return;
116  if (!VertexMap.GetVertexIndex(iPoint)) return;
117  for (unsigned long iVar = 0; iVar < nVar; iVar++) FlowTraction(iPoint,iVar) = val_flowTraction[iVar];
118  }
119

123  inline void Add_FlowTraction(unsigned long iPoint, const su2double *val_flowTraction) override {
124  if (!fsi_analysis) return;
125  if (!VertexMap.GetVertexIndex(iPoint)) return;
126  for (unsigned long iVar = 0; iVar < nVar; iVar++) FlowTraction(iPoint,iVar) += val_flowTraction[iVar];
127  }
