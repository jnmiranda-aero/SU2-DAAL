33/*!
34 * \class CFEABoundVariable
35 * \brief Class that adds storage of boundary variables (tractions) to CFEAVariable.
36 * \note Member variables are allocated only for points marked as "vertex" i.e. on a boundary.
37 * A map is constructed so that variables can be referenced by iPoint instead of iVertex.
38 * \ingroup Structural Finite Element Analysis Variables
39 * \author R. Sanchez.
40 * \version 8.0.1 "Harrier"
41 */
42class CFEABoundVariable final : public CFEAVariable {
43protected:
44
45 MatrixType FlowTraction; /*!< \brief Traction from the fluid field. */
46 MatrixType FlowTraction_n; /*!< \brief Traction from the fluid field at time n. */
47
48 MatrixType Residual_Ext_Surf; /*!< \brief Term of the residual due to external forces. */
49 MatrixType Residual_Ext_Surf_n; /*!< \brief Term of the residual due to external forces at time n. */
50
51 CVertexMap<unsigned> VertexMap; /*!< \brief Object that controls accesses to the variables of this class. */
52
53 bool fsi_analysis = false; /*!< \brief If flow tractions are available. */
54
55public:
56 /*!
57 * \brief Constructor of the class.
58 * \param[in] val_fea - Values of the fea solution (initialization value).
59 * \param[in] npoint - Number of points/nodes/vertices in the domain.
60 * \param[in] ndim - Number of dimensions of the problem.
61 * \param[in] nvar - Number of variables of the problem.
62 * \param[in] config - Definition of the particular problem.
63 */
64 CFEABoundVariable(const su2double *val_fea, unsigned long npoint, unsigned long ndim, unsigned long nvar, CConfig *config);
65
66 /*!
67 * \brief Destructor of the class.
68 */
69 ~CFEABoundVariable() override = default;
70
71 /*!
72 * \brief Allocate member variables for points marked as vertex (via "Set_isVertex").
73 * \param[in] config - Definition of the particular problem.
74 */
75 void AllocateBoundaryVariables(CConfig *config);
76
77 /*!
78 * \brief Add surface load to the residual term
79 */
80 inline void Add_SurfaceLoad_Res(unsigned long iPoint, const su2double *val_surfForce) override {
81 if (!VertexMap.GetVertexIndex(iPoint)) return;
82 for (unsigned long iVar = 0; iVar < nVar; iVar++) Residual_Ext_Surf(iPoint,iVar) += val_surfForce[iVar];
83 }
84
85 /*!
86 * \brief Get the residual term due to surface load
87 */
88 inline su2double Get_SurfaceLoad_Res(unsigned long iPoint, unsigned long iVar) const override {
89 if (!VertexMap.GetVertexIndex(iPoint)) return 0.0;
90 return Residual_Ext_Surf(iPoint,iVar);
91 }
92
93 /*!
94 * \brief Clear the surface load residual
95 */
96 inline void Clear_SurfaceLoad_Res() override;
97
98 /*!
99 * \brief Store the surface load as the load for the previous time step.
100 */
101 void Set_SurfaceLoad_Res_n() override;
102
103 /*!
104 * \brief Get the surface load from the previous time step.
105 */
106 inline su2double Get_SurfaceLoad_Res_n(unsigned long iPoint, unsigned long iVar) const override {
107 if (!VertexMap.GetVertexIndex(iPoint)) return 0.0;
108 return Residual_Ext_Surf_n(iPoint,iVar);
109 }
110
111 /*!
112 * \brief Set the flow traction at a node on the structural side
113 */
114 inline void Set_FlowTraction(unsigned long iPoint, const su2double *val_flowTraction) override {
115 if (!fsi_analysis) return;
116 if (!VertexMap.GetVertexIndex(iPoint)) return;
117 for (unsigned long iVar = 0; iVar < nVar; iVar++) FlowTraction(iPoint,iVar) = val_flowTraction[iVar];
118 }
119
120 /*!
121 * \brief Add a value to the flow traction at a node on the structural side
122 */
123 inline void Add_FlowTraction(unsigned long iPoint, const su2double *val_flowTraction) override {
124 if (!fsi_analysis) return;
125 if (!VertexMap.GetVertexIndex(iPoint)) return;
126 for (unsigned long iVar = 0; iVar < nVar; iVar++) FlowTraction(iPoint,iVar) += val_flowTraction[iVar];
127 }
128
129 /*!
130 * \brief Get the residual term due to the flow traction
131 */
132 inline su2double Get_FlowTraction(unsigned long iPoint, unsigned long iVar) const override {
133 if (!fsi_analysis) return 0.0;
134 if (!VertexMap.GetVertexIndex(iPoint)) return 0.0;
135 return FlowTraction(iPoint,iVar);
136 }
137
138 /*!
139 * \brief Set the value of the flow traction at the previous time step.
140 */
141 void Set_FlowTraction_n() override;
142
143 /*!
144 * \brief Retrieve the value of the flow traction from the previous time step.
145 */
146 inline su2double Get_FlowTraction_n(unsigned long iPoint, unsigned long iVar) const override {
147 if (!fsi_analysis) return 0.0;
148 if (!VertexMap.GetVertexIndex(iPoint)) return 0.0;
149 return FlowTraction_n(iPoint,iVar);
150 }
151
152 /*!
153 * \brief Register the flow tractions as input variable.
154 */
155 void RegisterFlowTraction(bool reset) override;
156
157 /*!
158 * \brief Extract the flow traction derivatives.
159 */
160 inline su2double ExtractFlowTractionSensitivity(unsigned long iPoint, unsigned long iDim) const override {
161 if (!fsi_analysis) return 0.0;
162 if (!VertexMap.GetVertexIndex(iPoint)) return 0.0;
163 return SU2_TYPE::GetDerivative(FlowTraction(iPoint,iDim));
164 }
165
166 /*!
167 * \brief Get whether a node is on the boundary
168 */
169 inline bool Get_isVertex(unsigned long iPoint) const override {
170 return VertexMap.GetIsVertex(iPoint);
171 }
172
173 /*!
174 * \brief Set whether a node is on the boundary
175 */
176 inline void Set_isVertex(unsigned long iPoint, bool isVertex) override {
177 VertexMap.SetIsVertex(iPoint,isVertex);
178 }
179
180};
