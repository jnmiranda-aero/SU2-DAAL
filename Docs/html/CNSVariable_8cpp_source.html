48 void CNSVariable::SetRoe_Dissipation_NTS(unsigned long iPoint,
49  su2double val_delta,
50  su2double val_const_DES){
51 
52  const su2double cnu = pow(0.09, 1.5),
53  ch1 = 3.0,
54  ch2 = 1.0,
55  ch3 = 2.0,
56  sigma_max = 1.0;
57 
58  unsigned long iDim;
59  su2double Omega, Omega_2 = 0.0, Baux, Gaux, Lturb, Kaux, Aaux;
60 
61  AD::StartPreacc();
62  AD::SetPreaccIn(Vorticity[iPoint], 3);
63  AD::SetPreaccIn(StrainMag(iPoint));
64  AD::SetPreaccIn(val_delta);
65  AD::SetPreaccIn(val_const_DES);
66  /*--- Density ---*/
67  AD::SetPreaccIn(Solution(iPoint,0));
68  /*--- Laminar viscosity --- */
69  AD::SetPreaccIn(Primitive(iPoint, indices.LaminarViscosity()));
70  /*--- Eddy viscosity ---*/
71  AD::SetPreaccIn(Primitive(iPoint, indices.EddyViscosity()));
72 
73  /*--- Central/upwind blending based on:
74  * Zhixiang Xiao, Jian Liu, Jingbo Huang, and Song Fu. "Numerical
75  * Dissipation Effects on Massive Separation Around Tandem Cylinders",
76  * AIAA Journal, Vol. 50, No. 5 (2012), pp. 1119-1136.
77  * https://doi.org/10.2514/1.J051299
78  * ---*/
79 
80  for (iDim = 0; iDim < 3; iDim++){
81  Omega_2 += pow(Vorticity(iPoint,iDim),2);
82  }
83  Omega = sqrt(Omega_2);
84 
85  Baux = (ch3 * Omega * max(StrainMag(iPoint), Omega)) /
86  max((pow(StrainMag(iPoint),2)+Omega_2)*0.5, 1E-20);
87  Gaux = tanh(pow(Baux,4.0));
88 
89  Kaux = max(sqrt((Omega_2 + pow(StrainMag(iPoint), 2))*0.5), 0.1 * inv_TimeScale);
90 
91  const su2double nu = GetLaminarViscosity(iPoint)/GetDensity(iPoint);
92  const su2double nu_t = GetEddyViscosity(iPoint)/GetDensity(iPoint);
93  Lturb = sqrt((nu + nu_t)/(cnu*Kaux));
94 
95  Aaux = ch2*max((val_const_DES*val_delta/Lturb)/Gaux - 0.5, 0.0);
96 
97  Roe_Dissipation(iPoint) = sigma_max * tanh(pow(Aaux, ch1));
98 
99  AD::SetPreaccOut(Roe_Dissipation(iPoint));
100  AD::EndPreacc();
101 
102 }

104 void CNSVariable::SetRoe_Dissipation_FD(unsigned long iPoint, su2double val_wall_dist){
105 
106  /*--- Constants for Roe Dissipation ---*/
107 
108  const passivedouble k2 = pow(0.41,2.0);
109 
110  AD::StartPreacc();
111  AD::SetPreaccIn(Gradient_Primitive[iPoint], nVar, nDim);
112  AD::SetPreaccIn(val_wall_dist);
113  /*--- Eddy viscosity ---*/
114  AD::SetPreaccIn(Primitive(iPoint, indices.EddyViscosity()));
115  /*--- Laminar viscosity --- */
116  AD::SetPreaccIn(Primitive(iPoint, indices.LaminarViscosity()));
117 
118  su2double uijuij = 0.0;
119 
120  for(unsigned long iDim = 0; iDim < nDim; ++iDim)
121  for(unsigned long jDim = 0; jDim < nDim; ++jDim)
122  uijuij += pow(Gradient_Primitive(iPoint, indices.Velocity()+iDim, jDim),2);
123 
124  uijuij = max(sqrt(uijuij),1e-10);
125 
126  const su2double nu = GetLaminarViscosity(iPoint)/GetDensity(iPoint);
127  const su2double nu_t = GetEddyViscosity(iPoint)/GetDensity(iPoint);
128  const su2double r_d = (nu + nu_t)/(uijuij*k2*pow(val_wall_dist,2));
129 
130  Roe_Dissipation(iPoint) = 1.0-tanh(pow(8.0*r_d,3.0));
131 
132  AD::SetPreaccOut(Roe_Dissipation(iPoint));
133  AD::EndPreacc();
134 }

136 bool CNSVariable::SetPrimVar(unsigned long iPoint, su2double eddy_visc, su2double turb_ke, CFluidModel *FluidModel) {
137 
138  bool RightVol = true;
139 
140  SetVelocity(iPoint); // Computes velocity and velocity^2
141  su2double density = GetDensity(iPoint);
142  su2double staticEnergy = GetEnergy(iPoint)-0.5*Velocity2(iPoint) - turb_ke;
143 
144  /*--- Check will be moved inside fluid model plus error description strings ---*/
145 
146  FluidModel->SetTDState_rhoe(density, staticEnergy);
147 
148  bool check_dens = SetDensity(iPoint);
149  bool check_press = SetPressure(iPoint, FluidModel->GetPressure());
150  bool check_sos = SetSoundSpeed(iPoint, FluidModel->GetSoundSpeed2());
151  bool check_temp = SetTemperature(iPoint, FluidModel->GetTemperature());
152 
153  /*--- Check that the solution has a physical meaning ---*/
154 
155  if (check_dens || check_press || check_sos || check_temp) {
156 
157  /*--- Copy the old solution ---*/
158 
159  for (unsigned long iVar = 0; iVar < nVar; iVar++)
160  Solution(iPoint,iVar) = Solution_Old(iPoint,iVar);
161 
162  /*--- Recompute the primitive variables ---*/
163 
164  SetVelocity(iPoint); // Computes velocity and velocity^2
165  density = GetDensity(iPoint);
166  staticEnergy = GetEnergy(iPoint)-0.5*Velocity2(iPoint) - turb_ke;
167 
168  /*--- Check will be moved inside fluid model plus error description strings ---*/
169 
170  FluidModel->SetTDState_rhoe(density, staticEnergy);
171 
172  SetDensity(iPoint);
173  SetPressure(iPoint, FluidModel->GetPressure());
174  SetSoundSpeed(iPoint, FluidModel->GetSoundSpeed2());
175  SetTemperature(iPoint, FluidModel->GetTemperature());
176 
177  RightVol = false;
178 
179  }
180 
181  /*--- Set enthalpy ---*/
182 
183  SetEnthalpy(iPoint); // Requires pressure computation.
184 
185  /*--- Set laminar viscosity ---*/
186 
187  SetLaminarViscosity(iPoint, FluidModel->GetLaminarViscosity());
188 
189  /*--- Set eddy viscosity ---*/
190 
191  SetEddyViscosity(iPoint, eddy_visc);
192 
193  /*--- Set thermal conductivity ---*/
194 
195  SetThermalConductivity(iPoint, FluidModel->GetThermalConductivity());
196 
197  /*--- Set specific heat ---*/
198 
199  SetSpecificHeatCp(iPoint, FluidModel->GetCp());
200 
201  /*--- Set look-up variables in case of data-driven fluid model ---*/
202  if (DataDrivenFluid) {
203  SetDataExtrapolation(iPoint, FluidModel->GetExtrapolation());
204  SetEntropy(iPoint, FluidModel->GetEntropy());
205  }
206 
207  return RightVol;
208 }
