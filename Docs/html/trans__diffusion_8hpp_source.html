62  /*!
63  * \brief LM transition model specific steps in the ComputeResidual method
64  * \param[in] config - Definition of the particular problem.
65  */
66  void FinishResidualCalc(const CConfig* config) override {
67  const bool implicit = config->GetKind_TimeIntScheme() == EULER_IMPLICIT;
68 
69  /*--- Compute mean effective dynamic viscosity ---*/
70  const su2double diff_i_gamma = Laminar_Viscosity_i + Eddy_Viscosity_i;
71  const su2double diff_j_gamma = Laminar_Viscosity_j + Eddy_Viscosity_j;
72  const su2double diff_i_ReThetaT = 2.0*(Laminar_Viscosity_i + Eddy_Viscosity_i);
73  const su2double diff_j_ReThetaT = 2.0*(Laminar_Viscosity_j + Eddy_Viscosity_j);
74 
75  const su2double diff_gamma = 0.5*(diff_i_gamma + diff_j_gamma);
76  const su2double diff_ReThetaT = 0.5*(diff_i_ReThetaT + diff_j_ReThetaT);
77 
78  Flux[0] = diff_gamma*Proj_Mean_GradScalarVar[0];
79  Flux[1] = diff_ReThetaT*Proj_Mean_GradScalarVar[1];
80 
81  /*--- For Jacobians -> Use of TSL (Thin Shear Layer) approx. to compute derivatives of the gradients ---*/
82  if (implicit) {
83  const su2double proj_on_rho_i = proj_vector_ij/Density_i;
84  Jacobian_i[0][0] = -diff_gamma*proj_on_rho_i; Jacobian_i[0][1] = 0.0;
85  Jacobian_i[1][0] = 0.0; Jacobian_i[1][1] = -diff_ReThetaT*proj_on_rho_i;
86 
87  const su2double proj_on_rho_j = proj_vector_ij/Density_j;
88  Jacobian_j[0][0] = diff_gamma*proj_on_rho_j; Jacobian_j[0][1] = 0.0;
89  Jacobian_j[1][0] = 0.0; Jacobian_j[1][1] = diff_ReThetaT*proj_on_rho_j;
90  }
91  }
