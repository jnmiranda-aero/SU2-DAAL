51void CSourceP1::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i, CConfig *config) {
52
53 /*--- Retrieve the energy at the node i ---*/
54 Energy_i = RadVar_i[0];
55
56 /*--- Retrieve the temperature at the node i ---*/
57 Temperature_i = V_i[nDim+1];
58
59 /*--- Compute the blackbody intensity for gray media ---*/
60 BlackBody_Intensity = 4.0*STEFAN_BOLTZMANN*pow(Temperature_i,4.0);
61
62 /*--- Source term from black-body and energy contributions ---*/
63 val_residual[0] = Absorption_Coeff * Volume * (BlackBody_Intensity - Energy_i);
64
65 /*--- Contribution to the Jacobian ---*/
66 if (implicit) {
67 val_Jacobian_i[0][0] = - Absorption_Coeff * Volume;
68 }
69
70}

79void CAvgGradCorrected_P1::ComputeResidual(su2double *val_residual, su2double **Jacobian_i,
80 su2double **Jacobian_j, CConfig *config) {
81 constexpr int nVar = 1;
82
83 AD::StartPreacc();
84 AD::SetPreaccIn(Coord_i, nDim); AD::SetPreaccIn(Coord_j, nDim);
85 AD::SetPreaccIn(Normal, nDim);
86 AD::SetPreaccIn(RadVar_i,nVar); AD::SetPreaccIn(RadVar_j,nVar);
87 AD::SetPreaccIn(RadVar_Grad_i,nVar,nDim); AD::SetPreaccIn(RadVar_Grad_j,nVar,nDim);
88
89 su2double NormalGrad[nVar], CorrectedGrad[nVar];
90
91 auto proj_vector_ij = ComputeProjectedGradient(nDim, nVar, Normal, Coord_i, Coord_j, RadVar_Grad_i,
92 RadVar_Grad_j, true, RadVar_i, RadVar_j,
93 NormalGrad, CorrectedGrad);
94
95 val_residual[0] = GammaP1*CorrectedGrad[0];
96
97 /*--- For Jacobians -> Use of TSL approx. to compute derivatives of the gradients ---*/
98
99 if (implicit) {
100 Jacobian_i[0][0] = -GammaP1*proj_vector_ij;
101 Jacobian_j[0][0] = GammaP1*proj_vector_ij;
102 }
103
104 AD::SetPreaccOut(val_residual, nVar);
105 AD::EndPreacc();
106}
