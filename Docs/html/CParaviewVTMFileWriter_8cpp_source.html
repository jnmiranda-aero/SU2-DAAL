50void CParaviewVTMFileWriter::WriteData(string val_filename){
51
52 /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
53 val_filename.append(fileExt);
54
55 /*--- If we are in the first zone, create new file and write the file header,
56 * otherwise append to already existing file ---*/
57
58 if (rank == MASTER_NODE){
59 ofstream multiBlockFile;
60 if (iZone == 0)
61 multiBlockFile.open (val_filename.c_str());
62 else
63 multiBlockFile.open(val_filename.c_str(), ios::app);
64
65 if (iZone == 0){
66 multiBlockFile << R"(<VTKFile type="vtkMultiBlockDataSet" version="1.0">)" << endl;
67 multiBlockFile << "<vtkMultiBlockDataSet>" << endl;
68 }
69
70 /*--- Write all blocks that have been added ---*/
71
72 multiBlockFile << output.str();
73
74 /*--- If we are in the last zone, write the additional data and close all blocks ---*/
75
76 if (iZone == nZone-1){
77 multiBlockFile << "</vtkMultiBlockDataSet>" << endl;
78 multiBlockFile << "<FieldData>" << endl;
79 multiBlockFile << "<DataArray type='Float32' Name='TimeValue'>" << endl;
80 multiBlockFile << curTime << endl;
81 multiBlockFile << "</DataArray>" << endl;
82 multiBlockFile << "</FieldData>" << endl;
83 multiBlockFile << "</VTKFile>" << endl;
84 }
85 multiBlockFile.close();
86 }

90void CParaviewVTMFileWriter::AddDataset(const string& foldername, string name, const string& file, CParallelDataSorter* dataSorter){
91
92 /*--- Construct the full file name incl. folder ---*/
93 /*--- Note that the folder name is simply the filename ---*/
94
95 string fullFilename = foldername + "/zone_" + to_string(iZone) + "/" + file;
96
97 /*--- Create an XML writer and dump data into file ---*/
98
99 CParaviewXMLFileWriter XMLWriter(dataSorter);
100 XMLWriter.WriteData(fullFilename);
101
102 /*--- Add the dataset to the vtm file ---*/
103
104 AddDataset(std::move(name), fullFilename + CParaviewXMLFileWriter::fileExt);
105
106 /*--- Update the bandwidth ---*/
107
108 nWrittenDatasets++;
109
110 accumulatedBandwidth += XMLWriter.GetBandwidth();
111
112 bandwidth = accumulatedBandwidth/nWrittenDatasets;
113}

117void CParaviewVTMFileWriter::WriteFolderData(const string& foldername, CConfig *config,
118 string multiZoneHeaderString,
119 CParallelDataSorter* volumeDataSorter,
120 CParallelDataSorter* surfaceDataSorter,
121 CGeometry *geometry){
122
123 if (rank == MASTER_NODE){
124#if defined(_WIN32) || defined(_WIN64) || defined (__WINDOWS__)
125 _mkdir(foldername.c_str());
126 _mkdir((foldername + "/zone_" + to_string(iZone)).c_str());
127#else
128 mkdir(foldername.c_str(), 0777);
129 //mkdir((this->folderName + "/zone_" + to_string(valiZone)).c_str(), 0777);
130 mkdir((foldername + "/zone_" + to_string(iZone)).c_str(), 0777);
131#endif
132 }
133
134 /*--- Open a block for the zone ---*/
135
136 StartBlock(std::move(multiZoneHeaderString));
137
138 StartBlock("Internal");
139 AddDataset(foldername,"Internal", "Internal", volumeDataSorter);
140 EndBlock();
141
142 /*--- Open a block for the boundary ---*/
143
144 StartBlock("Boundary");
145
146 /*--- Loop over all markers used in the config file ---*/
147
148 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_CfgFile(); iMarker++){
149
150 /*--- Get the name of the marker ---*/
151
152 string markerTag = config->GetMarker_CfgFile_TagBound(iMarker);
153
154 /*--- If the current marker can be found on this partition store its name.
155 * Note that we have to provide a vector of markers to the sorter routine, although we only do
156 * one marker at a time, i.e. ::marker always contains one item. ---*/
157
158 vector<string> marker;
159 for (unsigned short jMarker = 0; jMarker < config->GetnMarker_All(); jMarker++){
160
161 /*--- We want to write all markers except send-receive markers ---*/
162
163 if (config->GetMarker_All_TagBound(jMarker) == markerTag &&
164 config->GetMarker_All_KindBC(jMarker) != SEND_RECEIVE){
165 marker.push_back(markerTag);
166 }
167 }
168
169 /*--- Only sort if there is at least one processor that has this marker ---*/
170
171 int globalMarkerSize = 0, localMarkerSize = marker.size();
172 SU2_MPI::Allreduce(&localMarkerSize, &globalMarkerSize, 1, MPI_INT, MPI_SUM, SU2_MPI::GetComm());
173
174 if (globalMarkerSize > 0){
175
176 /*--- Sort connectivity of the current marker ---*/
177
178 surfaceDataSorter->SortConnectivity(config, geometry, marker);
179 surfaceDataSorter->SortOutputData();
180
181 /*--- Add the dataset ---*/
182
183 AddDataset(foldername, markerTag, markerTag, surfaceDataSorter);
184
185 }
186 }
187
188 /*--- End "Boundary" block ---*/
189 EndBlock();
190 /*--- End "Zone" block ---*/
191 EndBlock();
192
193
194}
