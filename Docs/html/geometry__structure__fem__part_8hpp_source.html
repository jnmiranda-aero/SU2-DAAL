35/*!
36 * \struct CUnsignedLong2T
37 * \brief Helper struct used to store two integral types as one entity.
38 */
39struct CUnsignedLong2T {
40 unsigned long long0; /*!< \brief First integer to store in this class. */
41 unsigned long long1; /*!< \brief Second integer to store in this class. */
42
43 CUnsignedLong2T(unsigned long a = 0, unsigned long b = 0) : long0(a), long1(b) {}
44
45 inline bool operator<(const CUnsignedLong2T& other) const {
46 if (long0 != other.long0) return (long0 < other.long0);
47 return (long1 < other.long1);
48 }
49
50 inline bool operator==(const CUnsignedLong2T& other) const {
51 return (long0 == other.long0) && (long1 == other.long1);
52 }
53};

75/*!
76 * \class CFaceOfElement
77 * \brief Class used in the partitioning of the FEM grid as well as the building of
78 the faces of DG. It stores a face of an element.
79 */
80class CFaceOfElement {
81 public:
82 unsigned short nCornerPoints; /*!< \brief Number of corner points of the face. */
83 unsigned long cornerPoints[4]; /*!< \brief Global ID's of ther corner points. */
84 unsigned long elemID0, elemID1; /*!< \brief Element ID's to the left and right. */
85 unsigned short nPolyGrid0, nPolyGrid1; /*!< \brief Polynomial degrees of the grid of the elements
86 to the left and right. */
87 unsigned short nPolySol0, nPolySol1; /*!< \brief Polynomial degrees of the solution of the elements
88 to the left and right. */
89 unsigned short nDOFsElem0, nDOFsElem1; /*!< \brief Number of DOFs of the elements to the left and right. */
90 unsigned short elemType0, elemType1; /*!< \brief Type of the elements to the left and right. */
91 unsigned short faceID0, faceID1; /*!< \brief The local face ID in the corresponding elements
92 to the left and right of the face. */
93 unsigned short periodicIndex; /*!< \brief Periodic indicator of the face. A value of 0 means no
94 periodic face. A value larger than 0 gives the index of
95 the periodic boundary + 1. */
96 unsigned short periodicIndexDonor; /*!< \brief Periodic indicator of the donor face. A value of 0 means no
97 periodic donor face. A value larger than 0 gives the index of
98 the periodic donor boundary + 1. */
99 short faceIndicator; /*!< \brief The corresponding boundary marker if this face is on a
100 boundary. For an internal face the value is -1,
101 while an invalidated face has the value -2. */
102 bool JacFaceIsConsideredConstant; /*!< \brief Whether or not the Jacobian of the transformation
103 to the standard element is considered constant. */
104 bool elem0IsOwner; /*!< \brief Whether or not the neighboring element 0 is the owner
105 of the face. If false, element 1 is the owner. */
106
107 /* Standard constructor and destructor. */
108 CFaceOfElement();
109 ~CFaceOfElement() {}
110
111 /* Alternative constructor to set the corner points. */
112 CFaceOfElement(const unsigned short VTK_Type, const unsigned short nPoly, const unsigned long* Nodes);
113
114 /* Copy constructor and assignment operator. */
115 inline CFaceOfElement(const CFaceOfElement& other) { Copy(other); }
116
117 inline CFaceOfElement& operator=(const CFaceOfElement& other) {
118 Copy(other);
119 return (*this);
120 }
121
122 /* Less than operator. Needed for the sorting and searching. */
123 bool operator<(const CFaceOfElement& other) const;
124
125 /* Equal operator. Needed for removing double entities. */
126 bool operator==(const CFaceOfElement& other) const;
127
128 /*--- Member function, which creates a unique numbering for the corner points.
129 A sort in increasing order is OK for this purpose. ---*/
130 inline void CreateUniqueNumbering(void) { std::sort(cornerPoints, cornerPoints + nCornerPoints); }
131
132 /*--- Member function, which creates a unique numbering for the corner points
133 while the orientation is taken into account. ---*/
134 void CreateUniqueNumberingWithOrientation(void);
135
136 private:
137 /*--- Copy function, which copies the data of the given object into the current object. ---*/
138 void Copy(const CFaceOfElement& other);
139};
