33 /*!
34  * \brief Symbols that need to be defined before including this header:
35  * \param[in] ARRAY_T - The desired specialization of simd::Array.
36  * \param[in] SCALAR_T - Scalar type.
37  * \param[in] REGISTER_T - Intrinsic type.
38  * \param[in] SIZE_TAG - Dummy object associated with the simd size.
39  * \param[in] ARRAY_BOILERPLATE - Generates the general ctors, access, etc.
40  * \note On top of that, the intrinsic functions must be wrapped to
41  * strip their type / size characteristics, e.g. _mm256_add_pd -> add_p,
42  * overload resolution will do the rest. The first four symbols are
43  * undefined once we are done using them.
44  */
45 template <>
46 class ARRAY_T {
47 #define FOREACH SU2_OMP_SIMD for (size_t k = 0; k < Size; ++k)
48  template <class F, class S>
49  FORCEINLINE static S second(F, S s) {
50  return s;
51  }
52 
53  public:
54  using Scalar = SCALAR_T;
55  using Register = REGISTER_T;
56  enum : size_t { Align = alignof(Register) };
57  enum : size_t { Size = sizeof(Register) / sizeof(Scalar) };
58 
59  /*--- The infamous union "hack", sue me. ---*/
60  union {
61  Register reg;
62  Scalar x_[Size];
63  };
64 
65  /*--- Same basic construction operations as the general case. ---*/
66 
67  ARRAY_BOILERPLATE
68 
69  /*--- Special construction using the "register type" directly. ---*/
70 
71  FORCEINLINE Array(Register y) { reg = y; }
72  FORCEINLINE Array(const Array& other) noexcept { reg = other.reg; }
73 
74  /*--- Specialized construction primitives. ---*/
75 
76  FORCEINLINE void bcast(Scalar x) { reg = set1_p(SIZE_TAG, x); }
77  FORCEINLINE void load(const Scalar* ptr) { reg = loadu_p(SIZE_TAG, ptr); }
78  FORCEINLINE void loada(const Scalar* ptr) { reg = load_p(SIZE_TAG, ptr); }
79  FORCEINLINE void store(Scalar* ptr) const { storeu_p(ptr, reg); }
80  FORCEINLINE void storea(Scalar* ptr) const { store_p(ptr, reg); }
81  FORCEINLINE void stream(Scalar* ptr) const { stream_p(ptr, reg); }
82  template <class T>
83  FORCEINLINE void gather(const Scalar* begin, const T& offsets) {
84  FOREACH x_[k] = begin[offsets[k]];
85  }
86 
87  /*--- Compound assignement operators. ---*/
88 
89 #define MAKE_COMPOUND(OP, IMPL) \
90  FORCEINLINE Array& operator OP(Scalar x) { \
91  reg = IMPL(reg, set1_p(SIZE_TAG, x)); \
92  return *this; \
93  } \
94  FORCEINLINE Array& operator OP(const Array& other) noexcept { \
95  reg = IMPL(reg, other.reg); \
96  return *this; \
97  }
98  MAKE_COMPOUND(=, second)
99  MAKE_COMPOUND(+=, add_p)
100  MAKE_COMPOUND(-=, sub_p)
101  MAKE_COMPOUND(*=, mul_p)
102  MAKE_COMPOUND(/=, div_p)
103 #undef MAKE_COMPOUND
104 
105 #undef FOREACH
106 };
