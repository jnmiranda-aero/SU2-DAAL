85 void PrintingToolbox::CTablePrinter::PrintHeader() {
86  if (print_header_top_line_) PrintHorizontalLine();
87  *out_stream_ << separator_;
88  int indent = 0;
89  for (int i = 0; i < GetNumColumns(); ++i) {
90  std::stringstream ss;
91 
92  ss << column_headers_.at(i).substr(0, column_widths_.at(i));
93 
94  indent = 0;
95 
96  if (align_ == LEFT)
97  *out_stream_ << std::left;
98  else if (align_ == RIGHT)
99  *out_stream_ << std::right;
100  else if (align_ == CENTER) {
101  *out_stream_ << std::right;
102  indent = (int)(column_widths_.at(i) - ss.str().size()) / 2;
103  }
104 
105  *out_stream_ << std::setw(column_widths_.at(i) - indent) << column_headers_.at(i).substr(0, column_widths_.at(i));
106  if (i != GetNumColumns() - 1) {
107  *out_stream_ << std::setw(indent + (int)inner_separator_.size()) << inner_separator_;
108  }
109  }
110 
111  *out_stream_ << std::setw((int)separator_.size() + indent) << separator_;
112  *out_stream_ << "\n";
113  if (print_header_bottom_line_) {
114  PrintHorizontalLine();
115  }
116 }
