251 /*!
252 * \brief Get the projected velocity in a unitary vector direction (compressible solver).
253 * \param[in] val_vector - Direction of projection.
254 * \return Value of the projected velocity.
255 */
256 inline su2double GetProjVel(unsigned long iPoint, const su2double *val_vector) const final {
257 su2double ProjVel = 0.0;
258 for (unsigned long iDim = 0; iDim < nDim; iDim++)
259 ProjVel += Primitive(iPoint,iDim+indices.Velocity())*val_vector[iDim];
260 return ProjVel;

263 /*!
264 * \brief Set the velocity vector from the solution.
265 * \param[in] val_velocity - Pointer to the velocity.
266 */
267 inline void SetVelocity(unsigned long iPoint) final {
268 Velocity2(iPoint) = 0.0;
269 for (unsigned long iDim = 0; iDim < nDim; iDim++) {
270 Primitive(iPoint,iDim+indices.Velocity()) = Solution(iPoint,iDim+1) / Solution(iPoint,0);
271 Velocity2(iPoint) += pow(Primitive(iPoint,iDim+indices.Velocity()),2);
272 }
273 }
