267  inline void SetVelocity(unsigned long iPoint) final {
268  Velocity2(iPoint) = 0.0;
269  for (unsigned long iDim = 0; iDim < nDim; iDim++) {
270  Primitive(iPoint,iDim+indices.Velocity()) = Solution(iPoint,iDim+1) / Solution(iPoint,0);
271  Velocity2(iPoint) += pow(Primitive(iPoint,iDim+indices.Velocity()),2);
272  }
