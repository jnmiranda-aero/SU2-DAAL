32 CCGNSMeshReaderFVM::CCGNSMeshReaderFVM(CConfig* val_config, unsigned short val_iZone, unsigned short val_nZone)
33  : CMeshReaderFVM(val_config, val_iZone, val_nZone) {
34 #ifdef HAVE_CGNS
35  OpenCGNSFile(config->GetMesh_FileName());
36 
37  /*--- Read the basic information about the database and zone(s). ---*/
38  ReadCGNSDatabaseMetadata();
39  ReadCGNSZoneMetadata();
40 
41  /*--- Read the point coordinates into linear partitions. ---*/
42  ReadCGNSPointCoordinates();
43 
44  /*--- Loop over all sections to access the grid connectivity. We
45  treat the interior and boundary elements with separate routines.
46  If we have found that this is a boundary section (we assume
47  that internal cells and boundary cells do not exist in the same
48  section together), the master node reads the boundary section.
49  Otherwise, all ranks read and communicate the interior sections. ---*/
50  ReadCGNSSectionMetadata();
51  numberOfMarkers = 0;
52  for (int s = 0; s < nSections; s++) {
53  if (isInterior[s]) {
54  ReadCGNSVolumeSection(s);
55  } else {
56  numberOfMarkers++;
57  ReadCGNSSurfaceSection(s);
58  }
59  }
60 
61  /*--- We have extracted all CGNS data. Close the CGNS file. ---*/
62  if (cg_close(cgnsFileID)) cg_error_exit();
63 
64  /*--- Put our CGNS data into the class data for the mesh reader. ---*/
65  ReformatCGNSVolumeConnectivity();
66  ReformatCGNSSurfaceConnectivity();
67 
68 #else
69  SU2_MPI::Error(string(" SU2 built without CGNS support. \n") + string(" To use CGNS, build SU2 accordingly."),
70  CURRENT_FUNCTION);
71 #endif
72 }

77 void CCGNSMeshReaderFVM::OpenCGNSFile(const string& val_filename) {
78  /*--- Check whether the supplied file is truly a CGNS file. ---*/
79 
80  int file_type;
81  float file_version;
82  if (cg_is_cgns(val_filename.c_str(), &file_type) != CG_OK) {
83  SU2_MPI::Error(val_filename + string(" was not found or is not a properly formatted") +
84  string(" CGNS file.\nNote that SU2 expects unstructured") +
85  string(" CGNS files in ADF data format."),
86  CURRENT_FUNCTION);
87  }
88 
89  /*--- Open the CGNS file for reading. The value of cgnsFileID returned
90  is the specific index number for this file and will be
91  repeatedly used in the function calls. ---*/
92 
93  if (cg_open(val_filename.c_str(), CG_MODE_READ, &cgnsFileID)) cg_error_exit();
94  if (rank == MASTER_NODE) {
95  cout << "Reading the CGNS file: ";
96  cout << val_filename.c_str() << "." << endl;
97  }
98  if (cg_version(cgnsFileID, &file_version)) cg_error_exit();
99  if (rank == MASTER_NODE) {
100  if (file_version < 4.0) {
101  cout
102  << "WARNING: The CGNS file version (" << file_version
103  << ") is old and may cause high memory usage issues, consider updating the file with the cgnsupdate tool.\n";
104  }
105  }

108 void CCGNSMeshReaderFVM::ReadCGNSDatabaseMetadata() {
109  /*--- Get the number of databases. This is the highest node
110  in the CGNS heirarchy. ---*/
111 
112  int nbases;
113  if (cg_nbases(cgnsFileID, &nbases)) cg_error_exit();
114  if (rank == MASTER_NODE) cout << "CGNS file contains " << nbases << " database(s)." << endl;
115 
116  /*--- Check if there is more than one database. Throw an
117  error if there is because this reader can currently
118  only handle one database. ---*/
119 
120  if (nbases > 1) {
121  SU2_MPI::Error("CGNS reader currently can only handle 1 database.", CURRENT_FUNCTION);
122  }
123 
124  /*--- Read the database. Note that the CGNS indexing starts at 1. ---*/
125 
126  int cell_dim, phys_dim;
127  char basename[CGNS_STRING_SIZE];
128  if (cg_base_read(cgnsFileID, cgnsBase, basename, &cell_dim, &phys_dim)) cg_error_exit();
129  if (rank == MASTER_NODE) {
130  cout << "Database " << cgnsBase << ", " << basename << ": ";
131  cout << " cell dimension of " << cell_dim << ", physical ";
132  cout << "dimension of " << phys_dim << "." << endl;
133  }
134 
135  /*--- Set the number of dimensions baed on cell_dim. ---*/
136 
137  dimension = (unsigned short)cell_dim;
138 }

140 void CCGNSMeshReaderFVM::ReadCGNSZoneMetadata() {
141  /*--- First, check all sections to find the element types and to
142  classify them as either surface or volume elements. We will also
143  perform some error checks here to avoid partitioning issues. ---*/
144 
145  /*--- Get the number of zones for this base. ---*/
146 
147  int nzones;
148  if (cg_nzones(cgnsFileID, cgnsBase, &nzones)) cg_error_exit();
149  if (rank == MASTER_NODE) {
150  cout << nzones << " total zone(s)." << endl;
151  }
152 
153  /*--- Check if there is more than one zone. Until we enable it, we
154  will require a single zone CGNS file. Multizone problems can still
155  be run with CGNS by using separate CGNS files for each zone. ---*/
156 
157  if (nzones > 1) {
158  SU2_MPI::Error(string("CGNS reader currently expects only 1 zone per CGNS file.") +
159  string("Multizone problems can be run with separate CGNS files for each zone."),
160  CURRENT_FUNCTION);
161  }
162 
163  /*--- Read the basic information for this zone, including
164  the name and the number of vertices, cells, and
165  boundary cells which are stored in the cgsize variable. ---*/
166 
167  vector<cgsize_t> cgsize(3);
168  ZoneType_t zonetype;
169  char zonename[CGNS_STRING_SIZE];
170  if (cg_zone_read(cgnsFileID, cgnsBase, cgnsZone, zonename, cgsize.data())) cg_error_exit();
171 
172  /*--- Rename the zone size information for clarity.
173  NOTE: The number of cells here may be only the number of
174  interior elements or it may be the total. This needs to
175  be counted explicitly later. ---*/
176 
177  numberOfGlobalPoints = cgsize[0];
178  int nElemCGNS = cgsize[1];
179 
180  /*--- Get some additional information about the current zone. ---*/
181 
182  if (cg_zone_type(cgnsFileID, cgnsBase, cgnsZone, &zonetype)) cg_error_exit();
183 
184  /*--- Check for an unstructured mesh. Throw an error if not found. ---*/
185 
186  if (zonetype != Unstructured)
187  SU2_MPI::Error("Structured CGNS zone found while unstructured expected.", CURRENT_FUNCTION);
188 
189  /*--- Print current zone info to the console. ---*/
190 
191  if (rank == MASTER_NODE) {
192  cout << "Zone " << cgnsZone << ", " << zonename << ": ";
193  cout << numberOfGlobalPoints << " total vertices, ";
194  cout << nElemCGNS << " total elements." << endl;
195  }
196 
197  /*--- Retrieve the number of grids in this zone. For now, we know
198  this is one, but to be more general, this will need to check and
199  allow for a loop over all grids. ---*/
200 
201  int ngrids;
202  if (cg_ngrids(cgnsFileID, cgnsBase, cgnsZone, &ngrids)) cg_error_exit();
203  if (ngrids > 1) {
204  SU2_MPI::Error("CGNS reader currently handles only 1 grid per zone.", CURRENT_FUNCTION);
205  }
206 }

208 void CCGNSMeshReaderFVM::ReadCGNSPointCoordinates() {
209  /*--- Compute the number of points that will be on each processor.
210  This is a linear partitioning with the addition of a simple load
211  balancing for any remainder points. ---*/
212 
213  CLinearPartitioner pointPartitioner(numberOfGlobalPoints, 0);
214 
215  /*--- Store the local number of nodes for this rank. ---*/
216 
217  numberOfLocalPoints = pointPartitioner.GetSizeOnRank(rank);
218 
219  /*--- Create buffer to hold the grid coordinates for our rank. ---*/
220 
221  localPointCoordinates.resize(dimension);
222  for (int k = 0; k < dimension; k++) localPointCoordinates[k].resize(numberOfLocalPoints, 0.0);
223 
224  /*--- Set the value of range_max to the total number of nodes in
225  the unstructured mesh. Also allocate memory for the temporary array
226  that will hold the grid coordinates as they are extracted. Note the
227  +1 for CGNS convention. ---*/
228 
229  cgsize_t range_min = (cgsize_t)pointPartitioner.GetFirstIndexOnRank(rank) + 1;
230  auto range_max = (cgsize_t)pointPartitioner.GetLastIndexOnRank(rank);
231 
232  /*--- Loop over each set of coordinates. ---*/
233 
234  for (int k = 0; k < dimension; k++) {
235  /*--- Read the coordinate info. This will retrieve the
236  data type (either RealSingle or RealDouble) as
237  well as the coordname which will specify the
238  type of data that it is based in the SIDS convention.
239  This might be "CoordinateX," for instance. ---*/
240 
241  char coordname[CGNS_STRING_SIZE];
242  DataType_t datatype;
243  if (cg_coord_info(cgnsFileID, cgnsBase, cgnsZone, k + 1, &datatype, coordname)) cg_error_exit();
244  if (rank == MASTER_NODE) {
245  cout << "Loading " << coordname;
246  if (size > SINGLE_NODE) {
247  cout << " values into linear partitions." << endl;
248  } else {
249  cout << " values." << endl;
250  }
251  }
252 
253  /*--- Check the coordinate name to decide the index for storage. ---*/
254 
255  unsigned short indC = 0;
256  if (string(coordname) == "CoordinateX")
257  indC = 0;
258  else if (string(coordname) == "CoordinateY")
259  indC = 1;
260  else if (string(coordname) == "CoordinateZ")
261  indC = 2;
262  else
263  SU2_MPI::Error(string("Unknown coordinate name, ") + coordname + string(", in the CGNS file."), CURRENT_FUNCTION);
264 
265  /*--- Now read our rank's chunk of coordinates from the file.
266  Ask for datatype RealDouble and let CGNS library do the translation
267  when RealSingle is found. ---*/
268 
269  if (cg_coord_read(cgnsFileID, cgnsBase, cgnsZone, coordname, RealDouble, &range_min, &range_max,
270  localPointCoordinates[indC].data()))
271  cg_error_exit();
272  }
273 }

275 void CCGNSMeshReaderFVM::ReadCGNSSectionMetadata() {
276  /*--- Begin section for retrieving the connectivity info. ---*/
277 
278  if ((rank == MASTER_NODE) && (size > SINGLE_NODE)) cout << "Distributing connectivity across all ranks." << endl;
279 
280  /*--- First check the number of sections. ---*/
281 
282  if (cg_nsections(cgnsFileID, cgnsBase, cgnsZone, &nSections)) cg_error_exit();
283  if (rank == MASTER_NODE) {
284  cout << "Number of connectivity sections is ";
285  cout << nSections << "." << endl;
286  }
287 
288  /*--- Prepare several data structures to hold the various
289  pieces of information describing each section. ---*/
290 
291  isInterior.resize(nSections);
292  nElems.resize(nSections, 0);
293  elemOffset.resize(nSections + 1, 0);
294  elemOffset[0] = 0;
295  connElems.resize(nSections);
296  sectionNames.resize(nSections, vector<char>(CGNS_STRING_SIZE));
297  numberOfGlobalElements = 0;
298 
299  for (int s = 0; s < nSections; s++) {
300  /*--- Read the connectivity details for this section. ---*/
301 
302  int nbndry, parent_flag, vtk_type;
303  cgsize_t startE, endE, sizeNeeded;
304  ElementType_t elemType;
305  if (cg_section_read(cgnsFileID, cgnsBase, cgnsZone, s + 1, sectionNames[s].data(), &elemType, &startE, &endE,
306  &nbndry, &parent_flag))
307  cg_error_exit();
308 
309  /*--- Compute the total element count in this section (global). ---*/
310 
311  unsigned long element_count = (endE - startE + 1);
312 
313  /* Get the details for the CGNS element type in this section. */
314 
315  string elem_name = GetCGNSElementType(elemType, vtk_type);
316 
317  /* We assume that each section contains interior elements by default.
318  If we find 1D elements in a 2D problem or 2D elements in a 3D
319  problem, then we know the section must contain boundary elements.
320  We assume that each section is composed of either entirely interior
321  or entirely boundary elements. */
322 
323  isInterior[s] = true;
324 
325  if (elemType == MIXED) {
326  /* For a mixed section, we check the type of the first element
327  so that we can correctly label this section as an interior or
328  boundary element section. Here, we also assume that a section
329  can not hold both interior and boundary elements. First, get
330  the size required to read a single element from the section. */
331 
332  if (cg_ElementPartialSize(cgnsFileID, cgnsBase, cgnsZone, s + 1, startE, startE, &sizeNeeded) != CG_OK)
333  cg_error_exit();
334 
335  /* A couple of auxiliary vectors for mixed element sections. */
336 
337  vector<cgsize_t> connElemCGNS(sizeNeeded);
338  vector<cgsize_t> connOffsetCGNS(2, 0);
339 
340  /* Retrieve the connectivity information for the first element. */
341 
342  if (cg_poly_elements_partial_read(cgnsFileID, cgnsBase, cgnsZone, s + 1, startE, startE, connElemCGNS.data(),
343  connOffsetCGNS.data(), nullptr) != CG_OK)
344  cg_error_exit();
345 
346  /* The element type is in the first position of the connectivity
347  information that we retrieved from the CGNS file. */
348 
349  elemType = ElementType_t(connElemCGNS[0]);
350  }
351 
352  /* Check for 1D elements in 2D problems, or for 2D elements in
353  3D problems. If found, mark the section as a boundary section. */
354 
355  if ((dimension == 2) && (elemType == BAR_2 || elemType == BAR_3)) isInterior[s] = false;
356  if ((dimension == 3) && (elemType == TRI_3 || elemType == QUAD_4)) isInterior[s] = false;
357 
358  /*--- Increment the global element offset for each section
359  based on whether or not this is a surface or volume section.
360  We also keep a running count of the total elements globally. ---*/
361 
362  elemOffset[s + 1] = elemOffset[s];
363  if (!isInterior[s])
364  elemOffset[s + 1] += element_count;
365  else
366  numberOfGlobalElements += element_count;
367 
368  /*--- Print some information to the console. ---*/
369 
370  if (rank == MASTER_NODE) {
371  cout << "Section " << string(sectionNames[s].data());
372  cout << " contains " << element_count << " elements";
373  cout << " of type " << elem_name << "." << endl;
374  }
375  }
376 }

378 void CCGNSMeshReaderFVM::ReadCGNSVolumeSection(int val_section) {
379  /*--- In this routine, each rank will read a chunk of the element
380  connectivity for a single specified section of the CGNS mesh file.
381  All operations are executed in parallel here and the reading of the
382  section proceeds based on a linear partitioning of the elements
383  across all ranks in the calculation. We will use partial reads of
384  the CGNS section from the CGNS API to accomplish this. Once each
385  rank has a linear chunk of the mesh, we will redistribute the
386  connectivity to match the linear partitioning of the grid points,
387  not the elements, since the points control the overall partitioning. ---*/
388 
389  int nbndry, parent_flag, npe, iProcessor;
390  unsigned long iElem = 0, iPoint = 0, iNode = 0, jNode = 0;
391  cgsize_t startE, endE;
392  ElementType_t elemType;
393  char sectionName[CGNS_STRING_SIZE];
394 
395  /*--- Read the connectivity details for this section.
396  Store the total number of elements in this section
397  to be used later for memory allocation. ---*/
398 
399  if (cg_section_read(cgnsFileID, cgnsBase, cgnsZone, val_section + 1, sectionName, &elemType, &startE, &endE, &nbndry,
400  &parent_flag))
401  cg_error_exit();
402 
403  /*--- Compute element linear partitioning ---*/
404 
405  unsigned long element_count = (endE - startE + 1);
406  CLinearPartitioner elementPartitioner(element_count, startE, true);
407 
408  /*--- Store the number of elements that this rank is responsible for
409  in the current section. ---*/
410 
411  nElems[val_section] = elementPartitioner.GetSizeOnRank(rank);
412 
413  /*--- Allocate some memory for the handling the connectivity
414  and auxiliary data that we need to communicate. ---*/
415 
416  vector<cgsize_t> elemTypes(nElems[val_section], 0);
417  vector<cgsize_t> nPoinPerElem(nElems[val_section], 0);
418  vector<cgsize_t> elemGlobalID(nElems[val_section], 0);
419 
420  /*--- Determine the size of the vector needed to read the connectivity
421  data from the CGNS file. Only call the CGNS API if we have a non-zero
422  number of elements on this rank. ---*/
423 
424  cgsize_t sizeNeeded = 0, sizeOffset = 0;
425  if (nElems[val_section] > 0) {
426  if (cg_ElementPartialSize(cgnsFileID, cgnsBase, cgnsZone, val_section + 1,
427  (cgsize_t)elementPartitioner.GetFirstIndexOnRank(rank),
428  (cgsize_t)elementPartitioner.GetLastIndexOnRank(rank), &sizeNeeded) != CG_OK)
429  cg_error_exit();
430  }
431 
432  /*--- Allocate the memory for the connectivity, the offset if needed
433  and read the data. ---*/
434 
435  vector<cgsize_t> connElemCGNS(sizeNeeded, 0);
436  if (elemType == MIXED || elemType == NFACE_n || elemType == NGON_n) {
437  sizeOffset = nElems[val_section] + 1;
438  }
439  vector<cgsize_t> connOffsetCGNS(sizeOffset, 0);
440 
441  /*--- Retrieve the connectivity information and store. Note that
442  we are only accessing our rank's piece of the data here in the
443  partial read function in the CGNS API. Only call the CGNS API
444  if we have a non-zero number of elements on this rank. ---*/
445 
446  if (nElems[val_section] > 0) {
447  if (elemType == MIXED || elemType == NFACE_n || elemType == NGON_n) {
448  if (cg_poly_elements_partial_read(cgnsFileID, cgnsBase, cgnsZone, val_section + 1,
449  (cgsize_t)elementPartitioner.GetFirstIndexOnRank(rank),
450  (cgsize_t)elementPartitioner.GetLastIndexOnRank(rank), connElemCGNS.data(),
451  connOffsetCGNS.data(), nullptr) != CG_OK)
452  cg_error_exit();
453  } else {
454  if (cg_elements_partial_read(
455  cgnsFileID, cgnsBase, cgnsZone, val_section + 1, (cgsize_t)elementPartitioner.GetFirstIndexOnRank(rank),
456  (cgsize_t)elementPartitioner.GetLastIndexOnRank(rank), connElemCGNS.data(), nullptr) != CG_OK)
457  cg_error_exit();
458  }
459  }
460 
461  /*--- Print some information to the console. ---*/
462 
463  if (rank == MASTER_NODE) {
464  cout << "Loading volume section " << string(sectionName);
465  cout << " from file." << endl;
466  }
467 
468  /*--- Find the number of nodes required to represent
469  this type of element. ---*/
470 
471  if (cg_npe(elemType, &npe)) cg_error_exit();
472 
473  /*--- Check whether the sections contains a mixture of multiple
474  element types, which will require special handling to get the
475  element type one-by-one when reading. ---*/
476 
477  bool isMixed = false;
478  if (elemType == MIXED) {
479  isMixed = true;
480  }
481 
482  /*--- Loop through all of the elements in this section to get more
483  information and to decide whether it has interior elements. ---*/
484 
485  unsigned long counterCGNS = 0;
486  for (iElem = 0; iElem < nElems[val_section]; iElem++) {
487  ElementType_t iElemType = elemType;
488 
489  /*--- If we have a mixed element section, we need to check the elem
490  type one-by-one. We also must manually advance the counter to the
491  next element's position in the buffer. ---*/
492 
493  if (isMixed) {
494  iElemType = ElementType_t(connElemCGNS[counterCGNS]);
495  npe = connOffsetCGNS[iElem + 1] - connOffsetCGNS[iElem] - 1;
496  counterCGNS++;
497  for (int jj = 0; jj < npe; jj++) counterCGNS++;
498  }
499 
500  /*--- Store the number of points per element for the current elem. ---*/
501 
502  nPoinPerElem[iElem] = npe;
503 
504  /*--- Store the global ID for this element. Note the -1 to move
505  from CGNS convention to SU2 convention. We also subtract off
506  an additional offset in case we have found boundary sections
507  prior to this one, in order to keep the internal element global
508  IDs indexed starting from zero. ---*/
509 
510  elemGlobalID[iElem] = (elementPartitioner.GetFirstIndexOnRank(rank) + iElem - elemOffset[val_section]);
511 
512  /* Get the VTK type for this element. */
513 
514  int vtk_type;
515  string elem_name = GetCGNSElementType(iElemType, vtk_type);
516  elemTypes[iElem] = vtk_type;
517  }
518 
519  /*--- Force free the memory for the conn offset from the CGNS file. ---*/
520 
521  vector<cgsize_t>().swap(connOffsetCGNS);
522 
523  /*--- These are internal elems. Allocate memory on each proc. ---*/
524 
525  vector<cgsize_t> connElemTemp(nElems[val_section] * SU2_CONN_SIZE, 0);
526 
527  /*--- Copy the connectivity into the larger array with a standard
528  format per element: [globalID vtkType n0 n1 n2 n3 n4 n5 n6 n7 n8]. ---*/
529 
530  counterCGNS = 0;
531  for (iElem = 0; iElem < nElems[val_section]; iElem++) {
532  /*--- Store the conn in chunks of SU2_CONN_SIZE for simplicity. ---*/
533 
534  unsigned long nn = iElem * SU2_CONN_SIZE;
535 
536  /*--- First, store the global element ID and the VTK type. ---*/
537 
538  connElemTemp[nn] = elemGlobalID[iElem];
539  nn++;
540  connElemTemp[nn] = elemTypes[iElem];
541  nn++;
542 
543  /*--- Store the connectivity values. Note we subtract one from
544  the CGNS 1-based convention. We may also need to remove the first
545  entry if this is a mixed element section. ---*/
546 
547  if (isMixed) counterCGNS++;
548  for (iNode = 0; iNode < (unsigned long)nPoinPerElem[iElem]; iNode++) {
549  connElemTemp[nn] = connElemCGNS[counterCGNS + iNode] - 1;
550  nn++;
551  }
552  counterCGNS += nPoinPerElem[iElem];
553  }
554 
555  /*--- Force free the memory for the conn from the CGNS file. ---*/
556 
557  vector<cgsize_t>().swap(connElemCGNS);
558 
559  /*--- We now have the connectivity stored in linearly partitioned
560  chunks. We need to loop through and decide how many elements we
561  must send to each rank in order to have all elements that
562  surround a particular "owned" node on each rank (i.e., elements
563  will appear on multiple ranks). First, initialize a counter
564  and flag. ---*/
565 
566  int* nElem_Send = new int[size + 1];
567  nElem_Send[0] = 0;
568  int* nElem_Recv = new int[size + 1];
569  nElem_Recv[0] = 0;
570  int* nElem_Flag = new int[size];
571 
572  for (iProcessor = 0; iProcessor < size; iProcessor++) {
573  nElem_Send[iProcessor] = 0;
574  nElem_Recv[iProcessor] = 0;
575  nElem_Flag[iProcessor] = -1;
576  }
577  nElem_Send[size] = 0;
578  nElem_Recv[size] = 0;
579 
580  /*--- Create a partitioner object to find the owning rank of points. ---*/
581 
582  CLinearPartitioner pointPartitioner(numberOfGlobalPoints, 0);
583 
584  for (iElem = 0; iElem < nElems[val_section]; iElem++) {
585  for (iNode = 0; iNode < (unsigned long)nPoinPerElem[iElem]; iNode++) {
586  /*--- Get the index of the current point. ---*/
587 
588  iPoint = connElemTemp[iElem * SU2_CONN_SIZE + SU2_CONN_SKIP + iNode];
589 
590  /*--- Search for the processor that owns this point. ---*/
591 
592  iProcessor = pointPartitioner.GetRankContainingIndex(iPoint);
593 
594  /*--- If we have not visited this element yet, increment our
595  number of elements that must be sent to a particular proc. ---*/
596 
597  if ((nElem_Flag[iProcessor] != (int)iElem)) {
598  nElem_Flag[iProcessor] = iElem;
599  nElem_Send[iProcessor + 1]++;
600  }
601  }
602  }
603 
604  /*--- Communicate the number of cells to be sent/recv'd amongst
605  all processors. After this communication, each proc knows how
606  many cells it will receive from each other processor. ---*/
607 
608  SU2_MPI::Alltoall(&(nElem_Send[1]), 1, MPI_INT, &(nElem_Recv[1]), 1, MPI_INT, SU2_MPI::GetComm());
609 
610  /*--- Prepare to send connectivities. First check how many
611  messages we will be sending and receiving. Here we also put
612  the counters into cumulative storage format to make the
613  communications simpler. ---*/
614 
615  unsigned long nSends = 0, nRecvs = 0;
616  for (iProcessor = 0; iProcessor < size; iProcessor++) nElem_Flag[iProcessor] = -1;
617 
618  for (iProcessor = 0; iProcessor < size; iProcessor++) {
619  if ((iProcessor != rank) && (nElem_Send[iProcessor + 1] > 0)) nSends++;
620  if ((iProcessor != rank) && (nElem_Recv[iProcessor + 1] > 0)) nRecvs++;
621 
622  nElem_Send[iProcessor + 1] += nElem_Send[iProcessor];
623  nElem_Recv[iProcessor + 1] += nElem_Recv[iProcessor];
624  }
625 
626  /*--- Allocate memory to hold the connectivity that we are
627  sending. Note that we are also sending the global ID and the
628  VTK element type in the first and second positions, respectively.
629  We have assumed a constant message size of a hex element (8 nodes)
630  + 2 extra values for the ID and VTK. ---*/
631 
632  unsigned long *connSend = nullptr, iSend = 0;
633  unsigned long sendSize = (unsigned long)SU2_CONN_SIZE * nElem_Send[size];
634  connSend = new unsigned long[sendSize];
635  for (iSend = 0; iSend < sendSize; iSend++) connSend[iSend] = 0;
636 
637  /*--- Create an index variable to keep track of our index
638  position as we load up the send buffer. ---*/
639 
640  vector<unsigned long> index(size);
641  for (iProcessor = 0; iProcessor < size; iProcessor++) index[iProcessor] = SU2_CONN_SIZE * nElem_Send[iProcessor];
642 
643  /*--- Loop through our elements and load the elems and their
644  additional data that we will send to the other procs. ---*/
645 
646  for (iElem = 0; iElem < (unsigned long)nElems[val_section]; iElem++) {
647  for (iNode = 0; iNode < (unsigned long)nPoinPerElem[iElem]; iNode++) {
648  /*--- Get the index of the current point. ---*/
649 
650  iPoint = connElemTemp[iElem * SU2_CONN_SIZE + SU2_CONN_SKIP + iNode];
651 
652  /*--- Search for the processor that owns this point ---*/
653 
654  iProcessor = pointPartitioner.GetRankContainingIndex(iPoint);
655 
656  /*--- Load connectivity into the buffer for sending ---*/
657 
658  if (nElem_Flag[iProcessor] != (int)iElem) {
659  nElem_Flag[iProcessor] = iElem;
660  unsigned long nn = index[iProcessor];
661 
662  /*--- Load the VTK type first into the conn array,
663  then the connectivity vals, and last, the global ID. ---*/
664 
665  for (jNode = 0; jNode < SU2_CONN_SIZE; jNode++) {
666  connSend[nn] = connElemTemp[iElem * SU2_CONN_SIZE + jNode];
667  nn++;
668  }
669 
670  /*--- Increment the index by the message length ---*/
671 
672  index[iProcessor] += SU2_CONN_SIZE;
673  }
674  }
675  }
676 
677  /*--- Force free memory after loading up the send buffer. ---*/
678 
679  vector<cgsize_t>().swap(connElemTemp);
680  vector<cgsize_t>().swap(elemTypes);
681  vector<cgsize_t>().swap(nPoinPerElem);
682  vector<cgsize_t>().swap(elemGlobalID);
683 
684  vector<unsigned long>().swap(index);
685 
686  /*--- Allocate the memory that we need for receiving the conn
687  values and then cue up the non-blocking receives. Note that
688  we do not include our own rank in the communications. We will
689  directly copy our own data later. ---*/
690 
691  unsigned long *connRecv = nullptr, iRecv = 0;
692  unsigned long recvSize = (unsigned long)SU2_CONN_SIZE * nElem_Recv[size];
693  connRecv = new unsigned long[recvSize];
694  for (iRecv = 0; iRecv < recvSize; iRecv++) connRecv[iRecv] = 0;
695 
696  /*--- Allocate memory for the MPI requests if we will communicate. ---*/
697 
698  SU2_MPI::Request* connSendReq = nullptr;
699  SU2_MPI::Request* connRecvReq = nullptr;
700 
701  if (nSends > 0) {
702  connSendReq = new SU2_MPI::Request[nSends];
703  }
704  if (nRecvs > 0) {
705  connRecvReq = new SU2_MPI::Request[nRecvs];
706  }
707 
708  /*--- Launch the non-blocking sends and receives. ---*/
709 
710  InitiateCommsAll(connSend, nElem_Send, connSendReq, connRecv, nElem_Recv, connRecvReq, SU2_CONN_SIZE,
711  COMM_TYPE_UNSIGNED_LONG);
712 
713  /*--- Copy the current rank's data into the recv buffer directly. ---*/
714 
715  iRecv = SU2_CONN_SIZE * nElem_Recv[rank];
716  unsigned long myStart = SU2_CONN_SIZE * nElem_Send[rank];
717  unsigned long myFinal = SU2_CONN_SIZE * nElem_Send[rank + 1];
718  for (iSend = myStart; iSend < myFinal; iSend++) {
719  connRecv[iRecv] = connSend[iSend];
720  iRecv++;
721  }
722 
723  /*--- Complete the non-blocking communications. ---*/
724 
725  CompleteCommsAll(nSends, connSendReq, nRecvs, connRecvReq);
726 
727  /*--- Store the connectivity for this rank in the proper data
728  structure. First, allocate the appropriate amount of memory
729  for this section, then write the recv'd values. ---*/
730 
731  if (nElem_Recv[size] > 0) {
732  connElems[val_section].resize(nElem_Recv[size] * SU2_CONN_SIZE, 0);
733  unsigned long count = 0;
734  for (iElem = 0; iElem < (unsigned long)nElem_Recv[size]; iElem++) {
735  for (iNode = 0; iNode < SU2_CONN_SIZE; iNode++) {
736  unsigned long nn = iElem * SU2_CONN_SIZE + iNode;
737  connElems[val_section][count] = (cgsize_t)connRecv[nn];
738  count++;
739  }
740  }
741 
742  /*--- Store the total number of elements the current rank
743  now has for the current section after completing the comms. ---*/
744 
745  nElems[val_section] = nElem_Recv[size];
746 
747  } else {
748  /*--- The current rank did not recv any elements from this
749  section. Set the count to zero and nullify the data structure. ---*/
750 
751  nElems[val_section] = 0;
752  connElems[val_section].resize(0);
753  }
754 
755  /*--- Free temporary memory from communications ---*/
756 
757  delete[] connSendReq;
758  delete[] connRecvReq;
759 
760  delete[] connSend;
761  delete[] connRecv;
762  delete[] nElem_Recv;
763  delete[] nElem_Send;
764  delete[] nElem_Flag;
765 }

767 void CCGNSMeshReaderFVM::ReadCGNSSurfaceSection(int val_section) {
768  /*--- In this routine, we access a CGNS surface section and have the
769  master rank load all of the surface conn. This can help avoid issues
770  where there are fewer elements than ranks on a surface. This is later
771  linearly partitioned. A limitation of this approach is that there
772  could be a memory bottleneck for extremely large grids, but this has
773  not been reached yet in practice. In that event, we can also read these
774  surface sections with a linear partitioning. ---*/
775 
776  int nbndry, parent_flag, npe;
777  unsigned long iElem = 0, iNode = 0;
778  cgsize_t startE, endE, ElementDataSize;
779  ElementType_t elemType;
780  char sectionName[CGNS_STRING_SIZE];
781 
782  if (rank == MASTER_NODE) {
783  /*--- Allocate some memory for the handling the connectivity
784  and auxiliary data that we are need to communicate. ---*/
785 
786  vector<cgsize_t> connElemCGNS(nElems[val_section] * SU2_CONN_SIZE, 0);
787  vector<unsigned short> elemTypes(nElems[val_section], 0);
788  vector<unsigned short> nPoinPerElem(nElems[val_section], 0);
789  vector<unsigned long> elemGlobalID(nElems[val_section], 0);
790 
791  /*--- Read the section info again ---*/
792 
793  if (cg_section_read(cgnsFileID, cgnsBase, cgnsZone, val_section + 1, sectionName, &elemType, &startE, &endE,
794  &nbndry, &parent_flag))
795  cg_error_exit();
796 
797  /*--- Print some information to the console. ---*/
798 
799  cout << "Loading surface section " << string(sectionName);
800  cout << " from file." << endl;
801 
802  /*--- Store the number of elems (all on the master). ---*/
803 
804  nElems[val_section] = (endE - startE + 1);
805 
806  /*--- Read and store the total amount of data that will be
807  listed when reading this section. ---*/
808 
809  if (cg_ElementDataSize(cgnsFileID, cgnsBase, cgnsZone, val_section + 1, &ElementDataSize)) cg_error_exit();
810 
811  /*--- Find the number of nodes required to represent
812  this type of element. ---*/
813 
814  if (cg_npe(elemType, &npe)) cg_error_exit();
815 
816  /*--- Check whether the sections contains a mixture of multiple
817  element types, which will require special handling to get the
818  element type one-by-one when reading. ---*/
819 
820  bool isMixed = false;
821  if (elemType == MIXED) {
822  isMixed = true;
823  }
824 
825  /*--- Allocate memory for accessing the connectivity and to
826  store it in the proper data structure for post-processing. ---*/
827 
828  vector<cgsize_t> connElemTemp(ElementDataSize, 0);
829 
830  /*--- Retrieve the connectivity information and store. ---*/
831 
832  if (elemType == MIXED || elemType == NGON_n || elemType == NFACE_n) {
833  vector<cgsize_t> connOffsetTemp(nElems[val_section] + 1, 0);
834  if (cg_poly_elements_partial_read(cgnsFileID, cgnsBase, cgnsZone, val_section + 1, startE, endE,
835  connElemTemp.data(), connOffsetTemp.data(), nullptr) != CG_OK)
836  cg_error_exit();
837  } else {
838  if (cg_elements_read(cgnsFileID, cgnsBase, cgnsZone, val_section + 1, connElemTemp.data(), nullptr))
839  cg_error_exit();
840  }
841 
842  /*--- Allocate the memory for the data structure used to carry
843  the connectivity for this section. ---*/
844 
845  connElems[val_section].resize(nElems[val_section] * SU2_CONN_SIZE, 0);
846 
847  unsigned long counterCGNS = 0;
848  for (iElem = 0; iElem < nElems[val_section]; iElem++) {
849  ElementType_t iElemType = elemType;
850 
851  /*--- If we have a mixed element section, we need to check the elem
852  type one-by-one. We also must manually advance the counter. ---*/
853 
854  if (isMixed) {
855  iElemType = ElementType_t(connElemTemp[counterCGNS]);
856  counterCGNS++;
857  }
858 
859  /*--- Get the VTK type for this element. ---*/
860 
861  int vtk_type;
862  string elem_name = GetCGNSElementType(iElemType, vtk_type);
863 
864  /*--- Get the number of nodes per element. ---*/
865 
866  cg_npe(iElemType, &npe);
867 
868  /*--- Load the surface element connectivity into the SU2 data
869  structure with format: [globalID VTK n1 n2 n3 n4 n5 n6 n7 n8].
870  We do not need a global ID for the surface elements, so we
871  simply set that to zero to maintain the same data structure
872  format as the interior elements. Note that we subtract 1 to
873  move from the CGNS 1-based indexing to SU2's zero-based. ---*/
874 
875  connElems[val_section][iElem * SU2_CONN_SIZE + 0] = 0;
876  connElems[val_section][iElem * SU2_CONN_SIZE + 1] = vtk_type;
877  for (iNode = 0; iNode < (unsigned long)npe; iNode++) {
878  unsigned long nn = iElem * SU2_CONN_SIZE + SU2_CONN_SKIP + iNode;
879  connElems[val_section][nn] = connElemTemp[counterCGNS] - 1;
880  counterCGNS++;
881  }
882  }
883 
884  } else {
885  /*--- We are not the master, so we resize to zero for safety. ---*/
886 
887  nElems[val_section] = 0;
888  connElems[val_section].resize(0);
889  }
890 }

1015 void CCGNSMeshReaderFVM::InitiateCommsAll(void* bufSend, const int* nElemSend, SU2_MPI::Request* sendReq, void* bufRecv,
1016  const int* nElemRecv, SU2_MPI::Request* recvReq, unsigned short countPerElem,
1017  unsigned short commType) {
1018  /*--- Local variables ---*/
1019 
1020  int iMessage, iProc, offset, nElem, count, source, dest, tag;
1021 
1022  /*--- Launch the non-blocking recv's first. ---*/
1023 
1024  iMessage = 0;
1025  for (iProc = 0; iProc < size; iProc++) {
1026  /*--- Post recv's only if another proc is sending us data. We do
1027  not communicate with ourselves or post recv's for zero length
1028  messages to keep overhead down. ---*/
1029 
1030  if ((nElemRecv[iProc + 1] > nElemRecv[iProc]) && (iProc != rank)) {
1031  /*--- Compute our location in the recv buffer. ---*/
1032 
1033  offset = countPerElem * nElemRecv[iProc];
1034 
1035  /*--- Take advantage of cumulative storage format to get the number
1036  of elems that we need to recv. ---*/
1037 
1038  nElem = nElemRecv[iProc + 1] - nElemRecv[iProc];
1039 
1040  /*--- Total count can include multiple pieces of data per element. ---*/
1041 
1042  count = countPerElem * nElem;
1043 
1044  /*--- Post non-blocking recv for this proc. ---*/
1045 
1046  source = iProc;
1047  tag = iProc + 1;
1048 
1049  switch (commType) {
1050  case COMM_TYPE_DOUBLE:
1051  SU2_MPI::Irecv(&(static_cast<su2double*>(bufRecv)[offset]), count, MPI_DOUBLE, source, tag,
1052  SU2_MPI::GetComm(), &(recvReq[iMessage]));
1053  break;
1054  case COMM_TYPE_UNSIGNED_LONG:
1055  SU2_MPI::Irecv(&(static_cast<unsigned long*>(bufRecv)[offset]), count, MPI_UNSIGNED_LONG, source, tag,
1056  SU2_MPI::GetComm(), &(recvReq[iMessage]));
1057  break;
1058  case COMM_TYPE_LONG:
1059  SU2_MPI::Irecv(&(static_cast<long*>(bufRecv)[offset]), count, MPI_LONG, source, tag, SU2_MPI::GetComm(),
1060  &(recvReq[iMessage]));
1061  break;
1062  case COMM_TYPE_UNSIGNED_SHORT:
1063  SU2_MPI::Irecv(&(static_cast<unsigned short*>(bufRecv)[offset]), count, MPI_UNSIGNED_SHORT, source, tag,
1064  SU2_MPI::GetComm(), &(recvReq[iMessage]));
1065  break;
1066  case COMM_TYPE_CHAR:
1067  SU2_MPI::Irecv(&(static_cast<char*>(bufRecv)[offset]), count, MPI_CHAR, source, tag, SU2_MPI::GetComm(),
1068  &(recvReq[iMessage]));
1069  break;
1070  case COMM_TYPE_SHORT:
1071  SU2_MPI::Irecv(&(static_cast<short*>(bufRecv)[offset]), count, MPI_SHORT, source, tag, SU2_MPI::GetComm(),
1072  &(recvReq[iMessage]));
1073  break;
1074  case COMM_TYPE_INT:
1075  SU2_MPI::Irecv(&(static_cast<int*>(bufRecv)[offset]), count, MPI_INT, source, tag, SU2_MPI::GetComm(),
1076  &(recvReq[iMessage]));
1077  break;
1078  default:
1079  break;
1080  }
1081 
1082  /*--- Increment message counter. ---*/
1083 
1084  iMessage++;
1085  }
1086  }
1087 
1088  /*--- Launch the non-blocking sends next. ---*/
1089 
1090  iMessage = 0;
1091  for (iProc = 0; iProc < size; iProc++) {
1092  /*--- Post sends only if we are sending another proc data. We do
1093  not communicate with ourselves or post sends for zero length
1094  messages to keep overhead down. ---*/
1095 
1096  if ((nElemSend[iProc + 1] > nElemSend[iProc]) && (iProc != rank)) {
1097  /*--- Compute our location in the send buffer. ---*/
1098 
1099  offset = countPerElem * nElemSend[iProc];
1100 
1101  /*--- Take advantage of cumulative storage format to get the number
1102  of elems that we need to send. ---*/
1103 
1104  nElem = nElemSend[iProc + 1] - nElemSend[iProc];
1105 
1106  /*--- Total count can include multiple pieces of data per element. ---*/
1107 
1108  count = countPerElem * nElem;
1109 
1110  /*--- Post non-blocking send for this proc. ---*/
1111 
1112  dest = iProc;
1113  tag = rank + 1;
1114 
1115  switch (commType) {
1116  case COMM_TYPE_DOUBLE:
1117  SU2_MPI::Isend(&(static_cast<su2double*>(bufSend)[offset]), count, MPI_DOUBLE, dest, tag, SU2_MPI::GetComm(),
1118  &(sendReq[iMessage]));
1119  break;
1120  case COMM_TYPE_UNSIGNED_LONG:
1121  SU2_MPI::Isend(&(static_cast<unsigned long*>(bufSend)[offset]), count, MPI_UNSIGNED_LONG, dest, tag,
1122  SU2_MPI::GetComm(), &(sendReq[iMessage]));
1123  break;
1124  case COMM_TYPE_LONG:
1125  SU2_MPI::Isend(&(static_cast<long*>(bufSend)[offset]), count, MPI_LONG, dest, tag, SU2_MPI::GetComm(),
1126  &(sendReq[iMessage]));
1127  break;
1128  case COMM_TYPE_UNSIGNED_SHORT:
1129  SU2_MPI::Isend(&(static_cast<unsigned short*>(bufSend)[offset]), count, MPI_UNSIGNED_SHORT, dest, tag,
1130  SU2_MPI::GetComm(), &(sendReq[iMessage]));
1131  break;
1132  case COMM_TYPE_CHAR:
1133  SU2_MPI::Isend(&(static_cast<char*>(bufSend)[offset]), count, MPI_CHAR, dest, tag, SU2_MPI::GetComm(),
1134  &(sendReq[iMessage]));
1135  break;
1136  case COMM_TYPE_SHORT:
1137  SU2_MPI::Isend(&(static_cast<short*>(bufSend)[offset]), count, MPI_SHORT, dest, tag, SU2_MPI::GetComm(),
1138  &(sendReq[iMessage]));
1139  break;
1140  case COMM_TYPE_INT:
1141  SU2_MPI::Isend(&(static_cast<int*>(bufSend)[offset]), count, MPI_INT, dest, tag, SU2_MPI::GetComm(),
1142  &(sendReq[iMessage]));
1143  break;
1144  default:
1145  break;
1146  }
1147 
1148  /*--- Increment message counter. ---*/
1149 
1150  iMessage++;
1151  }
1152  }

1155 void CCGNSMeshReaderFVM::CompleteCommsAll(int nSends, SU2_MPI::Request* sendReq, int nRecvs,
1156  SU2_MPI::Request* recvReq) {
1157  /*--- Local variables ---*/
1158 
1159  int ind, iSend, iRecv;
1160  SU2_MPI::Status status;
1161 
1162  /*--- Wait for the non-blocking sends to complete. ---*/
1163
