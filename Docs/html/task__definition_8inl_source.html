40inline CTaskDefinition::CTaskDefinition(SOLVER_TASK val_task,

42 int val_ind0MustBeCompleted,
43 int val_ind1MustBeCompleted,
44 int val_ind2MustBeCompleted,
45 int val_ind3MustBeCompleted,
46 int val_ind4MustBeCompleted) {
47
48 /* Copy the data from the arguments. */
49 task = val_task;
50 timeLevel = val_timeLevel;
51 indMustBeCompleted[0] = val_ind0MustBeCompleted;
52 indMustBeCompleted[1] = val_ind1MustBeCompleted;
53 indMustBeCompleted[2] = val_ind2MustBeCompleted;
54 indMustBeCompleted[3] = val_ind3MustBeCompleted;
55 indMustBeCompleted[4] = val_ind4MustBeCompleted;
56
57 /* Make sure that the -1 values are numbered last. */
58 sort(indMustBeCompleted, indMustBeCompleted+5, greater<int>());
59
60 /* Determine the actual number of tasks that must be completed. */
61 for(nIndMustBeCompleted=0; nIndMustBeCompleted<5; ++nIndMustBeCompleted) {
62 if(indMustBeCompleted[nIndMustBeCompleted] < 0) break;
63 }
64
65 /* Initialize the other member variables to their default values. */
66 intPointADER = 0;
67 secondPartTimeIntADER = false;
68}
