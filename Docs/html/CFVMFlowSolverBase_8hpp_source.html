281 /*!
282 * \brief Sums edge fluxes (if required) and computes the global error counter.
283 * \param[in] pausePreacc - Whether preaccumulation was paused durin.
284 * \param[in] localCounter - Thread-local error counter.
285 * \param[in,out] config - Used to set the global error counter.
286 */
287 inline void FinalizeResidualComputation(const CGeometry *geometry, bool pausePreacc,
288 unsigned long localCounter, CConfig* config) {
289
290 /*--- Restore preaccumulation and adjoint evaluation state. ---*/
291 AD::ResumePreaccumulation(pausePreacc);
292 if (!ReducerStrategy) AD::EndNoSharedReading();
293
294 if (ReducerStrategy) {
295 SumEdgeFluxes(geometry);
296 if (config->GetKind_TimeIntScheme() == EULER_IMPLICIT) {
297 Jacobian.SetDiagonalAsColumnSum();
298 }
299 }
300
301 /*--- Warning message about non-physical reconstructions. ---*/
302 if ((MGLevel == MESH_0) && (config->GetComm_Level() == COMM_FULL)) {
303 /*--- Add counter results for all threads. ---*/
304 SU2_OMP_ATOMIC
305 ErrorCounter += localCounter;
306
307 /*--- Add counter results for all ranks. ---*/
308 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
309 {
310 localCounter = ErrorCounter;
311 SU2_MPI::Reduce(&localCounter, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE, SU2_MPI::GetComm());
312 config->SetNonphysical_Reconstr(ErrorCounter);
313 }
314 END_SU2_OMP_SAFE_GLOBAL_ACCESS
315 }
316 }

364 /*!
365 * \brief Generic implementation to compute the time step based on CFL and conv/visc eigenvalues.
366 * \param[in] geometry - Geometrical definition of the problem.
367 * \param[in] solver_container - Container vector with all the solutions.
368 * \param[in] config - Definition of the particular problem.
369 * \param[in] iMesh - Index of the mesh in multigrid computations.
370 * \param[in] Iteration - Value of the current iteration.
371 * \tparam SoundSpeedFunc - Function object to compute speed of sound.
372 * \tparam LambdaViscFunc - Function object to compute the viscous lambda.
373 * \note Both functors need to implement (nodes,iPoint,jPoint) for edges, and (nodes,iPoint) for vertices.
374 */
375 template<class SoundSpeedFunc, class LambdaViscFunc>
376 FORCEINLINE void SetTime_Step_impl(const SoundSpeedFunc& soundSpeed,
377 const LambdaViscFunc& lambdaVisc,
378 CGeometry *geometry,
379 CSolver **solver_container,
380 CConfig *config,
381 unsigned short iMesh,
382 unsigned long Iteration) {
383
384 const bool viscous = config->GetViscous();
385 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
386 const bool time_stepping = (config->GetTime_Marching() == TIME_MARCHING::TIME_STEPPING);
387 const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
388 (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
389 const su2double K_v = 0.25;
390
391 /*--- Init thread-shared variables to compute min/max values.
392 * Critical sections are used for this instead of reduction
393 * clauses for compatibility with OpenMP 2.0 (Windows...). ---*/
394
395 ompMasterAssignBarrier(Min_Delta_Time,1e30, Max_Delta_Time,0.0, Global_Delta_UnstTimeND,1e30);
396
397 /*--- Loop domain points. ---*/
398
399 SU2_OMP_FOR_DYN(omp_chunk_size)
400 for (auto iPoint = 0ul; iPoint < nPointDomain; ++iPoint) {
401
402 /*--- Set maximum eigenvalues to zero. ---*/
403
404 nodes->SetMax_Lambda_Inv(iPoint,0.0);
405
406 if (viscous)
407 nodes->SetMax_Lambda_Visc(iPoint,0.0);
408
409 /*--- Loop over the neighbors of point i. ---*/
410
411 for (unsigned short iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); ++iNeigh)
412 {
413 auto jPoint = geometry->nodes->GetPoint(iPoint,iNeigh);
414
415 auto iEdge = geometry->nodes->GetEdge(iPoint,iNeigh);
416 auto Normal = geometry->edges->GetNormal(iEdge);
417 auto Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
418
419 /*--- Mean Values ---*/
420
421 su2double Mean_ProjVel = 0.5 * (nodes->GetProjVel(iPoint,Normal) + nodes->GetProjVel(jPoint,Normal));
422 su2double Mean_SoundSpeed = soundSpeed(*nodes, iPoint, jPoint) * sqrt(Area2);
423
424 /*--- Adjustment for grid movement ---*/
425
426 if (dynamic_grid) {
427 const su2double *GridVel_i = geometry->nodes->GetGridVel(iPoint);
428 const su2double *GridVel_j = geometry->nodes->GetGridVel(jPoint);
429
430 for (unsigned short iDim = 0; iDim < nDim; iDim++)
431 Mean_ProjVel -= 0.5 * (GridVel_i[iDim] + GridVel_j[iDim]) * Normal[iDim];
432 }
433
434 /*--- Inviscid contribution ---*/
435
436 su2double Lambda = fabs(Mean_ProjVel) + Mean_SoundSpeed;
437 nodes->AddMax_Lambda_Inv(iPoint,Lambda);
438
439 /*--- Viscous contribution ---*/
440
441 if (!viscous) continue;
442
443 Lambda = lambdaVisc(*nodes, iPoint, jPoint) * Area2;
444 nodes->AddMax_Lambda_Visc(iPoint, Lambda);
445 }
446
447 }
448 END_SU2_OMP_FOR
449
450 /*--- Loop boundary edges ---*/
451
452 for (unsigned short iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
453 if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
454 (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
455 (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
456
457 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
458 for (auto iVertex = 0ul; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
459
460 /*--- Point identification, Normal vector and area ---*/
461
462 auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
463
464 if (!geometry->nodes->GetDomain(iPoint)) continue;
465
466 auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
467 auto Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
468
469 /*--- Mean Values ---*/
470
471 su2double ProjVel = nodes->GetProjVel(iPoint,Normal);
472 su2double SoundSpeed = soundSpeed(*nodes, iPoint) * sqrt(Area2);
473
474 /*--- Adjustment for grid movement ---*/
475
476 if (dynamic_grid) {
477 ProjVel -= GeometryToolbox::DotProduct(nDim, Normal, geometry->nodes->GetGridVel(iPoint));
478 }
479
480 /*--- Inviscid contribution ---*/
481
482 su2double Lambda = fabs(ProjVel) + SoundSpeed;
483 nodes->AddMax_Lambda_Inv(iPoint, Lambda);
484
485 /*--- Viscous contribution ---*/
486
487 if (!viscous) continue;
488
489 Lambda = lambdaVisc(*nodes,iPoint) * Area2;
490 nodes->AddMax_Lambda_Visc(iPoint, Lambda);
491 }
492 END_SU2_OMP_FOR
493 }
494 }
495
496 /*--- Each element uses their own speed, steady state simulation. ---*/
497 {
498 /*--- Thread-local variables for min/max reduction. ---*/
499 su2double minDt = 1e30, maxDt = 0.0;
500
501 SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
502 for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
503
504 su2double Vol = geometry->nodes->GetVolume(iPoint);
505
506 if (Vol != 0.0) {
507 su2double Local_Delta_Time = nodes->GetLocalCFL(iPoint)*Vol / nodes->GetMax_Lambda_Inv(iPoint);
508
509 if(viscous) {
510 su2double dt_visc = nodes->GetLocalCFL(iPoint)*K_v*Vol*Vol / nodes->GetMax_Lambda_Visc(iPoint);
511 Local_Delta_Time = min(Local_Delta_Time, dt_visc);
512 }
513
514 minDt = min(minDt, Local_Delta_Time);
515 maxDt = max(maxDt, Local_Delta_Time);
516
517 nodes->SetDelta_Time(iPoint, min(Local_Delta_Time, config->GetMax_DeltaTime()));
518 }
519 else {
520 nodes->SetDelta_Time(iPoint,0.0);
521 }
522 }
523 END_SU2_OMP_FOR
524 /*--- Min/max over threads. ---*/
525 SU2_OMP_CRITICAL
526 {
527 Min_Delta_Time = min(Min_Delta_Time, minDt);
528 Max_Delta_Time = max(Max_Delta_Time, maxDt);
529 Global_Delta_Time = Min_Delta_Time;
530 }
531 END_SU2_OMP_CRITICAL
532 }
533
534 /*--- Compute the min/max dt (in parallel, now over mpi ranks). ---*/
535
536 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
537 if (config->GetComm_Level() == COMM_FULL) {
538 su2double rbuf_time;
539 SU2_MPI::Allreduce(&Min_Delta_Time, &rbuf_time, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
540 Min_Delta_Time = rbuf_time;
541
542 SU2_MPI::Allreduce(&Max_Delta_Time, &rbuf_time, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
543 Max_Delta_Time = rbuf_time;
544 }
545 } END_SU2_OMP_SAFE_GLOBAL_ACCESS
546
547 /*--- For exact time solution use the minimum delta time of the whole mesh. ---*/
548 if (time_stepping) {
549
550 /*--- If the unsteady CFL is set to zero, it uses the defined unsteady time step,
551 * otherwise it computes the time step based on the unsteady CFL. ---*/
552
553 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
554 {
555 if (config->GetUnst_CFL() == 0.0) {
556 Global_Delta_Time = config->GetDelta_UnstTime();
557 }
558 else {
559 Global_Delta_Time = Min_Delta_Time;
560 }
561 Max_Delta_Time = Global_Delta_Time;
562
563 config->SetDelta_UnstTimeND(Global_Delta_Time);
564 }
565 END_SU2_OMP_SAFE_GLOBAL_ACCESS
566
567 /*--- Sets the regular CFL equal to the unsteady CFL. ---*/
568
569 SU2_OMP_FOR_STAT(omp_chunk_size)
570 for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
571 nodes->SetLocalCFL(iPoint, config->GetUnst_CFL());
572 nodes->SetDelta_Time(iPoint, Global_Delta_Time);
573 }
574 END_SU2_OMP_FOR
575
576 }
577
578 /*--- Recompute the unsteady time step for the dual time strategy if the unsteady CFL is diferent from 0.
579 * This is only done once because in dual time the time step cannot be variable. ---*/
580
581 if (dual_time && (Iteration == config->GetRestart_Iter()) && (config->GetUnst_CFL() != 0.0) && (iMesh == MESH_0)) {
582
583 /*--- Thread-local variable for reduction. ---*/
584 su2double glbDtND = 1e30;
585
586 SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
587 for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
588 glbDtND = min(glbDtND, config->GetUnst_CFL()*Global_Delta_Time / nodes->GetLocalCFL(iPoint));
589 }
590 END_SU2_OMP_FOR
591 SU2_OMP_CRITICAL
592 Global_Delta_UnstTimeND = min(Global_Delta_UnstTimeND, glbDtND);
593 END_SU2_OMP_CRITICAL
594
595 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
596 {
597 SU2_MPI::Allreduce(&Global_Delta_UnstTimeND, &glbDtND, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
598 Global_Delta_UnstTimeND = glbDtND;
599
600 config->SetDelta_UnstTimeND(Global_Delta_UnstTimeND);
601 }
602 END_SU2_OMP_SAFE_GLOBAL_ACCESS
603 }
604
605 /*--- The pseudo local time (explicit integration) cannot be greater than the physical time ---*/
606
607 if (dual_time && !implicit) {
608 SU2_OMP_FOR_STAT(omp_chunk_size)
609 for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
610 su2double dt = min((2.0/3.0)*config->GetDelta_UnstTimeND(), nodes->GetDelta_Time(iPoint));
611 nodes->SetDelta_Time(iPoint, dt);
612 }
613 END_SU2_OMP_FOR
614 }

617 /*!
618 * \brief Compute the max eigenvalue, gemeric implementation.
619 * \param[in] geometry - Geometrical definition of the problem.
620 * \param[in] config - Definition of the particular problem.
621 * \tparam SoundSpeedFunc - Function object to compute speed of sound.
622 * \note Functor needs to implement (nodes,iPoint,jPoint) for edges, and (nodes,iPoint) for vertices.
623 */
624 template<class SoundSpeedFunc>
625 FORCEINLINE void SetMax_Eigenvalue_impl(const SoundSpeedFunc& soundSpeed, CGeometry *geometry, const CConfig *config) {
626
627 /*--- Loop domain points. ---*/
628
629 SU2_OMP_FOR_DYN(omp_chunk_size)
630 for (unsigned long iPoint = 0; iPoint < nPointDomain; ++iPoint) {
631
632 /*--- Set eigenvalues to zero. ---*/
633 nodes->SetLambda(iPoint,0.0);
634
635 /*--- Loop over the neighbors of point i. ---*/
636 for (unsigned short iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); ++iNeigh)
637 {
638 auto jPoint = geometry->nodes->GetPoint(iPoint, iNeigh);
639
640 auto iEdge = geometry->nodes->GetEdge(iPoint, iNeigh);
641 auto Normal = geometry->edges->GetNormal(iEdge);
642 su2double Area = GeometryToolbox::Norm(nDim, Normal);
643
644 /*--- Mean Values ---*/
645
646 su2double Mean_ProjVel = 0.5 * (nodes->GetProjVel(iPoint,Normal) + nodes->GetProjVel(jPoint,Normal));
647 su2double Mean_SoundSpeed = soundSpeed(*nodes, iPoint, jPoint) * Area;
648
649 /*--- Adjustment for grid movement ---*/
650
651 if (dynamic_grid) {
652 const su2double *GridVel_i = geometry->nodes->GetGridVel(iPoint);
653 const su2double *GridVel_j = geometry->nodes->GetGridVel(jPoint);
654
655 for (unsigned short iDim = 0; iDim < nDim; iDim++)
656 Mean_ProjVel -= 0.5 * (GridVel_i[iDim] + GridVel_j[iDim]) * Normal[iDim];
657 }
658
659 /*--- Inviscid contribution ---*/
660
661 nodes->AddLambda(iPoint, fabs(Mean_ProjVel) + Mean_SoundSpeed);
662 }
663 }
664 END_SU2_OMP_FOR
665
666 /*--- Loop boundary edges ---*/
667
668 for (unsigned short iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
669 if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
670 (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
671 (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
672
673 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
674 for (unsigned long iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
675
676 /*--- Point identification, Normal vector and area ---*/
677
678 auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
679
680 if (!geometry->nodes->GetDomain(iPoint)) continue;
681
682 auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
683 su2double Area = GeometryToolbox::Norm(nDim, Normal);
684
685 /*--- Mean Values ---*/
686
687 su2double Mean_ProjVel = nodes->GetProjVel(iPoint,Normal);
688 su2double Mean_SoundSpeed = soundSpeed(*nodes, iPoint) * Area;
689
690 /*--- Adjustment for grid movement ---*/
691
692 if (dynamic_grid) {
693 Mean_ProjVel -= GeometryToolbox::DotProduct(nDim, Normal, geometry->nodes->GetGridVel(iPoint));
694 }
695
696 /*--- Inviscid contribution ---*/
697
698 nodes->AddLambda(iPoint, fabs(Mean_ProjVel) + Mean_SoundSpeed);
699 }
700 END_SU2_OMP_FOR
701 }
702 }
703
704 /*--- Correct the eigenvalue values across any periodic boundaries. ---*/
705
706 for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic()/2; iPeriodic++) {
707 InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_MAX_EIG);
708 CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_MAX_EIG);
709 }
710
711 /*--- MPI parallelization ---*/
712
713 InitiateComms(geometry, config, MAX_EIGENVALUE);
714 CompleteComms(geometry, config, MAX_EIGENVALUE);
715 }

717 /*!
718 * \brief Compute the dissipation sensor for centered schemes.
719 * \param[in] geometry - Geometrical definition of the problem.
720 * \param[in] config - Definition of the particular problem.
721 * \tparam SensVarFunc - Function object implementing (nodes, iPoint) to return the sensor variable, e.g. pressure.
722 */
723 template<class SensVarFunc>
724 FORCEINLINE void SetCentered_Dissipation_Sensor_impl(const SensVarFunc& sensVar,
725 CGeometry *geometry, const CConfig *config) {
726
727 /*--- We can access memory more efficiently if there are no periodic boundaries. ---*/
728
729 const bool isPeriodic = (config->GetnMarker_Periodic() > 0);
730
731 /*--- Loop domain points. ---*/
732
733 SU2_OMP_FOR_DYN(omp_chunk_size)
734 for (unsigned long iPoint = 0; iPoint < nPointDomain; ++iPoint) {
735
736 const bool boundary_i = geometry->nodes->GetPhysicalBoundary(iPoint);
737 const su2double sensVar_i = sensVar(*nodes, iPoint);
738
739 /*--- Initialize. ---*/
740 iPoint_UndLapl[iPoint] = 0.0;
741 jPoint_UndLapl[iPoint] = 0.0;
742
743 /*--- Loop over the neighbors of point i. ---*/
744 for (auto jPoint : geometry->nodes->GetPoints(iPoint))
745 {
746 bool boundary_j = geometry->nodes->GetPhysicalBoundary(jPoint);
747
748 /*--- If iPoint is boundary it only takes contributions from other boundary points. ---*/
749 if (boundary_i && !boundary_j) continue;
750
751 su2double sensVar_j = sensVar(*nodes, jPoint);
752
753 /*--- Dissipation sensor, add variable difference and variable sum. ---*/
754 iPoint_UndLapl[iPoint] += sensVar_j - sensVar_i;
755 jPoint_UndLapl[iPoint] += sensVar_j + sensVar_i;
756 }
757
758 if (!isPeriodic) {
759 /*--- Every neighbor is accounted for, sensor can be computed. ---*/
760 nodes->SetSensor(iPoint, fabs(iPoint_UndLapl[iPoint]) / jPoint_UndLapl[iPoint]);
761 }
762 }
763 END_SU2_OMP_FOR
764
765 if (isPeriodic) {
766 /*--- Correct the sensor values across any periodic boundaries. ---*/
767
768 for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic()/2; iPeriodic++) {
769 InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_SENSOR);
770 CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_SENSOR);
771 }
772
773 /*--- Set final pressure switch for each point ---*/
774
775 SU2_OMP_FOR_STAT(omp_chunk_size)
776 for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++)
777 nodes->SetSensor(iPoint, fabs(iPoint_UndLapl[iPoint]) / jPoint_UndLapl[iPoint]);
778 END_SU2_OMP_FOR
779 }
780
781 /*--- MPI parallelization ---*/
782
783 InitiateComms(geometry, config, SENSOR);
784 CompleteComms(geometry, config, SENSOR);
785
786 }

788 /*!
789 * \brief Generic implementation of explicit iterations with a preconditioner.
790 * \note The preconditioner is a functor implementing the methods:
791 * - compute(config, iPoint): Should prepare the preconditioner for iPoint.
792 * - apply(iVar, residual[], resTruncError[]): Apply it to compute the iVar update.
793 * See Explicit_Iteration for the general form of the preconditioner.
794 */
795 template<ENUM_TIME_INT IntegrationType, class ResidualPrecond>
796 void Explicit_Iteration_impl(ResidualPrecond& preconditioner, CGeometry *geometry,
797 CSolver **solver_container, CConfig *config, unsigned short iRKStep) {
798
799 static_assert(IntegrationType == CLASSICAL_RK4_EXPLICIT ||
800 IntegrationType == RUNGE_KUTTA_EXPLICIT ||
801 IntegrationType == EULER_EXPLICIT, "");
802
803 const bool adjoint = config->GetContinuous_Adjoint();
804
805 const su2double RK_AlphaCoeff = config->Get_Alpha_RKStep(iRKStep);
806
807 /*--- Hard-coded classical RK4 coefficients. Will be added to config. ---*/
808 const su2double RK_FuncCoeff[] = {1.0/6.0, 1.0/3.0, 1.0/3.0, 1.0/6.0};
809 const su2double RK_TimeCoeff[] = {0.5, 0.5, 1.0, 1.0};
810
811 /*--- Local residual variables for current thread ---*/
812 su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
813 unsigned long idxMax[MAXNVAR] = {0};
814
815 /*--- Update the solution and residuals ---*/
816
817 if (!adjoint) {
818 SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
819 for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
820
821 su2double Vol = geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint);
822 su2double Delta = nodes->GetDelta_Time(iPoint) / Vol;
823
824 const su2double* Res_TruncError = nodes->GetResTruncError(iPoint);
825 const su2double* Residual = LinSysRes.GetBlock(iPoint);
826
827 preconditioner.compute(config, iPoint);
828
829 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
830
831 su2double Res = preconditioner.apply(iVar, Residual, Res_TruncError);
832
833 /*--- "Static" switch which should be optimized at compile time. ---*/
834 switch(IntegrationType) {
835
836 case EULER_EXPLICIT:
837 nodes->AddSolution(iPoint,iVar, -Res*Delta);
838 break;
839
840 case RUNGE_KUTTA_EXPLICIT:
841 nodes->AddSolution(iPoint, iVar, -Res*Delta*RK_AlphaCoeff);
842 break;
843
844 case CLASSICAL_RK4_EXPLICIT:
845 {
846 su2double tmp_time = -1.0*RK_TimeCoeff[iRKStep]*Delta;
847 su2double tmp_func = -1.0*RK_FuncCoeff[iRKStep]*Delta;
848
849 if (iRKStep < 3) {
850 /* Base Solution Update */
851 nodes->AddSolution(iPoint,iVar, tmp_time*Res);
852
853 /* New Solution Update */
854 nodes->AddSolution_New(iPoint,iVar, tmp_func*Res);
855 } else {
856 nodes->SetSolution(iPoint, iVar, nodes->GetSolution_New(iPoint, iVar) + tmp_func*Res);
857 }
858 }
859 break;
860 }
861
862 /*--- Update residual information for current thread. ---*/
863 ResidualReductions_PerThread(iPoint, iVar, Res, resRMS, resMax, idxMax);
864 }
865 }
866 END_SU2_OMP_FOR
867 /*--- Reduce residual information over all threads in this rank. ---*/
868 ResidualReductions_FromAllThreads(geometry, config, resRMS, resMax, idxMax);
869
870 }
871
872 /*--- MPI solution ---*/
873
874 InitiateComms(geometry, config, SOLUTION);
875 CompleteComms(geometry, config, SOLUTION);
876
877 if (!adjoint) {
878 /*--- For verification cases, compute the global error metrics. ---*/
879 ComputeVerificationError(geometry, config);
880 }
881
882 }

886 */
887 template<ENUM_TIME_INT IntegrationType>
888 FORCEINLINE void Explicit_Iteration(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iRKStep) {
889 struct Identity {
890 FORCEINLINE void compute(const CConfig*, unsigned long) {}
891 FORCEINLINE su2double apply(unsigned short iVar, const su2double* res, const su2double* resTrunc) {
892 return res[iVar] + resTrunc[iVar];

899 /*!
900 * \brief Generic implementation to prepare an implicit iteration with an optional preconditioner applied to the diagonal.
901 * \tparam DiagonalPrecond - A function object implementing:
902 * - active: A boolean variable to determine if the preconditioner should be used.
903 * - (config, iPoint, delta): Compute and return a matrix type compatible with the Jacobian matrix,
904 * where "delta" is V/dt.
905 */
906 template<class DiagonalPrecond>
907 void PrepareImplicitIteration_impl(DiagonalPrecond& preconditioner, CGeometry *geometry, CConfig *config) {
908
909 const bool implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
910
911 /*--- Local residual variables for current thread ---*/
912 su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
913 unsigned long idxMax[MAXNVAR] = {0};
914
915 /*--- Add pseudotime term to Jacobian. ---*/
916
917 if (implicit) {
918 SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
919 for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
920
921 /*--- Modify matrix diagonal to improve diagonal dominance. ---*/
922
923 if (nodes->GetDelta_Time(iPoint) != 0.0) {
924
925 su2double Vol = geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint);
926
927 su2double Delta = Vol / nodes->GetDelta_Time(iPoint);
928
929 if (preconditioner.active)
930 Jacobian.AddBlock2Diag(iPoint, preconditioner(config, iPoint, Delta));
931 else
932 Jacobian.AddVal2Diag(iPoint, Delta);
933 }
934 else {
935 Jacobian.SetVal2Diag(iPoint, 1.0);
936 }
937 }
938 END_SU2_OMP_FOR
939 }
940
941 /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
942
943 SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
944 for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
945
946 /*--- Multigrid contribution to residual. ---*/
947
948 su2double* local_Res_TruncError = nodes->GetResTruncError(iPoint);
949
950 if (nodes->GetDelta_Time(iPoint) == 0.0) {
951 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
952 LinSysRes(iPoint,iVar) = 0.0;
953 local_Res_TruncError[iVar] = 0.0;
954 }
955 }
956
957 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
958 unsigned long total_index = iPoint*nVar + iVar;
959 LinSysRes[total_index] = - (LinSysRes[total_index] + local_Res_TruncError[iVar]);
960 LinSysSol[total_index] = 0.0;
961
962 /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
963 ResidualReductions_PerThread(iPoint, iVar, LinSysRes[total_index], resRMS, resMax, idxMax);
964 }
965 }
966 END_SU2_OMP_FOR
967
968 /*--- "Add" residuals from all threads to global residual variables. ---*/
969 ResidualReductions_FromAllThreads(geometry, config, resRMS, resMax, idxMax);
970
971 }

974 * \brief Generic implementation to complete an implicit iteration, i.e. update the solution.
975 * \tparam compute_ur - Whether to use automatic under-relaxation for the update.
976 */
977 template<bool compute_ur>
978 void CompleteImplicitIteration_impl(CGeometry *geometry, CConfig *config) {
979
980 if (compute_ur) ComputeUnderRelaxationFactor(config);

982 /*--- Update solution with under-relaxation and communicate it. ---*/

985 SU2_OMP_FOR_STAT(omp_chunk_size)
986 for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
987 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
988 nodes->AddSolution(iPoint, iVar, nodes->GetUnderRelaxation(iPoint)*LinSysSol[iPoint*nVar+iVar]);

999 InitiateComms(geometry, config, SOLUTION);
1000 CompleteComms(geometry, config, SOLUTION);
1001
1002 /*--- For verification cases, compute the global error metrics. ---*/
1003 ComputeVerificationError(geometry, config);
1004 }
