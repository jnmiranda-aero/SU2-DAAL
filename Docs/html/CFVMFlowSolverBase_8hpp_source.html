287  inline void FinalizeResidualComputation(const CGeometry *geometry, bool pausePreacc,
288  unsigned long localCounter, CConfig* config) {
289 
290  /*--- Restore preaccumulation and adjoint evaluation state. ---*/
291  AD::ResumePreaccumulation(pausePreacc);
292  if (!ReducerStrategy) AD::EndNoSharedReading();
293 
294  if (ReducerStrategy) {
295  SumEdgeFluxes(geometry);
296  if (config->GetKind_TimeIntScheme() == EULER_IMPLICIT) {
297  Jacobian.SetDiagonalAsColumnSum();
298  }
299  }
300 
301  /*--- Warning message about non-physical reconstructions. ---*/
302  if ((MGLevel == MESH_0) && (config->GetComm_Level() == COMM_FULL)) {
303  /*--- Add counter results for all threads. ---*/
304  SU2_OMP_ATOMIC
305  ErrorCounter += localCounter;
306 
307  /*--- Add counter results for all ranks. ---*/
308  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
309  {
310  localCounter = ErrorCounter;
311  SU2_MPI::Reduce(&localCounter, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE, SU2_MPI::GetComm());
312  config->SetNonphysical_Reconstr(ErrorCounter);
313  }
314  END_SU2_OMP_SAFE_GLOBAL_ACCESS
315  }

364  /*!
365  * \brief Generic implementation to compute the time step based on CFL and conv/visc eigenvalues.
366  * \param[in] geometry - Geometrical definition of the problem.
367  * \param[in] solver_container - Container vector with all the solutions.
368  * \param[in] config - Definition of the particular problem.
369  * \param[in] iMesh - Index of the mesh in multigrid computations.
370  * \param[in] Iteration - Value of the current iteration.
371  * \tparam SoundSpeedFunc - Function object to compute speed of sound.
372  * \tparam LambdaViscFunc - Function object to compute the viscous lambda.
373  * \note Both functors need to implement (nodes,iPoint,jPoint) for edges, and (nodes,iPoint) for vertices.
374  */
375  template<class SoundSpeedFunc, class LambdaViscFunc>
376  FORCEINLINE void SetTime_Step_impl(const SoundSpeedFunc& soundSpeed,
377  const LambdaViscFunc& lambdaVisc,
378  CGeometry *geometry,
379  CSolver **solver_container,
380  CConfig *config,
381  unsigned short iMesh,
382  unsigned long Iteration) {
383 
384  const bool viscous = config->GetViscous();
385  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
386  const bool time_stepping = (config->GetTime_Marching() == TIME_MARCHING::TIME_STEPPING);
387  const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
388  (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
389  const su2double K_v = 0.25;
390 
391  /*--- Init thread-shared variables to compute min/max values.
392  * Critical sections are used for this instead of reduction
393  * clauses for compatibility with OpenMP 2.0 (Windows...). ---*/
394 
395  ompMasterAssignBarrier(Min_Delta_Time,1e30, Max_Delta_Time,0.0, Global_Delta_UnstTimeND,1e30);
396 
397  /*--- Loop domain points. ---*/
398 
399  SU2_OMP_FOR_DYN(omp_chunk_size)
400  for (auto iPoint = 0ul; iPoint < nPointDomain; ++iPoint) {
401 
402  /*--- Set maximum eigenvalues to zero. ---*/
403 
404  nodes->SetMax_Lambda_Inv(iPoint,0.0);
405 
406  if (viscous)
407  nodes->SetMax_Lambda_Visc(iPoint,0.0);
408 
409  /*--- Loop over the neighbors of point i. ---*/
410 
411  for (unsigned short iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); ++iNeigh)
412  {
413  auto jPoint = geometry->nodes->GetPoint(iPoint,iNeigh);
414 
415  auto iEdge = geometry->nodes->GetEdge(iPoint,iNeigh);
416  auto Normal = geometry->edges->GetNormal(iEdge);
417  auto Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
418 
419  /*--- Mean Values ---*/
420 
421  su2double Mean_ProjVel = 0.5 * (nodes->GetProjVel(iPoint,Normal) + nodes->GetProjVel(jPoint,Normal));
422  su2double Mean_SoundSpeed = soundSpeed(*nodes, iPoint, jPoint) * sqrt(Area2);
423 
424  /*--- Adjustment for grid movement ---*/
425 
426  if (dynamic_grid) {
427  const su2double *GridVel_i = geometry->nodes->GetGridVel(iPoint);
428  const su2double *GridVel_j = geometry->nodes->GetGridVel(jPoint);
429 
430  for (unsigned short iDim = 0; iDim < nDim; iDim++)
431  Mean_ProjVel -= 0.5 * (GridVel_i[iDim] + GridVel_j[iDim]) * Normal[iDim];
432  }
433 
434  /*--- Inviscid contribution ---*/
435 
436  su2double Lambda = fabs(Mean_ProjVel) + Mean_SoundSpeed;
437  nodes->AddMax_Lambda_Inv(iPoint,Lambda);
438 
439  /*--- Viscous contribution ---*/
440 
441  if (!viscous) continue;
442 
443  Lambda = lambdaVisc(*nodes, iPoint, jPoint) * Area2;
444  nodes->AddMax_Lambda_Visc(iPoint, Lambda);
445  }
446 
447  }
448  END_SU2_OMP_FOR
449 
450  /*--- Loop boundary edges ---*/
451 
452  for (unsigned short iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
453  if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
454  (config->GetMarker_All_KindBC(iMarker) != SYMMETRY_PLANE) &&
455  (config->GetMarker_All_KindBC(iMarker) != EULER_WALL) &&
456  (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
457  (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
458 
459  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
460  for (auto iVertex = 0ul; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
461 
462  /*--- Point identification, Normal vector and area ---*/
463 
464  auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
465 
466  if (!geometry->nodes->GetDomain(iPoint)) continue;
467 
468  auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
469  auto Area2 = GeometryToolbox::SquaredNorm(nDim, Normal);
470 
471  /*--- Mean Values ---*/
472 
473  su2double ProjVel = nodes->GetProjVel(iPoint,Normal);
474  su2double SoundSpeed = soundSpeed(*nodes, iPoint) * sqrt(Area2);
475 
476  /*--- Adjustment for grid movement ---*/
477 
478  if (dynamic_grid) {
479  ProjVel -= GeometryToolbox::DotProduct(nDim, Normal, geometry->nodes->GetGridVel(iPoint));
480  }
481 
482  /*--- Inviscid contribution ---*/
483 
484  su2double Lambda = fabs(ProjVel) + SoundSpeed;
485  nodes->AddMax_Lambda_Inv(iPoint, Lambda);
486 
487  /*--- Viscous contribution ---*/
488 
489  if (!viscous) continue;
490 
491  Lambda = lambdaVisc(*nodes,iPoint) * Area2;
492  nodes->AddMax_Lambda_Visc(iPoint, Lambda);
493  }
494  END_SU2_OMP_FOR
495  }
496  }
497 
498  /*--- Each element uses their own speed, steady state simulation. ---*/
499  {
500  /*--- Thread-local variables for min/max reduction. ---*/
501  su2double minDt = 1e30, maxDt = 0.0;
502 
503  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
504  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
505 
506  su2double Vol = geometry->nodes->GetVolume(iPoint);
507 
508  if (Vol != 0.0) {
509  su2double Local_Delta_Time = nodes->GetLocalCFL(iPoint)*Vol / nodes->GetMax_Lambda_Inv(iPoint);
510 
511  if(viscous) {
512  su2double dt_visc = nodes->GetLocalCFL(iPoint)*K_v*Vol*Vol / nodes->GetMax_Lambda_Visc(iPoint);
513  Local_Delta_Time = min(Local_Delta_Time, dt_visc);
514  }
515 
516  minDt = min(minDt, Local_Delta_Time);
517  maxDt = max(maxDt, Local_Delta_Time);
518 
519  nodes->SetDelta_Time(iPoint, min(Local_Delta_Time, config->GetMax_DeltaTime()));
520  }
521  else {
522  nodes->SetDelta_Time(iPoint,0.0);
523  }
524  }
525  END_SU2_OMP_FOR
526  /*--- Min/max over threads. ---*/
527  SU2_OMP_CRITICAL
528  {
529  Min_Delta_Time = min(Min_Delta_Time, minDt);
530  Max_Delta_Time = max(Max_Delta_Time, maxDt);
531  Global_Delta_Time = Min_Delta_Time;
532  }
533  END_SU2_OMP_CRITICAL
534  }
535 
536  /*--- Compute the min/max dt (in parallel, now over mpi ranks). ---*/
537 
538  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
539  if (config->GetComm_Level() == COMM_FULL) {
540  su2double rbuf_time;
541  SU2_MPI::Allreduce(&Min_Delta_Time, &rbuf_time, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
542  Min_Delta_Time = rbuf_time;
543 
544  SU2_MPI::Allreduce(&Max_Delta_Time, &rbuf_time, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
545  Max_Delta_Time = rbuf_time;
546  }
547  } END_SU2_OMP_SAFE_GLOBAL_ACCESS
548 
549  /*--- For exact time solution use the minimum delta time of the whole mesh. ---*/
550  if (time_stepping) {
551 
552  /*--- If the unsteady CFL is set to zero, it uses the defined unsteady time step,
553  * otherwise it computes the time step based on the unsteady CFL. ---*/
554 
555  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
556  {
557  if (config->GetUnst_CFL() == 0.0) {
558  Global_Delta_Time = config->GetDelta_UnstTime();
559  }
560  else {
561  Global_Delta_Time = Min_Delta_Time;
562  }
563  Max_Delta_Time = Global_Delta_Time;
564 
565  config->SetDelta_UnstTimeND(Global_Delta_Time);
566  }
567  END_SU2_OMP_SAFE_GLOBAL_ACCESS
568 
569  /*--- Sets the regular CFL equal to the unsteady CFL. ---*/
570 
571  SU2_OMP_FOR_STAT(omp_chunk_size)
572  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
573  nodes->SetLocalCFL(iPoint, config->GetUnst_CFL());
574  nodes->SetDelta_Time(iPoint, Global_Delta_Time);
575  }
576  END_SU2_OMP_FOR
577 
578  }
579 
580  /*--- Recompute the unsteady time step for the dual time strategy if the unsteady CFL is diferent from 0.
581  * This is only done once because in dual time the time step cannot be variable. ---*/
582 
583  if (dual_time && (Iteration == config->GetRestart_Iter()) && (config->GetUnst_CFL() != 0.0) && (iMesh == MESH_0)) {
584 
585  /*--- Thread-local variable for reduction. ---*/
586  su2double glbDtND = 1e30;
587 
588  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
589  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
590  glbDtND = min(glbDtND, config->GetUnst_CFL()*Global_Delta_Time / nodes->GetLocalCFL(iPoint));
591  }
592  END_SU2_OMP_FOR
593  SU2_OMP_CRITICAL
594  Global_Delta_UnstTimeND = min(Global_Delta_UnstTimeND, glbDtND);
595  END_SU2_OMP_CRITICAL
596 
597  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
598  {
599  SU2_MPI::Allreduce(&Global_Delta_UnstTimeND, &glbDtND, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
600  Global_Delta_UnstTimeND = glbDtND;
601 
602  config->SetDelta_UnstTimeND(Global_Delta_UnstTimeND);
603  }
604  END_SU2_OMP_SAFE_GLOBAL_ACCESS
605  }
606 
607  /*--- The pseudo local time (explicit integration) cannot be greater than the physical time ---*/
608 
609  if (dual_time && !implicit) {
610  SU2_OMP_FOR_STAT(omp_chunk_size)
611  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
612  su2double dt = min((2.0/3.0)*config->GetDelta_UnstTimeND(), nodes->GetDelta_Time(iPoint));
613  nodes->SetDelta_Time(iPoint, dt);
614  }
615  END_SU2_OMP_FOR
616  }

619  /*!
620  * \brief Compute the max eigenvalue, gemeric implementation.
621  * \param[in] geometry - Geometrical definition of the problem.
622  * \param[in] config - Definition of the particular problem.
623  * \tparam SoundSpeedFunc - Function object to compute speed of sound.
624  * \note Functor needs to implement (nodes,iPoint,jPoint) for edges, and (nodes,iPoint) for vertices.
625  */
626  template<class SoundSpeedFunc>
627  FORCEINLINE void SetMax_Eigenvalue_impl(const SoundSpeedFunc& soundSpeed, CGeometry *geometry, const CConfig *config) {
628 
629  /*--- Loop domain points. ---*/
630 
631  SU2_OMP_FOR_DYN(omp_chunk_size)
632  for (unsigned long iPoint = 0; iPoint < nPointDomain; ++iPoint) {
633 
634  /*--- Set eigenvalues to zero. ---*/
635  nodes->SetLambda(iPoint,0.0);
636 
637  /*--- Loop over the neighbors of point i. ---*/
638  for (unsigned short iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); ++iNeigh)
639  {
640  auto jPoint = geometry->nodes->GetPoint(iPoint, iNeigh);
641 
642  auto iEdge = geometry->nodes->GetEdge(iPoint, iNeigh);
643  auto Normal = geometry->edges->GetNormal(iEdge);
644  su2double Area = GeometryToolbox::Norm(nDim, Normal);
645 
646  /*--- Mean Values ---*/
647 
648  su2double Mean_ProjVel = 0.5 * (nodes->GetProjVel(iPoint,Normal) + nodes->GetProjVel(jPoint,Normal));
649  su2double Mean_SoundSpeed = soundSpeed(*nodes, iPoint, jPoint) * Area;
650 
651  /*--- Adjustment for grid movement ---*/
652 
653  if (dynamic_grid) {
654  const su2double *GridVel_i = geometry->nodes->GetGridVel(iPoint);
655  const su2double *GridVel_j = geometry->nodes->GetGridVel(jPoint);
656 
657  for (unsigned short iDim = 0; iDim < nDim; iDim++)
658  Mean_ProjVel -= 0.5 * (GridVel_i[iDim] + GridVel_j[iDim]) * Normal[iDim];
659  }
660 
661  /*--- Inviscid contribution ---*/
662 
663  nodes->AddLambda(iPoint, fabs(Mean_ProjVel) + Mean_SoundSpeed);
664  }
665  }
666  END_SU2_OMP_FOR
667 
668  /*--- Loop boundary edges ---*/
669 
670  for (unsigned short iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
671  if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
672  (config->GetMarker_All_KindBC(iMarker) != SYMMETRY_PLANE) &&
673  (config->GetMarker_All_KindBC(iMarker) != EULER_WALL) &&
674  (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
675  (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
676 
677  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
678  for (unsigned long iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
679 
680  /*--- Point identification, Normal vector and area ---*/
681 
682  auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
683 
684  if (!geometry->nodes->GetDomain(iPoint)) continue;
685 
686  auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
687  su2double Area = GeometryToolbox::Norm(nDim, Normal);
688 
689  /*--- Mean Values ---*/
690 
691  su2double Mean_ProjVel = nodes->GetProjVel(iPoint,Normal);
692  su2double Mean_SoundSpeed = soundSpeed(*nodes, iPoint) * Area;
693 
694  /*--- Adjustment for grid movement ---*/
695 
696  if (dynamic_grid) {
697  Mean_ProjVel -= GeometryToolbox::DotProduct(nDim, Normal, geometry->nodes->GetGridVel(iPoint));
698  }
699 
700  /*--- Inviscid contribution ---*/
701 
702  nodes->AddLambda(iPoint, fabs(Mean_ProjVel) + Mean_SoundSpeed);
703  }
704  END_SU2_OMP_FOR
705  }
706  }
707 
708  /*--- Correct the eigenvalue values across any periodic boundaries. ---*/
709 
710  for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic()/2; iPeriodic++) {
711  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_MAX_EIG);
712  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_MAX_EIG);
713  }
714 
715  /*--- MPI parallelization ---*/
716 
717  InitiateComms(geometry, config, MPI_QUANTITIES::MAX_EIGENVALUE);
718  CompleteComms(geometry, config, MPI_QUANTITIES::MAX_EIGENVALUE);
719  }

721  /*!
722  * \brief Compute the dissipation sensor for centered schemes.
723  * \param[in] geometry - Geometrical definition of the problem.
724  * \param[in] config - Definition of the particular problem.
725  * \tparam SensVarFunc - Function object implementing (nodes, iPoint) to return the sensor variable, e.g. pressure.
726  */
727  template<class SensVarFunc>
728  FORCEINLINE void SetCentered_Dissipation_Sensor_impl(const SensVarFunc& sensVar,
729  CGeometry *geometry, const CConfig *config) {
730 
731  /*--- We can access memory more efficiently if there are no periodic boundaries. ---*/
732 
733  const bool isPeriodic = (config->GetnMarker_Periodic() > 0);
734 
735  /*--- Loop domain points. ---*/
736 
737  SU2_OMP_FOR_DYN(omp_chunk_size)
738  for (unsigned long iPoint = 0; iPoint < nPointDomain; ++iPoint) {
739 
740  const bool boundary_i = geometry->nodes->GetPhysicalBoundary(iPoint);
741  const su2double sensVar_i = sensVar(*nodes, iPoint);
742 
743  /*--- Initialize. ---*/
744  iPoint_UndLapl[iPoint] = 0.0;
745  jPoint_UndLapl[iPoint] = 0.0;
746 
747  /*--- Loop over the neighbors of point i. ---*/
748  for (auto jPoint : geometry->nodes->GetPoints(iPoint))
749  {
750  bool boundary_j = geometry->nodes->GetPhysicalBoundary(jPoint);
751 
752  /*--- If iPoint is boundary it only takes contributions from other boundary points. ---*/
753  if (boundary_i && !boundary_j) continue;
754 
755  su2double sensVar_j = sensVar(*nodes, jPoint);
756 
757  /*--- Dissipation sensor, add variable difference and variable sum. ---*/
758  iPoint_UndLapl[iPoint] += sensVar_j - sensVar_i;
759  jPoint_UndLapl[iPoint] += sensVar_j + sensVar_i;
760  }
761 
762  if (!isPeriodic) {
763  /*--- Every neighbor is accounted for, sensor can be computed. ---*/
764  nodes->SetSensor(iPoint, fabs(iPoint_UndLapl[iPoint]) / jPoint_UndLapl[iPoint]);
765  }
766  }
767  END_SU2_OMP_FOR
768 
769  if (isPeriodic) {
770  /*--- Correct the sensor values across any periodic boundaries. ---*/
771 
772  for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic()/2; iPeriodic++) {
773  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_SENSOR);
774  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_SENSOR);
775  }
776 
777  /*--- Set final pressure switch for each point ---*/
778 
779  SU2_OMP_FOR_STAT(omp_chunk_size)
780  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++)
781  nodes->SetSensor(iPoint, fabs(iPoint_UndLapl[iPoint]) / jPoint_UndLapl[iPoint]);
782  END_SU2_OMP_FOR
783  }
784 
785  /*--- MPI parallelization ---*/
786 
787  InitiateComms(geometry, config, MPI_QUANTITIES::SENSOR);
788  CompleteComms(geometry, config, MPI_QUANTITIES::SENSOR);
789 
790  }

792  /*!
793  * \brief Generic implementation of explicit iterations with a preconditioner.
794  * \note The preconditioner is a functor implementing the methods:
795  * - compute(config, iPoint): Should prepare the preconditioner for iPoint.
796  * - apply(iVar, residual[], resTruncError[]): Apply it to compute the iVar update.
797  * See Explicit_Iteration for the general form of the preconditioner.
798  */
799  template<ENUM_TIME_INT IntegrationType, class ResidualPrecond>
800  void Explicit_Iteration_impl(ResidualPrecond& preconditioner, CGeometry *geometry,
801  CSolver **solver_container, CConfig *config, unsigned short iRKStep) {
802 
803  static_assert(IntegrationType == CLASSICAL_RK4_EXPLICIT ||
804  IntegrationType == RUNGE_KUTTA_EXPLICIT ||
805  IntegrationType == EULER_EXPLICIT, "");
806 
807  const bool adjoint = config->GetContinuous_Adjoint();
808 
809  const su2double RK_AlphaCoeff = config->Get_Alpha_RKStep(iRKStep);
810 
811  /*--- Hard-coded classical RK4 coefficients. Will be added to config. ---*/
812  const su2double RK_FuncCoeff[] = {1.0/6.0, 1.0/3.0, 1.0/3.0, 1.0/6.0};
813  const su2double RK_TimeCoeff[] = {0.5, 0.5, 1.0, 1.0};
814 
815  /*--- Local residual variables for current thread ---*/
816  su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
817  unsigned long idxMax[MAXNVAR] = {0};
818 
819  /*--- Update the solution and residuals ---*/
820 
821  if (!adjoint) {
822  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
823  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
824 
825  su2double Vol = geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint);
826  su2double Delta = nodes->GetDelta_Time(iPoint) / Vol;
827 
828  const su2double* Res_TruncError = nodes->GetResTruncError(iPoint);
829  const su2double* Residual = LinSysRes.GetBlock(iPoint);
830 
831  preconditioner.compute(config, iPoint);
832 
833  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
834 
835  su2double Res = preconditioner.apply(iVar, Residual, Res_TruncError);
836 
837  /*--- "Static" switch which should be optimized at compile time. ---*/
838  switch(IntegrationType) {
839 
840  case EULER_EXPLICIT:
841  nodes->AddSolution(iPoint,iVar, -Res*Delta);
842  break;
843 
844  case RUNGE_KUTTA_EXPLICIT:
845  nodes->AddSolution(iPoint, iVar, -Res*Delta*RK_AlphaCoeff);
846  break;
847 
848  case CLASSICAL_RK4_EXPLICIT:
849  {
850  su2double tmp_time = -1.0*RK_TimeCoeff[iRKStep]*Delta;
851  su2double tmp_func = -1.0*RK_FuncCoeff[iRKStep]*Delta;
852 
853  if (iRKStep < 3) {
854  /* Base Solution Update */
855  nodes->AddSolution(iPoint,iVar, tmp_time*Res);
856 
857  /* New Solution Update */
858  nodes->AddSolution_New(iPoint,iVar, tmp_func*Res);
859  } else {
860  nodes->SetSolution(iPoint, iVar, nodes->GetSolution_New(iPoint, iVar) + tmp_func*Res);
861  }
862  }
863  break;
864  }
865 
866  /*--- Update residual information for current thread. ---*/
867  ResidualReductions_PerThread(iPoint, iVar, Res, resRMS, resMax, idxMax);
868  }
869  }
870  END_SU2_OMP_FOR
871  /*--- Reduce residual information over all threads in this rank. ---*/
872  ResidualReductions_FromAllThreads(geometry, config, resRMS, resMax, idxMax);
873 
874  }
875 
876  /*--- MPI solution ---*/
877 
878  InitiateComms(geometry, config, MPI_QUANTITIES::SOLUTION);
879  CompleteComms(geometry, config, MPI_QUANTITIES::SOLUTION);
880 
881  if (!adjoint) {
882  /*--- For verification cases, compute the global error metrics. ---*/
883  ComputeVerificationError(geometry, config);
884  }
885 
886  }

903  /*!
904  * \brief Generic implementation to prepare an implicit iteration with an optional preconditioner applied to the diagonal.
905  * \tparam DiagonalPrecond - A function object implementing:
906  * - active: A boolean variable to determine if the preconditioner should be used.
907  * - (config, iPoint, delta): Compute and return a matrix type compatible with the Jacobian matrix,
908  * where "delta" is V/dt.
909  */
910  template<class DiagonalPrecond>
911  void PrepareImplicitIteration_impl(DiagonalPrecond& preconditioner, CGeometry *geometry, CConfig *config) {
912 
913  const bool implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
914 
915  /*--- Local residual variables for current thread ---*/
916  su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
917  unsigned long idxMax[MAXNVAR] = {0};
918 
919  /*--- Add pseudotime term to Jacobian. ---*/
920 
921  if (implicit) {
922  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
923  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
924 
925  /*--- Modify matrix diagonal to improve diagonal dominance. ---*/
926 
927  if (nodes->GetDelta_Time(iPoint) != 0.0) {
928 
929  su2double Vol = geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint);
930 
931  su2double Delta = Vol / nodes->GetDelta_Time(iPoint);
932 
933  if (preconditioner.active)
934  Jacobian.AddBlock2Diag(iPoint, preconditioner(config, iPoint, Delta));
935  else
936  Jacobian.AddVal2Diag(iPoint, Delta);
937  }
938  else {
939  Jacobian.SetVal2Diag(iPoint, 1.0);
940  }
941  }
942  END_SU2_OMP_FOR
943  }
944 
945  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
946 
947  SU2_OMP_FOR_(schedule(static,omp_chunk_size) SU2_NOWAIT)
948  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
949 
950  /*--- Multigrid contribution to residual. ---*/
951 
952  su2double* local_Res_TruncError = nodes->GetResTruncError(iPoint);
953 
954  if (nodes->GetDelta_Time(iPoint) == 0.0) {
955  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
956  LinSysRes(iPoint,iVar) = 0.0;
957  local_Res_TruncError[iVar] = 0.0;
958  }
959  }
960 
961  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
962  unsigned long total_index = iPoint*nVar + iVar;
963  LinSysRes[total_index] = - (LinSysRes[total_index] + local_Res_TruncError[iVar]);
964  LinSysSol[total_index] = 0.0;
965 
966  /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
967  ResidualReductions_PerThread(iPoint, iVar, LinSysRes[total_index], resRMS, resMax, idxMax);
968  }
969  }
970  END_SU2_OMP_FOR
971 
972  /*--- "Add" residuals from all threads to global residual variables. ---*/
973  ResidualReductions_FromAllThreads(geometry, config, resRMS, resMax, idxMax);
974 
975  }
