32void Partition_Analysis(CGeometry *geometry, CConfig *config) {
33
34 /*--- This routine does a quick and dirty output of the total
35 vertices, ghost vertices, total elements, ghost elements, etc.,
36 so that we can analyze the partition quality. ---*/
37
38 unsigned short nMarker = config->GetnMarker_All();
39 unsigned short iMarker, iNodes, MarkerS, MarkerR;
40 unsigned long iElem, iPoint, nVertexS, nVertexR;
41 unsigned long nNeighbors = 0, nSendTotal = 0, nRecvTotal = 0;
42 unsigned long nPointTotal=0, nPointGhost=0, nElemTotal=0;
43 unsigned long nElemHalo=0, nEdge=0, nElemBound=0;
44 int iRank;
45 int rank = MASTER_NODE;
46 int size = SINGLE_NODE;
47
48#ifdef HAVE_MPI
49 SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &rank);
50 SU2_MPI::Comm_size(SU2_MPI::GetComm(), &size);
51#endif
52
53 nPointTotal = geometry->GetnPoint();
54 nPointGhost = geometry->GetnPoint() - geometry->GetnPointDomain();
55 nElemTotal = geometry->GetnElem();
56 nEdge = geometry->GetnEdge();
57
58 for (iMarker = 0; iMarker < nMarker; iMarker++) {
59 nElemBound += geometry->GetnElem_Bound(iMarker);
60 if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) &&
61 (config->GetMarker_All_SendRecv(iMarker) > 0)) {
62 MarkerS = iMarker; MarkerR = iMarker+1;
63 nVertexS = geometry->nVertex[MarkerS];
64 nVertexR = geometry->nVertex[MarkerR];
65 nNeighbors++;
66 nSendTotal += nVertexS;
67 nRecvTotal += nVertexR;
68 }
69 }
70
71 bool *isHalo = new bool[geometry->GetnElem()];
72 for (iElem = 0; iElem < geometry->GetnElem(); iElem++) {
73 isHalo[iElem] = false;
74 for (iNodes = 0; iNodes < geometry->elem[iElem]->GetnNodes(); iNodes++) {
75 iPoint = geometry->elem[iElem]->GetNode(iNodes);
76 if (!geometry->nodes->GetDomain(iPoint)) isHalo[iElem] = true;
77 }
78 }
79
80 for (iElem = 0; iElem < geometry->GetnElem(); iElem++) {
81 if (isHalo[iElem]) nElemHalo++;
82 }
83
84 unsigned long *row_ptr = nullptr, nnz;
85 unsigned short *nNeigh = nullptr;
86 vector<unsigned long> vneighs;
87
88 /*--- Don't delete *row_ptr, *col_ind because they are
89 asigned to the Jacobian structure. ---*/
90
91 /*--- Compute the number of neighbors ---*/
92
93 nNeigh = new unsigned short [geometry->GetnPoint()];
94 for (iPoint = 0; iPoint < geometry->GetnPoint(); iPoint++) {
95 // +1 -> to include diagonal element
96 nNeigh[iPoint] = (geometry->nodes->GetnPoint(iPoint)+1);
97 }
98
99 /*--- Create row_ptr structure, using the number of neighbors ---*/
100
101 row_ptr = new unsigned long [geometry->GetnPoint()+1];
102 row_ptr[0] = 0;
103 for (iPoint = 0; iPoint < geometry->GetnPoint(); iPoint++)
104 row_ptr[iPoint+1] = row_ptr[iPoint] + nNeigh[iPoint];
105 nnz = row_ptr[geometry->GetnPoint()];
106
107 delete [] row_ptr;
108 delete [] nNeigh;
109
110 /*--- Now put this info into a CSV file for processing ---*/
111
112 ofstream Profile_File;
113 Profile_File.precision(15);
114
115 if (rank == MASTER_NODE) {
116 /*--- Prepare and open the file ---*/
117 Profile_File.open("partitioning.csv");
118 /*--- Create the CSV header ---*/
119 Profile_File << R"("Rank", "nNeighbors", "nPointTotal", "nEdge", "nPointGhost", "nSendTotal", "nRecvTotal", "nElemTotal", "nElemBoundary", "nElemHalo", "nnz")" << endl;
120 Profile_File.close();
121 }
122 SU2_MPI::Barrier(SU2_MPI::GetComm());
123
124 /*--- Loop through the map and write the results to the file ---*/
125
126 for (iRank = 0; iRank < size; iRank++) {
127 if (rank == iRank) {
128 Profile_File.open("partitioning.csv", ios::out | ios::app);
129 Profile_File << rank << ", " << nNeighbors << ", " << nPointTotal << ", " << nEdge << "," << nPointGhost << ", " << nSendTotal << ", " << nRecvTotal << ", " << nElemTotal << "," << nElemBound << ", " << nElemHalo << ", " << nnz << endl;
130 Profile_File.close();
131 }
132 SU2_MPI::Barrier(SU2_MPI::GetComm());
133 }
134
135 delete [] isHalo;
136
137}

139void Partition_Analysis_FEM(CGeometry *geometry, CConfig *config) {
140
141 /*--- This routine does a quick and dirty output of the total
142 vertices, ghost vertices, total elements, ghost elements, etc.,
143 so that we can analyze the partition quality. ---*/
144
145 unsigned long nNeighSend = 0, nNeighRecv = 0;
146 unsigned long nElemOwned = 0, nElemSendTotal = 0, nElemRecvTotal = 0;
147 unsigned long nDOFOwned = 0, nDOFSendTotal = 0, nDOFRecvTotal = 0;
148
149 int iRank;
150 int rank = MASTER_NODE;
151 int size = SINGLE_NODE;
152
153#ifdef HAVE_MPI
154 SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &rank);
155 SU2_MPI::Comm_size(SU2_MPI::GetComm(), &size);
156#endif
157
158 /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
159 geometrical information for the FEM DG solver. ---*/
160 auto *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
161
162 unsigned long nVolElemOwned = DGGeometry->GetNVolElemOwned();
163 CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
164
165 /*--- Determine the number of owned elements and DOFs. ---*/
166 nElemOwned = nVolElemOwned;
167 for(unsigned long l=0; l<nVolElemOwned; ++l) {
168 nDOFOwned += volElem[l].nDOFsSol;
169 }
170
171 /*--- Get the communication information from DG_Geometry. Note that for a
172 FEM DG discretization the communication entities of FEMGeometry contain
173 the volume elements. ---*/
174 const vector<int> &ranksSend = DGGeometry->GetRanksSend();
175 const vector<int> &ranksRecv = DGGeometry->GetRanksRecv();
176 const vector<vector<unsigned long> > &elementsSend = DGGeometry->GetEntitiesSend();
177 const vector<vector<unsigned long> > &elementsRecv = DGGeometry->GetEntitiesRecv();
178
179 nNeighSend = ranksSend.size();
180 nNeighRecv = ranksRecv.size();
181
182 /*--- Determine the total number of elements and DOFS to be send. ---*/
183 for(unsigned long i=0; i<ranksSend.size(); ++i) {
184
185 const auto nElemSend = (unsigned int)elementsSend[i].size();
186
187 nElemSendTotal += nElemSend;
188
189 for(unsigned int j=0; j<nElemSend; ++j) {
190 const unsigned long jj = elementsSend[i][j];
191 nDOFSendTotal += volElem[jj].nDOFsSol;
192 }
193 }
194
195 /*--- Determine the total number of elements and DOFS to be received. ---*/
196 for(unsigned long i=0; i<ranksRecv.size(); ++i) {
197
198 const auto nElemRecv = (unsigned int)elementsRecv[i].size();
199
200 nElemRecvTotal += nElemRecv;
201
202 for(unsigned int j=0; j<nElemRecv; ++j) {
203 const unsigned long jj = elementsRecv[i][j];
204 nDOFRecvTotal += volElem[jj].nDOFsSol;
205 }
206
207 }
208
209 /*--- Now put this info into a CSV file for processing ---*/
210
211 ofstream Profile_File;
212 Profile_File.precision(15);
213
214 if (rank == MASTER_NODE) {
215 /*--- Prepare and open the file ---*/
216 Profile_File.open("partitioning.csv");
217 /*--- Create the CSV header ---*/
218 Profile_File << R"("Rank", "nNeighSend", "nNeighRecv", "nElemOwned", "nElemSendTotal", "nElemRecvTotal", "nDOFOwned", "nDOFSendTotal", "nDOFRecvTotal")" << endl;
219 Profile_File.close();
220 }
221 SU2_MPI::Barrier(SU2_MPI::GetComm());
222
223 /*--- Loop through the map and write the results to the file ---*/
224
225 for (iRank = 0; iRank < size; iRank++) {
226 if (rank == iRank) {
227 Profile_File.open("partitioning.csv", ios::out | ios::app);
228 Profile_File << rank << ", " << nNeighSend << ", " << nNeighRecv << ", " << nElemOwned << ", "
229 << nElemSendTotal << ", " << nElemRecvTotal << ", " << nDOFOwned << ", "
230 << nDOFSendTotal << ", " << nDOFRecvTotal << endl;
231 Profile_File.close();
232 }
233 SU2_MPI::Barrier(SU2_MPI::GetComm());
234 }
