40CIteration* CIterationFactory::CreateIteration(MAIN_SOLVER kindSolver, const CConfig* config){
41
42 CIteration *iteration = nullptr;
43
44 const auto rank = SU2_MPI::GetRank();
45
46 /*--- Loop over all zones and instantiate the physics iteration. ---*/
47
48 switch (kindSolver) {
49
50 case MAIN_SOLVER::EULER: case MAIN_SOLVER::NAVIER_STOKES: case MAIN_SOLVER::RANS:
51 case MAIN_SOLVER::INC_EULER: case MAIN_SOLVER::INC_NAVIER_STOKES: case MAIN_SOLVER::INC_RANS:
52 case MAIN_SOLVER::NEMO_EULER: case MAIN_SOLVER::NEMO_NAVIER_STOKES:
53 if(config->GetBoolTurbomachinery()){
54 if (rank == MASTER_NODE)
55 cout << "Euler/Navier-Stokes/RANS turbomachinery fluid iteration." << endl;
56 iteration = new CTurboIteration(config);
57
58 }
59 else{
60 if (rank == MASTER_NODE)
61 cout << "Euler/Navier-Stokes/RANS fluid iteration." << endl;
62 iteration = new CFluidIteration(config);
63 }
64 break;
65
66 case MAIN_SOLVER::FEM_EULER: case MAIN_SOLVER::FEM_NAVIER_STOKES: case MAIN_SOLVER::FEM_RANS: case MAIN_SOLVER::FEM_LES:
67 if (rank == MASTER_NODE)
68 cout << "Finite element Euler/Navier-Stokes/RANS/LES flow iteration." << endl;
69 iteration = new CFEMFluidIteration(config);
70 break;
71
72 case MAIN_SOLVER::HEAT_EQUATION:
73 if (rank == MASTER_NODE)
74 cout << "Heat iteration (finite volume method)." << endl;
75 iteration = new CHeatIteration(config);
76 break;
77
78 case MAIN_SOLVER::FEM_ELASTICITY:
79 if (rank == MASTER_NODE)
80 cout << "FEM iteration." << endl;
81 iteration = new CFEAIteration(config);
82 break;
83
84 case MAIN_SOLVER::ADJ_EULER: case MAIN_SOLVER::ADJ_NAVIER_STOKES: case MAIN_SOLVER::ADJ_RANS:
85 if (rank == MASTER_NODE)
86 cout << "Adjoint Euler/Navier-Stokes/RANS fluid iteration." << endl;
87 iteration = new CAdjFluidIteration(config);
88 break;
89
90 case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
91 case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
92 if (rank == MASTER_NODE)
93 cout << "Discrete adjoint Euler/Navier-Stokes/RANS fluid iteration." << endl;
94 iteration = new CDiscAdjFluidIteration(config);
95 break;
96
97 case MAIN_SOLVER::DISC_ADJ_FEM_EULER : case MAIN_SOLVER::DISC_ADJ_FEM_NS : case MAIN_SOLVER::DISC_ADJ_FEM_RANS :
98 if (rank == MASTER_NODE)
99 cout << "Discrete adjoint finite element Euler/Navier-Stokes/RANS fluid iteration." << endl;
100 iteration = new CDiscAdjFluidIteration(config);
101 break;
102
103 case MAIN_SOLVER::DISC_ADJ_FEM:
104 if (rank == MASTER_NODE)
105 cout << "Discrete adjoint FEM structural iteration." << endl;
106 iteration = new CDiscAdjFEAIteration(config);
107 break;
108
109 case MAIN_SOLVER::DISC_ADJ_HEAT:
110 if (rank == MASTER_NODE)
111 cout << "Discrete adjoint heat iteration." << endl;
112 iteration = new CDiscAdjHeatIteration(config);
113 break;
114
115 case MAIN_SOLVER::NONE: case MAIN_SOLVER::TEMPLATE_SOLVER: case MAIN_SOLVER::MULTIPHYSICS:
116 SU2_MPI::Error("No iteration found for specified solver.", CURRENT_FUNCTION);
117 break;
118 }
119
120 return iteration;
121}
