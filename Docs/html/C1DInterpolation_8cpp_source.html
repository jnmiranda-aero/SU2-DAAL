55 void CCubicSpline::SetSpline(const vector<su2double>& X, const vector<su2double>& Data) {
56  C1DInterpolation::SetSpline(X, Data);
57 
58  const int N = x.size();
59 
60  /*--- Alias the vectors of coefficients to build the tridiagonal system. ---*/
61 
62  auto& lower = b;
63  b.resize(N);
64  auto& main = c;
65  c.resize(N);
66  auto& upper = d;
67  d.resize(N);
68  vector<su2double> rhs(N);
69 
70  /*--- Main part of the tridiagonal system. ---*/
71 
72  for (int i = 1; i < N - 1; i++) {
73  lower[i] = x[i] - x[i - 1];
74  main[i] = (x[i + 1] - x[i - 1]) * 2;
75  upper[i] = x[i + 1] - x[i];
76  rhs[i] = 6 * ((y[i + 1] - y[i]) / upper[i] - (y[i] - y[i - 1]) / lower[i]);
77  }
78 
79  /*--- Start condition. ---*/
80 
81  if (startDer == SECOND) {
82  main[0] = 1.0;
83  upper[0] = 0;
84  rhs[0] = startVal;
85  } else { // FIRST
86  main[0] = 2 * lower[1];
87  upper[0] = lower[1];
88  rhs[0] = 6 * ((y[1] - y[0]) / lower[1] - startVal);
89  }
90 
91  /*--- End condition. ---*/
92 
93  if (endDer == SECOND) {
94  main[N - 1] = 1.0;
95  lower[N - 1] = 0;
96  rhs[N - 1] = endVal;
97  } else { // FIRST
98  main[N - 1] = 2 * upper[N - 2];
99  lower[N - 1] = upper[N - 2];
100  rhs[N - 1] = 6 * (endVal - (y[N - 1] - y[N - 2]) / upper[N - 2]);
101  }
102 
103  /*--- Solve system for 2nd derivative at the knots. ---*/
104 
105  CBlasStructure::tdma(lower, main, upper, rhs);
106  const auto& d2y = rhs;
107 
108  /*--- Compute the polynomial coefficients. ---*/
109 
110  for (int i = 0; i < N - 1; i++) {
111  su2double dx = x[i + 1] - x[i];
112  c[i] = 0.5 * d2y[i];
113  d[i] = (d2y[i + 1] - d2y[i]) / (6 * dx);
114  b[i] = (y[i + 1] - y[i]) / dx - (c[i] + d[i] * dx) * dx;
115  }
116 }

118 void CAkimaInterpolation::SetSpline(const vector<su2double>& X, const vector<su2double>& Data) {
119  C1DInterpolation::SetSpline(X, Data);
120 
121  const int n = X.size();
122  vector<su2double> h(n - 1);
123  vector<su2double> p(n - 1);
124 
125  /*---calculating finite differences (h) and gradients (p) ---*/
126  for (int i = 0; i < n - 1; i++) {
127  h[i] = X[i + 1] - X[i];
128  p[i] = (Data[i + 1] - Data[i]) / h[i];
129  }
130 
131  /*--- b,c,d are the akima spline's cofficient for the cubic equation ---*/
132  b.resize(n);
133  c.resize(n);
134  d.resize(n);
135 
136  b[0] = p[0];
137  b[1] = (p[0] + p[1]) / 2;
138  b[n - 1] = p[n - 2];
139  b[n - 2] = (p[n - 2] + p[n - 3]) / 2;
140 
141  for (int i = 2; i < n - 2; i++) {
142  su2double w1 = fabs(p[i + 1] - p[i]), w2 = fabs(p[i - 1] - p[i - 2]);
143  if (w1 + w2 < 0.0001) {
144  b[i] = (p[i - 1] + p[i]) / 2;
145  } else {
146  b[i] = (w1 * p[i - 1] + w2 * p[i]) / (w1 + w2);
147  }
148  }
149 
150  for (int i = 0; i < n - 1; i++) {
151  c[i] = (3 * p[i] - 2 * b[i] - b[i + 1]) / h[i];
152  d[i] = (b[i + 1] + b[i] - 2 * p[i]) / h[i] / h[i];
153  }
154  b.back() = c.back() = 0;
155 }

157 vector<su2double> CorrectedInletValues(const vector<su2double>& Inlet_Interpolated, su2double Theta,
158  unsigned short nDim, const su2double* Coord, unsigned short nVar_Turb,
159  INLET_INTERP_TYPE Interpolation_Type) {
160  unsigned short size_columns = Inlet_Interpolated.size() + nDim;
161  vector<su2double> Inlet_Values(size_columns);
162  su2double unit_r, unit_Theta, unit_m, Alpha, Phi;
163 
164  /*---For x,y,z,T,P columns---*/
165  for (int i = 0; i < nDim; i++) Inlet_Values[i] = Coord[i];
166 
167  for (int i = nDim; i < nDim + 2; i++) Inlet_Values[i] = Inlet_Interpolated[i - 2];
168 
169  /*---For turbulence variables columns---*/
170  if (nVar_Turb == 1)
171  Inlet_Values[nDim + 5] = Inlet_Interpolated[5];
172  else if (nVar_Turb == 2) {
173  Inlet_Values[nDim + 5] = Inlet_Interpolated[5];
174  Inlet_Values[nDim + 6] = Inlet_Interpolated[6];
175  }
176 
177  /*--- Correct for Interpolation Type now ---*/
178  switch (Interpolation_Type) {
179  case (INLET_INTERP_TYPE::VR_VTHETA):
180  unit_r = Inlet_Interpolated[nDim];
181  unit_Theta = Inlet_Interpolated[nDim + 1];
182  break;
183 
184  case (INLET_INTERP_TYPE::ALPHA_PHI):
185  Alpha = Inlet_Interpolated[nDim] * PI_NUMBER / 180;
186  Phi = Inlet_Interpolated[nDim + 1] * PI_NUMBER / 180;
187  unit_m = sqrt(1 / (1 + pow(tan(Alpha), 2)));
188  unit_Theta = tan(Alpha) * unit_m;
189  unit_r = unit_m * sin(Phi);
190  break;
191 
192  default:
193  unit_Theta = unit_r = 0.0;
194  break;
195  }
196 
197  /*--- Converting from cylindrical to cartesian unit vectors ---*/
198  Inlet_Values[nDim + 2] = unit_r * cos(Theta) - unit_Theta * sin(Theta);
199  Inlet_Values[nDim + 3] = unit_r * sin(Theta) + unit_Theta * cos(Theta);
200  Inlet_Values[nDim + 4] = sqrt(1 - pow(unit_r, 2) - pow(unit_Theta, 2));
201 
202  return Inlet_Values;
203 }
