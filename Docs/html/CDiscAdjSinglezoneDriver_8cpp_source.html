130 void CDiscAdjSinglezoneDriver::Preprocess(unsigned long TimeIter) {
131 
132  /*--- Set the current time iteration in the config and also in the driver
133  * because the python interface doesn't offer an explicit way of doing it. ---*/
134 
135  this->TimeIter = TimeIter;
136  config_container[ZONE_0]->SetTimeIter(TimeIter);
137 
138  /*--- Preprocess the adjoint iteration ---*/
139 
140  iteration->Preprocess(output_container[ZONE_0], integration_container, geometry_container,
141  solver_container, numerics_container, config_container,
142  surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
143 
144  /*--- For the adjoint iteration we need the derivatives of the iteration function with
145  *--- respect to the conservative variables. Since these derivatives do not change in the steady state case
146  *--- we only have to record if the current recording is different from the main variables. ---*/
147 
148  if (RecordingState != MainVariables){
149  MainRecording();
150  }

154 void CDiscAdjSinglezoneDriver::Run() {
155 
156  CQuasiNewtonInvLeastSquares<passivedouble> fixPtCorrector;
157  if (config->GetnQuasiNewtonSamples() > 1) {
158  fixPtCorrector.resize(config->GetnQuasiNewtonSamples(),
159  geometry_container[ZONE_0][INST_0][MESH_0]->GetnPoint(),
160  GetTotalNumberOfVariables(ZONE_0,true),
161  geometry_container[ZONE_0][INST_0][MESH_0]->GetnPointDomain());
162 
163  if (TimeIter != 0) GetAllSolutions(ZONE_0, true, fixPtCorrector);
164  }
165 
166  for (auto Adjoint_Iter = 0ul; Adjoint_Iter < nAdjoint_Iter; Adjoint_Iter++) {
167 
168  /*--- Initialize the adjoint of the output variables of the iteration with the adjoint solution
169  *--- of the previous iteration. The values are passed to the AD tool.
170  *--- Issues with iteration number should be dealt with once the output structure is in place. ---*/
171 
172  config->SetInnerIter(Adjoint_Iter);
173 
174  iteration->InitializeAdjoint(solver_container, geometry_container, config_container, ZONE_0, INST_0);
175 
176  /*--- Initialize the adjoint of the objective function with 1.0. ---*/
177 
178  SetAdjObjFunction();
179 
180  /*--- Interpret the stored information by calling the corresponding routine of the AD tool. ---*/
181 
182  AD::ComputeAdjoint();
183 
184  /*--- Extract the computed adjoint values of the input variables and store them for the next iteration. ---*/
185 
186  iteration->IterateDiscAdj(geometry_container, solver_container,
187  config_container, ZONE_0, INST_0, false);
188 
189  /*--- Monitor the pseudo-time ---*/
190 
191  StopCalc = iteration->Monitor(output_container[ZONE_0], integration_container, geometry_container,
192  solver_container, numerics_container, config_container,
193  surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
194 
195  /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
196 
197  AD::ClearAdjoints();
198 
199  /*--- Output files for steady state simulations. ---*/
200 
201  if (!config->GetTime_Domain()) {
202  iteration->Output(output_container[ZONE_0], geometry_container, solver_container,
203  config_container, Adjoint_Iter, false, ZONE_0, INST_0);
204  }
205 
206  if (StopCalc) break;
207 
208  /*--- Correct the solution with the quasi-Newton approach. ---*/
209 
210  if (fixPtCorrector.size()) {
211  GetAllSolutions(ZONE_0, true, fixPtCorrector.FPresult());
212  SetAllSolutions(ZONE_0, true, fixPtCorrector.compute());
213  }
214 
215  }

248 void CDiscAdjSinglezoneDriver::SetRecording(RECORDING kind_recording){
249 
250  AD::Reset();
251 
252  /*--- Prepare for recording by resetting the solution to the initial converged solution. ---*/
253 
254  for (unsigned short iSol=0; iSol < MAX_SOLS; iSol++) {
255  for (unsigned short iMesh = 0; iMesh <= config_container[ZONE_0]->GetnMGLevels(); iMesh++) {
256  auto solver = solver_container[ZONE_0][INST_0][iMesh][iSol];
257  if (solver && solver->GetAdjoint()) {
258  SU2_OMP_PARALLEL_(if (solver->GetHasHybridParallel()))
259  solver->SetRecording(geometry_container[ZONE_0][INST_0][iMesh], config_container[ZONE_0]);
260  END_SU2_OMP_PARALLEL
261  }
262  }
263  }
264 
265  if (rank == MASTER_NODE) {
266  cout << "\n-------------------------------------------------------------------------\n";
267  switch(kind_recording) {
268  case RECORDING::CLEAR_INDICES: cout << "Clearing the computational graph." << endl; break;
269  case RECORDING::MESH_COORDS: cout << "Storing computational graph wrt MESH COORDINATES." << endl; break;
270  case RECORDING::SOLUTION_VARIABLES:
271  cout << "Direct iteration to store the primal computational graph." << endl;
272  cout << "Computing residuals to check the convergence of the direct problem." << endl; break;
273  default: break;
274  }
275  }
276 
277  /*---Enable recording and register input of the iteration --- */
278 
279  if (kind_recording != RECORDING::CLEAR_INDICES){
280 
281  AD::StartRecording();
282 
283  iteration->RegisterInput(solver_container, geometry_container, config_container, ZONE_0, INST_0, kind_recording);
284  }
285 
286  /*--- Set the dependencies of the iteration ---*/
287 
288  iteration->SetDependencies(solver_container, geometry_container, numerics_container, config_container, ZONE_0,
289  INST_0, kind_recording);
290 
291  /*--- Do one iteration of the direct solver ---*/
292 
293  DirectRun(kind_recording);
294 
295  /*--- Store the recording state ---*/
296 
297  RecordingState = kind_recording;
298 
299  /*--- Register Output of the iteration ---*/
300 
301  iteration->RegisterOutput(solver_container, geometry_container, config_container, ZONE_0, INST_0);
302 
303  /*--- Extract the objective function and store it --- */
304 
305  SetObjFunction();
306 
307  if (kind_recording != RECORDING::CLEAR_INDICES && config_container[ZONE_0]->GetWrt_AD_Statistics()) {
308  AD::PrintStatistics(SU2_MPI::GetComm(), rank == MASTER_NODE);
309  }
310 
311  AD::StopRecording();
312 
313 }

315 void CDiscAdjSinglezoneDriver::SetAdjObjFunction(){
316  su2double seeding = 1.0;
317 
318  if (config->GetTime_Domain()) {
319  const auto IterAvg_Obj = config->GetIter_Avg_Objective();
320  if (TimeIter < IterAvg_Obj) {
321  /*--- Default behavior when no window is chosen is to use Square-Windowing, i.e. the numerator equals 1.0 ---*/
322  auto windowEvaluator = CWindowingTools();
323  const su2double weight = windowEvaluator.GetWndWeight(config->GetKindWindow(), TimeIter, IterAvg_Obj - 1);
324  seeding = weight / IterAvg_Obj;
325  }
326  else {
327  seeding = 0.0;
328  }
329  }
330  if (rank == MASTER_NODE) {
331  SU2_TYPE::SetDerivative(ObjFunc, SU2_TYPE::GetValue(seeding));
332  } else {
333  SU2_TYPE::SetDerivative(ObjFunc, 0.0);
334  }
335 }

337 void CDiscAdjSinglezoneDriver::SetObjFunction(){
338 
339  ObjFunc = 0.0;
340 
341  /*--- Specific scalar objective functions ---*/
342 
343  switch (config->GetKind_Solver()) {
344  case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
345  case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
346  case MAIN_SOLVER::DISC_ADJ_FEM_EULER: case MAIN_SOLVER::DISC_ADJ_FEM_NS: case MAIN_SOLVER::DISC_ADJ_FEM_RANS:
347 
348  /*--- Surface based obj. function ---*/
349 
350  direct_output->SetHistoryOutput(geometry, solver, config, config->GetTimeIter(),
351  config->GetOuterIter(), config->GetInnerIter());
352  ObjFunc += solver[FLOW_SOL]->GetTotal_ComboObj();
353  break;
354 
355  case MAIN_SOLVER::DISC_ADJ_HEAT:
356  direct_output->SetHistoryOutput(geometry, solver, config, config->GetTimeIter(),
357  config->GetOuterIter(), config->GetInnerIter());
358  ObjFunc = solver[HEAT_SOL]->GetTotal_ComboObj();
359  break;
360 
361  case MAIN_SOLVER::DISC_ADJ_FEM:
362  solver[FEA_SOL]->Postprocessing(geometry, config, numerics_container[ZONE_0][INST_0][MESH_0][FEA_SOL], true);
363 
364  direct_output->SetHistoryOutput(geometry, solver, config, config->GetTimeIter(),
365  config->GetOuterIter(), config->GetInnerIter());
366  ObjFunc = solver[FEA_SOL]->GetTotal_ComboObj();
367  break;
368 
369  default:
370  break;
371  }
372 
373  if (rank == MASTER_NODE){
374  AD::RegisterOutput(ObjFunc);
375  }
