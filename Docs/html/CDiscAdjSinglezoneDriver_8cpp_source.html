36CDiscAdjSinglezoneDriver::CDiscAdjSinglezoneDriver(char* confFile,
37 unsigned short val_nZone,
38 SU2_Comm MPICommunicator) : CSinglezoneDriver(confFile,
39 val_nZone,
40 MPICommunicator) {
41
42
43 /*--- Store the number of internal iterations that will be run by the adjoint solver ---*/
44 nAdjoint_Iter = config_container[ZONE_0]->GetnInner_Iter();
45
46 /*--- Store the pointers ---*/
47 config = config_container[ZONE_0];
48 iteration = iteration_container[ZONE_0][INST_0];
49 solver = solver_container[ZONE_0][INST_0][MESH_0];
50 numerics = numerics_container[ZONE_0][INST_0][MESH_0];
51 geometry = geometry_container[ZONE_0][INST_0][MESH_0];
52 integration = integration_container[ZONE_0][INST_0];
53
54 /*--- Store the recording state ---*/
55 RecordingState = RECORDING::CLEAR_INDICES;
56
57 /*--- Initialize the direct iteration ---*/
58
59 switch (config->GetKind_Solver()) {
60
61 case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
62 case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
63 if (rank == MASTER_NODE)
64 cout << "Direct iteration: Euler/Navier-Stokes/RANS equation." << endl;
65
66 if (config->GetBoolTurbomachinery()) {
67 direct_iteration = new CTurboIteration(config);
68 }
69 else { direct_iteration = CIterationFactory::CreateIteration(MAIN_SOLVER::EULER, config); }
70
71 if (config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) {
72 direct_output = COutputFactory::CreateOutput(MAIN_SOLVER::EULER, config, nDim);
73 }
74 else { direct_output = COutputFactory::CreateOutput(MAIN_SOLVER::INC_EULER, config, nDim); }
75
76 MainVariables = RECORDING::SOLUTION_VARIABLES;
77 if (config->GetDeform_Mesh()) {
78 SecondaryVariables = RECORDING::MESH_DEFORM;
79 }
80 else { SecondaryVariables = RECORDING::MESH_COORDS; }
81 MainSolver = ADJFLOW_SOL;
82 break;
83
84 case MAIN_SOLVER::DISC_ADJ_FEM_EULER : case MAIN_SOLVER::DISC_ADJ_FEM_NS : case MAIN_SOLVER::DISC_ADJ_FEM_RANS :
85 if (rank == MASTER_NODE)
86 cout << "Direct iteration: Euler/Navier-Stokes/RANS equation." << endl;
87 direct_iteration = CIterationFactory::CreateIteration(MAIN_SOLVER::FEM_EULER, config);
88 direct_output = COutputFactory::CreateOutput(MAIN_SOLVER::FEM_EULER, config, nDim);
89 MainVariables = RECORDING::SOLUTION_VARIABLES;
90 SecondaryVariables = RECORDING::MESH_COORDS;
91 MainSolver = ADJFLOW_SOL;
92 break;
93
94 case MAIN_SOLVER::DISC_ADJ_FEM:
95 if (rank == MASTER_NODE)
96 cout << "Direct iteration: elasticity equation." << endl;
97 direct_iteration = CIterationFactory::CreateIteration(MAIN_SOLVER::FEM_ELASTICITY, config);
98 direct_output = COutputFactory::CreateOutput(MAIN_SOLVER::FEM_ELASTICITY, config, nDim);
99 MainVariables = RECORDING::SOLUTION_VARIABLES;
100 SecondaryVariables = RECORDING::MESH_COORDS;
101 MainSolver = ADJFEA_SOL;
102 break;
103
104 case MAIN_SOLVER::DISC_ADJ_HEAT:
105 if (rank == MASTER_NODE)
106 cout << "Direct iteration: heat equation." << endl;
107 direct_iteration = CIterationFactory::CreateIteration(MAIN_SOLVER::HEAT_EQUATION, config);
108 direct_output = COutputFactory::CreateOutput(MAIN_SOLVER::HEAT_EQUATION, config, nDim);
109 MainVariables = RECORDING::SOLUTION_VARIABLES;
110 SecondaryVariables = RECORDING::MESH_COORDS;
111 MainSolver = ADJHEAT_SOL;
112 break;
113
114 default:
115 break;
116
117 }
118
119 direct_output->PreprocessHistoryOutput(config, false);
120

130void CDiscAdjSinglezoneDriver::Preprocess(unsigned long TimeIter) {
131
132 /*--- Set the current time iteration in the config and also in the driver
133 * because the python interface doesn't offer an explicit way of doing it. ---*/
134
135 this->TimeIter = TimeIter;
136 config_container[ZONE_0]->SetTimeIter(TimeIter);
137
138 /*--- Preprocess the adjoint iteration ---*/
139
140 iteration->Preprocess(output_container[ZONE_0], integration_container, geometry_container,
141 solver_container, numerics_container, config_container,
142 surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
143
144 /*--- For the adjoint iteration we need the derivatives of the iteration function with
145 *--- respect to the conservative variables. Since these derivatives do not change in the steady state case
146 *--- we only have to record if the current recording is different from the main variables. ---*/
147
148 if (RecordingState != MainVariables){
149 MainRecording();
150 }

154void CDiscAdjSinglezoneDriver::Run() {
155
156 CQuasiNewtonInvLeastSquares<passivedouble> fixPtCorrector;
157 if (config->GetnQuasiNewtonSamples() > 1) {
158 fixPtCorrector.resize(config->GetnQuasiNewtonSamples(),
159 geometry_container[ZONE_0][INST_0][MESH_0]->GetnPoint(),
160 GetTotalNumberOfVariables(ZONE_0,true),
161 geometry_container[ZONE_0][INST_0][MESH_0]->GetnPointDomain());
162
163 if (TimeIter != 0) GetAllSolutions(ZONE_0, true, fixPtCorrector);
164 }
165
166 for (auto Adjoint_Iter = 0ul; Adjoint_Iter < nAdjoint_Iter; Adjoint_Iter++) {
167
168 /*--- Initialize the adjoint of the output variables of the iteration with the adjoint solution
169 *--- of the previous iteration. The values are passed to the AD tool.
170 *--- Issues with iteration number should be dealt with once the output structure is in place. ---*/
171
172 config->SetInnerIter(Adjoint_Iter);
173
174 iteration->InitializeAdjoint(solver_container, geometry_container, config_container, ZONE_0, INST_0);
175
176 /*--- Initialize the adjoint of the objective function with 1.0. ---*/
177
178 SetAdjObjFunction();
179
180 /*--- Interpret the stored information by calling the corresponding routine of the AD tool. ---*/
181
182 AD::ComputeAdjoint();
183
184 /*--- Extract the computed adjoint values of the input variables and store them for the next iteration. ---*/
185
186 iteration->IterateDiscAdj(geometry_container, solver_container,
187 config_container, ZONE_0, INST_0, false);
188
189 /*--- Monitor the pseudo-time ---*/
190
191 StopCalc = iteration->Monitor(output_container[ZONE_0], integration_container, geometry_container,
192 solver_container, numerics_container, config_container,
193 surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
194
195 /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
196
197 AD::ClearAdjoints();
198
199 /*--- Output files for steady state simulations. ---*/
200
201 if (!config->GetTime_Domain()) {
202 iteration->Output(output_container[ZONE_0], geometry_container, solver_container,
203 config_container, Adjoint_Iter, false, ZONE_0, INST_0);
204 }
205
206 if (StopCalc) break;
207
208 /*--- Correct the solution with the quasi-Newton approach. ---*/
209
210 if (fixPtCorrector.size()) {
211 GetAllSolutions(ZONE_0, true, fixPtCorrector.FPresult());
212 SetAllSolutions(ZONE_0, true, fixPtCorrector.compute());
213 }
214
215 }

219void CDiscAdjSinglezoneDriver::Postprocess() {
220
221 switch(config->GetKind_Solver())
222 {
223 case MAIN_SOLVER::DISC_ADJ_EULER : case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES : case MAIN_SOLVER::DISC_ADJ_RANS :
224 case MAIN_SOLVER::DISC_ADJ_INC_EULER : case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES : case MAIN_SOLVER::DISC_ADJ_INC_RANS :
225 case MAIN_SOLVER::DISC_ADJ_HEAT :
226
227 /*--- Compute the geometrical sensitivities ---*/
228 SecondaryRecording();
229 break;
230
231 case MAIN_SOLVER::DISC_ADJ_FEM :
232
233 /*--- Compute the geometrical sensitivities ---*/
234 SecondaryRecording();
235
236 iteration->Postprocess(output_container[ZONE_0], integration_container, geometry_container,
237 solver_container, numerics_container, config_container,
238 surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
239 break;
240
241 default:
242 break;
243
244 }//switch

248void CDiscAdjSinglezoneDriver::SetRecording(RECORDING kind_recording){
249
250 AD::Reset();
251
252 /*--- Prepare for recording by resetting the solution to the initial converged solution. ---*/
253
254 for (unsigned short iSol=0; iSol < MAX_SOLS; iSol++) {
255 for (unsigned short iMesh = 0; iMesh <= config_container[ZONE_0]->GetnMGLevels(); iMesh++) {
256 auto solver = solver_container[ZONE_0][INST_0][iMesh][iSol];
257 if (solver && solver->GetAdjoint()) {
258 SU2_OMP_PARALLEL_(if (solver->GetHasHybridParallel()))
259 solver->SetRecording(geometry_container[ZONE_0][INST_0][iMesh], config_container[ZONE_0]);
260 END_SU2_OMP_PARALLEL
261 }
262 }
263 }
264
265 if (rank == MASTER_NODE) {
266 cout << "\n-------------------------------------------------------------------------\n";
267 switch(kind_recording) {
268 case RECORDING::CLEAR_INDICES: cout << "Clearing the computational graph." << endl; break;
269 case RECORDING::MESH_COORDS: cout << "Storing computational graph wrt MESH COORDINATES." << endl; break;
270 case RECORDING::SOLUTION_VARIABLES:
271 cout << "Direct iteration to store the primal computational graph." << endl;
272 cout << "Computing residuals to check the convergence of the direct problem." << endl; break;
273 default: break;
274 }
275 }
276
277 /*---Enable recording and register input of the iteration --- */
278
279 if (kind_recording != RECORDING::CLEAR_INDICES){
280
281 AD::StartRecording();
282
283 iteration->RegisterInput(solver_container, geometry_container, config_container, ZONE_0, INST_0, kind_recording);
284 }
285
286 /*--- Set the dependencies of the iteration ---*/
287
288 iteration->SetDependencies(solver_container, geometry_container, numerics_container, config_container, ZONE_0,
289 INST_0, kind_recording);
290
291 /*--- Do one iteration of the direct solver ---*/
292
293 DirectRun(kind_recording);
294
295 /*--- Store the recording state ---*/
296
297 RecordingState = kind_recording;
298
299 /*--- Register Output of the iteration ---*/
300
301 iteration->RegisterOutput(solver_container, geometry_container, config_container, ZONE_0, INST_0);
302
303 /*--- Extract the objective function and store it --- */
304
305 SetObjFunction();
306
307 if (kind_recording != RECORDING::CLEAR_INDICES && config_container[ZONE_0]->GetWrt_AD_Statistics()) {
308 if (rank == MASTER_NODE) AD::PrintStatistics();
309#ifdef CODI_REVERSE_TYPE
310 if (size > SINGLE_NODE) {
311 su2double myMem = AD::getTape().getTapeValues().getUsedMemorySize(), totMem = 0.0;
312 SU2_MPI::Allreduce(&myMem, &totMem, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
313 if (rank == MASTER_NODE) {
314 cout << "MPI\n";
315 cout << "-------------------------------------\n";
316 cout << " Total memory used : " << totMem << " MB\n";
317 cout << "-------------------------------------\n" << endl;
318 }
319 }
320#endif
321 }
322
323 AD::StopRecording();
324

327void CDiscAdjSinglezoneDriver::SetAdjObjFunction(){
328 su2double seeding = 1.0;
329
330 if (config->GetTime_Domain()) {
331 const auto IterAvg_Obj = config->GetIter_Avg_Objective();
332 if (TimeIter < IterAvg_Obj) {
333 /*--- Default behavior when no window is chosen is to use Square-Windowing, i.e. the numerator equals 1.0 ---*/
334 auto windowEvaluator = CWindowingTools();
335 const su2double weight = windowEvaluator.GetWndWeight(config->GetKindWindow(), TimeIter, IterAvg_Obj - 1);
336 seeding = weight / IterAvg_Obj;
337 }
338 else {
339 seeding = 0.0;
340 }
341 }
342 if (rank == MASTER_NODE) {
343 SU2_TYPE::SetDerivative(ObjFunc, SU2_TYPE::GetValue(seeding));
344 } else {
345 SU2_TYPE::SetDerivative(ObjFunc, 0.0);
346 }
347}

349void CDiscAdjSinglezoneDriver::SetObjFunction(){
350
351 ObjFunc = 0.0;
352
353 /*--- Specific scalar objective functions ---*/
354
355 switch (config->GetKind_Solver()) {
356 case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
357 case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
358 case MAIN_SOLVER::DISC_ADJ_FEM_EULER: case MAIN_SOLVER::DISC_ADJ_FEM_NS: case MAIN_SOLVER::DISC_ADJ_FEM_RANS:
359
360 /*--- Surface based obj. function ---*/
361
362 direct_output->SetHistoryOutput(geometry, solver, config, config->GetTimeIter(),
363 config->GetOuterIter(), config->GetInnerIter());
364 ObjFunc += solver[FLOW_SOL]->GetTotal_ComboObj();
365 break;
366
367 case MAIN_SOLVER::DISC_ADJ_HEAT:
368 direct_output->SetHistoryOutput(geometry, solver, config, config->GetTimeIter(),
369 config->GetOuterIter(), config->GetInnerIter());
370 ObjFunc = solver[HEAT_SOL]->GetTotal_ComboObj();
371 break;
372
373 case MAIN_SOLVER::DISC_ADJ_FEM:
374 solver[FEA_SOL]->Postprocessing(geometry, config, numerics_container[ZONE_0][INST_0][MESH_0][FEA_SOL], true);
375
376 direct_output->SetHistoryOutput(geometry, solver, config, config->GetTimeIter(),
377 config->GetOuterIter(), config->GetInnerIter());
378 ObjFunc = solver[FEA_SOL]->GetTotal_ComboObj();
379 break;
380
381 default:
382 break;
383 }
384
385 if (rank == MASTER_NODE){
386 AD::RegisterOutput(ObjFunc);
387 }

391void CDiscAdjSinglezoneDriver::DirectRun(RECORDING kind_recording){
392
393 /*--- Mesh movement ---*/
394
395 direct_iteration->SetMesh_Deformation(geometry_container[ZONE_0][INST_0], solver, numerics, config, kind_recording);
396
397 /*--- Zone preprocessing ---*/
398
399 direct_iteration->Preprocess(direct_output, integration_container, geometry_container, solver_container, numerics_container, config_container, surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
400
401 /*--- Iterate the direct solver ---*/
402
403 direct_iteration->Iterate(direct_output, integration_container, geometry_container, solver_container, numerics_container, config_container, surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
404
405 /*--- Postprocess the direct solver ---*/
406
407 direct_iteration->Postprocess(direct_output, integration_container, geometry_container, solver_container, numerics_container, config_container, surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
408
409 /*--- Print the direct residual to screen ---*/
410
411 PrintDirectResidual(kind_recording);

427void CDiscAdjSinglezoneDriver::SecondaryRecording(){
428 /*--- SetRecording stores the computational graph on one iteration of the direct problem. Calling it with
429 * RECORDING::CLEAR_INDICES as argument ensures that all information from a previous recording is removed. ---*/
430
431 SetRecording(RECORDING::CLEAR_INDICES);
432
433 /*--- Store the computational graph of one direct iteration with the secondary variables as input. ---*/
434
435 SetRecording(SecondaryVariables);
436
437 /*--- Initialize the adjoint of the output variables of the iteration with the adjoint solution
438 * of the current iteration. The values are passed to the AD tool. ---*/
439

441
442 /*--- Initialize the adjoint of the objective function with 1.0. ---*/
443
444 SetAdjObjFunction();
445
446 /*--- Interpret the stored information by calling the corresponding routine of the AD tool. ---*/
447
448 AD::ComputeAdjoint();
449
450 /*--- Extract the computed sensitivity values. ---*/
451
452 if (SecondaryVariables == RECORDING::MESH_COORDS) {
453 solver[MainSolver]->SetSensitivity(geometry, config);
454 }
455 else { // MESH_DEFORM
456 solver[ADJMESH_SOL]->SetSensitivity(geometry, config, solver[MainSolver]);
457 }
458
459 /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
460
461 AD::ClearAdjoints();
