67 /*!
68 * \brief Species transport specific steps in the ComputeResidual method
69 * \param[in] config - Definition of the particular problem.
70 */
71 void FinishResidualCalc(const CConfig* config) override {
72 for (auto iVar = 0u; iVar < nVar; iVar++) {
73
74 const su2double Diffusivity_Lam = 0.5 * (Density_i * Diffusion_Coeff_i[iVar] + Density_j * Diffusion_Coeff_j[iVar]);
75
76 su2double Diffusivity_Turb = 0.0;
77
78 if (turbulence) {
79 const su2double Sc_t = config->GetSchmidt_Number_Turbulent();
80 Diffusivity_Turb = 0.5 * (Eddy_Viscosity_i / Sc_t + Eddy_Viscosity_j / Sc_t);
81 }
82
83 const su2double Diffusivity = Diffusivity_Lam + Diffusivity_Turb;
84
85 Flux[iVar] = Diffusivity * Proj_Mean_GradScalarVar[iVar];
86
87 /*--- Use TSL approx. to compute derivatives of the gradients. ---*/
88
89 /*--- Off-diagonal entries are all zero. ---*/
90 const su2double proj_on_rhoi = proj_vector_ij / Density_i;
91 Jacobian_i[iVar][iVar] = -Diffusivity * proj_on_rhoi;
92
93 const su2double proj_on_rhoj = proj_vector_ij / Density_j;
94 Jacobian_j[iVar][iVar] = Diffusivity * proj_on_rhoj;
95
96 } // iVar
97 }
