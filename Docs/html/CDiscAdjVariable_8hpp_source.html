32/*!
33 * \class CDiscAdjVariable
34 * \ingroup DiscAdj
35 * \brief Main class for defining the variables of the adjoint solver.
36 * \author T. Albring.
37 */
38class CDiscAdjVariable : public CVariable {
39private:
40 MatrixType Sensitivity; /*!< \brief Vector holding the derivative of target functional with respect to the coordinates at this node. */
41 MatrixType Solution_Direct; /*!< \brief Stores the primal solution of the current timestep in order to be able to reset. */
42 MatrixType DualTime_Derivative; /*!< \brief Container holding all/sum-of dual time contributions to the adjoint variable. */
43 MatrixType DualTime_Derivative_n; /*!< \brief Container holding dual time contributions to the adjoint variable used in the next timestep. */
44
45public:
46 /*!
47 * \brief Constructor of the class.
48 * \param[in] sol - Pointer to the adjoint value (initialization value).
49 * \param[in] npoint - Number of points/nodes/vertices in the domain.
50 * \param[in] ndim - Number of dimensions of the problem.
51 * \param[in] nvar - Number of variables of the problem.
52 * \param[in] config - Definition of the particular problem.
53 */
54 CDiscAdjVariable(const su2double* sol, unsigned long npoint, unsigned long ndim, unsigned long nvar, CConfig *config);
55
56 /*!
57 * \brief Allocate extra adjoint variables.
58 * \param[in] nVarExtra - Number of extra variables.
59 */
60 void AllocateAdjointSolutionExtra(unsigned long nVarExtra);
61
62 /*!
63 * \brief Set the sensitivity at the node
64 */
65 inline void SetSensitivity(unsigned long iPoint, unsigned long iDim, su2double val) final {
66 Sensitivity(iPoint,iDim) = val;
67 }
68
69 /*!
70 * \brief Get the Sensitivity at the node
71 */
72 inline su2double GetSensitivity(unsigned long iPoint, unsigned long iDim) const final {
73 return Sensitivity(iPoint,iDim);
74 }
75 inline const MatrixType& GetSensitivity() const final { return Sensitivity; }
76
77 /*!
78 * \brief Set/store the dual time contributions to the adjoint variable.
79 * Contains sum of contributions from 2 timesteps for dual time 2nd order.
80 */
81 inline void SetDual_Time_Derivative(unsigned long iPoint, unsigned long iVar, su2double der) {
82 DualTime_Derivative(iPoint,iVar) = der;
83 }
84
85 /*!
86 * \brief Set/store the dual time contributions to the adjoint variable for upcoming timestep.
87 */
88 inline void SetDual_Time_Derivative_n(unsigned long iPoint, unsigned long iVar, su2double der) {
89 DualTime_Derivative_n(iPoint,iVar) = der;
90 }
91
92 /*!
93 * \brief Return the dual time contributions to the adjoint variable.
94 * Contains sum of contributions from 2 timesteps for dual time 2nd order.
95 */
96 inline su2double GetDual_Time_Derivative(unsigned long iPoint, unsigned long iVar) const {
97 return DualTime_Derivative(iPoint,iVar);
98 }
99
100 /*!
101 * \brief Return the dual time contributions to the adjoint variable for upcoming timestep.
102 */
103 inline su2double GetDual_Time_Derivative_n(unsigned long iPoint, unsigned long iVar) const {
104 return DualTime_Derivative_n(iPoint,iVar);
105 }
106
107 /*!
108 * \brief Set/store the primal solution for all variables of one point.
109 */
110 inline void SetSolution_Direct(unsigned long iPoint, const su2double *val_solution_direct) final {
111 for (unsigned long iVar = 0; iVar < nVar; iVar++)
112 Solution_Direct(iPoint,iVar) = val_solution_direct[iVar];
113 }
114
115 /*!
116 * \brief Returns the primal solution for all variables of one point.
117 */
118 inline su2double* GetSolution_Direct(unsigned long iPoint) final {
119 return Solution_Direct[iPoint];
120 }
121
122 /*!
123 * \brief Set Dual-time derivative contributions to the external.
124 */
125 void Set_External_To_DualTimeDer() final;
126
127};
