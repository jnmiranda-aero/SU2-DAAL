191 /*!
192 * \brief Set the center of gravity of an element (including edges).
193 * \param[in] nDim - Number of dimensions (2 or 3).
194 * \param[in] val_coord - Coordinates of the element.
195 */
196 template <class T>
197 inline su2double* SetCoord_CG(unsigned short nDim, const T& val_coord) {
198 for (unsigned short iDim = 0; iDim < nDim; iDim++) {
199 Coord_CG[iDim] = 0.0;
200 for (unsigned short iNode = 0; iNode < GetnNodes(); iNode++)
201 Coord_CG[iDim] += val_coord[iNode][iDim] / su2double(GetnNodes());
202 }
203 return Coord_CG;
204 }

446/*! \class CPrimalGridWithConnectivity
447 * \brief Override the connectivity getters of CPrimalGrid.
448 *
449 * \details Non-FEM primal grid classes like CLine, CTriangle etc
450 * are derived from CPrimalGridWithConnectivity<CLineConnectivity> etc.
451 * The Connectivity class must have the following static members of
452 * type unsigned short / unsigned short array, typically constexpr:
453 * - nNodes: Number of nodes of the element.
454 * - nFaces: Number of faces of the element.
455 * - nNodesFace[]: Number of nodes of each face of the element.
456 * - maxNodesFace: Maximum number of nodes for a face.
457 * - Faces[][]: Matrix to store the local nodes of all the faces.
458 * - nNeighbor_Nodes[]: Number of neighbor nodes of each node of the element.
459 * - Neighbor_Nodes[][]: Matrix to store the neighbors of all the nodes.
460 * - VTK_Type: Type of element using VTK nomenclature.
461 *
462 * The getter functions for connectivity in CPrimalGrid have final overrides
463 * in this class, accessing the static members of Connectivity.
464 *
465 * \tparam Connectivity - class defining the connectivity structure
466 */
467template <typename Connectivity>
468class CPrimalGridWithConnectivity : public CPrimalGrid {
469 public:
470 CPrimalGridWithConnectivity(bool FEM) : CPrimalGrid(FEM, Connectivity::nNodes, Connectivity::nFaces) {}
471
472 inline unsigned short GetnNodes() const final { return Connectivity::nNodes; }
473
474 inline unsigned short GetnFaces() const final { return Connectivity::nFaces; }
475
476 inline unsigned short GetnNodesFace(unsigned short val_face) const final {
477 assert(val_face < Connectivity::nFaces);
478 return Connectivity::nNodesFace[val_face];
479 }
480
481 inline unsigned short GetMaxNodesFace() const final { return Connectivity::maxNodesFace; }
482
483 inline unsigned short GetFaces(unsigned short val_face, unsigned short val_index) const final {
484 assert(val_face < GetnFaces() && val_index < GetnNodesFace(val_face));
485 return Connectivity::Faces[val_face][val_index];
486 }
487
488 inline unsigned short GetnNeighbor_Nodes(unsigned short val_node) const final {
489 assert(val_node < Connectivity::nNodes);
490 return Connectivity::nNeighbor_Nodes[val_node];
491 }
492
493 inline unsigned short GetNeighbor_Nodes(unsigned short val_node, unsigned short val_index) const final {
494 assert(val_node < Connectivity::nNodes && val_index < GetnNeighbor_Nodes(val_node));
495 return Connectivity::Neighbor_Nodes[val_node][val_index];
496 }
497
498 inline unsigned short GetVTK_Type() const final { return Connectivity::VTK_Type; }
499};
