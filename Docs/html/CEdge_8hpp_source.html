39class CEdge {
40 static_assert(su2activematrix::IsRowMajor, "Needed to return normal as pointer.");
41
42 private:
43 using Index = unsigned long;
44 using NodeArray = C2DContainer<Index, Index, StorageType::ColumnMajor, 64, DynamicSize, 2>;
45 NodeArray Nodes; /*!< \brief Vector to store the node indices of the edge. */
46 su2activematrix Normal; /*!< \brief Normal (area) of the edge. */
47 const Index nEdge, nEdgeSIMD;

49 friend class CPhysicalGeometry;

51 public:
52 enum NodePosition : unsigned long { LEFT = 0, RIGHT = 1 };

55 * \brief Constructor of the class.

66 /*!
67 * \brief Get left/right node index defining the edge.
68 * \param[in] iEdge - Edge index.
69 * \param[in] iNode - Node index 0 or 1, LEFT or RIGHT.
70 * \return Index of the node that composes the edge.
71 */
72 inline unsigned long GetNode(unsigned long iEdge, unsigned long iNode) const { return Nodes(iEdge, iNode); }
73
74 /*!
75 * \brief SIMD version of GetNode, iNode returned for contiguous iEdges.
76 */
77 template <class T, size_t N>
78 FORCEINLINE simd::Array<T, N> GetNode(simd::Array<T, N> iEdge, unsigned long iNode) const {
79 return simd::Array<T, N>(&Nodes(iEdge[0], iNode));
80 }
81
82 /*!
83 * \brief Sets the tail of "Nodes" to repeat one of the last edges.
84 * \note This is needed when using the SIMD version of GetNode and
85 * the number of edges is not a multiple of the simd width.
86 */
87 void SetPaddingNodes() {
88 for (auto iEdge = nEdge; iEdge < nEdgeSIMD; ++iEdge) {
89 /*--- Pad nodes by repeating the first edge in the last SIMD group. ---*/
90 const auto iEdge0 = nEdgeSIMD - simd::preferredLen<su2double>();
91 Nodes(iEdge, LEFT) = Nodes(iEdge0, LEFT);
92 Nodes(iEdge, RIGHT) = Nodes(iEdge0, RIGHT);
93 }
94 }
95
96 /*!
97 * \brief Set the node indices of an edge.
98 * \param[in] iEdge - Edge index.
99 * \param[in] iPoint - Index of left node.
100 * \param[in] jPoint - Index of right node.
101 */
102 inline void SetNodes(unsigned long iEdge, unsigned long iPoint, unsigned long jPoint) {
103 Nodes(iEdge, LEFT) = iPoint;
104 Nodes(iEdge, RIGHT) = jPoint;
105 }
106
107 /*!
108 * \brief Get the number of nodes of an edge (2).

110 inline unsigned long GetnNodes() const { return 2; }

155 /*!
156 * \brief Copy the the normal vector of a face.
157 * \param[in] iEdge - Edge index.
158 * \param[out] normal - Object into which the normal (dimensional) will be copied.
159 */
160 template <class T>
161 inline void GetNormal(unsigned long iEdge, T& normal) const {
162 for (auto iDim = 0ul; iDim < Normal.cols(); iDim++) normal[iDim] = Normal(iEdge, iDim);
163 }
164
165 /*!
166 * \brief Get the normal to a face of the control volume asociated with an edge.
167 * \param[in] iEdge - Edge index.
168 * \return Dimensional normal vector, the modulus is the area of the face.
169 */
170 inline const su2double* GetNormal(unsigned long iEdge) const { return Normal[iEdge]; }
171
172 /*!
173 * \brief Get the entire matrix of edge normals.
174 */
175 inline const su2activematrix& GetNormal() const { return Normal; }

177 /*!
178 * \brief Initialize normal vector to 0.

180 void SetZeroValues(void);

182 /*!
183 * \brief Set the normal vector of an edge.
184 * \param[in] iEdge - Edge index.
185 * \param[in] normal - Vector to initialize the normal vector.
186 * \return Value of the normal vector.
187 */
188 template <class T>
189 void SetNormal(unsigned long iEdge, const T& normal) {
190 for (auto iDim = 0ul; iDim < Normal.cols(); ++iDim) Normal(iEdge, iDim) = normal[iDim];
191 }
192
193 /*!
194 * \brief Add a vector to the normal vector of an edge.
195 * \param[in] iEdge - Edge index.
196 * \param[in] normal - Vector to add to the normal vector.
197 */
198 template <class T>
199 void AddNormal(unsigned long iEdge, const T& normal) {
200 for (auto iDim = 0ul; iDim < Normal.cols(); ++iDim) Normal(iEdge, iDim) += normal[iDim];
201 }
202
203 /*!
204 * \brief Subtract a vector to the normal vector of an edge.
205 * \param[in] iEdge - Edge index.
206 * \param[in] normal - Vector to add to the normal vector.
207 */
208 template <class T>
209 void SubNormal(unsigned long iEdge, const T& normal) {
210 for (auto iDim = 0ul; iDim < Normal.cols(); ++iDim) Normal(iEdge, iDim) -= normal[iDim];
211 }
