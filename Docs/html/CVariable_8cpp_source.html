31CVariable::CVariable(unsigned long npoint, unsigned long nvar, const CConfig *config) {
32
33 /*--- Initialize the number of solution variables. This version
34 of the constructor will be used primarily for converting the
35 restart files into solution files (SU2_SOL). ---*/
36 nPoint = npoint;
37 nVar = nvar;
38
39 /*--- Allocate the solution array. ---*/
40 Solution.resize(nPoint,nVar) = su2double(0.0);
41
42 if (config->GetMultizone_Problem())
43 Solution_BGS_k.resize(nPoint,nVar) = su2double(0.0);
44
45}

47CVariable::CVariable(unsigned long npoint, unsigned long ndim, unsigned long nvar,
48 const CConfig *config, bool adjoint) {
49
50 /*--- Initializate the number of dimension and number of variables ---*/
51 nPoint = npoint;
52 nDim = ndim;
53 nVar = nvar;
54
55 /*--- Allocate fields common to all problems. Do not allocate fields
56 that are specific to one solver, i.e. not common, in this class. ---*/
57 Solution.resize(nPoint,nVar) = su2double(0.0);
58
59 Solution_Old.resize(nPoint,nVar) = su2double(0.0);
60
61 if (config->GetTime_Domain())
62 Solution_time_n.resize(nPoint,nVar) = su2double(0.0);
63
64 if (config->GetTime_Marching() != TIME_MARCHING::STEADY)
65 Solution_time_n1.resize(nPoint,nVar) = su2double(0.0);
66
67 if (config->GetDiscrete_Adjoint()) {
68 if (adjoint && config->GetMultizone_Problem())
69 External.resize(nPoint,nVar) = su2double(0.0);
70
71 if (!adjoint) {
72 AD_InputIndex.resize(nPoint,nVar) = -1;
73 AD_OutputIndex.resize(nPoint,nVar) = -1;
74 }
75 }
76
77 if (config->GetMultizone_Problem())
78 Solution_BGS_k.resize(nPoint,nVar) = su2double(0.0);
79}
