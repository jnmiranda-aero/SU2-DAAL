50 /*!
51  * \class CCompressedSparsePattern
52  * \brief A simple class to store adjacency information in a
53  * compressed format suitable for sparse matrix operations.
54  * If built for row-major storage the inner indices are column indices
55  * and the pattern should be used as (row,icol), otherwise as (col,irow).
56  */
57 template <typename Index_t>
58 class CCompressedSparsePattern {

64  su2vector<Index_t> m_diagPtr; /*!< \brief Position of the diagonal entry. */
65  su2vector<Index_t> m_innerIdxTransp; /*!< \brief Position of the transpose non zero entries, requires symmetry. */

86  /*!
87  * \brief Construct with default inner indices.
88  * \param[in] outerPtrBegin - Start of outer pointers.
89  * \param[in] outerPtrEnd - End of outer pointers.
90  * \param[in] defaultInnerIdx - Default value for inner indices.
91  */
92  template <class Iterator>
93  CCompressedSparsePattern(Iterator outerPtrBegin, Iterator outerPtrEnd, Index_t defaultInnerIdx) {
94  const auto size = outerPtrEnd - outerPtrBegin;
95  m_outerPtr.resize(size);
96  Index_t k = 0;
97  for (auto it = outerPtrBegin; it != outerPtrEnd; ++it) m_outerPtr(k++) = *it;
98 
99  m_innerIdx.resize(m_outerPtr(size - 1)) = defaultInnerIdx;
100  }

102  /*!
103  * \brief Construct from rvalue refs.
104  * \note This is the most efficient constructor as no data copy occurs.
105  * \param[in] outerPtr - Outer index pointers.
106  * \param[in] innerIdx - Inner indices.
107  */
108  CCompressedSparsePattern(su2vector<Index_t>&& outerPtr, su2vector<Index_t>&& innerIdx)
109  : m_outerPtr(outerPtr), m_innerIdx(innerIdx) {
110  /*--- perform a basic sanity check ---*/
111  assert(m_innerIdx.size() == m_outerPtr(m_outerPtr.size() - 1));

114  /*!
115  * \brief Construct from vector-like objects of any type with
116  * methods "size()" and "data()" (returning a pointer).
117  * \param[in] outerPtr - Outer index pointers.
118  * \param[in] innerIdx - Inner indices.
119  */
120  template <class T>
121  CCompressedSparsePattern(const T& outerPtr, const T& innerIdx) {
122  m_outerPtr.resize(outerPtr.size());
123  for (Index_t i = 0; i < outerPtr.size(); ++i) m_outerPtr(i) = outerPtr.data()[i];
124 
125  m_innerIdx.resize(innerIdx.size());
126  for (Index_t i = 0; i < innerIdx.size(); ++i) m_innerIdx(i) = innerIdx.data()[i];
127 
128  /*--- perform a basic sanity check ---*/
129  assert(m_innerIdx.size() == m_outerPtr(m_outerPtr.size() - 1));
130  }

132  /*!
133  * \brief Build from a "list of lists" type object
134  * \param[in] lil - An object with operator [][] and method "size" e.g. vector<vector<?> >.
135  */
136  template <class T>
137  CCompressedSparsePattern(const T& lil) {
138  m_outerPtr.resize(lil.size() + 1);
139  m_outerPtr(0) = 0;
140  for (Index_t i = 1; i < Index_t(m_outerPtr.size()); ++i) m_outerPtr(i) = m_outerPtr(i - 1) + lil[i - 1].size();
141 
142  m_innerIdx.resize(m_outerPtr(lil.size()));
143  Index_t k = 0;
144  for (Index_t i = 0; i < Index_t(lil.size()); ++i)
145  for (Index_t j = 0; j < Index_t(lil[i].size()); ++j) m_innerIdx(k++) = lil[i][j];
146  }

148  /*!
149  * \brief Build a list of pointers to the diagonal entries of the pattern.
150  */
151  void buildDiagPtr() {
152  if (!m_diagPtr.empty()) return;
153 
154  m_diagPtr.resize(getOuterSize());
155 
156  SU2_OMP_PARALLEL_(for schedule(static,roundUpDiv(getOuterSize(),omp_get_max_threads())))
157  for (Index_t k = 0; k < getOuterSize(); ++k) m_diagPtr(k) = findInnerIdx(k, k);
158  END_SU2_OMP_PARALLEL
159  }
160 
161  /*!
162  * \brief Build a list of pointers to the transpose entries of the pattern, requires symmetry.
163  */
164  void buildTransposePtr() {
165  if (!m_innerIdxTransp.empty()) return;
166 
167  m_innerIdxTransp.resize(getNumNonZeros());
168 
169  SU2_OMP_PARALLEL_(for schedule(static,roundUpDiv(getOuterSize(),omp_get_max_threads())))
170  for (Index_t i = 0; i < getOuterSize(); ++i) {
171  for (Index_t k = m_outerPtr(i); k < m_outerPtr(i + 1); ++k) {
172  auto j = m_innerIdx(k);
173  m_innerIdxTransp(k) = findInnerIdx(j, i);
174  assert(m_innerIdxTransp(k) != m_innerIdx.size() && "The pattern is not symmetric.");
175  }
176  }
177  END_SU2_OMP_PARALLEL
178  }

229  /*!
230  * \param[in] iOuterIdx - Outer index (row/col).
231  * \param[in] iInnerIdx - Inner index (col/row).
232  * \return Absolute position of non zero entry (iOuterIdx,iInnerIdx),
233  * or NNZ if position does not belong to the pattern.
234  */
235  inline Index_t findInnerIdx(Index_t iOuterIdx, Index_t iInnerIdx) const {
236  for (Index_t k = m_outerPtr(iOuterIdx); k < m_outerPtr(iOuterIdx + 1); ++k)
237  if (m_innerIdx(k) == iInnerIdx) return k;
238  return m_innerIdx.size();
239  }

241  /*!
242  * \param[in] iOuterIdx - Outer index (row/col).
243  * \param[in] iInnerIdx - Inner index (col/row).
244  * \return True if (iOuterIdx,iInnerIdx) exists, i.e. is non zero.
245  */
246  inline bool isNonZero(Index_t iOuterIdx, Index_t iInnerIdx) const {
247  return findInnerIdx(iOuterIdx, iInnerIdx) < m_innerIdx.size();

250  /*!
251  * \param[in] iOuterIdx - Outer index (row/col).
252  * \param[in] iInnerIdx - Inner index (col/row).
253  * \return Absolute position of non zero entry (iOuterIdx,iInnerIdx).
254  * \note This method is only safe if the entry exists.
255  */
256  inline Index_t quickFindInnerIdx(Index_t iOuterIdx, Index_t iInnerIdx) const {
257  assert(isNonZero(iOuterIdx, iInnerIdx) && "Error, j does not belong to NZ(i).");
258  Index_t k = m_outerPtr(iOuterIdx);
259  while (m_innerIdx(k) != iInnerIdx) ++k;
260  return k;
261  }

340 /*!
341  * \brief Build a sparse pattern from geometry information, of type FVM or FEM,
342  * for a given fill-level. At fill-level N, the immediate neighbors of the
343  * points in level N-1 are also considered neighbors of the base point.
344  * The resulting pattern is that of A^{N+1} where A is the sparse matrix
345  * of immediate neighbors.
346  * \note Algorithm is equivalent to the implementation by F. Palacios,
347  * A. Bueno, and T. Economon from CSysMatrix.
348  * \param[in] geometry - Definition of the grid.
349  * \param[in] type - Of connectivity.
350  * \param[in] fillLvl - Target degree of neighborhood (immediate neighbors always added).
351  * \return Compressed-Storage-Row sparse pattern.
352  */
353 template <class Geometry_t, typename Index_t>
354 CCompressedSparsePattern<Index_t> buildCSRPattern(Geometry_t& geometry, ConnectivityType type, Index_t fillLvl) {
355  Index_t nPoint = geometry.GetnPoint();
356 
357  std::vector<Index_t> outerPtr(nPoint + 1);
358  std::vector<Index_t> innerIdx;
359  innerIdx.reserve(nPoint); // at least this much space is needed
360 
361  for (Index_t iPoint = 0; iPoint < nPoint; ++iPoint) {
362  /*--- Inner indices for iPoint start here. ---*/
363  outerPtr[iPoint] = innerIdx.size();
364 
365  /*--- Use a set to avoid duplication and keep ascending order. ---*/
366  std::set<Index_t> neighbors;
367 
368  /*--- Insert base point. ---*/
369  neighbors.insert(iPoint);
370 
371  /*--- Neighbors added in previous level. ---*/
372  std::set<Index_t> addedNeighbors(neighbors);
373 
374  for (Index_t iLevel = 0;; ++iLevel) {
375  /*--- New points added in this level. ---*/
376  std::set<Index_t> newNeighbors;
377 
378  /*--- For each point previously added, add its level 0
379  * neighbors, not duplicating any existing neighbor. ---*/
380  for (auto jPoint : addedNeighbors) {
381  if (type == ConnectivityType::FiniteVolume) {
382  /*--- For FVM we know the neighbors of point j directly. ---*/
383  for (Index_t kPoint : geometry.nodes->GetPoints(jPoint))
384  if (neighbors.count(kPoint) == 0) // no duplication
385  newNeighbors.insert(kPoint);
386  } else // FiniteElement
387  {
388  /*--- For FEM we need the nodes of all elements that contain point j. ---*/
389  for (auto iElem : geometry.nodes->GetElems(jPoint)) {
390  auto elem = geometry.elem[iElem];
391 
392  for (unsigned short iNode = 0; iNode < elem->GetnNodes(); ++iNode) {
393  Index_t kPoint = elem->GetNode(iNode);
394 
395  if (neighbors.count(kPoint) == 0) // no duplication
396  newNeighbors.insert(kPoint);
397  }
398  }
399  }
400  }
401 
402  neighbors.insert(newNeighbors.begin(), newNeighbors.end());
403 
404  if (iLevel >= fillLvl) break;
405 
406  /*--- For the next level we get the neighbours of the new points. ---*/
407  addedNeighbors = newNeighbors;
408  }
409 
410  /*--- Store final sparse pattern for iPoint. ---*/
411  innerIdx.insert(innerIdx.end(), neighbors.begin(), neighbors.end());
412  }
413  outerPtr.back() = innerIdx.size();
414 
415  /*--- Return pattern as CCompressedSparsePattern object. ---*/
416  return CCompressedSparsePattern<Index_t>(outerPtr, innerIdx);
417 }

419 /*!
420  * \brief Build a lookup table of the absolute positions of the non zero entries
421  * of a compressed sparse pattern, accessed when visiting the FVM edges
422  * of a grid. The table can then be used for fast access (avoids searches)
423  * to the non zero entries of a sparse matrix associated with the pattern.
424  * \param[in] geometry - Definition of the grid.
425  * \param[in] pattern - Sparse pattern.
426  * \return nEdge by 2 matrix.
427  */
428 template <class Geometry_t, typename Index_t>
429 CEdgeToNonZeroMap<Index_t> mapEdgesToSparsePattern(Geometry_t& geometry,
430  const CCompressedSparsePattern<Index_t>& pattern) {
431  assert(!pattern.empty());
432 
433  CEdgeToNonZeroMap<Index_t> edgeMap(geometry.GetnEdge(), 2);
434 
435  for (Index_t iEdge = 0; iEdge < geometry.GetnEdge(); ++iEdge) {
436  Index_t iPoint = geometry.edges->GetNode(iEdge, 0);
437  Index_t jPoint = geometry.edges->GetNode(iEdge, 1);
438 
439  edgeMap(iEdge, 0) = pattern.quickFindInnerIdx(iPoint, jPoint);
440  edgeMap(iEdge, 1) = pattern.quickFindInnerIdx(jPoint, iPoint);
441  }
442 
443  return edgeMap;
444 }

446 /*!
447  * \brief Create the natural coloring (equivalent to the normal sequential loop
448  * order) for a given number of inner indexes.
449  * \note This is to reduce overhead in "OpenMP-ready" code when only 1 thread is used.
450  * \param[in] numInnerIndexes - Number of indexes that are to be colored.
451  * \return Natural (sequential) coloring of the inner indices.
452  */
453 template <class T = CCompressedSparsePatternUL, class Index_t = typename T::IndexType>
454 T createNaturalColoring(Index_t numInnerIndexes) {
455  /*--- One color. ---*/
456  su2vector<Index_t> outerPtr(2);
457  outerPtr(0) = 0;
458  outerPtr(1) = numInnerIndexes;
459 
460  /*--- Containing all indexes in ascending order. ---*/
461  su2vector<Index_t> innerIdx(numInnerIndexes);
462  std::iota(innerIdx.data(), innerIdx.data() + numInnerIndexes, 0);
463 
464  return T(std::move(outerPtr), std::move(innerIdx));

467 /*!
468  * \brief Color contiguous groups of outer indices of a sparse pattern such that
469  * within each color, any two groups do not have inner indices in common.
470  * \note Within a group, two outer indices will generally have common inner indices.
471  * The coloring is returned as a compressed sparse pattern where the colors
472  * are outer indices, and the outer indices of the input pattern are the
473  * inner indices of the coloring. A simple greedy algorithm is used.
474  * Using a sparse pattern as input allows "anything" to be colored e.g.
475  * FVM edges, FEM elements, the rows/columns of a sparse matrix, etc.
476  * \note The worst that can happen in this method is needing an unreasonable number
477  * of colors, or too much memory due to a large range of the inner indices.
478  * The last two template parameters limit both, in case of failure an empty
479  * pattern is returned.
480  * \param[in] pattern - Sparse pattern to be colored.
481  * \param[in] groupSize - Size of the outer index groups, default 1.
482  * \param[in] balanceColors - Try to balance number of indexes per color,
483  * tends to result in worse locality (thus false by default).
484  * \param[out] indexColor - Optional, vector with colors given to the outer indices.
485  * \return Coloring in the same type of the input pattern.
486  */
487 template <typename Color_t = unsigned char, size_t MaxColors = 255, size_t MaxMB = 128, class T>
488 T colorSparsePattern(const T& pattern, size_t groupSize = 1, bool balanceColors = false,
489  std::vector<Color_t>* indexColor = nullptr) {
490  static_assert(std::is_integral<Color_t>::value, "");
491  static_assert(std::numeric_limits<Color_t>::max() >= MaxColors, "");
492 
493  using Index_t = typename T::IndexType;
494 
495  const Index_t grpSz = groupSize;
496  const Index_t nOuter = pattern.getOuterSize();
497 
498  /*--- Trivial case. ---*/
499  if (groupSize >= nOuter) return createNaturalColoring(nOuter);
500 
501  const Index_t minIdx = pattern.getMinInnerIdx();
502  const Index_t nInner = pattern.getMaxInnerIdx() + 1 - minIdx;
503 
504  /*--- Check the max memory condition (<< 23 is to count bits). ---*/
505  if (size_t(nInner) > (MaxMB << 23)) return T();
506 
507  /*--- Vector with the color given to each outer index. ---*/
508  std::vector<Color_t> idxColor(nOuter);
509 
510  /*--- Start with one color, with no indices assigned. ---*/
511  std::vector<Index_t> colorSize(1, 0);
512  Color_t nColor = 1;
513 
514  {
515  /*--- For each color keep track of the inner indices that are in it. ---*/
516  std::vector<std::vector<bool> > innerInColor;
517  innerInColor.emplace_back(nInner, false);
518 
519  /*--- Order in which we look for space in the colors to insert a new group. ---*/
520  std::vector<Color_t> searchOrder(MaxColors);
521 
522  auto outerPtr = pattern.outerPtr();
523  auto innerIdx = pattern.innerIdx();
524 
525  for (Index_t iOuter = 0; iOuter < nOuter; iOuter += grpSz) {
526  Index_t grpEnd = std::min(iOuter + grpSz, nOuter);
527 
528  searchOrder.resize(nColor);
529  std::iota(searchOrder.begin(), searchOrder.end(), 0);
530 
531  /*--- Balance sizes by looking for space in smaller colors first. ---*/
532  if (balanceColors) {
533  std::sort(searchOrder.begin(), searchOrder.end(),
534  [&colorSize](Color_t a, Color_t b) { return colorSize[a] < colorSize[b]; });
535  }
536 
537  auto it = searchOrder.begin();
538 
539  for (; it != searchOrder.end(); ++it) {
540  bool free = true;
541  /*--- Traverse entire group as a large outer index. ---*/
542  for (Index_t k = outerPtr[iOuter]; k < outerPtr[grpEnd] && free; ++k) {
543  free = !innerInColor[*it][innerIdx[k] - minIdx];
544  }
545  /*--- If none of the inner indices in the group appears in
546  * this color yet, it is assigned to the group. ---*/
547  if (free) break;
548  }
549 
550  Color_t color;
551 
552  if (it != searchOrder.end()) {
553  /*--- Found a free color. ---*/
554  color = *it;
555  } else {
556  /*--- No color was free, make space for a new one. ---*/
557  color = nColor++;
558  if (nColor == MaxColors) return T();
559  colorSize.push_back(0);
560  innerInColor.emplace_back(nInner, false);
561  }
562 
563  /*--- Assign color to group. ---*/
564  for (Index_t k = iOuter; k < grpEnd; ++k) idxColor[k] = color;
565 
566  /*--- Mark the inner indices of the group as belonging to the color. ---*/
567  for (Index_t k = outerPtr[iOuter]; k < outerPtr[grpEnd]; ++k) {
568  innerInColor[color][innerIdx[k] - minIdx] = true;
569  }
570 
571  /*--- Update count for the assigned color. ---*/
572  colorSize[color] += grpEnd - iOuter;
573  }
574  } // matrix of bools goes out of scope
575 
576  /*--- Compress the coloring information. ---*/
577 
578  su2vector<Index_t> colorPtr(nColor + 1);
579  colorPtr(0) = 0;
580  su2vector<Index_t> outerIdx(nOuter);
581

649 /*!
650  * \brief Computes the efficiency of a grid coloring for given number of threads and chunk size.
651  */
652 template <class SparsePattern>
653 su2double coloringEfficiency(const SparsePattern& coloring, int numThreads, int chunkSize) {
654  using Index_t = typename SparsePattern::IndexType;
655 
656  /*--- Ideally compute time is proportional to total work over number of threads. ---*/
657  su2double ideal = coloring.getNumNonZeros() / su2double(numThreads);
658 
659  /*--- In practice the total work is quantized first by colors and then by chunks. ---*/
660  Index_t real = 0;
661  for (Index_t color = 0; color < coloring.getOuterSize(); ++color)
662  real += chunkSize * roundUpDiv(roundUpDiv(coloring.getNumNonZeros(color), chunkSize), numThreads);
663 
664  return ideal / real;
665 }
