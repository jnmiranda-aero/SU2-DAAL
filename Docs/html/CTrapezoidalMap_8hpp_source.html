88 * \param[in] val_y - y-coordinate or first independent variable

93
94 /*!
95 * \brief determine if the x-coordinate falls within the bounds xmin,xmax of the table
96 * \param[in] val_x - x-coordinate or first independent variable
97 * \param[out] bool - true if val_x is within (xmin,xmax)
98 */
99 inline bool IsInsideHullX(su2double val_x) {
100 return (val_x >= unique_bands_x.front()) && (val_x <= unique_bands_x.back());
101 }
102
103 /*!

107 double GetMemoryFootprint() const { return memory_footprint; }
