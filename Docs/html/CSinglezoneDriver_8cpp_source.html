46 void CSinglezoneDriver::StartSolver() {
47 
48  StartTime = SU2_MPI::Wtime();
49 
50  config_container[ZONE_0]->Set_StartTime(StartTime);
51 
52  /*--- Main external loop of the solver. Runs for the number of time steps required. ---*/
53 
54  if (rank == MASTER_NODE)
55  cout << endl <<"------------------------------ Begin Solver -----------------------------" << endl;
56 
57  if (rank == MASTER_NODE){
58  cout << endl <<"Simulation Run using the Single-zone Driver" << endl;
59  if (driver_config->GetTime_Domain())
60  cout << "The simulation will run for "
61  << driver_config->GetnTime_Iter() - config_container[ZONE_0]->GetRestart_Iter() << " time steps." << endl;
62  }
63 
64  /*--- Set the initial time iteration to the restart iteration. ---*/
65  if (config_container[ZONE_0]->GetRestart() && driver_config->GetTime_Domain())
66  TimeIter = config_container[ZONE_0]->GetRestart_Iter();
67 
68  /*--- Run the problem until the number of time iterations required is reached. ---*/
69  /*--- or until a SIGTERM signal stops the loop. We catch SIGTERM and exit gracefully ---*/
70  while ( TimeIter < config_container[ZONE_0]->GetnTime_Iter()) {
71  /*--- Perform some preprocessing before starting the time-step simulation. ---*/
72 
73  Preprocess(TimeIter);
74 
75  /*--- Run a time-step iteration of the single-zone problem. ---*/
76 
77  Run();
78 
79  /*--- Perform some postprocessing on the solution before the update ---*/
80 
81  Postprocess();
82 
83  /*--- Update the solution for dual time stepping strategy ---*/
84 
85  Update();
86 
87  /*--- Monitor the computations after each iteration. ---*/
88 
89  Monitor(TimeIter);
90 
91  /*--- Output the solution in files. ---*/
92 
93  Output(TimeIter);
94 
95  /*--- If the convergence criteria has been met, terminate the simulation. ---*/
96 
97  if (StopCalc) break;
98 
99  TimeIter++;
100 
101  }
102 
103 }

105 void CSinglezoneDriver::Preprocess(unsigned long TimeIter) {
106 
107  /*--- Set the current time iteration in the config and also in the driver
108  * because the python interface doesn't offer an explicit way of doing it. ---*/
109 
110  this->TimeIter = TimeIter;
111  config_container[ZONE_0]->SetTimeIter(TimeIter);
112 
113  /*--- Store the current physical time in the config container, as
114  this can be used for verification / MMS. This should also be more
115  general once the drivers are more stable. ---*/
116 
117  if (config_container[ZONE_0]->GetTime_Marching() != TIME_MARCHING::STEADY)
118  config_container[ZONE_0]->SetPhysicalTime(static_cast<su2double>(TimeIter)*config_container[ZONE_0]->GetDelta_UnstTimeND());
119  else
120  config_container[ZONE_0]->SetPhysicalTime(0.0);
121 
122 
123  /*--- Set the initial condition for EULER/N-S/RANS ---------------------------------------------*/
124  if (config_container[ZONE_0]->GetFluidProblem()) {
125  solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->SetInitialCondition(geometry_container[ZONE_0][INST_0],
126  solver_container[ZONE_0][INST_0],
127  config_container[ZONE_0], TimeIter);
128  }
129  if (config_container[ZONE_0]->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
130  solver_container[ZONE_0][INST_0][MESH_0][SPECIES_SOL]->SetInitialCondition(geometry_container[ZONE_0][INST_0],
131  solver_container[ZONE_0][INST_0],
132  config_container[ZONE_0], TimeIter);
133  }
134  else if (config_container[ZONE_0]->GetHeatProblem()) {
135  /*--- Set the initial condition for HEAT equation ---------------------------------------------*/
136  solver_container[ZONE_0][INST_0][MESH_0][HEAT_SOL]->SetInitialCondition(geometry_container[ZONE_0][INST_0],
137  solver_container[ZONE_0][INST_0],
138  config_container[ZONE_0], TimeIter);
139  }
140 
141  SU2_MPI::Barrier(SU2_MPI::GetComm());
142 
143  /*--- Run a predictor step ---*/
144  if (config_container[ZONE_0]->GetPredictor())
145  iteration_container[ZONE_0][INST_0]->Predictor(output_container[ZONE_0], integration_container, geometry_container, solver_container,
146  numerics_container, config_container, surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
147 
148  /*--- Perform a dynamic mesh update if required. ---*/
149  /*--- For the Disc.Adj. of a case with (rigidly) moving grid, the appropriate
150  mesh cordinates are read from the restart files. ---*/
151  if (!(config_container[ZONE_0]->GetGrid_Movement() && config_container[ZONE_0]->GetDiscrete_Adjoint()))
152  DynamicMeshUpdate(TimeIter);
153 
154 }

156 void CSinglezoneDriver::Run() {
157 
158  unsigned long OuterIter = 0;
159  config_container[ZONE_0]->SetOuterIter(OuterIter);
160 
161  /*--- Iterate the zone as a block, either to convergence or to a max number of iterations ---*/
162  iteration_container[ZONE_0][INST_0]->Solve(output_container[ZONE_0], integration_container, geometry_container, solver_container,
163  numerics_container, config_container, surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);

207  wrote_files = true;

209

211 
212  StopTime = SU2_MPI::Wtime();

217 
218  StartTime = SU2_MPI::Wtime();
219  }
220 
221  config_container[ZONE_0]->Set_StartTime(StartTime);

224 void CSinglezoneDriver::DynamicMeshUpdate(unsigned long TimeIter) {
225 
226  auto iteration = iteration_container[ZONE_0][INST_0];
227 
228  /*--- Legacy dynamic mesh update - Only if GRID_MOVEMENT = YES ---*/
229  if (config_container[ZONE_0]->GetGrid_Movement()) {
230  iteration->SetGrid_Movement(geometry_container[ZONE_0][INST_0],surface_movement[ZONE_0],
231  grid_movement[ZONE_0][INST_0], solver_container[ZONE_0][INST_0],
232  config_container[ZONE_0], 0, TimeIter);
233  }
234 
235  /*--- New solver - all the other routines in SetGrid_Movement should be adapted to this one ---*/
236  /*--- Works if DEFORM_MESH = YES ---*/
237  iteration->SetMesh_Deformation(geometry_container[ZONE_0][INST_0],
238  solver_container[ZONE_0][INST_0][MESH_0],
239  numerics_container[ZONE_0][INST_0][MESH_0],
240  config_container[ZONE_0], RECORDING::CLEAR_INDICES);
241 
242  /*--- Update the wall distances if the mesh was deformed. ---*/
243  if (config_container[ZONE_0]->GetGrid_Movement() ||
244  config_container[ZONE_0]->GetDeform_Mesh()) {
245  CGeometry::ComputeWallDistance(config_container, geometry_container);
246  }

249 bool CSinglezoneDriver::Monitor(unsigned long TimeIter){
250 
251  unsigned long nInnerIter, InnerIter, nTimeIter;
252  su2double MaxTime, CurTime;
253  bool TimeDomain, InnerConvergence, TimeConvergence, FinalTimeReached, MaxIterationsReached;
254 
255  nInnerIter = config_container[ZONE_0]->GetnInner_Iter();
256  InnerIter = config_container[ZONE_0]->GetInnerIter();
257  nTimeIter = config_container[ZONE_0]->GetnTime_Iter();
258  MaxTime = config_container[ZONE_0]->GetMax_Time();
259  CurTime = output_container[ZONE_0]->GetHistoryFieldValue("CUR_TIME");
260 
261  TimeDomain = config_container[ZONE_0]->GetTime_Domain();
262 
263 
264  /*--- Check whether the inner solver has converged --- */
265 
266  if (TimeDomain == NO){
267 
268  InnerConvergence = output_container[ZONE_0]->GetConvergence();
269  MaxIterationsReached = InnerIter+1 >= nInnerIter;
270 
271  if ((MaxIterationsReached || InnerConvergence) && (rank == MASTER_NODE)) {
272  cout << "\n----------------------------- Solver Exit -------------------------------" << endl;
273  if (InnerConvergence) cout << "All convergence criteria satisfied." << endl;
274  else cout << "\nMaximum number of iterations reached (ITER = " << nInnerIter << ") before convergence." << endl;
275  output_container[ZONE_0]->PrintConvergenceSummary();
276  cout << "-------------------------------------------------------------------------" << endl;
277  }
278 
279  StopCalc = MaxIterationsReached || InnerConvergence;
280  }
281 
282  if (TimeDomain == YES) {
283 
284  /*--- Check whether the outer time integration has reached the final time ---*/
285 
286  TimeConvergence = GetTimeConvergence();
287 
288  FinalTimeReached = CurTime >= MaxTime;
289  MaxIterationsReached = TimeIter+1 >= nTimeIter;
290 
291  if ((FinalTimeReached || MaxIterationsReached || TimeConvergence) && (rank == MASTER_NODE)){
292  cout << "\n----------------------------- Solver Exit -------------------------------";
293  if (TimeConvergence) cout << "\nAll windowed time-averaged convergence criteria are fullfilled." << endl;
294  if (FinalTimeReached) cout << "\nMaximum time reached (MAX_TIME = " << MaxTime << "s)." << endl;
295  if (MaxIterationsReached) cout << "\nMaximum number of time iterations reached (TIME_ITER = " << nTimeIter << ")." << endl;
296  cout << "-------------------------------------------------------------------------" << endl;
297  }
298  StopCalc = FinalTimeReached || MaxIterationsReached || TimeConvergence;
299  }
300 
301  /*--- Reset the inner convergence --- */
302 
303  output_container[ZONE_0]->SetConvergence(false);
304 
305  /*--- Increase the total iteration count --- */
306 
307  IterCount += config_container[ZONE_0]->GetInnerIter()+1;
308 
309  return StopCalc;
310 }
