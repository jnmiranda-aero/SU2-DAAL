91  /*!
92  * \brief Compute the viscous flux for the scalar equation at a particular edge.
93  * \tparam SolverSpecificNumericsFunc - lambda-function, that implements solver specific contributions to numerics.
94  * \note The functor has to implement (iPoint, jPoint)
95  * \param[in] iEdge - Edge for which we want to compute the flux
96  * \param[in] geometry - Geometrical definition of the problem.
97  * \param[in] solver_container - Container vector with all the solutions.
98  * \param[in] numerics - Description of the numerical method.
99  * \param[in] config - Definition of the particular problem.
100  */
101  template <class SolverSpecificNumericsFunc>
102  FORCEINLINE void Viscous_Residual_impl(const SolverSpecificNumericsFunc& SolverSpecificNumerics, const unsigned long iEdge,
103  const CGeometry* geometry, CSolver** solver_container, CNumerics* numerics,
104  const CConfig* config) {
105  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
106  CFlowVariable* flowNodes = solver_container[FLOW_SOL] ?
107  su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes()) : nullptr;
108 
109  /*--- Points in edge ---*/
110 
111  auto iPoint = geometry->edges->GetNode(iEdge, 0);
112  auto jPoint = geometry->edges->GetNode(iEdge, 1);
113 
114  /*--- Points coordinates, and normal vector ---*/
115 
116  numerics->SetCoord(geometry->nodes->GetCoord(iPoint), geometry->nodes->GetCoord(jPoint));
117  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
118 
119  /*--- Conservative variables w/o reconstruction ---*/
120 
121  if (flowNodes) {
122  numerics->SetPrimitive(flowNodes->GetPrimitive(iPoint), flowNodes->GetPrimitive(jPoint));
123  }
124 
125  /*--- Turbulent variables w/o reconstruction, and its gradients ---*/
126 
127  numerics->SetScalarVar(nodes->GetSolution(iPoint), nodes->GetSolution(jPoint));
128  numerics->SetScalarVarGradient(nodes->GetGradient(iPoint), nodes->GetGradient(jPoint));
129 
130  /*--- Call Numerics contribution which are Solver-Specifc. Implemented in the caller: Viscous_Residual. ---*/
131 
132  SolverSpecificNumerics(iPoint, jPoint);
133 
134  /*--- Compute residual, and Jacobians ---*/
135 
136  auto residual = numerics->ComputeResidual(config);
137 
138  if (ReducerStrategy) {
139  EdgeFluxes.SubtractBlock(iEdge, residual);
140  if (implicit) Jacobian.UpdateBlocksSub(iEdge, residual.jacobian_i, residual.jacobian_j);
141  } else {
142  LinSysRes.SubtractBlock(iPoint, residual);
143  LinSysRes.AddBlock(jPoint, residual);
144  if (implicit) Jacobian.UpdateBlocksSub(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
145  }
146  }

148  /*!
149  * \brief Generic implementation of the fluid interface boundary condition for scalar solvers.
150  * \tparam SolverSpecificNumericsFunc - lambda that implements solver specific contributions to viscous numerics.
151  * \note The functor has to implement (iPoint)
152  * \param[in] geometry - Geometrical definition of the problem.
153  * \param[in] solver_container - Container vector with all the solutions.
154  * \param[in] conv_numerics - Description of the numerical method.
155  * \param[in] visc_numerics - Description of the numerical method.
156  * \param[in] config - Definition of the particular problem.
157  */
158  template <class SolverSpecificNumericsFunc>
159  void BC_Fluid_Interface_impl(const SolverSpecificNumericsFunc& SolverSpecificNumerics, CGeometry *geometry,
160  CSolver **solver_container, CNumerics *conv_numerics, CNumerics *visc_numerics,
161  CConfig *config) {
162  if (solver_container[FLOW_SOL] == nullptr) return;
163 
164  const auto nPrimVar = solver_container[FLOW_SOL]->GetnPrimVar();
165  su2activevector PrimVar_j(nPrimVar);
166  su2double solution_j[MAXNVAR] = {0.0};
167 
168  for (auto iMarker = 0u; iMarker < config->GetnMarker_All(); iMarker++) {
169 
170  if (config->GetMarker_All_KindBC(iMarker) != FLUID_INTERFACE) continue;
171 
172  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
173  for (auto iVertex = 0u; iVertex < geometry->nVertex[iMarker]; iVertex++) {
174 
175  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
176 
177  if (!geometry->nodes->GetDomain(iPoint)) continue;
178 
179  const auto Point_Normal = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
180  const auto nDonorVertex = GetnSlidingStates(iMarker,iVertex);
181 
182  su2double Normal[MAXNDIM] = {0.0};
183  for (auto iDim = 0u; iDim < nDim; iDim++)
184  Normal[iDim] = -geometry->vertex[iMarker][iVertex]->GetNormal()[iDim];
185 
186  su2double* PrimVar_i = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
187 
188  auto* Jacobian_i = Jacobian.GetBlock(iPoint, iPoint);
189 
190  /*--- Loop over the nDonorVertexes and compute the averaged flux ---*/
191 
192  for (auto jVertex = 0; jVertex < nDonorVertex; jVertex++) {
193 
194  for (auto iVar = 0u; iVar < nPrimVar; iVar++)
195  PrimVar_j[iVar] = solver_container[FLOW_SOL]->GetSlidingState(iMarker, iVertex, iVar, jVertex);
196 
197  /*--- Get the weight computed in the interpolator class for the j-th donor vertex ---*/
198 
199  const su2double weight = solver_container[FLOW_SOL]->GetSlidingState(iMarker, iVertex, nPrimVar, jVertex);
200 
201  /*--- Set primitive variables ---*/
202 
203  conv_numerics->SetPrimitive( PrimVar_i, PrimVar_j.data() );
204 
205  /*--- Set the scalar variable states ---*/
206 
207  for (auto iVar = 0u; iVar < nVar; ++iVar)
208  solution_j[iVar] = GetSlidingState(iMarker, iVertex, iVar, jVertex);
209 
210  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), solution_j);
211 
212  /*--- Set the normal vector ---*/
213 
214  conv_numerics->SetNormal(Normal);
215 
216  if (dynamic_grid)
217  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
218 
219  if (conv_numerics->GetBoundedScalar()) {
220  const su2double* velocity = &PrimVar_j[prim_idx.Velocity()];
221  const su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
222  conv_numerics->SetMassFlux(BoundedScalarBCFlux(iPoint, true, density, velocity, Normal));
223  }
224 
225  auto residual = conv_numerics->ComputeResidual(config);
226 
227  /*--- Accumulate the residuals to compute the average ---*/
228 
229  for (auto iVar = 0u; iVar < nVar; iVar++) {
230  LinSysRes(iPoint,iVar) += weight*residual[iVar];
231  for (auto jVar = 0u; jVar < nVar; jVar++)
232  Jacobian_i[iVar*nVar+jVar] += SU2_TYPE::GetValue(weight*residual.jacobian_i[iVar][jVar]);
233  }
234  }
235 
236  /*--- Set the normal vector and the coordinates ---*/
237 
238  visc_numerics->SetNormal(Normal);
239  su2double Coord_Reflected[MAXNDIM];
240  GeometryToolbox::PointPointReflect(nDim, geometry->nodes->GetCoord(Point_Normal),
241  geometry->nodes->GetCoord(iPoint), Coord_Reflected);
242  visc_numerics->SetCoord(geometry->nodes->GetCoord(iPoint), Coord_Reflected);
243 
244  /*--- Primitive variables ---*/
245 
246  visc_numerics->SetPrimitive(PrimVar_i, PrimVar_j.data());
247 
248  /*--- Scalar variables and their gradients ---*/
249 
250  visc_numerics->SetScalarVar(nodes->GetSolution(iPoint), solution_j);
251  visc_numerics->SetScalarVarGradient(nodes->GetGradient(iPoint), nodes->GetGradient(iPoint));
252 
253  /*--- Allow derived solvers to set more variables in numerics. ---*/
254 
255  SolverSpecificNumerics(iPoint);
256 
257  /*--- Compute and update residual ---*/
258 
259  auto residual = visc_numerics->ComputeResidual(config);
260 
261  LinSysRes.SubtractBlock(iPoint, residual);
262 
263  /*--- Jacobian contribution for implicit integration ---*/
264 
265  Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
266 
267  }
268  END_SU2_OMP_FOR
269  }

272  /*!
273  * \brief Applies a convective flux correction to negate the effects of flow divergence at a BC node.
274  * \note This function should be used for nodes that are part of a boundary marker, it computes a mass flux
275  * from density and velocity at the node, and the outward-pointing normal (-1 * normal of vertex).
276  * \return The mass flux.
277  */
278  inline su2double BoundedScalarBCFlux(unsigned long iPoint, bool implicit, const su2double& density,
279  const su2double* velocity, const su2double* normal) {
280  const su2double edgeMassFlux = density * GeometryToolbox::DotProduct(nDim, velocity, normal);
281  LinSysRes.AddBlock(iPoint, nodes->GetSolution(iPoint), -edgeMassFlux);
282  if (implicit) Jacobian.AddVal2Diag(iPoint, -edgeMassFlux);
283  return edgeMassFlux;
284  }

529  /*!
530  * \brief Allocates the final pointer of SlidingState depending on how many donor vertex donate to it. That number is stored in SlidingStateNodes[val_marker][val_vertex].
531  * \param[in] val_marker - marker index
532  * \param[in] val_vertex - vertex index
533  */
534  inline void SetSlidingStateStructure(unsigned short val_marker, unsigned long val_vertex) final {
535  int iVar;
536 
537  for( iVar = 0; iVar < nVar+1; iVar++){
