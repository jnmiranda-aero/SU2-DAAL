35void CCGNSFileWriter::WriteData(string val_filename) {
36
37#ifdef HAVE_CGNS
38
39 /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
40 val_filename.append(fileExt);
41 /*--- Open the CGNS file for writing. ---*/
42 InitializeMeshFile(val_filename);
43
44 /*--- Write point coordinates. ---*/
45 WriteField(0, "CoordinateX");
46 WriteField(1, "CoordinateY");
47 if (nDim == 3) WriteField(2, "CoordinateZ");
48
49 /*--- Write mesh connectivity. ---*/
50 if (nDim == 2) {
51 WriteConnectivity(LINE, "Lines");
52 WriteConnectivity(TRIANGLE, "Triangles");
53 WriteConnectivity(QUADRILATERAL, "Quadrilaterals");
54 }
55 if (nDim == 3) {
56 WriteConnectivity(TRIANGLE, "Triangles");
57 WriteConnectivity(QUADRILATERAL, "Quadrilaterals");
58 WriteConnectivity(TETRAHEDRON, "Tetrahedra");
59 WriteConnectivity(PYRAMID, "Pyramids");
60 WriteConnectivity(PRISM, "Prisms");
61 WriteConnectivity(HEXAHEDRON, "Hexahedra");
62 }
63
64 /*--- Initialize and write fields. ---*/
65 InitializeFields();
66
67 const auto& fieldNames = dataSorter->GetFieldNames();
68 for (unsigned long i = nDim; i < fieldNames.size(); ++i) {
69 WriteField(i, fieldNames[i]);
70 }
71
72 /*--- Close the CGNS file. ---*/
73 if (rank == MASTER_NODE) CallCGNS(cg_close(cgnsFileID));
74
75#endif
76}

79void CCGNSFileWriter::InitializeMeshFile(const string& val_filename) {
80 if (!dataSorter->GetConnectivitySorted()) {
81 SU2_MPI::Error("Connectivity must be sorted.", CURRENT_FUNCTION);
82 }
83
84 nLocalPoints = dataSorter->GetnPoints();
85 nDim = dataSorter->GetnDim();
86 GlobalElem = static_cast<cgsize_t>(dataSorter->GetnElemGlobal());
87 GlobalPoint = static_cast<cgsize_t>(dataSorter->GetnPointsGlobal());
88 cumulative = 0;
89
90 /*--- If surface file cell dimension is decreased. ---*/
91 const auto nCell = static_cast<int>(nDim - isSurface);
92
93 if (rank == MASTER_NODE) {
94 /*--- Remove the previous file if present. ---*/
95 remove(val_filename.c_str());
96
97 /*--- Create CGNS file and open in write mode. ---*/
98 CallCGNS(cg_open(val_filename.c_str(), CG_MODE_WRITE, &cgnsFileID));
99
100 /*--- Create Base. ---*/
101 CallCGNS(cg_base_write(cgnsFileID, "Base", nCell, nDim, &cgnsBase));
102
103 /*--- Create Zone. ---*/
104 array<cgsize_t, 3> zoneData;
105
106 zoneData[0] = GlobalPoint;
107 zoneData[1] = GlobalElem;
108 zoneData[2] = 0;
109
110 CallCGNS(cg_zone_write(cgnsFileID, cgnsBase, "Zone", zoneData.data(), Unstructured, &cgnsZone));
111 }
112}

114void CCGNSFileWriter::WriteField(int iField, const string& FieldName) {
115 /*--- Check if field is coordinate. ---*/
116 const bool isCoord = iField < nDim;
117
118 /*--- Create send buffer. ---*/
119 sendBufferField.resize(nLocalPoints);
120
121 for (unsigned long iPoint = 0; iPoint < nLocalPoints; iPoint++) {
122 sendBufferField[iPoint] = static_cast<dataPrecision>(dataSorter->GetData(iField, iPoint));
123 }
124
125 if (rank != MASTER_NODE) {
126 SU2_MPI::Send(sendBufferField.data(), nLocalPoints * sizeof(dataPrecision), MPI_CHAR, MASTER_NODE, 0,
127 SU2_MPI::GetComm());
128 return;
129 }
130
131 /*--- Coordinate vector is written in blocks, one for each process. ---*/
132 cgsize_t nodeBegin = 1;
133 auto nodeEnd = static_cast<cgsize_t>(nLocalPoints);
134 if (nLocalPoints > 0) {
135 if (isCoord) {
136 int CoordinateNumber;
137 CallCGNS(cg_coord_partial_write(cgnsFileID, cgnsBase, cgnsZone, dataType, FieldName.c_str(), &nodeBegin, &nodeEnd,
138 sendBufferField.data(), &CoordinateNumber));
139 } else {
140 int fieldNumber;
141 CallCGNS(cg_field_partial_write(cgnsFileID, cgnsBase, cgnsZone, cgnsFields, dataType, FieldName.c_str(), &nodeBegin,
142 &nodeEnd, sendBufferField.data(), &fieldNumber));
143 }
144 }
145
146 for (int i = 0; i < size; ++i) {
147 if (i == MASTER_NODE) continue;
148 /*--- In CGNS numbering starts form 1 and ranges are inclusive ---*/
149 nodeBegin = static_cast<cgsize_t>(dataSorter->GetnPointCumulative(i) + 1);
150 nodeEnd = static_cast<cgsize_t>(dataSorter->GetnPointCumulative(i + 1));
151
152 const auto recvSize = static_cast<int>(nodeEnd - nodeBegin + 1);
153 recvBufferField.resize(recvSize);
154
155 SU2_MPI::Recv(recvBufferField.data(), recvSize * sizeof(dataPrecision), MPI_CHAR, i, 0, SU2_MPI::GetComm(),
156 MPI_STATUS_IGNORE);
157 if (recvSize <= 0) continue;
158 if (isCoord) {
159 int CoordinateNumber;
160 CallCGNS(cg_coord_partial_write(cgnsFileID, cgnsBase, cgnsZone, dataType, FieldName.c_str(), &nodeBegin, &nodeEnd,
161 recvBufferField.data(), &CoordinateNumber));
162 } else {
163 int fieldNumber;
164 CallCGNS(cg_field_partial_write(cgnsFileID, cgnsBase, cgnsZone, cgnsFields, dataType, FieldName.c_str(),
165 &nodeBegin, &nodeEnd, recvBufferField.data(), &fieldNumber));
166 }
167 }

170void CCGNSFileWriter::WriteConnectivity(GEO_TYPE type, const string& SectionName) {
171 const auto nTotElem = dataSorter->GetnElemGlobal(type);
172 if (nTotElem == 0) return;
173
174 /*--- Create a new CGNS node to store connectivity. ---*/
175 const auto elementType = GetCGNSType(type);
176
177 cgsize_t firstElem = cumulative + 1;
178 cgsize_t endElem = cumulative + static_cast<cgsize_t>(nTotElem);
179
180 int cgnsSection;
181 if (rank == MASTER_NODE)
182 CallCGNS(cg_section_partial_write(cgnsFileID, cgnsBase, cgnsZone, SectionName.c_str(), elementType, firstElem,
183 endElem, 0, &cgnsSection));
184
185 /*--- Retrieve element distribution among processes. ---*/
186 const auto nLocalElem = dataSorter->GetnElem(type);
187
188 vector<unsigned long> distElem(size);
189
190 SU2_MPI::Allgather(&nLocalElem, 1, MPI_UNSIGNED_LONG, distElem.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
191
192 firstElem = cumulative + 1;
193 endElem = cumulative + static_cast<cgsize_t>(distElem[rank]);
194
195 /*--- Connectivity is stored in send buffer. ---*/
196 const auto nPointsElem = nPointsOfElementType(type);
197 sendBufferConnectivity.resize(nLocalElem * nPointsElem);
198
199 for (unsigned long iElem = 0; iElem < nLocalElem; iElem++) {
200 for (unsigned long iPoint = 0; iPoint < nPointsElem; iPoint++) {
201 sendBufferConnectivity[iPoint + nPointsElem * iElem] =
202 static_cast<cgsize_t>(dataSorter->GetElemConnectivity(type, iElem, iPoint));
203 }
204 }
205
206 const auto bufferSize = static_cast<int>(nLocalElem * nPointsElem * sizeof(cgsize_t));
207 if (rank != MASTER_NODE) {
208 SU2_MPI::Send(sendBufferConnectivity.data(), bufferSize, MPI_CHAR, MASTER_NODE, 1, SU2_MPI::GetComm());
209 return;
210 }
211
212 /*--- Connectivity vector is written in blocks, one for each process. ---*/
213 if (nLocalElem > 0)
214 CallCGNS(cg_elements_partial_write(cgnsFileID, cgnsBase, cgnsZone, cgnsSection, firstElem, endElem,
215 sendBufferConnectivity.data()));
216
217 for (int i = 0; i < size; ++i) {
218 if (i == MASTER_NODE) continue;
219 /*--- In CGNS numbering starts form 1 and ranges are inclusive ---*/
220 firstElem = endElem + 1;
221 endElem += static_cast<cgsize_t>(distElem[i]);
222 const auto recvSize = static_cast<int>((endElem - firstElem + 1) * nPointsElem);
223 recvBufferConnectivity.resize(recvSize);
224
225 const auto recvByte = static_cast<int>(recvBufferConnectivity.size() * sizeof(cgsize_t));
226 SU2_MPI::Recv(recvBufferConnectivity.data(), recvByte, MPI_CHAR, i, 1, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
227
228 if (!recvBufferConnectivity.empty())
229 CallCGNS(cg_elements_partial_write(cgnsFileID, cgnsBase, cgnsZone, cgnsSection, firstElem, endElem,
230 recvBufferConnectivity.data()));
231 }
232 cumulative += static_cast<cgsize_t>(nTotElem);
233}
