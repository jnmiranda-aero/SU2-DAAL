137 const su2double Velocity_BLEdge = sqrt((pow(Vel_inf_Mag,2)) + (((2 * Gamma) / (Gamma - 1)) * (1 - (pow(Pressure_i / Pressure_inf,(1 - (1 / Gamma)))))*(Pressure_i / Density_inf))); // Boundary Layer edge velocity magnitude req'd to calculate edge Mach number for compressibilty correction to LM model (doi:10.2514/6.2022-1542) - jnmiranda-ucd-daal

150 AD::SetPreaccIn(V_i[idx.Density()], V_i[idx.LaminarViscosity()], V_i[idx.EddyViscosity()]);
151
152 Density_i = V_i[idx.Density()];
153 Laminar_Viscosity_i = V_i[idx.LaminarViscosity()];
154 Eddy_Viscosity_i = V_i[idx.EddyViscosity()];
155
156 Residual[0] = 0.0;
157 Residual[1] = 0.0;
158 Jacobian_i[0][0] = 0.0;
159 Jacobian_i[0][1] = 0.0;
160 Jacobian_i[1][0] = 0.0;
161 Jacobian_i[1][1] = 0.0;
162
163 if (dist_i > 1e-10) {
164 su2double Tu = 1.0;
165 if (TurbFamily == TURB_FAMILY::KW) Tu = max(100.0 * sqrt(2.0 * ScalarVar_i[0] / 3.0) / Velocity_Mag, 0.027);
166 if (TurbFamily == TURB_FAMILY::SA) Tu = config->GetTurbulenceIntensity_FreeStream() * 100;
167
168 /*--- Corr_RetC correlation*/
169 const su2double Corr_Rec = TransCorrelations.ReThetaC_Correlations(config, V_i, Tu, TransVar_i[1]);
170 // const su2double Corr_Rec = TransCorrelations.ReThetaC_Correlations(config, Tu, TransVar_i[1]); // Mach corrected Re_theta_c: Re_theta_c_compressible. Correction applied in correlations definitions - jnmiranda-ucd-daal
171 // const su2double Corr_Rec = TransCorrelations.ReThetaC_Correlations(Tu, TransVar_i[1]); // Mach corrected Re_theta_c: Re_theta_c_compressible. Correction applied in correlations definitions - jnmiranda-ucd-daal
172 // const su2double Corr_Rec_comp = Corr_Rec / C_Me; // this isnt really needed anymore as it would require modifying this in other places of the code. Instead correction is applied in the correlations- just remember that this is no longer the OG Corr_Rec but the compressible one
173
174 /*--- F_length correlation*/
175 const su2double Corr_F_length = TransCorrelations.FLength_Correlations(config, Tu, TransVar_i[1]);
176
177 /*--- F_length ---*/
178 su2double F_length = 0.0;
179 if (TurbFamily == TURB_FAMILY::KW) {
180 const su2double r_omega = Density_i * dist_i * dist_i * ScalarVar_i[1] / Laminar_Viscosity_i;
181 const su2double f_sub = exp(-pow(r_omega / 200.0, 2));
182 F_length = Corr_F_length * (1. - f_sub) + 40.0 * f_sub;
183 }
184 if (TurbFamily == TURB_FAMILY::SA) F_length = Corr_F_length;
185
186 /*--- F_onset ---*/
187 su2double R_t = 1.0;
188 if (TurbFamily == TURB_FAMILY::KW) R_t = Density_i * ScalarVar_i[0] / Laminar_Viscosity_i / ScalarVar_i[1];
189 if (TurbFamily == TURB_FAMILY::SA) R_t = Eddy_Viscosity_i / Laminar_Viscosity_i;
190
191 const su2double Re_v = Density_i * dist_i * dist_i * StrainMag_i / Laminar_Viscosity_i;
192 const su2double F_onset1 = Re_v / (2.193 * Corr_Rec); // Original F_length1 using incompressible correlation for Re_theta_c. This is now the old F_length1. - jnmiranda-ucd-daal
193 // const su2double F_onset1 = Re_v / (2.193 * Corr_Rec_comp); // Mach corrected F_length1 using Re_theta_c_compressible - jnmiranda-ucd-daal - no longer needed but keeping here to remind myself how I initially did it
194 su2double F_onset2 = 1.0;
195 su2double F_onset3 = 1.0;
196 if (TurbFamily == TURB_FAMILY::KW) {
197 F_onset2 = min(max(F_onset1, pow(F_onset1, 4.0)), 2.0);
198 F_onset3 = max(1.0 - pow(R_t / 2.5, 3.0), 0.0);
199 }
200 if (TurbFamily == TURB_FAMILY::SA) {
201 F_onset2 = min(max(F_onset1, pow(F_onset1, 4.0)), 4.0);
202 F_onset3 = max(2.0 - pow(R_t / 2.5, 3.0), 0.0);
203 }
204 const su2double F_onset = max(F_onset2 - F_onset3, 0.0);
205
206 /*-- Gradient of velocity magnitude ---*/
207
208 su2double dU_dx = 0.5 / Velocity_Mag * (2. * vel_u * PrimVar_Grad_i[1][0] + 2. * vel_v * PrimVar_Grad_i[2][0]);
209 if (nDim == 3) dU_dx += 0.5 / Velocity_Mag * (2. * vel_w * PrimVar_Grad_i[3][0]);
210
211 su2double dU_dy = 0.5 / Velocity_Mag * (2. * vel_u * PrimVar_Grad_i[1][1] + 2. * vel_v * PrimVar_Grad_i[2][1]);
212 if (nDim == 3) dU_dy += 0.5 / Velocity_Mag * (2. * vel_w * PrimVar_Grad_i[3][1]);
213
214 su2double dU_dz = 0.0;
215 if (nDim == 3)
216 dU_dz =
217 0.5 / Velocity_Mag *
218 (2. * vel_u * PrimVar_Grad_i[1][2] + 2. * vel_v * PrimVar_Grad_i[2][2] + 2. * vel_w * PrimVar_Grad_i[3][2]);
219
220 su2double du_ds = vel_u / Velocity_Mag * dU_dx + vel_v / Velocity_Mag * dU_dy;
221 if (nDim == 3) du_ds += vel_w / Velocity_Mag * dU_dz;
222
223 /*-- Calculate blending function f_theta --*/
224 su2double time_scale = 500.0 * Laminar_Viscosity_i / Density_i / Velocity_Mag / Velocity_Mag;
225 if (options.LM2015)
226 time_scale = min(time_scale,
227 Density_i * LocalGridLength_i * LocalGridLength_i / (Laminar_Viscosity_i + Eddy_Viscosity_i));
228 const su2double theta_bl = TransVar_i[1] * Laminar_Viscosity_i / Density_i / Velocity_Mag;
229 const su2double delta_bl = 7.5 * theta_bl;
230 const su2double delta = 50.0 * VorticityMag * dist_i / Velocity_Mag * delta_bl + 1e-20;
231
232 su2double f_wake = 0.0;
233 if (TurbFamily == TURB_FAMILY::KW) {
234 const su2double re_omega = Density_i * ScalarVar_i[1] * dist_i * dist_i / Laminar_Viscosity_i;
235 f_wake = exp(-pow(re_omega / (1.0e+05), 2));
236 }
237 if (TurbFamily == TURB_FAMILY::SA) f_wake = 1.0;
238
239 const su2double var1 = (TransVar_i[0] - 1.0 / c_e2) / (1.0 - 1.0 / c_e2);
240 const su2double var2 = 1.0 - pow(var1, 2.0);
241 const su2double f_theta = min(max(f_wake * exp(-pow(dist_i / delta, 4)), var2), 1.0);
242 const su2double f_turb = exp(-pow(R_t / 4, 4));
243
244 su2double f_theta_2 = 0.0;
245 if (options.LM2015)
246 f_theta_2 = min(f_wake * exp(-pow(dist_i / delta, 4.0)), 1.0);
247
248 /*--- Corr_Ret correlation*/
249 const su2double Corr_Ret_lim = 20.0;
250 su2double f_lambda = 1.0;
251
252 su2double Retheta_Error = 200.0, Retheta_old = 0.0;
253 su2double lambda = 0.0;
254 su2double Corr_Ret = 20.0;
255
256 for (int iter = 0; iter < 100; iter++) {
257 su2double theta = Corr_Ret * Laminar_Viscosity_i / Density_i / Velocity_Mag;
258 lambda = Density_i * theta * theta / Laminar_Viscosity_i * du_ds;
259 lambda = min(max(-0.1, lambda), 0.1);
260
261 if (lambda <= 0.0) {
262 f_lambda = 1. - (-12.986 * lambda - 123.66 * lambda * lambda - 405.689 * lambda * lambda * lambda) *
263 exp(-pow(Tu / 1.5, 1.5));
264 } else {
265 f_lambda = 1. + 0.275 * (1. - exp(-35. * lambda)) * exp(-Tu / 0.5);
266 }
267
268 if (Tu <= 1.3) {
269 Corr_Ret = f_lambda * (1173.51 - 589.428 * Tu + 0.2196 / Tu / Tu);
270 } else {
271 Corr_Ret = 331.5 * f_lambda * pow(Tu - 0.5658, -0.671);
272 }
273 Corr_Ret = max(Corr_Ret, Corr_Ret_lim);
274
275 Retheta_Error = fabs(Retheta_old - Corr_Ret) / Retheta_old;
276
277 if (Retheta_Error < 0.0000001) {
278 break;
279 }
280
281 Retheta_old = Corr_Ret;
282 }
283
284 su2double Corr_Ret_comp = Corr_Ret * f_Me; // Compressibility correction to Re_theta_t (doi:10.2514/6.2022-1542) - jnmiranda-ucd-daal
285
286 /*-- Corr_RetT_SCF Correlations--*/
287 su2double ReThetat_SCF = 0.0;
288 if (options.LM2015) {
289 su2double VelocityNormalized[3];
290 VelocityNormalized[0] = vel_u / Velocity_Mag;
291 VelocityNormalized[1] = vel_v / Velocity_Mag;
292 if (nDim == 3) VelocityNormalized[2] = vel_w / Velocity_Mag;
293
294 su2double StreamwiseVort = 0.0;
295 for (auto iDim = 0u; iDim < nDim; iDim++) {
296 StreamwiseVort += VelocityNormalized[iDim] * Vorticity_i[iDim];
297 }
298 StreamwiseVort = abs(StreamwiseVort);
299
300 const su2double H_CF = StreamwiseVort * dist_i / Velocity_Mag;
301 const su2double DeltaH_CF = H_CF * (1.0 + min(Eddy_Viscosity_i / Laminar_Viscosity_i, 0.4));
302 const su2double DeltaH_CF_Minus = max(-1.0 * (0.1066 - DeltaH_CF), 0.0);
303 const su2double DeltaH_CF_Plus = max(0.1066 - DeltaH_CF, 0.0);
304 const su2double fDeltaH_CF_Minus = 75.0 * tanh(DeltaH_CF_Minus / 0.0125);
305 const su2double fDeltaH_CF_Plus = 6200 * DeltaH_CF_Plus + 50000 * DeltaH_CF_Plus * DeltaH_CF_Plus;
306
307 const su2double toll = 1e-5;
308 su2double error = toll + 1.0;
309 su2double thetat_SCF = 0.0;
310 su2double rethetat_SCF_old = 20.0;
311 const int nMax = 100;
312
313 int iter;
314 for (iter = 0; iter < nMax && error > toll; iter++) {
315 thetat_SCF = rethetat_SCF_old * Laminar_Viscosity_i / (Density_i * (Velocity_Mag / 0.82));
316 thetat_SCF = max(1e-20, thetat_SCF);
317
318 ReThetat_SCF = -35.088 * log(hRoughness / thetat_SCF) + 319.51 + fDeltaH_CF_Plus - fDeltaH_CF_Minus;
319
320 error = abs(ReThetat_SCF - rethetat_SCF_old) / rethetat_SCF_old;
321
322 rethetat_SCF_old = ReThetat_SCF;
323 }
324 }

326 // su2double ReThetat_SCF_comp = ReThetat_SCF * f_Me; // I dont know if this would work since the authors from (doi:10.2514/6.2022-1542) did not actually do this. A function would be needed here (although f_Me might work) - jnmiranda-ucd-daal
327
328 /*-- production term of Intermittency(Gamma) --*/
329 const su2double Pg =
330 F_length * c_a1 * Density_i * StrainMag_i * sqrt(F_onset * TransVar_i[0]) * (1.0 - c_e1 * TransVar_i[0]);
331
332 /*-- destruction term of Intermittency(Gamma) --*/
333 const su2double Dg = c_a2 * Density_i * VorticityMag * TransVar_i[0] * f_turb * (c_e2 * TransVar_i[0] - 1.0);
334
335 /*-- production term of ReThetaT --*/
336 // const su2double PRethetat = c_theta * Density_i / time_scale * (Corr_Ret - TransVar_i[1]) * (1.0 - f_theta);
337 const su2double PRethetat = c_theta * Density_i / time_scale * (Corr_Ret_comp - TransVar_i[1]) * (1.0 - f_theta); // compressibility correction to Re_theta_t added
338
339 /*-- destruction term of ReThetaT --*/
340 // It should not be with the minus sign but I put for consistency
341 su2double DRethetat = 0.0;
342 if (options.LM2015)
343 DRethetat = -c_theta * (Density_i / time_scale) * c_CF * min(ReThetat_SCF - TransVar_i[1], 0.0) * f_theta_2;
344
345 /*--- Source ---*/
346 Residual[0] += (Pg - Dg) * Volume;
347 Residual[1] += (PRethetat - DRethetat) * Volume;
348
349 /*--- Implicit part ---*/
350 Jacobian_i[0][0] = (F_length * c_a1 * StrainMag_i * sqrt(F_onset) *
351 (0.5 * pow(TransVar_i[0], -0.5) - 1.5 * c_e1 * pow(TransVar_i[0], 0.5)) -
352 c_a2 * VorticityMag * f_turb * (2.0 * c_e2 * TransVar_i[0] - 1.0)) *
353 Volume;
354 Jacobian_i[0][1] = 0.0;
355 Jacobian_i[1][0] = 0.0;
356 Jacobian_i[1][1] = -c_theta / time_scale * (1.0 - f_theta) * Volume;
357 if (options.LM2015 && ReThetat_SCF - TransVar_i[1] < 0)
358 Jacobian_i[1][1] += (c_theta / time_scale) * c_CF * f_theta_2 * Volume;
359 }
360
361 AD::SetPreaccOut(Residual, nVar);
362 AD::EndPreacc();
363
364 return ResidualType<>(Residual, Jacobian_i, nullptr);
365 }
