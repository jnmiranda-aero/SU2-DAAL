45/*!
46 * \class CAvgGrad_Scalar
47 * \brief Template class for computing viscous residual of scalar values
48 * \details This class serves as a template for the scalar viscous residual
49 * classes. The general structure of a viscous residual calculation is the
50 * same for many different models, which leads to a lot of repeated code.
51 * By using the template design pattern, these sections of repeated code are
52 * moved to a shared base class, and the specifics of each model
53 * are implemented by derived classes. In order to add a new residual
54 * calculation for a viscous residual, extend this class and implement
55 * the pure virtual functions with model-specific behavior.
56 * \ingroup ViscDiscr
57 * \author C. Pederson, A. Bueno, and F. Palacios
58 */
59template <class FlowIndices>
60class CAvgGrad_Scalar : public CNumerics {
61 protected:
62 enum : unsigned short {MAXNVAR = 8};
63
64 const FlowIndices idx; /*!< \brief Object to manage the access to the flow primitives. */
65 su2double Proj_Mean_GradScalarVar[MAXNVAR]; /*!< \brief Mean_gradScalarVar DOT normal, corrected if required. */
66 su2double proj_vector_ij = 0.0; /*!< \brief (Edge_Vector DOT normal)/|Edge_Vector|^2 */
67 su2double Flux[MAXNVAR]; /*!< \brief Final result, diffusive flux/residual. */
68 su2double* Jacobian_i[MAXNVAR]; /*!< \brief Flux Jacobian w.r.t. node i. */
69 su2double* Jacobian_j[MAXNVAR]; /*!< \brief Flux Jacobian w.r.t. node j. */
70 su2double JacobianBuffer[2*MAXNVAR*MAXNVAR];/*!< \brief Static storage for the two Jacobians. */
71
72 const bool correct_gradient = false, incompressible = false;
73
74 /*!
75 * \brief A pure virtual function; Adds any extra variables to AD
76 */
77 virtual void ExtraADPreaccIn() = 0;
78
79 /*!
80 * \brief Model-specific steps in the ComputeResidual method, derived classes
81 * should compute the Flux and Jacobians (i/j) inside this method.
82 * \param[in] config - Definition of the particular problem.
83 */
84 virtual void FinishResidualCalc(const CConfig* config) = 0;
85
86 public:
87 /*!
88 * \brief Constructor of the class.
89 * \param[in] val_nDim - Number of dimensions of the problem.
90 * \param[in] val_nVar - Number of variables of the problem.
91 * \param[in] correct_gradient - Whether to correct gradient for skewness.
92 * \param[in] config - Definition of the particular problem.
93 */
94 CAvgGrad_Scalar(unsigned short val_nDim, unsigned short val_nVar, bool correct_grad,
95 const CConfig* config)
96 : CNumerics(val_nDim, val_nVar, config),
97 idx(val_nDim, config->GetnSpecies()),
98 correct_gradient(correct_grad),
99 incompressible(config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) {
100 if (nVar > MAXNVAR) {
101 SU2_MPI::Error("Static arrays are too small.", CURRENT_FUNCTION);
102 }
103 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
104 Jacobian_i[iVar] = &JacobianBuffer[iVar * nVar];
105 Jacobian_j[iVar] = &JacobianBuffer[iVar * nVar + MAXNVAR * MAXNVAR];
106 }
107
108 /*--- Initialize the JacobianBuffer to zero. ---*/
109 for (unsigned short iVar = 0; iVar < 2*MAXNVAR*MAXNVAR; iVar++) {
110 JacobianBuffer[iVar] = 0.0;
111 }
112 }
113
114 /*!
115 * \brief Compute the viscous residual using an average of gradients without correction.
116 * \param[in] config - Definition of the particular problem.
117 * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
118 */
119 ResidualType<> ComputeResidual(const CConfig* config) final {
120 AD::StartPreacc();
121 AD::SetPreaccIn(Coord_i, nDim);
122 AD::SetPreaccIn(Coord_j, nDim);
123 AD::SetPreaccIn(Normal, nDim);
124 AD::SetPreaccIn(ScalarVar_Grad_i, nVar, nDim);
125 AD::SetPreaccIn(ScalarVar_Grad_j, nVar, nDim);
126 if (correct_gradient) {
127 AD::SetPreaccIn(ScalarVar_i, nVar);
128 AD::SetPreaccIn(ScalarVar_j, nVar);
129 }
130 if (!std::is_same<FlowIndices, CNoFlowIndices>::value) {
131 AD::SetPreaccIn(V_i[idx.Density()], V_i[idx.LaminarViscosity()], V_i[idx.EddyViscosity()]);
132 AD::SetPreaccIn(V_j[idx.Density()], V_j[idx.LaminarViscosity()], V_j[idx.EddyViscosity()]);
133
134 Density_i = V_i[idx.Density()];
135 Density_j = V_j[idx.Density()];
136 Laminar_Viscosity_i = V_i[idx.LaminarViscosity()];
137 Laminar_Viscosity_j = V_j[idx.LaminarViscosity()];
138 Eddy_Viscosity_i = V_i[idx.EddyViscosity()];
139 Eddy_Viscosity_j = V_j[idx.EddyViscosity()];
140 }
141
142 ExtraADPreaccIn();
143
144 su2double ProjGradScalarVarNoCorr[MAXNVAR];
145 proj_vector_ij = ComputeProjectedGradient(nDim, nVar, Normal, Coord_i, Coord_j, ScalarVar_Grad_i, ScalarVar_Grad_j,
146 correct_gradient, ScalarVar_i, ScalarVar_j, ProjGradScalarVarNoCorr,
147 Proj_Mean_GradScalarVar);
148 FinishResidualCalc(config);
149
150 AD::SetPreaccOut(Flux, nVar);
151 AD::EndPreacc();
152
153 return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
154 }
155};
