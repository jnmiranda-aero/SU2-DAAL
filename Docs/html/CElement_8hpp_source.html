35 /*!
36  * \class CElement
37  * \ingroup FemAlgos
38  * \brief Abstract class for defining finite elements.
39  * \note Usage: Element instances are used to compute gradients (in reference or current
40  * coordinates), element matrices (stiffness, mass, etc.), and nodal residuals.
41  * The use-case is to set the coordinates, compute the gradients, and then
42  * incrementally build the matrices using the various "Add_XXX" methods.
43  * \note Implementation: This base class provides a generic interface to an element,
44  * so that client numerics do not need to care about specific details (e.g. shape
45  * functions). However, more efficient code can be generated by caring about them,
46  * we try to strike a balance by tailoring the gradient computation to the specific
47  * sizes (number of integration points, nodes, and dimensions) of derived elements.
48  * This is done via the intermediate template class CElementWithKnownSizes from
49  * which derived elements can inherit the implementation of "ComputeGrad_XXX".
50  * To do so their constructors should consist of:
51  * - Call CElementWithKnownSizes's ctor to allocate parent class members.
52  * - Populate the "GaussWeights", and set the shape function values (Ni) in "GaussPoint".
53  * - Populate the "GaussCoords" and "dNiXj" fields of CElementWithKnownSizes.
54  * - Define the coordinates for nodal extrapolation of quantities at the Gauss points.
55  * Derived elements can also implement cheap methods of computing their area/volume.
56  * \author R. Sanchez
57  */

519 /*!
520  * \class CElementWithKnownSizes
521  * \ingroup FemAlgos
522  * \brief Templated class to implement the computation of gradients for specific element sizes.
523  * \author P. Gomes, R. Sanchez
524  */
525 template <unsigned short NGAUSS, unsigned short NNODE, unsigned short NDIM>
526 class CElementWithKnownSizes : public CElement {
527  private:
528  FORCEINLINE static su2double JacobianAdjoint(const su2double Jacobian[][1], su2double ad[][1]) {
529  /*--- Adjoint to Jacobian, we put 1.0 here so that ad/detJac is the inverse later ---*/
530  ad[0][0] = 1.0;
531  /*--- Determinant of Jacobian ---*/
532  return Jacobian[0][0];
533  }
534 
535  FORCEINLINE static su2double JacobianAdjoint(const su2double Jacobian[][2], su2double ad[][2]) {
536  ad[0][0] = Jacobian[1][1];
537  ad[0][1] = -Jacobian[0][1];
538  ad[1][0] = -Jacobian[1][0];
539  ad[1][1] = Jacobian[0][0];
540  /*--- Determinant of Jacobian ---*/
541  return ad[0][0] * ad[1][1] - ad[0][1] * ad[1][0];
542  }
543 
544  FORCEINLINE static su2double JacobianAdjoint(const su2double Jacobian[][3], su2double ad[][3]) {
545  ad[0][0] = Jacobian[1][1] * Jacobian[2][2] - Jacobian[1][2] * Jacobian[2][1];
546  ad[0][1] = Jacobian[0][2] * Jacobian[2][1] - Jacobian[0][1] * Jacobian[2][2];
547  ad[0][2] = Jacobian[0][1] * Jacobian[1][2] - Jacobian[0][2] * Jacobian[1][1];
548  ad[1][0] = Jacobian[1][2] * Jacobian[2][0] - Jacobian[1][0] * Jacobian[2][2];
549  ad[1][1] = Jacobian[0][0] * Jacobian[2][2] - Jacobian[0][2] * Jacobian[2][0];
550  ad[1][2] = Jacobian[0][2] * Jacobian[1][0] - Jacobian[0][0] * Jacobian[1][2];
551  ad[2][0] = Jacobian[1][0] * Jacobian[2][1] - Jacobian[1][1] * Jacobian[2][0];
552  ad[2][1] = Jacobian[0][1] * Jacobian[2][0] - Jacobian[0][0] * Jacobian[2][1];
553  ad[2][2] = Jacobian[0][0] * Jacobian[1][1] - Jacobian[0][1] * Jacobian[1][0];
554  /*--- Determinant of Jacobian ---*/
555  return Jacobian[0][0] * ad[0][0] + Jacobian[0][1] * ad[1][0] + Jacobian[0][2] * ad[2][0];

571  /*!
572  * \brief Implementation of gradient computation leveraging the static sizes.
573  * \param[in] FRAME - template, REFERENCE or CURRENT coordinates.
574  */
575  template <FrameType FRAME>
576  void ComputeGrad_impl(void) {
577  su2double Jacobian[NDIM][NDIM], ad[NDIM][NDIM];
578  unsigned short iNode, iDim, jDim, iGauss;
579 
580  /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
581  for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed) ---*/
582  const su2activematrix& Coord = (FRAME == REFERENCE) ? RefCoord : CurrentCoord;
583 
584  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
585  /*--- Jacobian transformation ---*/
586  /*--- This does dX/dXi transpose ---*/
587 
588  for (iDim = 0; iDim < NDIM; iDim++)
589  for (jDim = 0; jDim < NDIM; jDim++) Jacobian[iDim][jDim] = 0.0;
590 
591  for (iNode = 0; iNode < NNODE; iNode++)
592  for (iDim = 0; iDim < NDIM; iDim++)
593  for (jDim = 0; jDim < NDIM; jDim++) Jacobian[iDim][jDim] += Coord(iNode, jDim) * dNiXj[iGauss][iNode][iDim];
594 
595  if (NDIM == 1) {
596  /*--- Obviously the Jacobian is the slope of the line ---*/
597  Jacobian[0][0] = (Coord[1][0] - Coord[0][0]);
598  }
599 
600  /*--- Adjoint to the Jacobian and determinant ---*/
601 
602  auto detJac = JacobianAdjoint(Jacobian, ad);
603 
604  if (FRAME == REFERENCE)
605  GaussPoint[iGauss].SetJ_X(detJac);
606  else
607  GaussPoint[iGauss].SetJ_x(detJac);
608 
609  /*--- Jacobian inverse (it was already computed as transpose) ---*/
610 
611  for (iDim = 0; iDim < NDIM; iDim++)
612  for (jDim = 0; jDim < NDIM; jDim++) Jacobian[iDim][jDim] = ad[iDim][jDim] / detJac;
613 
614  /*--- Derivatives with respect to global coordinates ---*/
615 
616  for (iNode = 0; iNode < NNODE; iNode++) {
617  for (iDim = 0; iDim < NDIM; iDim++) {
618  su2double GradNi_Xj = 0.0;
619  for (jDim = 0; jDim < NDIM; jDim++) GradNi_Xj += Jacobian[iDim][jDim] * dNiXj[iGauss][iNode][jDim];
620 
621  if (FRAME == REFERENCE)
622  GaussPoint[iGauss].SetGradNi_Xj(GradNi_Xj, iDim, iNode);
623  else
624  GaussPoint[iGauss].SetGradNi_xj(GradNi_Xj, iDim, iNode);
625  }
626  }
627  }
628  }
629 
630  template <FrameType FRAME>
631  void ComputeGrad_impl_surf_embedded() {
632  /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
633  for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed) ---*/
634  const su2activematrix& Coord = (FRAME == REFERENCE) ? RefCoord : CurrentCoord;
635 
636  if (NDIM == 1) {
637  unsigned short iNode, iDim, iGauss;
638  su2double Jacobian[2];
639  su2double val_grad;
640 
641  Jacobian[0] = (Coord[1][0] - Coord[0][0]);
642  Jacobian[1] = (Coord[1][1] - Coord[0][1]);
643  const su2double JTJ = Jacobian[0] * Jacobian[0] + Jacobian[1] * Jacobian[1];
644  const su2double volJacobian = sqrt(JTJ);
645 
646  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
647  GaussPoint[iGauss].SetJ_X(volJacobian);
648 
649  for (iNode = 0; iNode < NNODE; iNode++) {
650  for (iDim = 0; iDim < NDIM + 1; iDim++) {
651  val_grad = Jacobian[iDim] * dNiXj[iGauss][iNode][0] / JTJ;
652  GaussPoint[iGauss].SetGradNi_Xj(val_grad, iDim, iNode);
653  }
654  }
655  }
656 
657  } else if (NDIM == 2) {
658  su2double Jacobian[NDIM + 1][NDIM], JTJ[NDIM][NDIM];
659  su2double volJacobian;
660  unsigned short iShape, iDim, jDim, jVertex, kDim, iGauss;
661 
662  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
663  /*--- Jacobian transformation ---*/
664  /*--- This does dX/dXi transpose ---*/
665  if (NNODE == 3) {
666  for (iDim = 0; iDim < NDIM + 1; iDim++) {
667  for (jVertex = 0; jVertex < NDIM; jVertex++) {
668  Jacobian[iDim][jVertex] = Coord[jVertex + 1][iDim] - Coord[0][iDim];
669  }
670  }
671  }
672  if (NNODE == 4) {
673  for (iDim = 0; iDim < NDIM + 1; iDim++) {
674  Jacobian[iDim][0] = (Coord[1][iDim] - Coord[0][iDim]) / 2;
675  Jacobian[iDim][1] = (Coord[3][iDim] - Coord[0][iDim]) / 2;
676  }
677  }
678 
679  for (iDim = 0; iDim < NDIM; iDim++) {
680  for (jDim = 0; jDim < NDIM; jDim++) {
681  JTJ[iDim][jDim] = 0.0;
682  for (kDim = 0; kDim < NDIM + 1; kDim++) {
683  JTJ[iDim][jDim] += Jacobian[kDim][iDim] * Jacobian[kDim][jDim];
684  }
685  }
686  }
687 
688  /*--- matrix volume of Jacobian ---*/
689  const su2double detJacobian = JTJ[0][0] * JTJ[1][1] - JTJ[0][1] * JTJ[1][0];
690 
691  volJacobian = sqrt(detJacobian);
692 
693  GaussPoint[iGauss].SetJ_X(volJacobian);
694 
695  /*--- Derivatives with respect to global coordinates ---*/
696 
697  su2double JTJinv[2][2];
698  JTJinv[0][0] = JTJ[1][1] / detJacobian;
699  JTJinv[1][1] = JTJ[0][0] / detJacobian;
700  JTJinv[0][1] = -JTJ[0][1] / detJacobian;
701  JTJinv[1][0] = -JTJ[1][0] / detJacobian;
702 
703  su2double Jdagger[2][3];
704 
705  for (iDim = 0; iDim < NDIM; iDim++) {
706  for (jDim = 0; jDim < NDIM + 1; jDim++) {
707  Jdagger[iDim][jDim] = 0.0;
708  for (kDim = 0; kDim < NDIM; kDim++) {
709  Jdagger[iDim][jDim] += JTJinv[iDim][kDim] * Jacobian[jDim][kDim];
710  }
711  }
712  }
713 
714  su2double GradOut[NDIM + 1];
715  for (iShape = 0; iShape < NNODE; iShape++) {
716  for (iDim = 0; iDim < NDIM + 1; iDim++) {
717  GradOut[iDim] = 0.0;
718  for (jDim = 0; jDim < NDIM; jDim++) {
719  GradOut[iDim] += Jdagger[jDim][iDim] * dNiXj[iGauss][iShape][jDim];
720  }
721  GaussPoint[iGauss].SetGradNi_Xj(GradOut[iDim], iDim, iShape);
722  }
723  }
724  }
725  }
726  }
