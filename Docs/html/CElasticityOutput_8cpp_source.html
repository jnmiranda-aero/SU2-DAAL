34CElasticityOutput::CElasticityOutput(CConfig *config, unsigned short nDim) : COutput(config, nDim, false) {
35
36 linear_analysis = (config->GetGeometricConditions() == STRUCT_DEFORMATION::SMALL);
37 nonlinear_analysis = (config->GetGeometricConditions() == STRUCT_DEFORMATION::LARGE);
38 dynamic = (config->GetTime_Domain());
39
40 /*--- Initialize number of variables ---*/
41 if (linear_analysis) nVar_FEM = nDim;
42 if (nonlinear_analysis) nVar_FEM = 3;
43
44 /*--- Default fields for screen output ---*/
45 if (nRequestedHistoryFields == 0){
46 RequestCommonHistory(dynamic);
47 nRequestedHistoryFields = requestedHistoryFields.size();
48 }
49
50 /*--- Default fields for screen output ---*/
51 if (nRequestedScreenFields == 0){
52 if (dynamic) requestedScreenFields.emplace_back("TIME_ITER");
53 if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
54 requestedScreenFields.emplace_back("INNER_ITER");
55 if(linear_analysis){
56 requestedScreenFields.emplace_back("RMS_DISP_X");
57 requestedScreenFields.emplace_back("RMS_DISP_Y");
58 requestedScreenFields.emplace_back("RMS_DISP_Z");
59 }
60 if(nonlinear_analysis){
61 requestedScreenFields.emplace_back("RMS_UTOL");
62 requestedScreenFields.emplace_back("RMS_RTOL");
63 requestedScreenFields.emplace_back("RMS_ETOL");
64 }
65 requestedScreenFields.emplace_back("VMS");
66 nRequestedScreenFields = requestedScreenFields.size();
67 }
68
69 /*--- Default fields for volume output ---*/
70 if (nRequestedVolumeFields == 0){
71 requestedVolumeFields.emplace_back("COORDINATES");
72 requestedVolumeFields.emplace_back("SOLUTION");
73 requestedVolumeFields.emplace_back("STRESS");
74 if (dynamic) {
75 requestedVolumeFields.emplace_back("VELOCITY");
76 requestedVolumeFields.emplace_back("ACCELERATION");
77 }
78 if (config->GetTopology_Optimization()) requestedVolumeFields.emplace_back("TOPOLOGY");
79 nRequestedVolumeFields = requestedVolumeFields.size();
80 }
81
82 stringstream ss;
83 ss << "Zone " << config->GetiZone() << " (Structure)";
84 multiZoneHeaderString = ss.str();
85
86 /*--- Set the volume filename --- */
87
88 volumeFilename = config->GetVolume_FileName();
89
90 /*--- Set the surface filename --- */
91
92 surfaceFilename = config->GetSurfCoeff_FileName();
93
94 /*--- Set the restart filename --- */
95
96 restartFilename = config->GetRestart_FileName();
97
98 /*--- Set the default convergence field --- */
99
100 if (convFields.empty()) {
101 if (linear_analysis) convFields.emplace_back("RMS_DISP_X");
102 if (nonlinear_analysis) convFields.emplace_back("RMS_UTOL");
103 }
104}

106void CElasticityOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
107
108 CSolver* fea_solver = solver[FEA_SOL];
109
110 /*--- Residuals: ---*/
111 /*--- Linear analysis: RMS of the displacements in the nDim coordinates ---*/
112 /*--- Nonlinear analysis: UTOL, RTOL and DTOL (defined in the Postprocessing function) ---*/
113
114
115 if (linear_analysis){
116 SetHistoryOutputValue("RMS_DISP_X", log10(fea_solver->GetRes_RMS(0)));
117 SetHistoryOutputValue("RMS_DISP_Y", log10(fea_solver->GetRes_RMS(1)));
118 if (nDim == 3){
119 SetHistoryOutputValue("RMS_DISP_Z", log10(fea_solver->GetRes_RMS(2)));
120 }
121 } else if (nonlinear_analysis){
122 SetHistoryOutputValue("RMS_UTOL", log10(fea_solver->GetRes_FEM(0)));
123 SetHistoryOutputValue("RMS_RTOL", log10(fea_solver->GetRes_FEM(1)));
124 SetHistoryOutputValue("RMS_ETOL", log10(fea_solver->GetRes_FEM(2)));
125 }
126
127 if (multiZone){
128 SetHistoryOutputValue("BGS_DISP_X", log10(fea_solver->GetRes_BGS(0)));
129 SetHistoryOutputValue("BGS_DISP_Y", log10(fea_solver->GetRes_BGS(1)));
130 if (nDim == 3) SetHistoryOutputValue("BGS_DISP_Z", log10(fea_solver->GetRes_BGS(2)));
131 }
132
133 SetHistoryOutputValue("VMS", fea_solver->GetTotal_CFEA());
134 SetHistoryOutputValue("LOAD_INCREMENT", fea_solver->GetLoad_Increment()*100);
135 SetHistoryOutputValue("LOAD_RAMP", fea_solver->GetForceCoeff());
136
137 SetHistoryOutputValue("LINSOL_ITER", fea_solver->GetIterLinSolver());
138 SetHistoryOutputValue("LINSOL_RESIDUAL", log10(fea_solver->GetResLinSolver()));
139
140 SetHistoryOutputValue("REFERENCE_NODE", fea_solver->GetTotal_OFRefNode());
141 SetHistoryOutputValue("TOPOL_COMPLIANCE", fea_solver->GetTotal_OFCompliance());
142 SetHistoryOutputValue("STRESS_PENALTY", fea_solver->GetTotal_OFStressPenalty());
143 if (config->GetRefGeom()) {
144 SetHistoryOutputValue("REFERENCE_GEOMETRY", fea_solver->GetTotal_OFRefGeom());
145 }
146 if (config->GetTopology_Optimization()) {
147 SetHistoryOutputValue("VOLUME_FRACTION", fea_solver->GetTotal_OFVolFrac());
148 SetHistoryOutputValue("TOPOL_DISCRETENESS", fea_solver->GetTotal_OFDiscreteness());
149 }
150
151 ComputeSimpleCustomOutputs(config);
152
153 /*--- Keep this as last, since it uses the history values that were set. ---*/
154 SetCustomAndComboObjectives(FEA_SOL, config, solver);
155
156}

195void CElasticityOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
196
197 CVariable* Node_Struc = solver[FEA_SOL]->GetNodes();
198 CPoint* Node_Geo = geometry->nodes;
199
200 SetVolumeOutputValue("COORD-X", iPoint, Node_Geo->GetCoord(iPoint, 0));
201 SetVolumeOutputValue("COORD-Y", iPoint, Node_Geo->GetCoord(iPoint, 1));
202 if (nDim == 3)
203 SetVolumeOutputValue("COORD-Z", iPoint, Node_Geo->GetCoord(iPoint, 2));
204
205 SetVolumeOutputValue("DISPLACEMENT-X", iPoint, Node_Struc->GetSolution(iPoint, 0));
206 SetVolumeOutputValue("DISPLACEMENT-Y", iPoint, Node_Struc->GetSolution(iPoint, 1));
207 if (nDim == 3) SetVolumeOutputValue("DISPLACEMENT-Z", iPoint, Node_Struc->GetSolution(iPoint, 2));
208
209 if(dynamic){
210 SetVolumeOutputValue("VELOCITY-X", iPoint, Node_Struc->GetSolution_Vel(iPoint, 0));
211 SetVolumeOutputValue("VELOCITY-Y", iPoint, Node_Struc->GetSolution_Vel(iPoint, 1));
212 if (nDim == 3) SetVolumeOutputValue("VELOCITY-Z", iPoint, Node_Struc->GetSolution_Vel(iPoint, 2));
213
214 SetVolumeOutputValue("ACCELERATION-X", iPoint, Node_Struc->GetSolution_Accel(iPoint, 0));
215 SetVolumeOutputValue("ACCELERATION-Y", iPoint, Node_Struc->GetSolution_Accel(iPoint, 1));
216 if (nDim == 3) SetVolumeOutputValue("ACCELERATION-Z", iPoint, Node_Struc->GetSolution_Accel(iPoint, 2));
217 }
218
219 SetVolumeOutputValue("STRESS-XX", iPoint, Node_Struc->GetStress_FEM(iPoint)[0]);
220 SetVolumeOutputValue("STRESS-YY", iPoint, Node_Struc->GetStress_FEM(iPoint)[1]);
221 SetVolumeOutputValue("STRESS-XY", iPoint, Node_Struc->GetStress_FEM(iPoint)[2]);
222 if (nDim == 3){
223 SetVolumeOutputValue("STRESS-ZZ", iPoint, Node_Struc->GetStress_FEM(iPoint)[3]);
224 SetVolumeOutputValue("STRESS-XZ", iPoint, Node_Struc->GetStress_FEM(iPoint)[4]);
225 SetVolumeOutputValue("STRESS-YZ", iPoint, Node_Struc->GetStress_FEM(iPoint)[5]);
226 }
227 SetVolumeOutputValue("VON_MISES_STRESS", iPoint, Node_Struc->GetVonMises_Stress(iPoint));
228
229 if (config->GetTopology_Optimization()) {
230 SetVolumeOutputValue("TOPOL_DENSITY", iPoint, Node_Struc->GetAuxVar(iPoint));
231 }
232}
