37 template <int N>
38 class CPolynomialConductivity final : public CConductivityModel {
39  public:
40  /*!
41  * \brief Constructor of the class.
42  */
43  CPolynomialConductivity(const su2double* val_coeffs) {
44  for (int i = 0; i < N; ++i) {
45  coeffs_[i] = val_coeffs[i];
46  }
47  }
48 
49  /*!
50  * \brief Set thermal conductivity.
51  */
52  void SetConductivity(su2double t, su2double rho, su2double mu_lam, su2double mu_turb, su2double cp,
53  su2double, su2double) override {
54  /* Evaluate the new kt from the coefficients and temperature. */
55  kt_ = coeffs_[0];
56  su2double t_i = 1.0;
57  for (int i = 1; i < N; ++i) {
58  t_i *= t;
59  kt_ += coeffs_[i] * t_i;
60  }
61  }
62 
63  private:
64  array<su2double, N> coeffs_; /*!< \brief Polynomial coefficients for conductivity as a function of temperature. */
65 };
