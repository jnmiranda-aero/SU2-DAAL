52CWallModel1DEQ::CWallModel1DEQ(CConfig* config, const string& Marker_Tag) : CWallModel(config) {
53 /* Retrieve the integer and floating point information for this
54 boundary marker. */
55 const unsigned short* intInfo = config->GetWallFunction_IntInfo(Marker_Tag);
56 const su2double* doubleInfo = config->GetWallFunction_DoubleInfo(Marker_Tag);
57
58 /* Copy the data into the member variables. */
59 numPoints = intInfo[0];
60 h_wm = doubleInfo[0];
61 expansionRatio = doubleInfo[1];
62
63 unsigned short nfa = numPoints + 1;
64 su2double currentHeight = 1.0;
65
66 /* Allocate the memory for the coordinates of the grid points used
67 in the 1D equilibrium wall model. */
68 y_cv.resize(numPoints);
69 y_fa.resize(numPoints + 1);
70
71 /* Determine the scaled version of the normal coordinates, where the
72 first normal coordinate is simply 1.0. */
73
74 y_fa[0] = 0.0;
75 for (unsigned short i = 1; i < nfa; ++i) {
76 y_fa[i] = y_fa[i - 1] + currentHeight;
77 y_cv[i - 1] = 0.5 * (y_fa[i] + y_fa[i - 1]);
78 currentHeight = currentHeight * expansionRatio;
79 }
80
81 su2double y_max = y_cv[numPoints - 1];
82 for (unsigned short i = 0; i < nfa; ++i) {
83 y_fa[i] = y_fa[i] / y_max * h_wm;
84 }
85 for (unsigned short i = 0; i < numPoints; ++i) {
86 y_cv[i] = y_cv[i] / y_max * h_wm;
87 }
88}

90void CWallModel1DEQ::WallShearStressAndHeatFlux(const su2double tExchange, const su2double velExchange,
91 const su2double muExchange, const su2double pExchange,
92 const su2double Wall_HeatFlux, const bool HeatFlux_Prescribed,
93 const su2double TWall, const bool Temperature_Prescribed,
94 CFluidModel* FluidModel, su2double& tauWall, su2double& qWall,
95 su2double& ViscosityWall, su2double& kOverCvWall) {
96 /* Set tau wall to initial guess
97 */
98 tauWall = 0.5;
99 qWall = 0.0;
100 ViscosityWall = 0.0;
101 kOverCvWall = 0.0;
102
103 /* Set some constants, assuming air at standard conditions
104 Todo: Read values below from the config file or from solver.
105 */
106 // su2double C_1 = 2.03929e-04;
107 su2double C_1 = 1.716e-5;
108 su2double S = 110.4;
109 su2double T_ref = 273.15;
110 su2double R = 287.058;
111 su2double A = 17;
112 su2double gamma = 1.4;
113 su2double c_p = (gamma * R) / (gamma - 1);
114 su2double c_v = R / (gamma - 1);
115 su2double h_wall = c_p * TWall;
116 su2double h_bc = c_p * tExchange;
117 unsigned short nfa = numPoints + 1;
118
119 /* Set up vectors
120 */
121 vector<su2double> T(nfa, tExchange);
122 vector<su2double> mu_fa(nfa, muExchange);
123 vector<su2double> tmp(nfa, 0.0);
124 vector<su2double> u(numPoints, 0.0);
125 vector<su2double> lower(numPoints - 1, 0.0);
126 vector<su2double> upper(numPoints - 1, 0.0);
127 vector<su2double> diagonal(numPoints, 0.0);
128 vector<su2double> rhs(numPoints, 0.0);
129
130 /* Set parameters for control
131 */
132 bool converged = false;
133 unsigned short iter = 0, max_iter = 25;
134 su2double tauWall_prev = 0.0, tol = 1e-3, aux_rhs = 0.0;
135 su2double qWall_prev = 0.0;
136 su2double mut, nu, mu_lam, rho, utau, y_plus, D;
137
138 while (!converged) {
139 iter += 1;
140 if (iter == max_iter) converged = true;
141
142 tauWall_prev = tauWall;
143 qWall_prev = qWall;
144
145 /* Calculate total viscosity
146 note: rho and mu_lam will be a function of temperature when solving an energy equation
147 */
148 for (unsigned short i = 0; i < nfa; ++i) {
149 mu_lam = C_1 * pow(T[i] / T_ref, 1.5) * ((T_ref + S) / (T[i] + S));
150 mu_fa[i] = mu_lam;
151 }
152
153 for (unsigned short i = 1; i < nfa; ++i) {
154 rho = pExchange / (R * T[i]);
155 nu = mu_fa[i] / rho;
156 utau = sqrt(tauWall / rho);
157 y_plus = y_fa[i] * utau / nu;
158 D = pow(1.0 - exp((-y_plus) / A), 2.0);
159 mut = rho * karman * y_fa[i] * utau * D;
160 mu_fa[i] += mut;
161 }
162
163 /* Momentum matrix
164 The solution vector is u at y_cv
165 */
166 lower.assign(numPoints - 1, 0.0);
167 upper.assign(numPoints - 1, 0.0);
168 diagonal.assign(numPoints, 0.0);
169 rhs.assign(numPoints, 0.0);
170
171 /* Top bc
172 */
173 diagonal[numPoints - 1] = 1.0;
174 rhs[numPoints - 1] = velExchange;
175
176 /* Internal cvs
177 */
178 for (unsigned short i = 1; i < (numPoints - 1); ++i) {
179 upper[i] = mu_fa[i + 1] / (y_cv[i + 1] - y_cv[i]);
180 lower[i - 1] = mu_fa[i] / (y_cv[i] - y_cv[i - 1]);
181 diagonal[i] = -1.0 * (upper[i] + lower[i - 1]);
182 }
183
184 /* Wall BC
185 */
186 upper[0] = mu_fa[1] / (y_cv[1] - y_cv[0]);
187 diagonal[0] = -1.0 * (upper[0] + mu_fa[0] / (y_cv[0] - y_fa[0]));
188 rhs[0] = 0.0;
189
190 /* Solve the matrix problem to get the velocity field
191 - rhs returns the solution
192 */
193#if (defined(HAVE_MKL) || defined(HAVE_LAPACK)) && !(defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE))
194 int info, nrhs = 1;
195
196 dgtsv_(&numPoints, &nrhs, lower.data(), diagonal.data(), upper.data(), rhs.data(), &numPoints, &info);
197 if (info != 0) SU2_MPI::Error("Unsuccessful call to dgtsv_", CURRENT_FUNCTION);
198#else
199 SU2_MPI::Error("Not compiled with MKL or LAPACK support", CURRENT_FUNCTION);
200#endif
201
202 u = rhs;
203
204 /* Update total viscosity
205 */
206 for (unsigned short i = 0; i < nfa; ++i) {
207 mu_lam = C_1 * pow(T[i] / T_ref, 1.5) * ((T_ref + S) / (T[i] + S));
208 mu_fa[i] = mu_lam;
209 tmp[i] = mu_lam / Pr_lam;
210 }
211 /* Update tauWall
212 */
213 tauWall = mu_fa[0] * (u[0] - 0.0) / (y_cv[0] - y_fa[0]);
214 for (unsigned short i = 1; i < nfa; ++i) {
215 rho = pExchange / (R * T[i]);
216 nu = mu_fa[i] / rho;
217 utau = sqrt(tauWall / rho);
218 y_plus = y_fa[i] * utau / nu;
219 D = pow(1.0 - exp((-y_plus) / A), 2.0);
220 mut = rho * karman * y_fa[i] * utau * D;
221 mu_fa[i] += mut;
222 tmp[i] += mut / Pr_turb;
223 }
224
225 /* Energy matrix
226 The Solution vector is Enthalpy at y_cv
227 */
228 lower.assign(numPoints - 1, 0.0);
229 upper.assign(numPoints - 1, 0.0);
230 diagonal.assign(numPoints, 0.0);
231 rhs.assign(numPoints, 0.0);
232
233 /* Internal cvs
234 */
235 for (unsigned short i = 1; i < (numPoints - 1); ++i) {
236 upper[i] = tmp[i + 1] / (y_cv[i + 1] - y_cv[i]);
237 lower[i - 1] = tmp[i] / (y_cv[i] - y_cv[i - 1]);
238 diagonal[i] = -1.0 * (upper[i] + lower[i - 1]);
239 }
240
241 /* Top BC
242 */
243 diagonal[numPoints - 1] = 1.0;
244
245 /* Wall BC
246 */
247 upper[0] = tmp[1] / (y_cv[1] - y_cv[0]);
248 diagonal[0] = -1.0 * (upper[0] + tmp[0] / (y_cv[0] - y_fa[0]));
249 aux_rhs = tmp[0] / (y_cv[0] - y_fa[0]);
250
251 /* RHS of the Energy equation
252 - Compute flux -- (mu + mu_t) * u * du/dy --
253 */
254
255 /* Zero flux at the wall
256 */
257 tmp[0] = 0.;
258 for (unsigned short i = 1; i < numPoints; ++i) {
259 tmp[i] = 0.5 * (mu_fa[i]) * (u[i] + u[i - 1]) * (u[i] - u[i - 1]) / (y_cv[i] - y_cv[i - 1]);
260 }
261 for (unsigned short i = 0; i < (numPoints - 1); ++i) {
262 rhs[i] = -tmp[i + 1] + tmp[i];
263 }
264
265 if (HeatFlux_Prescribed) {
266 /* dT/dy = 0 -> Twall = T[1] */
267 h_wall = c_p * T[1];
268 }
269
270 rhs[0] -= aux_rhs * h_wall;
271 rhs[numPoints - 1] = h_bc;
272
273 /* Solve the matrix problem to get the Enthalpy field
274 */
275#if (defined(HAVE_MKL) || defined(HAVE_LAPACK)) && !(defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE))
276 dgtsv_(&numPoints, &nrhs, lower.data(), diagonal.data(), upper.data(), rhs.data(), &numPoints, &info);
277 if (info != 0) SU2_MPI::Error("Unsuccessful call to dgtsv_", CURRENT_FUNCTION);
278#else
279 SU2_MPI::Error("Not compiled with MKL or LAPACK support", CURRENT_FUNCTION);
280#endif
281
282 /* Get Temperature from enthalpy
283 - Temperature will be at face
284 */
285 T[0] = h_wall / c_p;
286 T[numPoints] = h_bc / c_p;
287 for (unsigned short i = 0; i < numPoints - 1; i++) {
288 T[i + 1] = 0.5 * (rhs[i] + rhs[i + 1]) / c_p;
289 }
290
291 /* Final update tauWall
292 */
293 mu_lam = C_1 * pow(T[0] / T_ref, 1.5) * ((T_ref + S) / (T[0] + S));
294
295 /* These quantities will be returned.
296 */
297 tauWall = mu_lam * (u[0] - 0.0) / (y_cv[0] - y_fa[0]);
298 qWall = mu_lam * (c_p / Pr_lam) * -(T[1] - T[0]) / (y_fa[1] - y_fa[0]);
299 ViscosityWall = mu_lam;
300 // kOverCvWall = c_p / c_v * (mu[0]/Pr_lam + muTurb[0]/Pr_turb);
301 kOverCvWall = c_p / c_v * (mu_lam / Pr_lam);
302
303 /* Final check of the Y+
304 */
305 rho = pExchange / (R * T[0]);
306 if (y_cv[0] * sqrt(tauWall / rho) / (mu_lam / rho) > 1.0)
307 SU2_MPI::Error("Y+ greater than one: Increase the number of points or growth ratio.", CURRENT_FUNCTION);
308
309 /* Define a norm
310 */
311 if (abs(1.0 - tauWall / tauWall_prev) < tol && abs(1.0 - qWall / qWall_prev) < tol) {
312 converged = true;
313 }
314 }
315}

326void CWallModelLogLaw::WallShearStressAndHeatFlux(const su2double tExchange, const su2double velExchange,
327 const su2double muExchange, const su2double pExchange,
328 const su2double Wall_HeatFlux, const bool HeatFlux_Prescribed,
329 const su2double Wall_Temperature, const bool Temperature_Prescribed,
330 CFluidModel* FluidModel, su2double& tauWall, su2double& qWall,
331 su2double& ViscosityWall, su2double& kOverCvWall) {
332 /* Set the wall temperature, depending whether or not the temperature
333 was prescribed and initialize the fluid model. */
334 const su2double TWall = Temperature_Prescribed ? Wall_Temperature : tExchange;
335
336 FluidModel->SetTDState_PT(pExchange, TWall);
337
338 /* Get the required data from the fluid model. */
339 const su2double rho_wall = FluidModel->GetDensity();
340 const su2double mu_wall = FluidModel->GetLaminarViscosity();
341 const su2double c_p = FluidModel->GetCp();
342 const su2double c_v = FluidModel->GetCv();
343 const su2double nu_wall = mu_wall / rho_wall;
344
345 /* Initial guess of the friction velocity. */
346 su2double u_tau = max(0.01 * velExchange, 1.e-5);
347
348 /* Set parameters for control of the Newton iteration. */
349 bool converged = false;
350 unsigned short iter = 0, max_iter = 50;
351 const su2double tol = 1e-3;
352
353 while (!converged) {
354 iter += 1;
355 if (iter == max_iter) converged = true;
356
357 const su2double u_tau0 = u_tau;
358 const su2double y_plus = u_tau0 * h_wm / nu_wall;
359
360 /* Reichardt boundary layer analytical law
361 fprime is the differentiation of the Reichardt law with repect to u_tau.
362 */
363 const su2double fval =
364 velExchange / u_tau0 -
365 ((C - log(karman) / karman) * (1.0 - exp(-y_plus / 11.0) - (y_plus / 11.0) * exp(-0.33 * y_plus))) -
366 log(karman * y_plus + 1.0) / karman;
367 const su2double fprime = -velExchange / pow(u_tau0, 2.0) +
368 (-C + log(karman) / karman) *
369 (-(1.0 / 11.0) * h_wm * exp(-0.33 * y_plus) / nu_wall +
370 (1.0 / 11.0) * h_wm * exp(-(1.0 / 11.0) * y_plus) / nu_wall +
371 (1.0 / 33.0) * u_tau0 * pow(h_wm, 2.0) * exp(-0.33 * y_plus) / pow(nu_wall, 2.0)) -
372 1.0 * h_wm / (nu_wall * (karman * y_plus + 1.0));
373
374 /* Newton method
375 */
376 const su2double newton_step = fval / fprime;
377 u_tau = u_tau0 - newton_step;
378
379 /* Define a norm
380 */
381 if (abs(1.0 - u_tau / u_tau0) < tol) converged = true;
382 }
383
384 tauWall = rho_wall * pow(u_tau, 2.0);
385
386 if (Temperature_Prescribed) {
387 /* The Kader's law will be used to approximate the variations of the temperature inside the boundary layer.
388 */
389 const su2double y_plus = u_tau * h_wm / nu_wall;
390 const su2double lhs = -((tExchange - TWall) * rho_wall * c_p * u_tau);
391 const su2double Gamma = -(0.01 * (Pr_lam * pow(y_plus, 4.0)) / (1.0 + 5.0 * y_plus * pow(Pr_lam, 3.0)));
392 const su2double rhs_1 = Pr_lam * y_plus * exp(Gamma);
393 const su2double rhs_2 =
394 (2.12 * log(1.0 + y_plus) + pow((3.85 * pow(Pr_lam, (1.0 / 3.0)) - 1.3), 2.0) + 2.12 * log(Pr_lam)) *
395 exp(1. / Gamma);
396 qWall = lhs / (rhs_1 + rhs_2);
397 } else {
398 qWall = Wall_HeatFlux;
399 }
400
401 ViscosityWall = mu_wall;
402 kOverCvWall = FluidModel->GetThermalConductivity() / c_v;
403}
