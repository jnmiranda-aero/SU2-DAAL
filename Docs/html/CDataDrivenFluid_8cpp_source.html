34 CDataDrivenFluid::CDataDrivenFluid(const CConfig* config, bool display) : CFluidModel() {
35  rank = SU2_MPI::GetRank();
36  Kind_DataDriven_Method = config->GetKind_DataDriven_Method();
37 
38  varname_rho = "Density";
39  varname_e = "Energy";
40 
41  /*--- Set up interpolation algorithm according to data-driven method. Currently only MLP's are supported. ---*/
42  switch (Kind_DataDriven_Method) {
43  case ENUM_DATADRIVEN_METHOD::MLP:
44 #ifdef USE_MLPCPP
45  lookup_mlp = new MLPToolbox::CLookUp_ANN(config->GetNDataDriven_Files(), config->GetDataDriven_FileNames());
46  if ((rank == MASTER_NODE) && display) lookup_mlp->DisplayNetworkInfo();
47 #else
48  SU2_MPI::Error("SU2 was not compiled with MLPCpp enabled (-Denable-mlpcpp=true).", CURRENT_FUNCTION);
49 #endif
50  break;
51  case ENUM_DATADRIVEN_METHOD::LUT:
52  lookup_table = new CLookUpTable(config->GetDataDriven_FileNames()[0], varname_rho, varname_e);
53  break;
54  default:
55  break;
56  }
57 
58  /*--- Relaxation factor and tolerance for Newton solvers. ---*/
59  Newton_Relaxation = config->GetRelaxation_DataDriven();
60  Newton_Tolerance = 1e-10;
61  MaxIter_Newton = 50;
62 
63  /*--- Preprocessing of inputs and outputs for the interpolation method. ---*/
64  MapInputs_to_Outputs();
65 
66  /*--- Compute approximate ideal gas properties ---*/
67  ComputeIdealGasQuantities();
68 }

139 void CDataDrivenFluid::SetTDState_rhoe(su2double rho, su2double e) {
140  /*--- Compute thermodynamic state based on density and energy. ---*/
141  Density = rho;
142  StaticEnergy = e;
143 
144  /*--- Clip density and energy values to prevent extrapolation. ---*/
145  Density = min(rho_max, max(rho_min, Density));
146  StaticEnergy = min(e_max, max(e_min, StaticEnergy));
147 
148  Evaluate_Dataset(Density, StaticEnergy);
149 
150  /*--- Compute speed of sound. ---*/
151  auto blue_term = (dsdrho_e * (2 - rho * pow(dsde_rho, -1) * d2sdedrho) + rho * d2sdrho2);
152  auto green_term = (-pow(dsde_rho, -1) * d2sde2 * dsdrho_e + d2sdedrho);
153 
154  SoundSpeed2 = -rho * pow(dsde_rho, -1) * (blue_term - rho * green_term * (dsdrho_e / dsde_rho));
155 
156  /*--- Compute primary flow variables. ---*/
157  Temperature = 1.0 / dsde_rho;
158  Pressure = -pow(rho, 2) * Temperature * dsdrho_e;
159  Density = rho;
160  StaticEnergy = e;
161  Enthalpy = e + Pressure / rho;
162 
163  /*--- Compute secondary flow variables ---*/
164  dTde_rho = -pow(dsde_rho, -2) * d2sde2;
165  dTdrho_e = -pow(dsde_rho, -2) * d2sdedrho;
166 
167  dPde_rho = -pow(rho, 2) * (dTde_rho * dsdrho_e + Temperature * d2sdedrho);
168  dPdrho_e = -2 * rho * Temperature * dsdrho_e - pow(rho, 2) * (dTdrho_e * dsdrho_e + Temperature * d2sdrho2);
169 
170  /*--- Compute enthalpy and entropy derivatives required for Giles boundary conditions. ---*/
171  dhdrho_e = -Pressure * pow(rho, -2) + dPdrho_e / rho;
172  dhde_rho = 1 + dPde_rho / rho;
173 
174  dhdrho_P = dhdrho_e - dhde_rho * (1 / dPde_rho) * dPdrho_e;
175  dhdP_rho = dhde_rho * (1 / dPde_rho);
176  dsdrho_P = dsdrho_e - dPdrho_e * (1 / dPde_rho) * dsde_rho;
177  dsdP_rho = dsde_rho / dPde_rho;
178 }

195 void CDataDrivenFluid::SetEnergy_Prho(su2double P, su2double rho) {
196  /*--- Run 1D Newton solver for pressure at constant density. ---*/
197  Density = rho;
198 
199  /*--- Approximate static energy through ideal gas law. ---*/
200  su2double e_idealgas = Cv_idealgas * (P / (R_idealgas * rho));
201  StaticEnergy = min(e_max, max(e_idealgas, e_min));
202 
203  Run_Newton_Solver(P, &Pressure, &StaticEnergy, &dPde_rho);
204 }

216 void CDataDrivenFluid::SetTDState_hs(su2double h, su2double s) {
217  /*--- Run 2D Newton solver for enthalpy and entropy. ---*/
218 
219  /*--- Approximate density and static energy through ideal gas law under isentropic assumption. ---*/
220  su2double T_init = h / Cp_idealgas;
221  su2double P_init = P_middle * pow(T_init / T_middle, gamma_idealgas/(gamma_idealgas - 1));
222 
223  e_start = h * Cv_idealgas / Cp_idealgas;
224  rho_start = P_init / (R_idealgas * T_init);
225  Run_Newton_Solver(h, s, &Enthalpy, &Entropy, &dhdrho_e, &dhde_rho, &dsdrho_e, &dsde_rho);
226 }

271 void CDataDrivenFluid::Run_Newton_Solver(su2double Y1_target, su2double Y2_target, su2double* Y1, su2double* Y2,
272  su2double* dY1drho, su2double* dY1de, su2double* dY2drho, su2double* dY2de) {
273  /*--- 2D Newton solver, computing the density and internal energy values corresponding to Y1_target and Y2_target.
274  * ---*/
275 
276  /*--- Setting initial values for density and energy. ---*/
277  su2double rho = rho_start, e = e_start;
278 
279  bool converged = false;
280  unsigned long Iter = 0;
281 
282  su2double delta_Y1, delta_Y2, delta_rho, delta_e, determinant;
283 
284  /*--- Initiating Newton solver ---*/
285  while (!converged && (Iter < MaxIter_Newton)) {
286  /*--- Determine thermodynamic state based on current density and energy. ---*/
287  SetTDState_rhoe(rho, e);
288 
289  /*--- Determine residuals. ---*/
290  delta_Y1 = *Y1 - Y1_target;
291  delta_Y2 = *Y2 - Y2_target;
292 
293  /*--- Continue iterative process if residuals are outside tolerances. ---*/
294  if ((abs(delta_Y1 / *Y1) < Newton_Tolerance) && (abs(delta_Y2 / *Y2) < Newton_Tolerance)) {
295  converged = true;
296  } else {
297  /*--- Compute step size for density and energy. ---*/
298  determinant = (*dY1drho) * (*dY2de) - (*dY1de) * (*dY2drho);
299 
300  delta_rho = (*dY2de * delta_Y1 - *dY1de * delta_Y2) / determinant;
301  delta_e = (-*dY2drho * delta_Y1 + *dY1drho * delta_Y2) / determinant;
302 
303  /*--- Update density and energy values. ---*/
304  rho -= Newton_Relaxation * delta_rho;
305  e -= Newton_Relaxation * delta_e;
306  }
307  Iter++;
308  }
309  nIter_Newton = Iter;
310 
311  /*--- Evaluation of final state. ---*/
312  SetTDState_rhoe(rho, e);
313 }

315 void CDataDrivenFluid::Run_Newton_Solver(su2double Y_target, su2double* Y, su2double* X, su2double* dYdX) {
316  /*--- 1D Newton solver, computing the density or internal energy value corresponding to Y_target. ---*/
317 
318  bool converged = false;
319  unsigned long Iter = 0;
320 
321  su2double delta_Y, delta_X;
322 
323  /*--- Initiating Newton solver. ---*/
324  while (!converged && (Iter < MaxIter_Newton)) {
325  /*--- Determine thermodynamic state based on current density and energy. ---*/
326  SetTDState_rhoe(Density, StaticEnergy);
327 
328  /*--- Determine residual ---*/
329  delta_Y = Y_target - *Y;
330 
331  /*--- Continue iterative process if residuals are outside tolerances. ---*/
332  if (abs(delta_Y / *Y) < Newton_Tolerance) {
333  converged = true;
334  } else {
335  delta_X = delta_Y / *dYdX;
336 
337  /*--- Update energy value ---*/
338  *X += Newton_Relaxation * delta_X;
339  }
340  Iter++;
341  }
342 
343  /*--- Calculate thermodynamic state based on converged values for density and energy. ---*/
344  SetTDState_rhoe(Density, StaticEnergy);
345 
346  nIter_Newton = Iter;
347 }

349 void CDataDrivenFluid::ComputeIdealGasQuantities() {
350  /*--- Compute approximate ideal gas properties from the middle of the reference data set. These properties are used to approximate the initial condition of the Newton solvers using the ideal gas law. ---*/
351  su2double rho_average = 1.0, e_average = 1.0;
352 
353  /*--- Obtain minimum and maximum density and static energy from data set. ---*/
354  switch (Kind_DataDriven_Method)
355  {
356  case ENUM_DATADRIVEN_METHOD::LUT:
357  rho_min = *lookup_table->GetTableLimitsX().first;
358  e_min = *lookup_table->GetTableLimitsY().first;
359  rho_max = *lookup_table->GetTableLimitsX().second;
360  e_max = *lookup_table->GetTableLimitsY().second;
361  rho_average = 0.5*(*lookup_table->GetTableLimitsX().first + *lookup_table->GetTableLimitsX().second);
362  e_average = 0.5*(*lookup_table->GetTableLimitsY().first + *lookup_table->GetTableLimitsY().second);
363  break;
364  case ENUM_DATADRIVEN_METHOD::MLP:
365 #ifdef USE_MLPCPP
366  rho_min = lookup_mlp->GetInputNorm(iomap_rhoe, idx_rho).first;
367  e_min = lookup_mlp->GetInputNorm(iomap_rhoe, idx_e).first;
368  rho_max = lookup_mlp->GetInputNorm(iomap_rhoe, idx_rho).second;
369  e_max = lookup_mlp->GetInputNorm(iomap_rhoe, idx_e).second;
370  rho_average = 0.5*(lookup_mlp->GetInputNorm(iomap_rhoe, idx_rho).first + lookup_mlp->GetInputNorm(iomap_rhoe, idx_rho).second);
371  e_average = 0.5*(lookup_mlp->GetInputNorm(iomap_rhoe, idx_e).first + lookup_mlp->GetInputNorm(iomap_rhoe, idx_e).second);
372 #endif
373  break;
374  default:
375  break;
376  }
377 
378  /*--- Compute thermodynamic state from middle of data set. ---*/
379  SetTDState_rhoe(rho_average, e_average);
380  P_middle = Pressure;
381  T_middle = Temperature;
382 
383  R_idealgas = P_middle / (rho_average * T_middle);
384  Cv_idealgas = e_average / T_middle;
385  Cp_idealgas = Enthalpy / T_middle;
386  gamma_idealgas = (R_idealgas / Cv_idealgas) + 1;
387 }
