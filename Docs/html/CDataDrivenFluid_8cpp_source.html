34CDataDrivenFluid::CDataDrivenFluid(const CConfig* config, bool display) : CFluidModel() {
35 rank = SU2_MPI::GetRank();
36 Kind_DataDriven_Method = config->GetKind_DataDriven_Method();
37
38 varname_rho = "Density";
39 varname_e = "Energy";
40
41 /*--- Set up interpolation algorithm according to data-driven method. Currently only MLP's are supported. ---*/
42 switch (Kind_DataDriven_Method) {
43 case ENUM_DATADRIVEN_METHOD::MLP:
44#ifdef USE_MLPCPP
45 lookup_mlp = new MLPToolbox::CLookUp_ANN(config->GetNDataDriven_Files(), config->GetDataDriven_FileNames());
46 if ((rank == MASTER_NODE) && display) lookup_mlp->DisplayNetworkInfo();
47#else
48 SU2_MPI::Error("SU2 was not compiled with MLPCpp enabled (-Denable-mlpcpp=true).", CURRENT_FUNCTION);
49#endif
50 break;
51 case ENUM_DATADRIVEN_METHOD::LUT:
52 lookup_table = new CLookUpTable(config->GetDataDriven_FileNames()[0], varname_rho, varname_e);
53 break;
54 default:
55 break;
56 }
57
58 /*--- Relaxation factor and tolerance for Newton solvers. ---*/
59 Newton_Relaxation = config->GetRelaxation_DataDriven();
60 Newton_Tolerance = 1e-10;
61 MaxIter_Newton = 50;
62
63 /*--- Preprocessing of inputs and outputs for the interpolation method. ---*/
64 MapInputs_to_Outputs();
65
66 /*--- Compute approximate ideal gas properties ---*/
67 ComputeIdealGasQuantities();
68}

85void CDataDrivenFluid::MapInputs_to_Outputs() {
86 /*--- Inputs of the data-driven method are density and internal energy. ---*/
87 input_names_rhoe.resize(2);
88 idx_rho = 0;
89 idx_e = 1;
90 input_names_rhoe[idx_rho] = varname_rho;
91 input_names_rhoe[idx_e] = varname_e;
92
93 /*--- Required outputs for the interpolation method are entropy and its partial derivatives with respect to energy and
94 * density. ---*/
95 size_t n_outputs = 6;
96 size_t idx_s = 0, idx_dsde_rho = 1, idx_dsdrho_e = 2, idx_d2sde2 = 3, idx_d2sdedrho = 4, idx_d2sdrho2 = 5;
97
98 outputs_rhoe.resize(n_outputs);
99 output_names_rhoe.resize(n_outputs);
100 output_names_rhoe[idx_s] = "s";
101 outputs_rhoe[idx_s] = &Entropy;
102 output_names_rhoe[idx_dsde_rho] = "dsde_rho";
103 outputs_rhoe[idx_dsde_rho] = &dsde_rho;
104 output_names_rhoe[idx_dsdrho_e] = "dsdrho_e";
105 outputs_rhoe[idx_dsdrho_e] = &dsdrho_e;

124void CDataDrivenFluid::SetTDState_rhoe(su2double rho, su2double e) {
125 /*--- Compute thermodynamic state based on density and energy. ---*/
126 Density = rho;
127 StaticEnergy = e;
128
129 /*--- Clip density and energy values to prevent extrapolation. ---*/
130 Density = min(rho_max, max(rho_min, Density));
131 StaticEnergy = min(e_max, max(e_min, StaticEnergy));
132
133 Evaluate_Dataset(Density, StaticEnergy);
134
135 /*--- Compute speed of sound. ---*/
136 auto blue_term = (dsdrho_e * (2 - rho * pow(dsde_rho, -1) * d2sdedrho) + rho * d2sdrho2);
137 auto green_term = (-pow(dsde_rho, -1) * d2sde2 * dsdrho_e + d2sdedrho);
138
139 SoundSpeed2 = -rho * pow(dsde_rho, -1) * (blue_term - rho * green_term * (dsdrho_e / dsde_rho));
140
141 /*--- Compute primary flow variables. ---*/
142 Temperature = 1.0 / dsde_rho;
143 Pressure = -pow(rho, 2) * Temperature * dsdrho_e;
144 Density = rho;
145 StaticEnergy = e;
146 Enthalpy = e + Pressure / rho;
147
148 /*--- Compute secondary flow variables ---*/
149 dTde_rho = -pow(dsde_rho, -2) * d2sde2;
150 dTdrho_e = -pow(dsde_rho, -2) * d2sdedrho;
151
152 dPde_rho = -pow(rho, 2) * (dTde_rho * dsdrho_e + Temperature * d2sdedrho);
153 dPdrho_e = -2 * rho * Temperature * dsdrho_e - pow(rho, 2) * (dTdrho_e * dsdrho_e + Temperature * d2sdrho2);
154
155 /*--- Compute enthalpy and entropy derivatives required for Giles boundary conditions. ---*/
156 dhdrho_e = -Pressure * pow(rho, -2) + dPdrho_e / rho;
157 dhde_rho = 1 + dPde_rho / rho;
158
159 dhdrho_P = dhdrho_e - dhde_rho * (1 / dPde_rho) * dPdrho_e;
160 dhdP_rho = dhde_rho * (1 / dPde_rho);
161 dsdrho_P = dsdrho_e - dPdrho_e * (1 / dPde_rho) * dsde_rho;
162 dsdP_rho = dsde_rho / dPde_rho;
163}

165void CDataDrivenFluid::SetTDState_PT(su2double P, su2double T) {
166
167 /*--- Approximate density and static energy with ideal gas law. ---*/
168 rho_start = P / (R_idealgas * T);
169 e_start = Cv_idealgas * T;
170
171 /*--- Run 2D Newton solver for pressure and temperature ---*/
172 Run_Newton_Solver(P, T, &Pressure, &Temperature, &dPdrho_e, &dPde_rho, &dTdrho_e, &dTde_rho);
173}

180void CDataDrivenFluid::SetEnergy_Prho(su2double P, su2double rho) {
181 /*--- Run 1D Newton solver for pressure at constant density. ---*/
182 Density = rho;
183
184 /*--- Approximate static energy through ideal gas law. ---*/
185 su2double e_idealgas = Cv_idealgas * (P / (R_idealgas * rho));
186 StaticEnergy = min(e_max, max(e_idealgas, e_min));
187
188 Run_Newton_Solver(P, &Pressure, &StaticEnergy, &dPde_rho);
189}

201void CDataDrivenFluid::SetTDState_hs(su2double h, su2double s) {
202 /*--- Run 2D Newton solver for enthalpy and entropy. ---*/
203
204 /*--- Approximate density and static energy through ideal gas law under isentropic assumption. ---*/
205 su2double T_init = h / Cp_idealgas;
206 su2double P_init = P_middle * pow(T_init / T_middle, gamma_idealgas/(gamma_idealgas - 1));
207
208 e_start = h * Cv_idealgas / Cp_idealgas;
209 rho_start = P_init / (R_idealgas * T_init);
210 Run_Newton_Solver(h, s, &Enthalpy, &Entropy, &dhdrho_e, &dhde_rho, &dsdrho_e, &dsde_rho);
211}

213void CDataDrivenFluid::SetTDState_Ps(su2double P, su2double s) {
214 /*--- Run 2D Newton solver for pressure and entropy ---*/
215
216 /*--- Approximate initial state through isentropic assumption and ideal gas law. ---*/
217 su2double T_init = T_middle * pow(P / P_middle, (gamma_idealgas - 1)/gamma_idealgas);
218 e_start = Cv_idealgas * T_init;
219 rho_start = P / (R_idealgas * T_init);
220
221 Run_Newton_Solver(P, s, &Pressure, &Entropy, &dPdrho_e, &dPde_rho, &dsdrho_e, &dsde_rho);
222}

267void CDataDrivenFluid::Run_Newton_Solver(su2double Y1_target, su2double Y2_target, su2double* Y1, su2double* Y2,
268 su2double* dY1drho, su2double* dY1de, su2double* dY2drho, su2double* dY2de) {
269 /*--- 2D Newton solver, computing the density and internal energy values corresponding to Y1_target and Y2_target.
270 * ---*/
271
272 /*--- Setting initial values for density and energy. ---*/
273 su2double rho = rho_start, e = e_start;
274
275 bool converged = false;
276 unsigned long Iter = 0;
277
278 su2double delta_Y1, delta_Y2, delta_rho, delta_e, determinant;
279
280 /*--- Initiating Newton solver ---*/
281 while (!converged && (Iter < MaxIter_Newton)) {
282 /*--- Determine thermodynamic state based on current density and energy. ---*/
283 SetTDState_rhoe(rho, e);
284
285 /*--- Determine residuals. ---*/
286 delta_Y1 = *Y1 - Y1_target;
287 delta_Y2 = *Y2 - Y2_target;
288
289 /*--- Continue iterative process if residuals are outside tolerances. ---*/
290 if ((abs(delta_Y1 / *Y1) < Newton_Tolerance) && (abs(delta_Y2 / *Y2) < Newton_Tolerance)) {
291 converged = true;
292 } else {
293 /*--- Compute step size for density and energy. ---*/
294 determinant = (*dY1drho) * (*dY2de) - (*dY1de) * (*dY2drho);
295
296 delta_rho = (*dY2de * delta_Y1 - *dY1de * delta_Y2) / determinant;
297 delta_e = (-*dY2drho * delta_Y1 + *dY1drho * delta_Y2) / determinant;
298
299 /*--- Update density and energy values. ---*/
300 rho -= Newton_Relaxation * delta_rho;
301 e -= Newton_Relaxation * delta_e;
302 }
303 Iter++;
304 }
305 nIter_Newton = Iter;
306
307 /*--- Evaluation of final state. ---*/
308 SetTDState_rhoe(rho, e);
309}

311void CDataDrivenFluid::Run_Newton_Solver(su2double Y_target, su2double* Y, su2double* X, su2double* dYdX) {
312 /*--- 1D Newton solver, computing the density or internal energy value corresponding to Y_target. ---*/
313
314 bool converged = false;
315 unsigned long Iter = 0;
316
317 su2double delta_Y, delta_X;
318
319 /*--- Initiating Newton solver. ---*/
320 while (!converged && (Iter < MaxIter_Newton)) {
321 /*--- Determine thermodynamic state based on current density and energy. ---*/
322 SetTDState_rhoe(Density, StaticEnergy);
323
324 /*--- Determine residual ---*/
325 delta_Y = Y_target - *Y;
326
327 /*--- Continue iterative process if residuals are outside tolerances. ---*/
328 if (abs(delta_Y / *Y) < Newton_Tolerance) {
329 converged = true;
330 } else {
331 delta_X = delta_Y / *dYdX;
332
333 /*--- Update energy value ---*/
334 *X += Newton_Relaxation * delta_X;
335 }
336 Iter++;
337 }
338
339 /*--- Calculate thermodynamic state based on converged values for density and energy. ---*/
340 SetTDState_rhoe(Density, StaticEnergy);
341
342 nIter_Newton = Iter;
343}

345void CDataDrivenFluid::ComputeIdealGasQuantities() {
346 /*--- Compute approximate ideal gas properties from the middle of the reference data set. These properties are used to approximate the initial condition of the Newton solvers using the ideal gas law. ---*/
347 su2double rho_average = 1.0, e_average = 1.0;
348
349 /*--- Obtain minimum and maximum density and static energy from data set. ---*/
350 switch (Kind_DataDriven_Method)
351 {
352 case ENUM_DATADRIVEN_METHOD::LUT:
353 rho_min = *lookup_table->GetTableLimitsX().first;
354 e_min = *lookup_table->GetTableLimitsY().first;
355 rho_max = *lookup_table->GetTableLimitsX().second;
356 e_max = *lookup_table->GetTableLimitsY().second;
357 rho_average = 0.5*(*lookup_table->GetTableLimitsX().first + *lookup_table->GetTableLimitsX().second);
358 e_average = 0.5*(*lookup_table->GetTableLimitsY().first + *lookup_table->GetTableLimitsY().second);
359 break;
360 case ENUM_DATADRIVEN_METHOD::MLP:
361#ifdef USE_MLPCPP
362 rho_min = lookup_mlp->GetInputNorm(iomap_rhoe, idx_rho).first;
363 e_min = lookup_mlp->GetInputNorm(iomap_rhoe, idx_e).first;
364 rho_max = lookup_mlp->GetInputNorm(iomap_rhoe, idx_rho).second;
365 e_max = lookup_mlp->GetInputNorm(iomap_rhoe, idx_e).second;
366 rho_average = 0.5*(lookup_mlp->GetInputNorm(iomap_rhoe, idx_rho).first + lookup_mlp->GetInputNorm(iomap_rhoe, idx_rho).second);
367 e_average = 0.5*(lookup_mlp->GetInputNorm(iomap_rhoe, idx_e).first + lookup_mlp->GetInputNorm(iomap_rhoe, idx_e).second);
368#endif
369 break;
370 default:
371 break;
372 }
373
374 /*--- Compute thermodynamic state from middle of data set. ---*/
375 SetTDState_rhoe(rho_average, e_average);
376 P_middle = Pressure;
377 T_middle = Temperature;
378
379 R_idealgas = P_middle / (rho_average * T_middle);
380 Cv_idealgas = e_average / T_middle;
381 Cp_idealgas = Enthalpy / T_middle;
382 gamma_idealgas = (R_idealgas / Cv_idealgas) + 1;
383}
