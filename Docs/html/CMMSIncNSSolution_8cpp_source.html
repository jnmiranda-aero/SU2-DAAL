32CMMSIncNSSolution::CMMSIncNSSolution(unsigned short val_nDim, unsigned short val_nVar, unsigned short val_iMesh,
33 CConfig* config)
34 : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
35 /*--- Write a message that the solution is initialized for the manufactured
36 solution for the incompressible Navier-Stokes equations. ---*/
37 if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
38 cout << endl;
39 cout << "Warning: Fluid properties and solution are being " << endl;
40 cout << " initialized for the manufactured solution " << endl;
41 cout << " of the incompressible Navier-Stokes equations!!!" << endl;
42 cout << endl << flush;
43 }
44
45 /*--- Coefficients, needed to determine the solution. ---*/
46 Viscosity = config->GetViscosity_FreeStreamND();
47 Density = config->GetDensity_FreeStreamND();
48 Temperature = config->GetTemperature_FreeStreamND();
49
50 /*--- Constants, which describe this manufactured solution. This is a
51 viscous solution where the primitive variables vary as a combination
52 of sine and cosine functions. The solution is from Salari K, and
53 Knupp P, "Code verification by the method of manufactured solutions,"
54 SAND 2000-1444, Sandia National Laboratories, Albuquerque, NM, 2000. ---*/
55
56 P_0 = 1.0;
57 u_0 = 1.0;
58 v_0 = 1.0;
59 epsilon = 0.001;
60
61 /*--- Perform some sanity and error checks for this solution here. ---*/
62 if (config->GetTime_Marching() != TIME_MARCHING::STEADY)
63 SU2_MPI::Error("Steady mode must be selected for the MMS incompressible NS case", CURRENT_FUNCTION);
64
65 if (Kind_Solver != MAIN_SOLVER::INC_EULER && Kind_Solver != MAIN_SOLVER::INC_NAVIER_STOKES &&
66 Kind_Solver != MAIN_SOLVER::INC_RANS)
67 SU2_MPI::Error("Incompressible flow equations must be selected for the MMS incompressible NS case",
68 CURRENT_FUNCTION);
69
70 if (Kind_Solver != MAIN_SOLVER::INC_NAVIER_STOKES)
71 SU2_MPI::Error("Navier Stokes equations must be selected for the MMS incompressible NS case", CURRENT_FUNCTION);
72
73 if (config->GetKind_FluidModel() != CONSTANT_DENSITY)
74 SU2_MPI::Error("Constant density fluid model must be selected for the MMS incompressible NS case",
75 CURRENT_FUNCTION);
76
77 if (config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
78 SU2_MPI::Error("Constant viscosity must be selected for the MMS incompressible NS case", CURRENT_FUNCTION);
79
80 if (config->GetEnergy_Equation())
81 SU2_MPI::Error("Energy equation must be disabled (isothermal) for the MMS incompressible NS case",
82 CURRENT_FUNCTION);
83}

92void CMMSIncNSSolution::GetSolution(const su2double* val_coords, const su2double val_t, su2double* val_solution) const {
93 /* Easier storage of the x- and y-coordinates. */
94 const su2double x = val_coords[0];
95 const su2double y = val_coords[1];
96
97 /* Compute the primitives from the defined solution. */
98 const su2double u = u_0 * (sin(x * x + y * y) + epsilon);
99 const su2double v = v_0 * (cos(x * x + y * y) + epsilon);
100 const su2double p = P_0 * (sin(x * x + y * y) + 2.0);
101
102 /* For the incompressible solver, we return the primitive variables
103 directly, as they are used for the working variables in the solver.
104 Note that the implementation below is valid for both 2D and 3D. */
105 val_solution[0] = p;
106 val_solution[1] = u;
107 val_solution[2] = v;
108 val_solution[3] = 0.0;
109 val_solution[nVar - 1] = Temperature;
110}

112void CMMSIncNSSolution::GetMMSSourceTerm(const su2double* val_coords, const su2double val_t,
113 su2double* val_source) const {
114 /*--- Easier storage of the x- and y-coordinates. ---*/
115 const su2double x = val_coords[0];
116 const su2double y = val_coords[1];
117
118 /*--- The expressions for the source terms are generated
119 automatically by the sympy package in python.---*/
120 val_source[0] = 2 * Density * (u_0 * x * cos(pow(x, 2) + pow(y, 2)) - v_0 * y * sin(pow(x, 2) + pow(y, 2)));
121 val_source[1] = 4 * Density * pow(u_0, 2) * x * (epsilon + sin(pow(x, 2) + pow(y, 2))) * cos(pow(x, 2) + pow(y, 2)) -
122 2 * Density * u_0 * v_0 * y * (epsilon + sin(pow(x, 2) + pow(y, 2))) * sin(pow(x, 2) + pow(y, 2)) +
123 2 * Density * u_0 * v_0 * y * (epsilon + cos(pow(x, 2) + pow(y, 2))) * cos(pow(x, 2) + pow(y, 2)) +
124 2 * P_0 * x * cos(pow(x, 2) + pow(y, 2)) -
125 0.666666666666667 * Viscosity *
126 (-8.0 * u_0 * pow(x, 2) * sin(pow(x, 2) + pow(y, 2)) + 4.0 * u_0 * cos(pow(x, 2) + pow(y, 2)) +
127 4 * v_0 * x * y * cos(pow(x, 2) + pow(y, 2))) +
128 2 * Viscosity *
129 (2 * u_0 * pow(y, 2) * sin(pow(x, 2) + pow(y, 2)) - u_0 * cos(pow(x, 2) + pow(y, 2)) +
130 2 * v_0 * x * y * cos(pow(x, 2) + pow(y, 2)));
131 val_source[2] = -2 * Density * u_0 * v_0 * x * (epsilon + sin(pow(x, 2) + pow(y, 2))) * sin(pow(x, 2) + pow(y, 2)) +
132 2 * Density * u_0 * v_0 * x * (epsilon + cos(pow(x, 2) + pow(y, 2))) * cos(pow(x, 2) + pow(y, 2)) -
133 4 * Density * pow(v_0, 2) * y * (epsilon + cos(pow(x, 2) + pow(y, 2))) * sin(pow(x, 2) + pow(y, 2)) +
134 2 * P_0 * y * cos(pow(x, 2) + pow(y, 2)) +
135 0.666666666666667 * Viscosity *
136 (-4 * u_0 * x * y * sin(pow(x, 2) + pow(y, 2)) +
137 8.0 * v_0 * pow(y, 2) * cos(pow(x, 2) + pow(y, 2)) + 4.0 * v_0 * sin(pow(x, 2) + pow(y, 2))) +
138 2 * Viscosity *
139 (2 * u_0 * x * y * sin(pow(x, 2) + pow(y, 2)) + 2 * v_0 * pow(x, 2) * cos(pow(x, 2) + pow(y, 2)) +
140 v_0 * sin(pow(x, 2) + pow(y, 2)));
141 val_source[3] = 0.0;
142 val_source[nVar - 1] = 0.0;
143}
