33 void CIteration::SetGrid_Movement(CGeometry** geometry, CSurfaceMovement* surface_movement,
34  CVolumetricMovement* grid_movement, CSolver*** solver, CConfig* config,
35  unsigned long IntIter, unsigned long TimeIter) {
36  unsigned short Kind_Grid_Movement = config->GetKind_GridMovement();
37  bool adjoint = config->GetContinuous_Adjoint();
38 
39  unsigned short val_iZone = config->GetiZone();
40 
41  /*--- Perform mesh movement depending on specified type ---*/
42  switch (Kind_Grid_Movement) {
43  case RIGID_MOTION:
44 
45  if (rank == MASTER_NODE) cout << endl << " Performing rigid mesh transformation." << endl;
46 
47  /*--- Move each node in the volume mesh using the specified type
48  of rigid mesh motion. These routines also compute analytic grid
49  velocities for the fine mesh. ---*/
50 
51  grid_movement->Rigid_Translation(geometry[MESH_0], config, val_iZone, TimeIter);
52  grid_movement->Rigid_Plunging(geometry[MESH_0], config, val_iZone, TimeIter);
53  grid_movement->Rigid_Pitching(geometry[MESH_0], config, val_iZone, TimeIter);
54  grid_movement->Rigid_Rotation(geometry[MESH_0], config, val_iZone, TimeIter);
55 
56  /*--- Update the multigrid structure after moving the finest grid,
57  including computing the grid velocities on the coarser levels. ---*/
58 
59  grid_movement->UpdateMultiGrid(geometry, config);
60 
61  break;
62 
63  case STEADY_TRANSLATION:
64  /*--- Set or update the translating frame mesh movement with the current translation rates,
65  * which might be altered via the python interface. ---*/
66 
67  if (rank == MASTER_NODE) cout << "\n Setting translational grid velocities." << endl;
68 
69  /*--- Set the translational velocity on all grid levels. ---*/
70 
71  for (auto iMGlevel = 0u; iMGlevel <= config->GetnMGLevels(); iMGlevel++)
72  geometry[iMGlevel]->SetTranslationalVelocity(config, iMGlevel == 0);
73 
74  break;
75 
76  case ROTATING_FRAME:
77  /*--- Set or update the rotating frame mesh movement with the current translation and rotation
78  * rates, which might be altered via the python interface. ---*/
79 
80  if (rank == MASTER_NODE) cout << "\n Setting rotating frame grid velocities." << endl;
81 
82  /*--- Set the grid velocities on all multigrid levels for a steadily
83  rotating reference frame. ---*/
84 
85  for (auto iMGlevel = 0u; iMGlevel <= config->GetnMGLevels(); iMGlevel++){
86  geometry[iMGlevel]->SetRotationalVelocity(config, iMGlevel == 0);
87  geometry[iMGlevel]->SetShroudVelocity(config);
88  }
89 
90  break;
91  }
92 
93  if (config->GetSurface_Movement(AEROELASTIC) || config->GetSurface_Movement(AEROELASTIC_RIGID_MOTION) || config->GetSurface_Movement(MOVING_WALL)) {
94  /*--- Apply rigid mesh transformation to entire grid first, if necessary ---*/
95 
96  if (IntIter == 0) {
97 
98  if (Kind_Grid_Movement == AEROELASTIC_RIGID_MOTION) {
99  if (rank == MASTER_NODE) cout << endl << " Performing rigid mesh transformation." << endl;
100 
101  /*--- Move each node in the volume mesh using the specified type
102  of rigid mesh motion. These routines also compute analytic grid
103  velocities for the fine mesh. ---*/
104 
105  grid_movement->Rigid_Translation(geometry[MESH_0], config, val_iZone, TimeIter);
106  grid_movement->Rigid_Plunging(geometry[MESH_0], config, val_iZone, TimeIter);
107  grid_movement->Rigid_Pitching(geometry[MESH_0], config, val_iZone, TimeIter);
108  grid_movement->Rigid_Rotation(geometry[MESH_0], config, val_iZone, TimeIter);
109 
110  /*--- Update the multigrid structure after moving the finest grid,
111  including computing the grid velocities on the coarser levels. ---*/
112 
113  grid_movement->UpdateMultiGrid(geometry, config);
114  }
115  if (config->GetSurface_Movement(MOVING_WALL)) {
116  for (auto iMGlevel = 0u; iMGlevel <= config->GetnMGLevels(); iMGlevel++) {
117  geometry[iMGlevel]->SetWallVelocity(config, iMGlevel == 0u);
118  }
119  }
120 
121  }
122 
123  /*--- Use the if statement to move the grid only at selected dual time step iterations. ---*/
124  else if (IntIter % config->GetAeroelasticIter() == 0) {
125  if (rank == MASTER_NODE) cout << endl << " Solving aeroelastic equations and updating surface positions." << endl;
126 
127  /*--- Solve the aeroelastic equations for the new node locations of the moving markers(surfaces) ---*/
128 
129  solver[MESH_0][FLOW_SOL]->Aeroelastic(surface_movement, geometry[MESH_0], config, TimeIter);
130 
131  /*--- Deform the volume grid around the new boundary locations ---*/
132 
133  if (rank == MASTER_NODE) cout << " Deforming the volume grid due to the aeroelastic movement." << endl;
134  grid_movement->SetVolume_Deformation(geometry[MESH_0], config, true);
135 
136  /*--- Update the grid velocities on the fine mesh using finite
137  differencing based on node coordinates at previous times. ---*/
138 
139  if (rank == MASTER_NODE) cout << " Computing grid velocities by finite differencing." << endl;
140  geometry[MESH_0]->SetGridVelocity(config);
141 
142  /*--- Update the multigrid structure after moving the finest grid,
143  including computing the grid velocities on the coarser levels. ---*/
144 
145  grid_movement->UpdateMultiGrid(geometry, config);
146  }
147  }
148 
149  if (config->GetSurface_Movement(EXTERNAL) || config->GetSurface_Movement(EXTERNAL_ROTATION)) {
150  /*--- Apply rigid rotation to entire grid first, if necessary ---*/
151 
152  if (Kind_Grid_Movement == EXTERNAL_ROTATION) {
153  if (rank == MASTER_NODE) cout << " Updating node locations by rigid rotation." << endl;
154  grid_movement->Rigid_Rotation(geometry[MESH_0], config, val_iZone, TimeIter);
155  }
156 
157  /*--- Load new surface node locations from external files ---*/
158 
159  if (rank == MASTER_NODE) cout << " Updating surface locations from file." << endl;
160  surface_movement->SetExternal_Deformation(geometry[MESH_0], config, val_iZone, TimeIter);
161 
162  /*--- Deform the volume grid around the new boundary locations ---*/
163 
164  if (rank == MASTER_NODE) cout << " Deforming the volume grid." << endl;
165  grid_movement->SetVolume_Deformation(geometry[MESH_0], config, true);
166 
167  /*--- Update the grid velocities on the fine mesh using finite
168  differencing based on node coordinates at previous times. ---*/
169 
170  if (!adjoint) {
171  if (rank == MASTER_NODE) cout << " Computing grid velocities by finite differencing." << endl;
172  geometry[MESH_0]->SetGridVelocity(config);
173  }
174 
175  /*--- Update the multigrid structure after moving the finest grid,
176  including computing the grid velocities on the coarser levels. ---*/
177 
178  grid_movement->UpdateMultiGrid(geometry, config);
179  }
180 }

182 void CIteration::SetMesh_Deformation(CGeometry** geometry, CSolver** solver, CNumerics*** numerics, CConfig* config,
183  RECORDING kind_recording) {
184  if (!config->GetDeform_Mesh()) return;
185 
186  /*--- Perform the elasticity mesh movement ---*/
187 
188  bool wasActive = false;
189  if ((kind_recording != RECORDING::MESH_DEFORM) && !config->GetMultizone_Problem()) {
190  /*--- In a primal run, AD::TapeActive returns a false ---*/
191  /*--- In any other recordings, the tape is passive during the deformation. ---*/
192  wasActive = AD::BeginPassive();
193  }
194 
195  /*--- Set the stiffness of each element mesh into the mesh numerics ---*/
196 
197  solver[MESH_SOL]->SetMesh_Stiffness(numerics[MESH_SOL], config);
198 
199  /*--- Deform the volume grid around the new boundary locations ---*/
200 
201  solver[MESH_SOL]->DeformMesh(geometry, numerics[MESH_SOL], config);
202 
203  /*--- Continue recording. ---*/
204  AD::EndPassive(wasActive);
205 }
