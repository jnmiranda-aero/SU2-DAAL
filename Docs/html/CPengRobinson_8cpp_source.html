30 CPengRobinson::CPengRobinson(su2double gamma, su2double R, su2double Pstar, su2double Tstar, su2double w)
31  : CIdealGas(gamma, R) {
32  a = 0.45724 * Gas_Constant * Gas_Constant * Tstar * Tstar / Pstar;
33  b = 0.0778 * Gas_Constant * Tstar / Pstar;
34  TstarCrit = Tstar;
35  Zed = 1.0;
36 
37  if (w <= 0.49)
38  k = 0.37464 + 1.54226 * w - 0.26992 * w * w;
39  else
40  k = 0.379642 + 1.48503 * w - 0.164423 * w * w + 0.016666 * w * w * w;
41 }

43 su2double CPengRobinson::alpha2(su2double T) const {
44  return (1 + k * (1 - sqrt(T / TstarCrit))) * (1 + k * (1 - sqrt(T / TstarCrit)));

47 su2double CPengRobinson::T_v_h(su2double v, su2double h) {
48  su2double fv, A, B, C, T, d, atanh;
49  su2double sqrt2 = sqrt(2.0);
50 
51  d = (v * v + 2 * b * v - b * b);
52 
53  atanh = (log(1.0 + (b * sqrt2 / (v + b))) - log(1.0 - (b * sqrt2 / (v + b)))) / 2.0;
54 
55  fv = atanh;
56 
57  A = Gas_Constant * (1 / Gamma_Minus_One + v / (v - b)) - a * v * k * k / (TstarCrit * d);
58  B = a * k * (k + 1) / sqrt(TstarCrit) * (fv / (b * sqrt2) + 2 * v / d);
59  C = h + a * (1 + k) * (1 + k) * (fv / (b * sqrt2) + v / d);
60 
61  T = (-B + sqrt(B * B + 4 * A * C)) / (2 * A); /// Only positive root considered
62 
63  return T * T;
64 }

66 su2double CPengRobinson::T_P_rho(su2double P, su2double rho) {
67  su2double A, B, C, T, vb1, vb2;
68  vb1 = (1 / rho - b);
69  vb2 = (1 / rho / rho + 2 * b / rho - b * b);
70 
71  A = Gas_Constant / vb1 - a * k * k / TstarCrit / vb2;
72 
73  B = 2 * a * k * (k + 1) / sqrt(TstarCrit) / vb2;
74 
75  C = -P - a * (1 + k) * (1 + k) / vb2;
76 
77  T = (-B + sqrt(B * B - 4 * A * C)) / (2 * A);
78  T *= T;
79  return T;
80 }

82 void CPengRobinson::SetTDState_rhoe(su2double rho, su2double e) {
83  su2double DpDd_T, DpDT_d, DeDd_T, Cv;
84  su2double A, B, C, sqrt2, fv, a2T, rho2, atanh;
85 
86  Density = rho;
87  StaticEnergy = e;
88 
89  AD::StartPreacc();
90  AD::SetPreaccIn(rho);
91  AD::SetPreaccIn(e);
92 
93  rho2 = rho * rho;
94  sqrt2 = sqrt(2.0);
95 
96  atanh = (log(1.0 + (rho * b * sqrt2 / (1 + rho * b))) - log(1.0 - (rho * b * sqrt2 / (1 + rho * b)))) / 2.0;
97 
98  fv = atanh;
99 
100  A = Gas_Constant / Gamma_Minus_One;
101  B = a * k * (k + 1) * fv / (b * sqrt2 * sqrt(TstarCrit));
102  C = a * (k + 1) * (k + 1) * fv / (b * sqrt2) + e;
103 
104  Temperature = (-B + sqrt(B * B + 4 * A * C)) / (2 * A); /// Only positive root considered
105  Temperature *= Temperature;
106 
107  a2T = alpha2(Temperature);
108 
109  A = (1 / rho2 + 2 * b / rho - b * b);
110  B = 1 / rho - b;
111 
112  Pressure = Temperature * Gas_Constant / B - a * a2T / A;
113 
114  Entropy = Gas_Constant / Gamma_Minus_One * log(Temperature) + Gas_Constant * log(B) -
115  a * sqrt(a2T) * k * fv / (b * sqrt2 * sqrt(Temperature * TstarCrit));
116 
117  DpDd_T = (Temperature * Gas_Constant / (B * B) - 2 * a * a2T * (1 / rho + b) / (A * A)) / (rho2);
118 
119  DpDT_d = Gas_Constant / B + a * k / A * sqrt(a2T / (Temperature * TstarCrit));
120 
121  Cv = Gas_Constant / Gamma_Minus_One + (a * k * (k + 1) * fv) / (2 * b * sqrt(2 * Temperature * TstarCrit));
122 
123  dPde_rho = DpDT_d / Cv;
124 
125  DeDd_T = -a * (1 + k) * sqrt(a2T) / A / (rho2);
126 
127  dPdrho_e = DpDd_T - dPde_rho * DeDd_T;
128 
129  SoundSpeed2 = dPdrho_e + Pressure / (rho2)*dPde_rho;
130 
131  dTde_rho = 1 / Cv;
132 
133  Zed = Pressure / (Gas_Constant * Temperature * rho);
134 
135  AD::SetPreaccOut(Temperature);
136  AD::SetPreaccOut(SoundSpeed2);
137  AD::SetPreaccOut(dPde_rho);
138  AD::SetPreaccOut(dPdrho_e);
139  AD::SetPreaccOut(Zed);
140  AD::SetPreaccOut(dTde_rho);
141  AD::SetPreaccOut(Pressure);
142  AD::SetPreaccOut(Entropy);
143  AD::EndPreacc();
144 }

146 void CPengRobinson::SetTDState_PT(su2double P, su2double T) {
147  su2double toll = 1e-6;
148  su2double A, B, Z, DZ = 1.0, F, F1, atanh;
149  su2double rho, fv, e;
150  su2double sqrt2 = sqrt(2.0);
151  unsigned short nmax = 20, count = 0;
152 
153  AD::StartPreacc();
154  AD::SetPreaccIn(P);
155  AD::SetPreaccIn(T);
156 
157  A = a * alpha2(T) * P / (T * Gas_Constant) / (T * Gas_Constant);
158  B = b * P / (T * Gas_Constant);
159 
160  if (Zed > 0.1)
161  Z = min(Zed, 0.99);
162  else
163  Z = 0.99;
164 
165  do {
166  F = Z * Z * Z + Z * Z * (B - 1.0) + Z * (A - 2 * B - 3 * B * B) + (B * B * B + B * B - A * B);
167  F1 = 3 * Z * Z + 2 * Z * (B - 1.0) + (A - 2 * B - 3 * B * B);
168  DZ = F / F1;
169  Z -= DZ;
170  } while (abs(DZ) > toll && count < nmax);
171 
172  if (count == nmax) {
173  cout << "Warning Newton-Raphson exceed number of max iteration in PT" << endl;
174  cout << "Compressibility factor " << Z << " would be substituted with " << Zed << endl;
175  }
176  // check if the solution is physical otherwise uses previous point solution
177  if (Z <= 1.0001 && Z >= 0.05 && count < nmax) Zed = Z;
178 
179  rho = P / (Zed * Gas_Constant * T);
180 
181  atanh = (log(1.0 + (rho * b * sqrt2 / (1 + rho * b))) - log(1.0 - (rho * b * sqrt2 / (1 + rho * b)))) / 2.0;
182 
183  fv = atanh;
184 
185  e = T * Gas_Constant / Gamma_Minus_One - a * (k + 1) * sqrt(alpha2(T)) * fv / (b * sqrt2);
186 
187  AD::SetPreaccOut(rho);
188  AD::SetPreaccOut(e);
189  AD::EndPreacc();
190 
191  SetTDState_rhoe(rho, e);
192 }

200 void CPengRobinson::SetTDState_hs(su2double h, su2double s) {
201  su2double T, fv, sqrt2 = sqrt(2.0), A;
202  su2double f, v, atanh;
203  su2double x1, x2, xmid, dx, fx1, fx2, fmid, rtb;
204  su2double toll = 1e-9, FACTOR = 0.2;
205  su2double cons_s, cons_h;
206  unsigned short countrtb = 0, NTRY = 100, ITMAX = 100;
207 
208  A = Gas_Constant / Gamma_Minus_One;
209  T = h * Gamma_Minus_One / Gas_Constant / Gamma;
210  v = exp(-1 / Gamma_Minus_One * log(T) + s / Gas_Constant);
211 
212  x1 = 0.2 * v;
213  x2 = 0.35 * v;
214 
215  T = T_v_h(x1, h);
216 
217  atanh = (log(1.0 + (b * sqrt2 / (x1 + b))) - log(1.0 - (b * sqrt2 / (x1 + b)))) / 2.0;
218  fv = atanh;
219 
220  fx1 = A * log(T) + Gas_Constant * log(x1 - b) - a * sqrt(alpha2(T)) * k * fv / (b * sqrt2 * sqrt(T * TstarCrit)) - s;
221  T = T_v_h(x2, h);
222 
223  atanh = (log(1.0 + (b * sqrt2 / (x2 + b))) - log(1.0 - (b * sqrt2 / (x2 + b)))) / 2.0;
224  fv = atanh;
225 
226  fx2 = A * log(T) + Gas_Constant * log(x2 - b) - a * sqrt(alpha2(T)) * k * fv / (b * sqrt2 * sqrt(T * TstarCrit)) - s;
227 
228  // zbrac algorithm NR
229 
230  for (int j = 1; j <= NTRY; j++) {
231  if (fx1 * fx2 > 0.0) {
232  if (fabs(fx1) < fabs(fx2)) {
233  x1 += FACTOR * (x1 - x2);
234  T = T_v_h(x1, h);
235  atanh = (log(1.0 + (b * sqrt2 / (x1 + b))) - log(1.0 - (b * sqrt2 / (x1 + b)))) / 2.0;
236  fv = atanh;
237  fx1 = A * log(T) + Gas_Constant * log(x1 - b) -
238  a * sqrt(alpha2(T)) * k * fv / (b * sqrt2 * sqrt(T * TstarCrit)) - s;
239  } else {
240  x2 += FACTOR * (x2 - x1);
241  T = T_v_h(x2, h);
242  atanh = (log(1.0 + (b * sqrt2 / (x2 + b))) - log(1.0 - (b * sqrt2 / (x2 + b)))) / 2.0;
243  fv = atanh;
244  fx2 = A * log(T) + Gas_Constant * log(x2 - b) -
245  a * sqrt(alpha2(T)) * k * fv / (b * sqrt2 * sqrt(T * TstarCrit)) - s;
246  }
247  }
248  }
249 
250  // rtbis algorithm NR
251 
252  f = fx1;
253  fmid = fx2;
254  if (f * fmid >= 0.0) {
255  cout << "Root must be bracketed for bisection in rtbis" << endl;
256  SetTDState_rhoT(Density, Temperature);
257  }
258  rtb = f < 0.0 ? (static_cast<void>(dx = x2 - x1), x1) : (static_cast<void>(dx = x1 - x2), x2);
259  do {
260  xmid = rtb + (dx *= 0.5);
261  T = T_v_h(xmid, h);
262  atanh = (log(1.0 + (b * sqrt2 / (xmid + b))) - log(1.0 - (b * sqrt2 / (xmid + b)))) / 2.0;
263  fv = atanh;
264  fmid = A * log(T) + Gas_Constant * log(xmid - b) -
265  a * sqrt(alpha2(T)) * k * fv / (b * sqrt2 * sqrt(T * TstarCrit)) - s;
266 
267  if (fmid <= 0.0) rtb = xmid;
268  countrtb++;
269  } while (abs(fmid) > toll && countrtb < ITMAX);
270 
271  v = xmid;
272  if (countrtb == ITMAX) {
273  cout << "Too many bisections in rtbis" << endl;
274  cout << countrtb << endl;
275  }
276  if (v != v) {
277  cout << "not physical solution found, h and s input " << h << " " << s << endl;
278  SetTDState_rhoT(Density, Temperature);
279  }
280 
281  T = T_v_h(v, h);
282  SetTDState_rhoT(1 / v, T);
283 
284  // consistency check
285  cons_h = abs(((StaticEnergy + Pressure / Density) - h) / h);
286  cons_s = abs((Entropy - s) / s);
287 
288  if (cons_h > 1e-4 || cons_s > 1e-4) {
289  cout << "TD consistency not verified in hs call" << endl;
290  }
291 }

293 void CPengRobinson::SetEnergy_Prho(su2double P, su2double rho) {
294  su2double ad;
295  su2double A, B, C, T, vb1, vb2, atanh;
296 
297  AD::StartPreacc();
298  AD::SetPreaccIn(P);
299  AD::SetPreaccIn(rho);
300 
301  vb1 = (1 / rho - b);
302  vb2 = (1 / rho / rho + 2 * b / rho - b * b);
303 
304  A = Gas_Constant / vb1 - a * k * k / TstarCrit / vb2;
305 
306  B = 2 * a * k * (k + 1) / sqrt(TstarCrit) / vb2;
307 
308  C = -P - a * (1 + k) * (1 + k) / vb2;
309 
310  T = (-B + sqrt(B * B - 4 * A * C)) / (2 * A);
311  T *= T;
312 
313  atanh = (log(1.0 + (rho * b * sqrt(2.0) / (1 + rho * b))) - log(1.0 - (rho * b * sqrt(2.0) / (1 + rho * b)))) / 2.0;
314  ad = a * (k + 1) * sqrt(alpha2(T)) / (b * sqrt(2.0)) * atanh;
315 
316  StaticEnergy = T * Gas_Constant / Gamma_Minus_One - ad;
317 
318  AD::SetPreaccOut(StaticEnergy);
319  AD::EndPreacc();

425 void CPengRobinson::ComputeDerivativeNRBC_Prho(su2double P, su2double rho) {
426  su2double dPdT_rho, dPdrho_T, dPds_rho, der1_alpha;
427 
428  SetTDState_Prho(P, rho);
429 
430  der1_alpha = -k / (2 * sqrt(TstarCrit * Temperature));
431  dPdT_rho = Gas_Constant * rho / (1.0 - rho * b) -
432  2 * a * rho * rho * sqrt(alpha2(Temperature)) * der1_alpha / (1 + 2 * b * rho - b * b * rho * rho);
433  dPdrho_T = Gas_Constant * Temperature / (1.0 - rho * b) / (1.0 - rho * b) -
434  2.0 * rho * a * alpha2(Temperature) * (1.0 + b * rho) / (1 + 2 * b * rho - b * b * rho * rho) /
435  (1 + 2 * b * rho - b * b * rho * rho);
436 
437  dhdrho_P = -dPdrho_e / dPde_rho - P / rho / rho;
438  dhdP_rho = 1.0 / dPde_rho + 1.0 / rho;
439  dPds_rho = rho * rho * (SoundSpeed2 - dPdrho_T) / dPdT_rho;
440  dsdP_rho = 1.0 / dPds_rho;
441  dsdrho_P = -SoundSpeed2 / dPds_rho;
442 }
