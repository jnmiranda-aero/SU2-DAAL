42 void CMultiGridQueue::AddCV(unsigned long newPoint, short numberNeighbors) {
43  const short maxNeighbors = QueueCV.size() - 1;
44 
45  /*--- Basic check ---*/
46  if (newPoint >= nPoint) {
47  SU2_MPI::Error("The index of the CV is greater than the size of the priority list.", CURRENT_FUNCTION);
48  }
49 
50  /*--- Resize the list ---*/
51  if (numberNeighbors > maxNeighbors) {
52  const size_t newSize = numberNeighbors + 1;
53  if (QueueCV.capacity() < newSize) QueueCV.reserve(2 * newSize);
54  QueueCV.resize(newSize);
55  }
56 
57  /*--- Find the point in the queue ---*/
58  const bool inQueue = (Priority[newPoint] == numberNeighbors);
59 
60  if (!inQueue) {
61  /*--- Add the control volume, and update the priority list ---*/
62  QueueCV[numberNeighbors].push_back(newPoint);
63  Priority[newPoint] = numberNeighbors;
64  }
65 }

67 void CMultiGridQueue::RemoveCV(unsigned long removePoint) {
68  /*--- Basic check ---*/
69  if (removePoint >= nPoint) {
70  SU2_MPI::Error("The index of the CV is greater than the size of the priority list.", CURRENT_FUNCTION);
71  }
72 
73  /*--- Find priority of the Control Volume. ---*/
74  const auto numberNeighbors = Priority[removePoint];
75  if (numberNeighbors == -1) ThrowPointNotInListError(removePoint);
76  Priority[removePoint] = -1;
77 
78  /*--- Find the point in the queue, if the queue is not changed we can exit. ---*/
79  if (!QueueCV[numberNeighbors].findAndErase(removePoint)) return;
80 
81  /*--- Check that the size of the queue is the right one. ---*/
82  /*--- Resize the queue, leaving at least one element in it. ---*/
83  auto sizeQueueCV = QueueCV.size();
84 
85  while (sizeQueueCV > 0) {
86  if (!QueueCV[--sizeQueueCV].empty()) break;
87  }
88  QueueCV.resize(sizeQueueCV + 1);
89 }

91 void CMultiGridQueue::MoveCV(unsigned long movePoint, short numberNeighbors) {
92  RightCV[movePoint] = (numberNeighbors >= 0);
93  numberNeighbors = max<short>(numberNeighbors, 0);
94 
95  /*--- Remove the control volume ---*/
96  RemoveCV(movePoint);
97 
98  /*--- Add a new control volume ---*/
99  AddCV(movePoint, numberNeighbors);

143 bool CMultiGridQueue::EmptyQueue() const {
144  /*--- In case there is only the no agglomerated elements (size 1), check
145  * if they can be agglomerated or if we have already finished. ---*/
146 
147  if (QueueCV.size() == 1) {
148  for (auto iPoint : QueueCV[0])
149  if ((iPoint != QueueType::ErasedValue) && RightCV[iPoint]) return false;
150  } else {
151  for (size_t iQ = 1; iQ < QueueCV.size(); ++iQ)
152  if (!QueueCV[iQ].empty()) return false;
153  }
154  return true;
155 }
