50COutput::COutput(const CConfig *config, unsigned short ndim, bool fem_output):
51 rank(SU2_MPI::GetRank()),
52 size(SU2_MPI::GetSize()),
53 nDim(ndim),
54 multiZone(config->GetMultizone_Problem()),
55 gridMovement(config->GetDynamic_Grid()),
56 femOutput(fem_output),
57 si_units(config->GetSystemMeasurements() == SI),
58 us_units(config->GetSystemMeasurements() == US) {
59
60 cauchyTimeConverged = false;
61
62 convergenceTable = new PrintingToolbox::CTablePrinter(&std::cout);
63 multiZoneHeaderTable = new PrintingToolbox::CTablePrinter(&std::cout);
64 fileWritingTable = new PrintingToolbox::CTablePrinter(&std::cout);
65 historyFileTable = new PrintingToolbox::CTablePrinter(&histFile, "");
66
67 /*--- Set default filenames ---*/
68
69 surfaceFilename = "surface";
70 volumeFilename = "volume";
71 restartFilename = "restart";
72
73 /*--- Retrieve the history filename ---*/
74
75 historyFilename = config->GetConv_FileName();
76
77 /*--- Add the correct file extension depending on the file format ---*/
78
79 string hist_ext = ".csv";
80 if (config->GetTabular_FileFormat() == TAB_OUTPUT::TAB_TECPLOT) hist_ext = ".dat";
81
82 /*--- Append the zone ID ---*/
83
84 historyFilename = config->GetMultizone_HistoryFileName(historyFilename, config->GetiZone(), hist_ext);
85
86 /*--- Append the restart iteration ---*/
87
88 if (config->GetTime_Domain() && config->GetRestart()) {
89 historyFilename = config->GetUnsteady_FileName(historyFilename, config->GetRestart_Iter(), hist_ext);
90 }
91
92 historySep = ",";
93
94 /*--- Initialize residual ---*/
95
96 rhoResNew = EPS;
97 rhoResOld = EPS;
98
99 nRequestedHistoryFields = config->GetnHistoryOutput();
100 for (unsigned short iField = 0; iField < nRequestedHistoryFields; iField++){
101 requestedHistoryFields.push_back(config->GetHistoryOutput_Field(iField));
102 }
103
104 nRequestedScreenFields = config->GetnScreenOutput();
105 for (unsigned short iField = 0; iField < nRequestedScreenFields; iField++){
106 requestedScreenFields.push_back(config->GetScreenOutput_Field(iField));
107 }
108
109 nRequestedVolumeFields = config->GetnVolumeOutput();
110 for (unsigned short iField = 0; iField < nRequestedVolumeFields; iField++){
111 requestedVolumeFields.push_back(config->GetVolumeOutput_Field(iField));
112 }
113
114 /*--- Default is to write history to file and screen --- */
115
116 noWriting = false;
117
118 /*--- Initialize convergence monitoring structure ---*/
119
120 nCauchy_Elems = config->GetCauchy_Elems();
121 cauchyEps = config->GetCauchy_Eps();
122 minLogResidual = config->GetMinLogResidual();
123
124 for (unsigned short iField = 0; iField < config->GetnConv_Field(); iField++){
125 convFields.emplace_back(config->GetConv_Field(iField));
126 }
127
128 newFunc = vector<su2double>(convFields.size());
129 oldFunc = vector<su2double>(convFields.size());
130 cauchySerie = vector<vector<su2double>>(convFields.size(), vector<su2double>(nCauchy_Elems, 0.0));
131 cauchyValue = 0.0;
132 convergence = false;
133
134 /*--- Initialize time convergence monitoring structure ---*/
135
136 nWndCauchy_Elems = config->GetWnd_Cauchy_Elems();
137 wndCauchyEps = config->GetWnd_Cauchy_Eps();
138
139 wndConvFields.reserve(config->GetnWndConv_Field());
140 for (unsigned short iField = 0; iField < config->GetnWndConv_Field(); iField++){
141 wndConvFields.emplace_back(config->GetWndConv_Field(iField));
142 }
143
144 WndOld_Func = vector<su2double>(wndConvFields.size());
145 WndNew_Func = vector<su2double>(wndConvFields.size());
146 WndCauchy_Serie = vector<vector<su2double>>(wndConvFields.size(), vector<su2double>(nWndCauchy_Elems, 0.0));
147 WndCauchy_Value = 0.0;
148 TimeConvergence = false;
149
150 /*--- Check that the number of cauchy elems is not too large ---*/
151
152 if (nCauchy_Elems > 1000){
153 SU2_MPI::Error("Number of Cauchy Elems must be smaller than 1000", CURRENT_FUNCTION);
154 }
155 if (nWndCauchy_Elems > 1000){
156 SU2_MPI::Error("Number of Time Cauchy Elems must be smaller than 1000", CURRENT_FUNCTION);
157 }
158
159 /*--- Initialize all convergence flags to false. ---*/
160
161 convergence = false;
162
163 buildFieldIndexCache = false;
164
165 curInnerIter = 0;
166 curOuterIter = 0;
167 curTimeIter = 0;
168
169 volumeDataSorter = nullptr;
170 surfaceDataSorter = nullptr;
171
172 headerNeeded = false;
173}

230void COutput::SetHistoryOutput(CGeometry ****geometry, CSolver *****solver, CConfig **config, std::shared_ptr<CTurbomachineryStagePerformance>(TurboStagePerf), std::shared_ptr<CTurboOutput> TurboPerf, unsigned short val_iZone, unsigned long TimeIter, unsigned long OuterIter, unsigned long InnerIter, unsigned short val_iInst){
231
232 unsigned long Iter= InnerIter;
233
234 if (config[ZONE_0]->GetMultizone_Problem())
235 Iter = OuterIter;
236
237 /*--- Turbomachinery Performance Screen summary output---*/
238 if (Iter%100 == 0 && rank == MASTER_NODE) {
239 SetTurboPerformance_Output(TurboPerf, config[val_iZone], TimeIter, OuterIter, InnerIter);
240 SetTurboMultiZonePerformance_Output(TurboStagePerf, TurboPerf, config[val_iZone]);
241 }
242
243 for (int iZone = 0; iZone < config[ZONE_0]->GetnZone(); iZone ++){
244 if (rank == MASTER_NODE) {
245 WriteTurboSpanwisePerformance(TurboPerf, geometry[iZone][val_iInst][MESH_0], config, iZone);
246 }
247 }
248
249 /*--- Update turboperformance history file*/
250 if (rank == MASTER_NODE){
251 LoadTurboHistoryData(TurboStagePerf, TurboPerf, config[val_iZone]);
252 }
253 SetHistoryOutput(geometry[val_iZone][val_iInst][MESH_0], solver[val_iZone][val_iInst][MESH_0], config[val_iZone], TimeIter, OuterIter,InnerIter);

293void COutput::SetupCustomHistoryOutput(const std::string& expression, CustomHistoryOutput& output) const {
294 std::vector<std::string> symbols;
295 output.expression = mel::Parse<passivedouble>(expression, symbols);
296
297 output.symbolValues.reserve(symbols.size());
298 for (const auto& symbol : symbols) {
299 const auto* ptr = GetPtrToHistoryOutput(symbol);
300 if (ptr == nullptr) {
301 SU2_MPI::Error(std::string("Invalid history output (") + symbol + std::string(") used in expression:\n") +
302 expression, CURRENT_FUNCTION);
303 }
304 output.symbolValues.push_back(ptr);
305 }
306 output.ready = true;
307}

366 /*--- File writer that will later be used to write the file to disk. Created below in the "switch" ---*/

369 /*--- If it is still present, strip the extension (suffix) from the filename ---*/
370 const auto lastindex = fileName.find_last_of('.');
371 fileName = fileName.substr(0, lastindex);
372
373 /*--- If the filename with appended iteration is set (depending on the WRT_*_OVERWRITE options)
374 * two files are writen, the normal one and a copy to avoid overwriting previous outputs. ---*/
375 string filename_iter, extension;
376
377 /*--- Write output information to screen ---*/
378
379 auto LogOutputFiles = [&](const std::string& message) {
380 if (rank == MASTER_NODE) {
381 (*fileWritingTable) << message << fileName + extension;
382 if (!filename_iter.empty()) (*fileWritingTable) << message + " + iter" << filename_iter + extension;
383 }
384 };
385
386 /*--- Write files depending on the format --- */
387
388 switch (format) {
389
390 case OUTPUT_TYPE::SURFACE_CSV:
391
392 extension = CSU2FileWriter::fileExt;
393
394 if (fileName.empty())
395 fileName = config->GetFilename(surfaceFilename, "", curTimeIter);
396
397 if (!config->GetWrt_Surface_Overwrite())
398 filename_iter = config->GetFilename_Iter(fileName, curInnerIter, curOuterIter);
399
400 surfaceDataSorter->SortConnectivity(config, geometry);
401 surfaceDataSorter->SortOutputData();
402
403 LogOutputFiles("CSV file");
404 fileWriter = new CSU2FileWriter(surfaceDataSorter);
405
406 break;
407
408 case OUTPUT_TYPE::RESTART_ASCII: case OUTPUT_TYPE::CSV:
409
410 extension = CSU2FileWriter::fileExt;
411
412 if (fileName.empty())
413 fileName = config->GetFilename(restartFilename, "", curTimeIter);
414
415 if (!config->GetWrt_Restart_Overwrite())
416 filename_iter = config->GetFilename_Iter(fileName, curInnerIter, curOuterIter);

872bool COutput::ConvergenceMonitoring(CConfig *config, unsigned long Iteration) {
873
874 convergence = true;
875
876 for (auto iField_Conv = 0ul; iField_Conv < convFields.size(); iField_Conv++) {
877
878 const auto& convField = convFields[iField_Conv];
879 const auto it = historyOutput_Map.find(convField);
880
881 if (it == historyOutput_Map.end()) continue;
882
883 const auto& field = it->second;
884 const su2double monitor = field.value;
885
886 /*--- Stop the simulation in case a nan appears, do not save the solution. ---*/
887 if (std::isnan(SU2_TYPE::GetValue(monitor))) {
888 SU2_MPI::Error("SU2 has diverged (NaN detected).", CURRENT_FUNCTION);
889 }
890
891 bool fieldConverged = false;
892
893 switch (field.fieldType) {
894
895 /*--- Cauchy based convergence criteria ---*/
896 case HistoryFieldType::COEFFICIENT: {
897
898 if (Iteration == 0) {
899 for (auto iCounter = 0ul; iCounter < nCauchy_Elems; iCounter++) {
900 cauchySerie[iField_Conv][iCounter] = 0.0;
901 }
902 newFunc[iField_Conv] = monitor;
903 }
904
905 oldFunc[iField_Conv] = newFunc[iField_Conv];
906 newFunc[iField_Conv] = monitor;
907 /*--- Automatically modify the scaling factor of relative Cauchy convergence for
908 * coefficients that are close to zero. Example: For the clean aircraft, the rolling
909 * moment coefficient MOMENT_X is close to zero and thus will never reach a relative
910 * cauchy convergence ->> dividing tiny numbers is not a good idea. Using absolute
911 * cauchy convergence is more robust in this case. ---*/
912 cauchyFunc = fabs(newFunc[iField_Conv] - oldFunc[iField_Conv]) / fmax(fabs(monitor), 0.1);
913
914 cauchySerie[iField_Conv][Iteration % nCauchy_Elems] = cauchyFunc;
915 cauchyValue = 0.0;
916 for (auto iCounter = 0ul; iCounter < nCauchy_Elems; iCounter++)
917 cauchyValue += cauchySerie[iField_Conv][iCounter];
918
919 cauchyValue /= nCauchy_Elems;
920
921 /*--- Start monitoring only if the current iteration
922 * is larger than the number of cauchy elements. --- */
923 fieldConverged = (cauchyValue < cauchyEps) && (Iteration >= nCauchy_Elems);
924
925 if (Iteration == 0) cauchyValue = 1.0;
926 SetHistoryOutputValue("CAUCHY_" + convField, cauchyValue);
927
928 } break;
929
930
931 /*--- Residual based convergence criteria ---*/
932 case HistoryFieldType::RESIDUAL:
933 case HistoryFieldType::AUTO_RESIDUAL:
934
935 fieldConverged = (Iteration != 0) && (monitor <= minLogResidual);
936 break;
937
938 default:
939 break;
940 }
941
942 convergence = fieldConverged && convergence;
943 }
944
945 /*--- Do not apply any convergence criteria if the number
946 * of iterations is less than a particular value. ---*/
947
948 if (convFields.empty() || Iteration < config->GetStartConv_Iter()) convergence = false;
949
950 /*--- Apply the same convergence criteria to all processors. ---*/
951
952 unsigned short local = convergence, global = 0;
953 SU2_MPI::Allreduce(&local, &global, 1, MPI_UNSIGNED_SHORT, MPI_MAX, SU2_MPI::GetComm());
954 convergence = global > 0;
955
956 return convergence;
957}

959bool COutput::MonitorTimeConvergence(CConfig *config, unsigned long TimeIteration) {
960
961 bool Inner_IterConv = GetConvergence() || config->GetnInner_Iter()-1 <= curInnerIter; //Check, if Inner_Iter is converged
962
963 if(TimeIteration == 0){
964 for (unsigned short iField_Conv = 0; iField_Conv < wndConvFields.size(); iField_Conv++){
965 const string WndConv_Field= wndConvFields[iField_Conv];
966 if (historyOutput_Map.count(WndConv_Field) > 0){
967 SetHistoryOutputValue("CAUCHY_"+ WndConv_Field, 1.0);
968 }
969 }
970 }
971 if(Inner_IterConv && TimeIteration >= config->GetStartWindowIteration()){
972 TimeConvergence = true;
973 unsigned short iCounter;
974
975 for (unsigned short iField_Conv = 0; iField_Conv < wndConvFields.size(); iField_Conv++){
976 bool fieldConverged = false;
977 const string WndConv_Field= wndConvFields[iField_Conv];
978
979 if (historyOutput_Map.count(WndConv_Field) > 0){
980 su2double monitor = historyOutput_Map[WndConv_Field].value;
981
982 /*--- Stop the simulation in case a nan appears, do not save the solution ---*/
983 if (std::isnan(SU2_TYPE::GetValue(monitor))) {
984 SU2_MPI::Error("SU2 has diverged (NaN detected).", CURRENT_FUNCTION);
985 }
986
987 /*--- Cauchy based convergence criteria ---*/
988
989 if (historyOutput_Map[WndConv_Field].fieldType == HistoryFieldType::AUTO_COEFFICIENT) { //TAVG values are AUTO_COEFF
990 if (TimeIteration == config->GetStartWindowIteration()){
991 for (iCounter = 0; iCounter < nWndCauchy_Elems; iCounter++){
992 WndCauchy_Serie[iField_Conv][iCounter] = 0.0;
993 }
994 WndNew_Func[iField_Conv] = monitor;
995 }
996 WndOld_Func[iField_Conv] = WndNew_Func[iField_Conv];
997 WndNew_Func[iField_Conv] = monitor;
998 WndCauchy_Func = fabs(WndNew_Func[iField_Conv] - WndOld_Func[iField_Conv]);
999 WndCauchy_Serie[iField_Conv][TimeIteration % nWndCauchy_Elems] = WndCauchy_Func;
1000 WndCauchy_Value = 1.0;
1001
1002 if (TimeIteration >= nWndCauchy_Elems+config->GetStartWindowIteration()){
1003 WndCauchy_Value = 0.0;
1004 for (iCounter = 0; iCounter < nWndCauchy_Elems; iCounter++){
1005 WndCauchy_Value += WndCauchy_Serie[iField_Conv][iCounter];
1006 }
1007 WndCauchy_Value /= nWndCauchy_Elems;
1008 }
1009 fieldConverged = WndCauchy_Value < wndCauchyEps;
1010
1011 /*--- Start monitoring only if the current iteration is larger than the
1012 * number of cauchy elements and the number of start-up iterations ---*/
1013
1014 if (TimeIteration < config->GetStartWindowIteration() + max(config->GetWnd_StartConv_Iter(), nWndCauchy_Elems)){
1015 fieldConverged = false;
1016 }
1017 SetHistoryOutputValue("CAUCHY_" + WndConv_Field, WndCauchy_Value);
1018 }
1019 TimeConvergence = fieldConverged && TimeConvergence;
1020 }
1021 }
1022
1023 /*--- Do not apply any convergence criterion if the option is disabled. */
1024 if(!config->GetWnd_Cauchy_Crit()){TimeConvergence = false;}
1025 if(wndConvFields.empty()){TimeConvergence = false;}
1026 }
1027 return TimeConvergence;
1028}
