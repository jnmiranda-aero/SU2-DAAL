308 void COutput::SetupCustomHistoryOutput(const std::string& expression, CustomHistoryOutput& output) const {
309  std::vector<std::string> symbols;
310  output.expression = mel::Parse<passivedouble>(expression, symbols);
311 
312  output.symbolValues.reserve(symbols.size());

319  output.symbolValues.push_back(ptr);

379 void COutput::WriteToFile(CConfig *config, CGeometry *geometry, OUTPUT_TYPE format, string fileName){
380 
381  /*--- File writer that will later be used to write the file to disk. Created below in the "switch" ---*/
382  CFileWriter *fileWriter = nullptr;
383 
384  /*--- If it is still present, strip the extension (suffix) from the filename ---*/
385  const auto lastindex = fileName.find_last_of('.');
386  fileName = fileName.substr(0, lastindex);
387 
388  /*--- If the filename with appended iteration is set (depending on the WRT_*_OVERWRITE options)
389  * two files are writen, the normal one and a copy to avoid overwriting previous outputs. ---*/
390  string filename_iter, extension;
391 
392  /*--- Write output information to screen ---*/
393 
394  auto LogOutputFiles = [&](const std::string& message) {
395  if (rank == MASTER_NODE) {
396  (*fileWritingTable) << message << fileName + extension;
397  if (!filename_iter.empty()) (*fileWritingTable) << message + " + iter" << filename_iter + extension;
398  }
399  };
400 
401  /*--- Write files depending on the format --- */
402 
403  switch (format) {
404 
405  case OUTPUT_TYPE::SURFACE_CSV:
406 
407  extension = CSU2FileWriter::fileExt;
408 
409  if (fileName.empty())
410  fileName = config->GetFilename(surfaceFilename, "", curTimeIter);
411 
412  if (!config->GetWrt_Surface_Overwrite())
413  filename_iter = config->GetFilename_Iter(fileName, curInnerIter, curOuterIter);
414 
415  surfaceDataSorter->SortConnectivity(config, geometry);
416  surfaceDataSorter->SortOutputData();
417 
418  LogOutputFiles("CSV file");
419  fileWriter = new CSU2FileWriter(surfaceDataSorter);
420 
421  break;

887 bool COutput::ConvergenceMonitoring(CConfig *config, unsigned long Iteration) {
888 
889  convergence = true;
890 
891  for (auto iField_Conv = 0ul; iField_Conv < convFields.size(); iField_Conv++) {
892 
893  const auto& convField = convFields[iField_Conv];
894  const auto it = historyOutput_Map.find(convField);
895 
896  if (it == historyOutput_Map.end()) continue;
897 
898  const auto& field = it->second;
899  const su2double monitor = field.value;
900 
901  /*--- Stop the simulation in case a nan appears, do not save the solution. ---*/
902  if (std::isnan(SU2_TYPE::GetValue(monitor))) {
903  SU2_MPI::Error("SU2 has diverged (NaN detected).", CURRENT_FUNCTION);
904  }
905 
906  bool fieldConverged = false;
907 
908  switch (field.fieldType) {
909 
910  /*--- Cauchy based convergence criteria ---*/
911  case HistoryFieldType::COEFFICIENT: {
912 
913  if (Iteration == 0) {
914  for (auto iCounter = 0ul; iCounter < nCauchy_Elems; iCounter++) {
915  cauchySerie[iField_Conv][iCounter] = 0.0;
916  }
917  newFunc[iField_Conv] = monitor;
918  }
919 
920  oldFunc[iField_Conv] = newFunc[iField_Conv];
921  newFunc[iField_Conv] = monitor;
922  /*--- Automatically modify the scaling factor of relative Cauchy convergence for
923  * coefficients that are close to zero. Example: For the clean aircraft, the rolling
924  * moment coefficient MOMENT_X is close to zero and thus will never reach a relative
925  * cauchy convergence ->> dividing tiny numbers is not a good idea. Using absolute
926  * cauchy convergence is more robust in this case. ---*/
927  cauchyFunc = fabs(newFunc[iField_Conv] - oldFunc[iField_Conv]) / fmax(fabs(monitor), 0.1);
928 
929  cauchySerie[iField_Conv][Iteration % nCauchy_Elems] = cauchyFunc;
930  cauchyValue = 0.0;
931  for (auto iCounter = 0ul; iCounter < nCauchy_Elems; iCounter++)
932  cauchyValue += cauchySerie[iField_Conv][iCounter];
933 
934  cauchyValue /= nCauchy_Elems;
935 
936  /*--- Start monitoring only if the current iteration
937  * is larger than the number of cauchy elements. --- */
938  fieldConverged = (cauchyValue < cauchyEps) && (Iteration >= nCauchy_Elems);
939 
940  if (Iteration == 0) cauchyValue = 1.0;
941  SetHistoryOutputValue("CAUCHY_" + convField, cauchyValue);
942 
943  } break;
944 
945 
946  /*--- Residual based convergence criteria ---*/
947  case HistoryFieldType::RESIDUAL:
948  case HistoryFieldType::AUTO_RESIDUAL:
949 
950  fieldConverged = (Iteration != 0) && (monitor <= minLogResidual);
951  break;
952 
953  default:
954  break;
955  }
956 
957  convergence = fieldConverged && convergence;
958  }
959 
960  /*--- Do not apply any convergence criteria if the number
961  * of iterations is less than a particular value. ---*/
962 
963  if (convFields.empty() || Iteration < config->GetStartConv_Iter()) convergence = false;
964 
965  /*--- If a SIGTERM signal is sent to one of the processes, we set convergence to true. ---*/
966  if (STOP) convergence = true;
967 
968  /*--- Apply the same convergence criteria to all processors. ---*/
969 
970  unsigned short local = convergence, global = 0;
971 
972  SU2_MPI::Allreduce(&local, &global, 1, MPI_UNSIGNED_SHORT, MPI_MAX, SU2_MPI::GetComm());
973  convergence = global > 0;
974 
975  return convergence;
976 }

978 bool COutput::MonitorTimeConvergence(CConfig *config, unsigned long TimeIteration) {
979 
980  bool Inner_IterConv = GetConvergence() || config->GetnInner_Iter()-1 <= curInnerIter; //Check, if Inner_Iter is converged
981 
982  if(TimeIteration == 0){
983  for (unsigned short iField_Conv = 0; iField_Conv < wndConvFields.size(); iField_Conv++){
984  const string WndConv_Field= wndConvFields[iField_Conv];

992  unsigned short iCounter;
993 
994  for (unsigned short iField_Conv = 0; iField_Conv < wndConvFields.size(); iField_Conv++){
995  bool fieldConverged = false;
996  const string WndConv_Field= wndConvFields[iField_Conv];
997

1002  if (std::isnan(SU2_TYPE::GetValue(monitor))) {
1003  SU2_MPI::Error("SU2 has diverged (NaN detected).", CURRENT_FUNCTION);
1004  }

1010  for (iCounter = 0; iCounter < nWndCauchy_Elems; iCounter++){
1011  WndCauchy_Serie[iField_Conv][iCounter] = 0.0;
1012  }
1013  WndNew_Func[iField_Conv] = monitor;
1014  }
1015  WndOld_Func[iField_Conv] = WndNew_Func[iField_Conv];
1016  WndNew_Func[iField_Conv] = monitor;
1017  WndCauchy_Func = fabs(WndNew_Func[iField_Conv] - WndOld_Func[iField_Conv]);
1018  WndCauchy_Serie[iField_Conv][TimeIteration % nWndCauchy_Elems] = WndCauchy_Func;
1019  WndCauchy_Value = 1.0;
1020 
1021  if (TimeIteration >= nWndCauchy_Elems+config->GetStartWindowIteration()){
1022  WndCauchy_Value = 0.0;
1023  for (iCounter = 0; iCounter < nWndCauchy_Elems; iCounter++){
1024  WndCauchy_Value += WndCauchy_Serie[iField_Conv][iCounter];
1025  }
1026  WndCauchy_Value /= nWndCauchy_Elems;
1027  }
1028  fieldConverged = WndCauchy_Value < wndCauchyEps;
1029 
1030  /*--- Start monitoring only if the current iteration is larger than the
1031  * number of cauchy elements and the number of start-up iterations ---*/
1032 
1033  if (TimeIteration < config->GetStartWindowIteration() + max(config->GetWnd_StartConv_Iter(), nWndCauchy_Elems)){
1034  fieldConverged = false;
1035  }
1036  SetHistoryOutputValue("CAUCHY_" + WndConv_Field, WndCauchy_Value);
1037  }
1038  TimeConvergence = fieldConverged && TimeConvergence;
1039  }
1040  }
1041 
1042  /*--- Do not apply any convergence criterion if the option is disabled. */
1043  if(!config->GetWnd_Cauchy_Crit()){TimeConvergence = false;}
1044  if(wndConvFields.empty()){TimeConvergence = false;}
1045  }
1046  return TimeConvergence;
