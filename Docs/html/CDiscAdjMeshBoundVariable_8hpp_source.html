34/*!

38class CDiscAdjMeshBoundVariable final : public CVariable {

44 CVertexMap<unsigned> VertexMap; /*!< \brief Object that controls accesses to the variables of this class. */

46public:

48 /*!

54 CDiscAdjMeshBoundVariable(unsigned long npoint, unsigned long ndim, CConfig *config);

59 ~CDiscAdjMeshBoundVariable() override = default;

65 void AllocateBoundaryVariables(CConfig *config);

67 /*!
68 * \brief Get the value of the displacement imposed at the boundary.
69 * \return Value of the boundary displacement.
70 */
71 inline const su2double* GetBoundDisp_Direct(unsigned long iPoint) const override {
72 if (!VertexMap.GetVertexIndex(iPoint)) return nullptr;
73 return Bound_Disp_Direct[iPoint];
74 }
75
76 /*!
77 * \brief Set the solution for the boundary displacements.
78 * \param[in] val_BoundDisp - Pointer to the boundary displacements.
79 */
80 inline void SetBoundDisp_Direct(unsigned long iPoint, const su2double *val_BoundDisp) override {
81 if (!VertexMap.GetVertexIndex(iPoint)) return;
82 for (unsigned long iDim = 0; iDim < nDim; iDim++)
83 Bound_Disp_Direct(iPoint,iDim) = val_BoundDisp[iDim];
84 }
85
86 /*!
87 * \brief Set the value of the sensitivity with respect to the undeformed coordinates.
88 * \param[in] val_sens - Pointer to the sensitivities of the boundary displacements.
89 */
90 inline void SetBoundDisp_Sens(unsigned long iPoint, const su2double *val_sens) override {
91 if (!VertexMap.GetVertexIndex(iPoint)) return;
92 for (unsigned long iDim = 0; iDim < nDim; iDim++)
93 Bound_Disp_Sens(iPoint,iDim) = val_sens[iDim];
94 }
95
96 /*!
97 * \brief Get the value of the sensitivity with respect to the undeformed coordinates.
98 * \param[in] iDim - Index of Mesh_Coord_Sens[nDim]
99 * \return Value of the original Mesh_Coord_Sens iDim.
100 */
101 inline su2double GetBoundDisp_Sens(unsigned long iPoint, unsigned long iDim) const override {
102 if (!VertexMap.GetVertexIndex(iPoint)) return 0.0;
103 return Bound_Disp_Sens(iPoint,iDim);
104 }
105
106 /*!
107 * \brief Get whether a node is on the boundary
108 */
109 inline bool Get_isVertex(unsigned long iPoint) const override {
110 return VertexMap.GetIsVertex(iPoint);
111 }
112
113 /*!
114 * \brief Set whether a node is on the boundary
115 */
116 inline void Set_isVertex(unsigned long iPoint, bool isVertex) override {
117 VertexMap.SetIsVertex(iPoint,isVertex);
118 }
119
120 /*!
121 * \brief Get the value of the BGS solution.
122 */
123 inline su2double Get_BGSSolution(unsigned long iPoint, unsigned long iDim) const override {
124 if (!VertexMap.GetVertexIndex(iPoint)) return 0.0;
125 return Bound_Disp_Sens(iPoint,iDim);
126 }
127
128 /*!
129 * \brief Set the value of the solution in the previous BGS subiteration.
130 */
131 void Set_BGSSolution_k() override;
132
133 /*!
134 * \brief Restore the previous BGS subiteration to solution.
135 */
136 void Restore_BGSSolution_k() override;
137
138 /*!
139 * \brief Get the value of the solution in the previous BGS subiteration.
140 * \param[out] val_solution - solution in the previous BGS subiteration.
141 */
142 inline su2double Get_BGSSolution_k(unsigned long iPoint, unsigned long iDim) const override {
143 if (!VertexMap.GetVertexIndex(iPoint)) return 0.0;
144 return Solution_BGS_k(iPoint,iDim);
145 }
146
147 /*!
148 * \brief Get the value of the solution in the previous BGS subiteration.
149 * \param[out] val_solution - solution in the previous BGS subiteration.
150 */
151 inline void Set_BGSSolution_k(unsigned long iPoint, unsigned long iDim, su2double val) override {
152 if (!VertexMap.GetVertexIndex(iPoint)) return;
153 Solution_BGS_k(iPoint,iDim) = val;
154 }
155
156};
