64void CNewtonIntegration::Setup() {
65
66 auto iparam = config->GetNewtonKrylovIntParam();
67 auto dparam = config->GetNewtonKrylovDblParam();
68
69 startupIters = iparam[0];
70 startupResidual = dparam[0];
71 precondIters = iparam[1];
72 precondTol = dparam[1];
73 tolRelaxFactor = iparam[2];
74 fullTolResidual = dparam[2];
75 finDiffStepND = SU2_TYPE::GetValue(dparam[3]);
76
77 const auto nVar = solvers[FLOW_SOL]->GetnVar();
78 const auto nPoint = geometry->GetnPoint();
79 const auto nPointDomain = geometry->GetnPointDomain();
80
81 omp_chunk_size = computeStaticChunkSize(nPoint, omp_get_max_threads(), 1024);
82
83 LinSolver.SetxIsZero(true);
84
85 LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
86
87 if (!std::is_same<Scalar,su2double>::value) {
88 LinSysSol.Initialize(nPoint, nPointDomain, nVar, nullptr);
89 }
90
91 /*--- Check if the solver is able to provide a linear preconditioner. ---*/
92 if (config->GetKind_TimeIntScheme() != EULER_IMPLICIT) return;
93
94 const auto kindPrec = static_cast<ENUM_LINEAR_SOLVER_PREC>(config->GetKind_Linear_Solver_Prec());
95
96 preconditioner = CPreconditioner<MixedScalar>::Create(kindPrec, solvers[FLOW_SOL]->Jacobian, geometry, config);
97
98 if (!std::is_same<Scalar,MixedScalar>::value) {
99 precondIn.Initialize(nPoint, nPointDomain, nVar, nullptr);
100 precondOut.Initialize(nPoint, nPointDomain, nVar, nullptr);
101 }
102
103 /*--- Only possible with a preconditioner. ---*/
104 startupPeriod = (startupIters > 0) || (startupResidual < 0.0);
105
106}

108void CNewtonIntegration::PerturbSolution(const CSysVector<Scalar>& dir, Scalar mag) {
109
110 SU2_OMP_FOR_STAT(omp_chunk_size)
111 for (auto iPoint = 0ul; iPoint < geometry->GetnPoint(); ++iPoint) {
112 SU2_OMP_SIMD
113 for (auto iVar = 0ul; iVar < solvers[FLOW_SOL]->GetnVar(); ++iVar)
114 solvers[FLOW_SOL]->GetNodes()->AddSolution(iPoint,iVar, mag*dir(iPoint,iVar));
115 }
116 END_SU2_OMP_FOR

119void CNewtonIntegration::ComputeResiduals(ResEvalType type) {
120
121 /*--- Save the default integration scheme, and force to explicit if required. ---*/
122 auto TimeIntScheme = config->GetKind_TimeIntScheme();
123 if (type == ResEvalType::EXPLICIT) {
124 SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetKind_TimeIntScheme(EULER_EXPLICIT);)
125 }
126
127 solvers[FLOW_SOL]->Preprocessing(geometry, solvers, config, MESH_0, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
128
129 if (type == ResEvalType::DEFAULT) {
130 solvers[FLOW_SOL]->SetTime_Step(geometry, solvers, config, MESH_0, config->GetTimeIter());
131 }
132
133 Space_Integration(geometry, solvers, numerics[FLOW_SOL], config, MESH_0, NO_RK_ITER, RUNTIME_FLOW_SYS);
134
135 /*--- Restore default. ---*/
136 if (type == ResEvalType::EXPLICIT) {
137 SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetKind_TimeIntScheme(TimeIntScheme);)
138 }

142void CNewtonIntegration::ComputeFinDiffStep() {
143
144 static su2double rmsSol;
145 su2double rmsSol_loc = 0.0;
146
147 SU2_OMP_MASTER
148 rmsSol = 0.0;
149 END_SU2_OMP_MASTER
150
151 SU2_OMP_FOR_STAT(omp_chunk_size)
152 for (auto iPoint = 0ul; iPoint < geometry->GetnPointDomain(); ++iPoint)
153 for (auto iVar = 0ul; iVar < solvers[FLOW_SOL]->GetnVar(); ++iVar)
154 rmsSol_loc += pow(solvers[FLOW_SOL]->GetNodes()->GetSolution(iPoint,iVar), 2);
155 END_SU2_OMP_FOR
156
157 atomicAdd(rmsSol_loc, rmsSol);
158
159 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
160 {
161 su2double t = rmsSol;
162 SU2_MPI::Allreduce(&t, &rmsSol, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
163 finDiffStep = finDiffStepND * max(1.0, sqrt(SU2_TYPE::GetValue(rmsSol) / geometry->GetGlobal_nPointDomain()));
164 }
165 END_SU2_OMP_SAFE_GLOBAL_ACCESS
166}

168void CNewtonIntegration::MultiGrid_Iteration(CGeometry ****geometry_, CSolver *****solvers_, CNumerics ******numerics_,
169 CConfig **config_, unsigned short EqSystem, unsigned short iZone,
170 unsigned short iInst) {
171 config = config_[iZone];
172 solvers = solvers_[iZone][iInst][MESH_0];
173 geometry = geometry_[iZone][iInst][MESH_0];
174 numerics = numerics_[iZone][iInst][MESH_0];
175
176 if (!setup) { Setup(); setup = true; }
177
178 SU2_OMP_PARALLEL_(if(solvers[FLOW_SOL]->GetHasHybridParallel())) {
179
180 /*--- Save the current solution to be able to perturb it. ---*/
181
182 solvers[FLOW_SOL]->Set_OldSolution();
183
184 /*--- Current residual. ---*/
185
186 ComputeResiduals(ResEvalType::DEFAULT);
187
188 /*--- Compute the approximate Jacobian for preconditioning. ---*/
189
190 solvers[FLOW_SOL]->PrepareImplicitIteration(geometry, solvers, config);
191
192 if (preconditioner) preconditioner->Build();
193
194 SU2_OMP_FOR_STAT(omp_chunk_size)
195 for (auto i = 0ul; i < LinSysRes.GetNElmDomain(); ++i)
196 LinSysRes[i] = SU2_TYPE::GetValue(solvers[FLOW_SOL]->LinSysRes[i]);
197 END_SU2_OMP_FOR
198
199 su2double residual = 0.0;
200 for (auto iVar = 0ul; iVar < LinSysRes.GetNVar(); ++iVar)
201 residual += log10(solvers[FLOW_SOL]->GetRes_RMS(iVar)) / LinSysRes.GetNVar();
202
203 /*--- Check if startup period should end after this iteration. ---*/
204
205 bool endStartup = false;
206
207 if (startupPeriod) {
208 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
209 firstResidual = max(firstResidual, residual);
210 if (startupIters) startupIters -= 1;
211 }
212 END_SU2_OMP_SAFE_GLOBAL_ACCESS
213 endStartup = (startupIters == 0) && (residual - firstResidual < startupResidual);
214 }
215
216 /*--- The NK solves are expensive, the tolerance is relaxed while the residuals are high. ---*/
217
218 Scalar toleranceFactor = 1.0;
219
220 if (!startupPeriod && tolRelaxFactor > 1 && fullTolResidual < 0.0) {
221 SU2_OMP_SAFE_GLOBAL_ACCESS(firstResidual = max(firstResidual, residual);)
222 su2double x = (residual - firstResidual) / fullTolResidual;
223 toleranceFactor = 1.0 + (tolRelaxFactor-1)*max(0.0, 1.0-SU2_TYPE::GetValue(x));
224 }
225
226 /*--- Solve for the solution update. ---*/
227
228 auto iter = config->GetLinear_Solver_Iter();
229 Scalar eps = SU2_TYPE::GetValue(config->GetLinear_Solver_Error());
230
231 auto& linSysSol = GetSolutionVec(solvers[FLOW_SOL]->LinSysSol);
232
233 if (startupPeriod) {
234 iter = Preconditioner_impl(LinSysRes, linSysSol, iter, eps);
235 }
236 else {
237 ComputeFinDiffStep();
238
239 eps *= toleranceFactor;
240 iter = LinSolver.FGMRES_LinSolver(LinSysRes, linSysSol, CMatrixFreeProductWrapper(this),
241 CPreconditionerWrapper(this), eps, iter, eps, false, config);
242 /*--- Scale back the residual to trick the CFL adaptation. ---*/
243 eps /= toleranceFactor;
244 }
245 SetSolutionResult(solvers[FLOW_SOL]->LinSysSol);
246
247 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
248 solvers[FLOW_SOL]->SetIterLinSolver(iter);
249 solvers[FLOW_SOL]->SetResLinSolver(eps);
250 }
251 END_SU2_OMP_SAFE_GLOBAL_ACCESS
252
253 /// TODO: Clever back-tracking and CFL adaptation based on residual reduction.
254
255 /*--- Update solution. ---*/
256
257 solvers[FLOW_SOL]->CompleteImplicitIteration(geometry, solvers, config);
258
259 /*--- Call the various post processings. ---*/
260
261 solvers[FLOW_SOL]->Preprocessing(geometry, solvers, config, MESH_0, NO_RK_ITER, RUNTIME_FLOW_SYS, true);
262
263 solvers[FLOW_SOL]->Postprocessing(geometry, solvers, config, MESH_0);
264
265 SU2_OMP_MASTER {
266 solvers[FLOW_SOL]->Pressure_Forces(geometry, config);
267 solvers[FLOW_SOL]->Momentum_Forces(geometry, config);
268 solvers[FLOW_SOL]->Friction_Forces(geometry, config);
269 }
270 END_SU2_OMP_MASTER
271
272 /*--- At the end of the startup period the CFL is reset to the initial value. ---*/
273
274 if (endStartup) {
275 SU2_OMP_MASTER {
276 startupPeriod = false;
277 firstResidual = residual;
278 }
279 END_SU2_OMP_MASTER
280 SU2_OMP_FOR_STAT(omp_chunk_size)
281 for (auto iPoint = 0ul; iPoint < geometry->GetnPoint(); ++iPoint)
282 solvers[FLOW_SOL]->GetNodes()->SetLocalCFL(iPoint, config->GetCFL(MESH_0));
283 END_SU2_OMP_FOR
284 }
285
286 }
287 END_SU2_OMP_PARALLEL
288}

290void CNewtonIntegration::MatrixFreeProduct(const CSysVector<Scalar>& u, CSysVector<Scalar>& v) {
291
292 Scalar factor = finDiffStep / u.norm();
293
294 PerturbSolution(u, factor);
295
296 ComputeResiduals(ResEvalType::EXPLICIT);
297
298 /*--- Finalize product. ---*/
299 factor = 1.0 / factor;
300
301 SU2_OMP_FOR_STAT(omp_chunk_size)
302 for (auto iPoint = 0ul; iPoint < geometry->GetnPointDomain(); ++iPoint) {
303 su2double delta = (geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint)) /
304 max(EPS, solvers[FLOW_SOL]->GetNodes()->GetDelta_Time(iPoint));
305 SU2_OMP_SIMD
306 for (auto iVar = 0ul; iVar < LinSysRes.GetNVar(); ++iVar) {
307 Scalar perturbRes = SU2_TYPE::GetValue(solvers[FLOW_SOL]->LinSysRes(iPoint,iVar));
308
309 /*--- The global residual had its sign flipped, so we add to get the difference. ---*/
310 v(iPoint,iVar) = (perturbRes + LinSysRes(iPoint,iVar)) * factor;
311
312 /*--- Pseudotime term of the true Jacobian. ---*/
313 v(iPoint,iVar) += SU2_TYPE::GetValue(delta) * u(iPoint,iVar);
314 }
315 }
316 END_SU2_OMP_FOR
317
318 CSysMatrixComms::Initiate(v, geometry, config);
319 CSysMatrixComms::Complete(v, geometry, config);
320}

322void CNewtonIntegration::Preconditioner(const CSysVector<Scalar>& u, CSysVector<Scalar>& v) const {
323
324 if (preconditioner) {
325 Scalar eps = SU2_TYPE::GetValue(precondTol);
326 Preconditioner_impl(u, v, precondIters, eps);
327 }
328 else {
329 /*--- Approximate diagonal preconditioner. ---*/
330
331 const bool dt1st = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
332 const bool dt2nd = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
333 const su2double dt = config->GetDelta_UnstTimeND() * (dt1st + 1.5 * dt2nd);
334
335 SU2_OMP_FOR_STAT(omp_chunk_size)
336 for (auto iPoint = 0ul; iPoint < geometry->GetnPointDomain(); ++iPoint) {
337 su2double delta = (solvers[FLOW_SOL]->GetNodes()->GetDelta_Time(iPoint) + dt) /
338 (geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint));
339 SU2_OMP_SIMD
340 for (auto iVar = 0ul; iVar < u.GetNVar(); ++iVar)
341 v(iPoint,iVar) = SU2_TYPE::GetValue(delta) * u(iPoint,iVar);
342 }
343 END_SU2_OMP_FOR
344
345 CSysMatrixComms::Initiate(v, geometry, config);
346 CSysMatrixComms::Complete(v, geometry, config);
347 }
