38class CPrimalGridBoundFEM final : public CPrimalGrid {
39 private:
40 unsigned long boundElemIDGlobal; /*!< \brief Global boundary element ID of this element. */

42 std::vector<unsigned long> donorElementsWallFunctions; /*!< \brief The global ID's of the donor elements

44
45 unsigned short VTK_Type; /*!< \brief Element type using the VTK convention. */

48

52 /*!
53 * \brief Constructor using data to initialize the boundary element.
54 * \param[in] val_elemGlobalID - Global boundary element ID of this element.
55 * \param[in] val_domainElementID - Global ID of the corresponding domain element.
56 * \param[in] val_VTK_Type - VTK type to indicate the element type
57 * \param[in] val_nPolyGrid - Polynomial degree to describe the geometry of the element.

63 std::vector<unsigned long>& val_nodes);
64
65 /*!

68 * \return Number of nodes that composes a face of an element.

70 inline unsigned short GetnNodesFace(unsigned short val_face) const override {
71 return std::numeric_limits<unsigned short>::max();
72 }
73
74 /*!
75 * \brief Get the face index of an element.
76 * \param[in] val_face - Local index of the face.

78 * \return Local (to the element) index of the nodes that compose the face.
79 */
80 inline unsigned short GetFaces(unsigned short val_face, unsigned short val_index) const override {
81 return std::numeric_limits<unsigned short>::max();
82 }
83

90 inline unsigned short GetNeighbor_Nodes(unsigned short val_node, unsigned short val_index) const override {
91 return std::numeric_limits<unsigned short>::max();
92 }
93
94 /*!
95 * \brief Get the number of nodes of an element.

100 /*!

184

190 donorElementsWallFunctions.push_back(donorElement);
191 }
192

194 * \brief Make available the number of donor elements for the wall function treatment.

198
199 /*!
200 * \brief Make available the pointer to the vector for the donor elements

205 inline const unsigned long* GetDonorsWallFunctions(void) const override { return donorElementsWallFunctions.data(); }
206
207 /*!
208 * \brief Set the global ID's of the donor elements for the wall function treatment.
209 * \param[in] donorElements - Vector, which contain the donor elements.
210 */
211 inline void SetDonorsWallFunctions(const std::vector<unsigned long>& donorElements) override {
212 donorElementsWallFunctions = donorElements;
213 }
214
215 /*!
216 * \brief Function to remove the multiple donors for the wall function treatment.
217 */
218 void RemoveMultipleDonorsWallFunctions(void) override;
219};
