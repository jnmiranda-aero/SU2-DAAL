39 /*!
40  * \class CPreconditioner
41  * \brief Abstract base class for defining a preconditioning operation.
42  * \author J. Hicken.
43  *
44  * See the remarks regarding the CMatrixVectorProduct class. The same
45  * idea applies here to the preconditioning operation.
46  */
47 template <class ScalarType>
48 class CPreconditioner {
49  public:
50  /*!
51  * \brief Destructor of the class
52  */
53  virtual ~CPreconditioner() = 0;
54 
55  /*!
56  * \brief Overload of operator (), applies the preconditioner to "u" storing the result in "v".
57  */
58  virtual void operator()(const CSysVector<ScalarType>& u, CSysVector<ScalarType>& v) const = 0;
59 
60  /*!
61  * \brief Generic "preprocessing" hook derived classes may implement to build the preconditioner.
62  */
63  virtual void Build() {}
64 
65  /*!
66  * \brief Return true to identify the identity preconditioner, may allow some solvers to be more efficient.
67  */
68  virtual bool IsIdentity() const { return false; }
69 
70  /*!
71  * \brief Factory method.
72  */
73  static CPreconditioner* Create(ENUM_LINEAR_SOLVER_PREC kind, CSysMatrix<ScalarType>& jacobian, CGeometry* geometry,
74  const CConfig* config);
75 };

77 CPreconditioner<ScalarType>::~CPreconditioner() {}

83 template <class ScalarType>
84 class CJacobiPreconditioner final : public CPreconditioner<ScalarType> {
85  private:
86  CSysMatrix<ScalarType>& sparse_matrix; /*!< \brief Pointer to matrix that defines the preconditioner. */
87  CGeometry* geometry; /*!< \brief Pointer to geometry associated with the matrix. */
88  const CConfig* config; /*!< \brief Pointer to problem configuration. */
89 
90  public:
91  /*!
92  * \brief Constructor of the class.
93  * \param[in] matrix_ref - Matrix reference that will be used to define the preconditioner.
94  * \param[in] geometry_ref - Geometry associated with the problem.
95  * \param[in] config_ref - Config of the problem.
96  */
97  inline CJacobiPreconditioner(CSysMatrix<ScalarType>& matrix_ref, CGeometry* geometry_ref, const CConfig* config_ref)
98  : sparse_matrix(matrix_ref) {
99  if ((geometry_ref == nullptr) || (config_ref == nullptr))
100  SU2_MPI::Error("Preconditioner needs to be built with valid references.", CURRENT_FUNCTION);
101  geometry = geometry_ref;
102  config = config_ref;
103  }
104 
105  /*!
106  * \note This class cannot be default constructed as that would leave us with invalid Pointers.
107  */
108  CJacobiPreconditioner() = delete;
109 
110  /*!
111  * \brief operator that defines the preconditioner operation
112  * \param[in] u - CSysVector that is being preconditioned
113  * \param[out] v - CSysVector that is the result of the preconditioning
114  */
115  inline void operator()(const CSysVector<ScalarType>& u, CSysVector<ScalarType>& v) const override {
116  sparse_matrix.ComputeJacobiPreconditioner(u, v, geometry, config);
117  }
118 
119  /*!
120  * \note Request the associated matrix to build the preconditioner.
121  */
122  inline void Build() override { sparse_matrix.BuildJacobiPreconditioner(); }
123 };

308 template <class ScalarType>
309 CPreconditioner<ScalarType>* CPreconditioner<ScalarType>::Create(ENUM_LINEAR_SOLVER_PREC kind,
310  CSysMatrix<ScalarType>& jacobian, CGeometry* geometry,
311  const CConfig* config) {
312  CPreconditioner<ScalarType>* prec = nullptr;
313 
314  switch (kind) {
315  case JACOBI:
316  prec = new CJacobiPreconditioner<ScalarType>(jacobian, geometry, config);
317  break;
318  case LINELET:
319  prec = new CLineletPreconditioner<ScalarType>(jacobian, geometry, config);
320  break;
321  case LU_SGS:
322  prec = new CLU_SGSPreconditioner<ScalarType>(jacobian, geometry, config);
323  break;
324  case ILU:
325  prec = new CILUPreconditioner<ScalarType>(jacobian, geometry, config);
326  break;
327  case PASTIX_ILU:
328  case PASTIX_LU_P:
329  case PASTIX_LDLT_P:
330  prec = new CPastixPreconditioner<ScalarType>(jacobian, geometry, config, kind);
331  break;
332  }
333 
334  return prec;
335 }
