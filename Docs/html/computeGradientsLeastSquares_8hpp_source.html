34/*!
35 * \brief Prepare Smatrix for 2D.
36 * \ingroup FvmAlgos
37 */
38FORCEINLINE void computeSmatrix(su2double r11, su2double r12, su2double r13,
39 su2double r22, su2double r23, su2double r33,
40 su2double detR2, su2double Smatrix[][2]) {
41 Smatrix[0][0] = (r12*r12+r22*r22)/detR2;
42 Smatrix[0][1] = -r11*r12/detR2;
43 Smatrix[1][1] = r11*r11/detR2;
44}

46/*!
47 * \brief Prepare Smatrix for 3D.
48 * \ingroup FvmAlgos
49 */
50FORCEINLINE void computeSmatrix(su2double r11, su2double r12, su2double r13,
51 su2double r22, su2double r23, su2double r33,
52 su2double detR2, su2double Smatrix[][3]) {
53 su2double z11 = r22*r33;
54 su2double z12 =-r12*r33;
55 su2double z13 = r12*r23-r13*r22;
56 su2double z22 = r11*r33;
57 su2double z23 =-r11*r23;
58 su2double z33 = r11*r22;
59
60 Smatrix[0][0] = (z11*z11+z12*z12+z13*z13)/detR2;
61 Smatrix[0][1] = (z12*z22+z13*z23)/detR2;
62 Smatrix[0][2] = (z13*z33)/detR2;
63 Smatrix[1][1] = (z22*z22+z23*z23)/detR2;
64 Smatrix[1][2] = (z23*z33)/detR2;
65 Smatrix[2][2] = (z33*z33)/detR2;
66}

68/*!
69 * \brief Solve the least-squares problem for one point.
70 * \ingroup FvmAlgos
71 * \note See detail::computeGradientsLeastSquares for the
72 * purpose of template "nDim" and "periodic".
73 */
74template<size_t nDim, bool periodic, class GradientType, class RMatrixType>
75FORCEINLINE void solveLeastSquares(size_t iPoint,
76 size_t varBegin,
77 size_t varEnd,
78 const RMatrixType& Rmatrix,
79 GradientType& gradient)
80{
81 const auto eps = pow(std::numeric_limits<passivedouble>::epsilon(),2);
82
83 /*--- Entries of upper triangular matrix R. ---*/
84
85 if (periodic) {
86 AD::StartPreacc();
87 AD::SetPreaccIn(Rmatrix(iPoint,0,0));
88 AD::SetPreaccIn(Rmatrix(iPoint,0,1));
89 AD::SetPreaccIn(Rmatrix(iPoint,1,1));
90 }
91
92 su2double r11 = Rmatrix(iPoint,0,0);
93 su2double r12 = Rmatrix(iPoint,0,1);
94 su2double r22 = Rmatrix(iPoint,1,1);
95 su2double r13 = 0.0, r23 = 0.0, r33 = 1.0;
96
97 r11 = sqrt(max(r11, eps));
98 r12 /= r11;
99 r22 = sqrt(max(r22 - r12*r12, eps));
100
101 if (nDim == 3) {
102 if (periodic) {
103 AD::SetPreaccIn(Rmatrix(iPoint,0,2));
104 AD::SetPreaccIn(Rmatrix(iPoint,1,2));
105 AD::SetPreaccIn(Rmatrix(iPoint,2,1));
106 AD::SetPreaccIn(Rmatrix(iPoint,2,2));
107 }
108
109 r13 = Rmatrix(iPoint,0,2);
110 r33 = Rmatrix(iPoint,2,2);
111 const auto r23_a = Rmatrix(iPoint,1,2);
112 const auto r23_b = Rmatrix(iPoint,2,1);
113
114 r13 /= r11;
115 r23 = r23_a/r22 - r23_b*r12/(r11*r22);
116 r33 = sqrt(max(r33 - r23*r23 - r13*r13, eps));
117 }
118
119 /*--- Compute determinant ---*/
120
121 const su2double detR2 = pow(r11*r22*r33, 2);
122
123 /*--- S matrix := inv(R)*traspose(inv(R)) ---*/
124
125 su2double Smatrix[nDim][nDim] = {{0.0}};
126
127 /*--- Detect singular matrix ---*/
128
129 if (detR2 > eps) {
130 computeSmatrix(r11, r12, r13, r22, r23, r33, detR2, Smatrix);
131 }
132
133 if (periodic) {
134 /*--- Stop preacc here as gradient is in/out. ---*/
135 for (size_t iDim = 0; iDim < nDim; ++iDim)
136 for (size_t jDim = iDim; jDim < nDim; ++jDim)
137 AD::SetPreaccOut(Smatrix[iDim][jDim]);
138 AD::EndPreacc();
139 }
140
141 /*--- Computation of the gradient: S*c ---*/
142
143 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
144 {
145 su2double Cvector[nDim] = {0.0};
146
147 for (size_t iDim = 0; iDim < nDim; ++iDim)
148 for (size_t jDim = 0; jDim < nDim; ++jDim)
149 Cvector[iDim] += Smatrix[min(iDim,jDim)][max(iDim,jDim)] * gradient(iPoint, iVar, jDim);
150
151 for (size_t iDim = 0; iDim < nDim; ++iDim)
152 gradient(iPoint, iVar, iDim) = Cvector[iDim];
153 }
154
155 if (!periodic) {
156 /*--- Stop preacc here instead as gradient is only out. ---*/
157 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
158 for (size_t iDim = 0; iDim < nDim; ++iDim)
159 AD::SetPreaccOut(gradient(iPoint, iVar, iDim));
160 AD::EndPreacc();
161 }
162}

164/*!
165 * \brief Compute the gradient of a field using inverse-distance-weighted or
166 * unweighted Least-Squares approximation.
167 * \ingroup FvmAlgos
168 * \note See notes from computeGradientsGreenGauss.hpp.
169 * \param[in] solver - Optional, solver associated with the field (used only for MPI).
170 * \param[in] kindMpiComm - Type of MPI communication required.
171 * \param[in] kindPeriodicComm - Type of periodic communication required.
172 * \param[in] geometry - Geometric grid properties.
173 * \param[in] weighted - Use inverse-distance weights.
174 * \param[in] config - Configuration of the problem, used to identify types of boundaries.
175 * \param[in] field - Generic object implementing operator (iPoint, iVar).
176 * \param[in] varBegin - Index of first variable for which to compute the gradient.
177 * \param[in] varEnd - Index of last variable for which to compute the gradient.
178 * \param[out] gradient - Generic object implementing operator (iPoint, iVar, iDim).
179 * \param[out] Rmatrix - Generic object implementing operator (iPoint, iDim, iDim).
180 */
181template<size_t nDim, class FieldType, class GradientType, class RMatrixType>
182void computeGradientsLeastSquares(CSolver* solver,
183 MPI_QUANTITIES kindMpiComm,
184 PERIODIC_QUANTITIES kindPeriodicComm,
185 CGeometry& geometry,
186 const CConfig& config,
187 bool weighted,
188 const FieldType& field,
189 size_t varBegin,
190 size_t varEnd,
191 GradientType& gradient,
192 RMatrixType& Rmatrix)
193{
194 const bool periodic = (solver != nullptr) && (config.GetnMarker_Periodic() > 0);
195
196 const size_t nPointDomain = geometry.GetnPointDomain();
197
198#ifdef HAVE_OMP
199 constexpr size_t OMP_MAX_CHUNK = 512;
200
201 size_t chunkSize = computeStaticChunkSize(nPointDomain,
202 omp_get_max_threads(), OMP_MAX_CHUNK);
203#endif
204
205 /*--- First loop over non-halo points of the grid. ---*/
206
207 SU2_OMP_FOR_DYN(chunkSize)
208 for (size_t iPoint = 0; iPoint < nPointDomain; ++iPoint)
209 {
210 auto nodes = geometry.nodes;
211 const auto coord_i = nodes->GetCoord(iPoint);
212
213 /*--- Cannot preaccumulate if hybrid parallel due to shared reading. ---*/
214 if (omp_get_num_threads() == 1) AD::StartPreacc();
215 AD::SetPreaccIn(coord_i, nDim);
216
217 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
218 AD::SetPreaccIn(field(iPoint,iVar));
219
220 /*--- Clear gradient and Rmatrix. ---*/
221
222 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
223 for (size_t iDim = 0; iDim < nDim; ++iDim)
224 gradient(iPoint, iVar, iDim) = 0.0;
225
226 for (size_t iDim = 0; iDim < nDim; ++iDim)
227 for (size_t jDim = 0; jDim < nDim; ++jDim)
228 Rmatrix(iPoint, iDim, jDim) = 0.0;
229
230
231 for (auto jPoint : nodes->GetPoints(iPoint))
232 {
233 const auto coord_j = geometry.nodes->GetCoord(jPoint);
234 AD::SetPreaccIn(coord_j, nDim);
235
236
237 /*--- Distance vector from iPoint to jPoint ---*/
238
239 su2double dist_ij[nDim] = {0.0};
240 GeometryToolbox::Distance(nDim, coord_j, coord_i, dist_ij);
241
242
243 /*--- Compute inverse weight, default 1 (unweighted). ---*/
244
245 su2double weight = 1.0;
246 if(weighted) weight = GeometryToolbox::SquaredNorm(nDim, dist_ij);
247
248 /*--- Summations for entries of upper triangular matrix R. ---*/
249
250 if (weight > 0.0)
251 {
252 weight = 1.0 / weight;
253
254 for (size_t iDim = 0; iDim < nDim; ++iDim)
255 for (size_t jDim = iDim; jDim < nDim; ++jDim)
256 Rmatrix(iPoint,iDim,jDim) += dist_ij[iDim]*dist_ij[jDim]*weight;
257
258 if (nDim == 3)
259 Rmatrix(iPoint,2,1) += dist_ij[0]*dist_ij[nDim-1]*weight;
260
261 /*--- Entries of c:= transpose(A)*b ---*/
262
263 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
264 {
265 AD::SetPreaccIn(field(jPoint,iVar));
266
267 su2double delta_ij = weight * (field(jPoint,iVar) - field(iPoint,iVar));
268
269 for (size_t iDim = 0; iDim < nDim; ++iDim)
270 gradient(iPoint, iVar, iDim) += dist_ij[iDim] * delta_ij;
271 }
272 }
273 }
274
275 if (periodic)
276 {
277 /*--- A second loop is required after periodic comms, checkpoint the preacc. ---*/
278
279 for (size_t iDim = 0; iDim < nDim; ++iDim)
280 for (size_t jDim = 0; jDim < nDim; ++jDim)
281 AD::SetPreaccOut(Rmatrix(iPoint, iDim, jDim));
282
283 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
284 for (size_t iDim = 0; iDim < nDim; ++iDim)
285 AD::SetPreaccOut(gradient(iPoint, iVar, iDim));
286
287 AD::EndPreacc();
288 }
289 else {
290 /*--- Periodic comms are not needed, solve the LS problem for iPoint. ---*/
291
292 solveLeastSquares<nDim, false>(iPoint, varBegin, varEnd, Rmatrix, gradient);
293 }
294 }
295 END_SU2_OMP_FOR
296
297 /*--- Correct the gradient values across any periodic boundaries. ---*/
298
299 if (periodic)
300 {
301 for (size_t iPeriodic = 1; iPeriodic <= config.GetnMarker_Periodic()/2; ++iPeriodic)
302 {
303 solver->InitiatePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm);
304 solver->CompletePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm);
305 }
306
307 /*--- Second loop over points of the grid to compute final gradient. ---*/
308
309 SU2_OMP_FOR_DYN(chunkSize)
310 for (size_t iPoint = 0; iPoint < nPointDomain; ++iPoint)
311 solveLeastSquares<nDim, true>(iPoint, varBegin, varEnd, Rmatrix, gradient);
312 END_SU2_OMP_FOR
313 }
314
315 /*--- If no solver was provided we do not communicate ---*/
316
317 if (solver != nullptr)
318 {
319 /*--- Obtain the gradients at halo points from the MPI ranks that own them. ---*/
320
321 solver->InitiateComms(&geometry, &config, kindMpiComm);
322 solver->CompleteComms(&geometry, &config, kindMpiComm);
323 }
324
325}
326} // end namespace

332template<class FieldType, class GradientType, class RMatrixType>
333void computeGradientsLeastSquares(CSolver* solver,
334 MPI_QUANTITIES kindMpiComm,
335 PERIODIC_QUANTITIES kindPeriodicComm,
336 CGeometry& geometry,
337 const CConfig& config,
338 bool weighted,
339 const FieldType& field,
340 size_t varBegin,
341 size_t varEnd,
342 GradientType& gradient,
343 RMatrixType& Rmatrix) {
344 switch (geometry.GetnDim()) {
345 case 2:
346 detail::computeGradientsLeastSquares<2>(solver, kindMpiComm, kindPeriodicComm, geometry, config,
347 weighted, field, varBegin, varEnd, gradient, Rmatrix);
348 break;
349 case 3:
350 detail::computeGradientsLeastSquares<3>(solver, kindMpiComm, kindPeriodicComm, geometry, config,
351 weighted, field, varBegin, varEnd, gradient, Rmatrix);
352 break;
353 default:
354 SU2_MPI::Error("Too many dimensions to compute gradients.", CURRENT_FUNCTION);
355 break;
356 }
