31 : CVariable(npoint, ndim, nvar, config),
32 Gradient_Reconstruction(config->GetReconstructionGradientRequired() ? Gradient_Aux : Gradient) {
33 /*--- Gradient related fields ---*/
34
35 Gradient.resize(nPoint, nVar, nDim, 0.0);
36

40

42 Rmatrix.resize(nPoint, nDim, nDim, 0.0);
43 }
44
45 /*--- Always allocate the slope limiter, and the auxiliary

47
48 Limiter.resize(nPoint, nVar) = su2double(0.0);
49 Solution_Max.resize(nPoint, nVar) = su2double(0.0);
50 Solution_Min.resize(nPoint, nVar) = su2double(0.0);
51
52 Delta_Time.resize(nPoint) = su2double(0.0);
53
54 /* Under-relaxation parameter. */
55 UnderRelaxation.resize(nPoint) = su2double(1.0);
56 LocalCFL.resize(nPoint) = su2double(0.0);
57
58 /*--- Allocate space for the harmonic balance source terms ---*/
59 if (config->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE) {
60 HB_Source.resize(nPoint, nVar) = su2double(0.0);
