38/*!
39 * \class CPrimitiveIndices
40 * \brief Abstract wrapper for the CIndices classes of CEulerVariable, CIncEulerVariable, etc.
41 * This should only be used when the concrete type of solver variable is not known.
42 * \note We are using some tricks to try to make this more efficient than virtual functions.
43 */
44template <class IndexType>
45struct CPrimitiveIndices {
46 CPrimitiveIndices(bool incompressible, bool nemo, IndexType nDim, IndexType nSpecies) {
47 /*--- Instantiate the right type of CIndices, without heap allocations. ---*/
48 if (incompressible) {
49 type_ = 0;
50 Construct<CIncEulerVariable::template CIndices<IndexType>>(nDim, nSpecies);
51 } else if (nemo) {
52 type_ = 1;
53 Construct<CNEMOEulerVariable::template CIndices<IndexType>>(nDim, nSpecies);
54 } else {
55 type_ = 2;
56 Construct<CEulerVariable::template CIndices<IndexType>>(nDim, nSpecies);
57 }
58 }
59
60 /*--- To use the CIndices object we stored on the buffer, we cast to the
61 * concrete type using the same logic used during construction. ---*/
62#define GET_INDEX_IMPL(NAME) \
63 inline IndexType NAME() const { \
64 switch (type_) { \
65 case 0: \
66 return reinterpret_cast<const CIncEulerVariable::template CIndices<IndexType>*>(data_)->NAME(); \
67 case 1: \
68 return reinterpret_cast<const CNEMOEulerVariable::template CIndices<IndexType>*>(data_)->NAME(); \
69 default: \
70 return reinterpret_cast<const CEulerVariable::template CIndices<IndexType>*>(data_)->NAME(); \
71 } \
72 }
73 GET_INDEX_IMPL(NDim)
74 GET_INDEX_IMPL(NSpecies)
75 GET_INDEX_IMPL(SpeciesDensities)
76 GET_INDEX_IMPL(Temperature)
77 GET_INDEX_IMPL(Velocity)
78 GET_INDEX_IMPL(Pressure)
79 GET_INDEX_IMPL(Density)
80 GET_INDEX_IMPL(Enthalpy)
81 GET_INDEX_IMPL(SoundSpeed)
82 GET_INDEX_IMPL(LaminarViscosity)
83 GET_INDEX_IMPL(EddyViscosity)
84 GET_INDEX_IMPL(ThermalConductivity)
85 GET_INDEX_IMPL(CpTotal)
86 GET_INDEX_IMPL(Temperature_ve)
87#undef GET_INDEX_IMPL
88
89 private:
90 template <class ConcreteIndices>
91 void Construct(IndexType nDim, IndexType nSpecies) {
92 /*--- Build the indices object in the static buffer owned by this class. ---*/
93 static_assert(sizeof(ConcreteIndices) <= 2 * sizeof(IndexType), "");
94 new(data_) ConcreteIndices(nDim, nSpecies);
95 }
96
97 alignas(sizeof(IndexType)) char data_[2 * sizeof(IndexType)]{}; /*!< \brief Space for the largest CIndices class. */
98 uint8_t type_;
99};

101/*!
102 * \brief Maps primitive variable names to their indices based on a CPrimitiveIndices object.
103 * \note The variables names we try to map here must be kept in sync with CPrimitiveIndices.
104 * All index classes should use numeric_limits::max for variables that are not available.
105 */
106template <class IndexType>
107std::map<std::string, IndexType> PrimitiveNameToIndexMap(const CPrimitiveIndices<IndexType>& idx) {
108 std::map<std::string, IndexType> nameToIndex;
109 const auto notAvailable = std::numeric_limits<IndexType>::max();
110#define INSERT_VAR_INDEX_PAIR(STRING, FUNCTION) \
111 if (idx.FUNCTION() != notAvailable) nameToIndex[STRING] = idx.FUNCTION();
112 INSERT_VAR_INDEX_PAIR("TEMPERATURE", Temperature)
113 INSERT_VAR_INDEX_PAIR("TEMPERATURE_VE", Temperature_ve)
114 INSERT_VAR_INDEX_PAIR("PRESSURE", Pressure)
115 INSERT_VAR_INDEX_PAIR("DENSITY", Density)
116 INSERT_VAR_INDEX_PAIR("ENTHALPY", Enthalpy)
117 INSERT_VAR_INDEX_PAIR("SOUND_SPEED", SoundSpeed)
118 INSERT_VAR_INDEX_PAIR("LAMINAR_VISCOSITY", LaminarViscosity)
119 INSERT_VAR_INDEX_PAIR("EDDY_VISCOSITY", EddyViscosity)
120 INSERT_VAR_INDEX_PAIR("THERMAL_CONDUCTIVITY", ThermalConductivity)
121 INSERT_VAR_INDEX_PAIR("CP_TOTAL", CpTotal)
122#undef INSERT_VAR_INDEX_PAIR
123 nameToIndex["VELOCITY_X"] = idx.Velocity();
124 nameToIndex["VELOCITY_Y"] = idx.Velocity() + 1;
125 if (idx.NDim() == 3) {
126 nameToIndex["VELOCITY_Z"] = idx.Velocity() + 2;
127 }
128 for (IndexType iSpecies = 0; iSpecies < idx.NSpecies(); ++iSpecies) {
129 nameToIndex["DENSITY_" + std::to_string(iSpecies)] = idx.SpeciesDensities() + iSpecies;
130 }
131 return nameToIndex;
132}
