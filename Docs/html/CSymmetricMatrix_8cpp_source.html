52 void CSymmetricMatrix::CholeskyDecompose() {
53 #ifndef HAVE_LAPACK
54  int j;
55  for (j = 0; j < Size(); ++j) {
56  passivedouble sum = 0.0;
57  for (int k = 0; k < j; ++k) sum -= pow(Get(j, k), 2);
58  sum += Get(j, j);
59  if (sum < 0.0) break; // not SPD
60  Set(j, j, sqrt(sum));
61 
62  for (int i = j + 1; i < Size(); ++i) {
63  passivedouble sum = 0.0;
64  for (int k = 0; k < j; ++k) sum -= Get(i, k) * Get(j, k);
65  sum += Get(i, j);
66  Set(i, j, sum / Get(j, j));
67  }
68  }
69  if (j != Size()) SU2_MPI::Error("LLT factorization failed.", CURRENT_FUNCTION);
70 #endif
71 }

73 void CSymmetricMatrix::CalcInv(bool is_spd) {
74 #ifndef HAVE_LAPACK
75  const int sz = Size();
76 
77  /*--- Compute inverse from decomposed matrices. ---*/
78  if (is_spd) {
79  CholeskyDecompose();
80 
81  /*--- Initialize inverse matrix. ---*/
82  CSymmetricMatrix inv(sz);
83 
84  /*--- Compute L inverse. ---*/
85  /*--- Solve smaller and smaller systems. ---*/
86  for (int j = 0; j < sz; ++j) {
87  /*--- Forward substitution. ---*/
88  inv(j, j) = 1.0 / Get(j, j);
89 
90  for (int i = j + 1; i < sz; ++i) {
91  passivedouble sum = 0.0;
92  for (int k = j; k < i; ++k) sum -= Get(i, k) * inv(k, j);
93  inv(i, j) = sum / Get(i, i);
94  }
95  } // L inverse in inv
96 
97  /*--- Multiply inversed matrices overwrite mat. ---*/
98  for (int j = 0; j < sz; ++j)
99  for (int i = j; i < sz; ++i) {
100  passivedouble sum = 0.0;
101  for (int k = i; k < sz; ++k) sum += inv(k, i) * inv(k, j);
102  Set(i, j, sum);
103  }
104  } else {
105  auto inv = StealData();
106  CBlasStructure::inverse(sz, inv);
107  mat = move(inv);
108  }
109 #endif
110 }

112 void CSymmetricMatrix::CalcInv_sytri() {
113 #ifdef HAVE_LAPACK
114  const char uplo = 'L';
115  const int sz = Size();
116  int info;
117  vector<int> ipiv(sz);
118 
119  /*--- Query the optimum work size. ---*/
120  int query = -1;
121  passivedouble tmp;
122  dsytrf_(&uplo, &sz, mat.data(), &sz, ipiv.data(), &tmp, &query, &info);
123  query = static_cast<int>(tmp);
124  vector<passivedouble> work(query);
125 
126  /*--- Factorize and invert. ---*/
127  dsytrf_(&uplo, &sz, mat.data(), &sz, ipiv.data(), work.data(), &query, &info);
128  if (info != 0) SU2_MPI::Error("LDLT factorization failed.", CURRENT_FUNCTION);
129  dsytri_(&uplo, &sz, mat.data(), &sz, ipiv.data(), work.data(), &info);
130  if (info != 0) SU2_MPI::Error("Inversion with LDLT factorization failed.", CURRENT_FUNCTION);
131 #endif
132 }

158 void CSymmetricMatrix::MatMatMult(const char side, const su2passivematrix& mat_in, su2passivematrix& mat_out) const {
159  /*--- Left side: mat_out = this * mat_in. ---*/
160  if (side == 'L' || side == 'l') {
161  const int M = Size(), N = mat_in.cols();
162  assert(M == static_cast<int>(mat_in.rows()));
163 
164  mat_out.resize(M, N);
165 
166 #ifdef HAVE_LAPACK
167  /*--- Right and lower because matrices are in row major order. ---*/
168  const char side = 'R', uplo = 'L';
169  const passivedouble alpha = 1.0, beta = 0.0;
170  DSYMM(&side, &uplo, &N, &M, &alpha, mat.data(), &M, mat_in.data(), &N, &beta, mat_out.data(), &N);
171 #else // Naive product
172  for (int i = 0; i < M; ++i)
173  for (int j = 0; j < N; ++j) {
174  mat_out(i, j) = 0.0;
175  for (int k = 0; k < M; ++k) mat_out(i, j) += Get(i, k) * mat_in(k, j);
176  }
177 #endif
178  }
179  /*--- Right_side: mat_out = mat_in * this. ---*/
180  else {
181  const int M = mat_in.rows(), N = Size();
182  assert(N == static_cast<int>(mat_in.cols()));
183 
184  mat_out.resize(M, N);
185 
186 #ifdef HAVE_LAPACK
187  /*--- Left and lower because matrices are in row major order. ---*/
188  const char side = 'L', uplo = 'L';
189  const passivedouble alpha = 1.0, beta = 0.0;
190  DSYMM(&side, &uplo, &N, &M, &alpha, mat.data(), &N, mat_in.data(), &N, &beta, mat_out.data(), &N);
191 #else // Naive product
192  for (int i = 0; i < M; ++i)
193  for (int j = 0; j < N; ++j) {
194  mat_out(i, j) = 0.0;
195  for (int k = 0; k < N; ++k) mat_out(i, j) += mat_in(i, k) * Get(k, j);
196  }
197 #endif
198  }
199 }
