32CParaviewBinaryFileWriter::CParaviewBinaryFileWriter(CParallelDataSorter *valDataSorter) :
33 CFileWriter(valDataSorter, fileExt){
34
35 /* Check for big endian. We have to swap bytes otherwise.
36 * Since size of character is 1 byte when the character pointer
37 * is de-referenced it will contain only first byte of integer. ---*/
38
39 bigEndian = false;
40 unsigned int i = 1;
41 char *c = (char*)&i;
42 bigEndian = *c == 0;

48void CParaviewBinaryFileWriter::WriteData(string val_filename){
49
50 if (!dataSorter->GetConnectivitySorted()){
51 SU2_MPI::Error("Connectivity must be sorted.", CURRENT_FUNCTION);
52 }
53
54 const vector<string>& fieldNames = dataSorter->GetFieldNames();
55
56 unsigned short iDim = 0, nDim = dataSorter->GetnDim();
57
58 unsigned long iPoint, iElem;
59
60 const int MAX_STRING_LENGTH = 255;
61 char str_buf[MAX_STRING_LENGTH];
62
63 const int NCOORDS = 3;
64
65 OpenMPIFile(val_filename);
66
67 string header = "# vtk DataFile Version 3.0\n"
68 "vtk output\n"
69 "BINARY\n"
70 "DATASET UNSTRUCTURED_GRID\n";
71
72 WriteMPIString(header, MASTER_NODE);
73
74 /*--- Communicate the number of total points that will be
75 written by each rank. After this communication, each proc knows how
76 many poinnts will be written before its location in the file and the
77 offsets can be correctly set. ---*/
78
79 unsigned long myPoint, GlobalPoint;
80
81 GlobalPoint = dataSorter->GetnPointsGlobal();
82 myPoint = dataSorter->GetnPoints();
83
84 SPRINTF(str_buf, "POINTS %i float\n", SU2_TYPE::Int(GlobalPoint));
85
86 WriteMPIString(string(str_buf), MASTER_NODE);
87
88 /*--- Load/write the 1D buffer of point coordinates. Note that we
89 always have 3 coordinate dimensions, even for 2D problems. ---*/
90
91 vector<float> dataBufferFloat(myPoint*NCOORDS);
92 for (iPoint = 0; iPoint < myPoint; iPoint++) {
93 for (iDim = 0; iDim < NCOORDS; iDim++) {
94 if (nDim == 2 && iDim == 2) {
95 dataBufferFloat[iPoint*NCOORDS + iDim] = 0.0;
96 } else {
97 auto val = (float)dataSorter->GetData(iDim, iPoint);
98 dataBufferFloat[iPoint*NCOORDS + iDim] = val;
99 }
100 }
101 }
102
103 if (!bigEndian) SwapBytes((char *)dataBufferFloat.data(), sizeof(float), myPoint*NCOORDS);
104
105 /*--- Compute various data sizes --- */
106
107 unsigned long sizeInBytesPerPoint = sizeof(float)*NCOORDS;
108 unsigned long sizeInBytesLocal = sizeInBytesPerPoint*myPoint;
109 unsigned long sizeInBytesGlobal = sizeInBytesPerPoint*GlobalPoint;
110 unsigned long offsetInBytes = sizeInBytesPerPoint*dataSorter->GetnPointCumulative(rank);
111
112 WriteMPIBinaryDataAll(dataBufferFloat.data(), sizeInBytesLocal, sizeInBytesGlobal, offsetInBytes);
113
114 /*--- Compute our local number of elements, the required storage,
115 and reduce the total number of elements and storage globally. ---*/
116
117 unsigned long myElem, myElemStorage, GlobalElem, GlobalElemStorage;
118
119 unsigned long nParallel_Line = dataSorter->GetnElem(LINE),
120 nParallel_Tria = dataSorter->GetnElem(TRIANGLE),
121 nParallel_Quad = dataSorter->GetnElem(QUADRILATERAL),
122 nParallel_Tetr = dataSorter->GetnElem(TETRAHEDRON),
123 nParallel_Hexa = dataSorter->GetnElem(HEXAHEDRON),
124 nParallel_Pris = dataSorter->GetnElem(PRISM),
125 nParallel_Pyra = dataSorter->GetnElem(PYRAMID);
126
127 myElem = dataSorter->GetnElem();
128 myElemStorage = dataSorter->GetnConn();
129 GlobalElem = dataSorter->GetnElemGlobal();
130 GlobalElemStorage = dataSorter->GetnConnGlobal();
131
132 SPRINTF(str_buf, "\nCELLS %i %i\n", SU2_TYPE::Int(GlobalElem),
133 SU2_TYPE::Int(GlobalElemStorage+GlobalElem));
134 WriteMPIString(str_buf, MASTER_NODE);
135
136 /*--- Load/write 1D buffers for the connectivity of each element type. ---*/
137
138 vector<int> connBuf(myElemStorage + myElem);
139 unsigned long iStorage = 0;
140 unsigned short iNode = 0;
141
142 auto copyToBuffer = [&](GEO_TYPE type, unsigned long nElem, unsigned short nPoints){
143 for (iElem = 0; iElem < nElem; iElem++) {
144 connBuf[iStorage+0] = nPoints;
145 for (iNode = 0; iNode < nPoints; iNode++){
146 connBuf[iStorage+iNode+1] = int(dataSorter->GetElemConnectivity(type, iElem, iNode)-1);
147 }
148 iStorage += nPoints + 1;
149 }
150 };
151
152 copyToBuffer(LINE, nParallel_Line, N_POINTS_LINE);
153 copyToBuffer(TRIANGLE, nParallel_Tria, N_POINTS_TRIANGLE);
154 copyToBuffer(QUADRILATERAL, nParallel_Quad, N_POINTS_QUADRILATERAL);
155 copyToBuffer(TETRAHEDRON, nParallel_Tetr, N_POINTS_TETRAHEDRON);
156 copyToBuffer(HEXAHEDRON, nParallel_Hexa, N_POINTS_HEXAHEDRON);
157 copyToBuffer(PRISM, nParallel_Pris, N_POINTS_PRISM);
158 copyToBuffer(PYRAMID, nParallel_Pyra, N_POINTS_PYRAMID);
159
160 if (!bigEndian) SwapBytes((char *)connBuf.data(), sizeof(int), myElemStorage+myElem);
161
162 /*--- Compute various data sizes --- */
163
164 sizeInBytesPerPoint = sizeof(int);
165 sizeInBytesLocal = sizeInBytesPerPoint*(myElemStorage + myElem);
166 sizeInBytesGlobal = sizeInBytesPerPoint*(GlobalElemStorage + GlobalElem);
167 offsetInBytes = sizeInBytesPerPoint*
168 (dataSorter->GetnElemConnCumulative(rank) + dataSorter->GetnElemCumulative(rank));
169
170 WriteMPIBinaryDataAll(connBuf.data(), sizeInBytesLocal, sizeInBytesGlobal, offsetInBytes);
171
172 SPRINTF (str_buf, "\nCELL_TYPES %i\n", SU2_TYPE::Int(GlobalElem));
173 WriteMPIString(str_buf, MASTER_NODE);
174
175 /*--- Load/write the cell type for all elements in the file. ---*/
176
177 vector<int> typeBuf(myElem);
178 auto typeIter = typeBuf.begin();
179
180 std::fill(typeIter, typeIter+nParallel_Line, LINE); typeIter += nParallel_Line;
181 std::fill(typeIter, typeIter+nParallel_Tria, TRIANGLE); typeIter += nParallel_Tria;
182 std::fill(typeIter, typeIter+nParallel_Quad, QUADRILATERAL); typeIter += nParallel_Quad;
183 std::fill(typeIter, typeIter+nParallel_Tetr, TETRAHEDRON); typeIter += nParallel_Tetr;
184 std::fill(typeIter, typeIter+nParallel_Hexa, HEXAHEDRON); typeIter += nParallel_Hexa;
185 std::fill(typeIter, typeIter+nParallel_Pris, PRISM); typeIter += nParallel_Pris;
186 std::fill(typeIter, typeIter+nParallel_Pyra, PYRAMID); typeIter += nParallel_Pyra;
187
188 if (!bigEndian) SwapBytes((char *)typeBuf.data(), sizeof(int), myElem);
189
190 /*--- Compute various data sizes --- */
191
192 sizeInBytesPerPoint = sizeof(int);
193 sizeInBytesLocal = sizeInBytesPerPoint*myElem;
194 sizeInBytesGlobal = sizeInBytesPerPoint*GlobalElem;
195 offsetInBytes = sizeInBytesPerPoint*dataSorter->GetnElemCumulative(rank);
196
197 WriteMPIBinaryDataAll(typeBuf.data(), sizeInBytesLocal, sizeInBytesGlobal, offsetInBytes);
198
199 SPRINTF (str_buf, "\nPOINT_DATA %i\n", SU2_TYPE::Int(GlobalPoint));
200 WriteMPIString(str_buf, MASTER_NODE);
201
202 /*--- Adjust container start location to avoid point coords. ---*/
203
204 unsigned short varStart = 2;
205 if (nDim == 3) varStart++;
206
207 /*--- Loop over all variables that have been registered in the output. ---*/
208
209 unsigned short iField, VarCounter = varStart;
210 for (iField = varStart; iField < fieldNames.size(); iField++) {
211
212 string fieldname = fieldNames[iField];
213 fieldname.erase(remove(fieldname.begin(), fieldname.end(), '"'),
214 fieldname.end());
215
216 /*--- Check whether this field is a vector or scalar. ---*/
217
218 bool output_variable = true, isVector = false;
219 size_t found = fieldNames[iField].find("_x");
220 if (found!=string::npos) {
221 output_variable = true;
222 isVector = true;
223 }
224 found = fieldNames[iField].find("_y");
225 if (found!=string::npos) {
226 /*--- We have found a vector, so skip the Y component. ---*/
227 output_variable = false;
228 VarCounter++;
229 }
230 found = fieldNames[iField].find("_z");
231 if (found!=string::npos) {
232 /*--- We have found a vector, so skip the Z component. ---*/
233 output_variable = false;
234 VarCounter++;
235 }
236
237 /*--- Write the point data as an <X,Y,Z> vector or a scalar. ---*/
238
239 if (output_variable && isVector) {
240
241 /*--- Adjust the string name to remove the leading "X-" ---*/
242
243 fieldname.erase(fieldname.end()-2,fieldname.end());
244
245 SPRINTF (str_buf, "\nVECTORS %s float\n", fieldname.c_str());
246 WriteMPIString(str_buf, MASTER_NODE);
247
248 /*--- Load up the buffer for writing this rank's vector data. ---*/
249
250 float val = 0.0;
251 for (iPoint = 0; iPoint < myPoint; iPoint++) {
252 for (iDim = 0; iDim < NCOORDS; iDim++) {
253 if (nDim == 2 && iDim == 2) {
254 dataBufferFloat[iPoint*NCOORDS + iDim] = 0.0;
255 } else {
256 val = (float)dataSorter->GetData(VarCounter+iDim,iPoint);
257 dataBufferFloat[iPoint*NCOORDS + iDim] = val;
258 }
259 }
260 }
261 if (!bigEndian)
262 SwapBytes((char *)dataBufferFloat.data(), sizeof(float), myPoint*NCOORDS);
263
264 /*--- Compute various data sizes --- */
265
266 sizeInBytesPerPoint = sizeof(float)*NCOORDS;
267 sizeInBytesLocal = sizeInBytesPerPoint*myPoint;
268 sizeInBytesGlobal = sizeInBytesPerPoint*GlobalPoint;
269 offsetInBytes = sizeInBytesPerPoint*dataSorter->GetnPointCumulative(rank);
270
271 WriteMPIBinaryDataAll(dataBufferFloat.data(), sizeInBytesLocal, sizeInBytesGlobal, offsetInBytes);
272
273 VarCounter++;
274
275 } else if (output_variable) {
276
277 SPRINTF (str_buf, "\nSCALARS %s float 1\n", fieldname.c_str());
278 WriteMPIString(str_buf, MASTER_NODE);
279 WriteMPIString("LOOKUP_TABLE default\n", MASTER_NODE);
280
281 /*--- For now, create a temp 1D buffer to load up the data for writing.
282 This will be replaced with a derived data type most likely. ---*/
283
284 for (iPoint = 0; iPoint < myPoint; iPoint++) {
285 auto val = (float)dataSorter->GetData(VarCounter,iPoint);
286 dataBufferFloat[iPoint] = val;
287 }
288
289 if (!bigEndian)
290 SwapBytes((char *)dataBufferFloat.data(), sizeof(float), myPoint);
291
292 /*--- Compute various data sizes --- */
293
294 sizeInBytesPerPoint = sizeof(float);
295 sizeInBytesLocal = sizeInBytesPerPoint*myPoint;
296 sizeInBytesGlobal = sizeInBytesPerPoint*GlobalPoint;
297 offsetInBytes = sizeInBytesPerPoint*dataSorter->GetnPointCumulative(rank);
298
299 WriteMPIBinaryDataAll(dataBufferFloat.data(), sizeInBytesLocal, sizeInBytesGlobal, offsetInBytes);
300
301 VarCounter++;
302 }
303
304 }
305
306 CloseMPIFile();
307
308}

312 big endian, which is expected for ParaView binary legacy format. ---*/

314void CParaviewBinaryFileWriter::SwapBytes(char *buffer, size_t nBytes, unsigned long nVar) {
315
316 /*--- Store half the number of bytes in kk. ---*/
317
318 const int kk = (int)nBytes/2;
319
320 /*--- Loop over the number of variables in the buffer. ---*/
321
322 for (int j = 0; j < (int)nVar; j++) {
323
324 /*--- Initialize ii and jj, which are used to store the
325 indices of the bytes to be swapped. ---*/
326
327 int ii = j*(int)nBytes;
328 int jj = ii + (int)nBytes - 1;
329
330 /*--- Swap the bytes. ---*/
331
332 for (int i = 0; i < kk; i++) {
333 char tmp = buffer[jj];
334 buffer[jj] = buffer[ii];
335 buffer[ii] = tmp;
336
337 ii++;
338 jj--;
339
340 }
341 }
