34/*!
35 * \class CIncIdealGasPolynomial
36 * \brief Child class for defining a custom incompressible ideal gas model.
37 * \author: T. Economon
38 */
39template <int N>
40class CIncIdealGasPolynomial final : public CFluidModel {
41 public:
42 /*!
43 * \brief Constructor of the class.
44 */
45 CIncIdealGasPolynomial(su2double val_gas_constant, su2double val_operating_pressure) {
46 /* In the incompressible ideal gas model, the thermodynamic pressure is decoupled
47 from the governing equations and held constant. The density is therefore only a
48 function of temperature variations. We also use a molecular weight (g/mol) and the
49 universal gas constant to compute the specific gas constant for the fluid. The gas
50 is incompressible, so Cp = Cv (gamma = 1). */
51 Gas_Constant = val_gas_constant;
52 Pressure = val_operating_pressure;
53 Gamma = 1.0;
54 }
55
56 /*!
57 * \brief Set the temperature polynomial coefficients for variable Cp.
58 * \param[in] config - configuration container for the problem.
59 */
60 void SetCpModel(const CConfig* config) override {
61 for (int i = 0; i < N; ++i) {
62 coeffs_[i] = config->GetCp_PolyCoeffND(i);
63 }
64 }
65
66 /*!
67 * \brief Set the Dimensionless State using Temperature.
68 * \param[in] t - Temperature value at the point.
69 */
70 void SetTDState_T(su2double t, const su2double *val_scalars = nullptr) override {
71 /* The EoS only depends upon temperature. */
72 Temperature = t;
73 Density = Pressure / (Temperature * Gas_Constant);
74
75 /* Evaluate the new Cp from the coefficients and temperature. */
76 Cp = coeffs_[0];
77 su2double t_i = 1.0;
78 for (int i = 1; i < N; ++i) {
79 t_i *= t;
80 Cp += coeffs_[i] * t_i;
81 }
82 Cv = Cp / Gamma;
83 }
84
85 private:
86 su2double Gas_Constant{0.0}; /*!< \brief Specific Gas Constant. */
87 su2double Gamma{0.0}; /*!< \brief Ratio of specific heats. */
88 array<su2double, N> coeffs_; /*!< \brief Polynomial coefficients for conductivity as a function of temperature. */
89};
