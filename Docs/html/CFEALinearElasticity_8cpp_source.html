39 void CFEALinearElasticity::Compute_Tangent_Matrix(CElement *element, const CConfig *config) {
40 
41  unsigned short iVar, jVar, kVar;
42  unsigned short iGauss, nGauss;
43  unsigned short iNode, jNode, nNode;
44  unsigned short iDim;
45  unsigned short bDim;
46 
47  su2double Weight, Jac_X;
48 
49  su2double AuxMatrix[3][6], *res_aux = new su2double[nVar];
50 
51  /*--- Set element properties and recompute the constitutive matrix, this is needed
52  for multiple material cases and for correct differentiation ---*/
53  SetElement_Properties(element, config);
54 
55  /*--- Register pre-accumulation inputs, material props and nodal coords ---*/
56  AD::StartPreacc();
57  AD::SetPreaccIn(E);
58  AD::SetPreaccIn(Nu);
59  element->SetPreaccIn_Coords();
60  /*--- Recompute Lame parameters as they depend on the material properties ---*/
61  Compute_Lame_Parameters();
62 
63  Compute_Constitutive_Matrix(element, config);
64 
65  /*--- Initialize auxiliary matrices ---*/
66 
67  bDim = (nDim == 2) ? DIM_STRAIN_2D : DIM_STRAIN_3D;
68 
69  for (iVar = 0; iVar < bDim; iVar++) {
70  for (jVar = 0; jVar < nDim; jVar++) {
71  Ba_Mat[iVar][jVar] = 0.0;
72  Bb_Mat[iVar][jVar] = 0.0;
73  }
74  }
75 
76  for (iVar = 0; iVar < 3; iVar++) {
77  for (jVar = 0; jVar < 6; jVar++) {
78  AuxMatrix[iVar][jVar] = 0.0;
79  }
80  }
81 
82  element->ClearElement(); /*--- Restarts the element: avoids adding over previous results in other elements --*/
83  element->ComputeGrad_Linear();
84  nNode = element->GetnNodes();
85  nGauss = element->GetnGaussPoints();
86 
87  for (iGauss = 0; iGauss < nGauss; iGauss++) {
88 
89  Weight = element->GetWeight(iGauss);
90  Jac_X = element->GetJ_X(iGauss);
91 
92  /*--- Retrieve the values of the gradients of the shape functions for each node ---*/
93  /*--- This avoids repeated operations ---*/
94  for (iNode = 0; iNode < nNode; iNode++) {
95  for (iDim = 0; iDim < nDim; iDim++) {
96  GradNi_Ref_Mat[iNode][iDim] = element->GetGradNi_X(iNode,iGauss,iDim);
97  }
98  }
99 
100  for (iNode = 0; iNode < nNode; iNode++) {
101 
102  if (nDim == 2) {
103  Ba_Mat[0][0] = GradNi_Ref_Mat[iNode][0];
104  Ba_Mat[1][1] = GradNi_Ref_Mat[iNode][1];
105  Ba_Mat[2][0] = GradNi_Ref_Mat[iNode][1];
106  Ba_Mat[2][1] = GradNi_Ref_Mat[iNode][0];
107  }
108  else {
109  Ba_Mat[0][0] = GradNi_Ref_Mat[iNode][0];
110  Ba_Mat[1][1] = GradNi_Ref_Mat[iNode][1];
111  Ba_Mat[2][2] = GradNi_Ref_Mat[iNode][2];
112  Ba_Mat[3][0] = GradNi_Ref_Mat[iNode][1];
113  Ba_Mat[3][1] = GradNi_Ref_Mat[iNode][0];
114  Ba_Mat[4][0] = GradNi_Ref_Mat[iNode][2];
115  Ba_Mat[4][2] = GradNi_Ref_Mat[iNode][0];
116  Ba_Mat[5][1] = GradNi_Ref_Mat[iNode][2];
117  Ba_Mat[5][2] = GradNi_Ref_Mat[iNode][1];
118  }
119 
120  /*--- Compute the BT.D Matrix ---*/
121 
122  for (iVar = 0; iVar < nDim; iVar++) {
123  for (jVar = 0; jVar < bDim; jVar++) {
124  AuxMatrix[iVar][jVar] = 0.0;
125  for (kVar = 0; kVar < bDim; kVar++) {
126  AuxMatrix[iVar][jVar] += Ba_Mat[kVar][iVar]*D_Mat[kVar][jVar];
127  }
128  }
129  }
130 
131  /*--- Assumming symmetry ---*/
132  for (jNode = iNode; jNode < nNode; jNode++) {
133  if (nDim == 2) {
134  Bb_Mat[0][0] = GradNi_Ref_Mat[jNode][0];
135  Bb_Mat[1][1] = GradNi_Ref_Mat[jNode][1];
136  Bb_Mat[2][0] = GradNi_Ref_Mat[jNode][1];
137  Bb_Mat[2][1] = GradNi_Ref_Mat[jNode][0];
138  }
139  else {
140  Bb_Mat[0][0] = GradNi_Ref_Mat[jNode][0];
141  Bb_Mat[1][1] = GradNi_Ref_Mat[jNode][1];
142  Bb_Mat[2][2] = GradNi_Ref_Mat[jNode][2];
143  Bb_Mat[3][0] = GradNi_Ref_Mat[jNode][1];
144  Bb_Mat[3][1] = GradNi_Ref_Mat[jNode][0];
145  Bb_Mat[4][0] = GradNi_Ref_Mat[jNode][2];
146  Bb_Mat[4][2] = GradNi_Ref_Mat[jNode][0];
147  Bb_Mat[5][1] = GradNi_Ref_Mat[jNode][2];
148  Bb_Mat[5][2] = GradNi_Ref_Mat[jNode][1];
149  }
150 
151  for (iVar = 0; iVar < nDim; iVar++) {
152  for (jVar = 0; jVar < nDim; jVar++) {
153  KAux_ab[iVar][jVar] = 0.0;
154  for (kVar = 0; kVar < bDim; kVar++) {
155  KAux_ab[iVar][jVar] += Weight * AuxMatrix[iVar][kVar] * Bb_Mat[kVar][jVar] * Jac_X;
156  }
157  }
158  }
159 
160  element->Add_Kab(iNode, jNode, KAux_ab);
161  /*--- Symmetric terms --*/
162  if (iNode != jNode) {
163  element->Add_Kab_T(jNode, iNode, KAux_ab);
164  }
165 
166  }
167 
168  }
169 
170  }
171 
172  /*--- Compute residual ---*/
173  for(iNode = 0; iNode<nNode; ++iNode)
174  {
175  for(jNode = 0; jNode<nNode; ++jNode)
176  {
177  const su2double *Kab = element->Get_Kab(iNode,jNode);
178 
179  for (iVar = 0; iVar < nVar; iVar++) {
180  res_aux[iVar] = 0.0;
181  for (jVar = 0; jVar < nVar; jVar++)
182  res_aux[iVar] += Kab[iVar*nVar+jVar]*
183  (element->GetCurr_Coord(jNode,jVar)-element->GetRef_Coord(jNode,jVar));
184  }
185  element->Add_Kt_a(iNode, res_aux);
186  }
187  }
188 
189  /*--- Register the stress residual as preaccumulation output ---*/
190  element->SetPreaccOut_Kt_a();
191  AD::EndPreacc();
192 
193  delete[] res_aux;
194 }

197 void CFEALinearElasticity::Compute_Constitutive_Matrix(CElement *element_container, const CConfig *config) {
198 
199  /*--- Compute the D Matrix (for plane stress and 2-D)---*/
200 
201 
202  if (nDim == 2) {
203  if (plane_stress) {
204  /*--- We enable plane stress cases ---*/
205 
206  D_Mat[0][0] = E/(1-Nu*Nu); D_Mat[0][1] = (E*Nu)/(1-Nu*Nu); D_Mat[0][2] = 0.0;
207  D_Mat[1][0] = (E*Nu)/(1-Nu*Nu); D_Mat[1][1] = E/(1-Nu*Nu); D_Mat[1][2] = 0.0;
208  D_Mat[2][0] = 0.0; D_Mat[2][1] = 0.0; D_Mat[2][2] = ((1-Nu)*E)/(2*(1-Nu*Nu));
209  }
210  else {
211  /*--- Assuming plane strain as a general case ---*/
212 
213  D_Mat[0][0] = Lambda + 2.0*Mu; D_Mat[0][1] = Lambda; D_Mat[0][2] = 0.0;
214  D_Mat[1][0] = Lambda; D_Mat[1][1] = Lambda + 2.0*Mu; D_Mat[1][2] = 0.0;
215  D_Mat[2][0] = 0.0; D_Mat[2][1] = 0.0; D_Mat[2][2] = Mu;
216  }
217 
218  }
219  else {
220 
221  su2double Lbda_2Mu = Lambda + 2.0*Mu;
222 
223  D_Mat[0][0] = Lbda_2Mu; D_Mat[0][1] = Lambda; D_Mat[0][2] = Lambda; D_Mat[0][3] = 0.0; D_Mat[0][4] = 0.0; D_Mat[0][5] = 0.0;
224  D_Mat[1][0] = Lambda; D_Mat[1][1] = Lbda_2Mu; D_Mat[1][2] = Lambda; D_Mat[1][3] = 0.0; D_Mat[1][4] = 0.0; D_Mat[1][5] = 0.0;
225  D_Mat[2][0] = Lambda; D_Mat[2][1] = Lambda; D_Mat[2][2] = Lbda_2Mu; D_Mat[2][3] = 0.0; D_Mat[2][4] = 0.0; D_Mat[2][5] = 0.0;
226  D_Mat[3][0] = 0.0; D_Mat[3][1] = 0.0; D_Mat[3][2] = 0.0; D_Mat[3][3] = Mu; D_Mat[3][4] = 0.0; D_Mat[3][5] = 0.0;
227  D_Mat[4][0] = 0.0; D_Mat[4][1] = 0.0; D_Mat[4][2] = 0.0; D_Mat[4][3] = 0.0; D_Mat[4][4] = Mu; D_Mat[4][5] = 0.0;
228  D_Mat[5][0] = 0.0; D_Mat[5][1] = 0.0; D_Mat[5][2] = 0.0; D_Mat[5][3] = 0.0; D_Mat[5][4] = 0.0; D_Mat[5][5] = Mu;
229 
230  }
231 
232 }

235 su2double CFEALinearElasticity::Compute_Averaged_NodalStress(CElement *element, const CConfig *config) {
236 
237  unsigned short iVar, jVar;
238  unsigned short iGauss, nGauss;
239  unsigned short iNode, nNode;
240  unsigned short iDim, bDim;
241 
242  su2double avgStress[DIM_STRAIN_3D] = {0.0};
243 
244  /*--- Set element properties and recompute the constitutive matrix, this is needed
245  for multiple material cases and for correct differentiation ---*/
246  SetElement_Properties(element, config);
247 
248  /*--- Register pre-accumulation inputs ---*/
249  AD::StartPreacc();
250  AD::SetPreaccIn(E);
251  AD::SetPreaccIn(Nu);
252  element->SetPreaccIn_Coords();
253  /*--- Recompute Lame parameters as they depend on the material properties ---*/
254  Compute_Lame_Parameters();
255 
256  Compute_Constitutive_Matrix(element, config);
257 
258  /*--- Initialize auxiliary matrices ---*/
259 
260  bDim = (nDim == 2) ? DIM_STRAIN_2D : DIM_STRAIN_3D;
261 
262  for (iVar = 0; iVar < bDim; iVar++) {
263  for (jVar = 0; jVar < nDim; jVar++) {
264  Ba_Mat[iVar][jVar] = 0.0;
265  }
266  }
267 
268  element->ClearStress(); /*--- Clears the stress in the element to avoid adding over previous results. --*/
269  element->ComputeGrad_Linear();
270  nNode = element->GetnNodes();
271  nGauss = element->GetnGaussPoints();
272 
273  for (iGauss = 0; iGauss < nGauss; iGauss++) {
274 
275  /*--- Retrieve the values of the gradients of the shape functions for each node ---*/
276  /*--- This avoids repeated operations ---*/
277  for (iNode = 0; iNode < nNode; iNode++) {
278  for (iDim = 0; iDim < nDim; iDim++) {
279  GradNi_Ref_Mat[iNode][iDim] = element->GetGradNi_X(iNode,iGauss,iDim);
280  nodalDisplacement[iNode][iDim] = element->GetCurr_Coord(iNode, iDim) - element->GetRef_Coord(iNode, iDim);
281  }
282  }
283 
284  su2double Strain[DIM_STRAIN_3D] = {0.0};
285 
286  for (iNode = 0; iNode < nNode; iNode++) {
287 
288  /*--- Set matrix B ---*/
289  if (nDim == 2) {
290  Ba_Mat[0][0] = GradNi_Ref_Mat[iNode][0];
291  Ba_Mat[1][1] = GradNi_Ref_Mat[iNode][1];
292  Ba_Mat[2][0] = GradNi_Ref_Mat[iNode][1];
293  Ba_Mat[2][1] = GradNi_Ref_Mat[iNode][0];
294  }
295  else {
296  Ba_Mat[0][0] = GradNi_Ref_Mat[iNode][0];
297  Ba_Mat[1][1] = GradNi_Ref_Mat[iNode][1];
298  Ba_Mat[2][2] = GradNi_Ref_Mat[iNode][2];
299  Ba_Mat[3][0] = GradNi_Ref_Mat[iNode][1];
300  Ba_Mat[3][1] = GradNi_Ref_Mat[iNode][0];
301  Ba_Mat[4][0] = GradNi_Ref_Mat[iNode][2];
302  Ba_Mat[4][2] = GradNi_Ref_Mat[iNode][0];
303  Ba_Mat[5][1] = GradNi_Ref_Mat[iNode][2];
304  Ba_Mat[5][2] = GradNi_Ref_Mat[iNode][1];
305  }
306 
307  /*--- Compute the Strain Vector as B*u ---*/
308 
309  for (iVar = 0; iVar < bDim; iVar++) {
310  for (jVar = 0; jVar < nDim; jVar++) {
311  Strain[iVar] += Ba_Mat[iVar][jVar]*nodalDisplacement[iNode][jVar];
312  }
313  }
314 
315  }
316 
317  /*--- Compute the Stress Vector as D*epsilon ---*/
318 
319  su2double Stress[DIM_STRAIN_3D] = {0.0};
320 
321  for (iVar = 0; iVar < bDim; iVar++) {
322  for (jVar = 0; jVar < bDim; jVar++) {
323  Stress[iVar] += D_Mat[iVar][jVar]*Strain[jVar];
324  }
325  avgStress[iVar] += Stress[iVar] / nGauss;
326  }
327 
328  for (iNode = 0; iNode < nNode; iNode++) {
329 
330  su2double Ni_Extrap = element->GetNi_Extrap(iNode, iGauss);
331 
332  if (nDim == 2) {
333  for(iVar = 0; iVar < 3; ++iVar)
334  element->Add_NodalStress(iNode, iVar, Stress[iVar] * Ni_Extrap);
335  }
336  else {
337  /*--- If nDim is 3 and we compute it this way, the 3rd component is the Szz,
338  * while in the output it is the 4th component for practical reasons. ---*/
339  element->Add_NodalStress(iNode, 0, Stress[0] * Ni_Extrap);
340  element->Add_NodalStress(iNode, 1, Stress[1] * Ni_Extrap);
341  element->Add_NodalStress(iNode, 2, Stress[3] * Ni_Extrap);
342  element->Add_NodalStress(iNode, 3, Stress[2] * Ni_Extrap);
343  element->Add_NodalStress(iNode, 4, Stress[4] * Ni_Extrap);
344  element->Add_NodalStress(iNode, 5, Stress[5] * Ni_Extrap);
345  }
346  }
347 
348  }
349 
350  if (nDim == 3) std::swap(avgStress[2], avgStress[3]);
351  auto elStress = VonMisesStress(nDim, avgStress);
352 
353  /*--- We only differentiate w.r.t. an avg VM stress for the element as
354  * considering all nodal stresses would use too much memory. ---*/
355  AD::SetPreaccOut(elStress);
356  AD::EndPreacc();
357 
358  return elStress;
359 }
