32 void CSysVector<ScalarType>::Initialize(unsigned long numBlk, unsigned long numBlkDomain, unsigned long numVar,
33  const ScalarType* val, bool valIsArray, bool errorIfParallel) {
34  if (errorIfParallel && omp_in_parallel()) {
35  assert(false);
36  SU2_MPI::Error("If this class were constructed in parallel its operations would be incorrect.", CURRENT_FUNCTION);
37  }
38 
39  if (omp_get_thread_num())
40  SU2_MPI::Error("Only the master thread is allowed to initialize the vector.", CURRENT_FUNCTION);
41 
42  if (nElm != numBlk * numVar) {
43  MemoryAllocation::aligned_free(vec_val);
44  vec_val = nullptr;
45  }
46 
47  nElm = numBlk * numVar;
48  nElmDomain = numBlkDomain * numVar;
49  nVar = numVar;
50 
51  omp_chunk_size = computeStaticChunkSize(nElm, omp_get_max_threads(), OMP_MAX_SIZE);
52 
53  if (vec_val == nullptr) vec_val = MemoryAllocation::aligned_alloc<ScalarType, true>(64, nElm * sizeof(ScalarType));
54 
55  if (val != nullptr) {
56  if (!valIsArray) {
57  for (auto i = 0ul; i < nElm; i++) vec_val[i] = *val;
58  } else {
59  for (auto i = 0ul; i < nElm; i++) vec_val[i] = val[i];
60  }
61  }
62 }
