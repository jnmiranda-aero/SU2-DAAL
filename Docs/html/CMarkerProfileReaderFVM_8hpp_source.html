170 inline vector<su2double> GetColumnForProfile(int val_iProfile, unsigned short iCol) const {
171 auto nRow = numberOfRowsInProfile[val_iProfile];
172 auto nCol = numberOfColumnsInProfile[val_iProfile];
173 vector<su2double> ColumnData(nRow);
174 for (unsigned long iRow = 0; iRow < nRow; iRow++)
175 ColumnData[iRow]=profileData[val_iProfile][iRow*nCol+iCol];
176 return ColumnData;
177 }
