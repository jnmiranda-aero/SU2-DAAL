32CBoxMeshReaderFVM::CBoxMeshReaderFVM(CConfig* val_config, unsigned short val_iZone, unsigned short val_nZone)
33 : CMeshReaderFVM(val_config, val_iZone, val_nZone) {
34 /* The box mesh is always 3D. */
35 dimension = 3;
36
37 /* Set the VTK type for the interior elements and the boundary elements. */
38 KindElem = HEXAHEDRON;
39 KindBound = QUADRILATERAL;
40
41 /* The number of nodes in the i and j directions. */
42 nNode = config->GetMeshBoxSize(0);
43 mNode = config->GetMeshBoxSize(1);
44 pNode = config->GetMeshBoxSize(2);
45
46 /* Lengths for non-square domains. */
47 Lx = config->GetMeshBoxLength(0);
48 Ly = config->GetMeshBoxLength(1);
49 Lz = config->GetMeshBoxLength(2);
50
51 /* Offsets in x and y directions from 0.0. */
52 Ox = config->GetMeshBoxOffset(0);
53 Oy = config->GetMeshBoxOffset(1);
54 Oz = config->GetMeshBoxOffset(2);
55
56 /* Compute and store the points, interior elements, and surface elements.
57 In these routines, we use a simple analytic formula to compute the
58 coordinates and the node numbering. We store only the points and interior
59 elements on our rank's linear partition, but the master stores the entire
60 set of surface connectivity. */
61 ComputeBoxPointCoordinates();
62 ComputeBoxVolumeConnectivity();
63 ComputeBoxSurfaceConnectivity();
64}

68void CBoxMeshReaderFVM::ComputeBoxPointCoordinates() {
69 /* Set the global count of points based on the grid dimensions. */
70 numberOfGlobalPoints = (nNode) * (mNode) * (pNode);
71
72 /* Get a partitioner to help with linear partitioning. */
73 CLinearPartitioner pointPartitioner(numberOfGlobalPoints, 0);
74
75 /* Determine number of local points */
76 for (unsigned long globalIndex = 0; globalIndex < numberOfGlobalPoints; globalIndex++) {
77 if ((int)pointPartitioner.GetRankContainingIndex(globalIndex) == rank) {
78 numberOfLocalPoints++;
79 }
80 }
81
82 /* Loop over our analytically defined of coordinates and store only
83 those that contain a node within our linear partition of points. */
84 localPointCoordinates.resize(dimension);
85 for (int k = 0; k < dimension; k++) localPointCoordinates[k].reserve(numberOfLocalPoints);
86
87 unsigned long globalIndex = 0;
88 for (unsigned long kNode = 0; kNode < pNode; kNode++) {
89 for (unsigned long jNode = 0; jNode < mNode; jNode++) {
90 for (unsigned long iNode = 0; iNode < nNode; iNode++) {
91 if ((int)pointPartitioner.GetRankContainingIndex(globalIndex) == rank) {
92 /* Store the coordinates more clearly. */
93 const passivedouble x = SU2_TYPE::GetValue(Lx * ((su2double)iNode) / ((su2double)(nNode - 1)) + Ox);
94 const passivedouble y = SU2_TYPE::GetValue(Ly * ((su2double)jNode) / ((su2double)(mNode - 1)) + Oy);
95 const passivedouble z = SU2_TYPE::GetValue(Lz * ((su2double)kNode) / ((su2double)(pNode - 1)) + Oz);
96
97 /* Load into the coordinate class data structure. */
98 localPointCoordinates[0].push_back(x);
99 localPointCoordinates[1].push_back(y);
100 localPointCoordinates[2].push_back(z);
101 }
102 globalIndex++;
103 }
104 }
105 }
106}

108void CBoxMeshReaderFVM::ComputeBoxVolumeConnectivity() {
109 /* Set the global count of elements based on the grid dimensions. */
110 numberOfGlobalElements = (nNode - 1) * (mNode - 1) * (pNode - 1);
111
112 /* Get a partitioner to help with linear partitioning. */
113 CLinearPartitioner pointPartitioner(numberOfGlobalPoints, 0);
114
115 /* Loop over our analytically defined of elements and store only those
116 that contain a node within our linear partition of points. */
117 numberOfLocalElements = 0;
118 vector<unsigned long> connectivity(N_POINTS_HEXAHEDRON, 0);
119 unsigned long globalIndex = 0;
120 for (unsigned long kNode = 0; kNode < pNode - 1; kNode++) {
121 for (unsigned long jNode = 0; jNode < mNode - 1; jNode++) {
122 for (unsigned long iNode = 0; iNode < nNode - 1; iNode++) {
123 /* Compute connectivity based on the i,j,k index. */
124 connectivity[0] = kNode * mNode * nNode + jNode * nNode + iNode;
125 connectivity[1] = kNode * mNode * nNode + jNode * nNode + iNode + 1;
126 connectivity[2] = kNode * mNode * nNode + (jNode + 1) * nNode + (iNode + 1);
127 connectivity[3] = kNode * mNode * nNode + (jNode + 1) * nNode + iNode;
128 connectivity[4] = (kNode + 1) * mNode * nNode + jNode * nNode + iNode;
129 connectivity[5] = (kNode + 1) * mNode * nNode + jNode * nNode + iNode + 1;
130 connectivity[6] = (kNode + 1) * mNode * nNode + (jNode + 1) * nNode + (iNode + 1);
131 connectivity[7] = (kNode + 1) * mNode * nNode + (jNode + 1) * nNode + iNode;
132
133 /* Check whether any of the points is in our linear partition. */
134 bool isOwned = false;
135 for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++) {
136 if ((int)pointPartitioner.GetRankContainingIndex(connectivity[i]) == rank) {
137 isOwned = true;
138 }
139 }
140
141 /* If so, we need to store the element locally. */
142 if (isOwned) {
143 localVolumeElementConnectivity.push_back(globalIndex);
144 localVolumeElementConnectivity.push_back(KindElem);
145 for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++) {
146 localVolumeElementConnectivity.push_back(connectivity[i]);
147 }
148 numberOfLocalElements++;
149 }
150 globalIndex++;
151 }
152 }
153 }

156void CBoxMeshReaderFVM::ComputeBoxSurfaceConnectivity() {
157 /* The rectangle alays has 4 markers. */
158 numberOfMarkers = 6;
159 surfaceElementConnectivity.resize(numberOfMarkers);
160 markerNames.resize(numberOfMarkers);
161
162 vector<unsigned long> connectivity(N_POINTS_HEXAHEDRON, 0);
163
164 /* Compute and store the 6 sets of connectivity. */
165
166 markerNames[0] = "x_minus";
167 if (rank == MASTER_NODE) {
168 for (unsigned long kNode = 0; kNode < pNode - 1; kNode++) {
169 for (unsigned long jNode = 0; jNode < mNode - 1; jNode++) {
170 connectivity[0] = kNode * mNode * nNode + jNode * nNode;
171 connectivity[1] = (kNode + 1) * mNode * nNode + jNode * nNode;
172 connectivity[2] = (kNode + 1) * mNode * nNode + (jNode + 1) * nNode;
173 connectivity[3] = kNode * mNode * nNode + (jNode + 1) * nNode;
174
175 surfaceElementConnectivity[0].push_back(0);
176 surfaceElementConnectivity[0].push_back(KindBound);
177 for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++)
178 surfaceElementConnectivity[0].push_back(connectivity[i]);
179 }
180 }
181 }
182
183 markerNames[1] = "x_plus";
184 if (rank == MASTER_NODE) {
185 for (unsigned long kNode = 0; kNode < pNode - 1; kNode++) {
186 for (unsigned long jNode = 0; jNode < mNode - 1; jNode++) {
187 connectivity[0] = kNode * mNode * nNode + jNode * nNode + (nNode - 1);
188 connectivity[1] = kNode * mNode * nNode + (jNode + 1) * nNode + (nNode - 1);
189 connectivity[2] = (kNode + 1) * mNode * nNode + (jNode + 1) * nNode + (nNode - 1);
190 connectivity[3] = (kNode + 1) * mNode * nNode + jNode * nNode + (nNode - 1);
191
192 surfaceElementConnectivity[1].push_back(0);
193 surfaceElementConnectivity[1].push_back(KindBound);
194 for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++)
195 surfaceElementConnectivity[1].push_back(connectivity[i]);
196 }
197 }
198 }
199
200 markerNames[2] = "y_minus";
201 if (rank == MASTER_NODE) {
202 for (unsigned long kNode = 0; kNode < pNode - 1; kNode++) {
203 for (unsigned long iNode = 0; iNode < nNode - 1; iNode++) {
204 connectivity[0] = kNode * mNode * nNode + iNode;
205 connectivity[1] = kNode * mNode * nNode + iNode + 1;
206 connectivity[2] = (kNode + 1) * mNode * nNode + iNode + 1;
207 connectivity[3] = (kNode + 1) * mNode * nNode + iNode;
208
209 surfaceElementConnectivity[2].push_back(0);
210 surfaceElementConnectivity[2].push_back(KindBound);
211 for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++)
212 surfaceElementConnectivity[2].push_back(connectivity[i]);
213 }
214 }
215 }
216
217 markerNames[3] = "y_plus";

220 for (unsigned long iNode = 0; iNode < nNode - 1; iNode++) {

239 connectivity[1] = jNode * nNode + iNode + 1;

254 for (unsigned long iNode = 0; iNode < nNode - 1; iNode++) {
