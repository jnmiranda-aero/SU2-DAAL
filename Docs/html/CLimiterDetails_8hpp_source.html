65 /*!
66  * \brief Common small functions used by limiters.
67  * \ingroup FvmAlgos
68  */
69 template<class Type = su2double>
70 struct LimiterHelpers
71 {
72  FORCEINLINE static Type epsilon() {return std::numeric_limits<passivedouble>::epsilon();}
73 
74  FORCEINLINE static Type venkatFunction(const Type& proj, const Type& delta, const Type& eps2)
75  {
76  Type y = delta*(delta+proj) + eps2;
77  return (y + delta*proj) / (y + 2*proj*proj);
78  }
79 
80  FORCEINLINE static Type vanAlbadaFunction(const Type& proj, const Type& delta, const Type& eps)
81  {
82  return delta * (2*proj + delta) / (4*pow(proj, 2) + pow(delta, 2) + eps);
83  }
84 
85  FORCEINLINE static Type raisedSine(const Type& dist)
86  {
87  Type factor = 0.5*(1.0+dist+sin(PI_NUMBER*dist)/PI_NUMBER);
88  return max(0.0, min(factor, 1.0));
89  }
90 
91  FORCEINLINE static Type r3Function(const Type& proj, const Type& delta, const Type& epsp)
92  {
93  Type Dp = fabs(delta);
94  Type Dm = fabs(proj);
95  if(Dp>(2.0*Dm)) return 1.0;
96  Type y = pow(Dp, 3) + epsp;
97  Type S3 = 4.0*Dm*Dm;
98  return (y + Dp*S3) / (y + Dm*(delta*delta+S3));
99  }
100 
101  FORCEINLINE static Type r4Function(const Type& proj, const Type& delta, const Type& epsp)
102  {
103  Type Dp = fabs(delta);
104  Type Dm = fabs(proj);
105  if(Dp>(2.0*Dm)) return 1.0;
106  Type y = pow(Dp, 4) + epsp;
107  Type S4 = 2.0*Dm*(Dp*Dp-2.0*Dm*(Dp-2.0*Dm));
108  return (y + Dp*S4) / (y + Dm*(pow(delta,3)+S4));
109  }
110 
111  FORCEINLINE static Type r5Function(const Type& proj, const Type& delta, const Type& epsp)
112  {
113  Type Dp = fabs(delta);
114  Type Dm = fabs(proj);
115  if(Dp>(2.0*Dm)) return 1.0;
116  Type y = pow(Dp, 5) + epsp;
117  Type S5 = 8.0*Dm*Dm*(Dp*Dp-2.0*Dm*(Dp-Dm));
118  return (y + Dp*S5) / (y + Dm*(pow(delta,4)+S5));
119  }
120 };

305 /*!
306  * \brief Venkatakrishnan-Wang specialization.
307  * \ingroup FvmAlgos
308  */
309 template<>
310 struct CLimiterDetails<LIMITER::VENKATAKRISHNAN_WANG>
311 {
312  static su2activevector sharedMin, sharedMax;
313  su2activevector eps2;
314 
315  /*!
316  * \brief Store the solution range based eps^2 parameter.
317  */
318  template<class FieldType>
319  inline void preprocess(CGeometry& geometry, const CConfig& config, size_t varBegin,
320  size_t varEnd, const FieldType& field)
321  {
322  /*--- Determine the max and min global value for each variable. ---*/
323 
324  su2double largeNum = 0.1*std::numeric_limits<passivedouble>::max();
325 
326  /*--- Allocate the static members (shared between threads) to
327  * perform the reduction across all threads in the rank. ---*/
328 
329  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
330  {
331  sharedMin.resize(varEnd) = largeNum;
332  sharedMax.resize(varEnd) =-largeNum;
333  }
334  END_SU2_OMP_SAFE_GLOBAL_ACCESS
335 
336  /*--- Per thread reduction. ---*/
337 
338  su2activevector localMin(varEnd), localMax(varEnd);
339  localMin = largeNum;
340  localMax =-largeNum;
341 
342  SU2_OMP_FOR_(schedule(static, 512) SU2_NOWAIT)
343  for(size_t iPoint = 0; iPoint < geometry.GetnPointDomain(); ++iPoint)
344  {
345  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
346  {
347  localMin(iVar) = min(localMin(iVar), field(iPoint, iVar));
348  localMax(iVar) = max(localMax(iVar), field(iPoint, iVar));
349  }
350  }
351  END_SU2_OMP_FOR
352 
353  /*--- Per rank reduction. ---*/
354 
355  SU2_OMP_CRITICAL
356  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
357  {
358  sharedMin(iVar) = min(sharedMin(iVar), localMin(iVar));
359  sharedMax(iVar) = max(sharedMax(iVar), localMax(iVar));
360  }
361  END_SU2_OMP_CRITICAL
362 
363  /*--- Global reduction. ---*/
364 
365  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
366  {
367  localMin = sharedMin;
368  SU2_MPI::Allreduce(localMin.data(), sharedMin.data(), varEnd, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
369 
370  localMax = sharedMax;
371  SU2_MPI::Allreduce(localMax.data(), sharedMax.data(), varEnd, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
372  }
373  END_SU2_OMP_SAFE_GLOBAL_ACCESS
374 
375  /*--- Compute eps^2 (each thread has its own copy of it). ---*/
376 
377  eps2.resize(varEnd);
378  su2double K = config.GetVenkat_LimiterCoeff();
379 
380  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
381  {
382  su2double range = sharedMax(iVar) - sharedMin(iVar);
383  eps2(iVar) = max(pow(K*range, 2), LimiterHelpers<>::epsilon());
384  }
385  }
386 
387  /*!
388  * \brief No geometric modification for this kind of limiter.
389  */
390  template<class... Ts>
391  inline su2double geometricFactor(Ts&...) const {return 1.0;}
392 
393  /*!
394  * \brief Smooth function that disables limiting in smooth regions.
395  */
396  inline su2double limiterFunction(size_t iVar, su2double proj, su2double delta) const
397  {
398  AD::SetPreaccIn(eps2(iVar));
399  return LimiterHelpers<>::venkatFunction(proj, delta, eps2(iVar));
400  }
401 };
