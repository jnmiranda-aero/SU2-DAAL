76 CNumerics::ResidualType<> CCentLax_NEMO::ComputeResidual(const CConfig *config) {
77 
78  /*--- Calculate geometrical quantities ---*/
79  Area = GeometryToolbox::Norm(nDim, Normal);
80 
81  for (auto iDim = 0ul; iDim < nDim; iDim++)
82  UnitNormal[iDim] = Normal[iDim]/Area;
83 
84  /*--- Rename for convenience ---*/
85  Density_i = V_i[RHO_INDEX]; Density_j = V_j[RHO_INDEX];
86  Enthalpy_i = V_i[H_INDEX]; Enthalpy_j = V_j[H_INDEX];
87  SoundSpeed_i = V_i[A_INDEX]; SoundSpeed_j = V_j[A_INDEX];
88 
89  /*--- Compute mean quantities for the variables ---*/
90  for (auto iVar = 0ul; iVar < nVar; iVar++)
91  MeanU[iVar] = 0.5*(U_i[iVar]+U_j[iVar]);
92  for (auto iVar = 0ul; iVar < nPrimVar; iVar++)
93  MeanV[iVar] = 0.5*(V_i[iVar]+V_j[iVar]);
94 
95  /*--- Compute NonEq specific variables ---*/
96  const auto& mean_eves = fluidmodel->ComputeSpeciesEve(MeanV[TVE_INDEX]);
97  fluidmodel->ComputedPdU(MeanV, mean_eves, MeandPdU);
98 
99  /*--- Get projected flux tensor ---*/
100  GetInviscidProjFlux(MeanU, MeanV, Normal, ProjFlux);
101 
102  /*--- Jacobians of the inviscid flux, scale = 0.5 because val_res ~ 0.5*(fc_i+fc_j)*Normal ---*/
103  if (implicit) {
104  GetInviscidProjJac(MeanU, MeanV, MeandPdU, Normal, 0.5, Jacobian_i);
105 
106  for (auto iVar = 0ul; iVar < nVar; iVar++)
107  for (auto jVar = 0ul; jVar < nVar; jVar++)
108  Jacobian_j[iVar][jVar] = Jacobian_i[iVar][jVar];
109  }
110 
111  /*--- Compute the local spectral radius and the stretching factor ---*/
112  ProjVelocity_i = GeometryToolbox::DotProduct(nDim, &V_i[VEL_INDEX],Normal);
113  ProjVelocity_j = GeometryToolbox::DotProduct(nDim, &V_j[VEL_INDEX],Normal);
114 
115  /*--- Dissipation --*/
116  Local_Lambda_i = (fabs(ProjVelocity_i)+SoundSpeed_i*Area);
117  Local_Lambda_j = (fabs(ProjVelocity_j)+SoundSpeed_j*Area);
118  MeanLambda = 0.5*(Local_Lambda_i+Local_Lambda_j);
119 
120  Phi_i = pow(Lambda_i/(4.0*MeanLambda+EPS),Param_p);
121  Phi_j = pow(Lambda_j/(4.0*MeanLambda+EPS),Param_p);
122  StretchingFactor = 4.0*Phi_i*Phi_j/(Phi_i+Phi_j+EPS);
123 
124  /*--- Computes differences btw. conservative variables ---*/
125  for (auto iVar = 0ul; iVar < nVar; iVar++)
126  Diff_U[iVar] = U_i[iVar] - U_j[iVar];
127  Diff_U[nSpecies+nDim] = Density_i*Enthalpy_i - Density_j*Enthalpy_j;
128 
129  /*--- Compute dissipation coefficient ---*/
130  sc0 = 3.0*(su2double(Neighbor_i)+su2double(Neighbor_j))/(su2double(Neighbor_i)*su2double(Neighbor_j));
131  Epsilon_0 = Param_Kappa_0*sc0*su2double(nDim)/3.0;
132 
133  /*--- Compute viscous part of the residual ---*/
134  for (auto iVar = 0ul; iVar < nVar; iVar++) {
135  Flux[iVar] = ProjFlux[iVar]+Epsilon_0*Diff_U[iVar]*StretchingFactor*MeanLambda;
136  }
137 
138  if (implicit) {
139 
140  cte = Epsilon_0*StretchingFactor*MeanLambda;
141 
142  for (auto iVar = 0u; iVar < nSpecies+nDim; iVar++) {
143  Jacobian_i[iVar][iVar] += cte;
144  Jacobian_j[iVar][iVar] -= cte;
145  }
146 
147  /*--- Last rows: CAREFUL!! You have differences of \rho_Enthalpy, not differences of \rho_Energy ---*/
148  for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
149  Jacobian_i[nSpecies+nDim][iSpecies] += cte*dPdU_i[iSpecies];
150  for (auto iDim = 0ul; iDim < nDim; iDim++)
151  Jacobian_i[nSpecies+nDim][nSpecies+iDim] += cte*dPdU_i[nSpecies+iDim];
152  Jacobian_i[nSpecies+nDim][nSpecies+nDim] += cte*(1+dPdU_i[nSpecies+nDim]);
153  Jacobian_i[nSpecies+nDim][nSpecies+nDim+1] += cte*dPdU_i[nSpecies+nDim+1];
154  Jacobian_i[nSpecies+nDim+1][nSpecies+nDim+1] += cte;
155 
156  /*--- Last row of Jacobian_j ---*/
157  for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
158  Jacobian_j[nSpecies+nDim][iSpecies] -= cte*dPdU_j[iSpecies];
159  for (auto iDim = 0ul; iDim < nDim; iDim++)
160  Jacobian_j[nSpecies+nDim][nSpecies+iDim] -= cte*dPdU_j[nSpecies+nDim];
161  Jacobian_j[nSpecies+nDim][nSpecies+nDim] -= cte*(1+dPdU_j[nSpecies+nDim]);
162  Jacobian_j[nSpecies+nDim][nSpecies+nDim+1] -= cte*dPdU_j[nSpecies+nDim+1];
163  Jacobian_j[nSpecies+nDim+1][nSpecies+nDim+1] -= cte;
164  }
165  return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
166 
167 }
