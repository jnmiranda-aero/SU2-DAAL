33CMultiGridGeometry::CMultiGridGeometry(CGeometry* fine_grid, CConfig* config, unsigned short iMesh) : CGeometry() {
34 nDim = fine_grid->GetnDim(); // Write the number of dimensions of the coarse grid.
35
36 /*--- Create a queue system to do the agglomeration
37 1st) More than two markers ---> Vertices (never agglomerate)
38 2nd) Two markers ---> Edges (agglomerate if same BC, never agglomerate if different BC)
39 3rd) One marker ---> Surface (always agglomarate)
40 4th) No marker ---> Internal Volume (always agglomarate) ---*/
41
42 /*--- Set a marker to indicate indirect agglomeration, for quads and hexs,
43 i.e. consider up to neighbors of neighbors of neighbors.
44 For other levels this information is propagated down during their construction. ---*/
45
46 if (iMesh == MESH_1) {
47 for (auto iPoint = 0ul; iPoint < fine_grid->GetnPoint(); iPoint++)
48 fine_grid->nodes->SetAgglomerate_Indirect(iPoint, false);
49
50 for (auto iElem = 0ul; iElem < fine_grid->GetnElem(); iElem++) {
51 if ((fine_grid->elem[iElem]->GetVTK_Type() == HEXAHEDRON) ||
52 (fine_grid->elem[iElem]->GetVTK_Type() == QUADRILATERAL)) {
53 for (auto iNode = 0u; iNode < fine_grid->elem[iElem]->GetnNodes(); iNode++) {
54 const auto iPoint = fine_grid->elem[iElem]->GetNode(iNode);
55 fine_grid->nodes->SetAgglomerate_Indirect(iPoint, true);
56 }
57 }
58 }
59 }
60
61 /*--- Create the coarse grid structure using as baseline the fine grid ---*/
62
63 CMultiGridQueue MGQueue_InnerCV(fine_grid->GetnPoint());
64 vector<unsigned long> Suitable_Indirect_Neighbors;
65
66 nodes = new CPoint(fine_grid->GetnPoint(), nDim, iMesh, config);
67
68 unsigned long Index_CoarseCV = 0;
69
70 /*--- The first step is the boundary agglomeration. ---*/
71
72 for (auto iMarker = 0u; iMarker < fine_grid->GetnMarker(); iMarker++) {
73 for (auto iVertex = 0ul; iVertex < fine_grid->GetnVertex(iMarker); iVertex++) {
74 const auto iPoint = fine_grid->vertex[iMarker][iVertex]->GetNode();
75
76 /*--- If the element has not been previously agglomerated and it
77 belongs to this physical domain, and it meets the geometrical
78 criteria, the agglomeration is studied. ---*/
79
80 if ((!fine_grid->nodes->GetAgglomerate(iPoint)) && (fine_grid->nodes->GetDomain(iPoint)) &&
81 (GeometricalCheck(iPoint, fine_grid, config))) {
82 unsigned short nChildren = 1;
83
84 /*--- We set an index for the parent control volume, this
85 also marks it as agglomerated. ---*/
86
87 fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
88
89 /*--- We add the seed point (child) to the parent control volume ---*/
90
91 nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
92 bool agglomerate_seed = true;
93 auto counter = 0;
94 unsigned short copy_marker[3] = {};
95 const auto marker_seed = iMarker;
96
97 /*--- For a particular point in the fine grid we save all the markers
98 that are in that point ---*/
99
100 for (auto jMarker = 0u; jMarker < fine_grid->GetnMarker() && counter < 3; jMarker++) {
101 if (fine_grid->nodes->GetVertex(iPoint, jMarker) != -1) {
102 copy_marker[counter] = jMarker;
103 counter++;
104 }
105 }
106
107 /*--- To aglomerate a vertex it must have only one physical bc!!
108 This can be improved. If there is only a marker, it is a good
109 candidate for agglomeration ---*/
110
111 if (counter == 1) agglomerate_seed = true;
112
113 /*--- If there are two markers, we will aglomerate if any of the
114 markers is SEND_RECEIVE ---*/
115
116 if (counter == 2) {
117 agglomerate_seed = (config->GetMarker_All_KindBC(copy_marker[0]) == SEND_RECEIVE) ||
118 (config->GetMarker_All_KindBC(copy_marker[1]) == SEND_RECEIVE);
119 }
120
121 /*--- If there are more than 2 markers, the aglomeration will be discarted ---*/
122
123 if (counter > 2) agglomerate_seed = false;
124
125 /*--- If the seed can be agglomerated, we try to agglomerate more points ---*/
126
127 if (agglomerate_seed) {
128 /*--- Now we do a sweep over all the nodes that surround the seed point ---*/
129
130 for (auto CVPoint : fine_grid->nodes->GetPoints(iPoint)) {
131 /*--- The new point can be agglomerated ---*/
132
133 if (SetBoundAgglomeration(CVPoint, marker_seed, fine_grid, config)) {
134 /*--- We set the value of the parent ---*/
135
136 fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
137
138 /*--- We set the value of the child ---*/
139
140 nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
141 nChildren++;
142 }
143 }
144
145 Suitable_Indirect_Neighbors.clear();
146
147 if (fine_grid->nodes->GetAgglomerate_Indirect(iPoint))
148 SetSuitableNeighbors(Suitable_Indirect_Neighbors, iPoint, Index_CoarseCV, fine_grid);
149
150 /*--- Now we do a sweep over all the indirect nodes that can be added ---*/
151
152 for (auto CVPoint : Suitable_Indirect_Neighbors) {
153 /*--- The new point can be agglomerated ---*/
154
155 if (SetBoundAgglomeration(CVPoint, marker_seed, fine_grid, config)) {
156 /*--- We set the value of the parent ---*/
157
158 fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
159
160 /*--- We set the indirect agglomeration information of the corse point
161 based on its children in the fine grid. ---*/
162
163 if (fine_grid->nodes->GetAgglomerate_Indirect(CVPoint))
164 nodes->SetAgglomerate_Indirect(Index_CoarseCV, true);
165
166 /*--- We set the value of the child ---*/
167
168 nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
169 nChildren++;
170 }
171 }
172 }
173
174 /*--- Update the number of children of the coarse control volume. ---*/
175
176 nodes->SetnChildren_CV(Index_CoarseCV, nChildren);
177 Index_CoarseCV++;
178 }
179 }
180 }
181
182 /*--- Do not agglomerate any leftover node with more than one physical boundary condition,
183 i.e. make one coarse CV with a single child. ---*/
184
185 for (auto iMarker = 0u; iMarker < fine_grid->GetnMarker(); iMarker++) {
186 for (auto iVertex = 0ul; iVertex < fine_grid->GetnVertex(iMarker); iVertex++) {
187 const auto iPoint = fine_grid->vertex[iMarker][iVertex]->GetNode();
188
189 if ((!fine_grid->nodes->GetAgglomerate(iPoint)) && (fine_grid->nodes->GetDomain(iPoint))) {
190 fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
191 nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
192 nodes->SetnChildren_CV(Index_CoarseCV, 1);
193 Index_CoarseCV++;
194 }
195 }
196 }
197
198 /*--- Update the queue with the results from the boundary agglomeration ---*/
199
200 for (auto iPoint = 0ul; iPoint < fine_grid->GetnPoint(); iPoint++) {
201 if (fine_grid->nodes->GetAgglomerate(iPoint)) {
202 MGQueue_InnerCV.RemoveCV(iPoint);
203
204 } else {
205 /*--- Count the number of agglomerated neighbors, and modify the queue,
206 Points with more agglomerated neighbors are processed first. ---*/
207
208 short priority = 0;
209 for (auto jPoint : fine_grid->nodes->GetPoints(iPoint)) {
210 priority += fine_grid->nodes->GetAgglomerate(jPoint);
211 }
212 MGQueue_InnerCV.MoveCV(iPoint, priority);
213 }
214 }
215
216 /*--- Agglomerate the domain points. ---*/
217
218 auto iteration = 0ul;
219 while (!MGQueue_InnerCV.EmptyQueue() && (iteration < fine_grid->GetnPoint())) {
220 const auto iPoint = MGQueue_InnerCV.NextCV();
221 iteration++;
222
223 /*--- If the element has not being previously agglomerated, belongs to the physical domain,
224 and satisfies several geometrical criteria then the seed CV is accepted for agglomeration. ---*/
225
226 if ((!fine_grid->nodes->GetAgglomerate(iPoint)) && (fine_grid->nodes->GetDomain(iPoint)) &&
227 (GeometricalCheck(iPoint, fine_grid, config))) {
228 unsigned short nChildren = 1;
229
230 /*--- We set an index for the parent control volume ---*/
231
232 fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
233
234 /*--- We add the seed point (child) to the parent control volume ---*/
235
236 nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
237
238 /*--- Update the queue with the seed point (remove the seed and
239 increase the priority of its neighbors) ---*/
240
241 MGQueue_InnerCV.Update(iPoint, fine_grid);
242
243 /*--- Now we do a sweep over all the nodes that surround the seed point ---*/
244
245 for (auto CVPoint : fine_grid->nodes->GetPoints(iPoint)) {
246 /*--- Determine if the CVPoint can be agglomerated ---*/
247
248 if ((!fine_grid->nodes->GetAgglomerate(CVPoint)) && (fine_grid->nodes->GetDomain(CVPoint)) &&
249 (GeometricalCheck(CVPoint, fine_grid, config))) {
250 /*--- We set the value of the parent ---*/
251
252 fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
253
254 /*--- We set the value of the child ---*/
255
256 nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
257 nChildren++;
258
259 /*--- Update the queue with the new control volume (remove the CV and
260 increase the priority of its neighbors) ---*/
261
262 MGQueue_InnerCV.Update(CVPoint, fine_grid);
263 }
264 }
265
266 /*--- Identify the indirect neighbors ---*/
267
268 Suitable_Indirect_Neighbors.clear();
269 if (fine_grid->nodes->GetAgglomerate_Indirect(iPoint))
270 SetSuitableNeighbors(Suitable_Indirect_Neighbors, iPoint, Index_CoarseCV, fine_grid);
271
272 /*--- Now we do a sweep over all the indirect nodes that can be added ---*/
273
274 for (auto CVPoint : Suitable_Indirect_Neighbors) {
275 /*--- The new point can be agglomerated ---*/
276
277 if ((!fine_grid->nodes->GetAgglomerate(CVPoint)) && (fine_grid->nodes->GetDomain(CVPoint))) {
278 /*--- We set the value of the parent ---*/
279
280 fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
281
282 /*--- We set the indirect agglomeration information ---*/
283
284 if (fine_grid->nodes->GetAgglomerate_Indirect(CVPoint)) nodes->SetAgglomerate_Indirect(Index_CoarseCV, true);
285
286 /*--- We set the value of the child ---*/
287
288 nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
289 nChildren++;
290
291 /*--- Update the queue with the new control volume (remove the CV and
292 increase the priority of the neighbors) ---*/
293
294 MGQueue_InnerCV.Update(CVPoint, fine_grid);
295 }
296 }
297
298 /*--- Update the number of control of childrens ---*/
299
300 nodes->SetnChildren_CV(Index_CoarseCV, nChildren);
301 Index_CoarseCV++;
302 } else {
303 /*--- The seed point can not be agglomerated because of size, domain, streching, etc.
304 move the point to the lowest priority ---*/
305
306 MGQueue_InnerCV.MoveCV(iPoint, -1);
307 }
308 }
309
310 /*--- Convert any point that was not agglomerated into a coarse point. ---*/
311
312 for (auto iPoint = 0ul; iPoint < fine_grid->GetnPoint(); iPoint++) {
313 if ((!fine_grid->nodes->GetAgglomerate(iPoint)) && (fine_grid->nodes->GetDomain(iPoint))) {
314 fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
315 if (fine_grid->nodes->GetAgglomerate_Indirect(iPoint)) nodes->SetAgglomerate_Indirect(Index_CoarseCV, true);
316 nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
317 nodes->SetnChildren_CV(Index_CoarseCV, 1);
318 Index_CoarseCV++;
319 }
320 }
321
322 nPointDomain = Index_CoarseCV;
323 nPoint = nPointDomain;
324
325 /*--- Check that there are no hanging nodes. Detect isolated points
326 (only 1 neighbor), and merge their children CV's with the neighbor. ---*/
327
328 SetPoint_Connectivity(fine_grid);
329
330 for (auto iCoarsePoint = 0ul; iCoarsePoint < nPointDomain; iCoarsePoint++) {
331 if (nodes->GetnPoint(iCoarsePoint) == 1) {
332 /*--- Find the neighbor of the isolated point. This neighbor is the right control volume ---*/
333
334 const auto iCoarsePoint_Complete = nodes->GetPoint(iCoarsePoint, 0);
335
336 /*--- Add the children to the connected control volume (and modify its parent indexing).
337 Identify the child CV from the finest grid and add it to the correct control volume.
338 Set the parent CV of iFinePoint. Instead of using the original one
339 (iCoarsePoint), use the new one (iCoarsePoint_Complete) ---*/
340
341 auto nChildren = nodes->GetnChildren_CV(iCoarsePoint_Complete);
342
343 for (auto iChildren = 0u; iChildren < nodes->GetnChildren_CV(iCoarsePoint); iChildren++) {
344 const auto iFinePoint = nodes->GetChildren_CV(iCoarsePoint, iChildren);
345 nodes->SetChildren_CV(iCoarsePoint_Complete, nChildren, iFinePoint);
346 nChildren++;
347 fine_grid->nodes->SetParent_CV(iFinePoint, iCoarsePoint_Complete);
348 }
349
350 /*--- Update the number of children control volumes ---*/
351
352 nodes->SetnChildren_CV(iCoarsePoint_Complete, nChildren);
353 nodes->SetnChildren_CV(iCoarsePoint, 0);
354 }
355 }
356
357 /*--- Reset the neighbor information. ---*/
358
359 nodes->ResetPoints();
360
361#ifdef HAVE_MPI
362 /*--- Dealing with MPI parallelization, the objective is that the received nodes must be agglomerated
363 in the same way as the donor (send) nodes. Send the node agglomeration information of the donor
364 (parent and children). The agglomerated halos of this rank are set according to the rank where
365 they are domain points. ---*/
366
367 for (auto iMarker = 0u; iMarker < config->GetnMarker_All(); iMarker++) {
368 if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
369 const auto MarkerS = iMarker;
370 const auto MarkerR = iMarker + 1;
371
372 const auto send_to = config->GetMarker_All_SendRecv(MarkerS) - 1;
373 const auto receive_from = abs(config->GetMarker_All_SendRecv(MarkerR)) - 1;
374
375 const auto nVertexS = fine_grid->nVertex[MarkerS];
376 const auto nVertexR = fine_grid->nVertex[MarkerR];
377
378 /*--- Allocate Receive and Send buffers ---*/
379
380 vector<unsigned long> Buffer_Receive_Children(nVertexR);
381 vector<unsigned long> Buffer_Send_Children(nVertexS);
382
383 vector<unsigned long> Buffer_Receive_Parent(nVertexR);
384 vector<unsigned long> Buffer_Send_Parent(nVertexS);
385
386 /*--- Copy the information that should be sent, child and parent indices. ---*/
387
388 for (auto iVertex = 0ul; iVertex < nVertexS; iVertex++) {
389 const auto iPoint = fine_grid->vertex[MarkerS][iVertex]->GetNode();
390 Buffer_Send_Children[iVertex] = iPoint;
391 Buffer_Send_Parent[iVertex] = fine_grid->nodes->GetParent_CV(iPoint);
392 }
393
394 /*--- Send/Receive information. ---*/
395
396 SU2_MPI::Sendrecv(Buffer_Send_Children.data(), nVertexS, MPI_UNSIGNED_LONG, send_to, 0,
397 Buffer_Receive_Children.data(), nVertexR, MPI_UNSIGNED_LONG, receive_from, 0,
398 SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
399 SU2_MPI::Sendrecv(Buffer_Send_Parent.data(), nVertexS, MPI_UNSIGNED_LONG, send_to, 1,
400 Buffer_Receive_Parent.data(), nVertexR, MPI_UNSIGNED_LONG, receive_from, 1, SU2_MPI::GetComm(),
401 MPI_STATUS_IGNORE);
402
403 /*--- Create a list of the parent nodes without duplicates. ---*/
404
405 auto Aux_Parent = Buffer_Receive_Parent;
406
407 sort(Aux_Parent.begin(), Aux_Parent.end());
408 auto it1 = unique(Aux_Parent.begin(), Aux_Parent.end());
409 Aux_Parent.resize(it1 - Aux_Parent.begin());
410
411 /*--- Create the local and remote vector for the parents and children CVs. ---*/
412
413 const auto& Parent_Remote = Buffer_Receive_Parent;
414 vector<unsigned long> Parent_Local(nVertexR);
415 vector<unsigned long> Children_Local(nVertexR);
416
417 for (auto iVertex = 0ul; iVertex < nVertexR; iVertex++) {
418 /*--- We use the same sorting as in the donor domain, i.e. the local parents
419 are numbered according to their order in the remote rank. ---*/
420
421 for (auto jVertex = 0ul; jVertex < Aux_Parent.size(); jVertex++) {
422 if (Parent_Remote[iVertex] == Aux_Parent[jVertex]) {
423 Parent_Local[iVertex] = jVertex + Index_CoarseCV;
424 break;
425 }
426 }
427 Children_Local[iVertex] = fine_grid->vertex[MarkerR][iVertex]->GetNode();
428 }
429
430 Index_CoarseCV += Aux_Parent.size();
431
432 vector<unsigned short> nChildren_MPI(Index_CoarseCV, 0);
433
434 /*--- Create the final structure ---*/
435 for (auto iVertex = 0ul; iVertex < nVertexR; iVertex++) {
436 const auto iPoint_Coarse = Parent_Local[iVertex];
437 const auto iPoint_Fine = Children_Local[iVertex];
438
439 /*--- Be careful, it is possible that a node changes the agglomeration configuration,
440 the priority is always when receiving the information. ---*/
441
442 fine_grid->nodes->SetParent_CV(iPoint_Fine, iPoint_Coarse);
443 nodes->SetChildren_CV(iPoint_Coarse, nChildren_MPI[iPoint_Coarse], iPoint_Fine);
444 nChildren_MPI[iPoint_Coarse]++;
445 nodes->SetnChildren_CV(iPoint_Coarse, nChildren_MPI[iPoint_Coarse]);
446 nodes->SetDomain(iPoint_Coarse, false);
447 }
448 }
449 }
450#endif // HAVE_MPI
451
452 /*--- Update the number of points after the MPI agglomeration ---*/
453
454 nPoint = Index_CoarseCV;
455
456 /*--- Console output with the summary of the agglomeration ---*/
457
458 unsigned long nPointFine = fine_grid->GetnPoint();
459 unsigned long Global_nPointCoarse, Global_nPointFine;
460
461 SU2_MPI::Allreduce(&nPoint, &Global_nPointCoarse, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
462 SU2_MPI::Allreduce(&nPointFine, &Global_nPointFine, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
463
464 SetGlobal_nPointDomain(Global_nPointCoarse);
465
466 if (iMesh != MESH_0) {
467 const su2double factor = 1.5;
468 const su2double Coeff = pow(su2double(Global_nPointFine) / Global_nPointCoarse, 1.0 / nDim);
469 const su2double CFL = factor * config->GetCFL(iMesh - 1) / Coeff;
470 config->SetCFL(iMesh, CFL);
471 }
472
473 const su2double ratio = su2double(Global_nPointFine) / su2double(Global_nPointCoarse);
474
475 if (((nDim == 2) && (ratio < 2.5)) || ((nDim == 3) && (ratio < 2.5))) {
476 config->SetMGLevels(iMesh - 1);
477 } else if (rank == MASTER_NODE) {
478 PrintingToolbox::CTablePrinter MGTable(&std::cout);
479 MGTable.AddColumn("MG Level", 10);
480 MGTable.AddColumn("CVs", 10);
481 MGTable.AddColumn("Aggl. Rate", 10);
482 MGTable.AddColumn("CFL", 10);
483 MGTable.SetAlign(PrintingToolbox::CTablePrinter::RIGHT);
484
485 if (iMesh == MESH_1) {
486 MGTable.PrintHeader();
487 MGTable << iMesh - 1 << Global_nPointFine << "1/1.00" << config->GetCFL(iMesh - 1);
488 }
489 stringstream ss;
490 ss << "1/" << std::setprecision(3) << ratio;
491 MGTable << iMesh << Global_nPointCoarse << ss.str() << config->GetCFL(iMesh);
492 if (iMesh == config->GetnMGLevels()) {
493 MGTable.PrintFooter();
494 }
495 }
496
497 edgeColorGroupSize = config->GetEdgeColoringGroupSize();
498}

500bool CMultiGridGeometry::SetBoundAgglomeration(unsigned long CVPoint, short marker_seed, const CGeometry* fine_grid,
501 const CConfig* config) const {
502 bool agglomerate_CV = false;
503
504 /*--- Basic condition, the point has not being previously agglomerated, it belongs to the domain,
505 and has passed some basic geometrical checks. ---*/
506
507 if ((!fine_grid->nodes->GetAgglomerate(CVPoint)) && (fine_grid->nodes->GetDomain(CVPoint)) &&
508 (GeometricalCheck(CVPoint, fine_grid, config))) {
509 /*--- If the point belongs to a boundary, its type must be compatible with the seed marker. ---*/
510
511 if (fine_grid->nodes->GetBoundary(CVPoint)) {
512 /*--- Identify the markers of the vertex that we want to agglomerate ---*/
513
514 int counter = 0;
515 unsigned short copy_marker[3] = {};
516 for (auto jMarker = 0u; jMarker < fine_grid->GetnMarker() && counter < 3; jMarker++) {
517 if (fine_grid->nodes->GetVertex(CVPoint, jMarker) != -1) {
518 copy_marker[counter] = jMarker;
519 counter++;
520 }
521 }
522
523 /*--- The basic condition is that the aglomerated vertex must have the same physical marker,
524 but eventually a send-receive condition ---*/
525
526 /*--- Only one marker in the vertex that is going to be aglomerated ---*/
527
528 if (counter == 1) {
529 /*--- We agglomerate if there is only a marker and is the same marker as the seed marker ---*/
530
531 if (copy_marker[0] == marker_seed) agglomerate_CV = true;
532
533 /*--- If there is only one marker, but the marker is the SEND_RECEIVE ---*/
534
535 if (config->GetMarker_All_KindBC(copy_marker[0]) == SEND_RECEIVE) agglomerate_CV = true;
536 }
537
538 /*--- If there are two markers in the vertex that is going to be aglomerated ---*/
539
540 if (counter == 2) {
541 /*--- First we verify that the seed is a physical boundary ---*/
542
543 if (config->GetMarker_All_KindBC(marker_seed) != SEND_RECEIVE) {
544 /*--- Then we check that one of the marker is equal to the seed marker, and the other is send/receive ---*/
545
546 if (((copy_marker[0] == marker_seed) && (config->GetMarker_All_KindBC(copy_marker[1]) == SEND_RECEIVE)) ||
547 ((config->GetMarker_All_KindBC(copy_marker[0]) == SEND_RECEIVE) && (copy_marker[1] == marker_seed)))
548 agglomerate_CV = true;
549 }
550 }
551
552 }
553
554 /*--- If the element belongs to the domain, it is allways aglomerated. ---*/
555
556 else {
557 agglomerate_CV = true;
558 }
559 }
560
561 return agglomerate_CV;
562}

564bool CMultiGridGeometry::GeometricalCheck(unsigned long iPoint, const CGeometry* fine_grid,
565 const CConfig* config) const {
566 su2double max_dimension = 1.2;
567
568 /*--- Evaluate the total size of the element ---*/
569
570 bool Volume = true;
571 su2double ratio = pow(fine_grid->nodes->GetVolume(iPoint), 1.0 / su2double(nDim)) * max_dimension;
572 su2double limit = pow(config->GetDomainVolume(), 1.0 / su2double(nDim));
573 if (ratio > limit) Volume = false;
574
575 /*--- Evaluate the stretching of the element ---*/
576
577 bool Stretching = true;
578
579 /* unsigned short iNode, iDim;
580 unsigned long jPoint;
581 su2double *Coord_i = fine_grid->nodes->GetCoord(iPoint);
582 su2double max_dist = 0.0 ; su2double min_dist = 1E20;
583 for (iNode = 0; iNode < fine_grid->nodes->GetnPoint(iPoint); iNode ++) {
584 jPoint = fine_grid->nodes->GetPoint(iPoint, iNode);
585 su2double *Coord_j = fine_grid->nodes->GetCoord(jPoint);
586 su2double distance = 0.0;
587 for (iDim = 0; iDim < nDim; iDim++)
588 distance += (Coord_j[iDim]-Coord_i[iDim])*(Coord_j[iDim]-Coord_i[iDim]);
589 distance = sqrt(distance);
590 max_dist = max(distance, max_dist);
591 min_dist = min(distance, min_dist);
592 }
593 if ( max_dist/min_dist > 100.0 ) Stretching = false;*/
594
595 return (Stretching && Volume);
596}

598void CMultiGridGeometry::SetSuitableNeighbors(vector<unsigned long>& Suitable_Indirect_Neighbors, unsigned long iPoint,
599 unsigned long Index_CoarseCV, const CGeometry* fine_grid) const {
600 /*--- Create a list with the first neighbors, including the seed. ---*/
601
602 vector<unsigned long> First_Neighbor_Points;
603 First_Neighbor_Points.push_back(iPoint);
604 for (auto jPoint : fine_grid->nodes->GetPoints(iPoint)) First_Neighbor_Points.push_back(jPoint);
605
606 /*--- Create a list with the second neighbors, without first, and seed neighbors. ---*/
607
608 vector<unsigned long> Second_Neighbor_Points, Second_Origin_Points, Suitable_Second_Neighbors;
609
610 for (auto jPoint : fine_grid->nodes->GetPoints(iPoint)) {
611 for (auto kPoint : fine_grid->nodes->GetPoints(jPoint)) {
612 /*--- Check that the second neighbor does not belong to the first neighbors or the seed. ---*/
613
614 auto end = First_Neighbor_Points.end();
615 if (find(First_Neighbor_Points.begin(), end, kPoint) == end) {
616 Second_Neighbor_Points.push_back(kPoint);
617 Second_Origin_Points.push_back(jPoint);
618 }
619 }
620 }
621
622 /*--- Identify those second neighbors that are repeated (candidates to be added).
623 For a mesh of quads this produces a 9-point stencil from the 5-point of direct
624 neighbors, and for hexs it produces a 27-point stencil. ---*/
625
626 for (auto iNeighbor = 0ul; iNeighbor < Second_Neighbor_Points.size(); iNeighbor++) {
627 for (auto jNeighbor = iNeighbor + 1; jNeighbor < Second_Neighbor_Points.size(); jNeighbor++) {
628 /*--- Repeated second neighbor with different origin ---*/
629
630 if ((Second_Neighbor_Points[iNeighbor] == Second_Neighbor_Points[jNeighbor]) &&
631 (Second_Origin_Points[iNeighbor] != Second_Origin_Points[jNeighbor])) {
632 Suitable_Indirect_Neighbors.push_back(Second_Neighbor_Points[iNeighbor]);
633
634 /*--- Create a list of suitable second neighbors, that we will use
635 to compute the third neighbors. --*/
636
637 Suitable_Second_Neighbors.push_back(Second_Neighbor_Points[iNeighbor]);
638 }
639 }
640 }
641
642 /*--- Remove duplicates ---*/
643
644 sort(Suitable_Second_Neighbors.begin(), Suitable_Second_Neighbors.end());
645 auto it1 = unique(Suitable_Second_Neighbors.begin(), Suitable_Second_Neighbors.end());
646 Suitable_Second_Neighbors.resize(it1 - Suitable_Second_Neighbors.begin());
647
648 /*--- Create a list with the third neighbors, without first, second, and seed neighbors. ---*/
649
650 /// TODO: This repeats the process above but I doubt it catches any more points.
651
652 vector<unsigned long> Third_Neighbor_Points, Third_Origin_Points;
653
654 for (auto kPoint : Suitable_Second_Neighbors) {
655 for (auto lPoint : fine_grid->nodes->GetPoints(kPoint)) {
656 /*--- Check that the third neighbor does not belong to the first neighbors or the seed ---*/
657
658 auto end1 = First_Neighbor_Points.end();
659 if (find(First_Neighbor_Points.begin(), end1, lPoint) != end1) continue;
660
661 /*--- Check that the third neighbor does not belong to the second neighbors ---*/
662
663 auto end2 = Suitable_Second_Neighbors.end();
664 if (find(Suitable_Second_Neighbors.begin(), end2, lPoint) != end2) continue;
665
666 Third_Neighbor_Points.push_back(lPoint);
667 Third_Origin_Points.push_back(kPoint);
668 }
669 }
670
671 /*--- Identify those third neighbors that are repeated (candidate to be added). ---*/
672
673 for (auto iNeighbor = 0ul; iNeighbor < Third_Neighbor_Points.size(); iNeighbor++) {
674 for (auto jNeighbor = iNeighbor + 1; jNeighbor < Third_Neighbor_Points.size(); jNeighbor++) {
675 /*--- Repeated third neighbor with different origin ---*/
676
677 if ((Third_Neighbor_Points[iNeighbor] == Third_Neighbor_Points[jNeighbor]) &&
678 (Third_Origin_Points[iNeighbor] != Third_Origin_Points[jNeighbor])) {
679 Suitable_Indirect_Neighbors.push_back(Third_Neighbor_Points[iNeighbor]);
680 }
681 }
682 }
683
684 /*--- Remove duplicates from the final list of Suitable Indirect Neighbors. ---*/
685
686 sort(Suitable_Indirect_Neighbors.begin(), Suitable_Indirect_Neighbors.end());
687 auto it2 = unique(Suitable_Indirect_Neighbors.begin(), Suitable_Indirect_Neighbors.end());
688 Suitable_Indirect_Neighbors.resize(it2 - Suitable_Indirect_Neighbors.begin());
689}

840void CMultiGridGeometry::SetControlVolume(const CGeometry* fine_grid, unsigned short action) {
841 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
842 unsigned long iFinePoint, iCoarsePoint, iEdge, iParent;
843 long FineEdge, CoarseEdge;
844 unsigned short iChildren;
845 bool change_face_orientation;
846 su2double Coarse_Volume, Area;
847
848 /*--- Compute the area of the coarse volume ---*/
849 for (iCoarsePoint = 0; iCoarsePoint < nPoint; iCoarsePoint++) {
850 nodes->SetVolume(iCoarsePoint, 0.0);
851 Coarse_Volume = 0.0;
852 for (iChildren = 0; iChildren < nodes->GetnChildren_CV(iCoarsePoint); iChildren++) {
853 iFinePoint = nodes->GetChildren_CV(iCoarsePoint, iChildren);
854 Coarse_Volume += fine_grid->nodes->GetVolume(iFinePoint);
855 }
856 nodes->SetVolume(iCoarsePoint, Coarse_Volume);
857 }
858
859 /*--- Update or not the values of faces at the edge ---*/
860 if (action != ALLOCATE) {
861 edges->SetZeroValues();
862 }
863
864 for (iCoarsePoint = 0; iCoarsePoint < nPoint; iCoarsePoint++)
865 for (iChildren = 0; iChildren < nodes->GetnChildren_CV(iCoarsePoint); iChildren++) {
866 iFinePoint = nodes->GetChildren_CV(iCoarsePoint, iChildren);
867
868 for (auto iFinePoint_Neighbor : fine_grid->nodes->GetPoints(iFinePoint)) {
869 iParent = fine_grid->nodes->GetParent_CV(iFinePoint_Neighbor);
870 if ((iParent != iCoarsePoint) && (iParent < iCoarsePoint)) {
871 FineEdge = fine_grid->FindEdge(iFinePoint, iFinePoint_Neighbor);
872
873 change_face_orientation = false;
874 if (iFinePoint < iFinePoint_Neighbor) change_face_orientation = true;
875

877
878 const auto Normal = fine_grid->edges->GetNormal(FineEdge);
879
880 if (change_face_orientation) {
881 edges->SubNormal(CoarseEdge, Normal);
882 } else {
883 edges->AddNormal(CoarseEdge, Normal);

885 }
886 }
887 }
888
889 /*--- Check if there is a normal with null area ---*/
890
891 for (iEdge = 0; iEdge < nEdge; iEdge++) {
892 const auto NormalFace = edges->GetNormal(iEdge);
893 Area = GeometryToolbox::Norm(nDim, NormalFace);
894 if (Area == 0.0) {
895 su2double DefaultNormal[3] = {EPS * EPS};
896 edges->SetNormal(iEdge, DefaultNormal);
897 }
898 }
899 }

1021void CMultiGridGeometry::FindNormal_Neighbor(const CConfig* config) {
1022 unsigned short iMarker, iDim;
1023 unsigned long iPoint, iVertex;
1024
1025 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
1026 if (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE &&
1027 config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY &&
1028 config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) {
1029 for (iVertex = 0; iVertex < nVertex[iMarker]; iVertex++) {
1030 iPoint = vertex[iMarker][iVertex]->GetNode();
1031
1032 /*--- If the node belong to the domain ---*/
1033 if (nodes->GetDomain(iPoint)) {
1034 /*--- Compute closest normal neighbor ---*/
1035 su2double cos_max, scalar_prod, norm_vect, norm_Normal, cos_alpha, diff_coord;
1036 unsigned long Point_Normal = 0;
1037 su2double* Normal = vertex[iMarker][iVertex]->GetNormal();
1038 cos_max = -1.0;
1039 for (auto jPoint : nodes->GetPoints(iPoint)) {
1040 scalar_prod = 0.0;
1041 norm_vect = 0.0;
1042 norm_Normal = 0.0;
1043 for (iDim = 0; iDim < nDim; iDim++) {
1044 diff_coord = nodes->GetCoord(jPoint, iDim) - nodes->GetCoord(iPoint, iDim);
1045 scalar_prod += diff_coord * Normal[iDim];
1046 norm_vect += diff_coord * diff_coord;
1047 norm_Normal += Normal[iDim] * Normal[iDim];
1048 }
1049 norm_vect = sqrt(norm_vect);
1050 norm_Normal = sqrt(norm_Normal);
1051 cos_alpha = scalar_prod / (norm_vect * norm_Normal);
1052
1053 /*--- Get maximum cosine (not minimum because normals are oriented inwards) ---*/
1054 if (cos_alpha >= cos_max) {
1055 Point_Normal = jPoint;
1056 cos_max = cos_alpha;
1057 }
1058 }
1059 vertex[iMarker][iVertex]->SetNormal_Neighbor(Point_Normal);
1060 }
1061 }
1062 }
