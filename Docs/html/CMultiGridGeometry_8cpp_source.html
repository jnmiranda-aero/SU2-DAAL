33 CMultiGridGeometry::CMultiGridGeometry(CGeometry* fine_grid, CConfig* config, unsigned short iMesh) : CGeometry() {
34  nDim = fine_grid->GetnDim(); // Write the number of dimensions of the coarse grid.
35 
36  /*--- Create a queue system to do the agglomeration
37  1st) More than two markers ---> Vertices (never agglomerate)
38  2nd) Two markers ---> Edges (agglomerate if same BC, never agglomerate if different BC)
39  3rd) One marker ---> Surface (always agglomarate)
40  4th) No marker ---> Internal Volume (always agglomarate) ---*/
41 
42  /*--- Set a marker to indicate indirect agglomeration, for quads and hexs,
43  i.e. consider up to neighbors of neighbors of neighbors.
44  For other levels this information is propagated down during their construction. ---*/
45 
46  if (iMesh == MESH_1) {
47  for (auto iPoint = 0ul; iPoint < fine_grid->GetnPoint(); iPoint++)
48  fine_grid->nodes->SetAgglomerate_Indirect(iPoint, false);
49 
50  for (auto iElem = 0ul; iElem < fine_grid->GetnElem(); iElem++) {
51  if ((fine_grid->elem[iElem]->GetVTK_Type() == HEXAHEDRON) ||
52  (fine_grid->elem[iElem]->GetVTK_Type() == QUADRILATERAL)) {
53  for (auto iNode = 0u; iNode < fine_grid->elem[iElem]->GetnNodes(); iNode++) {
54  const auto iPoint = fine_grid->elem[iElem]->GetNode(iNode);
55  fine_grid->nodes->SetAgglomerate_Indirect(iPoint, true);
56  }
57  }
58  }
59  }
60 
61  /*--- Create the coarse grid structure using as baseline the fine grid ---*/
62 
63  CMultiGridQueue MGQueue_InnerCV(fine_grid->GetnPoint());
64  vector<unsigned long> Suitable_Indirect_Neighbors;
65 
66  nodes = new CPoint(fine_grid->GetnPoint(), nDim, iMesh, config);
67 
68  unsigned long Index_CoarseCV = 0;
69 
70  /*--- The first step is the boundary agglomeration. ---*/
71 
72  for (auto iMarker = 0u; iMarker < fine_grid->GetnMarker(); iMarker++) {
73  for (auto iVertex = 0ul; iVertex < fine_grid->GetnVertex(iMarker); iVertex++) {
74  const auto iPoint = fine_grid->vertex[iMarker][iVertex]->GetNode();
75 
76  /*--- If the element has not been previously agglomerated and it
77  belongs to this physical domain, and it meets the geometrical
78  criteria, the agglomeration is studied. ---*/
79 
80  if ((!fine_grid->nodes->GetAgglomerate(iPoint)) && (fine_grid->nodes->GetDomain(iPoint)) &&
81  (GeometricalCheck(iPoint, fine_grid, config))) {
82  unsigned short nChildren = 1;
83 
84  /*--- We set an index for the parent control volume, this
85  also marks it as agglomerated. ---*/
86 
87  fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
88 
89  /*--- We add the seed point (child) to the parent control volume ---*/
90 
91  nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
92  bool agglomerate_seed = true;
93  auto counter = 0;
94  unsigned short copy_marker[3] = {};
95  const auto marker_seed = iMarker;
96 
97  /*--- For a particular point in the fine grid we save all the markers
98  that are in that point ---*/
99 
100  for (auto jMarker = 0u; jMarker < fine_grid->GetnMarker() && counter < 3; jMarker++) {
101  if (fine_grid->nodes->GetVertex(iPoint, jMarker) != -1) {
102  copy_marker[counter] = jMarker;
103  counter++;
104  }
105  }
106 
107  /*--- To aglomerate a vertex it must have only one physical bc!!
108  This can be improved. If there is only a marker, it is a good
109  candidate for agglomeration ---*/
110 
111  if (counter == 1) {
112  agglomerate_seed = true;
113 
114  /*--- Euler walls can be curved and agglomerating them leads to difficulties ---*/
115  if (config->GetMarker_All_KindBC(marker_seed) == EULER_WALL) agglomerate_seed = false;
116  }
117  /*--- If there are two markers, we will agglomerate if any of the
118  markers is SEND_RECEIVE ---*/
119 
120  if (counter == 2) {
121  agglomerate_seed = (config->GetMarker_All_KindBC(copy_marker[0]) == SEND_RECEIVE) ||
122  (config->GetMarker_All_KindBC(copy_marker[1]) == SEND_RECEIVE);
123 
124  /* --- Euler walls can also not be agglomerated when the point has 2 markers ---*/
125  if ((config->GetMarker_All_KindBC(copy_marker[0]) == EULER_WALL) ||
126  (config->GetMarker_All_KindBC(copy_marker[1]) == EULER_WALL)) {
127  agglomerate_seed = false;
128  }
129  }
130 
131  /*--- If there are more than 2 markers, the aglomeration will be discarded ---*/
132 
133  if (counter > 2) agglomerate_seed = false;
134 
135  /*--- If the seed can be agglomerated, we try to agglomerate more points ---*/
136 
137  if (agglomerate_seed) {
138  /*--- Now we do a sweep over all the nodes that surround the seed point ---*/
139 
140  for (auto CVPoint : fine_grid->nodes->GetPoints(iPoint)) {
141  /*--- The new point can be agglomerated ---*/
142 
143  if (SetBoundAgglomeration(CVPoint, marker_seed, fine_grid, config)) {
144  /*--- We set the value of the parent ---*/
145 
146  fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
147 
148  /*--- We set the value of the child ---*/
149 
150  nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
151  nChildren++;
152  }
153  }
154 
155  Suitable_Indirect_Neighbors.clear();
156 
157  if (fine_grid->nodes->GetAgglomerate_Indirect(iPoint))
158  SetSuitableNeighbors(Suitable_Indirect_Neighbors, iPoint, Index_CoarseCV, fine_grid);
159 
160  /*--- Now we do a sweep over all the indirect nodes that can be added ---*/
161 
162  for (auto CVPoint : Suitable_Indirect_Neighbors) {
163  /*--- The new point can be agglomerated ---*/
164 
165  if (SetBoundAgglomeration(CVPoint, marker_seed, fine_grid, config)) {
166  /*--- We set the value of the parent ---*/
167 
168  fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
169 
170  /*--- We set the indirect agglomeration information of the corse point
171  based on its children in the fine grid. ---*/
172 
173  if (fine_grid->nodes->GetAgglomerate_Indirect(CVPoint))
174  nodes->SetAgglomerate_Indirect(Index_CoarseCV, true);
175 
176  /*--- We set the value of the child ---*/
177 
178  nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
179  nChildren++;
180  }
181  }
182  }
183 
184  /*--- Update the number of children of the coarse control volume. ---*/
185 
186  nodes->SetnChildren_CV(Index_CoarseCV, nChildren);
187  Index_CoarseCV++;
188  }
189  }
190  }
191 
192  /*--- Do not agglomerate any leftover node with more than one physical boundary condition,
193  i.e. make one coarse CV with a single child. ---*/
194 
195  for (auto iMarker = 0u; iMarker < fine_grid->GetnMarker(); iMarker++) {
196  for (auto iVertex = 0ul; iVertex < fine_grid->GetnVertex(iMarker); iVertex++) {
197  const auto iPoint = fine_grid->vertex[iMarker][iVertex]->GetNode();
198 
199  if ((!fine_grid->nodes->GetAgglomerate(iPoint)) && (fine_grid->nodes->GetDomain(iPoint))) {
200  fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
201  nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
202  nodes->SetnChildren_CV(Index_CoarseCV, 1);
203  Index_CoarseCV++;
204  }
205  }
206  }
207 
208  /*--- Update the queue with the results from the boundary agglomeration ---*/
209 
210  for (auto iPoint = 0ul; iPoint < fine_grid->GetnPoint(); iPoint++) {
211  if (fine_grid->nodes->GetAgglomerate(iPoint)) {
212  MGQueue_InnerCV.RemoveCV(iPoint);
213 
214  } else {
215  /*--- Count the number of agglomerated neighbors, and modify the queue,
216  Points with more agglomerated neighbors are processed first. ---*/
217 
218  short priority = 0;
219  for (auto jPoint : fine_grid->nodes->GetPoints(iPoint)) {
220  priority += fine_grid->nodes->GetAgglomerate(jPoint);
221  }
222  MGQueue_InnerCV.MoveCV(iPoint, priority);
223  }
224  }
225 
226  /*--- Agglomerate the domain points. ---*/
227 
228  auto iteration = 0ul;
229  while (!MGQueue_InnerCV.EmptyQueue() && (iteration < fine_grid->GetnPoint())) {
230  const auto iPoint = MGQueue_InnerCV.NextCV();
231  iteration++;
232 
233  /*--- If the element has not been previously agglomerated, belongs to the physical domain,
234  and satisfies several geometrical criteria then the seed CV is accepted for agglomeration. ---*/
235 
236  if ((!fine_grid->nodes->GetAgglomerate(iPoint)) && (fine_grid->nodes->GetDomain(iPoint)) &&
237  (GeometricalCheck(iPoint, fine_grid, config))) {
238  unsigned short nChildren = 1;
239 
240  /*--- We set an index for the parent control volume ---*/
241 
242  fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
243 
244  /*--- We add the seed point (child) to the parent control volume ---*/
245 
246  nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
247 
248  /*--- Update the queue with the seed point (remove the seed and
249  increase the priority of its neighbors) ---*/
250 
251  MGQueue_InnerCV.Update(iPoint, fine_grid);
252 
253  /*--- Now we do a sweep over all the nodes that surround the seed point ---*/
254 
255  for (auto CVPoint : fine_grid->nodes->GetPoints(iPoint)) {
256  /*--- Determine if the CVPoint can be agglomerated ---*/
257 
258  if ((!fine_grid->nodes->GetAgglomerate(CVPoint)) && (fine_grid->nodes->GetDomain(CVPoint)) &&
259  (GeometricalCheck(CVPoint, fine_grid, config))) {
260  /*--- We set the value of the parent ---*/
261 
262  fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
263 
264  /*--- We set the value of the child ---*/
265 
266  nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
267  nChildren++;
268 
269  /*--- Update the queue with the new control volume (remove the CV and
270  increase the priority of its neighbors) ---*/
271 
272  MGQueue_InnerCV.Update(CVPoint, fine_grid);
273  }
274  }
275 
276  /*--- Identify the indirect neighbors ---*/
277 
278  Suitable_Indirect_Neighbors.clear();
279  if (fine_grid->nodes->GetAgglomerate_Indirect(iPoint))
280  SetSuitableNeighbors(Suitable_Indirect_Neighbors, iPoint, Index_CoarseCV, fine_grid);
281 
282  /*--- Now we do a sweep over all the indirect nodes that can be added ---*/
283 
284  for (auto CVPoint : Suitable_Indirect_Neighbors) {
285  /*--- The new point can be agglomerated ---*/
286 
287  if ((!fine_grid->nodes->GetAgglomerate(CVPoint)) && (fine_grid->nodes->GetDomain(CVPoint))) {
288  /*--- We set the value of the parent ---*/
289 
290  fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
291 
292  /*--- We set the indirect agglomeration information ---*/
293 
294  if (fine_grid->nodes->GetAgglomerate_Indirect(CVPoint)) nodes->SetAgglomerate_Indirect(Index_CoarseCV, true);
295 
296  /*--- We set the value of the child ---*/
297 
298  nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
299  nChildren++;
300 
301  /*--- Update the queue with the new control volume (remove the CV and
302  increase the priority of the neighbors) ---*/
303 
304  MGQueue_InnerCV.Update(CVPoint, fine_grid);
305  }
306  }
307 
308  /*--- Update the number of control of childrens ---*/
309 
310  nodes->SetnChildren_CV(Index_CoarseCV, nChildren);
311  Index_CoarseCV++;
312  } else {
313  /*--- The seed point can not be agglomerated because of size, domain, streching, etc.
314  move the point to the lowest priority ---*/
315 
316  MGQueue_InnerCV.MoveCV(iPoint, -1);
317  }
318  }
319 
320  /*--- Convert any point that was not agglomerated into a coarse point. ---*/
321 
322  for (auto iPoint = 0ul; iPoint < fine_grid->GetnPoint(); iPoint++) {
323  if ((!fine_grid->nodes->GetAgglomerate(iPoint)) && (fine_grid->nodes->GetDomain(iPoint))) {
324  fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
325  if (fine_grid->nodes->GetAgglomerate_Indirect(iPoint)) nodes->SetAgglomerate_Indirect(Index_CoarseCV, true);
326  nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
327  nodes->SetnChildren_CV(Index_CoarseCV, 1);
328  Index_CoarseCV++;
329  }
330  }
331 
332  nPointDomain = Index_CoarseCV;
333  nPoint = nPointDomain;
334 
335  /*--- Check that there are no hanging nodes. Detect isolated points
336  (only 1 neighbor), and merge their children CV's with the neighbor. ---*/
337 
338  SetPoint_Connectivity(fine_grid);
339 
340  for (auto iCoarsePoint = 0ul; iCoarsePoint < nPointDomain; iCoarsePoint++) {
341  if (nodes->GetnPoint(iCoarsePoint) == 1) {
342  /*--- Find the neighbor of the isolated point. This neighbor is the right control volume ---*/
343 
344  const auto iCoarsePoint_Complete = nodes->GetPoint(iCoarsePoint, 0);
345 
346  /*--- Add the children to the connected control volume (and modify its parent indexing).
347  Identify the child CV from the finest grid and add it to the correct control volume.
348  Set the parent CV of iFinePoint. Instead of using the original one
349  (iCoarsePoint), use the new one (iCoarsePoint_Complete) ---*/
350 
351  auto nChildren = nodes->GetnChildren_CV(iCoarsePoint_Complete);
352 
353  for (auto iChildren = 0u; iChildren < nodes->GetnChildren_CV(iCoarsePoint); iChildren++) {
354  const auto iFinePoint = nodes->GetChildren_CV(iCoarsePoint, iChildren);
355  nodes->SetChildren_CV(iCoarsePoint_Complete, nChildren, iFinePoint);
356  nChildren++;
357  fine_grid->nodes->SetParent_CV(iFinePoint, iCoarsePoint_Complete);
358  }
359 
360  /*--- Update the number of children control volumes ---*/
361 
362  nodes->SetnChildren_CV(iCoarsePoint_Complete, nChildren);
363  nodes->SetnChildren_CV(iCoarsePoint, 0);
364  }
365  }
366 
367  /*--- Reset the neighbor information. ---*/
368 
369  nodes->ResetPoints();
370 
371 #ifdef HAVE_MPI
372  /*--- Dealing with MPI parallelization, the objective is that the received nodes must be agglomerated
373  in the same way as the donor (send) nodes. Send the node agglomeration information of the donor
374  (parent and children). The agglomerated halos of this rank are set according to the rank where
375  they are domain points. ---*/
376 
377  for (auto iMarker = 0u; iMarker < config->GetnMarker_All(); iMarker++) {
378  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
379  const auto MarkerS = iMarker;
380  const auto MarkerR = iMarker + 1;
381 
382  const auto send_to = config->GetMarker_All_SendRecv(MarkerS) - 1;
383  const auto receive_from = abs(config->GetMarker_All_SendRecv(MarkerR)) - 1;
384 
385  const auto nVertexS = fine_grid->nVertex[MarkerS];
386  const auto nVertexR = fine_grid->nVertex[MarkerR];
387 
388  /*--- Allocate Receive and Send buffers ---*/
389 
390  vector<unsigned long> Buffer_Receive_Children(nVertexR);
391  vector<unsigned long> Buffer_Send_Children(nVertexS);
392 
393  vector<unsigned long> Buffer_Receive_Parent(nVertexR);
394  vector<unsigned long> Buffer_Send_Parent(nVertexS);
395 
396  /*--- Copy the information that should be sent, child and parent indices. ---*/
397 
398  for (auto iVertex = 0ul; iVertex < nVertexS; iVertex++) {
399  const auto iPoint = fine_grid->vertex[MarkerS][iVertex]->GetNode();
400  Buffer_Send_Children[iVertex] = iPoint;
401  Buffer_Send_Parent[iVertex] = fine_grid->nodes->GetParent_CV(iPoint);
402  }
403 
404  /*--- Send/Receive information. ---*/
405 
406  SU2_MPI::Sendrecv(Buffer_Send_Children.data(), nVertexS, MPI_UNSIGNED_LONG, send_to, 0,
407  Buffer_Receive_Children.data(), nVertexR, MPI_UNSIGNED_LONG, receive_from, 0,
408  SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
409  SU2_MPI::Sendrecv(Buffer_Send_Parent.data(), nVertexS, MPI_UNSIGNED_LONG, send_to, 1,
410  Buffer_Receive_Parent.data(), nVertexR, MPI_UNSIGNED_LONG, receive_from, 1, SU2_MPI::GetComm(),
411  MPI_STATUS_IGNORE);
412 
413  /*--- Create a list of the parent nodes without duplicates. ---*/
414 
415  auto Aux_Parent = Buffer_Receive_Parent;
416 
417  sort(Aux_Parent.begin(), Aux_Parent.end());
418  auto it1 = unique(Aux_Parent.begin(), Aux_Parent.end());
419  Aux_Parent.resize(it1 - Aux_Parent.begin());
420 
421  /*--- Create the local and remote vector for the parents and children CVs. ---*/
422 
423  const auto& Parent_Remote = Buffer_Receive_Parent;
424  vector<unsigned long> Parent_Local(nVertexR);
425  vector<unsigned long> Children_Local(nVertexR);
426 
427  for (auto iVertex = 0ul; iVertex < nVertexR; iVertex++) {
428  /*--- We use the same sorting as in the donor domain, i.e. the local parents
429  are numbered according to their order in the remote rank. ---*/
430 
431  for (auto jVertex = 0ul; jVertex < Aux_Parent.size(); jVertex++) {
432  if (Parent_Remote[iVertex] == Aux_Parent[jVertex]) {
433  Parent_Local[iVertex] = jVertex + Index_CoarseCV;
434  break;
435  }
436  }
437  Children_Local[iVertex] = fine_grid->vertex[MarkerR][iVertex]->GetNode();
438  }
439 
440  Index_CoarseCV += Aux_Parent.size();
441 
442  vector<unsigned short> nChildren_MPI(Index_CoarseCV, 0);
443 
444  /*--- Create the final structure ---*/
445  for (auto iVertex = 0ul; iVertex < nVertexR; iVertex++) {
446  const auto iPoint_Coarse = Parent_Local[iVertex];
447  const auto iPoint_Fine = Children_Local[iVertex];
448 
449  /*--- Be careful, it is possible that a node changes the agglomeration configuration,
450  the priority is always when receiving the information. ---*/
451  fine_grid->nodes->SetParent_CV(iPoint_Fine, iPoint_Coarse);
452  nodes->SetChildren_CV(iPoint_Coarse, nChildren_MPI[iPoint_Coarse], iPoint_Fine);
453  nChildren_MPI[iPoint_Coarse]++;
454  nodes->SetnChildren_CV(iPoint_Coarse, nChildren_MPI[iPoint_Coarse]);
455  nodes->SetDomain(iPoint_Coarse, false);
456  }
457  }
458  }
459 #endif // HAVE_MPI
460 
461  /*--- Update the number of points after the MPI agglomeration ---*/
462 
463  nPoint = Index_CoarseCV;
464 
465  /*--- Console output with the summary of the agglomeration ---*/
466 
467  unsigned long nPointFine = fine_grid->GetnPoint();
468  unsigned long Global_nPointCoarse, Global_nPointFine;
469 
470  SU2_MPI::Allreduce(&nPoint, &Global_nPointCoarse, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
471  SU2_MPI::Allreduce(&nPointFine, &Global_nPointFine, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
472 
473  SetGlobal_nPointDomain(Global_nPointCoarse);
474 
475  if (iMesh != MESH_0) {
476  const su2double factor = 1.5;
477  const su2double Coeff = pow(su2double(Global_nPointFine) / Global_nPointCoarse, 1.0 / nDim);
478  const su2double CFL = factor * config->GetCFL(iMesh - 1) / Coeff;
479  config->SetCFL(iMesh, CFL);
480  }
481 
482  const su2double ratio = su2double(Global_nPointFine) / su2double(Global_nPointCoarse);
483 
484  if (((nDim == 2) && (ratio < 2.5)) || ((nDim == 3) && (ratio < 2.5))) {
485  config->SetMGLevels(iMesh - 1);
486  } else if (rank == MASTER_NODE) {
487  PrintingToolbox::CTablePrinter MGTable(&std::cout);
488  MGTable.AddColumn("MG Level", 10);
489  MGTable.AddColumn("CVs", 10);
490  MGTable.AddColumn("Aggl. Rate", 10);
491  MGTable.AddColumn("CFL", 10);
492  MGTable.SetAlign(PrintingToolbox::CTablePrinter::RIGHT);
493 
494  if (iMesh == MESH_1) {
495  MGTable.PrintHeader();
496  MGTable << iMesh - 1 << Global_nPointFine << "1/1.00" << config->GetCFL(iMesh - 1);
497  }
498  stringstream ss;
499  ss << "1/" << std::setprecision(3) << ratio;
500  MGTable << iMesh << Global_nPointCoarse << ss.str() << config->GetCFL(iMesh);
501  if (iMesh == config->GetnMGLevels()) {
502  MGTable.PrintFooter();
503  }
504  }
505 
506  edgeColorGroupSize = config->GetEdgeColoringGroupSize();
507 }

509 bool CMultiGridGeometry::SetBoundAgglomeration(unsigned long CVPoint, short marker_seed, const CGeometry* fine_grid,
510  const CConfig* config) const {
511  bool agglomerate_CV = false;
512 
513  /*--- Basic condition, the point has not been previously agglomerated, it belongs to the domain,
514  and has passed some basic geometrical checks. ---*/
515 
516  if ((!fine_grid->nodes->GetAgglomerate(CVPoint)) && (fine_grid->nodes->GetDomain(CVPoint)) &&
517  (GeometricalCheck(CVPoint, fine_grid, config))) {
518  /*--- If the point belongs to a boundary, its type must be compatible with the seed marker. ---*/
519 
520  if (fine_grid->nodes->GetBoundary(CVPoint)) {
521  /*--- Identify the markers of the vertex that we want to agglomerate ---*/
522 
523  // count number of markers on the agglomeration candidate
524  int counter = 0;
525  unsigned short copy_marker[3] = {};
526  for (auto jMarker = 0u; jMarker < fine_grid->GetnMarker() && counter < 3; jMarker++) {
527  if (fine_grid->nodes->GetVertex(CVPoint, jMarker) != -1) {
528  copy_marker[counter] = jMarker;
529  counter++;
530  }
531  }
532 
533  /*--- The basic condition is that the aglomerated vertex must have the same physical marker,
534  but eventually a send-receive condition ---*/
535 
536  /*--- Only one marker in the vertex that is going to be aglomerated ---*/
537 
538  if (counter == 1) {
539  /*--- We agglomerate if there is only one marker and it is the same marker as the seed marker ---*/
540  // note that this should be the same marker id, not just the same marker type
541  if (copy_marker[0] == marker_seed) agglomerate_CV = true;
542 
543  /*--- If there is only one marker, but the marker is the SEND_RECEIVE ---*/
544 
545  if (config->GetMarker_All_KindBC(copy_marker[0]) == SEND_RECEIVE) {
546  agglomerate_CV = true;
547  }
548 
549  if ((config->GetMarker_All_KindBC(marker_seed) == SYMMETRY_PLANE) ||
550  (config->GetMarker_All_KindBC(marker_seed) == EULER_WALL)) {
551  if (config->GetMarker_All_KindBC(copy_marker[0]) == SEND_RECEIVE) {
552  agglomerate_CV = false;
553  }
554  }
555  }
556 
557  /*--- If there are two markers in the vertex that is going to be aglomerated ---*/
558 
559  if (counter == 2) {
560  /*--- First we verify that the seed is a physical boundary ---*/
561 
562  if (config->GetMarker_All_KindBC(marker_seed) != SEND_RECEIVE) {
563  /*--- Then we check that one of the markers is equal to the seed marker, and the other is send/receive ---*/
564 
565  if (((copy_marker[0] == marker_seed) && (config->GetMarker_All_KindBC(copy_marker[1]) == SEND_RECEIVE)) ||

622 void CMultiGridGeometry::SetSuitableNeighbors(vector<unsigned long>& Suitable_Indirect_Neighbors, unsigned long iPoint,
623  unsigned long Index_CoarseCV, const CGeometry* fine_grid) const {
624  /*--- Create a list with the first neighbors, including the seed. ---*/
625 
626  vector<unsigned long> First_Neighbor_Points;
627  First_Neighbor_Points.push_back(iPoint);
628  for (auto jPoint : fine_grid->nodes->GetPoints(iPoint)) First_Neighbor_Points.push_back(jPoint);
629 
630  /*--- Create a list with the second neighbors, without first, and seed neighbors. ---*/
631 
632  vector<unsigned long> Second_Neighbor_Points, Second_Origin_Points, Suitable_Second_Neighbors;
633 
634  for (auto jPoint : fine_grid->nodes->GetPoints(iPoint)) {
635  for (auto kPoint : fine_grid->nodes->GetPoints(jPoint)) {
636  /*--- Check that the second neighbor does not belong to the first neighbors or the seed. ---*/
637 
638  auto end = First_Neighbor_Points.end();
639  if (find(First_Neighbor_Points.begin(), end, kPoint) == end) {
640  Second_Neighbor_Points.push_back(kPoint);
641  Second_Origin_Points.push_back(jPoint);
642  }
643  }
644  }
645 
646  /*--- Identify those second neighbors that are repeated (candidates to be added).
647  For a mesh of quads this produces a 9-point stencil from the 5-point of direct
648  neighbors, and for hexs it produces a 27-point stencil. ---*/
649 
650  for (auto iNeighbor = 0ul; iNeighbor < Second_Neighbor_Points.size(); iNeighbor++) {
651  for (auto jNeighbor = iNeighbor + 1; jNeighbor < Second_Neighbor_Points.size(); jNeighbor++) {
652  /*--- Repeated second neighbor with different origin ---*/
653 
654  if ((Second_Neighbor_Points[iNeighbor] == Second_Neighbor_Points[jNeighbor]) &&
655  (Second_Origin_Points[iNeighbor] != Second_Origin_Points[jNeighbor])) {
656  Suitable_Indirect_Neighbors.push_back(Second_Neighbor_Points[iNeighbor]);
657 
658  /*--- Create a list of suitable second neighbors, that we will use
659  to compute the third neighbors. --*/
660 
661  Suitable_Second_Neighbors.push_back(Second_Neighbor_Points[iNeighbor]);
662  }
663  }
664  }
665 
666  /*--- Remove duplicates ---*/
667 
668  sort(Suitable_Second_Neighbors.begin(), Suitable_Second_Neighbors.end());
669  auto it1 = unique(Suitable_Second_Neighbors.begin(), Suitable_Second_Neighbors.end());
670  Suitable_Second_Neighbors.resize(it1 - Suitable_Second_Neighbors.begin());
671 
672  /*--- Create a list with the third neighbors, without first, second, and seed neighbors. ---*/
673 
674  /// TODO: This repeats the process above but I doubt it catches any more points.
675 
676  vector<unsigned long> Third_Neighbor_Points, Third_Origin_Points;
677 
678  for (auto kPoint : Suitable_Second_Neighbors) {
679  for (auto lPoint : fine_grid->nodes->GetPoints(kPoint)) {
680  /*--- Check that the third neighbor does not belong to the first neighbors or the seed ---*/
681 
682  auto end1 = First_Neighbor_Points.end();
683  if (find(First_Neighbor_Points.begin(), end1, lPoint) != end1) continue;
684 
685  /*--- Check that the third neighbor does not belong to the second neighbors ---*/
686 
687  auto end2 = Suitable_Second_Neighbors.end();
688  if (find(Suitable_Second_Neighbors.begin(), end2, lPoint) != end2) continue;
689 
690  Third_Neighbor_Points.push_back(lPoint);
691  Third_Origin_Points.push_back(kPoint);
692  }
693  }
694 
695  /*--- Identify those third neighbors that are repeated (candidate to be added). ---*/
696 
697  for (auto iNeighbor = 0ul; iNeighbor < Third_Neighbor_Points.size(); iNeighbor++) {
698  for (auto jNeighbor = iNeighbor + 1; jNeighbor < Third_Neighbor_Points.size(); jNeighbor++) {
699  /*--- Repeated third neighbor with different origin ---*/
700 
701  if ((Third_Neighbor_Points[iNeighbor] == Third_Neighbor_Points[jNeighbor]) &&
702  (Third_Origin_Points[iNeighbor] != Third_Origin_Points[jNeighbor])) {
703  Suitable_Indirect_Neighbors.push_back(Third_Neighbor_Points[iNeighbor]);
704  }
705  }
706  }
707 
708  /*--- Remove duplicates from the final list of Suitable Indirect Neighbors. ---*/
709 
710  sort(Suitable_Indirect_Neighbors.begin(), Suitable_Indirect_Neighbors.end());
711  auto it2 = unique(Suitable_Indirect_Neighbors.begin(), Suitable_Indirect_Neighbors.end());
712  Suitable_Indirect_Neighbors.resize(it2 - Suitable_Indirect_Neighbors.begin());
713 }

1048 void CMultiGridGeometry::FindNormal_Neighbor(const CConfig* config) {
1049  unsigned short iMarker, iDim;
1050  unsigned long iPoint, iVertex;
1051 
1052  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
1053  if (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE &&
1054  config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY &&
1055  config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) {
1056  for (iVertex = 0; iVertex < nVertex[iMarker]; iVertex++) {
1057  iPoint = vertex[iMarker][iVertex]->GetNode();
1058 
1059  /*--- If the node belong to the domain ---*/
1060  if (nodes->GetDomain(iPoint)) {
1061  /*--- Compute closest normal neighbor ---*/
1062  su2double cos_max, scalar_prod, norm_vect, norm_Normal, cos_alpha, diff_coord;
1063  unsigned long Point_Normal = 0;
1064  su2double* Normal = vertex[iMarker][iVertex]->GetNormal();
1065  cos_max = -1.0;
1066  for (auto jPoint : nodes->GetPoints(iPoint)) {
1067  scalar_prod = 0.0;
1068  norm_vect = 0.0;
1069  norm_Normal = 0.0;
1070  for (iDim = 0; iDim < nDim; iDim++) {
1071  diff_coord = nodes->GetCoord(jPoint, iDim) - nodes->GetCoord(iPoint, iDim);
1072  scalar_prod += diff_coord * Normal[iDim];
1073  norm_vect += diff_coord * diff_coord;
1074  norm_Normal += Normal[iDim] * Normal[iDim];
1075  }
1076  norm_vect = sqrt(norm_vect);
1077  norm_Normal = sqrt(norm_Normal);
1078  cos_alpha = scalar_prod / (norm_vect * norm_Normal);
1079
