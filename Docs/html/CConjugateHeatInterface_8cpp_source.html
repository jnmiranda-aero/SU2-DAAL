39 void CConjugateHeatInterface::GetDonor_Variable(CSolver *donor_solution, CGeometry *donor_geometry,
40  const CConfig *donor_config, unsigned long Marker_Donor,
41  unsigned long Vertex_Donor, unsigned long Point_Donor) {
42 
43  /*--- Compute distance of donor point to PointNormal for T-gradient/heatflux computation ---*/
44  const auto nDim = donor_geometry->GetnDim();
45  const auto Coord = donor_geometry->nodes->GetCoord(Point_Donor);
46  const auto PointNormal = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNormal_Neighbor();
47  const auto Coord_Normal = donor_geometry->nodes->GetCoord(PointNormal);
48 
49  su2double Edge_Vector[MAXNDIM] = {0.0};
50  GeometryToolbox::Distance(nDim, Coord_Normal, Coord, Edge_Vector);
51  const su2double dist = GeometryToolbox::Norm(nDim, Edge_Vector);
52 
53  /*--- Retrieve temperature solution and its gradient ---*/
54 
55  const su2double Twall = donor_solution->GetNodes()->GetTemperature(Point_Donor);
56  const su2double Tnormal = donor_solution->GetNodes()->GetTemperature(PointNormal);
57 
58  const su2double dTdn = (Twall - Tnormal)/dist;
59  // TODO: Check if these improve accuracy, if needed at all
60  // const auto Normal = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNormal();
61  // su2double Area = GeometryToolbox::Norm(nDim, Normal);
62  // for (iDim = 0; iDim < nDim; iDim++) {
63  // dTdn += (Twall - Tnormal)/dist * (Edge_Vector[iDim]/dist) * (Normal[iDim]/Area);
64  // }
65 
66  /*--- Calculate the heat flux density (temperature gradient times thermal conductivity) and
67  thermal conductivity divided by distance. ---*/
68  su2double thermal_conductivity = 0.0;
69  su2double heat_flux_density = 0.0;
70  su2double conductivity_over_dist = 0.0;
71 
72  const bool compressible_flow = (donor_config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE);
73  const bool incompressible_flow = (donor_config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
74  (donor_config->GetEnergy_Equation() || (donor_config->GetKind_FluidModel() == ENUM_FLUIDMODEL::FLUID_FLAMELET));
75 
76  if (compressible_flow) {
77 
78  const su2double Gamma = donor_config->GetGamma();
79  const su2double Gas_Constant = donor_config->GetGas_ConstantND();
80  const su2double Cp = (Gamma / (Gamma - 1.0)) * Gas_Constant;
81 
82  const su2double Prandtl_Lam = donor_config->GetPrandtl_Lam();
83  const su2double laminar_viscosity = donor_solution->GetNodes()->GetLaminarViscosity(Point_Donor); // TDE check for consistency
84 
85  const su2double thermal_conductivityND = Cp*(laminar_viscosity/Prandtl_Lam);
86  heat_flux_density = thermal_conductivityND*dTdn;
87 
88  if ((donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::DIRECT_TEMPERATURE_ROBIN_HEATFLUX) ||
89  (donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {
90 
91  thermal_conductivity = thermal_conductivityND*donor_config->GetViscosity_Ref();
92  conductivity_over_dist = thermal_conductivity/dist;
93  }
94  }
95  else if (incompressible_flow) {
96 
97  const auto iPoint = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNode();
98 
99  const su2double thermal_conductivityND = donor_solution->GetNodes()->GetThermalConductivity(iPoint);
100  heat_flux_density = thermal_conductivityND*dTdn;
101 
102  if ((donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::DIRECT_TEMPERATURE_ROBIN_HEATFLUX) ||
103  (donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {
104 
105  switch (donor_config->GetKind_ConductivityModel()) {
106 
107  case CONDUCTIVITYMODEL::CONSTANT:
108  case CONDUCTIVITYMODEL::FLAMELET:
109  case CONDUCTIVITYMODEL::COOLPROP:
110  thermal_conductivity = thermal_conductivityND*donor_config->GetThermal_Conductivity_Ref();
111  break;
112  case CONDUCTIVITYMODEL::CONSTANT_PRANDTL:
113  thermal_conductivity = thermal_conductivityND*donor_config->GetGas_Constant_Ref()
114  *donor_config->GetViscosity_Ref();
115  break;
116  case CONDUCTIVITYMODEL::POLYNOMIAL:
117  SU2_MPI::Error("Polynomial Conductivity model not implemented for CHT interface.", CURRENT_FUNCTION);
118  break;
119  }
120 
121  conductivity_over_dist = thermal_conductivity/dist;
122  }
123  }
124  else if (donor_config->GetHeatProblem()) {
125 
126  /*--- Heat solver stand-alone case ---*/
127 
128  const su2double thermal_diffusivity = donor_config->GetThermalDiffusivity();
129  heat_flux_density = thermal_diffusivity*dTdn;
130 
131 
132  if ((donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::DIRECT_TEMPERATURE_ROBIN_HEATFLUX) ||
133  (donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {
134 
135  /*--- Apply contact resistance to solid-to-solid heat transfer boundary ---*/
136  const su2double rho_cp_solid = donor_config->GetSpecific_Heat_Cp()*donor_config->GetMaterialDensity(0);
137  thermal_conductivity = thermal_diffusivity * rho_cp_solid;
138  conductivity_over_dist = thermal_conductivity/(dist + thermal_conductivity * ContactResistance);
139  }
140  }
141 
142  /*--- Set the conjugate heat variables that are transferred by default ---*/
143 
144  Donor_Variable[0] = Twall*donor_config->GetTemperature_Ref();
145  Donor_Variable[1] = heat_flux_density*donor_config->GetHeat_Flux_Ref();
146 
147  /*--- We only need these for the Robin BC option ---*/
148 
149  if ((donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::DIRECT_TEMPERATURE_ROBIN_HEATFLUX) ||
150  (donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {
151 
152  Donor_Variable[2] = conductivity_over_dist;
153  Donor_Variable[3] = Tnormal*donor_config->GetTemperature_Ref();
154  }
155 
156 }

169

172  target_solution->SetConjugateHeatVariable(Marker_Target, Vertex_Target, 3,
173  target_config->GetRelaxation_Factor_CHT(), Target_Variable[3]);
174  }
