36 CRadP1Solver::CRadP1Solver(CGeometry* geometry, CConfig *config) : CRadSolver(geometry, config) {
37 
38  unsigned short iVar;
39  unsigned short direct_diff = config->GetDirectDiff();
40  bool multizone = config->GetMultizone_Problem();
41 
42  nDim = geometry->GetnDim();
43  nPoint = geometry->GetnPoint();
44  nPointDomain = geometry->GetnPointDomain();
45  nVar = 1;
46 
47  /*--- Initialize nVarGrad for deallocation ---*/
48 
49  nVarGrad = nVar;
50 
51  Residual = new su2double[nVar];
52  Solution = new su2double[nVar];
53 
54  Res_Visc = new su2double[nVar];
55 
56  /*--- Define some structures for locating max residuals ---*/
57 
58  Residual_RMS.resize(nVar,0.0);
59  Residual_Max.resize(nVar,0.0);
60  Point_Max.resize(nVar,0);
61  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
62 
63  /*--- Jacobians and vector structures for implicit computations ---*/
64 
65  if (config->GetKind_TimeIntScheme_Radiation() == EULER_IMPLICIT) {
66 
67  Jacobian_i = new su2double* [nVar];
68  Jacobian_j = new su2double* [nVar];
69  for (iVar = 0; iVar < nVar; iVar++) {
70  Jacobian_i[iVar] = new su2double [nVar];
71  Jacobian_j[iVar] = new su2double [nVar];
72  }
73 
74  if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (P1 radiation equation)." << endl;
75  Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config);
76 
77  }
78 
79  /*--- Solution and residual vectors ---*/
80 
81  LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
82  LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
83 
84  /*--- Read farfield conditions from config ---*/
85  Temperature_Inf = config->GetTemperature_FreeStreamND();
86 
87  /*--- Initialize the secondary values for direct derivative approxiations ---*/
88 
89  switch(direct_diff){
90  case NO_DERIVATIVE: case D_DENSITY:
91  case D_PRESSURE: case D_VISCOSITY:
92  case D_MACH: case D_AOA:
93  case D_SIDESLIP: case D_REYNOLDS:
94  case D_TURB2LAM: case D_DESIGN:
95  /*--- Not necessary here ---*/
96  break;
97  case D_TEMPERATURE:
98  SU2_TYPE::SetDerivative(Temperature_Inf, 1.0);
99  break;
100  default:
101  break;
102  }
103 
104  SetTemperature_Inf(Temperature_Inf);
105 
106  /*--- Initialize the BGS residuals. ---*/
107  if (multizone){
108  Residual_BGS.resize(nVar,1.0);
109  Residual_Max_BGS.resize(nVar,1.0);
110  Point_Max_BGS.resize(nVar,0);
111  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
112  }
113 
114  /*--- Always instantiate and initialize the variable to a zero value. ---*/
115  su2double init_val;
116  switch(config->GetKind_P1_Init()){
117  case P1_INIT::ZERO: init_val = 0.0; break;
118  case P1_INIT::TEMPERATURE: init_val = 4.0*STEFAN_BOLTZMANN*pow(config->GetInc_Temperature_Init(),4.0); break;
119  default: init_val = 0.0; break;
120  }
121 
122  nodes = new CRadP1Variable(init_val, nPoint, nDim, nVar, config);
123  SetBaseClassPointerToNodes();
124 
125  /*--- Initialize the structure for volumetric heat source ---*/
126  if(config->GetHeatSource()){
127  SetVolumetricHeatSource(geometry, config);
128  }
129 
130  SolverName = "RAD";
131 }

162 void CRadP1Solver::Postprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh) {
163 
164  unsigned long iPoint;
165  su2double Energy, Temperature;
166  su2double SourceTerm, SourceTerm_Derivative;
167 
168  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
169 
170  /*--- Retrieve the radiative energy ---*/
171  Energy = nodes->GetSolution(iPoint, 0);
172 
173  /*--- Retrieve temperature from the flow solver ---*/
174  Temperature = solver_container[FLOW_SOL]->GetNodes()->GetTemperature(iPoint);
175 
176  /*--- Compute the divergence of the radiative flux ---*/
177  SourceTerm = Absorption_Coeff*(Energy - 4.0*STEFAN_BOLTZMANN*pow(Temperature,4.0));
178 
179  /*--- Compute the derivative of the source term with respect to the temperature ---*/
180  SourceTerm_Derivative = - 16.0*Absorption_Coeff*STEFAN_BOLTZMANN*pow(Temperature,3.0);
181 
182  /*--- Store the source term and its derivative ---*/
183  nodes->SetRadiative_SourceTerm(iPoint, 0, SourceTerm);
184  nodes->SetRadiative_SourceTerm(iPoint, 1, SourceTerm_Derivative);
185 
186  }

190 void CRadP1Solver::Viscous_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
191  CConfig *config, unsigned short iMesh, unsigned short iRKStep) {
192 
193  CNumerics* numerics = numerics_container[VISC_TERM];
194 
195  unsigned long iEdge, iPoint, jPoint;
196 
197  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
198 
199  /*--- Points in edge ---*/
200 
201  iPoint = geometry->edges->GetNode(iEdge,0);
202  jPoint = geometry->edges->GetNode(iEdge,1);
203 
204  /*--- Points coordinates, and normal vector ---*/
205 
206  numerics->SetCoord(geometry->nodes->GetCoord(iPoint),
207  geometry->nodes->GetCoord(jPoint));
208  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
209 
210  /*--- Radiation variables w/o reconstruction, and its gradients ---*/
211 
212  numerics->SetRadVar(nodes->GetSolution(iPoint), nodes->GetSolution(jPoint));
213  numerics->SetRadVarGradient(nodes->GetGradient(iPoint), nodes->GetGradient(jPoint));
214 
215  /*--- Compute residual, and Jacobians ---*/
216 
217  numerics->ComputeResidual(Residual, Jacobian_i, Jacobian_j, config);
218 
219  /*--- Add and subtract residual, and update Jacobian ---*/
220 
221  LinSysRes.SubtractBlock(iPoint, Residual);
222  LinSysRes.AddBlock(jPoint, Residual);
223  Jacobian.UpdateBlocksSub(iEdge, iPoint, jPoint, Jacobian_i, Jacobian_j);
224 
225  }

261 void CRadP1Solver::BC_Isothermal_Wall(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics, CNumerics *visc_numerics, CConfig *config,
262  unsigned short val_marker) {
263 
264  unsigned short iVar, jVar;
265  unsigned long iVertex, iPoint;
266 
267  su2double Theta, Ib_w, Radiative_Energy;
268  su2double *Normal, Area, Wall_Emissivity;
269  su2double Radiative_Heat_Flux;
270  su2double Twall;
271 
272  bool implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
273 
274  /*--- Identify the boundary by string name ---*/
275  string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
276 
277  /*--- Get the specified wall emissivity from config ---*/
278  Wall_Emissivity = config->GetWall_Emissivity(Marker_Tag);
279 
280  /*--- Compute the constant for the wall theta ---*/
281  Theta = Wall_Emissivity / (2.0*(2.0 - Wall_Emissivity));
282 
283  /*--- Retrieve the specified wall temperature ---*/
284  Twall = config->GetIsothermal_Temperature(Marker_Tag)/config->GetTemperature_Ref();
285 
286  /*--- Loop over all of the vertices on this boundary marker ---*/
287 
288  for (iVertex = 0; iVertex < geometry->nVertex[val_marker]; iVertex++) {
289  iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
290 
291  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
292 
293  if (geometry->nodes->GetDomain(iPoint)) {
294 
295  /*--- Compute dual-grid area and boundary normal ---*/
296  Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
297 
298  Area = GeometryToolbox::Norm(nDim, Normal);
299 
300  // Weak application of the boundary condition
301 
302  /*--- Initialize the viscous residuals to zero ---*/
303  for (iVar = 0; iVar < nVar; iVar++) {
304  Res_Visc[iVar] = 0.0;
305  if (implicit) {
306  for (jVar = 0; jVar < nVar; jVar++)
307  Jacobian_i[iVar][jVar] = 0.0;
308  }
309  }
310 
311  /*--- Apply a weak boundary condition for the radiative transfer equation. ---*/
312 
313  /*--- Compute the blackbody intensity at the wall. ---*/
314  Ib_w = 4.0*STEFAN_BOLTZMANN*pow(Twall,4.0);
315 
316  /*--- Compute the radiative heat flux. ---*/
317  Radiative_Energy = nodes->GetSolution(iPoint, 0);
318  Radiative_Heat_Flux = 1.0*Theta*(Ib_w - Radiative_Energy);
319 
320  /*--- Compute the Viscous contribution to the residual ---*/
321  Res_Visc[0] = Radiative_Heat_Flux*Area;
322 
323  /*--- Apply to the residual vector ---*/
324  LinSysRes.SubtractBlock(iPoint, Res_Visc);
325 
326  /*--- Compute the Jacobian contribution. ---*/
327  if (implicit) {
328  Jacobian_i[0][0] = - Theta;
329  Jacobian.SubtractBlock2Diag(iPoint, Jacobian_i);
330  }
331  }
332  }

486 void CRadP1Solver::ImplicitEuler_Iteration(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
487 
488  unsigned short iVar;
489  unsigned long iPoint, total_index, IterLinSol = 0;
490  su2double Vol;
491  su2double Delta;
492 
493  /*--- Set maximum residual to zero ---*/
494 
495  SetResToZero();
496 
497  /*--- Build implicit system ---*/
498 
499  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
500 
501  /*--- Read the volume ---*/
502 
503  Vol = geometry->nodes->GetVolume(iPoint);
504 
505  /*--- Modify matrix diagonal to assure diagonal dominance ---*/
506 
507  if (nodes->GetDelta_Time(iPoint) != 0.0) {
508  Delta = Vol / nodes->GetDelta_Time(iPoint);
509  Jacobian.AddVal2Diag(iPoint, Delta);
510  }
511  else {
512  Jacobian.SetVal2Diag(iPoint, 1.0);
513  for (iVar = 0; iVar < nVar; iVar++) {
514  total_index = iPoint*nVar + iVar;
515  LinSysRes[total_index] = 0.0;
516  }
517  }
518 
519  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
520 
521  for (iVar = 0; iVar < nVar; iVar++) {
522  total_index = iPoint*nVar+iVar;
523  LinSysRes[total_index] = - (LinSysRes[total_index]);
524  LinSysSol[total_index] = 0.0;
525  Residual_RMS[iVar] += LinSysRes[total_index]*LinSysRes[total_index];
526  AddRes_Max(iVar, fabs(LinSysRes[total_index]), geometry->nodes->GetGlobalIndex(iPoint), geometry->nodes->GetCoord(iPoint));
527  }
528  }
529 
530  /*--- Initialize residual and solution at the ghost points ---*/
531 
532  for (iPoint = nPointDomain; iPoint < nPoint; iPoint++) {
533  for (iVar = 0; iVar < nVar; iVar++) {
534  total_index = iPoint*nVar + iVar;
535  LinSysRes[total_index] = 0.0;
536  LinSysSol[total_index] = 0.0;
537  }
538  }
539 
540  /*--- Solve or smooth the linear system ---*/
541 
542  IterLinSol = System.Solve(Jacobian, LinSysRes, LinSysSol, geometry, config);
543 
544  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
545  for (iVar = 0; iVar < nVar; iVar++) {
546  nodes->AddSolution(iPoint, iVar, LinSysSol[iPoint*nVar+iVar]);
547  }
548  }
549 
550  /*--- The the number of iterations of the linear solver ---*/
551 
552  SetIterLinSolver(IterLinSol);
553 
554  /*--- MPI solution ---*/
555 
556  InitiateComms(geometry, config, MPI_QUANTITIES::SOLUTION);
557  CompleteComms(geometry, config, MPI_QUANTITIES::SOLUTION);
558 
559  /*--- Compute the root mean square residual ---*/
560 
561  SetResidual_RMS(geometry, config);
562 
563 }

565 void CRadP1Solver::SetTime_Step(CGeometry *geometry, CSolver **solver_container, CConfig *config,
566  unsigned short iMesh, unsigned long Iteration) {
567 
568  unsigned short iMarker;
569  unsigned long iEdge, iVertex, iPoint = 0, jPoint = 0;
570  su2double Area, Vol, Lambda;
571  su2double Global_Delta_Time = 1E6, Local_Delta_Time = 0.0, K_v = 0.25;
572  su2double CFL = config->GetCFL_Rad();
573  su2double GammaP1 = 1.0 / (3.0*(Absorption_Coeff + Scattering_Coeff));
574  const su2double* Normal;
575 
576  /*--- Compute spectral radius based on thermal conductivity ---*/
577 
578  Min_Delta_Time = 1.E6; Max_Delta_Time = 0.0;
579 
580  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
581  nodes->SetMax_Lambda_Visc(iPoint, 0.0);
582  }
583 
584  /*--- Loop interior edges ---*/
585 
586  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
587 
588  iPoint = geometry->edges->GetNode(iEdge,0);
589  jPoint = geometry->edges->GetNode(iEdge,1);
590 
591  /*--- Get the edge's normal vector to compute the edge's area ---*/
592  Normal = geometry->edges->GetNormal(iEdge);
593  Area = GeometryToolbox::Norm(nDim, Normal);
594 
595  /*--- Viscous contribution ---*/
596 
597  Lambda = GammaP1*Area*Area;
598  if (geometry->nodes->GetDomain(iPoint)) nodes->AddMax_Lambda_Visc(iPoint, Lambda);
599  if (geometry->nodes->GetDomain(jPoint)) nodes->AddMax_Lambda_Visc(jPoint, Lambda);
600 
601  }
602 
603  /*--- Loop boundary edges ---*/
604 
605  for (iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
606  for (iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
607 
608  /*--- Point identification, Normal vector and area ---*/
609 
610  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
611  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
612  Area = GeometryToolbox::Norm(nDim, Normal);
613 
614  /*--- Viscous contribution ---*/
615 
616  Lambda = GammaP1*Area*Area;
617  if (geometry->nodes->GetDomain(iPoint)) nodes->AddMax_Lambda_Visc(iPoint, Lambda);
618 
619  }
620  }
621 
622  /*--- Each element uses their own speed, steady state simulation ---*/
623 
624  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
625 
626  Vol = geometry->nodes->GetVolume(iPoint);
627 
628  if (Vol != 0.0) {
629 
630  /*--- Time step setting method ---*/
631 
632  Local_Delta_Time = CFL*K_v*Vol*Vol/ nodes->GetMax_Lambda_Visc(iPoint);
633 
634  /*--- Min-Max-Logic ---*/
635 
636  Global_Delta_Time = min(Global_Delta_Time, Local_Delta_Time);
637  Min_Delta_Time = min(Min_Delta_Time, Local_Delta_Time);
638  Max_Delta_Time = max(Max_Delta_Time, Local_Delta_Time);
639  if (Local_Delta_Time > config->GetMax_DeltaTime())
640  Local_Delta_Time = config->GetMax_DeltaTime();
641 
642  nodes->SetDelta_Time(iPoint, Local_Delta_Time);
643  }
644  else {
645  nodes->SetDelta_Time(iPoint, 0.0);
646  }
647  }
648 
649  /*--- Compute the max and the min dt (in parallel) ---*/
650  if (config->GetComm_Level() == COMM_FULL) {
651 
652  su2double sbuf_time;
653  sbuf_time = Min_Delta_Time;
654  SU2_MPI::Allreduce(&sbuf_time, &Min_Delta_Time, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
655 
656  sbuf_time = Max_Delta_Time;
657  SU2_MPI::Allreduce(&sbuf_time, &Max_Delta_Time, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
658  }
