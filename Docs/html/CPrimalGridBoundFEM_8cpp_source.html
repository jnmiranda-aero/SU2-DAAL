30CPrimalGridBoundFEM::CPrimalGridBoundFEM(unsigned long val_elemGlobalID, unsigned long val_domainElementID,
31 unsigned short val_VTK_Type, unsigned short val_nPolyGrid,
32 unsigned short val_nDOFsGrid, std::vector<unsigned long>& val_nodes)
33 : CPrimalGrid(true, val_nDOFsGrid, 1) {
34 /*--- Store the integer data in the member variables of this object. ---*/
35 VTK_Type = val_VTK_Type;
36
37 nPolyGrid = val_nPolyGrid;
38 nDOFsGrid = val_nDOFsGrid;
39
40 boundElemIDGlobal = val_elemGlobalID;
41 GlobalIndex_DomainElement = val_domainElementID;
42
43 /*--- Copy face structure of the element from val_nodes. ---*/
44
45 for (unsigned short i = 0; i < nDOFsGrid; i++) Nodes[i] = val_nodes[i];
46
47 /*--- For a linear quadrilateral the two last node numbers must be swapped,
48 such that the element numbering is consistent with the FEM solver. ---*/
49
50 if (nPolyGrid == 1 && VTK_Type == QUADRILATERAL) std::swap(Nodes[2], Nodes[3]);
51}

53void CPrimalGridBoundFEM::GetLocalCornerPointsFace(unsigned short elementType, unsigned short nPoly,
54 unsigned short nDOFs, unsigned short& nPointsPerFace,
55 unsigned long faceConn[]) {
56 /*--- Determine the element type and set the face data accordingly.
57 The faceConn values are local to the element. ---*/
58
59 switch (elementType) {
60 case LINE:
61 nPointsPerFace = 2;
62 faceConn[0] = 0;
63 faceConn[1] = nPoly;
64 break;
65
66 case TRIANGLE:
67 nPointsPerFace = 3;
68 faceConn[0] = 0;
69 faceConn[1] = nPoly;
70 faceConn[2] = nDOFs - 1;
71 break;
72
73 case QUADRILATERAL:
74 unsigned short nn2 = nPoly * (nPoly + 1);
75 nPointsPerFace = 4;
76 faceConn[0] = 0;
77 faceConn[1] = nPoly;
78 faceConn[2] = nDOFs - 1;
79 faceConn[3] = nn2;
80 break;
81 }
82}

84void CPrimalGridBoundFEM::GetCornerPointsAllFaces(unsigned short& nFaces, unsigned short nPointsPerFace[],
85 unsigned long faceConn[6][4]) const {
86 /*--- For compatibility reasons with the base class, nFaces is also an
87 argument to this function. However, it is always 1, because the
88 boundary element is the face. ---*/
89 nFaces = 1;
90
91 /*--- Get the corner points of the face local to the element. ---*/
92
93 unsigned long thisFaceConn[4] = {0, 0, 0, 0};
94 GetLocalCornerPointsFace(VTK_Type, nPolyGrid, nDOFsGrid, nPointsPerFace[0], thisFaceConn);
95
96 /*--- Convert the local values of thisFaceConn to global values. ---*/
97
98 for (unsigned short j = 0; j < nPointsPerFace[0]; ++j) faceConn[0][j] = Nodes[thisFaceConn[j]];
99}
