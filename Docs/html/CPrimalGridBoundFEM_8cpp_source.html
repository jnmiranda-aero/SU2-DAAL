53 void CPrimalGridBoundFEM::GetLocalCornerPointsFace(unsigned short elementType, unsigned short nPoly,
54  unsigned short nDOFs, unsigned short& nPointsPerFace,
55  unsigned long faceConn[]) {
56  /*--- Determine the element type and set the face data accordingly.
57  The faceConn values are local to the element. ---*/
58 
59  switch (elementType) {
60  case LINE:
61  nPointsPerFace = 2;
62  faceConn[0] = 0;
63  faceConn[1] = nPoly;
64  break;
65 
66  case TRIANGLE:
67  nPointsPerFace = 3;
68  faceConn[0] = 0;
69  faceConn[1] = nPoly;
70  faceConn[2] = nDOFs - 1;
71  break;
72 
73  case QUADRILATERAL:
74  unsigned short nn2 = nPoly * (nPoly + 1);
75  nPointsPerFace = 4;
76  faceConn[0] = 0;
77  faceConn[1] = nPoly;
78  faceConn[2] = nDOFs - 1;
79  faceConn[3] = nn2;
80  break;
81  }
82 }

84 void CPrimalGridBoundFEM::GetCornerPointsAllFaces(unsigned short& nFaces, unsigned short nPointsPerFace[],
85  unsigned long faceConn[6][4]) const {
86  /*--- For compatibility reasons with the base class, nFaces is also an
87  argument to this function. However, it is always 1, because the
88  boundary element is the face. ---*/
89  nFaces = 1;
90 
91  /*--- Get the corner points of the face local to the element. ---*/
92 
93  unsigned long thisFaceConn[4] = {0, 0, 0, 0};
94  GetLocalCornerPointsFace(VTK_Type, nPolyGrid, nDOFsGrid, nPointsPerFace[0], thisFaceConn);
95 
96  /*--- Convert the local values of thisFaceConn to global values. ---*/
97 
98  for (unsigned short j = 0; j < nPointsPerFace[0]; ++j) faceConn[0][j] = Nodes[thisFaceConn[j]];
99 }
