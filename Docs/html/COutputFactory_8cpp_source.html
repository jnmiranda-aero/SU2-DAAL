42 COutput* COutputFactory::CreateOutput(MAIN_SOLVER kindSolver, CConfig* config, int nDim){
43 
44  COutput* output = nullptr;
45 
46  switch(kindSolver){
47  case MAIN_SOLVER::EULER: case MAIN_SOLVER::NAVIER_STOKES: case MAIN_SOLVER::RANS:
48  output = new CFlowCompOutput(config, nDim);
49  break;
50  case MAIN_SOLVER::INC_EULER: case MAIN_SOLVER::INC_NAVIER_STOKES: case MAIN_SOLVER::INC_RANS:
51  output = new CFlowIncOutput(config, nDim);
52  break;
53  case MAIN_SOLVER::NEMO_EULER: case MAIN_SOLVER::NEMO_NAVIER_STOKES:
54  output = new CNEMOCompOutput(config, nDim);
55  break;
56  case MAIN_SOLVER::HEAT_EQUATION:
57  output = new CHeatOutput(config, nDim);
58  break;
59  case MAIN_SOLVER::FEM_ELASTICITY:
60  output = new CElasticityOutput(config, nDim);
61  break;
62  case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
63  case MAIN_SOLVER::ADJ_EULER: case MAIN_SOLVER::ADJ_NAVIER_STOKES: case MAIN_SOLVER::ADJ_RANS:
64  output = new CAdjFlowCompOutput(config, nDim);
65  break;
66  case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
67  output = new CAdjFlowIncOutput(config, nDim);
68  break;
69  case MAIN_SOLVER::DISC_ADJ_FEM:
70  output = new CAdjElasticityOutput(config, nDim);
71  break;
72  case MAIN_SOLVER::DISC_ADJ_HEAT:
73  output = new CAdjHeatOutput(config, nDim);
74  break;
75  case MAIN_SOLVER::FEM_EULER: case MAIN_SOLVER::FEM_LES: case MAIN_SOLVER::FEM_RANS: case MAIN_SOLVER::FEM_NAVIER_STOKES:
76  output = new CFlowCompFEMOutput(config, nDim);
77  break;
78  default:
79  output = new COutput(config, nDim, false);
80  break;
81  }
82 
83  return output;
84 }
