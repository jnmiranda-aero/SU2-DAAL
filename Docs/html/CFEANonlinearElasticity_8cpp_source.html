32CFEANonlinearElasticity::CFEANonlinearElasticity(unsigned short val_nDim, unsigned short val_nVar,
33 const CConfig *config) : CFEAElasticity(val_nDim, val_nVar, config) {
34
35 nearly_incompressible = (config->GetMaterialCompressibility() == STRUCT_COMPRESS::NEARLY_INCOMP);
36
37 unsigned short iVar;
38
39 F_Mat = new su2double *[3];
40 b_Mat = new su2double *[3];
41 FmT_Mat = new su2double *[3];
42 Stress_Tensor = new su2double *[3];
43 for (iVar = 0; iVar < 3; iVar++) {
44 F_Mat[iVar] = new su2double [3];
45 b_Mat[iVar] = new su2double [3];
46 FmT_Mat[iVar] = new su2double [3];
47 Stress_Tensor[iVar] = new su2double [3];
48 }
49
50 KAux_t_a = new su2double [nDim];
51
52 KAux_P_ab = new su2double* [nDim];
53 for (iVar = 0; iVar < nDim; iVar++) {
54 KAux_P_ab[iVar] = new su2double[nDim];
55 }
56
57 unsigned short nNodes = (nDim==2) ? NNODES_2D : NNODES_3D;
58
59 currentCoord = new su2double* [nNodes];
60 for (iVar = 0; iVar < nNodes; iVar++)
61 currentCoord[iVar] = new su2double[nDim];
62
63 J_F = 1.0; J_F_Iso = 1.0;
64 f33 = 1.0;
65
66 C10 = Mu/2.0;
67 D1 = 2.0/Kappa;
68
69 F_Mat_Iso = nullptr;
70 b_Mat_Iso = nullptr;
71
72 F_Mat_Iso = new su2double *[3];
73 b_Mat_Iso = new su2double *[3];
74 for (iVar = 0; iVar < 3; iVar++){
75 F_Mat_Iso[iVar] = new su2double [3];
76 b_Mat_Iso[iVar] = new su2double [3];
77 }
78
79 maxwell_stress = config->GetDE_Effects();
80
81 ke_DE = 0.0;
82 nElectric_Field = 0;
83 nDim_Electric_Field = 0;
84
85 EField_Ref_Unit = nullptr;
86 EField_Ref_Mod = nullptr;
87 EField_Curr_Unit = nullptr;
88
89 if (maxwell_stress) {
90
91 const su2double *Electric_Field_Dir = config->Get_Electric_Field_Dir();
92 unsigned short iVar, iDim;
93 su2double ref_Efield_mod;
94
95 // Initialize value of the electric field in the reference configuration
96 EFieldMod_Ref = 0.0;
97
98 ke_DE = config->GetElectric_Constant(0);
99
100 nDim_Electric_Field = config->GetnDim_Electric_Field();
101
102 if (nDim != nDim_Electric_Field) {
103 SU2_MPI::Error("The electric field dimensions do not agree with the geometry.", CURRENT_FUNCTION);
104 }
105
106 /*--- DV_Val: Vector to store the value of the design variable. ---*/
107
108 nElectric_Field = config->GetnElectric_Field();
109
110 /*--- We initialize the modulus ---*/
111 ref_Efield_mod = 0.0;
112 /*--- Normalize the electric field vector ---*/
113 for (iDim = 0; iDim < nDim_Electric_Field; iDim++) {
114 ref_Efield_mod += Electric_Field_Dir[iDim]*Electric_Field_Dir[iDim];
115 }
116 ref_Efield_mod = sqrt(ref_Efield_mod);
117
118 if (ref_Efield_mod == 0) {
119 SU2_MPI::Error("The electric field has not been defined.", CURRENT_FUNCTION);
120 }
121
122 /*--- Initialize pointer for the electric field ---*/
123 EField_Ref_Unit = new su2double[nDim_Electric_Field];
124 /*--- Assign values to the auxiliary Electric_Field structure ---*/
125 for (iDim = 0; iDim < nDim_Electric_Field; iDim++) {
126 EField_Ref_Unit[iDim] = Electric_Field_Dir[iDim]/ref_Efield_mod;
127 }
128
129 /*--- Auxiliary vector for hosting the electric field modulus in the reference configuration ---*/
130 EField_Ref_Mod = new su2double[nElectric_Field];
131 for (iVar = 0; iVar < nElectric_Field; iVar++)
132 EField_Ref_Mod[iVar] = config->Get_Electric_Field_Mod(iVar);
133
134 /*--- Auxiliary vector for computing the electric field in the current configuration ---*/
135 EField_Curr_Unit = new su2double[nDim_Electric_Field];
136 for (iDim = 0; iDim < nDim_Electric_Field; iDim++) {
137 EField_Curr_Unit[iDim] = 0.0;
138 }
139
140 /*--- Auxiliary vector for storing the electric field constant ---*/
141 unsigned short nElectric_Constant = config->GetnElectric_Constant();
142 ke_DE_i = new su2double[nElectric_Field];
143
144 if (nElectric_Constant == nElectric_Field)
145 for (iVar = 0; iVar < nElectric_Field; iVar++) ke_DE_i[iVar] = config->GetElectric_Constant(iVar);
146 else
147 for (iVar = 0; iVar < nElectric_Field; iVar++) ke_DE_i[iVar] = config->GetElectric_Constant(0);
148
149 switch (config->GetDV_FEA()) {
150 case YOUNG_MODULUS:
151 case POISSON_RATIO:
152 case DENSITY_VAL:
153 case DEAD_WEIGHT:
154 case ELECTRIC_FIELD:
155 break;
156 default:
157 switch (config->GetDirectDiff()){
158 case D_EFIELD:
159 SU2_TYPE::SetDerivative(EField_Ref_Mod[config->GetnID_DV()],1.0);
160 break;
161 default:
162 break;
163 }
164 break;
165 }
166
167 }

223void CFEANonlinearElasticity::Compute_Tangent_Matrix(CElement *element, const CConfig *config) {
224
225 unsigned short iVar, jVar, kVar;
226 unsigned short iGauss, nGauss;
227 unsigned short iNode, jNode, nNode;
228 unsigned short iDim, bDim;
229
230 su2double Ks_Aux_ab;
231
232 su2double Weight, Jac_x;
233
234 su2double AuxMatrixKc[3][6];
235 su2double AuxMatrixKs[3];
236
237 /*--- TODO: Initialize values for the material model considered ---*/
238// cout << "PROPERTY: " << element->Get_iProp() << " and DV " << element->Get_iDV() << endl;
239 SetElement_Properties(element, config);
240 if (maxwell_stress) SetElectric_Properties(element, config);
241
242 /*--- Register pre-accumulation inputs, material props and nodal coords ---*/
243 AD::StartPreacc();
244 AD::SetPreaccIn(E);
245 AD::SetPreaccIn(Nu);
246 if (maxwell_stress) {
247 AD::SetPreaccIn(EFieldMod_Ref);
248 AD::SetPreaccIn(ke_DE);
249 }
250 element->SetPreaccIn_Coords();
251 /*--- Recompute Lame parameters as they depend on the material properties ---*/
252 Compute_Lame_Parameters();
253
254 /*-----------------------------------------------------------*/
255
256 /*--- Initialize auxiliary matrices ---*/
257
258 bDim = (nDim == 2) ? DIM_STRAIN_2D : DIM_STRAIN_3D;
259
260 for (iVar = 0; iVar < bDim; iVar++) {
261 for (jVar = 0; jVar < nDim; jVar++) {
262 Ba_Mat[iVar][jVar] = 0.0;
263 Bb_Mat[iVar][jVar] = 0.0;
264 }
265 }
266
267 for (iVar = 0; iVar < 3; iVar++) {
268 for (jVar = 0; jVar < 6; jVar++) {
269 AuxMatrixKc[iVar][jVar] = 0.0;
270 }
271 }
272
273 for (iVar = 0; iVar < 3; iVar++) {
274 AuxMatrixKs[iVar] = 0.0;
275 }
276
277 element->ClearElement(); /*--- Restarts the element to avoid adding over previous results. --*/
278 element->ComputeGrad_Linear();
279 element->ComputeGrad_NonLinear();
280
281 nNode = element->GetnNodes();
282 nGauss = element->GetnGaussPoints();
283
284 /*--- Full integration of the constitutive and stress term ---*/
285
286 for (iGauss = 0; iGauss < nGauss; iGauss++) {
287
288 Weight = element->GetWeight(iGauss);
289 Jac_x = element->GetJ_x(iGauss);
290
291 /*--- Initialize the deformation gradient for each Gauss Point ---*/
292
293 for (iVar = 0; iVar < 3; iVar++) {
294 for (jVar = 0; jVar < 3; jVar++) {
295 F_Mat[iVar][jVar] = 0.0;
296 b_Mat[iVar][jVar] = 0.0;
297 }
298 }
299
300 /*--- Retrieve the values of the gradients of the shape functions for each node ---*/
301 /*--- This avoids repeated operations ---*/
302
303 for (iNode = 0; iNode < nNode; iNode++) {
304
305 for (iDim = 0; iDim < nDim; iDim++) {
306 GradNi_Ref_Mat[iNode][iDim] = element->GetGradNi_X(iNode,iGauss,iDim);
307 GradNi_Curr_Mat[iNode][iDim] = element->GetGradNi_x(iNode,iGauss,iDim);
308 currentCoord[iNode][iDim] = element->GetCurr_Coord(iNode, iDim);
309 }
310
311 /*--- Compute the deformation gradient ---*/
312
313 for (iVar = 0; iVar < nDim; iVar++) {
314 for (jVar = 0; jVar < nDim; jVar++) {
315 F_Mat[iVar][jVar] += currentCoord[iNode][iVar]*GradNi_Ref_Mat[iNode][jVar];
316 }
317 }
318 }
319
320 if (nDim == 2) {
321 if (plane_stress) {
322 // Compute the value of the term 33 for the deformation gradient
323 Compute_Plane_Stress_Term(element, config);
324 F_Mat[2][2] = f33;
325 }
326 else { // plane strain
327 F_Mat[2][2] = 1.0;
328 }
329 }
330
331 /*--- Determinant of F --> Jacobian of the transformation ---*/
332
333 J_F = F_Mat[0][0]*F_Mat[1][1]*F_Mat[2][2]+
334 F_Mat[0][1]*F_Mat[1][2]*F_Mat[2][0]+
335 F_Mat[0][2]*F_Mat[1][0]*F_Mat[2][1]-
336 F_Mat[0][2]*F_Mat[1][1]*F_Mat[2][0]-
337 F_Mat[1][2]*F_Mat[2][1]*F_Mat[0][0]-
338 F_Mat[2][2]*F_Mat[0][1]*F_Mat[1][0];
339
340 /*--- Compute the left Cauchy deformation tensor ---*/
341
342 for (iVar = 0; iVar < 3; iVar++) {
343 for (jVar = 0; jVar < 3; jVar++) {
344 for (kVar = 0; kVar < 3; kVar++) {
345 b_Mat[iVar][jVar] += F_Mat[iVar][kVar]*F_Mat[jVar][kVar];
346 }
347 }
348 }
349
350 /*--- Compute the constitutive matrix ---*/
351
352 Compute_Stress_Tensor(element, config);
353// if (maxwell_stress) Add_MaxwellStress(element, config);
354 Compute_Constitutive_Matrix(element, config);
355
356
357 for (iNode = 0; iNode < nNode; iNode++) {
358
359 /*--------------------------------------------------------------------------------*/
360 /*---------------------------- NODAL STRESS TERM ---------------------------------*/
361 /*--------------------------------------------------------------------------------*/
362 /*--- Compute the nodal stress term for each gaussian point and for each node, ---*/
363 /*--- and add it to the element structure to be retrieved from the solver ---*/
364
365 for (iVar = 0; iVar < nDim; iVar++) {
366 KAux_t_a[iVar] = 0.0;
367 for (jVar = 0; jVar < nDim; jVar++) {
368 KAux_t_a[iVar] += Weight * Stress_Tensor[iVar][jVar] * GradNi_Curr_Mat[iNode][jVar] * Jac_x;
369 }
370 }
371
372 element->Add_Kt_a(iNode, KAux_t_a);
373
374 /*--------------------------------------------------------------------------------*/
375 /*----------------------- CONSTITUTIVE AND STRESS TERM ---------------------------*/
376 /*--------------------------------------------------------------------------------*/
377
378 if (nDim == 2) {
379 Ba_Mat[0][0] = GradNi_Curr_Mat[iNode][0];
380 Ba_Mat[1][1] = GradNi_Curr_Mat[iNode][1];
381 Ba_Mat[2][0] = GradNi_Curr_Mat[iNode][1];
382 Ba_Mat[2][1] = GradNi_Curr_Mat[iNode][0];
383 }
384 else if (nDim ==3) {
385 Ba_Mat[0][0] = GradNi_Curr_Mat[iNode][0];
386 Ba_Mat[1][1] = GradNi_Curr_Mat[iNode][1];
387 Ba_Mat[2][2] = GradNi_Curr_Mat[iNode][2];
388 Ba_Mat[3][0] = GradNi_Curr_Mat[iNode][1];
389 Ba_Mat[3][1] = GradNi_Curr_Mat[iNode][0];
390 Ba_Mat[4][0] = GradNi_Curr_Mat[iNode][2];
391 Ba_Mat[4][2] = GradNi_Curr_Mat[iNode][0];
392 Ba_Mat[5][1] = GradNi_Curr_Mat[iNode][2];
393 Ba_Mat[5][2] = GradNi_Curr_Mat[iNode][1];
394 }
395
396 /*--- Compute the BT.D Matrix ---*/
397
398 for (iVar = 0; iVar < nDim; iVar++) {
399 for (jVar = 0; jVar < bDim; jVar++) {
400 AuxMatrixKc[iVar][jVar] = 0.0;
401 for (kVar = 0; kVar < bDim; kVar++) {
402 AuxMatrixKc[iVar][jVar] += Ba_Mat[kVar][iVar]*D_Mat[kVar][jVar];
403 }
404 }
405 }
406
407 /*--- Compute the BT.D Matrix ---*/
408
409 for (iVar = 0; iVar < nDim; iVar++) {
410 AuxMatrixKs[iVar] = 0.0;
411 for (jVar = 0; jVar < nDim; jVar++) {
412 AuxMatrixKs[iVar] += GradNi_Curr_Mat[iNode][jVar]*Stress_Tensor[jVar][iVar]; // DOUBLE CHECK
413 }
414 }
415
416 /*--- Assumming symmetry ---*/
417 for (jNode = iNode; jNode < nNode; jNode++) {
418 if (nDim == 2) {
419 Bb_Mat[0][0] = GradNi_Curr_Mat[jNode][0];
420 Bb_Mat[1][1] = GradNi_Curr_Mat[jNode][1];
421 Bb_Mat[2][0] = GradNi_Curr_Mat[jNode][1];
422 Bb_Mat[2][1] = GradNi_Curr_Mat[jNode][0];
423 }
424 else if (nDim ==3) {
425 Bb_Mat[0][0] = GradNi_Curr_Mat[jNode][0];
426 Bb_Mat[1][1] = GradNi_Curr_Mat[jNode][1];
427 Bb_Mat[2][2] = GradNi_Curr_Mat[jNode][2];
428 Bb_Mat[3][0] = GradNi_Curr_Mat[jNode][1];
429 Bb_Mat[3][1] = GradNi_Curr_Mat[jNode][0];
430 Bb_Mat[4][0] = GradNi_Curr_Mat[jNode][2];
431 Bb_Mat[4][2] = GradNi_Curr_Mat[jNode][0];
432 Bb_Mat[5][1] = GradNi_Curr_Mat[jNode][2];
433 Bb_Mat[5][2] = GradNi_Curr_Mat[jNode][1];
434 }
435
436 /*--- KAux_ab is the term for the constitutive part of the tangent matrix ---*/
437 for (iVar = 0; iVar < nDim; iVar++) {
438 for (jVar = 0; jVar < nDim; jVar++) {
439 KAux_ab[iVar][jVar] = 0.0;
440 for (kVar = 0; kVar < bDim; kVar++) {
441 KAux_ab[iVar][jVar] += Weight * AuxMatrixKc[iVar][kVar] * Bb_Mat[kVar][jVar] * Jac_x;
442 }
443 }
444 }
445
446 /*--- Ks_Aux_ab is the term for the constitutive part of the tangent matrix ---*/
447 Ks_Aux_ab = 0.0;
448 for (iVar = 0; iVar < nDim; iVar++) {
449 Ks_Aux_ab += Weight * AuxMatrixKs[iVar] * GradNi_Curr_Mat[jNode][iVar] * Jac_x;
450 }
451
452 element->Add_Kab(iNode, jNode, KAux_ab);
453 element->Add_Ks_ab(iNode, jNode, Ks_Aux_ab);
454 /*--- Symmetric terms --*/
455 if (iNode != jNode) {
456 element->Add_Kab_T(jNode, iNode, KAux_ab);
457 element->Add_Ks_ab(jNode, iNode, Ks_Aux_ab);
458 }
459
460 }
461
462 }
463
464 }
465
466 /*--- Register the stress residual as preaccumulation output ---*/
467 element->SetPreaccOut_Kt_a();
468 AD::EndPreacc();
469
470}

601void CFEANonlinearElasticity::Add_MaxwellStress(CElement *element, const CConfig *config) {
602
603// Adds the Maxwell stress to the output of the stress Sxx, Syy, Szz, SVM...
604
605 unsigned short iDim, jDim;
606
607 su2double E0 = 0.0, E1 = 0.0, E2 = 0.0;
608 su2double E0_2 = 0.0, E1_2 = 0.0, E2_2 = 0.0;
609 su2double E_2 = 0.0;
610
611 Compute_FmT_Mat();
612
613 for (iDim = 0; iDim < nDim; iDim++){
614 EField_Curr_Unit[iDim] = 0.0;
615 for (jDim = 0; jDim < nDim; jDim++){
616 EField_Curr_Unit[iDim] += FmT_Mat[iDim][jDim] * EField_Ref_Unit[jDim];
617 }
618 }
619
620 E0 = EFieldMod_Ref*EField_Curr_Unit[0]; E0_2 = pow(E0,2);
621 E1 = EFieldMod_Ref*EField_Curr_Unit[1]; E1_2 = pow(E1,2);

623

625
626 Stress_Tensor[0][0] += ke_DE*(E0_2-0.5*E_2); Stress_Tensor[0][1] += ke_DE*E0*E1; Stress_Tensor[0][2] += ke_DE*E0*E2;
627 Stress_Tensor[1][0] += ke_DE*E1*E0; Stress_Tensor[1][1] += ke_DE*(E1_2-0.5*E_2); Stress_Tensor[1][2] += ke_DE*E1*E2;
628 Stress_Tensor[2][0] += ke_DE*E2*E0; Stress_Tensor[2][1] += ke_DE*E2*E1; Stress_Tensor[2][2] += ke_DE*(E2_2-0.5*E_2);

747su2double CFEANonlinearElasticity::Compute_Averaged_NodalStress(CElement *element, const CConfig *config) {
748
749 unsigned short iVar, jVar, kVar;
750 unsigned short iGauss, nGauss;
751 unsigned short iDim, iNode, nNode;
752
753 su2double avgStress[DIM_STRAIN_3D] = {0.0};
754
755 /*--- TODO: Initialize values for the material model considered ---*/
756 SetElement_Properties(element, config);
757 if (maxwell_stress) SetElectric_Properties(element, config);
758 /*-----------------------------------------------------------*/
759
760 /*--- Register pre-accumulation inputs ---*/
761 /*--- WARNING: Outputs must be registered outside of this method, this allows more
762 * flexibility in selecting what is captured by AD, capturing the entire stress
763 * tensor would use more memory than that used by the stress residuals. ---*/
764 AD::StartPreacc();
765 AD::SetPreaccIn(E);
766 AD::SetPreaccIn(Nu);
767 if (maxwell_stress) {
768 AD::SetPreaccIn(EFieldMod_Ref);
769 AD::SetPreaccIn(ke_DE);
770 }
771 element->SetPreaccIn_Coords();
772 /*--- Recompute Lame parameters as they depend on the material properties ---*/
773 Compute_Lame_Parameters();
774
775 su2double Weight, Jac_x;
776
777 element->ClearStress();
778 element->ClearElement(); /*--- Restarts the element to avoid adding over previous results. ---*/
779 element->ComputeGrad_Linear();
780 element->ComputeGrad_NonLinear();
781
782 nNode = element->GetnNodes();
783 nGauss = element->GetnGaussPoints();
784
785 /*--- Computation of the deformation gradient ---*/
786
787 for (iGauss = 0; iGauss < nGauss; iGauss++) {
788
789 Weight = element->GetWeight(iGauss);
790 Jac_x = element->GetJ_x(iGauss);
791
792 /*--- Initialize the deformation gradient for each Gauss Point ---*/
793
794 for (iVar = 0; iVar < 3; iVar++) {
795 for (jVar = 0; jVar < 3; jVar++) {
796 F_Mat[iVar][jVar] = 0.0;
797 b_Mat[iVar][jVar] = 0.0;
798 }
799 }
800
801 /*--- Retrieve the values of the gradients of the shape functions for each node ---*/
802 /*--- This avoids repeated operations ---*/
803
804 for (iNode = 0; iNode < nNode; iNode++) {
805
806 for (iDim = 0; iDim < nDim; iDim++) {
807 currentCoord[iNode][iDim] = element->GetCurr_Coord(iNode,iDim);
808 GradNi_Ref_Mat[iNode][iDim] = element->GetGradNi_X(iNode,iGauss,iDim);
809 GradNi_Curr_Mat[iNode][iDim] = element->GetGradNi_x(iNode,iGauss,iDim);
810 }
811
812 /*--- Compute the deformation gradient ---*/
813
814 for (iVar = 0; iVar < nDim; iVar++) {
815 for (jVar = 0; jVar < nDim; jVar++) {
816 F_Mat[iVar][jVar] += currentCoord[iNode][iVar]*GradNi_Ref_Mat[iNode][jVar];
817 }
818 }
819 }
820
821 if (nDim == 2) {
822 if (plane_stress) {
823 // Compute the value of the term 33 for the deformation gradient
824 Compute_Plane_Stress_Term(element, config);
825 F_Mat[2][2] = f33;
826 }
827 else { // plane strain
828 F_Mat[2][2] = 1.0;
829 }
830 }
831
832 /*--- Determinant of F --> Jacobian of the transformation ---*/
833
834 J_F = F_Mat[0][0]*F_Mat[1][1]*F_Mat[2][2]+
835 F_Mat[0][1]*F_Mat[1][2]*F_Mat[2][0]+
836 F_Mat[0][2]*F_Mat[1][0]*F_Mat[2][1]-
837 F_Mat[0][2]*F_Mat[1][1]*F_Mat[2][0]-
838 F_Mat[1][2]*F_Mat[2][1]*F_Mat[0][0]-
839 F_Mat[2][2]*F_Mat[0][1]*F_Mat[1][0];
840
841 /*--- Compute the left Cauchy deformation tensor ---*/
842
843 for (iVar = 0; iVar < 3; iVar++) {
844 for (jVar = 0; jVar < 3; jVar++) {
845 for (kVar = 0; kVar < 3; kVar++) {
846 b_Mat[iVar][jVar] += F_Mat[iVar][kVar]*F_Mat[jVar][kVar];
847 }
848 }
849 }
850
851 /*--- Compute the stress tensor ---*/
852
853 Compute_Stress_Tensor(element, config);
854 if (maxwell_stress) Add_MaxwellStress(element, config);
855
856 avgStress[0] += Stress_Tensor[0][0] / nGauss;
857 avgStress[1] += Stress_Tensor[1][1] / nGauss;
858 avgStress[2] += Stress_Tensor[0][1] / nGauss;
859 if (nDim == 3) {
860 avgStress[3] += Stress_Tensor[2][2] / nGauss;
861 avgStress[4] += Stress_Tensor[0][2] / nGauss;
862 avgStress[5] += Stress_Tensor[1][2] / nGauss;
863 }
864
865 for (iNode = 0; iNode < nNode; iNode++) {
866
867 /*--- Compute the nodal stress term for each gaussian point and for each node, ---*/
868 /*--- and add it to the element structure to be retrieved from the solver ---*/
869
870 for (iVar = 0; iVar < nDim; iVar++) {
871 KAux_t_a[iVar] = 0.0;
872 for (jVar = 0; jVar < nDim; jVar++) {
873 KAux_t_a[iVar] += Weight * Stress_Tensor[iVar][jVar] * GradNi_Curr_Mat[iNode][jVar] * Jac_x;
874 }
875 }
876
877 element->Add_Kt_a(iNode, KAux_t_a);
878
879 /*--- Compute the average nodal stresses for each node ---*/
880
881 su2double Ni_Extrap = element->GetNi_Extrap(iNode, iGauss);
882
883 element->Add_NodalStress(iNode, 0, Stress_Tensor[0][0] * Ni_Extrap);
884 element->Add_NodalStress(iNode, 1, Stress_Tensor[1][1] * Ni_Extrap);
885 element->Add_NodalStress(iNode, 2, Stress_Tensor[0][1] * Ni_Extrap);
886 if (nDim == 3) {
887 element->Add_NodalStress(iNode, 3, Stress_Tensor[2][2] * Ni_Extrap);
888 element->Add_NodalStress(iNode, 4, Stress_Tensor[0][2] * Ni_Extrap);
889 element->Add_NodalStress(iNode, 5, Stress_Tensor[1][2] * Ni_Extrap);
890 }
891
892 }
893
894 }
895
896 auto elStress = VonMisesStress(nDim, avgStress);
897
898 /*--- We only differentiate w.r.t. an avg VM stress for the element as
899 * considering all nodal stresses would use too much memory. ---*/
900 AD::SetPreaccOut(elStress);
901 AD::EndPreacc();
902
903 return elStress;
904}
