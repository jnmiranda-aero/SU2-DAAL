56 * \brief Compute Re_theta_c from correlations.
57 * \param[in] Tu - Turbulence intensity.

63 // su2double ReThetaC_Correlations(const CConfig* config, const su2double Tu, const su2double Re_theta_t) const {
64 su2double ReThetaC_Correlations(const CConfig* config, const su2double* V_i, const su2double Tu, const su2double Re_theta_t) const {
65
66 //
67 //========================================================================================================================================================================
68 // My compressibility mods (they probably wont work tbh but its ok)
69 //========================================================================================================================================================================
70 //
71 const su2double Gamma = config->GetGamma();
72 const su2double Gas_Constant = config->GetGas_Constant();
73
74 const su2double* vel_inf = config->GetVelocity_FreeStream();
75 const su2double Vel_inf_u = vel_inf[0]; // Use the first component of the velocity (x-direction)
76 const su2double Vel_inf_v = vel_inf[1]; // Use the first component of the velocity (z-direction)
77 const su2double Vel_inf_w = vel_inf[2]; // Use the first component of the velocity (y-direction)
78 const su2double Vel_inf_Mag = sqrt(Vel_inf_u * Vel_inf_u + Vel_inf_v * Vel_inf_v + Vel_inf_w * Vel_inf_w);
79
80 const su2double Density_inf = config->GetDensity_FreeStream(); //V_i[idx.Density()];//maybe?
81 const su2double Pressure_inf = config->GetPressure_FreeStream();
82 const su2double Temperature_Inf = config->GetTemperature_FreeStream();
83
84 const su2double Pressure_i = V_i[idx.Pressure()]; //ask Prof Badrya about mesh node for P and rho (what node should I get P and rho from)
85
86 const su2double Velocity_BLEdge = sqrt((pow(Vel_inf_Mag,2)) + (((2 * Gamma) / (Gamma - 1)) * (1 - (pow(Pressure_i / Pressure_inf,(1 - (1 / Gamma)))))*(Pressure_i / Density_inf))); // Boundary Layer edge velocity magnitude req'd to calculate edge Mach number for compressibilty correction to LM model (doi:10.2514/6.2022-1542) - jnmiranda-ucd-daal
87 const su2double Speed_of_Sound_BLEdge = sqrt((Gamma * Gas_Constant * Temperature_Inf) * (pow(Pressure_i / Pressure_inf,((1 - Gamma) / Gamma))));
88 // const su2double Speed_of_Sound_BLEdge = sqrt(Gamma * Pressure_i / Density_i); // BL edge speed of sound (doi:10.2514/6.2022-1542) - jnmiranda-ucd-daal
89 const su2double Mach_BLEdge = Velocity_BLEdge / Speed_of_Sound_BLEdge; // BL edge Mach number
90
91 const su2double C_Me = 1.0 - 0.06124 * Mach_BLEdge + 0.2402 * pow(Mach_BLEdge, 2) - 0.00346 * pow(Mach_BLEdge, 3); // (doi:10.2514/6.2022-1542) - jnmiranda-ucd-daal
92 const su2double f_Me = 1.0105 - 0.3046 * Mach_BLEdge + 1.1646 * pow(Mach_BLEdge, 2) - 0.3605 * pow(Mach_BLEdge, 3); // Polynomial function to apply compressibility correction to LM. Used to calculate and modify Re_theta_t (which affects F_length_1) (doi:10.2514/6.2022-1542) - jnmiranda-ucd-daal
93 // || const su2double Re_theta_t_comp = Re_theta_t_Original * f_Me; // Compressibility correction to Re_theta_t (doi:10.2514/6.2022-1542) - jnmiranda-ucd-daal
94 // || const su2double Corr_Ret_comp = Corr_Ret * f_Me; // Compressibility correction to Re_theta_t (doi:10.2514/6.2022-1542) - jnmiranda-ucd-daal
95 //
96 //========================================================================================================================================================================
97 //
98
99 su2double rethetac = 0.0;
100
101 switch (options.Correlation) {
102 case TURB_TRANS_CORRELATION::MALAN: {
103 rethetac = min(0.615 * Re_theta_t + 61.5, Re_theta_t);
104 break;
105 }
106
107 case TURB_TRANS_CORRELATION::SULUKSNA: {
108 rethetac = min(0.1 * exp(-0.0022 * Re_theta_t + 12), 300.0);
109 break;
110 }
111
112 case TURB_TRANS_CORRELATION::KRAUSE: {
113 rethetac = 0.91 * Re_theta_t + 5.32;
114 break;
115 }
116
117 case TURB_TRANS_CORRELATION::KRAUSE_HYPER: {
118 const su2double FirstTerm = -0.042 * pow(Tu, 3);
119 const su2double SecondTerm = 0.4233 * pow(Tu, 2);
120 const su2double ThirdTerm = 0.0118 * pow(Tu, 1);
121 rethetac = Re_theta_t / (FirstTerm + SecondTerm + ThirdTerm + 1.0744);
122 break;
123 }
124
125 case TURB_TRANS_CORRELATION::MEDIDA_BAEDER: {
126 const su2double FirstTerm = 4.45 * pow(Tu, 3);
127 const su2double SecondTerm = 5.7 * pow(Tu, 2);
128 const su2double ThirdTerm = 1.37 * pow(Tu, 1);
129 rethetac = (FirstTerm - SecondTerm + ThirdTerm + 0.585) * Re_theta_t;
130 break;
131 }
132
133 case TURB_TRANS_CORRELATION::MEDIDA: {
134 rethetac = 0.62 * Re_theta_t;
135 break;
136 }
137
138 case TURB_TRANS_CORRELATION::MENTER_LANGTRY: {
139 if (Re_theta_t <= 1870) {
140 const su2double FirstTerm = (-396.035 * pow(10, -2));
141 const su2double SecondTerm = (10120.656 * pow(10, -4)) * Re_theta_t;
142 const su2double ThirdTerm = (-868.230 * pow(10, -6)) * pow(Re_theta_t, 2);
143 const su2double ForthTerm = (696.506 * pow(10, -9)) * pow(Re_theta_t, 3);
144 const su2double FifthTerm = (-174.105 * pow(10, -12)) * pow(Re_theta_t, 4);
145 rethetac = FirstTerm + SecondTerm + ThirdTerm + ForthTerm + FifthTerm;
146 } else {
147 rethetac = Re_theta_t - (593.11 + 0.482 * (Re_theta_t - 1870.0));
148 }
149
150 break;
151 }
152 case TURB_TRANS_CORRELATION::DEFAULT:
153 SU2_MPI::Error("Transition correlation is set to DEFAULT but no default value has ben set in the code.",
154 CURRENT_FUNCTION);
155 break;
156 }
157
158 rethetac = rethetac / C_Me; // Compressibility correction is applied here to the OG Re_theta_c. This is easier than modifying Re_theta_c in all other functions. Instead I think it's better to apply the correction where Re_theta_c is defined
159
160 return rethetac;
161 }

163 /*!
164 * \brief Compute FLength from correlations.
165 * \param[in] Tu - Turbulence intensity.
166 * \param[in] Re_theta_t - Re_theta_t (TransVar[1]).
167 * \param[out] F_length1 - Value for the F_length1 variable.
168 */
169 // su2double FLength_Correlations(const su2double Tu, const su2double Re_theta_t) const {
170 su2double FLength_Correlations(const CConfig config, const su2double Tu, const su2double Re_theta_t) const {
171 su2double F_length1 = 0.0;
172
173 switch (options.Correlation) {
174 case TURB_TRANS_CORRELATION::MALAN: {
175 F_length1 = min(exp(7.168 - 0.01173 * Re_theta_t) + 0.5, 300.0);
176 break;
177 }
178
179 case TURB_TRANS_CORRELATION::SULUKSNA: {
180 const su2double FirstTerm = -pow(0.025 * Re_theta_t, 2) + 1.47 * Re_theta_t - 120.0;
181 F_length1 = min(max(FirstTerm, 125.0), Re_theta_t);
182 break;
183 }
184
185 case TURB_TRANS_CORRELATION::KRAUSE: {
186 F_length1 = 3.39 * Re_theta_t + 55.03;
187 break;
188 }
189
190 case TURB_TRANS_CORRELATION::KRAUSE_HYPER: {
191 if (Tu <= 1.) {
192 F_length1 = log(Re_theta_t + 1) / Tu;
193 } else {
194 const su2double FirstTerm = 0.2337 * pow(Tu, 2);
195 const su2double SecondTerm = -1.3493 * pow(Tu, 1);
196 F_length1 = log(Re_theta_t + 1) * (FirstTerm + SecondTerm + 2.1449);
197 }
198 break;
199 }
200
201 case TURB_TRANS_CORRELATION::MEDIDA_BAEDER: {
202 const su2double FirstTerm = 0.171 * pow(Tu, 2);
203 const su2double SecondTerm = 0.0083 * pow(Tu, 1);
204 F_length1 = (FirstTerm - SecondTerm + 0.0306);
205 break;
206 }
207
208 case TURB_TRANS_CORRELATION::MEDIDA: {
209 F_length1 = 40;
210 break;
211 }
212
213 case TURB_TRANS_CORRELATION::MENTER_LANGTRY: {
214 if (Re_theta_t < 400) {
215 F_length1 = 39.8189 + (-119.270 * pow(10, -4)) * Re_theta_t +
216 (-132.567 * pow(10, -6)) * Re_theta_t * Re_theta_t;
217 } else if (Re_theta_t < 596) {
218 F_length1 = 263.404 + (-123.939 * pow(10, -2)) * Re_theta_t +
219 (194.548 * pow(10, -5)) * pow(Re_theta_t, 2) +
220 (-101.695 * pow(10, -8)) * pow(Re_theta_t, 3);
221 } else if (Re_theta_t < 1200) {
222 F_length1 = 0.5 - (3.0 * pow(10, -4)) * (Re_theta_t - 596.0);
223 } else {
224 F_length1 = 0.3188;
225 }
226 break;
227 }
228 case TURB_TRANS_CORRELATION::DEFAULT:
229 SU2_MPI::Error("Transition correlation is set to DEFAULT but no default value has ben set in the code.",
230 CURRENT_FUNCTION);
231 break;
232 }
233
234 return F_length1;
235 }
