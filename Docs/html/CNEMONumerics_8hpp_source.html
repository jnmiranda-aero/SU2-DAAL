165  /*!
166  * \brief TSL-Approximation of Viscous NS Jacobians for arbitrary equations of state.
167  */
168  template <unsigned short NVAR, unsigned short NSPECIES>
169  void ComputeViscousJacs_impl(const su2double *val_Mean_PrimVar,
170  su2double *val_Mean_Eve,
171  const su2double *val_Mean_Cvve,
172  const su2double *val_diffusion_coeff,
173  su2double val_laminar_viscosity,
174  su2double val_eddy_viscosity,
175  su2double val_thermal_conductivity,
176  su2double val_thermal_conductivity_ve,
177  su2double val_dist_ij, const su2double *val_normal,
178  su2double val_dS, const su2double *val_Fv,
179  su2double **val_Jac_i, su2double **val_Jac_j,
180  const CConfig *config){
181 
182  /*--- Initialize matrices ---*/
183  Matrix<NVAR, NVAR> dFdVi, dFdVj, dVdUi, dVdUj;
184 
185  /*--- Need to avoid size of 1 otherwise the type becomes a vector (without [i][j]). ---*/
186  constexpr unsigned short nSpeciesDummy = (NSPECIES == 1) ? 2 : NSPECIES;
187  Matrix<nSpeciesDummy, nSpeciesDummy> dJdr_i, dJdr_j;
188  Matrix<1,NSPECIES> Ys, Ys_i, Ys_j;
189 
190  /*--- Play tricks on the compiler, in static mode use NVAR from the template, in dynamic mode
191  use nVar from the class or from the arguments. ---*/
192  const auto nVar = (NVAR != DynamicSize) ? NVAR : this->nVar;
193  const auto nSpecies = (NSPECIES != DynamicSize) ? NSPECIES : this->nSpecies;
194 
195  /*--- Deduce nDim from the other variables to allow the compiler to also unroll the loops over nDim. ---*/
196  const auto nDim = nVar - nSpecies - 2;
197 
198  // Allocate and initialize, for the static case the compiler optimizes this away.
199  dFdVi.resize(nVar, nVar) = su2double(0.0);
200  dFdVj.resize(nVar, nVar) = su2double(0.0);
201  dVdUi.resize(nVar, nVar) = su2double(0.0);
202  dVdUj.resize(nVar, nVar) = su2double(0.0);
203  if (NSPECIES == DynamicSize) {
204  dJdr_i.resize(nSpecies,nSpecies) = su2double(0.0);
205  dJdr_j.resize(nSpecies,nSpecies) = su2double(0.0);
206  Ys.resize(1,nSpecies) = su2double(0.0);
207  Ys_i.resize(1,nSpecies) = su2double(0.0);
208  Ys_j.resize(1,nSpecies) = su2double(0.0);
209  }
210 
211  /*--- Calculate preliminary geometric quantities ---*/
212  su2double dij = val_dist_ij;
213  su2double theta = GeometryToolbox::SquaredNorm(nDim, val_normal);
214 
215  /*--- Rename for convenience ---*/
216  su2double rho_i = V_i[RHO_INDEX];
217  su2double rho_j = V_j[RHO_INDEX];
218  su2double T = val_Mean_PrimVar[T_INDEX];
219  su2double Tve = val_Mean_PrimVar[TVE_INDEX];
220  su2double mu = val_laminar_viscosity;
221  su2double ktr = val_thermal_conductivity;
222  su2double kve = val_thermal_conductivity_ve;
223  su2double RuSI= UNIVERSAL_GAS_CONSTANT;
224  su2double Ru = 1000.0*RuSI;
225  const auto& Ds = val_diffusion_coeff;
226  const auto& hs = fluidmodel->ComputeSpeciesEnthalpy(T, Tve, val_Mean_Eve);
227  const auto& Cvtr = fluidmodel->GetSpeciesCvTraRot();
228  const auto& Ms = fluidmodel->GetSpeciesMolarMass();
229 
230  /*--- Calculate useful diffusion parameters ---*/
231  // Mass fraction and summation of diffusion fluxes
232  su2double sumY = 0.0;
233  su2double sumY_i = 0.0;
234  su2double sumY_j = 0.0;
235  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
236  Ys[iSpecies] = val_Mean_PrimVar[RHOS_INDEX+iSpecies];
237  Ys_i[iSpecies] = V_i[RHOS_INDEX+iSpecies]/V_i[RHO_INDEX];
238  Ys_j[iSpecies] = V_j[RHOS_INDEX+iSpecies]/V_j[RHO_INDEX];
239 
240  sumY_i += Ds[iSpecies]*theta/dij*Ys_i[iSpecies];
241  sumY_j += Ds[iSpecies]*theta/dij*Ys_j[iSpecies];
242  sumY += Ds[iSpecies]*theta/dij*(Ys_j[iSpecies]-Ys_i[iSpecies]);
243  }
244 
245  su2double vel[MAXNDIM];
246  for (auto iDim = 0; iDim < nDim; iDim++)
247  vel[iDim] = val_Mean_PrimVar[VEL_INDEX+iDim];
248 
249  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
250  for (auto jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
251 
252  // first term
253  dJdr_j[iSpecies][jSpecies] += 0.5*(Ds[iSpecies]*theta/dij *
254  (Ys_j[iSpecies]*rho_i/rho_j +
255  Ys_i[iSpecies]));
256  dJdr_i[iSpecies][jSpecies] += -0.5*(Ds[iSpecies]*theta/dij *
257  (Ys_j[iSpecies] +
258  Ys_i[iSpecies]*rho_j/rho_i));
259 
260  // second term
261  dJdr_j[iSpecies][jSpecies] +=
262  0.25*(Ys_i[iSpecies] - rho_i/rho_j*Ys_j[iSpecies])*sumY
263  + 0.25*(Ys_i[iSpecies]+Ys_j[iSpecies])*(rho_i+rho_j)*Ds[jSpecies]*theta/(dij*rho_j)
264  - 0.25*(Ys_i[iSpecies]+Ys_j[iSpecies])*(rho_i+rho_j)*sumY_j/rho_j;
265 
266  dJdr_i[iSpecies][jSpecies] +=
267  0.25*(-rho_j/rho_i*Ys_i[iSpecies]+Ys_j[iSpecies])*sumY
268  - 0.25*(Ys_i[iSpecies]+Ys_j[iSpecies])*(rho_i+rho_j)*Ds[jSpecies]*theta/(dij*rho_i)
269  + 0.25*(Ys_i[iSpecies]+Ys_j[iSpecies])*(rho_i+rho_j)*sumY_i/rho_i;
270  }
271 
272  // first term
273  dJdr_j[iSpecies][iSpecies] += -0.5*Ds[iSpecies]*theta/dij*(1+rho_i/rho_j);
274  dJdr_i[iSpecies][iSpecies] += 0.5*Ds[iSpecies]*theta/dij*(1+rho_j/rho_i);
275 
276  // second term
277  dJdr_j[iSpecies][iSpecies] += 0.25*(1.0+rho_i/rho_j)*sumY;
278  dJdr_i[iSpecies][iSpecies] += 0.25*(1.0+rho_j/rho_i)*sumY;
279  }
280 
281  /*--- Calculate transformation matrix ---*/
282  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
283  dVdUi[iSpecies][iSpecies] = 1.0;
284  dVdUj[iSpecies][iSpecies] = 1.0;
285  }
286  for (auto iDim = 0; iDim < nDim; iDim++) {
287  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
288  dVdUi[nSpecies+iDim][iSpecies] = -V_i[VEL_INDEX+iDim]/V_i[RHO_INDEX];
289  dVdUj[nSpecies+iDim][iSpecies] = -V_j[VEL_INDEX+iDim]/V_j[RHO_INDEX];
290  }
291  dVdUi[nSpecies+iDim][nSpecies+iDim] = 1.0/V_i[RHO_INDEX];
292  dVdUj[nSpecies+iDim][nSpecies+iDim] = 1.0/V_j[RHO_INDEX];
293  }
294  for (auto iVar = 0; iVar < nVar; iVar++) {
295  dVdUi[nSpecies+nDim][iVar] = dTdU_i[iVar];
296  dVdUj[nSpecies+nDim][iVar] = dTdU_j[iVar];
297  dVdUi[nSpecies+nDim+1][iVar] = dTvedU_i[iVar];
298  dVdUj[nSpecies+nDim+1][iVar] = dTvedU_j[iVar];
299  }
300 
301  if (nDim == 2) {
302 
303  /*--- Geometry parameters ---*/
304  su2double thetax = theta + val_normal[0]*val_normal[0]/3.0;
305  su2double thetay = theta + val_normal[1]*val_normal[1]/3.0;
306  su2double etaz = val_normal[0]*val_normal[1]/3.0;
307  su2double pix = mu/dij * (thetax*vel[0] + etaz*vel[1] );
308  su2double piy = mu/dij * (etaz*vel[0] + thetay*vel[1]);
309 
310  /*--- Populate primitive Jacobian ---*/
311 
312  // X-momentum
313  dFdVj[nSpecies][nSpecies] = mu*thetax/dij*val_dS;
314  dFdVj[nSpecies][nSpecies+1] = mu*etaz/dij*val_dS;
315 
316  // Y-momentum
317  dFdVj[nSpecies+1][nSpecies] = mu*etaz/dij*val_dS;
318  dFdVj[nSpecies+1][nSpecies+1] = mu*thetay/dij*val_dS;
319 
320  // Energy
321  dFdVj[nSpecies+2][nSpecies] = pix*val_dS;
322  dFdVj[nSpecies+2][nSpecies+1] = piy*val_dS;
323  dFdVj[nSpecies+2][nSpecies+2] = ktr*theta/dij*val_dS;
324  dFdVj[nSpecies+2][nSpecies+3] = kve*theta/dij*val_dS;
325 
326  // Vib-el Energy
327  dFdVj[nSpecies+3][nSpecies+3] = kve*theta/dij*val_dS;
328 
329  for (auto iVar = 0; iVar < nVar; iVar++)
330  for (auto jVar = 0; jVar < nVar; jVar++)
331  dFdVi[iVar][jVar] = -dFdVj[iVar][jVar];
332 
333  // Common terms
334  dFdVi[nSpecies+2][nSpecies] += 0.5*val_Fv[nSpecies];
335  dFdVj[nSpecies+2][nSpecies] += 0.5*val_Fv[nSpecies];
336  dFdVi[nSpecies+2][nSpecies+1] += 0.5*val_Fv[nSpecies+1];
337  dFdVj[nSpecies+2][nSpecies+1] += 0.5*val_Fv[nSpecies+1];
338  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
339  dFdVi[nSpecies+2][nSpecies+2] += 0.5*val_Fv[iSpecies]*(Ru/Ms[iSpecies] +
340  Cvtr[iSpecies] );
341  dFdVj[nSpecies+2][nSpecies+2] += 0.5*val_Fv[iSpecies]*(Ru/Ms[iSpecies] +
342  Cvtr[iSpecies] );
343  dFdVi[nSpecies+2][nSpecies+3] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];
344  dFdVj[nSpecies+2][nSpecies+3] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];
345  dFdVi[nSpecies+3][nSpecies+3] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];
346  dFdVj[nSpecies+3][nSpecies+3] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];
347  }
348 
349  // Unique terms
350  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
351  for (auto jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
352  dFdVj[iSpecies][jSpecies] += -dJdr_j[iSpecies][jSpecies]*val_dS;
353  dFdVi[iSpecies][jSpecies] += -dJdr_i[iSpecies][jSpecies]*val_dS;
354  dFdVj[nSpecies+2][iSpecies] += -dJdr_j[jSpecies][iSpecies]*hs[jSpecies]*val_dS;
355  dFdVi[nSpecies+2][iSpecies] += -dJdr_i[jSpecies][iSpecies]*hs[jSpecies]*val_dS;
356  dFdVj[nSpecies+3][iSpecies] += -dJdr_j[jSpecies][iSpecies]*val_Mean_Eve[jSpecies]*val_dS;
357  dFdVi[nSpecies+3][iSpecies] += -dJdr_i[jSpecies][iSpecies]*val_Mean_Eve[jSpecies]*val_dS;
358  }
359  }
360 
361  } //nDim == 2
362  else {
363 
364  /*--- Geometry parameters ---*/
365  su2double thetax = theta + val_normal[0]*val_normal[0]/3.0;
366  su2double thetay = theta + val_normal[1]*val_normal[1]/3.0;
367  su2double thetaz = theta + val_normal[2]*val_normal[2]/3.0;
368  su2double etax = val_normal[1]*val_normal[2]/3.0;
369  su2double etay = val_normal[0]*val_normal[2]/3.0;
370  su2double etaz = val_normal[0]*val_normal[1]/3.0;
371  su2double pix = mu/dij * (thetax*vel[0] + etaz*vel[1] + etay*vel[2] );
372  su2double piy = mu/dij * (etaz*vel[0] + thetay*vel[1] + etax*vel[2] );
373  su2double piz = mu/dij * (etay*vel[0] + etax*vel[1] + thetaz*vel[2]);
374 
375  /*--- Populate primitive Jacobian ---*/
376 
377  // X-momentum
378  dFdVj[nSpecies][nSpecies] = mu*thetax/dij*val_dS;
379  dFdVj[nSpecies][nSpecies+1] = mu*etaz/dij*val_dS;
380  dFdVj[nSpecies][nSpecies+2] = mu*etay/dij*val_dS;
381 
382  // Y-momentum
383  dFdVj[nSpecies+1][nSpecies] = mu*etaz/dij*val_dS;
384  dFdVj[nSpecies+1][nSpecies+1] = mu*thetay/dij*val_dS;
385  dFdVj[nSpecies+1][nSpecies+2] = mu*etax/dij*val_dS;
386 
387  // Z-momentum
388  dFdVj[nSpecies+2][nSpecies] = mu*etay/dij*val_dS;
389  dFdVj[nSpecies+2][nSpecies+1] = mu*etax/dij*val_dS;
390  dFdVj[nSpecies+2][nSpecies+2] = mu*thetaz/dij*val_dS;
391 
392  // Energy
393  dFdVj[nSpecies+3][nSpecies] = pix*val_dS;
394  dFdVj[nSpecies+3][nSpecies+1] = piy*val_dS;
395  dFdVj[nSpecies+3][nSpecies+2] = piz*val_dS;
396  dFdVj[nSpecies+3][nSpecies+3] = ktr*theta/dij*val_dS;
397  dFdVj[nSpecies+3][nSpecies+4] = kve*theta/dij*val_dS;
398 
399  // Vib.-el energy
400  dFdVj[nSpecies+4][nSpecies+4] = kve*theta/dij*val_dS;
401 
402  for (auto iVar = 0; iVar < nVar; iVar++)
403  for (auto jVar = 0; jVar < nVar; jVar++)
404  dFdVi[iVar][jVar] = -dFdVj[iVar][jVar];
405 
406  // Common terms
407  for (auto iDim = 0; iDim < nDim; iDim++) {
408  dFdVi[nSpecies+3][nSpecies+iDim] += 0.5*val_Fv[nSpecies+iDim];
409  dFdVj[nSpecies+3][nSpecies+iDim] += 0.5*val_Fv[nSpecies+iDim];
410  }
411  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
412  dFdVi[nSpecies+3][nSpecies+3] += 0.5*val_Fv[iSpecies]*(Ru/Ms[iSpecies] +
413  Cvtr[iSpecies] );
414  dFdVj[nSpecies+3][nSpecies+3] += 0.5*val_Fv[iSpecies]*(Ru/Ms[iSpecies] +
415  Cvtr[iSpecies] );
