69 su2double CRadialBasisFunction::Get_RadialBasisValue(RADIAL_BASIS type, const su2double radius, const su2double dist) {
70  su2double rbf = dist / radius;
71 
72  switch (type) {
73  case RADIAL_BASIS::WENDLAND_C2:
74  if (rbf < 1)
75  rbf = pow(pow((1 - rbf), 2), 2) * (4 * rbf + 1); // double use of pow(x,2) for optimization
76  else
77  rbf = 0.0;
78  break;
79 
80  case RADIAL_BASIS::GAUSSIAN:
81  rbf = exp(-rbf * rbf);
82  break;
83 
84  case RADIAL_BASIS::THIN_PLATE_SPLINE:
85  if (rbf < numeric_limits<float>::min())
86  rbf = 0.0;
87  else
88  rbf *= rbf * log(rbf);
89  break;
90 
91  case RADIAL_BASIS::MULTI_QUADRIC:
92  case RADIAL_BASIS::INV_MULTI_QUADRIC:
93  rbf = sqrt(1.0 + rbf * rbf);
94  if (type == RADIAL_BASIS::INV_MULTI_QUADRIC) rbf = 1.0 / rbf;
95  break;
96  }
97 
98  return rbf;
99 }

101 void CRadialBasisFunction::SetTransferCoeff(const CConfig* const* config) {
102  /*--- RBF options. ---*/
103  const auto kindRBF = config[donorZone]->GetKindRadialBasisFunction();
104  const bool usePolynomial = config[donorZone]->GetRadialBasisFunctionPolynomialOption();
105  const su2double paramRBF = config[donorZone]->GetRadialBasisFunctionParameter();
106  const su2double pruneTol = config[donorZone]->GetRadialBasisFunctionPruneTol();
107 
108  const auto nMarkerInt = config[donorZone]->GetMarker_n_ZoneInterface() / 2;
109  const int nDim = donor_geometry->GetnDim();
110 
111  const int nProcessor = size;
112  Buffer_Receive_nVertex_Donor = new unsigned long[nProcessor];
113 
114  targetVertices.resize(config[targetZone]->GetnMarker_All());
115 
116  /*--- Process interface patches in parallel, fetch all donor point coordinates,
117  * then distribute interpolation matrix computation over ranks and threads.
118  * To avoid repeating calls to Collect_VertexInfo we also save the global
119  * indices of the donor points and the mpi rank index that owns them. ---*/
120 
121  vector<su2activematrix> donorCoordinates(nMarkerInt);
122  vector<vector<long> > donorGlobalPoint(nMarkerInt);
123  vector<vector<int> > donorProcessor(nMarkerInt);
124  vector<int> assignedProcessor(nMarkerInt, -1);
125  vector<unsigned long> totalWork(nProcessor, 0);
126 
127  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; ++iMarkerInt) {
128  /*--- On the donor side: find the tag of the boundary sharing the interface. ---*/
129  const auto markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
130 
131  /*--- On the target side: find the tag of the boundary sharing the interface. ---*/
132  const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
133 
134  /*--- If the zone does not contain the interface continue to the next pair of markers. ---*/
135  if (!CheckInterfaceBoundary(markDonor, markTarget)) continue;
136 
137  unsigned long nVertexDonor = 0;
138  if (markDonor != -1) nVertexDonor = donor_geometry->GetnVertex(markDonor);
139 
140  /*--- Sets MaxLocalVertex_Donor, Buffer_Receive_nVertex_Donor. ---*/
141  Determine_ArraySize(markDonor, markTarget, nVertexDonor, nDim);
142 
143  /*--- Compute total number of donor vertices. ---*/
144  const auto nGlobalVertexDonor =
145  accumulate(Buffer_Receive_nVertex_Donor, Buffer_Receive_nVertex_Donor + nProcessor, 0ul);
146 
147  /*--- Gather coordinates and global point indices. ---*/
148  Buffer_Send_Coord.resize(MaxLocalVertex_Donor, nDim);
149  Buffer_Send_GlobalPoint.resize(MaxLocalVertex_Donor);
150  Buffer_Receive_Coord.resize(nProcessor * MaxLocalVertex_Donor, nDim);
151  Buffer_Receive_GlobalPoint.resize(nProcessor * MaxLocalVertex_Donor);
152 
153  Collect_VertexInfo(markDonor, markTarget, nVertexDonor, nDim);
154 
155  /*--- Compresses the gathered donor point information to simplify computations. ---*/
156  auto& donorCoord = donorCoordinates[iMarkerInt];
157  auto& donorPoint = donorGlobalPoint[iMarkerInt];
158  auto& donorProc = donorProcessor[iMarkerInt];
159  donorCoord.resize(nGlobalVertexDonor, nDim);
160  donorPoint.resize(nGlobalVertexDonor);
161  donorProc.resize(nGlobalVertexDonor);
162 
163  auto iCount = 0ul;
164  for (int iProcessor = 0; iProcessor < nProcessor; ++iProcessor) {
165  auto offset = iProcessor * MaxLocalVertex_Donor;
166  for (auto iVertex = 0ul; iVertex < Buffer_Receive_nVertex_Donor[iProcessor]; ++iVertex) {
167  for (int iDim = 0; iDim < nDim; ++iDim) donorCoord(iCount, iDim) = Buffer_Receive_Coord(offset + iVertex, iDim);
168  donorPoint[iCount] = Buffer_Receive_GlobalPoint[offset + iVertex];
169  donorProc[iCount] = iProcessor;
170  ++iCount;
171  }
172  }
173  assert((iCount == nGlobalVertexDonor) && "Global donor point count mismatch.");
174 
175  /*--- Give an MPI-independent order to the points (required due to high condition
176  * number of the RBF matrix, avoids diff results with diff number of ranks. ---*/
177  vector<int> order(nGlobalVertexDonor);
178  iota(order.begin(), order.end(), 0);
179  sort(order.begin(), order.end(), [&donorPoint](int i, int j) { return donorPoint[i] < donorPoint[j]; });
180 
181  for (int i = 0; i < int(nGlobalVertexDonor); ++i) {
182  int j = order[i];
183  while (j < i) j = order[j];
184  if (i == j) continue;
185  swap(donorProc[i], donorProc[j]);
186  swap(donorPoint[i], donorPoint[j]);
187  for (int iDim = 0; iDim < nDim; ++iDim) swap(donorCoord(i, iDim), donorCoord(j, iDim));
188  }
189 
190  /*--- Static work scheduling over ranks based on which one has less work currently. ---*/
191  int iProcessor = 0;
192  for (int i = 1; i < nProcessor; ++i)
193  if (totalWork[i] < totalWork[iProcessor]) iProcessor = i;
194 
195  totalWork[iProcessor] += pow(nGlobalVertexDonor, 3); // based on matrix inversion.
196 
197  assignedProcessor[iMarkerInt] = iProcessor;
198  }
199  delete[] Buffer_Receive_nVertex_Donor;
200 
201  /*--- Compute the interpolation matrices for each patch of coordinates
202  * assigned to the rank. Subdivide work further by threads. ---*/
203  vector<int> nPolynomialVec(nMarkerInt, -1);
204  vector<vector<int> > keepPolynomialRowVec(nMarkerInt, vector<int>(nDim, 1));
205  vector<su2passivematrix> CinvTrucVec(nMarkerInt);
206 
207  SU2_OMP_PARALLEL_(for schedule(dynamic,1))
208  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; ++iMarkerInt) {
209  if (rank == assignedProcessor[iMarkerInt]) {
210  ComputeGeneratorMatrix(kindRBF, usePolynomial, paramRBF, donorCoordinates[iMarkerInt], nPolynomialVec[iMarkerInt],
211  keepPolynomialRowVec[iMarkerInt], CinvTrucVec[iMarkerInt]);
212  }
213  }
214  END_SU2_OMP_PARALLEL
215 
216  /*--- Final loop over interface markers to compute the interpolation coefficients. ---*/
217 
218  /*--- Initialize variables for interpolation statistics. ---*/
219  unsigned long totalTargetPoints = 0, totalDonorPoints = 0, denseSize = 0;
220  MinDonors = 1 << 30;
221  MaxDonors = 0;
222  MaxCorrection = 0.0;
223  AvgCorrection = 0.0;
224 
225  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
226  /*--- Identify the rank that computed the interpolation matrix for this marker. ---*/
227  const int iProcessor = assignedProcessor[iMarkerInt];
228  /*--- If no processor was assigned to work, the zone does not contain the interface. ---*/
229  if (iProcessor < 0) continue;
230 
231  /*--- Setup target information. ---*/
232  const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
233  unsigned long nVertexTarget = 0;
234  if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex(markTarget);
235 
236  /*--- Set references to donor information. ---*/
237  auto& donorCoord = donorCoordinates[iMarkerInt];
238  auto& donorPoint = donorGlobalPoint[iMarkerInt];
239  auto& donorProc = donorProcessor[iMarkerInt];
240 
241  auto& C_inv_trunc = CinvTrucVec[iMarkerInt];
242  auto& nPolynomial = nPolynomialVec[iMarkerInt];
243  auto& keepPolynomialRow = keepPolynomialRowVec[iMarkerInt];
244 
245  const auto nGlobalVertexDonor = donorCoord.rows();
246 
247 #ifdef HAVE_MPI
248  /*--- For simplicity, broadcast small information about the interpolation matrix. ---*/
249  SU2_MPI::Bcast(&nPolynomial, 1, MPI_INT, iProcessor, SU2_MPI::GetComm());
250  SU2_MPI::Bcast(keepPolynomialRow.data(), nDim, MPI_INT, iProcessor, SU2_MPI::GetComm());
251 
252  /*--- Send C_inv_trunc only to the ranks that need it (those with target points),
253  * partial broadcast. MPI wrapper not used due to passive double. ---*/
254  vector<unsigned long> allNumVertex(nProcessor);
255  SU2_MPI::Allgather(&nVertexTarget, 1, MPI_UNSIGNED_LONG, allNumVertex.data(), 1, MPI_UNSIGNED_LONG,
256  SU2_MPI::GetComm());
257 
258  if (rank == iProcessor) {
259  for (int jProcessor = 0; jProcessor < nProcessor; ++jProcessor)
260  if ((jProcessor != iProcessor) && (allNumVertex[jProcessor] != 0))
261  MPI_Send(C_inv_trunc.data(), C_inv_trunc.size(), MPI_DOUBLE, jProcessor, 0, SU2_MPI::GetComm());
262  } else if (nVertexTarget != 0) {
263  C_inv_trunc.resize(1 + nPolynomial + nGlobalVertexDonor, nGlobalVertexDonor);
264  MPI_Recv(C_inv_trunc.data(), C_inv_trunc.size(), MPI_DOUBLE, iProcessor, 0, SU2_MPI::GetComm(),
265  MPI_STATUS_IGNORE);
266  }
267 #endif
268 
269  /*--- Compute interpolation matrix (H). This is a large matrix-matrix product with
270  * the generator matrix (C_inv_trunc) on the right. We avoid instantiation
271  * of the entire function matrix (A) and of the result (H), but work
272  * on a slab (set of rows) of A/H to amortize accesses to C_inv_trunc. ---*/
273 
274  /*--- Fetch target vertex coordinates. ---*/
275 
276  if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
277  vector<const su2double*> targetCoord(nVertexTarget);
278 
279  for (auto iVertexTarget = 0ul; iVertexTarget < nVertexTarget; ++iVertexTarget) {
280  const auto pointTarget = target_geometry->vertex[markTarget][iVertexTarget]->GetNode();
281  targetCoord[iVertexTarget] = target_geometry->nodes->GetCoord(pointTarget);
282  }
283  totalTargetPoints += nVertexTarget;
284  denseSize += nVertexTarget * nGlobalVertexDonor;
285 
286  /*--- Distribute target slabs over the threads in the rank for processing. ---*/
287 
288  SU2_OMP_PARALLEL
289  if (nVertexTarget > 0) {
290  constexpr unsigned long targetSlabSize = 32;
291 
292  su2passivematrix funcMat(targetSlabSize, 1 + nPolynomial + nGlobalVertexDonor);
293  su2passivematrix interpMat(targetSlabSize, nGlobalVertexDonor);
294 
295  /*--- Thread-local variables for statistics. ---*/
296  unsigned long minDonors = 1 << 30, maxDonors = 0, totalDonors = 0;
297  passivedouble sumCorr = 0.0, maxCorr = 0.0;
298 
299  SU2_OMP_FOR_DYN(1)
300  for (auto iVertexTarget = 0ul; iVertexTarget < nVertexTarget; iVertexTarget += targetSlabSize) {
301  const auto iLastVertex = min(nVertexTarget, iVertexTarget + targetSlabSize);
302  const auto slabSize = iLastVertex - iVertexTarget;
303 
304  /*--- Prepare matrix of functions A (the targets to donors matrix). ---*/
305 
306  /*--- Polynominal part: ---*/
307  if (usePolynomial) {
308  /*--- Constant term. ---*/
309  for (auto k = 0ul; k < slabSize; ++k) funcMat(k, 0) = 1.0;
310 
311  /*--- Linear terms. ---*/
312  for (int iDim = 0, idx = 1; iDim < nDim; ++iDim) {
313  /*--- Of which one may have been excluded. ---*/
314  if (!keepPolynomialRow[iDim]) continue;
315  for (auto k = 0ul; k < slabSize; ++k)
316  funcMat(k, idx) = SU2_TYPE::GetValue(targetCoord[iVertexTarget + k][iDim]);
317  idx += 1;
318  }
319  }
320  /*--- RBF terms: ---*/
321  for (auto iVertexDonor = 0ul; iVertexDonor < nGlobalVertexDonor; ++iVertexDonor) {
322  for (auto k = 0ul; k < slabSize; ++k) {
323  auto dist = GeometryToolbox::Distance(nDim, targetCoord[iVertexTarget + k], donorCoord[iVertexDonor]);
324  auto rbf = Get_RadialBasisValue(kindRBF, paramRBF, dist);
325  funcMat(k, 1 + nPolynomial + iVertexDonor) = SU2_TYPE::GetValue(rbf);
326  }
327  }
328 
329  /*--- Compute slab of the interpolation matrix. ---*/
330 #ifdef HAVE_LAPACK
331  /*--- interpMat = funcMat * C_inv_trunc, but order of gemm arguments
332  * is swapped due to row-major storage of su2passivematrix. ---*/
333  const char op = 'N';
334  const int M = interpMat.cols(), N = slabSize, K = funcMat.cols();
335  // lda = C_inv_trunc.cols() = M; ldb = funcMat.cols() = K; ldc = interpMat.cols() = M;
336  const passivedouble alpha = 1.0, beta = 0.0;
337  DGEMM(&op, &op, &M, &N, &K, &alpha, C_inv_trunc[0], &M, funcMat[0], &K, &beta, interpMat[0], &M);
338 #else
339  /*--- Naive product, loop order considers short-wide
340  * nature of funcMat and interpMat. ---*/
341  interpMat = 0.0;
342  for (auto k = 0ul; k < funcMat.cols(); ++k)
343  for (auto i = 0ul; i < slabSize; ++i)
344  for (auto j = 0ul; j < interpMat.cols(); ++j) interpMat(i, j) += funcMat(i, k) * C_inv_trunc(k, j);
345 #endif
346  /*--- Set interpolation coefficients. ---*/
347 
348  for (auto k = 0ul; k < slabSize; ++k) {
349  auto& targetVertex = targetVertices[markTarget][iVertexTarget + k];
350 
351  /*--- Prune small coefficients. ---*/
352  auto info = PruneSmallCoefficients(SU2_TYPE::GetValue(pruneTol), interpMat.cols(), interpMat[k]);
353  auto nnz = info.first;
354  totalDonors += nnz;
355  minDonors = min(minDonors, nnz);
356  maxDonors = max(maxDonors, nnz);
357  auto corr = fabs(info.second - 1.0); // far from 1 either way is bad;
358  sumCorr += corr;
359  maxCorr = max(maxCorr, corr);
360 
361  /*--- Allocate and set donor information for this target point. ---*/
362  targetVertex.resize(nnz);
363 
364  for (unsigned long iVertex = 0, iSet = 0; iVertex < nGlobalVertexDonor; ++iVertex) {
365  auto coeff = interpMat(k, iVertex);
366  if (fabs(coeff) > 0.0) {
367  targetVertex.processor[iSet] = donorProc[iVertex];
368  targetVertex.globalPoint[iSet] = donorPoint[iVertex];
369  targetVertex.coefficient[iSet] = coeff;
370  ++iSet;
371  }
372  }
373  }
374  } // end target vertex loop
375  END_SU2_OMP_FOR
376  SU2_OMP_CRITICAL {
377  totalDonorPoints += totalDonors;
378  MinDonors = min(MinDonors, minDonors);
379  MaxDonors = max(MaxDonors, maxDonors);
380  AvgCorrection += sumCorr;
381  MaxCorrection = max(MaxCorrection, maxCorr);
382  }
383  END_SU2_OMP_CRITICAL
384  }
385  END_SU2_OMP_PARALLEL
386 
387  /*--- Free global data that will no longer be used. ---*/
388  donorCoord.resize(0, 0);
389  vector<long>().swap(donorPoint);
390  vector<int>().swap(donorProc);
391  C_inv_trunc.resize(0, 0);
392 
393  } // end loop over interface markers
394 
395  /*--- Final reduction of interpolation statistics and basic sanity checks. ---*/
396  auto Reduce = [](SU2_MPI::Op op, unsigned long& val) {
397  auto tmp = val;
398  SU2_MPI::Allreduce(&tmp, &val, 1, MPI_UNSIGNED_LONG, op, SU2_MPI::GetComm());
399  };
400  Reduce(MPI_SUM, totalTargetPoints);
401  Reduce(MPI_SUM, totalDonorPoints);
402  Reduce(MPI_SUM, denseSize);
403  Reduce(MPI_MIN, MinDonors);
404  Reduce(MPI_MAX, MaxDonors);
405 #ifdef HAVE_MPI
406  passivedouble tmp1 = AvgCorrection, tmp2 = MaxCorrection;
407  MPI_Allreduce(&tmp1, &AvgCorrection, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
408  MPI_Allreduce(&tmp2, &MaxCorrection, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
409 #endif
410  if (totalTargetPoints == 0) SU2_MPI::Error("Somehow there are no target interpolation points.", CURRENT_FUNCTION);
411 
412  if (MinDonors == 0)
413  SU2_MPI::Error(
414  "One or more target points have no donors, either:\n"
415  " - The interface surfaces are not in contact.\n"
416  " - The RBF radius is too small.\n"
417  " - The pruning tolerance is too aggressive.",
418  CURRENT_FUNCTION);
419 
420  MaxCorrection += 1.0; // put back the reference "1"
421  AvgCorrection = AvgCorrection / totalTargetPoints + 1.0;
422  AvgDonors = totalDonorPoints / totalTargetPoints;
423  Density = totalDonorPoints / (0.01 * denseSize);
424 }

426 void CRadialBasisFunction::ComputeGeneratorMatrix(RADIAL_BASIS type, bool usePolynomial, su2double radius,
427  const su2activematrix& coords, int& nPolynomial,
428  vector<int>& keepPolynomialRow, su2passivematrix& C_inv_trunc) {
429  const su2double interfaceCoordTol = 1e6 * numeric_limits<passivedouble>::epsilon();
430 
431  const int nVertexDonor = coords.rows();
432  const int nDim = coords.cols();
433 
434  /*--- Populate interpolation kernel. ---*/
435  CSymmetricMatrix global_M(nVertexDonor);
436 
437  for (int iVertex = 0; iVertex < nVertexDonor; ++iVertex)
438  for (int jVertex = iVertex; jVertex < nVertexDonor; ++jVertex)
439  global_M(iVertex, jVertex) = SU2_TYPE::GetValue(
440  Get_RadialBasisValue(type, radius, GeometryToolbox::Distance(nDim, coords[iVertex], coords[jVertex])));
441 
442  /*--- Invert M matrix (operation is in-place). ---*/
443  const bool kernelIsSPD = (type == RADIAL_BASIS::WENDLAND_C2) || (type == RADIAL_BASIS::GAUSSIAN) ||
444  (type == RADIAL_BASIS::INV_MULTI_QUADRIC);
445  global_M.Invert(kernelIsSPD);
446 
447  /*--- Compute C_inv_trunc. ---*/
448  if (usePolynomial) {
449  /*--- Fill P matrix (P for points, with an extra top row of ones). ---*/
450  su2passivematrix P(1 + nDim, nVertexDonor);
451 
452  for (int iVertex = 0; iVertex < nVertexDonor; iVertex++) {
453  P(0, iVertex) = 1.0;
454  for (int iDim = 0; iDim < nDim; ++iDim) P(1 + iDim, iVertex) = SU2_TYPE::GetValue(coords(iVertex, iDim));
455  }
456 
457  /*--- Check if points lie on a plane and remove one coordinate from P if so. ---*/
458  nPolynomial = CheckPolynomialTerms(interfaceCoordTol, keepPolynomialRow, P);
459 
460  /*--- Compute Q = P * M^-1 ---*/
461  su2passivematrix Q;
462  global_M.MatMatMult('R', P, Q);
463 
464  /*--- Compute Mp = (Q * P^T)^-1 ---*/
465  CSymmetricMatrix Mp(nPolynomial + 1);
466 
467  for (int i = 0; i <= nPolynomial; ++i)
468  for (int j = i; j <= nPolynomial; ++j) {
469  Mp(i, j) = 0.0;
470  for (int k = 0; k < nVertexDonor; ++k) Mp(i, j) += Q(i, k) * P(j, k);
471  }
472  Mp.Invert(false);
473 
474  /*--- Compute M_p * Q, the top part of C_inv_trunc. ---*/
475  su2passivematrix C_inv_top;
476  Mp.MatMatMult('L', Q, C_inv_top);
477 
478  /*--- Compute M^-1 - Q^T * C_inv_top, the bottom (main) part of C_inv_trunc. ---*/
479  su2passivematrix C_inv_bot = global_M.StealData();
480 #ifdef HAVE_LAPACK
481  /*--- Order of gemm arguments swapped due to row-major storage. ---*/
482  const char opa = 'N', opb = 'T';
483  const int M = nVertexDonor, N = nVertexDonor, K = nPolynomial + 1;
484  // lda = C_inv_top.cols() = M; ldb = Q.cols() = M; ldc = C_inv_bot.cols() = M;
485  const passivedouble alpha = -1.0, beta = 1.0;
486  DGEMM(&opa, &opb, &M, &N, &K, &alpha, C_inv_top[0], &M, Q[0], &M, &beta, C_inv_bot[0], &M);
487 #else // naive product
488  for (int i = 0; i < nVertexDonor; ++i)
489  for (int j = 0; j < nVertexDonor; ++j)
490  for (int k = 0; k <= nPolynomial; ++k) C_inv_bot(i, j) -= Q(k, i) * C_inv_top(k, j);
491 #endif
492 
493  /*--- Merge top and bottom of C_inv_trunc. More intrusive memory
494  * management, or separate handling of top and bottom, would
495  * avoid these copies (and associated temporary vars). ---*/
496  C_inv_trunc.resize(1 + nPolynomial + nVertexDonor, nVertexDonor);
497  memcpy(C_inv_trunc[0], C_inv_top.data(), C_inv_top.size() * sizeof(passivedouble));
498  memcpy(C_inv_trunc[1 + nPolynomial], C_inv_bot.data(), C_inv_bot.size() * sizeof(passivedouble));
499  } else {
500  /*--- No polynomial term used in the interpolation, C_inv_trunc = M^-1. ---*/
501

503

507 int CRadialBasisFunction::CheckPolynomialTerms(su2double max_diff_tol, vector<int>& keep_row, su2passivematrix& P) {
508  const int m = P.rows();
509  const int n = P.cols();
510 
511  /*--- The first row of P is all ones and we do not care about it for this analysis. ---*/
512  const int n_rows = m - 1;
513  keep_row.resize(n_rows);
514 
515  /*--- By default assume points are not on a plane (all rows kept). ---*/
516  int n_polynomial = n_rows;
517  keep_row.resize(n_rows);
518  for (int i = 0; i < n_rows; ++i) keep_row[i] = 1;
519 
520  /*--- Fit a plane through the points in P. ---*/
521 
522  /*--- Compute P times its transpose and invert. ---*/
523  CSymmetricMatrix PPT(n_rows);
524 
525  for (int i = 0; i < n_rows; ++i)
526  for (int j = i; j < n_rows; ++j) {
527  PPT(i, j) = 0.0;
528  for (int k = 0; k < n; ++k) PPT(i, j) += P(i + 1, k) * P(j + 1, k);
529  }
530  PPT.Invert(true);
531 
532  /*--- RHS for the least squares fit (vector of ones times P). ---*/
533  vector<passivedouble> rhs(n_rows, 0.0), coeff(n_rows);
534 
535  for (int i = 0; i < n_rows; ++i)
536  for (int j = 0; j < n; ++j) rhs[i] += P(i + 1, j);
537 
538  /*--- Multiply the RHS by the inverse thus obtaining the coefficients. ---*/
539  PPT.MatVecMult(rhs.begin(), coeff.begin());
540 
541  /*--- Determine the maximum deviation of the points from the fitted plane. ---*/
542  passivedouble max_diff = 0.0;
543 
544  for (int j = 0; j < n; ++j) {
545  passivedouble sum = 0.0;
546  for (int i = 0; i < n_rows; ++i) sum += coeff[i] * P(i + 1, j);
547 
548  /*--- 1.0 is the arbitrary constant we are assuming when fitting
549  the plane, i.e. the vector of ones used to generate the RHS. ---*/
550  max_diff = max(abs(1.0 - sum), max_diff);
551  }
552 
553  /*--- If points lie on plane remove row associated with the maximum coefficient. ---*/
554  if (max_diff < max_diff_tol) {
555  /*--- Find the max coeff and mark the corresponding row for removal. ---*/
556  int remove_row = 0;
557  for (int i = 1; i < n_rows; ++i)
558  if (abs(coeff[i]) > abs(coeff[remove_row])) remove_row = i;
559 
560  /*--- Mark row as removed and adjust number of polynomial terms. ---*/
561  n_polynomial = n_rows - 1;
562  keep_row[remove_row] = 0;
563 
564  /*--- Truncated P by shifting rows "up". ---*/
565  for (auto i = remove_row + 1; i < m - 1; ++i)
566  for (int j = 0; j < n; ++j) P(i, j) = P(i + 1, j);
567  }
568 
569  return n_polynomial;
570 }
