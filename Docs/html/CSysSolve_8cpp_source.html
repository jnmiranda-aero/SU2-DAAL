64template <class ScalarType>
65void CSysSolve<ScalarType>::ApplyGivens(ScalarType s, ScalarType c, ScalarType& h1, ScalarType& h2) const {
66 ScalarType temp = c * h1 + s * h2;
67 h2 = c * h2 - s * h1;
68 h1 = temp;
69}

71template <class ScalarType>
72void CSysSolve<ScalarType>::GenerateGivens(ScalarType& dx, ScalarType& dy, ScalarType& s, ScalarType& c) const {
73 if ((dx == 0.0) && (dy == 0.0)) {
74 c = 1.0;
75 s = 0.0;
76 } else if (fabs(dy) > fabs(dx)) {
77 ScalarType tmp = dx / dy;
78 dx = sqrt(1.0 + tmp * tmp);
79 s = Sign(1.0 / dx, dy);
80 c = tmp * s;
81 } else if (fabs(dy) <= fabs(dx)) {
82 ScalarType tmp = dy / dx;
83 dy = sqrt(1.0 + tmp * tmp);
84 c = Sign(1.0 / dy, dx);
85 s = tmp * c;
86 } else {
87 // dx and/or dy must be invalid
88 dx = 0.0;
89 dy = 0.0;
90 c = 1.0;
91 s = 0.0;
92 }
93 dx = fabs(dx * dy);
94 dy = 0.0;
95}

97template <class ScalarType>
98void CSysSolve<ScalarType>::SolveReduced(int n, const su2matrix<ScalarType>& Hsbg, const su2vector<ScalarType>& rhs,
99 su2vector<ScalarType>& x) const {
100 // initialize...
101 for (int i = 0; i < n; i++) x[i] = rhs[i];
102 // ... and backsolve
103 for (int i = n - 1; i >= 0; i--) {
104 x[i] /= Hsbg(i, i);
105 for (int j = i - 1; j >= 0; j--) {
106 x[j] -= Hsbg(j, i) * x[i];
107 }
108 }
109}

111template <class ScalarType>
112void CSysSolve<ScalarType>::ModGramSchmidt(bool shared_hsbg, int i, su2matrix<ScalarType>& Hsbg,
113 vector<CSysVector<ScalarType> >& w) const {
114 const auto thread = omp_get_thread_num();
115
116 /*--- If Hsbg is shared by multiple threads calling this function, only one
117 * thread can write into it. If Hsbg is private, all threads need to write. ---*/
118
119 auto SetHsbg = [&](int row, int col, const ScalarType& value) {
120 if (!shared_hsbg || thread == 0) {
121 Hsbg(row, col) = value;
122 }
123 };
124
125 /*--- Parameter for reorthonormalization ---*/
126
127 const ScalarType reorth = 0.98;
128
129 /*--- Get the norm of the vector being orthogonalized, and find the
130 threshold for re-orthogonalization ---*/
131
132 ScalarType nrm = w[i + 1].squaredNorm();
133 ScalarType thr = nrm * reorth;
134
135 /*--- The norm of w[i+1] < 0.0 or w[i+1] = NaN ---*/
136
137 if ((nrm <= 0.0) || (nrm != nrm)) {
138 /*--- nrm is the result of a dot product, communications are implicitly handled. ---*/
139 SU2_MPI::Error("FGMRES orthogonalization failed, linear solver diverged.", CURRENT_FUNCTION);
140 }
141
142 /*--- Begin main Gram-Schmidt loop ---*/
143
144 for (int k = 0; k < i + 1; k++) {
145 ScalarType prod = w[i + 1].dot(w[k]);
146 ScalarType h_ki = prod;
147 w[i + 1] -= prod * w[k];
148
149 /*--- Check if reorthogonalization is necessary ---*/
150
151 if (prod * prod > thr) {
152 prod = w[i + 1].dot(w[k]);
153 h_ki += prod;
154 w[i + 1] -= prod * w[k];
155 }
156 SetHsbg(k, i, h_ki);
157
158 /*--- Update the norm and check its size ---*/
159
160 nrm -= pow(h_ki, 2);
161 nrm = max<ScalarType>(nrm, 0.0);
162 thr = nrm * reorth;
163 }
164
165 /*--- Test the resulting vector ---*/
166
167 nrm = w[i + 1].norm();
168 SetHsbg(i + 1, i, nrm);
169
170 /*--- Scale the resulting vector ---*/
171
172 w[i + 1] /= nrm;
173}

201template <class ScalarType>
202unsigned long CSysSolve<ScalarType>::CG_LinSolver(const CSysVector<ScalarType>& b, CSysVector<ScalarType>& x,
203 const CMatrixVectorProduct<ScalarType>& mat_vec,
204 const CPreconditioner<ScalarType>& precond, ScalarType tol,
205 unsigned long m, ScalarType& residual, bool monitoring,
206 const CConfig* config) const {
207 const bool masterRank = (SU2_MPI::GetRank() == MASTER_NODE);
208 ScalarType norm_r = 0.0, norm0 = 0.0;
209 unsigned long i = 0;
210
211 /*--- Check the subspace size ---*/
212
213 if (m < 1) {
214 SU2_MPI::Error("Number of linear solver iterations must be greater than 0.", CURRENT_FUNCTION);
215 }
216
217 /*--- Allocate if not allocated yet, only one thread can
218 * do this since the working vectors are shared. ---*/
219
220 if (!cg_ready) {
221 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
222 auto nVar = b.GetNVar();
223 auto nBlk = b.GetNBlk();
224 auto nBlkDomain = b.GetNBlkDomain();
225
226 A_x.Initialize(nBlk, nBlkDomain, nVar, nullptr);
227 r.Initialize(nBlk, nBlkDomain, nVar, nullptr);
228 z.Initialize(nBlk, nBlkDomain, nVar, nullptr);
229 p.Initialize(nBlk, nBlkDomain, nVar, nullptr);
230
231 cg_ready = true;
232 }
233 END_SU2_OMP_SAFE_GLOBAL_ACCESS
234 }
235
236 /*--- Calculate the initial residual, compute norm, and check if system is already solved ---*/
237
238 if (!xIsZero) {
239 mat_vec(x, A_x);
240 r = b - A_x;
241 } else {
242 r = b;
243 }
244
245 /*--- Only compute the residuals in full communication mode. ---*/
246
247 if (config->GetComm_Level() == COMM_FULL) {
248 norm_r = r.norm();
249 norm0 = b.norm();
250
251 /*--- Set the norm to the initial initial residual value ---*/
252
253 if (tol_type == LinearToleranceType::RELATIVE) norm0 = norm_r;
254
255 if ((norm_r < tol * norm0) || (norm_r < eps)) {
256 if (masterRank && (lin_sol_mode != LINEAR_SOLVER_MODE::MESH_DEFORM)) {
257 SU2_OMP_MASTER
258 cout << "CSysSolve::ConjugateGradient(): system solved by initial guess." << endl;
259 END_SU2_OMP_MASTER
260 }
261 return 0;
262 }
263
264 /*--- Output header information including initial residual ---*/
265
266 if (monitoring && masterRank) {
267 SU2_OMP_MASTER {
268 WriteHeader("CG", tol, norm_r);
269 WriteHistory(i, norm_r / norm0);
270 }
271 END_SU2_OMP_MASTER
272 }
273 }
274
275 precond(r, z);
276 p = z;
277 ScalarType r_dot_z = r.dot(z);
278
279 /*--- Loop over all search directions ---*/
280
281 for (i = 0; i < m; i++) {
282 /*--- Apply matrix to p to build Krylov subspace ---*/
283
284 mat_vec(p, A_x);
285
286 /*--- Calculate step-length alpha ---*/
287
288 ScalarType alpha = r_dot_z / A_x.dot(p);
289
290 /*--- Update solution and residual: ---*/
291
292 x += alpha * p;
293 r -= alpha * A_x;
294
295 /*--- Only compute the residuals in full communication mode. ---*/
296
297 if (config->GetComm_Level() == COMM_FULL) {
298 /*--- Check if solution has converged, else output the relative residual if necessary ---*/
299
300 norm_r = r.norm();
301 if (norm_r < tol * norm0) break;
302 if (((monitoring) && (masterRank)) && ((i + 1) % monitorFreq == 0)) {
303 SU2_OMP_MASTER
304 WriteHistory(i + 1, norm_r / norm0);
305 END_SU2_OMP_MASTER
306 }
307 }
308
309 precond(r, z);
310
311 /*--- Calculate Gram-Schmidt coefficient, beta = (r_{i+1}, z_{i+1}) / (r_{i}, z_{i}) ---*/
312
313 ScalarType beta = r_dot_z;
314 r_dot_z = r.dot(z);
315 beta = r_dot_z / beta;
316
317 /*--- Gram-Schmidt orthogonalization. ---*/
318
319 p = beta * p + z;
320 }
321
322 /*--- Recalculate final residual (this should be optional) ---*/
323
324 if ((monitoring) && (config->GetComm_Level() == COMM_FULL)) {
325 if (masterRank) {
326 SU2_OMP_MASTER
327 WriteFinalResidual("CG", i, norm_r / norm0);
328 END_SU2_OMP_MASTER
329 }
330
331 if (recomputeRes) {
332 mat_vec(x, A_x);
333 r = b - A_x;
334 ScalarType true_res = r.norm();
335
336 if (fabs(true_res - norm_r) > tol * 10.0) {
337 if (masterRank) {
338 SU2_OMP_MASTER
339 WriteWarning(norm_r, true_res, tol);
340 END_SU2_OMP_MASTER
341 }
342 }
343 }
344 }
345
346 residual = norm_r / norm0;
347 return i;
348}

350template <class ScalarType>
351unsigned long CSysSolve<ScalarType>::FGMRES_LinSolver(const CSysVector<ScalarType>& b, CSysVector<ScalarType>& x,
352 const CMatrixVectorProduct<ScalarType>& mat_vec,
353 const CPreconditioner<ScalarType>& precond, ScalarType tol,
354 unsigned long m, ScalarType& residual, bool monitoring,
355 const CConfig* config) const {
356 const bool masterRank = (SU2_MPI::GetRank() == MASTER_NODE);
357 const bool flexible = !precond.IsIdentity();
358 /*--- If we call the solver outside of a parallel region, but the number of threads allows,
359 * we still want to parallelize some of the expensive operations. ---*/
360 const bool nestedParallel = !omp_in_parallel() && omp_get_max_threads() > 1;
361
362 /*--- Check the subspace size ---*/
363
364 if (m < 1) {
365 SU2_MPI::Error("Number of linear solver iterations must be greater than 0.", CURRENT_FUNCTION);
366 }
367
368 if (m > 5000) {
369 SU2_MPI::Error("FGMRES subspace is too large.", CURRENT_FUNCTION);
370 }
371
372 /*--- Allocate if not allocated yet ---*/
373
374 if (W.size() <= m || (flexible && Z.size() <= m)) {
375 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
376 W.resize(m + 1);
377 for (auto& w : W) w.Initialize(x.GetNBlk(), x.GetNBlkDomain(), x.GetNVar(), nullptr);
378 if (flexible) {
379 Z.resize(m + 1);
380 for (auto& z : Z) z.Initialize(x.GetNBlk(), x.GetNBlkDomain(), x.GetNVar(), nullptr);
381 }
382 }
383 END_SU2_OMP_SAFE_GLOBAL_ACCESS
384 }
385
386 /*--- Define various arrays. In parallel, each thread of each rank has and works
387 on its own thread, since calculations on these arrays are based on dot products
388 (reduced across all threads and ranks) all threads do the same computations. ---*/
389
390 su2vector<ScalarType> g(m + 1), sn(m + 1), cs(m + 1), y(m);
391 g = ScalarType(0);
392 sn = ScalarType(0);
393 cs = ScalarType(0);
394 y = ScalarType(0);
395 su2matrix<ScalarType> H(m + 1, m);
396 H = ScalarType(0);
397
398 /*--- Calculate the norm of the rhs vector. ---*/
399
400 ScalarType norm0 = b.norm();
401
402 /*--- Calculate the initial residual (actually the negative residual) and compute its norm. ---*/
403
404 if (!xIsZero) {
405 mat_vec(x, W[0]);
406 W[0] -= b;
407 } else {
408 W[0] = -b;
409 }
410
411 ScalarType beta = W[0].norm();
412
413 /*--- Set the norm to the initial initial residual value ---*/
414
415 if (tol_type == LinearToleranceType::RELATIVE) norm0 = beta;
416
417 if ((beta < tol * norm0) || (beta < eps)) {
418 /*--- System is already solved ---*/
419
420 if (masterRank) {
421 SU2_OMP_MASTER
422 cout << "CSysSolve::FGMRES(): system solved by initial guess." << endl;
423 END_SU2_OMP_MASTER
424 }
425 residual = beta;
426 return 0;
427 }
428
429 /*--- Normalize residual to get w_{0} (the negative sign is because w[0]
430 holds the negative residual, as mentioned above). ---*/
431
432 W[0] /= -beta;
433
434 /*--- Initialize the RHS of the reduced system ---*/
435
436 g[0] = beta;
437
438 /*--- Output header information including initial residual ---*/
439
440 unsigned long i = 0;
441 if ((monitoring) && (masterRank)) {
442 SU2_OMP_MASTER {
443 WriteHeader("FGMRES", tol, beta);
444 WriteHistory(i, beta / norm0);
445 }
446 END_SU2_OMP_MASTER
447 }
448
449 /*--- Loop over all search directions ---*/
450
451 for (i = 0; i < m; i++) {
452 /*--- Check if solution has converged ---*/
453
454 if (beta < tol * norm0) break;
455
456 if (flexible) {
457 /*--- Precondition the CSysVector w[i] and store result in z[i] ---*/
458
459 precond(W[i], Z[i]);
460
461 /*--- Add to Krylov subspace ---*/
462
463 mat_vec(Z[i], W[i + 1]);
464 } else {
465 mat_vec(W[i], W[i + 1]);
466 }
467
468 /*--- Modified Gram-Schmidt orthogonalization ---*/
469
470 if (nestedParallel) {
471 /*--- "omp parallel if" does not work well here ---*/
472 SU2_OMP_PARALLEL
473 ModGramSchmidt(true, i, H, W);
474 END_SU2_OMP_PARALLEL
475 } else {
476 ModGramSchmidt(false, i, H, W);
477 }
478
479 /*--- Apply old Givens rotations to new column of the Hessenberg matrix then generate the
480 new Givens rotation matrix and apply it to the last two elements of H[:][i] and g ---*/
481
482 for (unsigned long k = 0; k < i; k++) ApplyGivens(sn[k], cs[k], H[k][i], H[k + 1][i]);
483 GenerateGivens(H[i][i], H[i + 1][i], sn[i], cs[i]);
484 ApplyGivens(sn[i], cs[i], g[i], g[i + 1]);
485
486 /*--- Set L2 norm of residual and check if solution has converged ---*/
487
488 beta = fabs(g[i + 1]);
489
490 /*--- Output the relative residual if necessary ---*/
491
492 if ((((monitoring) && (masterRank)) && ((i + 1) % monitorFreq == 0))) {
493 SU2_OMP_MASTER
494 WriteHistory(i + 1, beta / norm0);
495 END_SU2_OMP_MASTER
496 }
497 }
498
499 /*--- Solve the least-squares system and update solution ---*/
500
501 SolveReduced(i, H, g, y);
502
503 const auto& basis = flexible ? Z : W;
504
505 if (nestedParallel) {
506 SU2_OMP_PARALLEL
507 for (unsigned long k = 0; k < i; k++) x += y[k] * basis[k];
508 END_SU2_OMP_PARALLEL
509 } else {
510 for (unsigned long k = 0; k < i; k++) x += y[k] * basis[k];
511 }
512
513 /*--- Recalculate final (neg.) residual (this should be optional) ---*/
514
515 if ((monitoring) && (config->GetComm_Level() == COMM_FULL)) {
516 if (masterRank) {
517 SU2_OMP_MASTER
518 WriteFinalResidual("FGMRES", i, beta / norm0);
519 END_SU2_OMP_MASTER
520 }
521
522 if (recomputeRes) {
523 mat_vec(x, W[0]);
524 W[0] -= b;
525 ScalarType res = W[0].norm();
526
527 if (fabs(res - beta) > tol * 10) {
528 if (masterRank) {
529 SU2_OMP_MASTER
530 WriteWarning(beta, res, tol);
531 END_SU2_OMP_MASTER
532 }
533 }
534 }
535 }
536
537 residual = beta / norm0;
538 return i;
539}

541template <class ScalarType>
542unsigned long CSysSolve<ScalarType>::RFGMRES_LinSolver(const CSysVector<ScalarType>& b, CSysVector<ScalarType>& x,
543 const CMatrixVectorProduct<ScalarType>& mat_vec,
544 const CPreconditioner<ScalarType>& precond, ScalarType tol,
545 unsigned long MaxIter, ScalarType& residual, bool monitoring,
546 const CConfig* config) {
547 const auto restartIter = config->GetLinear_Solver_Restart_Frequency();
548
549 SU2_OMP_MASTER {
550 xIsZero = false;
551 tol_type = LinearToleranceType::ABSOLUTE;
552 }
553 END_SU2_OMP_MASTER
554
555 const ScalarType norm0 = b.norm(); // <- Has a barrier
556
557 for (auto totalIter = 0ul; totalIter < MaxIter;) {
558 /*--- Enforce a hard limit on total number of iterations ---*/
559 auto iterLimit = min(restartIter, MaxIter - totalIter);
560 auto iter = FGMRES_LinSolver(b, x, mat_vec, precond, tol, iterLimit, residual, monitoring, config);
561 totalIter += iter;
562 if (residual <= tol * norm0 || iter < iterLimit) return totalIter;
563 }
564 return 0;
565}

567template <class ScalarType>
568unsigned long CSysSolve<ScalarType>::BCGSTAB_LinSolver(const CSysVector<ScalarType>& b, CSysVector<ScalarType>& x,
569 const CMatrixVectorProduct<ScalarType>& mat_vec,
570 const CPreconditioner<ScalarType>& precond, ScalarType tol,
571 unsigned long m, ScalarType& residual, bool monitoring,
572 const CConfig* config) const {
573 const bool masterRank = (SU2_MPI::GetRank() == MASTER_NODE);
574 ScalarType norm_r = 0.0, norm0 = 0.0;
575 unsigned long i = 0;
576
577 /*--- Check the subspace size ---*/
578
579 if (m < 1) {
580 SU2_MPI::Error("Number of linear solver iterations must be greater than 0.", CURRENT_FUNCTION);
581 }
582
583 /*--- Allocate if not allocated yet ---*/
584
585 if (!bcg_ready) {
586 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
587 auto nVar = b.GetNVar();
588 auto nBlk = b.GetNBlk();
589 auto nBlkDomain = b.GetNBlkDomain();
590
591 A_x.Initialize(nBlk, nBlkDomain, nVar, nullptr);
592 r_0.Initialize(nBlk, nBlkDomain, nVar, nullptr);
593 r.Initialize(nBlk, nBlkDomain, nVar, nullptr);
594 p.Initialize(nBlk, nBlkDomain, nVar, nullptr);
595 v.Initialize(nBlk, nBlkDomain, nVar, nullptr);
596 z.Initialize(nBlk, nBlkDomain, nVar, nullptr);
597
598 bcg_ready = true;
599 }
600 END_SU2_OMP_SAFE_GLOBAL_ACCESS
601 }
602
603 /*--- Calculate the initial residual, compute norm, and check if system is already solved ---*/
604
605 if (!xIsZero) {
606 mat_vec(x, A_x);
607 r = b - A_x;
608 } else {
609 r = b;
610 }
611
612 /*--- Only compute the residuals in full communication mode. ---*/
613
614 if (config->GetComm_Level() == COMM_FULL) {
615 norm_r = r.norm();
616 norm0 = b.norm();
617
618 /*--- Set the norm to the initial initial residual value ---*/
619
620 if (tol_type == LinearToleranceType::RELATIVE) norm0 = norm_r;
621
622 if ((norm_r < tol * norm0) || (norm_r < eps)) {
623 if (masterRank) {
624 SU2_OMP_MASTER
625 cout << "CSysSolve::BCGSTAB(): system solved by initial guess." << endl;
626 END_SU2_OMP_MASTER
627 }
628 return 0;
629 }
630
631 /*--- Output header information including initial residual ---*/
632
633 if ((monitoring) && (masterRank)) {
634 SU2_OMP_MASTER {
635 WriteHeader("BCGSTAB", tol, norm_r);
636 WriteHistory(i, norm_r / norm0);
637 }
638 END_SU2_OMP_MASTER
639 }
640 }
641
642 /*--- Initialization ---*/
643
644 ScalarType alpha = 1.0, omega = 1.0, rho = 1.0, rho_prime = 1.0;
645 p = ScalarType(0.0);
646 v = ScalarType(0.0);
647 r_0 = r;
648
649 /*--- Loop over all search directions ---*/
650
651 for (i = 0; i < m; i++) {
652 /*--- Compute rho_prime ---*/
653
654 rho_prime = rho;
655
656 /*--- Compute rho_i ---*/
657
658 rho = r.dot(r_0);
659
660 /*--- Compute beta ---*/
661
662 ScalarType beta = (rho / rho_prime) * (alpha / omega);
663
664 /*--- Update p ---*/
665
666 p = beta * (p - omega * v) + r;
667
668 /*--- Preconditioning step ---*/
669
670 precond(p, z);
671 mat_vec(z, v);
672
673 /*--- Calculate step-length alpha ---*/
674
675 ScalarType r_0_v = r_0.dot(v);
676 alpha = rho / r_0_v;
677
678 /*--- Update solution and residual ---*/
679
680 x += alpha * z;
681 r -= alpha * v;
682
683 /*--- Preconditioning step ---*/
684
685 precond(r, z);
686 mat_vec(z, A_x);
687
688 /*--- Calculate step-length omega, avoid division by 0. ---*/
689
690 omega = A_x.squaredNorm();
691 if (omega == ScalarType(0)) break;
692 omega = A_x.dot(r) / omega;
693
694 /*--- Update solution and residual ---*/
695
696 x += omega * z;
697 r -= omega * A_x;
698
699 /*--- Only compute the residuals in full communication mode. ---*/
700
701 if (config->GetComm_Level() == COMM_FULL) {
702 /*--- Check if solution has converged, else output the relative residual if necessary ---*/
703
704 norm_r = r.norm();
705 if (norm_r < tol * norm0) break;
706 if (((monitoring) && (masterRank)) && ((i + 1) % monitorFreq == 0)) {
707 SU2_OMP_MASTER
708 WriteHistory(i + 1, norm_r / norm0);
709 END_SU2_OMP_MASTER
710 }
711 }
712 }
713
714 /*--- Recalculate final residual (this should be optional) ---*/
715
716 if ((monitoring) && (config->GetComm_Level() == COMM_FULL)) {
717 if (masterRank) {
718 SU2_OMP_MASTER
719 WriteFinalResidual("BCGSTAB", i, norm_r / norm0);
720 END_SU2_OMP_MASTER
721 }
722
723 if (recomputeRes) {
724 mat_vec(x, A_x);
725 r = b - A_x;
726 ScalarType true_res = r.norm();
727
728 if ((fabs(true_res - norm_r) > tol * 10.0) && (masterRank)) {
729 SU2_OMP_MASTER
730 WriteWarning(norm_r, true_res, tol);
731 END_SU2_OMP_MASTER
732 }
733 }
734 }
735
736 residual = norm_r / norm0;
737 return i;
738}

740template <class ScalarType>
741unsigned long CSysSolve<ScalarType>::Smoother_LinSolver(const CSysVector<ScalarType>& b, CSysVector<ScalarType>& x,
742 const CMatrixVectorProduct<ScalarType>& mat_vec,
743 const CPreconditioner<ScalarType>& precond, ScalarType tol,
744 unsigned long m, ScalarType& residual, bool monitoring,
745 const CConfig* config) const {
746 const bool masterRank = (SU2_MPI::GetRank() == MASTER_NODE);
747 const bool fix_iter_mode = tol < eps;
748 ScalarType norm_r = 0.0, norm0 = 0.0;
749 unsigned long i = 0;
750
751 /*--- Relaxation factor, see comments inside the loop over the smoothing iterations. ---*/
752 const ScalarType omega = SU2_TYPE::GetValue(config->GetLinear_Solver_Smoother_Relaxation());
753
754 if (m < 1) {
755 SU2_MPI::Error("Number of linear solver iterations must be greater than 0.", CURRENT_FUNCTION);
756 }
757
758 /*--- Allocate vectors for residual (r), solution increment (z), and matrix-vector
759 product (A_x), for the latter two this is done only on the first call to the method. ---*/
760
761 if (!smooth_ready) {
762 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
763 auto nVar = b.GetNVar();
764 auto nBlk = b.GetNBlk();
765 auto nBlkDomain = b.GetNBlkDomain();
766
767 A_x.Initialize(nBlk, nBlkDomain, nVar, nullptr);
768 r.Initialize(nBlk, nBlkDomain, nVar, nullptr);
769 z.Initialize(nBlk, nBlkDomain, nVar, nullptr);
770
771 smooth_ready = true;
772 }
773 END_SU2_OMP_SAFE_GLOBAL_ACCESS
774 }
775
776 /*--- Compute the initial residual and check if the system is already solved (if in COMM_FULL mode). ---*/
777
778 if (!xIsZero) {
779 mat_vec(x, A_x);
780 r = b - A_x;
781 } else {
782 r = b;
783 }
784
785 /*--- Only compute the residuals in full communication mode. ---*/
786
787 if (config->GetComm_Level() == COMM_FULL) {
788 norm_r = r.norm();
789 norm0 = b.norm();
790
791 /*--- Set the norm to the initial initial residual value ---*/
792
793 if (tol_type == LinearToleranceType::RELATIVE) norm0 = norm_r;
794
795 if ((norm_r < tol * norm0) || (norm_r < eps)) {
796 if (masterRank) {
797 SU2_OMP_MASTER
798 cout << "CSysSolve::Smoother_LinSolver(): system solved by initial guess." << endl;
799 END_SU2_OMP_MASTER
800 }
801 return 0;
802 }
803
804 /*--- Output header information including initial residual. ---*/
805
806 if ((monitoring) && (masterRank)) {
807 SU2_OMP_MASTER {
808 WriteHeader("Smoother", tol, norm_r);
809 WriteHistory(i, norm_r / norm0);
810 }
811 END_SU2_OMP_MASTER
812 }
813 }
814
815 /*--- Smoothing Iterations ---*/
816
817 for (i = 0; i < m; i++) {
818 /*--- Compute the solution increment (z), or "search" direction, by applying
819 the preconditioner to the residual, i.e. z = M^{-1} * r. ---*/
820
821 precond(r, z);
822
823 /*--- The increment will be added to the solution (with relaxation omega)
824 to update the residual, needed to compute the next increment, we get the
825 product of the matrix with the direction (into A_x) and subtract from the
826 current residual, the system is linear so this saves some computation
827 compared to re-evaluating r = b-A*x. ---*/
828
829 mat_vec(z, A_x);
830
831 /*--- Update solution and residual with relaxation omega. Mathematically this
832 is a modified Richardson iteration for the left-preconditioned system
833 M^{-1}(b-A*x) which converges if ||I-w*M^{-1}*A|| < 1. Combining this method
834 with a Gauss-Seidel preconditioner and w>1 is NOT equivalent to SOR. ---*/
835
836 x += omega * z;
837 r -= omega * A_x;
838
839 /*--- Only compute the residuals in full communication mode. ---*/
840 /*--- Check if solution has converged, else output the relative residual if necessary. ---*/
841
842 if (!fix_iter_mode && config->GetComm_Level() == COMM_FULL) {
843 norm_r = r.norm();
844 if (norm_r < tol * norm0) break;
845 if (((monitoring) && (masterRank)) && ((i + 1) % monitorFreq == 0)) {
846 SU2_OMP_MASTER
847 WriteHistory(i + 1, norm_r / norm0);
848 END_SU2_OMP_MASTER
849 }
850 }
851 }
852
853 if (fix_iter_mode) norm_r = r.norm();
854
855 if ((monitoring) && (masterRank) && (config->GetComm_Level() == COMM_FULL)) {
856 SU2_OMP_MASTER
857 WriteFinalResidual("Smoother", i, norm_r / norm0);
858 END_SU2_OMP_MASTER
859 }
860
861 residual = norm_r / norm0;
862 return i;
863}

865template <class ScalarType>
866unsigned long CSysSolve<ScalarType>::Solve(CSysMatrix<ScalarType>& Jacobian, const CSysVector<su2double>& LinSysRes,
867 CSysVector<su2double>& LinSysSol, CGeometry* geometry,
868 const CConfig* config) {
869 /*---
870 A word about the templated types. It is assumed that the residual and solution vectors are always of su2doubles,
871 meaning that they are active in the discrete adjoint. The same assumption is made in SetExternalSolve.
872 When the Jacobian is passive (and therefore not compatible with the vectors) we go through the "HandleTemporaries"
873 mechanisms. Note that CG, BCGSTAB, and FGMRES, all expect the vector to be compatible with the Product and
874 Preconditioner (and therefore with the Matrix). Likewise for Solve_b (which is used by CSysSolve_b).
875 There are no provisions here for active Matrix and passive Vectors as that makes no sense since we only handle the
876 derivatives of the residual in CSysSolve_b.
877 ---*/
878
879 unsigned short KindSolver, KindPrecond;
880 unsigned long MaxIter;
881 ScalarType SolverTol;
882 bool ScreenOutput;
883
884 switch (lin_sol_mode) {
885 /*--- Mesh Deformation mode ---*/
886 case LINEAR_SOLVER_MODE::MESH_DEFORM: {
887 KindSolver = config->GetKind_Deform_Linear_Solver();
888 KindPrecond = config->GetKind_Deform_Linear_Solver_Prec();
889 MaxIter = config->GetDeform_Linear_Solver_Iter();
890 SolverTol = SU2_TYPE::GetValue(config->GetDeform_Linear_Solver_Error());
891 ScreenOutput = config->GetDeform_Output();
892 break;
893 }
894
895 /*--- Gradient Smoothing mode ---*/
896 case LINEAR_SOLVER_MODE::GRADIENT_MODE: {
897 KindSolver = config->GetKind_Grad_Linear_Solver();
898 KindPrecond = config->GetKind_Grad_Linear_Solver_Prec();
899 MaxIter = config->GetGrad_Linear_Solver_Iter();
900 SolverTol = SU2_TYPE::GetValue(config->GetGrad_Linear_Solver_Error());
901 ScreenOutput = true;
902 break;
903 }
904
905 /*--- Normal mode
906 * assumes that 'lin_sol_mode==LINEAR_SOLVER_MODE::STANDARD', but does not enforce it to avoid compiler warning.
907 * ---*/
908 default: {
909 KindSolver = config->GetKind_Linear_Solver();
910 KindPrecond = config->GetKind_Linear_Solver_Prec();
911 MaxIter = config->GetLinear_Solver_Iter();
912 SolverTol = SU2_TYPE::GetValue(config->GetLinear_Solver_Error());
913 ScreenOutput = false;
914 break;
915 }
916 }
917
918 /*--- Stop the recording for the linear solver ---*/
919 bool TapeActive = NO;
920
921 if (config->GetDiscrete_Adjoint()) {
922#ifdef CODI_REVERSE_TYPE
923
924 TapeActive = AD::TapeActive();
925
926 /*--- Declare external function inputs, outputs, and data ---*/
927 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
928 AD::SetExtFuncIn(&LinSysRes[0], LinSysRes.GetLocSize());
929 AD::SetExtFuncOut(&LinSysSol[0], LinSysSol.GetLocSize());
930 AD::FuncHelper.addUserData(&LinSysRes);
931 AD::FuncHelper.addUserData(&LinSysSol);
932 AD::FuncHelper.addUserData(&Jacobian);
933 AD::FuncHelper.addUserData(geometry);
934 AD::FuncHelper.addUserData(config);
935 AD::FuncHelper.addUserData(this);
936 }
937 END_SU2_OMP_SAFE_GLOBAL_ACCESS
938#endif
939 }
940
941 unsigned long IterLinSol = 0;
942
943 /*--- Declaration of the external function ---*/
944 auto externalFunction = [&]() {
945 /*--- Create matrix-vector product, preconditioner, and solve the linear system ---*/
946
947 HandleTemporariesIn(LinSysRes, LinSysSol);
948
949 auto mat_vec = CSysMatrixVectorProduct<ScalarType>(Jacobian, geometry, config);
950
951 const auto kindPrec = static_cast<ENUM_LINEAR_SOLVER_PREC>(KindPrecond);
952
953 auto precond = CPreconditioner<ScalarType>::Create(kindPrec, Jacobian, geometry, config);
954
955 /*--- Build preconditioner. ---*/
956
957 precond->Build();
958
959 /*--- Solve system. ---*/
960
961 ScalarType residual = 0.0;
962
963 switch (KindSolver) {
964 case BCGSTAB:
965 IterLinSol = BCGSTAB_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual,
966 ScreenOutput, config);
967 break;
968 case FGMRES:
969 IterLinSol = FGMRES_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual,
970 ScreenOutput, config);
971 break;
972 case RESTARTED_FGMRES:
973 IterLinSol = RFGMRES_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual,
974 ScreenOutput, config);
975 break;
976 case CONJUGATE_GRADIENT:
977 IterLinSol = CG_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual,
978 ScreenOutput, config);
979 break;
980 case SMOOTHER:
981 IterLinSol = Smoother_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual,
982 ScreenOutput, config);
983 break;
984 case PASTIX_LDLT:
985 case PASTIX_LU:
986 Jacobian.BuildPastixPreconditioner(geometry, config, KindSolver);
987 Jacobian.ComputePastixPreconditioner(*LinSysRes_ptr, *LinSysSol_ptr, geometry, config);
988 IterLinSol = 1;
989 residual = 1e-20;
990 break;
991 default:
992 SU2_MPI::Error("Unknown type of linear solver.", CURRENT_FUNCTION);
993 }
994
995 SU2_OMP_MASTER {
996 Residual = residual;
997 Iterations = IterLinSol;
998 }
999 END_SU2_OMP_MASTER
1000
1001 HandleTemporariesOut(LinSysSol);
1002
1003 delete precond;
1004
1005 if (TapeActive) {
1006 /*--- To keep the behavior of SU2_DOT, but not strictly required since jacobian is symmetric(?). ---*/
1007 const bool RequiresTranspose =
1008 ((lin_sol_mode != LINEAR_SOLVER_MODE::MESH_DEFORM) || (config->GetKind_SU2() == SU2_COMPONENT::SU2_DOT));
1009
1010 if (lin_sol_mode == LINEAR_SOLVER_MODE::MESH_DEFORM)
1011 KindPrecond = config->GetKind_Deform_Linear_Solver_Prec();
1012 else if (lin_sol_mode == LINEAR_SOLVER_MODE::GRADIENT_MODE)
1013 KindPrecond = config->GetKind_Grad_Linear_Solver_Prec();
1014 else
1015 KindPrecond = config->GetKind_DiscAdj_Linear_Prec();
1016
1017 /*--- Build preconditioner for the transposed Jacobian ---*/
1018
1019 if (RequiresTranspose) Jacobian.TransposeInPlace();
1020
1021 switch (KindPrecond) {
1022 case ILU:
1023 if (RequiresTranspose) Jacobian.BuildILUPreconditioner();
1024 break;
1025 case JACOBI:
1026 case LINELET:
1027 if (RequiresTranspose) Jacobian.BuildJacobiPreconditioner();
1028 break;
1029 case LU_SGS:
1030 /*--- Nothing to build. ---*/
1031 break;
1032 case PASTIX_ILU:
1033 case PASTIX_LU_P:
1034 case PASTIX_LDLT_P:
1035 /*--- It was already built. ---*/
1036 break;
1037 default:
1038 SU2_MPI::Error("The specified preconditioner is not yet implemented for the discrete adjoint method.",
1039 CURRENT_FUNCTION);
1040 break;
1041 }
1042 }
1043 }; /*--- Finish declaration of the external function ---*/
1044
1045#ifdef CODI_REVERSE_TYPE
1046 /*--- Call the external function with appropriate AD handling ---*/
1047 AD::FuncHelper.callPrimalFuncWithADType(externalFunction);
1048
1049 AD::FuncHelper.addToTape(CSysSolve_b<ScalarType>::Solve_b);
1050#else
1051 /*--- Without reverse AD, call the external function directly ---*/
1052 externalFunction();
1053#endif
1054
1055 return IterLinSol;
1056}

1086 }

1090 * ---*/

1125 ScreenOutput, config);
1126 break;
1127 case RESTARTED_FGMRES:
1128 IterLinSol = RFGMRES_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual,
1129 ScreenOutput, config);
1130 break;
1131 case BCGSTAB:

1133 ScreenOutput, config);
1134 break;
1135 case CONJUGATE_GRADIENT:
1136 IterLinSol = CG_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual,
1137 ScreenOutput, config);
1138 break;
1139 case SMOOTHER:

1141 ScreenOutput, config);
1142 break;
1143 case PASTIX_LDLT:
1144 case PASTIX_LU:
1145 if (directCall) Jacobian.BuildPastixPreconditioner(geometry, config, KindSolver);
1146 Jacobian.ComputePastixPreconditioner(*LinSysRes_ptr, *LinSysSol_ptr, geometry, config);
1147 IterLinSol = 1;
1148 residual = 1e-20;
1149 break;
1150 default:

1152 break;
1153 }

1156
1157 delete precond;
1158
1159 SU2_OMP_MASTER {
1160 Residual = residual;
1161 Iterations = IterLinSol;
1162 }
1163 END_SU2_OMP_MASTER

1166}
