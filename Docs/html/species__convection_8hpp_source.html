61  /*!
62  * \brief Species transport specific steps in the ComputeResidual method
63  * \param[in] config - Definition of the particular problem.
64  */
65  void FinishResidualCalc(const CConfig* config) override {
66  for (auto iVar = 0u; iVar < nVar; iVar++) {
67  Flux[iVar] = a0 * V_i[idx.Density()] * ScalarVar_i[iVar] + a1 * V_j[idx.Density()] * ScalarVar_j[iVar];
68 
69  /*--- Jacobians are taken wrt rho*Y not Y alone in the species solver. ---*/
70  /*--- Off-diagonal entries are zero. ---*/
71  Jacobian_i[iVar][iVar] = a0;
72  Jacobian_j[iVar][iVar] = a1;
73  } // iVar
74  }
