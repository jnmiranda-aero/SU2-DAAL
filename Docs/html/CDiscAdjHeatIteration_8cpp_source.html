31 void CDiscAdjHeatIteration::Preprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
32  CSolver***** solver, CNumerics****** numerics, CConfig** config,
33  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
34  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
35 
36  const auto TimeIter = config[val_iZone]->GetTimeIter();
37  const bool dual_time_1st = (config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
38  const bool dual_time_2nd = (config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
39  const bool dual_time = (dual_time_1st || dual_time_2nd);
40 
41  auto solvers = solver[val_iZone][val_iInst];
42 
43  /*--- For the unsteady adjoint, load direct solutions from restart files. ---*/
44 
45  if (config[val_iZone]->GetTime_Marching() != TIME_MARCHING::STEADY) {
46  const int Direct_Iter = static_cast<int>(config[val_iZone]->GetUnst_AdjointIter()) -
47  static_cast<int>(TimeIter) - 2 + dual_time;
48 
49  /*--- For dual-time stepping we want to load the already converged solution at previous timesteps.
50  * In general we only load one file and shift the previously loaded solutions, on the first we
51  * load one or two more (depending on dual time order). ---*/
52 
53  if (dual_time_2nd) {
54  LoadUnsteady_Solution(geometry, solver, config, val_iZone, val_iInst, Direct_Iter - 2);
55  } else if (dual_time_1st) {
56  LoadUnsteady_Solution(geometry, solver, config, val_iZone, val_iInst, Direct_Iter - 1);
57  }
58 
59  if (TimeIter == 0) {
60  /*--- Push solution back one level. ---*/
61  if (dual_time) {
62  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
63  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n();
64  }
65  }
66 
67  /*--- If required load another time step. Push the previous time step to n-1 and the
68  loaded time step to n. ---*/
69  if (dual_time_2nd) {
70  LoadUnsteady_Solution(geometry, solver, config, val_iZone, val_iInst, Direct_Iter - 1);
71 
72  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
73  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n1();
74  solvers[iMesh][HEAT_SOL]->GetNodes()->Set_Solution_time_n();
75  }
76  }
77 
78  /*--- Load current solution. ---*/
79  LoadUnsteady_Solution(geometry, solver, config, val_iZone, val_iInst, Direct_Iter);
80  }
81  if ((TimeIter > 0) && dual_time) {
82  /*--- Temporarily store the loaded solution in the Solution_Old array ---*/
83  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++)
84  solvers[iMesh][HEAT_SOL]->Set_OldSolution();
85 
86  /*--- Move timestep n to current solution. ---*/
87  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
88  for (auto iPoint = 0ul; iPoint < geometry[val_iZone][val_iInst][iMesh]->GetnPoint(); iPoint++) {
89  solvers[iMesh][HEAT_SOL]->GetNodes()->SetSolution(
90  iPoint, solvers[iMesh][HEAT_SOL]->GetNodes()->GetSolution_time_n(iPoint));
91  }
92  }
93 
94  /*--- Finally, place the loaded solution in the correct place (n or n-1 depending on order). ---*/
95  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
96  auto* heatSol = solvers[iMesh][HEAT_SOL];
97  for (auto iPoint = 0ul; iPoint < geometry[val_iZone][val_iInst][iMesh]->GetnPoint(); iPoint++) {
98  if (dual_time_2nd) {
99  /*--- If required also move timestep n-1 to timestep n. ---*/
100  heatSol->GetNodes()->Set_Solution_time_n(iPoint, heatSol->GetNodes()->GetSolution_time_n1(iPoint));
101  heatSol->GetNodes()->Set_Solution_time_n1(iPoint, heatSol->GetNodes()->GetSolution_Old(iPoint));
102  } else {
103  heatSol->GetNodes()->Set_Solution_time_n(iPoint, heatSol->GetNodes()->GetSolution_Old(iPoint));
104  }
105  }
106  }
107  }
108  }
109 
110  /*--- Store flow solution also in the adjoint solver in order to be able to reset it later ---*/
111 
112  if (TimeIter == 0 || dual_time) {
113  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
114  for (auto iPoint = 0ul; iPoint < geometry[val_iZone][val_iInst][iMesh]->GetnPoint(); iPoint++) {
115  solvers[iMesh][ADJHEAT_SOL]->GetNodes()->SetSolution_Direct(
116  iPoint, solvers[iMesh][HEAT_SOL]->GetNodes()->GetSolution(iPoint));
117  }
118  }
119  }
120 
121  solvers[MESH_0][ADJHEAT_SOL]->Preprocessing(
122  geometry[val_iZone][val_iInst][MESH_0], solvers[MESH_0], config[val_iZone], MESH_0, 0, RUNTIME_ADJHEAT_SYS, false);
123 }

125 void CDiscAdjHeatIteration::LoadUnsteady_Solution(CGeometry**** geometry, CSolver***** solver, CConfig** config,
126  unsigned short val_iZone, unsigned short val_iInst,
127  int val_DirectIter) {
128  auto solvers = solver[val_iZone][val_iInst];
129  auto geometries = geometry[val_iZone][val_iInst];
130 
131  if (val_DirectIter >= 0) {
132  if (rank == MASTER_NODE)
133  cout << " Loading heat solution from direct iteration " << val_DirectIter << " for zone " << val_iZone << "." << endl;
134 
135  solvers[MESH_0][HEAT_SOL]->LoadRestart(geometries, solvers, config[val_iZone], val_DirectIter, false);
136  }
137  else {
138  /*--- If there is no solution file we set the freestream condition ---*/
139  if (rank == MASTER_NODE)
140  cout << " Setting freestream conditions at direct iteration " << val_DirectIter << " for zone " << val_iZone << "." << endl;
141 
142  for (auto iMesh = 0u; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
143  solvers[iMesh][HEAT_SOL]->SetFreeStream_Solution(config[val_iZone]);
144  solvers[iMesh][HEAT_SOL]->Postprocessing(geometries[iMesh], solvers[iMesh], config[val_iZone], iMesh);
145  }
146  }
147 }

167 void CDiscAdjHeatIteration::RegisterInput(CSolver***** solver, CGeometry**** geometry, CConfig** config,
168  unsigned short iZone, unsigned short iInst, RECORDING kind_recording) {
169  auto solvers0 = solver[iZone][iInst][MESH_0];
170  auto geometry0 = geometry[iZone][iInst][MESH_0];
171 
172  if (kind_recording == RECORDING::SOLUTION_VARIABLES || kind_recording == RECORDING::SOLUTION_AND_MESH) {
173  /*--- Register flow and turbulent variables as input ---*/
174 
175  solvers0[ADJHEAT_SOL]->RegisterSolution(geometry0, config[iZone]);
176 
177  solvers0[ADJHEAT_SOL]->RegisterVariables(geometry0, config[iZone]);
178  }
179  else if (kind_recording == RECORDING::MESH_COORDS) {
180  /*--- Register node coordinates as input ---*/
181 
182  geometry0->RegisterCoordinates();
183  }
184  else if (kind_recording == RECORDING::MESH_DEFORM) {
185  /*--- Register the variables of the mesh deformation ---*/
186  /*--- Undeformed mesh coordinates ---*/
187  solvers0[ADJMESH_SOL]->RegisterSolution(geometry0, config[iZone]);
188 
189  /*--- Boundary displacements ---*/
190  solvers0[ADJMESH_SOL]->RegisterVariables(geometry0, config[iZone]);
191  }
192 }

194 void CDiscAdjHeatIteration::SetDependencies(CSolver***** solver, CGeometry**** geometry, CNumerics****** numerics,
195  CConfig** config, unsigned short iZone, unsigned short iInst,
196  RECORDING kind_recording) {
197 
198  auto solvers0 = solver[iZone][iInst][MESH_0];
199  auto geometries = geometry[iZone][iInst];
200 
201  if ((kind_recording == RECORDING::MESH_COORDS) ||
202  (kind_recording == RECORDING::CLEAR_INDICES) ||
203  (kind_recording == RECORDING::SOLUTION_AND_MESH)) {
204  /*--- Update geometry to get the influence on other geometry variables (normals, volume etc) ---*/
205 
206  CGeometry::UpdateGeometry(geometries, config[iZone]);
207 
208  CGeometry::ComputeWallDistance(config, geometry);
209  }
210 
211  solvers0[HEAT_SOL]->Set_Heatflux_Areas(geometries[MESH_0], config[iZone]);
212 
213  solvers0[HEAT_SOL]->Preprocessing(geometries[MESH_0], solvers0, config[iZone], MESH_0, NO_RK_ITER, RUNTIME_HEAT_SYS, true);
214  solvers0[HEAT_SOL]->Postprocessing(geometries[MESH_0], solvers0, config[iZone], MESH_0);
215 
216  solvers0[HEAT_SOL]->InitiateComms(geometries[MESH_0], config[iZone], MPI_QUANTITIES::SOLUTION);
217  solvers0[HEAT_SOL]->CompleteComms(geometries[MESH_0], config[iZone], MPI_QUANTITIES::SOLUTION);
218 }
