49 /*!
50  * \brief Aligned memory allocation compatible across platforms.
51  * \param[in] alignment, in bytes, of the memory being allocated.
52  * \param[in] size, also in bytes.
53  * \tparam ZeroInit, initialize memory to 0.
54  * \return Pointer to memory, always use su2::aligned_free to deallocate.
55  */
56 template <class T, bool ZeroInit = false>
57 inline T* aligned_alloc(size_t alignment, size_t size) noexcept {
58  assert(is_power_of_two(alignment));
59 
60  if (alignment < alignof(void*)) alignment = alignof(void*);
61 
62  size = round_up(alignment, size);
63 
64  void* ptr = nullptr;
65 
66 #if defined(__APPLE__)
67  if (::posix_memalign(&ptr, alignment, size) != 0) {
68  ptr = nullptr;
69  }
70 #elif defined(_WIN32)
71  ptr = _aligned_malloc(size, alignment);
72 #else
73  ptr = ::aligned_alloc(alignment, size);
74 #endif
75  if (ZeroInit) memset(ptr, 0, size);
76  return static_cast<T*>(ptr);
77 }
