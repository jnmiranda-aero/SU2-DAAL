31/*!
32 * \brief A wrapper funtion that calls specialized implementations depending
33 * on "LimiterKind". Those implementations are generated by instantiating
34 * versions of "computeLimiters_impl" with appropriate specializations
35 * of "CLimiterDetails". See corresponding hpp files for further details.
36 * \ingroup FvmAlgos
37 */
38template<class FieldType, class GradientType>
39void computeLimiters(LIMITER LimiterKind,
40 CSolver* solver,
41 MPI_QUANTITIES kindMpiComm,
42 PERIODIC_QUANTITIES kindPeriodicComm1,
43 PERIODIC_QUANTITIES kindPeriodicComm2,
44 CGeometry& geometry,
45 const CConfig& config,
46 size_t varBegin,
47 size_t varEnd,
48 const FieldType& field,
49 const GradientType& gradient,
50 FieldType& fieldMin,
51 FieldType& fieldMax,
52 FieldType& limiter)
53{
54 if (geometry.GetnDim() != 2 && geometry.GetnDim() != 3)
55 SU2_MPI::Error("Too many dimensions to compute limiters.", CURRENT_FUNCTION);
56
57#define INSTANTIATE(KIND)\
58if (geometry.GetnDim() == 2) {\
59 computeLimiters_impl<2,KIND>(solver, kindMpiComm, kindPeriodicComm1, kindPeriodicComm2, geometry,\
60 config, varBegin, varEnd, field, gradient, fieldMin, fieldMax, limiter);\
61} else {\
62 computeLimiters_impl<3,KIND>(solver, kindMpiComm, kindPeriodicComm1, kindPeriodicComm2, geometry,\
63 config, varBegin, varEnd, field, gradient, fieldMin, fieldMax, limiter);\
64}
65 switch (LimiterKind) {
66 case LIMITER::NONE:
67 {
68 SU2_OMP_FOR_STAT(512)
69 for(size_t iPoint = 0; iPoint < geometry.GetnPoint(); ++iPoint)
70 for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
71 limiter(iPoint, iVar) = 1.0;
72 END_SU2_OMP_FOR
73 break;
74 }
75 case LIMITER::BARTH_JESPERSEN:
76 {
77 INSTANTIATE(LIMITER::BARTH_JESPERSEN);
78 break;
79 }
80 case LIMITER::VENKATAKRISHNAN:
81 {
82 INSTANTIATE(LIMITER::VENKATAKRISHNAN);
83 break;
84 }
85 case LIMITER::NISHIKAWA_R3:
86 {
87 INSTANTIATE(LIMITER::NISHIKAWA_R3);
88 break;
89 }
90 case LIMITER::NISHIKAWA_R4:
91 {
92 INSTANTIATE(LIMITER::NISHIKAWA_R4);
93 break;
94 }
95 case LIMITER::NISHIKAWA_R5:
96 {
97 INSTANTIATE(LIMITER::NISHIKAWA_R5);
98 break;
99 }
100 case LIMITER::VENKATAKRISHNAN_WANG:
101 {
102 INSTANTIATE(LIMITER::VENKATAKRISHNAN_WANG);
103 break;
104 }
105 case LIMITER::WALL_DISTANCE:
106 {
107 INSTANTIATE(LIMITER::WALL_DISTANCE);
108 break;
109 }
110 case LIMITER::SHARP_EDGES:
111 {
112 INSTANTIATE(LIMITER::SHARP_EDGES);
113 break;
114 }
115 default:
116 {
117 SU2_MPI::Error("Unknown limiter type.", CURRENT_FUNCTION);
118 break;
119 }
120 }
121#undef INSTANTIATE
122}
