53 {
54  if (geometry.GetnDim() != 2 && geometry.GetnDim() != 3)
55  SU2_MPI::Error("Too many dimensions to compute limiters.", CURRENT_FUNCTION);
56 
57 #define INSTANTIATE(KIND)\
58 if (geometry.GetnDim() == 2) {\
59  computeLimiters_impl<2,KIND>(solver, kindMpiComm, kindPeriodicComm1, kindPeriodicComm2, geometry,\
60  config, varBegin, varEnd, field, gradient, fieldMin, fieldMax, limiter);\
61 } else {\
62  computeLimiters_impl<3,KIND>(solver, kindMpiComm, kindPeriodicComm1, kindPeriodicComm2, geometry,\
63  config, varBegin, varEnd, field, gradient, fieldMin, fieldMax, limiter);\
64 }
65  switch (LimiterKind) {
66  case LIMITER::NONE:
67  {
68  SU2_OMP_FOR_STAT(512)
69  for(size_t iPoint = 0; iPoint < geometry.GetnPoint(); ++iPoint)
70  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
71  limiter(iPoint, iVar) = 1.0;
72  END_SU2_OMP_FOR
73  break;
74  }
75  case LIMITER::BARTH_JESPERSEN:
76  {
77  INSTANTIATE(LIMITER::BARTH_JESPERSEN);

79  }

89  }

93  break;
94  }
95  case LIMITER::NISHIKAWA_R5:
96  {
97  INSTANTIATE(LIMITER::NISHIKAWA_R5);
98  break;
99  }
100  case LIMITER::VENKATAKRISHNAN_WANG:
101  {
