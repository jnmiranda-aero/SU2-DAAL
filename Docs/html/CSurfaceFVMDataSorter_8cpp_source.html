70 /*--- Prepare to check and communicate the nodes that each proc has
71 locally from the surface connectivity. ---*/
72
73 int *nElem_Send = new int[size+1](); nElem_Send[0] = 0;
74 int *nElem_Recv = new int[size+1](); nElem_Recv[0] = 0;

78 nElem_Send[ii] = 0;
79 nElem_Recv[ii] = 0;

82 nElem_Send[size] = 0; nElem_Recv[size] = 0;

84 /*--- Loop through our local line elements and check where each
85 of the grid nodes resides based on global index. ---*/
86
87 for (int ii = 0; ii < (int)nElemLine; ii++) {
88 for ( int jj = 0; jj < N_POINTS_LINE; jj++ ) {
89
90 /*--- Get global index. Note the -1 since it was 1-based for viz. ---*/

92 iNode = ii*N_POINTS_LINE+jj;

100 number of elements that must be sent to a particular proc. ---*/

103 nElem_Flag[iProcessor] = iNode;
104 nElem_Send[iProcessor+1]++;

107 }

110 /*--- Reset out flags and then loop through our local triangle surface
111 elements performing the same check for where each grid node resides. ---*/
112
113 for (int ii=0; ii < size; ii++) nElem_Flag[ii]= -1;
114
115 for (int ii = 0; ii < (int)nElemTria; ii++) {
116 for ( int jj = 0; jj < N_POINTS_TRIANGLE; jj++ ) {
117
118 /*--- Get global index. Note the -1 since it was 1-based for viz. ---*/
119
120 iNode = ii*N_POINTS_TRIANGLE + jj;

124

138 /*--- Reset out flags and then loop through our local quad surface
139 elements performing the same check for where each grid node resides. ---*/
140
141 for (int ii=0; ii < size; ii++) nElem_Flag[ii]= -1;
142
143 for (int ii = 0; ii < (int)nElemQuad; ii++) {
144 for ( int jj = 0; jj < N_POINTS_QUADRILATERAL; jj++ ) {
145
146 /*--- Get global index. Note the -1 since it was 1-based for viz. ---*/

149 Global_Index = Conn_Quad_Par[iNode]-1;

152
153 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
154
155 /*--- If we have not visited this element yet, increment our
156 number of elements that must be sent to a particular proc. ---*/

159 nElem_Flag[iProcessor] = iNode;
160 nElem_Send[iProcessor+1]++;

166 /*--- Communicate the number of nodes to be sent/recv'd amongst
167 all processors. After this communication, each proc knows how
168 many nodes it will receive from each other processor. ---*/
169
170 SU2_MPI::Alltoall(&(nElem_Send[1]), 1, MPI_INT,
171 &(nElem_Recv[1]), 1, MPI_INT, SU2_MPI::GetComm());

173 /*--- Prepare to send. First check how many
174 messages we will be sending and receiving. Here we also put
175 the counters into cumulative storage format to make the
176 communications simpler. ---*/
177
178 int nSends = 0, nRecvs = 0;
179 for (int ii=0; ii < size; ii++) nElem_Flag[ii] = -1;
180
181 for (int ii = 0; ii < size; ii++) {
182 if ((ii != rank) && (nElem_Send[ii+1] > 0)) nSends++;
183 if ((ii != rank) && (nElem_Recv[ii+1] > 0)) nRecvs++;
184
185 nElem_Send[ii+1] += nElem_Send[ii];
186 nElem_Recv[ii+1] += nElem_Recv[ii];
187 }

189 /*--- Allocate arrays for sending the global ID. ---*/
190
191 auto *idSend = new unsigned long[nElem_Send[size]]();
192
193 /*--- Create an index variable to keep track of our index
194 positions as we load up the send buffer. ---*/
195
196 auto *idIndex = new unsigned long[size]();
197 for (int ii=0; ii < size; ii++) idIndex[ii] = nElem_Send[ii];
198
199 /*--- Now loop back through the local connectivities for the surface
200 elements and load up the global IDs for sending to their home proc. ---*/
201
202 for (int ii = 0; ii < (int)nElemLine; ii++) {
203 for ( int jj = 0; jj < N_POINTS_LINE; jj++ ) {
204
205 /*--- Get global index. Note the -1 since it was 1-based for viz. ---*/
206
207 iNode = ii*N_POINTS_LINE+jj;
208 Global_Index = Conn_Line_Par[iNode]-1;
209
210 /*--- Search for the processor that owns this point ---*/
211
212 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
213
214 /*--- Load global ID into the buffer for sending ---*/
215
216 if (nElem_Flag[iProcessor] != iNode) {
217
218 nElem_Flag[iProcessor] = iNode;
219 unsigned long nn = idIndex[iProcessor];
220
221 /*--- Load the connectivity values. ---*/
222
223 idSend[nn] = Global_Index; nn++;
224
225 /*--- Increment the index by the message length ---*/
226
227 idIndex[iProcessor]++;
228
229 }
230
231 }
232 }
233
234 for (int ii=0; ii < size; ii++) nElem_Flag[ii]= -1;
235
236 for (int ii = 0; ii < (int)nElemTria; ii++) {
237 for ( int jj = 0; jj < N_POINTS_TRIANGLE; jj++ ) {
238
239 /*--- Get global index. Note the -1 since it was 1-based for viz. ---*/
240
241 iNode = ii*N_POINTS_TRIANGLE + jj;
242 Global_Index = Conn_Tria_Par[iNode]-1;
243
244 /*--- Search for the processor that owns this point ---*/
245
246 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
247
248 /*--- Load global ID into the buffer for sending ---*/
249
250 if (nElem_Flag[iProcessor] != iNode) {
251
252 nElem_Flag[iProcessor] = iNode;
253 unsigned long nn = idIndex[iProcessor];
254
255 /*--- Load the connectivity values. ---*/
256
257 idSend[nn] = Global_Index; nn++;
258
259 /*--- Increment the index by the message length ---*/
260
261 idIndex[iProcessor]++;
262
263 }
264
265 }
266 }
267
268 for (int ii=0; ii < size; ii++) nElem_Flag[ii]= -1;
269
270 for (int ii = 0; ii < (int)nElemQuad; ii++) {
271 for ( int jj = 0; jj < N_POINTS_QUADRILATERAL; jj++ ) {
272
273 /*--- Get global index. Note the -1 since it was 1-based for viz. ---*/
274
275 iNode = ii*N_POINTS_QUADRILATERAL+jj;
276 Global_Index = Conn_Quad_Par[iNode]-1;
277
278 /*--- Search for the processor that owns this point ---*/
279
280 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
281
282 /*--- Load global ID into the buffer for sending ---*/
283
284 if (nElem_Flag[iProcessor] != iNode) {
285
286 nElem_Flag[iProcessor] = iNode;
287 unsigned long nn = idIndex[iProcessor];

289 /*--- Load the connectivity values. ---*/

293 /*--- Increment the index by the message length ---*/

295 idIndex[iProcessor]++;

297 }
298
299 }
300 }

302 /*--- Allocate the memory that we need for receiving the global IDs
303 values and then cue up the non-blocking receives. Note that
304 we do not include our own rank in the communications. We will
305 directly copy our own data later. ---*/
306
307 auto *idRecv = new unsigned long[nElem_Recv[size]]();
308
309#ifdef HAVE_MPI
310 /*--- We need double the number of messages to send both the conn.
311 and the flags for the halo cells. ---*/
312
313 send_req = new SU2_MPI::Request[nSends];
314 recv_req = new SU2_MPI::Request[nRecvs];
315
316 /*--- Launch the non-blocking recv's for the global IDs. ---*/
317
318 unsigned long iMessage = 0;
319 for (int ii=0; ii<size; ii++) {
320 if ((ii != rank) && (nElem_Recv[ii+1] > nElem_Recv[ii])) {
321 int ll = nElem_Recv[ii];
322 int kk = nElem_Recv[ii+1] - nElem_Recv[ii];
323 int count = kk;
324 int source = ii;
325 int tag = ii + 1;
326 SU2_MPI::Irecv(&(idRecv[ll]), count, MPI_UNSIGNED_LONG, source, tag,
327 SU2_MPI::GetComm(), &(recv_req[iMessage]));
328 iMessage++;
329 }
330 }
331
332 /*--- Launch the non-blocking sends of the global IDs. ---*/
333
334 iMessage = 0;
335 for (int ii=0; ii<size; ii++) {
336 if ((ii != rank) && (nElem_Send[ii+1] > nElem_Send[ii])) {
337 int ll = nElem_Send[ii];
338 int kk = nElem_Send[ii+1] - nElem_Send[ii];
339 int count = kk;
340 int dest = ii;
341 int tag = rank + 1;
342 SU2_MPI::Isend(&(idSend[ll]), count, MPI_UNSIGNED_LONG, dest, tag,
343 SU2_MPI::GetComm(), &(send_req[iMessage]));
344 iMessage++;
345 }
346 }
347#endif
348
349 /*--- Copy my own rank's data into the recv buffer directly. ---*/
350
351 int mm = nElem_Recv[rank];
352 int ll = nElem_Send[rank];
353 int kk = nElem_Send[rank+1];
354
355 for (int nn=ll; nn<kk; nn++, mm++) idRecv[mm] = idSend[nn];
356
357 /*--- Wait for the non-blocking sends and recvs to complete. ---*/
358
359#ifdef HAVE_MPI
360 int number = nSends;
361 for (int ii = 0; ii < number; ii++)
362 SU2_MPI::Waitany(number, send_req, &ind, &status);
363
364 number = nRecvs;
365 for (int ii = 0; ii < number; ii++)
366 SU2_MPI::Waitany(number, recv_req, &ind, &status);
367
368 delete [] send_req;
369 delete [] recv_req;
370#endif

372 /*--------------------------------------------------------------------------*/
373 /*--- Step 2: Each proc now knows which is its local grid nodes from ---*/
374 /*--- the entire volume solution are part of the surface. We ---*/
375 /*--- now apply a mask to extract just those points on the ---*/
376 /*--- surface. We also need to perform a renumbering so that ---*/
377 /*--- the surface data (nodes and connectivity) have their ---*/
378 /*--- own global numbering. This is important for writing ---*/
379 /*--- output files in a later routine. ---*/
380 /*--------------------------------------------------------------------------*/
381
382 /*--- Create a local data structure that acts as a mask to extract the
383 set of points within the local set that are on the surface. ---*/
384
385 int *surfPoint = new int[volumeSorter->GetnPoints()];
386 for (iPoint = 0; iPoint < volumeSorter->GetnPoints(); iPoint++) surfPoint[iPoint] = -1;
387
388 for (int ii = 0; ii < nElem_Recv[size]; ii++) {
389 surfPoint[(int)idRecv[ii] - volumeSorter->GetNodeBegin(rank)] = (int)idRecv[ii];
390 }
391
392 /*--- First, add up the number of surface points I have on my rank. ---*/
393
394 nPoints = 0;
395 Renumber2Global.clear();
396
397 for (iPoint = 0; iPoint < volumeSorter->GetnPoints(); iPoint++) {
398 if (surfPoint[iPoint] != -1) {
399
400 /*--- Save the global index values for CSV output. ---*/
401
402 Renumber2Global[nPoints] = surfPoint[iPoint];
403
404 /*--- Increment total number of surface points found locally. ---*/
405
406 nPoints++;
407
408 }
409 }
410
411 /*--- Communicate this number of local surface points to all other
412 processors so that it can be used to create offsets for the new
413 global numbering for the surface points. ---*/
414
415 nPoint_Send[0] = 0;
416 nPoint_Recv[0] = 0;
417
418 for (int ii=1; ii < size+1; ii++) nPoint_Send[ii]= (int)nPoints;
419
420 SU2_MPI::Alltoall(&(nPoint_Send[1]), 1, MPI_INT,
421 &(nPoint_Recv[1]), 1, MPI_INT, SU2_MPI::GetComm());
422
423 /*--- Go to cumulative storage format to compute the offsets. ---*/
424
425 for (int ii = 0; ii < size; ii++) {
426 nPoint_Send[ii+1] += nPoint_Send[ii];
427 nPoint_Recv[ii+1] += nPoint_Recv[ii];
428 }
429
430 /*--- Now that we know the number of local surface points that we have,
431 we can allocate the new data structure to hold these points alone. Here,
432 we also copy the data for those points from our volume data structure. ---*/
433
434 delete [] dataBuffer;
435 dataBuffer = new passivedouble[nPoints*VARS_PER_POINT];
436
437 for (int jj = 0; jj < VARS_PER_POINT; jj++) {
438 count = 0;
439 for (int ii = 0; ii < (int)volumeSorter->GetnPoints(); ii++) {
440 if (surfPoint[ii] !=-1) {
441 dataBuffer[count*VARS_PER_POINT + jj] = volumeSorter->GetData(jj,ii);
442 count++;
443 }
444 }
445 }
446 /*--- Reduce the total number of surf points we have. This will be
447 needed for writing the surface solution files later. ---*/
448
449 SU2_MPI::Allreduce(&nPoints, &nPointsGlobal, 1,
450 MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
451
452 /*--- Now that we know every proc's global offset for the number of
453 surface points, we can create the new global numbering. Here, we
454 create a new mapping using two arrays, which will need to be
455 communicated. We use our mask again here. ---*/
456
457 auto *globalP = new unsigned long[nPoints]();
458 auto *renumbP = new unsigned long[nPoints]();
459
460 count = 0;
461 for (iPoint = 0; iPoint < volumeSorter->GetnPoints(); iPoint++) {
462 if (surfPoint[iPoint] != -1) {
463 globalP[count] = surfPoint[iPoint];
464 renumbP[count] = count + nPoint_Recv[rank];
465 count++;
466 }
467 }
468
469 /*--------------------------------------------------------------------------*/
470 /*--- Step 3: Communicate the arrays with the new global surface point ---*/
471 /*--- numbering to the procs that hold the connectivity for ---*/
472 /*--- each element. This will be done in two phases. First, ---*/
473 /*--- we send the arrays around to the other procs based on ---*/
474 /*--- the linear partitioning for the elems. This gets us ---*/
475 /*--- most of the way there, however, due to the type of ---*/
476 /*--- linear partitioning for the elements, there may exist ---*/
477 /*--- elements that have nodes outside of the linear part. ---*/
478 /*--- bounds. This is because the elems are distributed based ---*/
479 /*--- on the node with the smallest global ID. ---*/
480 /*--------------------------------------------------------------------------*/
481
482 /*--- Reset our flags and counters ---*/
483
484 for (int ii=0; ii < size; ii++) {
485 nElem_Send[ii] = 0;
486 nElem_Recv[ii] = 0;
487 nElem_Flag[ii]= -1;
488 }
489 nElem_Send[size] = 0; nElem_Recv[size] = 0;

491 /*--- Loop through my local surface nodes, find which proc the global
492 value lives on, then communicate the global ID and remumbered value. ---*/
493
494 for (int ii = 0; ii < (int)nPoints; ii++) {
495
496 Global_Index = globalP[ii];
497
498 /*--- Search for the processor that owns this point ---*/
499

501

503 number of elements that must be sent to a particular proc. ---*/

678#ifdef HAVE_MPI
679 number = 2*nSends;
680 for (int ii = 0; ii < number; ii++)
681 SU2_MPI::Waitany(number, send_req, &ind, &status);
682
683 number = 2*nRecvs;
684 for (int ii = 0; ii < number; ii++)
685 SU2_MPI::Waitany(number, recv_req, &ind, &status);
686
687 delete [] send_req;
688 delete [] recv_req;
689#endif
690
691 /*-- Now update my local connectivitiy for the surface with the new
692 numbering. Create a new mapping for global -> renumber for nodes. Note
693 the adding of 1 back in here for the eventual viz. purposes. ---*/
694
695 map<unsigned long,unsigned long> Global2Renumber;
696 for (int ii = 0; ii < nElem_Recv[size]; ii++) {
697 Global2Renumber[globalRecv[ii]] = renumbRecv[ii] + 1;
698 }
699
700
701 /*--- The final step is one last pass over all elements to check
702 for points outside of the linear partitions of the elements. Again,
703 note that elems were distributed based on their smallest global ID,
704 so some nodes of the elem may have global IDs lying outside of the
705 linear partitioning. We need to recover the mapping for these
706 outliers. We loop over all local surface elements to find these. ---*/
707
708 vector<unsigned long>::iterator it;
709 vector<unsigned long> outliers;
710
711 for (int ii = 0; ii < (int)nElemLine; ii++) {
712 for ( int jj = 0; jj < N_POINTS_LINE; jj++ ) {
713
714 iNode = ii*N_POINTS_LINE+jj;
715 Global_Index = Conn_Line_Par[iNode]-1;
716
717 /*--- Search for the processor that owns this point ---*/
718
719 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
720
721 /*--- Store the global ID if it is outside our own linear partition. ---*/
722
723 if ((iProcessor != (unsigned long)rank)) {
724 outliers.push_back(Global_Index);
725 }
726
727 }
728 }
729
730 for (int ii=0; ii < size; ii++) nElem_Flag[ii]= -1;
731
732 for (int ii = 0; ii < (int)nElemTria; ii++) {
733 for ( int jj = 0; jj < N_POINTS_TRIANGLE; jj++ ) {
734
735 iNode = ii*N_POINTS_TRIANGLE + jj;
736 Global_Index = Conn_Tria_Par[iNode]-1;
737
738 /*--- Search for the processor that owns this point ---*/
739
740 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
741
742 /*--- Store the global ID if it is outside our own linear partition. ---*/
743
744 if ((iProcessor != (unsigned long)rank)) {
745 outliers.push_back(Global_Index);
746 }
747
748 }
749 }
750
751 for (int ii=0; ii < size; ii++) nElem_Flag[ii]= -1;
752
753 for (int ii = 0; ii < (int)nElemQuad; ii++) {
754 for ( int jj = 0; jj < N_POINTS_QUADRILATERAL; jj++ ) {
755
756 iNode = ii*N_POINTS_QUADRILATERAL+jj;
757 Global_Index = Conn_Quad_Par[iNode]-1;
758
759 /*--- Search for the processor that owns this point ---*/
760
761 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
762
763 /*--- Store the global ID if it is outside our own linear partition. ---*/
764
765 if ((iProcessor != (unsigned long)rank)) {
766 outliers.push_back(Global_Index);
767 }
768
769 }
770 }
771
772 /*--- Create a unique list of global IDs that fall outside of our procs
773 linear partition. ---*/
774
775 sort(outliers.begin(), outliers.end());
776 it = unique(outliers.begin(), outliers.end());
777 outliers.resize(it - outliers.begin());
778
779 /*--- Now loop over the outliers and communicate to those procs that
780 hold the new numbering for our outlier points. We need to ask for the
781 new numbering from these procs. ---*/
782
783 for (int ii=0; ii < size; ii++) {
784 nElem_Send[ii] = 0;
785 nElem_Recv[ii] = 0;
786 nElem_Flag[ii]= -1;
787 }
788 nElem_Send[size] = 0; nElem_Recv[size] = 0;
789
790 for (int ii = 0; ii < (int)outliers.size(); ii++) {
791
792 Global_Index = outliers[ii];
793
794 /*--- Search for the processor that owns this point ---*/
795
796 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
797
798 /*--- If we have not visited this element yet, increment our
799 number of elements that must be sent to a particular proc. ---*/
800
801 if ((nElem_Flag[iProcessor] != ii)) {
802 nElem_Flag[iProcessor] = ii;
803 nElem_Send[iProcessor+1]++;
804 }
805
806 }
807
808 /*--- Communicate the number of cells to be sent/recv'd amongst
809 all processors. After this communication, each proc knows how
810 many cells it will receive from each other processor. ---*/
811
812 SU2_MPI::Alltoall(&(nElem_Send[1]), 1, MPI_INT,
813 &(nElem_Recv[1]), 1, MPI_INT, SU2_MPI::GetComm());
814
815 /*--- Prepare to send connectivities. First check how many
816 messages we will be sending and receiving. Here we also put
817 the counters into cumulative storage format to make the
818 communications simpler. ---*/
819
820 nSends = 0; nRecvs = 0;
821 for (int ii=0; ii < size; ii++) nElem_Flag[ii] = -1;
822
823 for (int ii = 0; ii < size; ii++) {
824 if ((ii != rank) && (nElem_Send[ii+1] > 0)) nSends++;
825 if ((ii != rank) && (nElem_Recv[ii+1] > 0)) nRecvs++;
826
827 nElem_Send[ii+1] += nElem_Send[ii];
828 nElem_Recv[ii+1] += nElem_Recv[ii];
829 }
830
831 delete [] idSend;
832 idSend = new unsigned long[nElem_Send[size]];
833 for (int ii = 0; ii < nElem_Send[size]; ii++)
834 idSend[ii] = 0;
835
836 /*--- Reset our index variable for reuse. ---*/
837
838 for (int ii=0; ii < size; ii++) idIndex[ii] = nElem_Send[ii];
839
840 /*--- Loop over the outliers again and load up the global IDs. ---*/
841
842 for (int ii = 0; ii < (int)outliers.size(); ii++) {
843
844 Global_Index = outliers[ii];
845
846 /*--- Search for the processor that owns this point ---*/
847
848 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
849
850 /*--- If we have not visited this element yet, increment our
851 number of elements that must be sent to a particular proc. ---*/
852
853 if ((nElem_Flag[iProcessor] != ii)) {
854
855 nElem_Flag[iProcessor] = ii;
856 unsigned long nn = idIndex[iProcessor];
857
858 /*--- Load the global ID values. ---*/
859
860 idSend[nn] = Global_Index; nn++;
861
862 /*--- Increment the index by the message length ---*/
863
864 idIndex[iProcessor]++;
865
866 }
867 }
868
869 /*--- Allocate the memory that we need for receiving the
870 values and then cue up the non-blocking receives. Note that
871 we do not include our own rank in the communications. We will
872 directly copy our own data later. ---*/
873
874 delete [] idRecv;
875 idRecv = new unsigned long[nElem_Recv[size]]();
876
877#ifdef HAVE_MPI
878 /*--- We need double the number of messages to send both the conn.
879 and the flags for the halo cells. ---*/
880
881 send_req = new SU2_MPI::Request[nSends];
882 recv_req = new SU2_MPI::Request[nRecvs];
883
884 /*--- Launch the non-blocking recv's for the connectivity. ---*/
885
886 iMessage = 0;
887 for (int ii=0; ii<size; ii++) {
888 if ((ii != rank) && (nElem_Recv[ii+1] > nElem_Recv[ii])) {
889 int ll = nElem_Recv[ii];
890 int kk = nElem_Recv[ii+1] - nElem_Recv[ii];
891 int count = kk;
892 int source = ii;
893 int tag = ii + 1;
894 SU2_MPI::Irecv(&(idRecv[ll]), count, MPI_UNSIGNED_LONG, source, tag,
895 SU2_MPI::GetComm(), &(recv_req[iMessage]));
896 iMessage++;
897 }
898 }
899
900 /*--- Launch the non-blocking sends of the connectivity. ---*/
901
902 iMessage = 0;
903 for (int ii=0; ii<size; ii++) {
904 if ((ii != rank) && (nElem_Send[ii+1] > nElem_Send[ii])) {
905 int ll = nElem_Send[ii];
906 int kk = nElem_Send[ii+1] - nElem_Send[ii];
907 int count = kk;
908 int dest = ii;
909 int tag = rank + 1;
910 SU2_MPI::Isend(&(idSend[ll]), count, MPI_UNSIGNED_LONG, dest, tag,
911 SU2_MPI::GetComm(), &(send_req[iMessage]));
912 iMessage++;
913 }
914 }
915#endif
916
917 /*--- Copy my own rank's data into the recv buffer directly. ---*/
918
919 mm = nElem_Recv[rank];
920 ll = nElem_Send[rank];
921 kk = nElem_Send[rank+1];
922
923 for (int nn=ll; nn<kk; nn++, mm++) idRecv[mm] = idSend[nn];
924
925 /*--- Wait for the non-blocking sends and recvs to complete. ---*/
926
927#ifdef HAVE_MPI
928 number = nSends;
929 for (int ii = 0; ii < number; ii++)
930 SU2_MPI::Waitany(number, send_req, &ind, &status);
931
932 number = nRecvs;
933 for (int ii = 0; ii < number; ii++)
934 SU2_MPI::Waitany(number, recv_req, &ind, &status);
935
936 delete [] send_req;
937 delete [] recv_req;
938#endif
939
940 /*--- The procs holding the outlier grid nodes now have the global IDs
941 that they need to have their renumbering shared. ---*/
942
943 for (int ii = 0; ii < nElem_Recv[size]; ii++) {
944 for (iPoint = 0; iPoint < nPoints; iPoint++) {
945 if (idRecv[ii] == globalP[iPoint]) {
946 idRecv[ii] = renumbP[iPoint];
947 }
948 }
949 }
950
951 /*--- Now simply reverse the last communication to give the renumbered IDs
952 back to the owner of the outlier points. Note everything is flipped. ---*/
953
954#ifdef HAVE_MPI
955 /*--- We need double the number of messages to send both the conn.
956 and the flags for the halo cells. ---*/
957
958 send_req = new SU2_MPI::Request[nRecvs];
959 recv_req = new SU2_MPI::Request[nSends];
960
961 /*--- Launch the non-blocking sends of the connectivity. ---*/
962
963 iMessage = 0;
964 for (int ii=0; ii<size; ii++) {
965 if ((ii != rank) && (nElem_Send[ii+1] > nElem_Send[ii])) {
966 int ll = nElem_Send[ii];
967 int kk = nElem_Send[ii+1] - nElem_Send[ii];
968 int count = kk;
969 int dest = ii;
970 int tag = ii + 1;
971 SU2_MPI::Irecv(&(idSend[ll]), count, MPI_UNSIGNED_LONG, dest, tag,
972 SU2_MPI::GetComm(), &(recv_req[iMessage]));
973 iMessage++;
974 }
975 }
976
977 /*--- Launch the non-blocking recv's for the connectivity. ---*/
978
979 iMessage = 0;
980 for (int ii=0; ii<size; ii++) {
981 if ((ii != rank) && (nElem_Recv[ii+1] > nElem_Recv[ii])) {
982 int ll = nElem_Recv[ii];
983 int kk = nElem_Recv[ii+1] - nElem_Recv[ii];
984 int count = kk;
985 int source = ii;
986 int tag = rank + 1;
987 SU2_MPI::Isend(&(idRecv[ll]), count, MPI_UNSIGNED_LONG, source, tag,
988 SU2_MPI::GetComm(), &(send_req[iMessage]));
989 iMessage++;
990 }
991 }
992#endif
993
994 /*--- Copy my own rank's data into the recv buffer directly. ---*/
995
996 mm = nElem_Send[rank];
997 ll = nElem_Recv[rank];
998 kk = nElem_Recv[rank+1];
999
1000 for (int nn=ll; nn<kk; nn++, mm++) idSend[mm] = idRecv[nn];
1001
1002 /*--- Wait for the non-blocking sends and recvs to complete. ---*/
1003
1004#ifdef HAVE_MPI
1005 number = nRecvs;
1006 for (int ii = 0; ii < number; ii++)
1007 SU2_MPI::Waitany(number, send_req, &ind, &status);
1008
1009 number = nSends;
1010 for (int ii = 0; ii < number; ii++)
1011 SU2_MPI::Waitany(number, recv_req, &ind, &status);
1012
1013 delete [] send_req;
1014 delete [] recv_req;
1015#endif
1016
1017 /*--- Add the renumbering for the outliers to the map from before carrying
1018 the global -> renumber transformation. Note that by construction,
1019 nElem_Send[ii] == outliers.size(). We also add in the 1 for viz. here. ---*/
1020
1021 for (int ii = 0; ii < nElem_Send[size]; ii++) {
1022 Global2Renumber[outliers[ii]] = idSend[ii] + 1;
1023 }
1024
1025 /*--- We can now overwrite the local connectivity for our surface elems
1026 using our completed map with the new global renumbering. Whew!! Note
1027 the -1 when accessing the conn from the map. ---*/
1028
1029 for (iElem = 0; iElem < nElemLine; iElem++) {
1030 iNode = (int)iElem*N_POINTS_LINE;
1031 Conn_Line_Par[iNode+0] = (int)Global2Renumber[Conn_Line_Par[iNode+0]-1];
1032 Conn_Line_Par[iNode+1] = (int)Global2Renumber[Conn_Line_Par[iNode+1]-1];
1033 }
1034
1035 for (iElem = 0; iElem < nElemTria; iElem++) {
1036 iNode = (int)iElem*N_POINTS_TRIANGLE;
1037 Conn_Tria_Par[iNode+0] = (int)Global2Renumber[Conn_Tria_Par[iNode+0]-1];
1038 Conn_Tria_Par[iNode+1] = (int)Global2Renumber[Conn_Tria_Par[iNode+1]-1];
1039 Conn_Tria_Par[iNode+2] = (int)Global2Renumber[Conn_Tria_Par[iNode+2]-1];
1040 }
1041
1042 for (iElem = 0; iElem < nElemQuad; iElem++) {
1043 iNode = (int)iElem*N_POINTS_QUADRILATERAL;
1044 Conn_Quad_Par[iNode+0] = (int)Global2Renumber[Conn_Quad_Par[iNode+0]-1];
1045 Conn_Quad_Par[iNode+1] = (int)Global2Renumber[Conn_Quad_Par[iNode+1]-1];
1046 Conn_Quad_Par[iNode+2] = (int)Global2Renumber[Conn_Quad_Par[iNode+2]-1];
1047 Conn_Quad_Par[iNode+3] = (int)Global2Renumber[Conn_Quad_Par[iNode+3]-1];
1048 }
1049
1050 /*--- Free temporary memory ---*/
1051
1052 delete [] idIndex;
1053 delete [] surfPoint;
1054 delete [] globalP;
1055 delete [] renumbP;
1056
1057 delete [] idSend;
1058 delete [] idRecv;
1059 delete [] globalSend;
1060 delete [] globalRecv;
1061 delete [] renumbSend;
1062 delete [] renumbRecv;
1063 delete [] nElem_Recv;
1064 delete [] nElem_Send;
1065 delete [] nElem_Flag;
1066 delete [] Local_Halo;
1067
1068}

1226 if ((ii != rank) && (nElem_Recv[ii+1] > 0)) nRecvs++;

1229 nElem_Recv[ii+1] += nElem_Recv[ii];

1399 SU2_MPI::GetComm(), &(send_req[iMessage+nSends]));

1401 }
1402 }
1403#endif

1406

1408 int ll = NODES_PER_ELEMENT*nElem_Send[rank];
1409 int kk = NODES_PER_ELEMENT*nElem_Send[rank+1];

1412

1414 ll = nElem_Send[rank];
1415 kk = nElem_Send[rank+1];
1416
1417 for (int nn=ll; nn<kk; nn++, mm++) haloRecv[mm] = haloSend[nn];

1420

1423 for (int ii = 0; ii < number; ii++)

1425
1426 number = 2*nRecvs;
1427 for (int ii = 0; ii < number; ii++)
1428 SU2_MPI::Waitany(number, recv_req, &ind, &status);
1429
1430 delete [] send_req;
1431 delete [] recv_req;
1432#endif
1433

1435 structure before post-processing below. Note that we add 1 here
1436 to the connectivity for vizualization packages. First, allocate
1437 appropriate amount of memory for this section. ---*/
1438
1439 if (nElem_Recv[size] > 0) Conn_Elem = new int[NODES_PER_ELEMENT*nElem_Recv[size]];
1440 int count = 0; nElem_Total = 0;
1441 for (int ii = 0; ii < nElem_Recv[size]; ii++) {
1442 if (!haloRecv[ii]) {
1443 nElem_Total++;
1444 for (int jj = 0; jj < NODES_PER_ELEMENT; jj++) {
1445 Conn_Elem[count] = (int)connRecv[ii*NODES_PER_ELEMENT+jj] + 1;
1446 count++;
1447 }
1448 }
1449 }

1452
1453 /*--- Store the particular global element count in the class data,
1454 and set the class data pointer to the connectivity array. ---*/
1455
1456 switch (Elem_Type) {

1459 Conn_Line_Par = Conn_Elem;
1460 break;
1461 case TRIANGLE:

1468 break;
1469 default:
1470 SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
1471 break;
1472 }

1477 delete [] connRecv;
1478 delete [] haloSend;
1479 delete [] haloRecv;
