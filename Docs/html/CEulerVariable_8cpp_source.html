31 CEulerVariable::CEulerVariable(su2double density, const su2double *velocity, su2double energy, unsigned long npoint,
32  unsigned long ndim, unsigned long nvar, const CConfig *config)
33  : CFlowVariable(npoint, ndim, nvar, ndim + 9, ndim + 4, config),
34  indices(ndim, 0) {
35 
36  const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
37  (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
38  const bool classical_rk4 = (config->GetKind_TimeIntScheme_Flow() == CLASSICAL_RK4_EXPLICIT);
39 
40  nSecondaryVar = config->GetViscous() ? 8 : 2,
41  nSecondaryVarGrad = 2;
42 
43  /*--- Solution initialization ---*/
44 
45  su2double val_solution[5] = {su2double(1.0), velocity[0], velocity[1], energy, energy};
46  if(nDim==3) val_solution[3] = velocity[2];
47 
48  for(unsigned long iPoint = 0; iPoint < nPoint; ++iPoint)
49  for (unsigned long iVar = 0; iVar < nVar; ++iVar)
50  Solution(iPoint,iVar) = density*val_solution[iVar];
51 
52  Solution_Old = Solution;
53 
54  if (classical_rk4) Solution_New = Solution;
55 
56  /*--- Allocate and initializate solution for dual time strategy ---*/
57 
58  if (dual_time) {
59  Solution_time_n = Solution;
60  Solution_time_n1 = Solution;
61  }
62 
63  Secondary.resize(nPoint,nSecondaryVar) = su2double(0.0);
64 
65  if (config->GetAxisymmetric()){
66  nAuxVar = 3;
67  Grad_AuxVar.resize(nPoint,nAuxVar,nDim,0.0);
68  AuxVar.resize(nPoint,nAuxVar) = su2double(0.0);
69  }
70 
71  if (config->GetWind_Gust()) {
72  WindGust.resize(nPoint,nDim);
73  }
74 
75  if (config->GetVorticityConfinement()) {
76  nAuxVar = 1;
77  Grad_AuxVar.resize(nPoint, nAuxVar, nDim, 0.0);
78  AuxVar.resize(nPoint, nAuxVar) = su2double(0.0);
79  }
80 
81  if (config->GetKind_FluidModel() == ENUM_FLUIDMODEL::DATADRIVEN_FLUID){
82  DataDrivenFluid = true;
83  DatasetExtrapolation.resize(nPoint) = 0;
84  NIterNewtonsolver.resize(nPoint) = 0;
85  FluidEntropy.resize(nPoint) = su2double(0.0);
86  }
87 }

89 bool CEulerVariable::SetPrimVar(unsigned long iPoint, CFluidModel *FluidModel) {
90 
91  bool RightVol = true;
92 
93  SetVelocity(iPoint); // Computes velocity and velocity^2
94  su2double density = GetDensity(iPoint);
95  su2double staticEnergy = GetEnergy(iPoint)-0.5*Velocity2(iPoint);
96 
97  /*--- Check will be moved inside fluid model plus error description strings ---*/
98 
99  FluidModel->SetTDState_rhoe(density, staticEnergy);
100 
101  bool check_dens = SetDensity(iPoint);
102  bool check_press = SetPressure(iPoint, FluidModel->GetPressure());
103  bool check_sos = SetSoundSpeed(iPoint, FluidModel->GetSoundSpeed2());
104  bool check_temp = SetTemperature(iPoint, FluidModel->GetTemperature());
105 
106  /*--- Check that the solution has a physical meaning ---*/
107 
108  if (check_dens || check_press || check_sos || check_temp) {
109 
110  /*--- Copy the old solution ---*/
111 
112  for (unsigned long iVar = 0; iVar < nVar; iVar++)
113  Solution(iPoint, iVar) = Solution_Old(iPoint, iVar);
114 
115  /*--- Recompute the primitive variables ---*/
116 
117  SetVelocity(iPoint); // Computes velocity and velocity^2
118  su2double density = GetDensity(iPoint);
119  su2double staticEnergy = GetEnergy(iPoint)-0.5*Velocity2(iPoint);
120  /* check will be moved inside fluid model plus error description strings*/
121  FluidModel->SetTDState_rhoe(density, staticEnergy);
122 
123  SetDensity(iPoint);
124  SetPressure(iPoint, FluidModel->GetPressure());
125  SetSoundSpeed(iPoint, FluidModel->GetSoundSpeed2());
126  SetTemperature(iPoint, FluidModel->GetTemperature());
127 
128  RightVol = false;
129 
130  }
131 
132  SetEnthalpy(iPoint); // Requires pressure computation.
133 
134  /*--- Set look-up variables in case of data-driven fluid model ---*/
135  if (DataDrivenFluid) {
136  SetDataExtrapolation(iPoint, FluidModel->GetExtrapolation());
137  SetEntropy(iPoint, FluidModel->GetEntropy());
138  }
139 
140  return RightVol;
141 }
