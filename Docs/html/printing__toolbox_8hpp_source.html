40/*!
41 * \class CTablePrinter
42 * \brief Class for writing output in a table.
43 *
44 * For example the output
45 *
46 * +-------------------------------------------+
47 * | MG Level| Presmooth|PostSmooth|CorrectSmo|
48 * +-------------------------------------------+
49 * | 0| 1| 0| 0|
50 * | 1| 1| 0| 0|
51 * | 2| 1| 0| 0|
52 * | 3| 1| 0| 0|
53 * +-------------------------------------------+
54 *
55 *
56 * can be generated with the code
57 *
58 * CTablePrinter MGTable(&std::cout);
59 * MGTable.AddColumn("MG Level", 10);
60 * MGTable.AddColumn("Presmooth", 10);
61 * MGTable.AddColumn("PostSmooth", 10);
62 * MGTable.AddColumn("CorrectSmooth", 10);
63 * MGTable.PrintHeader();
64 * for (unsigned short iLevel = 0; iLevel < nMGLevels+1; iLevel++) {
65 * MGTable << iLevel << MG_PreSmooth[iLevel] << MG_PostSmooth[iLevel] << MG_CorrecSmooth[iLevel];
66 * }
67 * MGTable.PrintFooter();
68 *
69 *
70 * \author T. Albring
71 */

142 template <typename T>
143 CTablePrinter& operator<<(T input) {
144 int indent = 0;
145
146 /* --- Set the left separator --- */
147 if (j_ == 0) *out_stream_ << separator_;
148
149 /* --- Determine and set the current alignment in the stream --- */
150 if (align_ == LEFT)
151 *out_stream_ << std::left;
152 else if (align_ == RIGHT || align_ == CENTER)
153 *out_stream_ << std::right;
154
155 /*--- Print the current column value to the stream --- */
156 *out_stream_ << std::setw(column_widths_.at(j_) - indent) << std::setprecision(precision_) << input;
157
158 /*--- Reset the column counter and if it is the last column,
159 * add also a line break ---*/
160 if (j_ == GetNumColumns() - 1) {
161 *out_stream_ << std::setw(indent + 1 + (int)separator_.size()) << separator_ + "\n";
162 i_ = i_ + 1;
163 j_ = 0;
164 } else {
165 *out_stream_ << std::setw(indent + (int)inner_separator_.size()) << inner_separator_;
166 j_ = j_ + 1;
167 }
168
169 return *this;
170 }

224inline std::vector<std::string> split(const std::string& s, char delimiter) {
225 std::vector<std::string> tokens;
226 std::string token;
227 std::istringstream tokenStream(s);
228 while (std::getline(tokenStream, token, delimiter)) {
229 tokens.push_back(token);
230 }
231 return tokens;
232}

258// trim from end of string (right)
259inline std::string& rtrim(std::string& s, const char* t = ws) {
260 s.erase(s.find_last_not_of(t) + 1);
261 return s;
262}
263
264// trim from beginning of string (left)
265inline std::string& ltrim(std::string& s, const char* t = ws) {
266 s.erase(0, s.find_first_not_of(t));
267 return s;
268}
269
270// trim from both ends of string (right then left)
271inline std::string& trim(std::string& s, const char* t = ws) { return ltrim(rtrim(s, t), t); }
