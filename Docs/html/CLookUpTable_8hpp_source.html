108 /*! \brief
109 * Returns the index to the variable in the lookup table.
110 */
111 inline unsigned int GetIndexOfVar(const std::string& nameVar) const {
112 int index = find(names_var.begin(), names_var.end(), nameVar) - names_var.begin();
113 if (index == int(names_var.size())) {
114 index = -1;
115 std::string error_msg = "Variable '";
116 error_msg.append(nameVar);
117 error_msg.append("' is not in the lookup table.");
118 SU2_MPI::Error(error_msg, CURRENT_FUNCTION);
119 }
120 return index;
121 }

261 /*!
262 * \brief Compute the area of a triangle given the 3 points of the triangle.
263 * \param[in] x1 - The coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
264 * \param[in] y1 - The coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
265 * \param[in] x2 - The coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
266 * \param[in] y2 - The coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
267 * \param[in] x3 - The coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
268 * \param[in] y3 - The coordinates of the points P1(x1,y1), P2(x2,y2) and P3(x3,y3).
269 * \returns The absolute value of the area of the triangle.
270 */
271 inline su2double TriArea(su2double x1, su2double y1, su2double x2, su2double y2, su2double x3, su2double y3) {
272 return abs((x1 * (y2 - y3) + x2 * (y3 - y1) + x3 * (y1 - y2)) * 0.5);
273 }
