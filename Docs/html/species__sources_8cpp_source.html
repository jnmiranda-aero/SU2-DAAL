69template <class T>
70CNumerics::ResidualType<> CSourceAxisymmetric_Species<T>::ComputeResidual(const CConfig* config) {
71 /*--- Preaccumulation ---*/
72 AD::StartPreacc();
73 AD::SetPreaccIn(ScalarVar_i, nVar);
74 AD::SetPreaccIn(Volume);
75
76 if (incompressible) {
77 AD::SetPreaccIn(V_i, nDim+6);
78 }
79 else {
80 AD::SetPreaccIn(V_i, nDim+7);
81 }
82
83 /*--- Initialization. ---*/
84 for (auto iVar = 0u; iVar < nVar; iVar++) {
85 residual[iVar] = 0.0;
86 for (auto jVar = 0; jVar < nVar; jVar++) {
87 jacobian[iVar][jVar] = 0.0;
88 }
89 }
90
91 /*--- Contribution due to 2D axisymmetric formulation ---*/
92 if (Coord_i[1] > EPS) {
93 AD::SetPreaccIn(Coord_i[1]);
94 AD::SetPreaccIn(Diffusion_Coeff_i, nVar);
95 AD::SetPreaccIn(ScalarVar_Grad_i, nVar, nDim);
96
97 const su2double yinv = 1.0 / Coord_i[1];
98
99 const su2double Density_i = V_i[idx.Density()];
100
101 su2double Velocity_i[3];
102 for (auto iDim = 0u; iDim < nDim; iDim++)
103 Velocity_i[iDim] = V_i[idx.Velocity() + iDim];
104
105 /*--- Inviscid component of the source term. When div(v)=0, this term does not occur ---*/
106
107 if (!bounded_scalar) {
108 for (auto iVar = 0u; iVar < nVar; iVar++)
109 residual[iVar] -= yinv * Volume * Density_i * ScalarVar_i[iVar] * Velocity_i[1];
110
111 if (implicit) {
112 for (auto iVar = 0u; iVar < nVar; iVar++) {
113 jacobian[iVar][iVar] -= yinv * Volume * Velocity_i[1];
114 }
115 }
116 }
117
118 /*--- Add the viscous terms if necessary. ---*/
119
120 if (config->GetViscous()) {
121 su2double Mass_Diffusivity_Tur = 0.0;
122 if (turbulence)
123 Mass_Diffusivity_Tur = V_i[idx.EddyViscosity()] / Sc_t;
124
125 for (auto iVar = 0u; iVar < nVar; iVar++) {
126 residual[iVar] += yinv * Volume * (Density_i * Diffusion_Coeff_i[iVar] + Mass_Diffusivity_Tur) * ScalarVar_Grad_i[iVar][1];
127 }
128 }
129
130 }
131
132 AD::SetPreaccOut(residual, nVar);
133 AD::EndPreacc();
134
135 return ResidualType<>(residual, jacobian, nullptr);
136}
