48void CSquareMatrixCM::Transpose() {
49 for (int j = 1; j < Size(); ++j)
50 for (int i = 0; i < j; ++i) swap(mat(i, j), mat(j, i));

53void CSquareMatrixCM::Invert() {
54#ifdef HAVE_LAPACK
55
56 /*--- Computation of the inverse using the Lapack routines. ---*/
57 int sz = Size();
58 int info;
59 vector<int> ipiv(sz);
60 vector<passivedouble> work(sz);
61
62 dgetrf_(&sz, &sz, mat.data(), &sz, ipiv.data(), &info);
63 if (info != 0) SU2_MPI::Error(string("Matrix is singular"), CURRENT_FUNCTION);
64
65 dgetri_(&sz, mat.data(), &sz, ipiv.data(), work.data(), &sz, &info);
66 if (info != 0) SU2_MPI::Error(string("Matrix inversion failed"), CURRENT_FUNCTION);
67
68#else
69 CBlasStructure::inverse(Size(), mat);
70#endif
71}

73void CSquareMatrixCM::MatMatMult(const char side, const ColMajorMatrix<passivedouble>& mat_in,
74 ColMajorMatrix<passivedouble>& mat_out) const {
75 /*--- Check the type of multiplication to be carried out. ---*/
76 if (side == 'L' || side == 'l') {
77 /*--- Left side: mat_out = this * mat_in. Set some sizes
78 and allocate the memory for mat_out. ---*/
79 const int M = Size(), N = mat_in.cols();
80 assert(M == static_cast<int>(mat_in.rows()));
81
82 mat_out.resize(M, N);
83
84#ifdef HAVE_LAPACK
85
86 /*--- The Lapack/blas function dgemm is used to carry out
87 the matrix matrix multiplication. ---*/
88 passivedouble alpha = 1.0, beta = 0.0;
89 char trans = 'N';
90
91 DGEMM(&trans, &trans, &M, &N, &M, &alpha, mat.data(), &M, mat_in.data(), &M, &beta, mat_out.data(), &M);
92#else
93 /*--- Naive product. ---*/
94 for (int i = 0; i < M; ++i) {
95 for (int j = 0; j < N; ++j) {
96 mat_out(i, j) = 0.0;
97 for (int k = 0; k < M; ++k) mat_out(i, j) += mat(i, k) * mat_in(k, j);
98 }
99 }
100#endif
101
102 } else {
103 /*--- Right_side: mat_out = mat_in * this. Set some sizes
104 and allocate the memory for mat_out. ---*/
105 const int M = mat_in.rows(), N = Size();
106 assert(N == static_cast<int>(mat_in.cols()));
107
108 mat_out.resize(M, N);
109
110#ifdef HAVE_LAPACK
111
112 /*--- The Lapack/blas function dgemm is used to carry out
113 the matrix matrix multiplication. ---*/
114 passivedouble alpha = 1.0, beta = 0.0;
115 char trans = 'N';
116
117 DGEMM(&trans, &trans, &M, &N, &N, &alpha, mat_in.data(), &M, mat.data(), &N, &beta, mat_out.data(), &M);
118#else
119 /*--- Naive product. ---*/
120 for (int i = 0; i < M; ++i) {
121 for (int j = 0; j < N; ++j) {
122 mat_out(i, j) = 0.0;
123 for (int k = 0; k < N; ++k) mat_out(i, j) += mat_in(i, k) * mat(k, j);
124 }
125 }
126#endif
127 }
