41/*!
42 * \class CWallModel
43 * \brief Base class for defining the LES wall model.
44 * \author: E. van der Weide, T. Economon, P. Urbanczyk
45 * \version 8.0.1 "Harrier"
46 */
47class CWallModel {
48 public:
49 /*!
50 * \brief Constructor of the class.
51 * \param[in] config - Definition of the particular problem.
52 */
53 CWallModel(CConfig* config);
54
55 /*!
56 * \brief Destructor of the class.
57 */
58 virtual ~CWallModel(void) = default;
59
60 /*!
61 * \brief Virtual function, which computes the wall shear stress and heat flux
62 from the data at the exchange location.
63 * \param[in] tExchange - Temperature at the exchange location.
64 * \param[in] velExchange - Velocity at the exchange location.
65 * \param[in] muExchange - Laminar viscosity at the exchange location.
66 * \param[in] pExchange - Pressure at the exchange location.
67 * \param[in] Wall_HeatFlux - Value of the wall heat flux, if prescribed.
68 * \param[in] HeatFlux_Prescribed - Whether or not the wall heat flux is prescribed.
69 * \param[in] Wall_Temperature - Value of the wall temperature, if prescribed.
70 * \param[in] Temperature_Prescribed - Wheter or not the wall temperature is prescribed.
71 * \param[in] FluidModel - Fluid model used in the solver.
72 * \param[out] tauWall - Wall shear stress, to be computed.
73 * \param[out] qWall - Wall hear flux, to be computed (if not prescribed).
74 * \param[out] ViscosityWall - Laminar viscosity at the wall, to be computed.
75 * \param[out] OverCvWall - Thermal conductivity divided by Cv at the wall,
76 to be computed.
77 */
78 virtual void WallShearStressAndHeatFlux(const su2double tExchange, const su2double velExchange,
79 const su2double muExchange, const su2double pExchange,
80 const su2double Wall_HeatFlux, const bool HeatFlux_Prescribed,
81 const su2double TWall, const bool Temperature_Prescribed,
82 CFluidModel* FluidModel, su2double& tauWall, su2double& qWall,
83 su2double& ViscosityWall, su2double& kOverCvWall);
84
85 protected:
86 su2double h_wm; /*!< \brief The thickness of the wall model. This is also basically the exchange location */
87 su2double Pr_lam; /*!< \brief Laminar Prandtl number. */
88 su2double Pr_turb; /*!< \brief Turbulent Prandtl number. */
89 su2double karman; /*!< \brief von Karman constant. */
90
91 private:
92 /*!
93 * \brief Default constructor of the class, disabled.
94 */
95 CWallModel(void);
96};
