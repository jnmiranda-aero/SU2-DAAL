84  /*!
85  * \brief Add contribution from diffusion due to axisymmetric formulation to 2D residual
86  */
87  inline void ResidualAxisymmetricDiffusion(su2double sigma) {
88  if (Coord_i[1] < EPS) return;
89 
90  const su2double yinv = 1.0 / Coord_i[1];
91  const su2double& nue = ScalarVar_i[0];
92 
93  const auto& density = V_i[idx.Density()];
94  const auto& laminar_viscosity = V_i[idx.LaminarViscosity()];
95 
96  const su2double nu = laminar_viscosity/density;
97 
98  su2double nu_e;
99 
100  if (options.version == SA_OPTIONS::NEG && nue < 0.0) {
101  const su2double cn1 = 16.0;
102  const su2double Xi = nue / nu;
103  const su2double fn = (cn1 + Xi*Xi*Xi) / (cn1 - Xi*Xi*Xi);
104  nu_e = nu + fn * nue;
105  } else {
106  nu_e = nu + nue;
107  }
108 
109  /* Diffusion source term */
110  const su2double dv_axi = (1.0/sigma)*nu_e*ScalarVar_Grad_i[0][1];
111 
112  Residual += yinv * dv_axi * Volume;
113  }

133  /*!
134  * \brief Residual for source term integration.
135  * \param[in] config - Definition of the particular problem.
136  * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
137  */
138  ResidualType<> ComputeResidual(const CConfig* config) override {
139  const auto& density = V_i[idx.Density()];
140  const auto& laminar_viscosity = V_i[idx.LaminarViscosity()];
141 
142  AD::StartPreacc();
143  AD::SetPreaccIn(density, laminar_viscosity, StrainMag_i, ScalarVar_i[0], Volume, dist_i, roughness_i);
144  AD::SetPreaccIn(Vorticity_i, 3);
145  AD::SetPreaccIn(PrimVar_Grad_i + idx.Velocity(), nDim, nDim);
146  AD::SetPreaccIn(ScalarVar_Grad_i[0], nDim);
147 
148  /*--- Common auxiliary variables and constants of the model. ---*/
149  CSAVariables var;
150 
151  Residual = 0.0;
152  Jacobian_i[0] = 0.0;
153 
154  if (dist_i > 1e-10) {
155 
156  var.dist_i_2 = pow(dist_i, 2);
157  const su2double nu = laminar_viscosity / density;
158  var.inv_k2_d2 = 1.0 / (var.k2 * var.dist_i_2);
159 
160  /*--- Modified values for roughness, roughness_i = 0 for smooth walls and Ji remains the same.
161  * Ref: Aupoix, B. and Spalart, P. R., "Extensions of the Spalart-Allmaras Turbulence Model to Account for Wall
162  * Roughness," International Journal of Heat and Fluid Flow, Vol. 24, 2003, pp. 454-462.
163  * See https://turbmodels.larc.nasa.gov/spalart.html#sarough for detailed explanation. ---*/
164  var.Ji = ScalarVar_i[0] / nu + var.cr1 * (roughness_i / (dist_i + EPS));
165  var.d_Ji = 1.0 / nu;
166 
167  const su2double Ji_2 = pow(var.Ji, 2);
168  const su2double Ji_3 = Ji_2 * var.Ji;
169 
170  var.fv1 = Ji_3 / (Ji_3 + var.cv1_3);
171  var.d_fv1 = 3 * Ji_2 * var.cv1_3 / (nu * pow(Ji_3 + var.cv1_3, 2));
172 
173  /*--- Using a modified relation so as to not change the Shat that depends on fv2.
174  * From NASA turb modeling resource and 2003 paper. ---*/
175  var.fv2 = 1 - ScalarVar_i[0] / (nu + ScalarVar_i[0] * var.fv1);
176  var.d_fv2 = -(1 / nu - Ji_2 * var.d_fv1) / pow(1 + var.Ji * var.fv1, 2);
177 
178  /*--- Evaluate Omega with a rotational correction term. ---*/
179 
180  Omega::get(Vorticity_i, nDim, PrimVar_Grad_i + idx.Velocity(), var);
181 
182  /*--- Compute modified vorticity ---*/
183  ModVort::get(ScalarVar_i[0], nu, var);
184  var.inv_Shat = 1.0 / var.Shat;
185  var.Prod = var.Shat;
186 
187  /*--- Dacles-Mariani et. al. rotation correction ("-R"). ---*/
188  if (options.rot) {
189  var.Prod += var.CRot * min(0.0, StrainMag_i - var.Omega);
190  /*--- Do not allow negative production for SA-neg. ---*/
191  if (ScalarVar_i[0] < 0) var.Prod = abs(var.Prod);
192  }
193 
194  /*--- Compute ft2 term ---*/
195  ft2::get(var);
196 
197  /*--- Compute auxiliary function r ---*/
198  rFunc::get(ScalarVar_i[0], var);
199 
200  var.g = var.r + var.cw2 * (pow(var.r, 6) - var.r);
201  var.g_6 = pow(var.g, 6);
202  var.glim = pow((1 + var.cw3_6) / (var.g_6 + var.cw3_6), 1.0 / 6.0);
203  var.fw = var.g * var.glim;
204 
205  var.d_g = var.d_r * (1 + var.cw2 * (6 * pow(var.r, 5) - 1));
206  var.d_fw = var.d_g * var.glim * (1 - var.g_6 / (var.g_6 + var.cw3_6));
207 
208  var.norm2_Grad = GeometryToolbox::SquaredNorm(nDim, ScalarVar_Grad_i[0]);
209 
210  if (options.bc) {
211  /*--- BC transition model (2020 revision). This should only be used with SA-noft2.
212  * TODO: Consider making this part of the "SourceTerms" template. ---*/
213  const su2double chi_1 = 0.002;
214  const su2double chi_2 = 50.0;
215 
216  /*--- Turbulence intensity is u'/U so we multiply by 100 to get percentage. ---*/
217  const su2double tu = 100.0 * config->GetTurbulenceIntensity_FreeStream();
218  const su2double nu_t = ScalarVar_i[0] * var.fv1;
219 
220  const su2double re_v = density * var.dist_i_2 / laminar_viscosity * var.Omega;
221  const su2double re_theta = re_v / 2.193;
222  /*--- Menter correlation. ---*/
223  const su2double re_theta_t = 803.73 * pow(tu + 0.6067, -1.027);
224 
225  const su2double term1 = sqrt(max(re_theta - re_theta_t, 0.0) / (chi_1 * re_theta_t));
226  const su2double term2 = sqrt(max((nu_t * chi_2) / nu, 0.0));
227  const su2double term_exponential = (term1 + term2);
228 
229  intermittency_eff_i = 1.0 - exp(-term_exponential);
230  var.intermittency = intermittency_eff_i;
231  var.interDestrFactor = 1;
232 
233  } else if (transition_LM){
234 
235  var.intermittency = intermittency_eff_i;
236  // Is wrong the reference from NASA?
237  // Original max(min(gamma, 0.5), 1.0) always gives 1 as result.
238  var.interDestrFactor = min(max(intermittency_i, 0.5), 1.0);
239 
240  } else {
241  /*--- Do not modify the production. ---*/
242  var.intermittency = 1.0;
243  var.interDestrFactor = 1.0;
244  }
245 
246  /*--- Compute production, destruction and cross production and jacobian ---*/
247  su2double Production = 0.0, Destruction = 0.0, CrossProduction = 0.0;
248  SourceTerms::get(ScalarVar_i[0], var, Production, Destruction, CrossProduction, Jacobian_i[0]);
249 
250  Residual = (Production - Destruction + CrossProduction) * Volume;
251 
252  if (axisymmetric) ResidualAxisymmetricDiffusion(var.sigma);
253 
254  Jacobian_i[0] *= Volume;
255  }
256 
257  AD::SetPreaccOut(Residual);
258  AD::EndPreacc();
259 
260  return ResidualType<>(&Residual, &Jacobian_i, nullptr);
261  }

289 struct Edw {
290  template <class MatrixType>

331 /*!
332  * \brief SA classes to compute the modified vorticity (\tilde{S}) and its derivative.
333  * \ingroup SourceDiscr
334  * \param[in] nue: SA variable.
335  * \param[in] nu: Laminar viscosity.
336  * \param[in,out] var: Common SA variables struct.
337  */
338 struct ModVort {
339 
340 /*! \brief Baseline. */
341 struct Bsl {
342  static void get(const su2double& nue, const su2double& nu, CSAVariables& var) {
343  const su2double Sbar = nue * var.fv2 * var.inv_k2_d2;
344  const su2double c2 = 0.7, c3 = 0.9;
345 
346  /*--- Limiting of \hat{S} based on "Modifications and Clarifications for the Implementation of the Spalart-Allmaras Turbulence Model"
347  * Note 1 option c in https://turbmodels.larc.nasa.gov/spalart.html ---*/
348  if (Sbar >= - c2 * var.Omega) {
349  var.Shat = var.Omega + Sbar;
350  } else {
351  const su2double Num = var.Omega * (c2 * c2 * var.Omega + c3 * Sbar);
352  const su2double Den = (c3 - 2 * c2) * var.Omega - Sbar;
353  var.Shat = var.Omega + Num / Den;
354  }
355  if (var.Shat <= 1e-10) {
356  var.Shat = 1e-10;
357  var.d_Shat = 0.0;
358  } else {
359  var.d_Shat = (var.fv2 + nue * var.d_fv2) * var.inv_k2_d2;
360  }
361  }
362 };
363 
364 /*! \brief Edward. */
365 struct Edw {
366  static void get(const su2double& nue, const su2double& nu, CSAVariables& var) {
367  var.Shat = max(var.Omega * ((1.0 / max(var.Ji, 1.0e-16)) + var.fv1), 1.0e-16);
368  var.Shat = max(var.Shat, 1.0e-10);
369  if (var.Shat <= 1.0e-10) {
370  var.d_Shat = 0.0;
371  } else {
372  var.d_Shat = -var.Omega * pow(var.Ji, -2) / nu + var.Omega * var.d_fv1;
373  }
374  }
375 };

425  * \brief SA source terms classes: production, destruction and cross-productions term and their derivative.
426  * \ingroup SourceDiscr
427  * \param[in] nue: SA variable.
428  * \param[in] var: Common SA variables struct.
429  * \param[out] production: Production term.
430  * \param[out] destruction: Destruction term.

432  * \param[out] jacobian: Derivative of the combined source term wrt nue.
433  */
434 struct SourceTerms {
435 
436 /*! \brief Baseline (Original SA model). */
437 struct Bsl {
438  static void get(const su2double& nue, const CSAVariables& var, su2double& production, su2double& destruction,
439  su2double& cross_production, su2double& jacobian) {
440  ComputeProduction(nue, var, production, jacobian);
441  ComputeDestruction(nue, var, destruction, jacobian);
442  ComputeCrossProduction(nue, var, cross_production, jacobian);
443  }
444 
445  static void ComputeProduction(const su2double& nue, const CSAVariables& var, su2double& production,
446  su2double& jacobian) {
447  const su2double factor = var.intermittency * var.cb1;
448  production = factor * (1.0 - var.ft2) * var.Prod * nue;
449  jacobian += factor * (-var.Prod * nue * var.d_ft2 + (1.0 - var.ft2) * (nue * var.d_Shat + var.Prod));
450  }
451 
452  static void ComputeDestruction(const su2double& nue, const CSAVariables& var, su2double& destruction,
453  su2double& jacobian) {
454  const su2double cb1_k2 = var.cb1 / var.k2;
455  const su2double factor = var.cw1 * var.fw - cb1_k2 * var.ft2;
456  destruction = var.interDestrFactor * factor * pow(nue, 2) / var.dist_i_2;
457  jacobian -= var.interDestrFactor * ((var.cw1 * var.d_fw - cb1_k2 * var.d_ft2) * pow(nue, 2) + factor * 2 * nue) / var.dist_i_2;
458  }
459 
460  static void ComputeCrossProduction(const su2double& nue, const CSAVariables& var, su2double& cross_production,
461  su2double&) {
462  cross_production = var.cb2_sigma * var.norm2_Grad;
463  /*--- No contribution to the jacobian. ---*/
464  }
465 };

467 /*! \brief Negative. */
468 struct Neg {
469  static void get(const su2double& nue, const CSAVariables& var, su2double& production, su2double& destruction,
470  su2double& cross_production, su2double& jacobian) {
471  if (nue > 0.0) {
472  Bsl::get(nue, var, production, destruction, cross_production, jacobian);
473  } else {
474  ComputeProduction(nue, var, production, jacobian);
475  ComputeDestruction(nue, var, destruction, jacobian);
476  ComputeCrossProduction(nue, var, cross_production, jacobian);
477  }

479 
480  static void ComputeProduction(const su2double& nue, const CSAVariables& var, su2double& production,
481  su2double& jacobian) {
482  const su2double dP_dnu = var.intermittency * var.cb1 * (1.0 - var.ct3) * var.Prod;
483  production = dP_dnu * nue;
484  jacobian += dP_dnu;
485  }
486 
487  static void ComputeDestruction(const su2double& nue, const CSAVariables& var, su2double& destruction,
488  su2double& jacobian) {
489  /*--- The destruction when nue < 0 is added instead of the usual subtraction, hence the negative sign. ---*/
490  const su2double dD_dnu = -var.cw1 * nue / var.dist_i_2;
491  destruction = dD_dnu * nue * var.interDestrFactor;
492  jacobian -= 2 * dD_dnu * var.interDestrFactor;
493  }
494 
495  static void ComputeCrossProduction(const su2double& nue, const CSAVariables& var, su2double& cross_production,
496  su2double& jacobian) {
497  Bsl::ComputeCrossProduction(nue, var, cross_production, jacobian);
498  }

535  /*!
536  * \brief Residual for source term integration.
537  * \param[in] config - Definition of the particular problem.
538  * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
539  */
540  ResidualType ComputeResidual(const CConfig* config) override {
541  /*--- Residual from standard SA ---*/
542  ParentClass::ComputeResidual(config);
543 
544  /*--- Compressibility Correction term ---*/
545  const auto& pressure = V_i[idx.Pressure()];
546  const auto& density = V_i[idx.Density()];
547  const su2double sound_speed = sqrt(pressure * Gamma / density);
548  su2double aux_cc = 0;
549  for (unsigned short iDim = 0; iDim < nDim; ++iDim) {
550  for (unsigned short jDim = 0; jDim < nDim; ++jDim) {
551  aux_cc += pow(PrimVar_Grad_i[idx.Velocity() + iDim][jDim], 2);
552  }
553  }

555  const su2double CompCorrection = 0.5 * ScalarVar_i[0] * d_CompCorrection;
556 
557  /*--- Axisymmetric contribution ---*/
558  if (this->axisymmetric && this->Coord_i[1] > EPS) {
559  const su2double yinv = 1.0 / this->Coord_i[1];
560  const su2double nue = ScalarVar_i[0];
561  const su2double v = V_i[idx.Velocity() + 1];
562 
563  const su2double d_axiCorrection = 2.0 * c5 * nue * pow(v * yinv / sound_speed, 2) * Volume;
564  const su2double axiCorrection = 0.5 * nue * d_axiCorrection;
565 
566  this->Residual -= axiCorrection;
567  this->Jacobian_i[0] -= d_axiCorrection;
568  }
569 
570  this->Residual -= CompCorrection;
571  this->Jacobian_i[0] -= d_CompCorrection;
572 
573  return ResidualType(&this->Residual, &this->Jacobian_i, nullptr);
574  }

774  * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
775  */
776  ResidualType<> ComputeResidual(const CConfig* config) override {
777  AD::StartPreacc();
778  AD::SetPreaccIn(StrainMag_i);
779  AD::SetPreaccIn(ScalarVar_i, nVar);
780  AD::SetPreaccIn(ScalarVar_Grad_i, nVar, nDim);
781  AD::SetPreaccIn(Volume);
782  AD::SetPreaccIn(dist_i);
783  AD::SetPreaccIn(F1_i);
784  AD::SetPreaccIn(F2_i);
785  AD::SetPreaccIn(CDkw_i);
786  AD::SetPreaccIn(PrimVar_Grad_i, nDim + idx.Velocity(), nDim);
787  AD::SetPreaccIn(Vorticity_i, 3);
788  AD::SetPreaccIn(V_i[idx.Density()], V_i[idx.LaminarViscosity()], V_i[idx.EddyViscosity()]);
789  AD::SetPreaccIn(V_i[idx.Velocity() + 1]);
790  AD::SetPreaccIn(V_i[idx.SoundSpeed()]);
791 
792  Density_i = V_i[idx.Density()];
793  Laminar_Viscosity_i = V_i[idx.LaminarViscosity()];
794  Eddy_Viscosity_i = V_i[idx.EddyViscosity()];
795 
796  Residual[0] = 0.0;
797  Residual[1] = 0.0;
798  Jacobian_i[0][0] = 0.0;
799  Jacobian_i[0][1] = 0.0;
800  Jacobian_i[1][0] = 0.0;
801  Jacobian_i[1][1] = 0.0;
802 
803  /*--- Computation of blended constants for the source terms ---*/
804 
805  const su2double alfa_blended = F1_i * alfa_1 + (1.0 - F1_i) * alfa_2;
806  const su2double beta_blended = F1_i * beta_1 + (1.0 - F1_i) * beta_2;
807 
808  su2double eff_intermittency = 1.0;
809 
810  if (config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
811  AD::SetPreaccIn(intermittency_eff_i);
812  eff_intermittency = intermittency_eff_i;
813  }
814 
815  if (dist_i > 1e-10) {
816 
817  su2double diverg = 0.0;
818  for (unsigned short iDim = 0; iDim < nDim; iDim++)
819  diverg += PrimVar_Grad_i[iDim + idx.Velocity()][iDim];
820  if (axisymmetric && Coord_i[1] > EPS) {
821  AD::SetPreaccIn(Coord_i[1]);
822  diverg += V_i[idx.Velocity() + 1] / Coord_i[1];
823  }
824 
825  /*--- If using UQ methodolgy, calculate production using perturbed Reynolds stress matrix ---*/
826 
827  const su2double VorticityMag = GeometryToolbox::Norm(3, Vorticity_i);
828  su2double P_Base = 0;
829  su2double zetaFMt = 0.0;
830  const su2double Mt = sqrt(2.0 * ScalarVar_i[0]) / V_i[idx.SoundSpeed()];
831 
832  /*--- Apply production term modifications ---*/
833  switch (sstParsedOptions.production) {
834  case SST_OPTIONS::UQ:
835  ComputePerturbedRSM(nDim, Eig_Val_Comp, uq_permute, uq_delta_b, uq_urlx, PrimVar_Grad_i + idx.Velocity(),
836  Density_i, Eddy_Viscosity_i, ScalarVar_i[0], MeanPerturbedRSM);
837  P_Base = PerturbedStrainMag(ScalarVar_i[0]);
838  break;
839 
840  case SST_OPTIONS::V:
841  P_Base = VorticityMag;
842  break;
843 
844  case SST_OPTIONS::KL:
845  P_Base = sqrt(StrainMag_i*VorticityMag);
846  break;
847 
848  case SST_OPTIONS::COMP_Wilcox:
849  P_Base = StrainMag_i;
850  if (Mt >= 0.25) {
851  zetaFMt = 2.0 * (Mt * Mt - 0.25 * 0.25);
852  }
853  break;
854 
855  case SST_OPTIONS::COMP_Sarkar:
856  P_Base = StrainMag_i;
857  if (Mt >= 0.25) {
858  zetaFMt = 0.5 * (Mt * Mt);
859  }
860  break;
861 
862  default:
863  /*--- Base production term for SST-1994 and SST-2003 ---*/
864  P_Base = StrainMag_i;
865  break;
866  }
867 
868  /*--- Production limiter. ---*/
869  const su2double prod_limit = prod_lim_const * beta_star * Density_i * ScalarVar_i[1] * ScalarVar_i[0];
870 
871  su2double P = Eddy_Viscosity_i * pow(P_Base, 2);
872  su2double pk = max(0.0, min(P, prod_limit));
873 
874  const auto& eddy_visc_var = sstParsedOptions.version == SST_OPTIONS::V1994 ? VorticityMag : StrainMag_i;
875  const su2double zeta = max(ScalarVar_i[1], eddy_visc_var * F2_i / a1);
876 
877  /*--- Production limiter only for V2003, recompute for V1994. ---*/
878  su2double pw;
879  if (sstParsedOptions.version == SST_OPTIONS::V1994) {
880  pw = alfa_blended * Density_i * pow(P_Base, 2);
881  } else {
882  pw = (alfa_blended * Density_i / Eddy_Viscosity_i) * pk;
883  }
884 
885  /*--- Sustaining terms, if desired. Note that if the production terms are
886  larger equal than the sustaining terms, the original formulation is
887  obtained again. This is in contrast to the version in literature
888  where the sustaining terms are simply added. This latter approach could
889  lead to problems for very big values of the free-stream turbulence
890  intensity. ---*/
891  if (sstParsedOptions.sust) {
892  const su2double sust_k = beta_star * Density_i * kAmb * omegaAmb;
893  const su2double sust_w = beta_blended * Density_i * omegaAmb * omegaAmb;
894  pk = max(pk, sust_k);
895  pw = max(pw, sust_w);
896  }
897 
898  if (sstParsedOptions.production == SST_OPTIONS::COMP_Sarkar) {
899  const su2double Dilatation_Sarkar = -0.15 * pk * Mt + 0.2 * beta_star * (1.0 +zetaFMt) * Density_i * ScalarVar_i[1] * ScalarVar_i[0] * Mt * Mt;
900  pk += Dilatation_Sarkar;
901  }
902 
903  /*--- Dissipation ---*/
904 
905  su2double dk = beta_star * Density_i * ScalarVar_i[1] * ScalarVar_i[0] * (1.0 + zetaFMt);
906  su2double dw = beta_blended * Density_i * ScalarVar_i[1] * ScalarVar_i[1] * (1.0 - 0.09/beta_blended * zetaFMt);
907 
908  /*--- LM model coupling with production and dissipation term for k transport equation---*/
909  if (config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
910  pk = pk * eff_intermittency;
911  dk = min(max(eff_intermittency, 0.1), 1.0) * dk;
912  }
913 
914  /*--- Add the production terms to the residuals. ---*/
915 
916  Residual[0] += pk * Volume;
917  Residual[1] += pw * Volume;
918 
919  /*--- Add the dissipation terms to the residuals.---*/
920 
921  Residual[0] -= dk * Volume;
922  Residual[1] -= dw * Volume;
923 
924  /*--- Cross diffusion ---*/
925 
926  Residual[1] += (1.0 - F1_i) * CDkw_i * Volume;
927 
928  /*--- Contribution due to 2D axisymmetric formulation ---*/
929 
930  if (axisymmetric) ResidualAxisymmetricConvectionDiffusion(alfa_blended, zeta);
931 
932  /*--- Implicit part ---*/
933 
934  Jacobian_i[0][0] = -beta_star * ScalarVar_i[1] * Volume * (1.0 + zetaFMt);
935  Jacobian_i[0][1] = -beta_star * ScalarVar_i[0] * Volume * (1.0 + zetaFMt);
936  Jacobian_i[1][0] = 0.0;
937  Jacobian_i[1][1] = -2.0 * beta_blended * ScalarVar_i[1] * Volume * (1.0 - 0.09/beta_blended * zetaFMt);
938  }
939 
940  AD::SetPreaccOut(Residual, nVar);
941  AD::EndPreacc();
942 
943  return ResidualType<>(Residual, Jacobian_i, nullptr);
944  }
