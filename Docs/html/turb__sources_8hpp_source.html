32/*!
33 * \class CSAVariables
34 * \ingroup SourceDiscr
35 * \brief Structure with SA common auxiliary functions and constants.
36 */
37struct CSAVariables {
38 /*--- List of constants ---*/
39 const su2double cv1_3 = pow(7.1, 3);
40 const su2double k2 = pow(0.41, 2);
41 const su2double cb1 = 0.1355;
42 const su2double cw2 = 0.3;
43 const su2double ct3 = 1.2;
44 const su2double ct4 = 0.5;
45 const su2double cw3_6 = pow(2, 6);
46 const su2double sigma = 2.0 / 3.0;
47 const su2double cb2 = 0.622;
48 const su2double cb2_sigma = cb2 / sigma;
49 const su2double cw1 = cb1 / k2 + (1 + cb2) / sigma;
50 const su2double cr1 = 0.5;
51 const su2double CRot = 1.0;
52 const su2double c2 = 0.7, c3 = 0.9;
53
54 /*--- List of auxiliary functions ---*/
55 su2double ft2, d_ft2, r, d_r, g, d_g, glim, fw, d_fw, Ji, d_Ji, S, Shat, d_Shat, fv1, d_fv1, fv2, d_fv2;
56
57 /*--- List of helpers ---*/
58 su2double Omega, dist_i_2, inv_k2_d2, inv_Shat, g_6, norm2_Grad;
59
60 su2double intermittency, interDestrFactor;
61};

63/*!
64 * \class CSourceBase_TurbSA
65 * \ingroup SourceDiscr
66 * \brief Class for integrating the source terms of the Spalart-Allmaras turbulence model equation.
67 * The variables that are subject to change in each variation/correction have their own class.
68 * \note Additional source terms (e.g. compressibility) are implemented as decorators.
69 */
70template <class FlowIndices, class Omega, class ft2, class ModVort, class rFunc, class SourceTerms>
71class CSourceBase_TurbSA : public CNumerics {
72 protected:
73
74 /*--- Residual and Jacobian ---*/
75 su2double Residual, *Jacobian_i;
76 su2double Jacobian_Buffer; /*!< \brief Static storage for the Jacobian (which needs to be pointer for return type). */
77
78 const FlowIndices idx; /*!< \brief Object to manage the access to the flow primitives. */
79 const SA_ParsedOptions options; /*!< \brief Struct with SA options. */
80
81 bool transition_LM;
82
83 public:
84 /*!
85 * \brief Constructor of the class.
86 * \param[in] nDim - Number of dimensions of the problem.
87 * \param[in] config - Definition of the particular problem.
88 */
89 CSourceBase_TurbSA(unsigned short nDim, const CConfig* config)
90 : CNumerics(nDim, 1, config),
91 idx(nDim, config->GetnSpecies()),
92 options(config->GetSAParsedOptions()),
93 transition_LM(config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
94 /*--- Setup the Jacobian pointer, we need to return su2double** but we know
95 * the Jacobian is 1x1 so we use this trick to avoid heap allocation. ---*/
96 Jacobian_i = &Jacobian_Buffer;
97 }
98
99
100 /*!
101 * \brief Residual for source term integration.
102 * \param[in] config - Definition of the particular problem.
103 * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
104 */
105 ResidualType<> ComputeResidual(const CConfig* config) override {
106 const auto& density = V_i[idx.Density()];
107 const auto& laminar_viscosity = V_i[idx.LaminarViscosity()];
108
109 AD::StartPreacc();
110 AD::SetPreaccIn(density, laminar_viscosity, StrainMag_i, ScalarVar_i[0], Volume, dist_i, roughness_i);
111 AD::SetPreaccIn(Vorticity_i, 3);
112 AD::SetPreaccIn(PrimVar_Grad_i + idx.Velocity(), nDim, nDim);
113 AD::SetPreaccIn(ScalarVar_Grad_i[0], nDim);
114
115 /*--- Common auxiliary variables and constants of the model. ---*/
116 CSAVariables var;
117
118 Residual = 0.0;
119 Jacobian_i[0] = 0.0;
120
121 /*--- Evaluate Omega with a rotational correction term. ---*/
122
123 Omega::get(Vorticity_i, nDim, PrimVar_Grad_i + idx.Velocity(), var);
124
125 /*--- Dacles-Mariani et. al. rotation correction ("-R"). ---*/
126 if (options.rot) {
127 var.Omega += var.CRot * min(0.0, StrainMag_i - var.Omega);
128 /*--- Do not allow negative production for SA-neg. ---*/
129 if (ScalarVar_i[0] < 0) var.Omega = abs(var.Omega);
130 }
131
132 if (dist_i > 1e-10) {
133 /*--- Vorticity ---*/
134 var.S = var.Omega;
135
136 var.dist_i_2 = pow(dist_i, 2);
137 const su2double nu = laminar_viscosity / density;
138 var.inv_k2_d2 = 1.0 / (var.k2 * var.dist_i_2);
139
140 /*--- Modified values for roughness, roughness_i = 0 for smooth walls and Ji remains the same.
141 * Ref: Aupoix, B. and Spalart, P. R., "Extensions of the Spalart-Allmaras Turbulence Model to Account for Wall
142 * Roughness," International Journal of Heat and Fluid Flow, Vol. 24, 2003, pp. 454-462.
143 * See https://turbmodels.larc.nasa.gov/spalart.html#sarough for detailed explanation. ---*/
144 var.Ji = ScalarVar_i[0] / nu + var.cr1 * (roughness_i / (dist_i + EPS));
145 var.d_Ji = 1.0 / nu;
146
147 const su2double Ji_2 = pow(var.Ji, 2);
148 const su2double Ji_3 = Ji_2 * var.Ji;
149
150 var.fv1 = Ji_3 / (Ji_3 + var.cv1_3);
151 var.d_fv1 = 3 * Ji_2 * var.cv1_3 / (nu * pow(Ji_3 + var.cv1_3, 2));
152
153 /*--- Using a modified relation so as to not change the Shat that depends on fv2.
154 * From NASA turb modeling resource and 2003 paper. ---*/
155 var.fv2 = 1 - ScalarVar_i[0] / (nu + ScalarVar_i[0] * var.fv1);
156 var.d_fv2 = -(1 / nu - Ji_2 * var.d_fv1) / pow(1 + var.Ji * var.fv1, 2);
157
158 /*--- Compute ft2 term ---*/
159 ft2::get(var);
160
161 /*--- Compute modified vorticity ---*/
162 ModVort::get(ScalarVar_i[0], nu, var);
163 var.inv_Shat = 1.0 / var.Shat;
164
165 /*--- Compute auxiliary function r ---*/
166 rFunc::get(ScalarVar_i[0], var);
167
168 var.g = var.r + var.cw2 * (pow(var.r, 6) - var.r);
169 var.g_6 = pow(var.g, 6);
170 var.glim = pow((1 + var.cw3_6) / (var.g_6 + var.cw3_6), 1.0 / 6.0);
171 var.fw = var.g * var.glim;
172
173 var.d_g = var.d_r * (1 + var.cw2 * (6 * pow(var.r, 5) - 1));
174 var.d_fw = var.d_g * var.glim * (1 - var.g_6 / (var.g_6 + var.cw3_6));
175
176 var.norm2_Grad = GeometryToolbox::SquaredNorm(nDim, ScalarVar_Grad_i[0]);
177
178 if (options.bc) {
179 /*--- BC transition model (2020 revision). This should only be used with SA-noft2.
180 * TODO: Consider making this part of the "SourceTerms" template. ---*/
181 const su2double chi_1 = 0.002;
182 const su2double chi_2 = 50.0;
183
184 /*--- Turbulence intensity is u'/U so we multiply by 100 to get percentage. ---*/
185 const su2double tu = 100.0 * config->GetTurbulenceIntensity_FreeStream();
186 const su2double nu_t = ScalarVar_i[0] * var.fv1;
187
188 const su2double re_v = density * var.dist_i_2 / laminar_viscosity * var.Omega;
189 const su2double re_theta = re_v / 2.193;
190 /*--- Menter correlation. ---*/
191 const su2double re_theta_t = 803.73 * pow(tu + 0.6067, -1.027);
192
193 const su2double term1 = sqrt(max(re_theta - re_theta_t, 0.0) / (chi_1 * re_theta_t));
194 const su2double term2 = sqrt(max((nu_t * chi_2) / nu, 0.0));
195 const su2double term_exponential = (term1 + term2);
196
197 intermittency_eff_i = 1.0 - exp(-term_exponential);
198 var.intermittency = intermittency_eff_i;
199 var.interDestrFactor = 1;
200
201 } else if (transition_LM){
202
203 var.intermittency = intermittency_eff_i;
204 //var.intermittency = 1.0;
205 // Is wrong the reference from NASA?
206 // Original max(min(gamma, 0.5), 1.0) always gives 1 as result.
207 var.interDestrFactor = min(max(intermittency_i, 0.5), 1.0);
208
209 } else {
210 /*--- Do not modify the production. ---*/
211 var.intermittency = 1.0;
212 var.interDestrFactor = 1.0;
213 }
214
215 /*--- Compute production, destruction and cross production and jacobian ---*/
216 su2double Production = 0.0, Destruction = 0.0, CrossProduction = 0.0;
217 SourceTerms::get(ScalarVar_i[0], var, Production, Destruction, CrossProduction, Jacobian_i[0]);
218
219 Residual = (Production - Destruction + CrossProduction) * Volume;
220 Jacobian_i[0] *= Volume;
221 }
222
223 AD::SetPreaccOut(Residual);
224 AD::EndPreacc();
225
226 return ResidualType<>(&Residual, &Jacobian_i, nullptr);
227 }
228};

236/*!
237 * \brief SA strain rate classes.
238 * \ingroup SourceDiscr
239 * \param[in] vorticity: Vorticity array.
240 * \param[in] nDim: Problem dimension.
241 * \param[in] velocity_grad: Velocity gradients.
242 * \param[out] var: Common SA variables struct (to set Omega).
243 */
244struct Omega {
245
246/*! \brief Baseline. */
247struct Bsl {
248 template <class MatrixType>
249 static void get(const su2double* vorticity, unsigned short, const MatrixType&, CSAVariables& var) {
250 var.Omega = GeometryToolbox::Norm(3, vorticity);
251 }
252};
253
254/*! \brief Edward. Here Omega is the Strain Rate. */
255struct Edw {
256 template <class MatrixType>
257 static void get(const su2double*, unsigned short nDim, const MatrixType& velocity_grad, CSAVariables& var) {
258 su2double Sbar = 0.0;
259 for (unsigned short iDim = 0; iDim < nDim; ++iDim) {
260 for (unsigned short jDim = 0; jDim < nDim; ++jDim) {
261 Sbar += (velocity_grad[iDim][jDim] + velocity_grad[jDim][iDim]) * velocity_grad[iDim][jDim];
262 }
263 }
264 for (unsigned short iDim = 0; iDim < nDim; ++iDim) {
265 Sbar -= (2.0 / 3.0) * pow(velocity_grad[iDim][iDim], 2);
266 }
267 var.Omega = sqrt(max(Sbar, 0.0));
268 }
269};
270};
271
272/*!
273 * \brief SA classes to set the ft2 term and its derivative.
274 * \ingroup SourceDiscr
275 * \param[in,out] var: Common SA variables struct.
276 */
277struct ft2 {
278
279/*! \brief No-ft2 term. */
280struct Zero {
281 static void get(CSAVariables& var) {
282 var.ft2 = 0.0;
283 var.d_ft2 = 0.0;
284 }
285};
286
287/*! \brief Non-zero ft2 term according to the literature. */
288struct Nonzero {
289 static void get(CSAVariables& var) {
290 const su2double xsi2 = pow(var.Ji, 2);
291 var.ft2 = var.ct3 * exp(-var.ct4 * xsi2);
292 var.d_ft2 = -2.0 * var.ct4 * var.Ji * var.ft2 * var.d_Ji;
293 }
294};
295};
296
297/*!
298 * \brief SA classes to compute the modified vorticity (\tilde{S}) and its derivative.
299 * \ingroup SourceDiscr
300 * \param[in] nue: SA variable.
301 * \param[in] nu: Laminar viscosity.
302 * \param[in,out] var: Common SA variables struct.
303 */
304struct ModVort {
305
306/*! \brief Baseline. */
307struct Bsl {
308 static void get(const su2double& nue, const su2double& nu, CSAVariables& var) {
309 const su2double Sbar = nue * var.fv2 * var.inv_k2_d2;
310 const su2double c2 = 0.7, c3 = 0.9;
311
312 /*--- Limiting of \hat{S} based on "Modifications and Clarifications for the Implementation of the Spalart-Allmaras Turbulence Model"
313 * Note 1 option c in https://turbmodels.larc.nasa.gov/spalart.html ---*/
314 if (Sbar >= - c2 * var.S) {
315 var.Shat = var.S + Sbar;
316 } else {
317 const su2double Num = var.S * (c2 * c2 * var.S + c3 * Sbar);
318 const su2double Den = (c3 - 2 * c2) * var.S - Sbar;
319 var.Shat = var.S + Num / Den;
320 }
321 if (var.Shat <= 1e-10) {
322 var.Shat = 1e-10;
323 var.d_Shat = 0.0;
324 } else {
325 var.d_Shat = (var.fv2 + nue * var.d_fv2) * var.inv_k2_d2;
326 }
327 }
328};
329
330/*! \brief Edward. */
331struct Edw {
332 static void get(const su2double& nue, const su2double& nu, CSAVariables& var) {
333 var.Shat = max(var.S * ((1.0 / max(var.Ji, 1.0e-16)) + var.fv1), 1.0e-16);
334 var.Shat = max(var.Shat, 1.0e-10);
335 if (var.Shat <= 1.0e-10) {
336 var.d_Shat = 0.0;
337 } else {
338 var.d_Shat = -var.S * pow(var.Ji, -2) / nu + var.S * var.d_fv1;
339 }
340 }
341};
342
343/*! \brief Negative. */
344struct Neg {
345 static void get(const su2double& nue, const su2double& nu, CSAVariables& var) {
346 if (nue > 0.0) {
347 // Baseline solution
348 Bsl::get(nue, nu, var);
349 } else {
350 var.Shat = 1.0e-10;
351 var.d_Shat = 0.0;
352 }
353 /*--- Don't check whether Sbar <>= -cv2*S.
354 * Steven R. Allmaras, Forrester T. Johnson and Philippe R. Spalart -
355 * "Modifications and Clarifications for the Implementation of the Spalart-Allmaras Turbulence Model" eq. 12
356 * No need for Sbar ---*/
357 }
358};
359};
360
361/*!
362 * \brief SA auxiliary function r and its derivative.
363 * \ingroup SourceDiscr
364 * \param[in] nue: SA variable.
365 * \param[in,out] var: Common SA variables struct.
366 */
367struct r {
368
369/*! \brief Baseline. */
370struct Bsl {
371 static void get(const su2double& nue, CSAVariables& var) {
372 var.r = min(nue * var.inv_Shat * var.inv_k2_d2, 10.0);
373 var.d_r = (var.Shat - nue * var.d_Shat) * pow(var.inv_Shat, 2) * var.inv_k2_d2;
374 if (var.r >= 10.0) var.d_r = 0.0;
375 }
376};
377
378/*! \brief Edward. */
379struct Edw {
380 static void get(const su2double& nue, CSAVariables& var) {
381 var.r = min(nue * var.inv_Shat * var.inv_k2_d2, 10.0);
382 var.r = tanh(var.r) / tanh(1.0);
383
384 var.d_r = (var.Shat - nue * var.d_Shat) * pow(var.inv_Shat, 2) * var.inv_k2_d2;
385 var.d_r = (1 - pow(tanh(var.r), 2.0)) * (var.d_r) / tanh(1.0);
386 }
387};
388};

390/*!
391 * \brief SA source terms classes: production, destruction and cross-productions term and their derivative.
392 * \ingroup SourceDiscr
393 * \param[in] nue: SA variable.
394 * \param[in] var: Common SA variables struct.
395 * \param[out] production: Production term.
396 * \param[out] destruction: Destruction term.
397 * \param[out] cross_production: CrossProduction term.
398 * \param[out] jacobian: Derivative of the combined source term wrt nue.
399 */
400struct SourceTerms {
401
402/*! \brief Baseline (Original SA model). */
403struct Bsl {
404 static void get(const su2double& nue, const CSAVariables& var, su2double& production, su2double& destruction,
405 su2double& cross_production, su2double& jacobian) {
406 ComputeProduction(nue, var, production, jacobian);
407 ComputeDestruction(nue, var, destruction, jacobian);
408 ComputeCrossProduction(nue, var, cross_production, jacobian);
409 }
410
411 static void ComputeProduction(const su2double& nue, const CSAVariables& var, su2double& production,
412 su2double& jacobian) {
413 const su2double factor = var.intermittency * var.cb1;
414 production = factor * (1.0 - var.ft2) * var.Shat * nue;
415 jacobian += factor * (-var.Shat * nue * var.d_ft2 + (1.0 - var.ft2) * (nue * var.d_Shat + var.Shat));
416 }
417
418 static void ComputeDestruction(const su2double& nue, const CSAVariables& var, su2double& destruction,
419 su2double& jacobian) {
420 const su2double cb1_k2 = var.cb1 / var.k2;
421 const su2double factor = var.cw1 * var.fw - cb1_k2 * var.ft2;
422 destruction = var.interDestrFactor * factor * pow(nue, 2) / var.dist_i_2;
423 jacobian -= var.interDestrFactor * ((var.cw1 * var.d_fw - cb1_k2 * var.d_ft2) * pow(nue, 2) + factor * 2 * nue) / var.dist_i_2;
424 }
425
426 static void ComputeCrossProduction(const su2double& nue, const CSAVariables& var, su2double& cross_production,
427 su2double&) {
428 cross_production = var.cb2_sigma * var.norm2_Grad;
429 /*--- No contribution to the jacobian. ---*/
430 }
431};
432
433/*! \brief Negative. */
434struct Neg {
435 static void get(const su2double& nue, const CSAVariables& var, su2double& production, su2double& destruction,
436 su2double& cross_production, su2double& jacobian) {
437 if (nue > 0.0) {
438 Bsl::get(nue, var, production, destruction, cross_production, jacobian);
439 } else {
440 ComputeProduction(nue, var, production, jacobian);
441 ComputeDestruction(nue, var, destruction, jacobian);
442 ComputeCrossProduction(nue, var, cross_production, jacobian);
443 }
444 }
445
446 static void ComputeProduction(const su2double& nue, const CSAVariables& var, su2double& production,
447 su2double& jacobian) {
448 const su2double dP_dnu = var.intermittency * var.cb1 * (1.0 - var.ct3) * var.S;
449 production = dP_dnu * nue;
450 jacobian += dP_dnu;
451 }
452
453 static void ComputeDestruction(const su2double& nue, const CSAVariables& var, su2double& destruction,
454 su2double& jacobian) {
455 /*--- The destruction when nue < 0 is added instead of the usual subtraction, hence the negative sign. ---*/
456 const su2double dD_dnu = -var.cw1 * nue / var.dist_i_2;
457 destruction = dD_dnu * nue * var.interDestrFactor;
458 jacobian -= 2 * dD_dnu * var.interDestrFactor;
459 }
460
461 static void ComputeCrossProduction(const su2double& nue, const CSAVariables& var, su2double& cross_production,
462 su2double& jacobian) {
463 Bsl::ComputeCrossProduction(nue, var, cross_production, jacobian);
464 }
465};
466};

472/*!
473 * \class CCompressibilityCorrection
474 * \ingroup SourceDiscr
475 * \brief Mixing Layer Compressibility Correction (SA-comp).
476 */
477template <class ParentClass>
478class CCompressibilityCorrection final : public ParentClass {
479 private:
480 using ParentClass::Gamma;
481 using ParentClass::idx;
482 using ParentClass::nDim;
483 using ParentClass::PrimVar_Grad_i;
484 using ParentClass::ScalarVar_i;
485 using ParentClass::V_i;
486 using ParentClass::Volume;
487
488 using ResidualType = typename ParentClass::template ResidualType<>;
489
490 const su2double c5 = 3.5;
491
492 public:
493 /*!
494 * \brief Constructor of the class.
495 * \param[in] nDim - Number of dimensions of the problem.
496 * \param[in] config - Definition of the particular problem.
497 */
498 CCompressibilityCorrection(unsigned short nDim, const CConfig* config)
499 : ParentClass(nDim, config) {}
500
501 /*!
502 * \brief Residual for source term integration.
503 * \param[in] config - Definition of the particular problem.
504 * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
505 */
506 ResidualType ComputeResidual(const CConfig* config) override {
507 /*--- Residual from standard SA ---*/
508 ParentClass::ComputeResidual(config);
509
510 /*--- Compressibility Correction term ---*/
511 const auto& pressure = V_i[idx.Pressure()];
512 const auto& density = V_i[idx.Density()];
513 const su2double sound_speed = sqrt(pressure * Gamma / density);
514 su2double aux_cc = 0;
515 for (unsigned short iDim = 0; iDim < nDim; ++iDim) {
516 for (unsigned short jDim = 0; jDim < nDim; ++jDim) {
517 aux_cc += pow(PrimVar_Grad_i[idx.Velocity() + iDim][jDim], 2);
518 }
519 }
520 const su2double d_CompCorrection = 2.0 * c5 * ScalarVar_i[0] / pow(sound_speed, 2) * aux_cc * Volume;
521 const su2double CompCorrection = 0.5 * ScalarVar_i[0] * d_CompCorrection;
522
523 this->Residual -= CompCorrection;
524 this->Jacobian_i[0] -= d_CompCorrection;
525
526 return ResidualType(&this->Residual, &this->Jacobian_i, nullptr);
527 }
528};

552template <class BaseType, class... Ts>
553CNumerics* AddCompressibilityCorrection(bool use_comp, Ts... args) {
554 if (use_comp) {
555 return new detail::CCompressibilityCorrection<BaseType>(args...);
556 } else {
557 return new BaseType(args...);
558 }
559}
560
561template <class BaseType, class... Ts>
562CNumerics* SAFactoryImpl(bool use_ft2, Ts... args) {
563 if (use_ft2) {
564 return AddCompressibilityCorrection<typename BaseType::template type<ft2::Nonzero>>(args...);
565 } else {
566 return AddCompressibilityCorrection<typename BaseType::template type<ft2::Zero>>(args...);
567 }

572/*!
573 * \brief Creates an SA source based on the version and modifications/correction in the config.
574 * \ingroup SourceDiscr
575 */
576template <class FlowIndices>
577CNumerics* SAFactory(unsigned short nDim, const CConfig* config) {
578 const auto options = config->GetSAParsedOptions();
579
580 switch (options.version) {
581 case SA_OPTIONS::NONE:

602 const bool sustaining_terms = false;
603 const bool axisymmetric = false;

605 /*--- Closure constants ---*/
606 const su2double sigma_k_1, sigma_k_2, sigma_w_1, sigma_w_2, beta_1, beta_2, beta_star, a1, alfa_1, alfa_2;
607 const su2double prod_lim_const;
608
609 /*--- Ambient values for SST-SUST. ---*/
610 const su2double kAmb, omegaAmb;
611
612 su2double F1_i, F2_i, CDkw_i;
613 su2double Residual[2];
614 su2double* Jacobian_i[2];
615 su2double Jacobian_Buffer[4]; /// Static storage for the Jacobian (which needs to be pointer for return type).
616
617 /*!
618 * \brief Get strain magnitude based on perturbed reynolds stress matrix.
619 * \param[in] turb_ke: turbulent kinetic energy of the node.
620 */
621 inline su2double PerturbedStrainMag(su2double turb_ke) const {
622 /*--- Compute norm of perturbed strain rate tensor. ---*/
623
624 su2double perturbedStrainMag = 0;
625 for (unsigned short iDim = 0; iDim < nDim; iDim++) {
626 for (unsigned short jDim = 0; jDim < nDim; jDim++) {
627 su2double StrainRate_ij = MeanPerturbedRSM[iDim][jDim] - TWO3 * turb_ke * delta[iDim][jDim];
628 StrainRate_ij = -StrainRate_ij * Density_i / (2 * Eddy_Viscosity_i);
629
630 perturbedStrainMag += pow(StrainRate_ij, 2.0);
631 }
632 }
633 return sqrt(2.0 * perturbedStrainMag);
634 }
635
636 /*!
637 * \brief Add contribution from convection and diffusion due to axisymmetric formulation to 2D residual
638 */
639 inline void ResidualAxisymmetricConvectionDiffusion(su2double alfa_blended, su2double zeta) {
640 if (Coord_i[1] < EPS) return;
641
642 const su2double yinv = 1.0 / Coord_i[1];
643 const su2double rhov = Density_i * V_i[idx.Velocity() + 1];
644 const su2double& k = ScalarVar_i[0];
645 const su2double& w = ScalarVar_i[1];
646
647 /*--- Compute blended constants ---*/
648 const su2double sigma_k_i = F1_i * sigma_k_1 + (1.0 - F1_i) * sigma_k_2;
649 const su2double sigma_w_i = F1_i * sigma_w_1 + (1.0 - F1_i) * sigma_w_2;
650
651 /*--- Convection-Diffusion ---*/
652 const su2double cdk_axi = rhov * k - (Laminar_Viscosity_i + sigma_k_i * Eddy_Viscosity_i) * ScalarVar_Grad_i[0][1];
653 const su2double cdw_axi = rhov * w - (Laminar_Viscosity_i + sigma_w_i * Eddy_Viscosity_i) * ScalarVar_Grad_i[1][1];
654
655 /*--- Add terms to the residuals ---*/
656
657 Residual[0] -= yinv * Volume * cdk_axi;
658 Residual[1] -= yinv * Volume * cdw_axi;
659
660 Jacobian_i[0][0] -= yinv * Volume * rhov;
661 Jacobian_i[0][1] -= 0.0;
662 Jacobian_i[1][0] -= 0.0;
663 Jacobian_i[1][1] -= yinv * Volume * rhov;
664
665 }

724 /*!
725 * \brief Residual for source term integration.
726 * \param[in] config - Definition of the particular problem.
727 * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
728 */
729 ResidualType<> ComputeResidual(const CConfig* config) override {
730 AD::StartPreacc();
731 AD::SetPreaccIn(StrainMag_i);
732 AD::SetPreaccIn(ScalarVar_i, nVar);
733 AD::SetPreaccIn(ScalarVar_Grad_i, nVar, nDim);
734 AD::SetPreaccIn(Volume);
735 AD::SetPreaccIn(dist_i);
736 AD::SetPreaccIn(F1_i);
737 AD::SetPreaccIn(F2_i);
738 AD::SetPreaccIn(CDkw_i);
739 AD::SetPreaccIn(PrimVar_Grad_i, nDim + idx.Velocity(), nDim);
740 AD::SetPreaccIn(Vorticity_i, 3);
741 AD::SetPreaccIn(V_i[idx.Density()], V_i[idx.LaminarViscosity()], V_i[idx.EddyViscosity()]);
742 AD::SetPreaccIn(V_i[idx.Velocity() + 1]);
743
744 Density_i = V_i[idx.Density()];
745 Laminar_Viscosity_i = V_i[idx.LaminarViscosity()];
746 Eddy_Viscosity_i = V_i[idx.EddyViscosity()];
747
748 Residual[0] = 0.0;
749 Residual[1] = 0.0;
750 Jacobian_i[0][0] = 0.0;
751 Jacobian_i[0][1] = 0.0;
752 Jacobian_i[1][0] = 0.0;
753 Jacobian_i[1][1] = 0.0;
754
755 /*--- Computation of blended constants for the source terms ---*/
756
757 const su2double alfa_blended = F1_i * alfa_1 + (1.0 - F1_i) * alfa_2;
758 const su2double beta_blended = F1_i * beta_1 + (1.0 - F1_i) * beta_2;
759
760 su2double eff_intermittency = 1.0;
761
762 if (config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
763 AD::SetPreaccIn(intermittency_eff_i);
764 eff_intermittency = intermittency_eff_i;
765 }
766
767 if (dist_i > 1e-10) {
768
769 su2double diverg = 0.0;
770 for (unsigned short iDim = 0; iDim < nDim; iDim++)
771 diverg += PrimVar_Grad_i[iDim + idx.Velocity()][iDim];
772 if (axisymmetric && Coord_i[1] > EPS) {
773 AD::SetPreaccIn(Coord_i[1]);
774 diverg += V_i[idx.Velocity() + 1] / Coord_i[1];
775 }
776
777 /*--- If using UQ methodolgy, calculate production using perturbed Reynolds stress matrix ---*/
778
779 const su2double VorticityMag = GeometryToolbox::Norm(3, Vorticity_i);
780 su2double P_Base = 0;
781
782 /*--- Apply production term modifications ---*/
783 switch (sstParsedOptions.production) {
784 case SST_OPTIONS::UQ:
785 ComputePerturbedRSM(nDim, Eig_Val_Comp, uq_permute, uq_delta_b, uq_urlx, PrimVar_Grad_i + idx.Velocity(),
786 Density_i, Eddy_Viscosity_i, ScalarVar_i[0], MeanPerturbedRSM);
787 P_Base = PerturbedStrainMag(ScalarVar_i[0]);
788 break;
789
790 case SST_OPTIONS::V:
791 P_Base = VorticityMag;
792 break;
793
794 case SST_OPTIONS::KL:
795 P_Base = sqrt(StrainMag_i*VorticityMag);
796 break;
797
798 default:
799 /*--- Base production term for SST-1994 and SST-2003 ---*/
800 P_Base = StrainMag_i;
801 break;
802 }
803
804 /*--- Production limiter. ---*/
805 const su2double prod_limit = prod_lim_const * beta_star * Density_i * ScalarVar_i[1] * ScalarVar_i[0];
806
807 su2double P = Eddy_Viscosity_i * pow(P_Base, 2);
808 su2double pk = max(0.0, min(P, prod_limit));
809
810 const auto& eddy_visc_var = sstParsedOptions.version == SST_OPTIONS::V1994 ? VorticityMag : StrainMag_i;
811 const su2double zeta = max(ScalarVar_i[1], eddy_visc_var * F2_i / a1);
812
813 /*--- Production limiter only for V2003, recompute for V1994. ---*/
814 su2double pw;
815 if (sstParsedOptions.version == SST_OPTIONS::V1994) {
816 pw = alfa_blended * Density_i * pow(P_Base, 2);
817 } else {
818 pw = (alfa_blended * Density_i / Eddy_Viscosity_i) * pk;
819 }
820
821 /*--- Sustaining terms, if desired. Note that if the production terms are
822 larger equal than the sustaining terms, the original formulation is
823 obtained again. This is in contrast to the version in literature
824 where the sustaining terms are simply added. This latter approach could
825 lead to problems for very big values of the free-stream turbulence
826 intensity. ---*/
827 if (sstParsedOptions.sust) {
828 const su2double sust_k = beta_star * Density_i * kAmb * omegaAmb;
829 const su2double sust_w = beta_blended * Density_i * omegaAmb * omegaAmb;
830 pk = max(pk, sust_k);
831 pw = max(pw, sust_w);
832 }
833
834 /*--- Dissipation ---*/
835
836 su2double dk = beta_star * Density_i * ScalarVar_i[1] * ScalarVar_i[0];
837 su2double dw = beta_blended * Density_i * ScalarVar_i[1] * ScalarVar_i[1];
838
839 /*--- LM model coupling with production and dissipation term for k transport equation---*/
840 if (config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
841 pk = pk * eff_intermittency;
842 dk = min(max(eff_intermittency, 0.1), 1.0) * dk;
843 }
844
845 /*--- Add the production terms to the residuals. ---*/
846
847 Residual[0] += pk * Volume;
848 Residual[1] += pw * Volume;
849
850 /*--- Add the dissipation terms to the residuals.---*/
851
852 Residual[0] -= dk * Volume;
853 Residual[1] -= dw * Volume;
854
855 /*--- Cross diffusion ---*/
856
857 Residual[1] += (1.0 - F1_i) * CDkw_i * Volume;
858
859 /*--- Contribution due to 2D axisymmetric formulation ---*/
860
861 if (axisymmetric) ResidualAxisymmetricConvectionDiffusion(alfa_blended, zeta);
862
863 /*--- Implicit part ---*/
864
865 Jacobian_i[0][0] = -beta_star * ScalarVar_i[1] * Volume;
866 Jacobian_i[0][1] = -beta_star * ScalarVar_i[0] * Volume;
867 Jacobian_i[1][0] = 0.0;
868 Jacobian_i[1][1] = -2.0 * beta_blended * ScalarVar_i[1] * Volume;
869 }
870
871 AD::SetPreaccOut(Residual, nVar);
872 AD::EndPreacc();
873
874 return ResidualType<>(Residual, Jacobian_i, nullptr);
875 }
