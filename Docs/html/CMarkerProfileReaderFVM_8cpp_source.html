32CMarkerProfileReaderFVM::CMarkerProfileReaderFVM(CGeometry *val_geometry,
33 CConfig *val_config,
34 string val_filename,
35 unsigned short val_kind_marker,
36 unsigned short val_number_vars,
37 vector<string> val_columnNames,
38 vector<string> val_columnValues) {
39
40 /*--- Store input values and pointers to class data. ---*/
41
42 rank = SU2_MPI::GetRank();
43 size = SU2_MPI::GetSize();
44
45 this->config = val_config;
46 this->geometry = val_geometry;
47 dimension = geometry->GetnDim();
48
49 filename = std::move(val_filename);
50 markerType = val_kind_marker;
51 numberOfVars = val_number_vars;
52 columnNames = std::move(val_columnNames);
53 columnValues = std::move(val_columnValues);
54
55 /* Attempt to open the specified file. */
56 ifstream profile_file;
57 profile_file.open(filename.data(), ios::in);
58
59 /* If the file is not found, then we merge the information necessary
60 and write a template marker profile file. Otherwise, we read and
61 store the information in the marker profile file. */
62
63 if (profile_file.fail()) {
64 MergeProfileMarkers();
65 WriteMarkerProfileTemplate();
66 SU2_MPI::Barrier(SU2_MPI::GetComm());
67 } else {
68 ReadMarkerProfile();
69 }

75void CMarkerProfileReaderFVM::ReadMarkerProfile() {
76
77 /*--- Open the profile file (we have already error checked) ---*/
78
79 ifstream profile_file;
80 profile_file.open(filename.data(), ios::in);
81 bool nmarkFound = false;
82 unsigned long skip = 0;
83
84 /*--- Identify the markers and data set in the profile file ---*/
85
86 string text_line;
87 /*--- We search the file until we find the keyword NMARK=. Data before NMARK= will be ignored.
88 This allows for some information in a header that will be ignored by the profile reader. ---*/
89 while (getline (profile_file, text_line)) {
90 /*--- read NMARK ---*/
91 string::size_type position = text_line.find ("NMARK=",0);
92 if (position != string::npos) {
93 nmarkFound = true;
94 text_line.erase (0,6); numberOfProfiles = atoi(text_line.c_str());
95
96 numberOfRowsInProfile.resize(numberOfProfiles);
97 numberOfColumnsInProfile.resize(numberOfProfiles);
98
99 for (unsigned short iMarker = 0 ; iMarker < numberOfProfiles; iMarker++) {
100
101 /*--- read MARKER_TAG ---*/
102 getline (profile_file, text_line);
103 text_line.erase (0,11);
104 for (unsigned short iChar = 0; iChar < 20; iChar++) {
105 position = text_line.find( ' ', 0 ); if (position != string::npos) text_line.erase (position,1);
106 position = text_line.find( '\r', 0 ); if (position != string::npos) text_line.erase (position,1);
107 position = text_line.find( '\n', 0 ); if (position != string::npos) text_line.erase (position,1);
108 }
109 profileTags.emplace_back(text_line.c_str());
110
111 /*--- read NROW ---*/
112 getline (profile_file, text_line);
113 text_line.erase (0,5); numberOfRowsInProfile[iMarker] = atoi(text_line.c_str());
114
115 /*--- read NCOL ---*/
116 getline (profile_file, text_line);
117 text_line.erase (0,5); numberOfColumnsInProfile[iMarker] = atoi(text_line.c_str());
118
119 /*--- read the column format description. This line is not required, so if we cannot find it, we just continue ---*/
120 getline (profile_file, text_line);
121 string::size_type dataheader = text_line.find ("# COORD",0);
122 if (dataheader == 0) {
123 skip = 0;
124 } else {
125 /*--- no header, but we have read a line, so we have to read one line less data ---*/
126 skip = 1;
127 }
128
129 /*--- Skip the data. This is read in the next loop. ---*/
130
131 for (unsigned long iRow = 0; iRow < (numberOfRowsInProfile[iMarker]-skip); iRow++) getline (profile_file, text_line);
132
133 }
134 }
135 }
136
137 profile_file.close();
138
139 if (!nmarkFound) {
140 SU2_MPI::Error("While opening profile file, no \"NMARK=\" specification was found", CURRENT_FUNCTION);
141 }
142
143 /*--- Compute array bounds and offsets. Allocate data structure. ---*/
144
145 profileData.resize(numberOfProfiles);
146 for (unsigned short iMarker = 0; iMarker < numberOfProfiles; iMarker++) {
147 profileData[iMarker].resize(numberOfRowsInProfile[iMarker]*numberOfColumnsInProfile[iMarker], 0.0);
148 }
149
150 /*--- Read all lines in the profile file and extract data. ---*/
151
152 profile_file.open(filename.data(), ios::in);
153
154 int counter = 0;
155 while (getline (profile_file, text_line)) {
156
157 string::size_type position = text_line.find ("NMARK=",0);
158 if (position != string::npos) {
159
160 for (unsigned short iMarker = 0; iMarker < numberOfProfiles; iMarker++) {
161
162 /*--- Skip the tag, nRow and nCol lines. ---*/
163
164 getline (profile_file, text_line);
165 getline (profile_file, text_line);
166 getline (profile_file, text_line);
167
168 /*--- if skip=0 then we can expect column format description ---*/
169 if (skip == 0) getline (profile_file, text_line);
170
171 /*--- Now read the data for each row and store. ---*/
172
173 for (unsigned long iRow = 0; iRow < numberOfRowsInProfile[iMarker]; iRow++) {
174
175 getline (profile_file, text_line);
176
177 istringstream point_line(text_line);
178
179 /*--- Store the values (starting with node coordinates) --*/
180
181 for (unsigned short iVar = 0; iVar < numberOfColumnsInProfile[iMarker]; iVar++)
182 point_line >> profileData[iMarker][iRow*numberOfColumnsInProfile[iMarker] + iVar];
183
184 /*--- Increment our local row counter. ---*/
185
186 counter++;
187
188 }
189 }
190 }
191 }
192
193 profile_file.close();
194
195}

197void CMarkerProfileReaderFVM::MergeProfileMarkers() {
198
199 /*--- Local variables needed on all processors ---*/
200
201 unsigned long iPoint, jPoint, kPoint;
202
203 int iProcessor, nProcessor = size;
204
205 unsigned long iVertex, iMarker;
206 unsigned long Buffer_Send_nPoin[1], *Buffer_Recv_nPoin = nullptr;
207 unsigned long nLocalPoint = 0, MaxLocalPoint = 0;
208 unsigned long maxProfiles = 0;
209 unsigned long index, iChar;
210
211 char str_buf[MAX_STRING_SIZE];
212 char name_buf[MAX_STRING_SIZE];
213 char value_buf[MAX_STRING_SIZE];
214
215 vector<string> Marker_Tags;
216
217 vector<unsigned long> nRowCum_Counter;
218
219 if (rank == MASTER_NODE) Buffer_Recv_nPoin = new unsigned long[nProcessor];
220
221 /*--- Search all boundaries on the present rank to count the number
222 of nodes found on profile markers. ---*/
223
224 nLocalPoint = 0; numberOfProfiles = 0;
225 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
226 if (config->GetMarker_All_KindBC(iMarker) == markerType) {
227
228 numberOfProfiles++;
229 for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
230 iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
231
232 /*--- Only communicate owned nodes to avoid duplicates. ---*/
233
234 if (geometry->nodes->GetDomain(iPoint))
235 nLocalPoint++;
236
237 }
238 }
239 }
240 Buffer_Send_nPoin[0] = nLocalPoint;
241
242 /*--- Communicate the total number of nodes on this domain. ---*/
243
244 SU2_MPI::Gather(&Buffer_Send_nPoin, 1, MPI_UNSIGNED_LONG,
245 Buffer_Recv_nPoin, 1, MPI_UNSIGNED_LONG, MASTER_NODE, SU2_MPI::GetComm());
246 SU2_MPI::Allreduce(&nLocalPoint, &MaxLocalPoint, 1, MPI_UNSIGNED_LONG, MPI_MAX, SU2_MPI::GetComm());
247 SU2_MPI::Reduce(&numberOfProfiles, &maxProfiles, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
248 SU2_MPI::GetComm());
249
250 /*--- Send and Recv buffers. ---*/
251
252 auto *Buffer_Send_X = new su2double[MaxLocalPoint];
253 su2double *Buffer_Recv_X = nullptr;
254
255 auto *Buffer_Send_Y = new su2double[MaxLocalPoint];
256 su2double *Buffer_Recv_Y = nullptr;
257
258 su2double *Buffer_Send_Z = nullptr, *Buffer_Recv_Z = nullptr;
259 if (dimension == 3) Buffer_Send_Z = new su2double[MaxLocalPoint];
260
261 char *Buffer_Send_Str = new char[MaxLocalPoint*MAX_STRING_SIZE];
262 char *Buffer_Recv_Str = nullptr;
263
264 char *Buffer_Send_Name = new char[MaxLocalPoint*MAX_STRING_SIZE];
265 char *Buffer_Recv_Name = nullptr;
266
267 char *Buffer_Send_Value = new char[MaxLocalPoint*MAX_STRING_SIZE];
268 char *Buffer_Recv_Value = nullptr;
269
270 /*--- Prepare the receive buffers in the master node only. ---*/
271
272 if (rank == MASTER_NODE) {
273
274 Buffer_Recv_X = new su2double[nProcessor*MaxLocalPoint];
275 Buffer_Recv_Y = new su2double[nProcessor*MaxLocalPoint];
276 if (dimension == 3) Buffer_Recv_Z = new su2double[nProcessor*MaxLocalPoint];
277
278 Buffer_Recv_Str = new char[nProcessor*MaxLocalPoint*MAX_STRING_SIZE];
279 Buffer_Recv_Name = new char[nProcessor*MaxLocalPoint*MAX_STRING_SIZE];
280 Buffer_Recv_Value = new char[nProcessor*MaxLocalPoint*MAX_STRING_SIZE];
281
282 /*--- Sum total number of nodes to be written and allocate arrays ---*/
283
284 unsigned long nGlobal_InletPoint = 0;
285 for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
286 nGlobal_InletPoint += Buffer_Recv_nPoin[iProcessor];
287 }
288
289 profileCoords.resize(maxProfiles);
290 for (iMarker = 0; iMarker < maxProfiles; iMarker++) {
291 profileCoords[iMarker].resize(dimension);
292 }
293
294 totalColumnNames.resize(maxProfiles);
295 totalColumnValues.resize(maxProfiles);
296
297 }
298
299 /*--- Main communication routine. Loop over each coordinate and perform
300 the MPI comm. Temporary 1-D buffers are used to send the coordinates at
301 all nodes on each partition to the master node. These are then unpacked
302 by the master and sorted by marker tag in one large n-dim. array. ---*/
303
304 su2double *Coords_Local; jPoint = 0;
305 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
306 if (config->GetMarker_All_KindBC(iMarker) == markerType) {
307
308 for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
309 iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
310
311 /*--- Only communicate owned nodes to avoid duplicates. ---*/
312
313 if (geometry->nodes->GetDomain(iPoint)) {
314
315 /*--- Retrieve local coordinates at this node. ---*/
316
317 Coords_Local = geometry->nodes->GetCoord(iPoint);
318
319 /*--- Load local coords into the temporary send buffer. ---*/
320
321 Buffer_Send_X[jPoint] = Coords_Local[0];
322 Buffer_Send_Y[jPoint] = Coords_Local[1];
323 if (dimension == 3) Buffer_Send_Z[jPoint] = Coords_Local[2];
324
325 /*--- If US system, the output should be in inches ---*/
326
327 if (config->GetSystemMeasurements() == US) {
328 Buffer_Send_X[jPoint] *= 12.0;
329 Buffer_Send_Y[jPoint] *= 12.0;
330 if (dimension == 3) Buffer_Send_Z[jPoint] *= 12.0;
331 }
332
333 /*--- Store the marker tag for this particular node. ---*/
334
335 SPRINTF(&Buffer_Send_Str[jPoint*MAX_STRING_SIZE], "%s",
336 config->GetMarker_All_TagBound(iMarker).c_str());
337
338 /*--- Store the column names ---*/
339
340 SPRINTF(&Buffer_Send_Name[jPoint*MAX_STRING_SIZE], "%s",
341 columnNames[iMarker].c_str());
342
343 /*--- Store the column values ---*/
344
345 SPRINTF(&Buffer_Send_Value[jPoint*MAX_STRING_SIZE], "%s",
346 columnValues[iMarker].c_str());
347
348 /*--- Increment jPoint as the counter. We need this because iPoint
349 may include halo nodes that we skip over during this loop. ---*/
350
351 jPoint++;
352
353 }
354 }
355 }
356 }
357
358 /*--- Gather the coordinate data on the master node using MPI. ---*/
359
360 SU2_MPI::Gather(Buffer_Send_X, (int)MaxLocalPoint, MPI_DOUBLE,
361 Buffer_Recv_X, (int)MaxLocalPoint, MPI_DOUBLE, MASTER_NODE, SU2_MPI::GetComm());
362 SU2_MPI::Gather(Buffer_Send_Y, (int)MaxLocalPoint, MPI_DOUBLE,
363 Buffer_Recv_Y, (int)MaxLocalPoint, MPI_DOUBLE, MASTER_NODE, SU2_MPI::GetComm());
364 if (dimension == 3) {
365 SU2_MPI::Gather(Buffer_Send_Z, (int)MaxLocalPoint, MPI_DOUBLE,
366 Buffer_Recv_Z, (int)MaxLocalPoint, MPI_DOUBLE, MASTER_NODE, SU2_MPI::GetComm());
367 }
368 SU2_MPI::Gather(Buffer_Send_Str, (int)MaxLocalPoint*MAX_STRING_SIZE, MPI_CHAR,
369 Buffer_Recv_Str, (int)MaxLocalPoint*MAX_STRING_SIZE, MPI_CHAR, MASTER_NODE, SU2_MPI::GetComm());
370
371 SU2_MPI::Gather(Buffer_Send_Name, (int)MaxLocalPoint*MAX_STRING_SIZE, MPI_CHAR,
372 Buffer_Recv_Name, (int)MaxLocalPoint*MAX_STRING_SIZE, MPI_CHAR, MASTER_NODE, SU2_MPI::GetComm());
373 SU2_MPI::Gather(Buffer_Send_Value, (int)MaxLocalPoint*MAX_STRING_SIZE, MPI_CHAR,
374 Buffer_Recv_Value, (int)MaxLocalPoint*MAX_STRING_SIZE, MPI_CHAR, MASTER_NODE, SU2_MPI::GetComm());
375
376 /*--- The master node unpacks and sorts this variable by marker tag. ---*/
377
378 if (rank == MASTER_NODE) {
379
380 profileTags.clear();
381 totalColumnNames.clear();
382 totalColumnValues.clear();
383
384 /*--- First, parse the marker tags to count how many total profile markers
385 we have now on the master. ---*/
386
387 for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
388 for (iPoint = 0; iPoint < Buffer_Recv_nPoin[iProcessor]; iPoint++) {
389 index = (iProcessor*MaxLocalPoint + iPoint)*MAX_STRING_SIZE;
390 for (iChar = 0; iChar < MAX_STRING_SIZE; iChar++) {
391 str_buf[iChar] = Buffer_Recv_Str[index + iChar];
392 name_buf[iChar] = Buffer_Recv_Name[index + iChar];
393 value_buf[iChar] = Buffer_Recv_Value[index + iChar];
394 }
395
396 Marker_Tags.push_back(str_buf);
397
398 /*--- only add if not already in the list ---*/
399 if (std::find(profileTags.begin(), profileTags.end(), str_buf) == profileTags.end()) {
400 profileTags.push_back(str_buf);
401 totalColumnNames.push_back(name_buf);
402 totalColumnValues.push_back(value_buf);
403 }
404
405 }
406
407 }
408
409 numberOfProfiles = profileTags.size();
410
411 /*--- Count the number of rows (nodes) per marker. ---*/
412
413 numberOfRowsInProfile.resize(maxProfiles,0.0);
414 jPoint = 0;
415 for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
416 for (iPoint = 0; iPoint < Buffer_Recv_nPoin[iProcessor]; iPoint++) {
417 for (iMarker = 0; iMarker < maxProfiles; iMarker++) {
418 if (profileTags[iMarker] == Marker_Tags[jPoint]) {
419
420 numberOfRowsInProfile[iMarker]++;
421 }
422 }
423 jPoint++;
424 }
425 }
426
427 /*--- Load up the coordinates, sorted into chunks per marker. ---*/
428
429 jPoint = 0; kPoint = 0;
430 for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
431 for (iPoint = 0; iPoint < Buffer_Recv_nPoin[iProcessor]; iPoint++) {
432 for (iMarker = 0; iMarker < maxProfiles; iMarker++) {
433
434 if (profileTags[iMarker] == Marker_Tags[kPoint]) {
435
436 /*--- Find our current index for this marker and store coords. ---*/
437
438 profileCoords[iMarker][0].push_back(Buffer_Recv_X[jPoint]);
439 profileCoords[iMarker][1].push_back(Buffer_Recv_Y[jPoint]);
440 if (dimension == 3)
441 profileCoords[iMarker][2].push_back(Buffer_Recv_Z[jPoint]);
442
443 }
444 }
445
446 /*--- Increment point counter for marker tags and data. ---*/
447
448 kPoint++;
449 jPoint++;
450
451 }
452
453 /*--- Adjust jPoint to index of next proc's data in the buffers. ---*/
454
455 jPoint = (iProcessor+1)*MaxLocalPoint;
456
457 }
458
459 }
460
461 /*--- Immediately release the temporary data buffers. ---*/
462 delete [] Buffer_Send_X;
463 delete [] Buffer_Send_Y;
464 delete [] Buffer_Send_Z;
465 delete [] Buffer_Send_Str;
466 delete [] Buffer_Send_Name;
467 delete [] Buffer_Send_Value;
468
469 if (rank == MASTER_NODE) {
470 delete [] Buffer_Recv_X;
471 delete [] Buffer_Recv_Y;
472 delete [] Buffer_Recv_Z;
473 delete [] Buffer_Recv_nPoin;
474 delete [] Buffer_Recv_Str;
475 delete [] Buffer_Recv_Name;
476 delete [] Buffer_Recv_Value;
477 }
478}

480void CMarkerProfileReaderFVM::WriteMarkerProfileTemplate() {
481
482 /*--- Count the number of columns that we have for this case.
483 Here, we have dimension entries for node coordinates and then the
484 total number of columns of data specified in the constructor. ---*/
485
486 const unsigned short nColumns = dimension + numberOfVars;
487
488 /*--- Write the profile file. Note that we have already merged
489 all of the information for the markers and coordinates previously
490 in the MergeProfileMarkers() routine and only the master writes. ---*/
491
492 if (rank == MASTER_NODE) {
493
494 ofstream node_file("example_"+filename);
495 node_file << "NMARK= " << numberOfProfiles << endl;
496
497 for (unsigned long iMarker = 0; iMarker < numberOfProfiles; iMarker++) {
498
499 /*--- Access the default data for this marker. ---*/
500 string Marker_Tag = profileTags[iMarker];
501
502 /*--- Header information for this marker. ---*/
503
504 node_file << "MARKER_TAG= " << Marker_Tag << endl;
505 node_file << "NROW=" << numberOfRowsInProfile[iMarker] << endl;
506 node_file << "NCOL=" << nColumns << endl;
507
508
509 /*--- header line (names of the columns) --- */
510 node_file << totalColumnNames[iMarker] << endl;
511
512 node_file << setprecision(15);
513 node_file << std::scientific;
514
515 /*--- Loop over the data structure and write the coords and vars. ---*/
516
517 for (unsigned long iPoint = 0; iPoint < numberOfRowsInProfile[iMarker]; iPoint++) {
518 for (unsigned short iDim = 0; iDim < dimension; iDim++) {
519 node_file << profileCoords[iMarker][iDim][iPoint] << "\t";
520 }
521
522 node_file << totalColumnValues[iMarker] << endl;
523
524 }
525
526
527 } // iMarker
528 node_file.close();
529
530 /*--- Print a message to inform the user about the template file. ---*/
531
532 stringstream err;
533 err << endl;
534 err << " Could not find the input file for the marker profile." << endl;
535 err << " Looked for: " << filename << "." << endl;
536 err << " Created a template profile file with default values" << endl;
537 err << " named example_" << filename << endl;
538 err << " You can use this file as a guide for making your own profile" << endl;
539 err << " specification." << endl << endl;
540 SU2_MPI::Error(err.str(), CURRENT_FUNCTION);
541
542 }
