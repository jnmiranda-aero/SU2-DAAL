67 void CDriverBase::InitializeContainers() {
68  /*--- Create pointers to all the classes that may be used by drivers. In general, the pointers are instantiated
69  * down a hierarchy over all zones, multi-grid levels, equation sets, and equation terms as described in the comments
70  * below. ---*/
71 
72  config_container = new CConfig*[nZone]();
73  output_container = new COutput*[nZone]();
74  geometry_container = new CGeometry***[nZone]();
75  solver_container = new CSolver****[nZone]();
76  numerics_container = new CNumerics*****[nZone]();
77  surface_movement = new CSurfaceMovement*[nZone]();
78  grid_movement = new CVolumetricMovement**[nZone]();
79 
80  nInst = new unsigned short[nZone];
81 
82  for (iZone = 0; iZone < nZone; iZone++) {
83  nInst[iZone] = 1;
84  }

87 void CDriverBase::CommonFinalize() {
88 
89  if (numerics_container != nullptr) {
90  for (iZone = 0; iZone < nZone; iZone++) {
91  delete[] numerics_container[iZone];
92  }
93  delete[] numerics_container;
94  }
95  if (rank == MASTER_NODE) cout << "Deleted CNumerics container." << endl;
96 
97  if (solver_container != nullptr) {
98  for (iZone = 0; iZone < nZone; iZone++) {
99  delete[] solver_container[iZone];
100  }
101  delete[] solver_container;
102  }
103  if (rank == MASTER_NODE) cout << "Deleted CSolver container." << endl;
104 
105  if (geometry_container != nullptr) {
106  for (iZone = 0; iZone < nZone; iZone++) {
107  if (geometry_container[iZone] != nullptr) {
108  for (iInst = 0; iInst < nInst[iZone]; iInst++) {
109  delete geometry_container[iZone][iInst][MESH_0];
110  delete[] geometry_container[iZone][iInst];
111  }
112  delete[] geometry_container[iZone];
113  }
114  }
115  delete[] geometry_container;
116  }
117  if (rank == MASTER_NODE) cout << "Deleted CGeometry container." << endl;
118 
119  if (surface_movement != nullptr) {
120  for (iZone = 0; iZone < nZone; iZone++) {
121  delete surface_movement[iZone];
122  }
123  delete[] surface_movement;
124  }
125  if (rank == MASTER_NODE) cout << "Deleted CSurfaceMovement class." << endl;
126 
127  if (grid_movement != nullptr) {
128  for (iZone = 0; iZone < nZone; iZone++) {
129  if (grid_movement[iZone] != nullptr) {
130  for (iInst = 0; iInst < nInst[iZone]; iInst++) {
131  delete grid_movement[iZone][iInst];
132  }
133  delete[] grid_movement[iZone];
134  }
135  }
136  delete[] grid_movement;
137  }
138  if (rank == MASTER_NODE) cout << "Deleted CVolumetricMovement class." << endl;
139 
140  if (config_container != nullptr) {
141  for (iZone = 0; iZone < nZone; iZone++) {
142  delete config_container[iZone];
143  }
144  delete[] config_container;
145  }
146  delete driver_config;
147  if (rank == MASTER_NODE) cout << "Deleted CConfig container." << endl;
148 
149  if (output_container != nullptr) {
150  for (iZone = 0; iZone < nZone; iZone++) {
151  delete output_container[iZone];
152  }
153  delete[] output_container;
154  }
155  if (rank == MASTER_NODE) cout << "Deleted COutput class." << endl;
156 
157  delete[] nInst;
158 }

306 vector<string> CDriverBase::GetCHTMarkerTags() const {
307  vector<string> tags;
308  const auto nMarker = main_config->GetnMarker_All();
309 
310  // The CHT markers can be identified as the markers that are customizable with a BC type HEAT_FLUX or ISOTHERMAL.
311  for (auto iMarker = 0u; iMarker < nMarker; iMarker++) {
312  if ((main_config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX ||
313  main_config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL) &&
314  main_config->GetMarker_All_PyCustom(iMarker)) {
315  tags.push_back(main_config->GetMarker_All_TagBound(iMarker));
316  }
317  }
318  return tags;

377 vector<passivedouble> CDriverBase::GetMarkerVertexNormals(unsigned short iMarker, unsigned long iVertex,
378  bool normalize) const {
379  if (iVertex >= GetNumberMarkerNodes(iMarker)) {
380  SU2_MPI::Error("Vertex index exceeds marker size.", CURRENT_FUNCTION);
381  }
382  const auto* normal = main_geometry->vertex[iMarker][iVertex]->GetNormal();
383  const su2double area = normalize ? GeometryToolbox::Norm(nDim, normal) : 1.0;
384 
385  vector<passivedouble> values(nDim, 0.0);
386 
387  for (auto iDim = 0u; iDim < nDim; iDim++) {
388  values[iDim] = SU2_TYPE::GetValue(normal[iDim] / area);
389  }
390  return values;

410 std::map<string, unsigned short> CDriverBase::GetFEASolutionIndices() const {
411  if (solver_container[selected_zone][INST_0][MESH_0][FEA_SOL] == nullptr) {
412  SU2_MPI::Error("The FEA solver does not exist.", CURRENT_FUNCTION);
413  }
414  const auto nDim = main_geometry->GetnDim();
415  std::map<string, unsigned short> names;
416  names["DISPLACEMENT_X"] = 0;
417  names["DISPLACEMENT_Y"] = 1;
418  if (nDim == 3) names["DISPLACEMENT_Z"] = 2;
419 
420  if (main_config->GetTime_Domain()) {
421  names["VELOCITY_X"] = nDim;
422  names["VELOCITY_Y"] = nDim + 1;
423  if (nDim == 3) names["VELOCITY_Z"] = nDim + 2;
424  names["ACCELERATION_X"] = 2 * nDim;
425  names["ACCELERATION_Y"] = 2 * nDim + 1;
426  if (nDim == 3) names["ACCELERATION_Z"] = 2 * nDim + 2;
427  }
