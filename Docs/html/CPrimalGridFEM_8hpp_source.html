69 unsigned long val_offDOfsSol, std::istringstream& elem_line);

71 /*!

82 CPrimalGridFEM(unsigned long val_elemGlobalID, unsigned short val_VTK_Type, unsigned short val_nPolyGrid,
83 unsigned short val_nPolySol, unsigned short val_nDOFsGrid, unsigned short val_nDOFsSol,
84 unsigned long val_offDOfsSol, const unsigned long* connGrid);

86 /*!
