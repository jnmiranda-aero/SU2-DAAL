33/*!
34 * \brief Class to store a dense general square matrix that uses the Column
35 * Major order storage format. The code should be compiled with
36 * LAPACK to use optimized matrix inversion and multiplication routines.
37 * \ingroup BLAS
38 */
39class CSquareMatrixCM {
40 static_assert(ColMajorMatrix<passivedouble>::Storage == StorageType::ColumnMajor,
41 "Column major storage is assumed for LAPACK.");
42
43 private:
44 ColMajorMatrix<passivedouble> mat; /*!< \brief Storage of the actual matrix. */
45
46 public:
47 /*!
48 * \brief Default constructor. Nothing to be done.
49 */
50 CSquareMatrixCM() = default;
51
52 /*!
53 * \overload
54 * \brief Overloaded constructor, which allocates the memory to store
55 * the matrix.
56 * \param[in] N - Number of rows and colums of the matrix.
57 */
58 CSquareMatrixCM(int N) { Initialize(N); }
59
60 /*!
61 * \brief Operator, which makes available the given matrix element as a reference.
62 * \param[in] i - Row index of the matrix element.
63 * \param[in] j - Column index of the matrix element.
64 * \return Reference to element (i,j).
65 */
66 inline passivedouble& operator()(int i, int j) { return mat(i, j); }
67
68 /*!
69 * \brief Operator, which makes available the given matrix element as a const reference.
70 * \param[in] i - Row index of the matrix element.
71 * \param[in] j - Column index of the matrix element.
72 * \return Constant reference to element (i,j).
73 */
74 inline const passivedouble& operator()(int i, int j) const { return mat(i, j); }
75
76 /*!
77 * \brief Function, which makes available a reference to the actual matrix.
78 * \return A reference to mat.
79 */
80 inline ColMajorMatrix<passivedouble>& GetMat() { return mat; }
81
82 /*!
83 * \brief Function, which makes available a const reference to the actual matrix.
84 * \return A const reference to mat.
85 */
86 inline const ColMajorMatrix<passivedouble>& GetMat() const { return mat; }
87
88 /*!
89 * \brief Function, which allocates the memory for the matrix.
90 * \param[in] N - Number of rows and colums of the matrix.
91 */
92 inline void Initialize(int N) { mat.resize(N, N); }
93
94 /*!
95 * \brief Function, which makes available the size of the matrix.
96 * \return The number of rows, columns of the matrix.
97 */
98 inline int Size() const { return mat.rows(); }
99
100 /*!
101 * \brief Function, which carries out the matrix produc of the current matrix
102 * with mat_in and stores the result in mat_out.
103 * \param[in] side - left: mat_out = this * mat_in, right: mat_out = mat_in * this
104 * \param[in] mat_in - Matrix to be multiplied by the current matrix.
105 * \param[out] mat_out - Matrix to store the result of the multiplication.
106 */
107 void MatMatMult(const char side, const ColMajorMatrix<passivedouble>& mat_in,
108 ColMajorMatrix<passivedouble>& mat_out) const;
109
110 /*!
111 * \brief Naive matrix-vector multiplication with general type.
112 */
113 template <class ForwardIt>
114 void MatVecMult(ForwardIt vec_in, ForwardIt vec_out) const {
115 for (int i = 0; i < Size(); ++i) {
116 *vec_out = 0.0;
117 auto vec = vec_in;
118 for (int k = 0; k < Size(); ++k) *vec_out += *(vec++) * mat(i, k);
119 ++vec_out;
120 }
121 }
122
123 /*!

125 */
126 void Invert();
127
128 /*!
129 * \brief Function, which transposes the matrix in-place.
130 */
131 void Transpose();
132};
