44void CNearestNeighbor::SetTransferCoeff(const CConfig* const* config) {
45 /*--- Desired number of donor points. ---*/
46 const auto nDonor = max<unsigned long>(config[donorZone]->GetNumNearestNeighbors(), 1);
47
48 /*--- Epsilon used to avoid division by zero. ---*/
49 const su2double eps = numeric_limits<passivedouble>::epsilon();
50
51 const int nProcessor = size;
52 const auto nMarkerInt = config[donorZone]->GetMarker_n_ZoneInterface() / 2;
53 const auto nDim = donor_geometry->GetnDim();
54
55 Buffer_Receive_nVertex_Donor = new unsigned long[nProcessor];
56
57 targetVertices.resize(config[targetZone]->GetnMarker_All());
58
59 vector<vector<DonorInfo> > DonorInfoVec(omp_get_max_threads());
60
61 /*--- Cycle over nMarkersInt interface to determine communication pattern. ---*/
62
63 AvgDistance = MaxDistance = 0.0;
64 unsigned long totalTargetPoints = 0;
65
66 for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
67 /*--- On the donor side: find the tag of the boundary sharing the interface. ---*/
68 const auto markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
69
70 /*--- On the target side: find the tag of the boundary sharing the interface. ---*/
71 const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
72
73 /*--- Checks if the zone contains the interface, if not continue to the next step. ---*/
74 if (!CheckInterfaceBoundary(markDonor, markTarget)) continue;
75
76 unsigned long nVertexDonor = 0, nVertexTarget = 0;
77 if (markDonor != -1) nVertexDonor = donor_geometry->GetnVertex(markDonor);
78 if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex(markTarget);
79
80 /*--- Sets MaxLocalVertex_Donor, Buffer_Receive_nVertex_Donor. ---*/
81 Determine_ArraySize(markDonor, markTarget, nVertexDonor, nDim);
82 if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
83
84 const auto nPossibleDonor =
85 accumulate(Buffer_Receive_nVertex_Donor, Buffer_Receive_nVertex_Donor + nProcessor, 0ul);
86
87 Buffer_Send_Coord.resize(MaxLocalVertex_Donor, nDim);
88 Buffer_Send_GlobalPoint.resize(MaxLocalVertex_Donor);
89 Buffer_Receive_Coord.resize(nProcessor * MaxLocalVertex_Donor, nDim);
90 Buffer_Receive_GlobalPoint.resize(nProcessor * MaxLocalVertex_Donor);
91
92 /*--- Collect coordinates and global point indices. ---*/
93 Collect_VertexInfo(markDonor, markTarget, nVertexDonor, nDim);
94
95 /*--- Find the closest donor points to each target. ---*/
96 SU2_OMP_PARALLEL {
97 /*--- Working array for this thread. ---*/
98 auto& donorInfo = DonorInfoVec[omp_get_thread_num()];
99 donorInfo.resize(nPossibleDonor);
100
101 su2double avgDist = 0.0, maxDist = 0.0;
102 unsigned long numTarget = 0;
103
104 SU2_OMP_FOR_DYN(roundUpDiv(nVertexTarget, 2 * omp_get_max_threads()))
105 for (auto iVertexTarget = 0ul; iVertexTarget < nVertexTarget; iVertexTarget++) {
106 auto& target_vertex = targetVertices[markTarget][iVertexTarget];
107 const auto Point_Target = target_geometry->vertex[markTarget][iVertexTarget]->GetNode();
108
109 if (!target_geometry->nodes->GetDomain(Point_Target)) continue;
110
111 /*--- Coordinates of the target point. ---*/
112 const su2double* Coord_i = target_geometry->nodes->GetCoord(Point_Target);
113
114 /*--- Compute all distances. ---*/
115 for (int iProcessor = 0, iDonor = 0; iProcessor < nProcessor; ++iProcessor) {
116 for (auto jVertex = 0ul; jVertex < Buffer_Receive_nVertex_Donor[iProcessor]; ++jVertex) {
117 const auto idx = iProcessor * MaxLocalVertex_Donor + jVertex;
118 const auto pGlobalPoint = Buffer_Receive_GlobalPoint[idx];
119 const su2double* Coord_j = Buffer_Receive_Coord[idx];
120 const auto dist2 = GeometryToolbox::SquaredDistance(nDim, Coord_i, Coord_j);
121
122 donorInfo[iDonor++] = DonorInfo(dist2, pGlobalPoint, iProcessor);
123 }
124 }
125
126 /*--- Find k closest points. ---*/
127 partial_sort(donorInfo.begin(), donorInfo.begin() + nDonor, donorInfo.end(),
128 [](const DonorInfo& a, const DonorInfo& b) {
129 /*--- Global index is used as tie-breaker to make sorted order independent of initial. ---*/
130 return (a.dist != b.dist) ? (a.dist < b.dist) : (a.pidx < b.pidx);
131 });
132
133 /*--- Update stats. ---*/
134 numTarget += 1;
135 su2double d = sqrt(donorInfo[0].dist);
136 avgDist += d;
137 maxDist = max(maxDist, d);
138
139 /*--- Compute interpolation numerators and denominator. ---*/
140 su2double denom = 0.0;
141 for (auto iDonor = 0ul; iDonor < nDonor; ++iDonor) {
142 donorInfo[iDonor].dist = 1.0 / (donorInfo[iDonor].dist + eps);
143 denom += donorInfo[iDonor].dist;
144 }
145
146 /*--- Set interpolation coefficients. ---*/
147 target_vertex.resize(nDonor);
148
149 for (auto iDonor = 0ul; iDonor < nDonor; ++iDonor) {
150 target_vertex.globalPoint[iDonor] = donorInfo[iDonor].pidx;
151 target_vertex.processor[iDonor] = donorInfo[iDonor].proc;
152 target_vertex.coefficient[iDonor] = donorInfo[iDonor].dist / denom;
153 }
154 }
155 END_SU2_OMP_FOR
156 SU2_OMP_CRITICAL {
157 totalTargetPoints += numTarget;
158 AvgDistance += avgDist;
159 MaxDistance = max(MaxDistance, maxDist);
160 }
161 END_SU2_OMP_CRITICAL
162 }
163 END_SU2_OMP_PARALLEL
164 }
165
166 delete[] Buffer_Receive_nVertex_Donor;
167
168 unsigned long tmp = totalTargetPoints;
169 SU2_MPI::Allreduce(&tmp, &totalTargetPoints, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
170 su2double tmp1 = AvgDistance, tmp2 = MaxDistance;
171 SU2_MPI::Allreduce(&tmp1, &AvgDistance, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
172 SU2_MPI::Allreduce(&tmp2, &MaxDistance, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
173 AvgDistance /= totalTargetPoints;
174}
