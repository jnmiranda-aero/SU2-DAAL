174su2double CHEXA8::ComputeVolume(const FrameType mode) const {
175 unsigned short iDim;
176 su2double r1[3] = {0.0, 0.0, 0.0}, r2[3] = {0.0, 0.0, 0.0}, r3[3] = {0.0, 0.0, 0.0},
177 CrossProduct[3] = {0.0, 0.0, 0.0};
178 su2double Volume = 0.0;
179
180 /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
181 for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed)---*/
182 const su2activematrix& Coord = (mode == REFERENCE) ? RefCoord : CurrentCoord;
183
184 for (iDim = 0; iDim < NDIM; iDim++) {
185 r1[iDim] = Coord[1][iDim] - Coord[0][iDim];
186 r2[iDim] = Coord[2][iDim] - Coord[0][iDim];
187 r3[iDim] = Coord[5][iDim] - Coord[0][iDim];
188 }
189
190 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
191 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
192 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
193
194 Volume = fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
195
196 for (iDim = 0; iDim < NDIM; iDim++) {
197 r1[iDim] = Coord[2][iDim] - Coord[0][iDim];
198 r2[iDim] = Coord[7][iDim] - Coord[0][iDim];
199 r3[iDim] = Coord[5][iDim] - Coord[0][iDim];
200 }
201
202 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
203 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
204 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
205
206 Volume += fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
207
208 for (iDim = 0; iDim < NDIM; iDim++) {
209 r1[iDim] = Coord[2][iDim] - Coord[0][iDim];
210 r2[iDim] = Coord[3][iDim] - Coord[0][iDim];
211 r3[iDim] = Coord[7][iDim] - Coord[0][iDim];
212 }
213
214 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
215 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
216 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
217
218 Volume += fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
219
220 for (iDim = 0; iDim < NDIM; iDim++) {
221 r1[iDim] = Coord[5][iDim] - Coord[0][iDim];
222 r2[iDim] = Coord[7][iDim] - Coord[0][iDim];
223 r3[iDim] = Coord[4][iDim] - Coord[0][iDim];
224 }
225
226 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
227 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
228 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
229
230 Volume += fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
231
232 for (iDim = 0; iDim < NDIM; iDim++) {
233 r1[iDim] = Coord[7][iDim] - Coord[2][iDim];
234 r2[iDim] = Coord[5][iDim] - Coord[2][iDim];
235 r3[iDim] = Coord[6][iDim] - Coord[2][iDim];
236 }
237
238 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
239 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
240 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
241
242 Volume += fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
243
244 return Volume;
245}
