93void CUpwAUSM_SLAU_Base_NEMO::ComputeJacobian(su2double** val_Jacobian_i, su2double** val_Jacobian_j) {
94 const auto& Ms = fluidmodel->GetSpeciesMolarMass();
95 const auto& Cvtr = fluidmodel->GetSpeciesCvTraRot();
96 const su2double Ru = 1000.0 * UNIVERSAL_GAS_CONSTANT;
97 rhoCvtr_i = V_i[RHOCVTR_INDEX];
98 rhoCvtr_j = V_j[RHOCVTR_INDEX];
99
100 /*--- Initialize the Jacobians ---*/
101 for (auto iVar = 0ul; iVar < nVar; iVar++) {
102 for (auto jVar = 0ul; jVar < nVar; jVar++) {
103 Jacobian_i[iVar][jVar] = 0.0;
104 Jacobian_j[iVar][jVar] = 0.0;
105 }
106 }
107
108 /*--- Determine proper flux and soundspeed ---*/
109 if (M_F >= 0.0)
110 Fc_LR = Fc_L;
111 else
112 Fc_LR = Fc_R;
113
114 su2double A_LR = 0;
115 if (M_F >= 0.0)
116 A_LR = A_F[0];
117 else
118 A_LR = A_F[1];
119
120 /*--- Sound speed derivatives: Species density ---*/
121
122 // Electrons only
123 for (auto iSpecies = 0ul; iSpecies < nEl; iSpecies++) {
124 da_L[iSpecies] = 1.0 / (2.0 * SoundSpeed_i * Density_i) * (1 + dPdU_i[nSpecies + nDim]) *
125 (dPdU_i[iSpecies] - Pressure_i / Density_i);
126 da_R[iSpecies] = 1.0 / (2.0 * SoundSpeed_j * Density_j) * (1 + dPdU_j[nSpecies + nDim]) *
127 (dPdU_j[iSpecies] - Pressure_j / Density_j);
128 }
129
130 // Heavy species
131 for (auto iSpecies = nEl; iSpecies < nSpecies; iSpecies++) {
132 da_L[iSpecies] =
133 1.0 / (2.0 * SoundSpeed_i) *
134 (1 / rhoCvtr_i * (Ru / Ms[iSpecies] - Cvtr[iSpecies] * dPdU_i[nSpecies + nDim]) * Pressure_i / Density_i +
135 1.0 / Density_i * (1.0 + dPdU_i[nSpecies + nDim]) * (dPdU_i[iSpecies] - Pressure_i / Density_i));
136 da_R[iSpecies] =
137 1.0 / (2.0 * SoundSpeed_j) *
138 (1 / rhoCvtr_j * (Ru / Ms[iSpecies] - Cvtr[iSpecies] * dPdU_j[nSpecies + nDim]) * Pressure_j / Density_j +
139 1.0 / Density_j * (1.0 + dPdU_j[nSpecies + nDim]) * (dPdU_j[iSpecies] - Pressure_j / Density_j));
140 }
141
142 /*--- Sound speed derivatives: Momentum ---*/
143 for (auto iDim = 0ul; iDim < nDim; iDim++) {
144 da_L[nSpecies + iDim] = -1.0 / (2.0 * Density_i * SoundSpeed_i) *
145 ((1.0 + dPdU_i[nSpecies + nDim]) * dPdU_i[nSpecies + nDim]) * Velocity_i[iDim];
146 da_R[nSpecies + iDim] = -1.0 / (2.0 * Density_j * SoundSpeed_j) *
147 ((1.0 + dPdU_j[nSpecies + nDim]) * dPdU_j[nSpecies + nDim]) * Velocity_j[iDim];
148 }
149
150 /*--- Sound speed derivatives: Energy ---*/
151 da_L[nSpecies + nDim] =
152 1.0 / (2.0 * Density_i * SoundSpeed_i) * ((1.0 + dPdU_i[nSpecies + nDim]) * dPdU_i[nSpecies + nDim]);
153 da_R[nSpecies + nDim] =
154 1.0 / (2.0 * Density_j * SoundSpeed_j) * ((1.0 + dPdU_j[nSpecies + nDim]) * dPdU_j[nSpecies + nDim]);
155
156 /*--- Sound speed derivatives: Vib-el energy ---*/
157 da_L[nSpecies + nDim + 1] =
158 1.0 / (2.0 * Density_i * SoundSpeed_i) * ((1.0 + dPdU_i[nSpecies + nDim]) * dPdU_i[nSpecies + nDim + 1]);
159 da_R[nSpecies + nDim + 1] =
160 1.0 / (2.0 * Density_j * SoundSpeed_j) * ((1.0 + dPdU_j[nSpecies + nDim]) * dPdU_j[nSpecies + nDim + 1]);
161
162 /*--- Left state Jacobian ---*/
163 if (M_F >= 0) {
164 /*--- Jacobian contribution: dFc terms ---*/
165 for (auto iVar = 0ul; iVar < nSpecies + nDim; iVar++) {
166 for (auto jVar = 0ul; jVar < nVar; jVar++) {
167 Jacobian_i[iVar][jVar] += M_F * Fc_L[iVar] * da_L[jVar];
168 }
169 Jacobian_i[iVar][iVar] += M_F * SoundSpeed_i;
170 }
171 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
172 Jacobian_i[nSpecies + nDim][iSpecies] +=
173 M_F * (dPdU_i[iSpecies] * SoundSpeed_i + Density_i * Enthalpy_i * da_L[iSpecies]);
174 }
175 for (auto iDim = 0ul; iDim < nDim; iDim++) {
176 Jacobian_i[nSpecies + nDim][nSpecies + iDim] +=
177 M_F *
178 (-dPdU_i[nSpecies + nDim] * Velocity_i[iDim] * SoundSpeed_i + Density_i * Enthalpy_i * da_L[nSpecies + iDim]);
179 }
180 Jacobian_i[nSpecies + nDim][nSpecies + nDim] +=
181 M_F * ((1.0 + dPdU_i[nSpecies + nDim]) * SoundSpeed_i + Density_i * Enthalpy_i * da_L[nSpecies + nDim]);
182 Jacobian_i[nSpecies + nDim][nSpecies + nDim + 1] +=
183 M_F * (dPdU_i[nSpecies + nDim + 1] * SoundSpeed_i + Density_i * Enthalpy_i * da_L[nSpecies + nDim + 1]);
184 for (auto jVar = 0ul; jVar < nVar; jVar++) {
185 Jacobian_i[nSpecies + nDim + 1][jVar] += M_F * Fc_L[nSpecies + nDim + 1] * da_L[jVar];
186 }
187 Jacobian_i[nSpecies + nDim + 1][nSpecies + nDim + 1] += M_F * SoundSpeed_i;
188 }
189
190 /*--- Calculate derivatives of the split pressure flux ---*/
191 if ((M_F >= 0) || ((M_F < 0) && (fabs(M_F) <= 1.0))) {
192 if (fabs(M_L) <= 1.0) {
193 /*--- Mach number ---*/
194 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
195 dM_LP[iSpecies] =
196 0.5 * (M_L + 1.0) *
197 (-ProjVelocity_i / (Density_i * SoundSpeed_i) - ProjVelocity_i * da_L[iSpecies] / (pow(SoundSpeed_i, 2)));
198 for (auto iDim = 0ul; iDim < nDim; iDim++)
199 dM_LP[nSpecies + iDim] = 0.5 * (M_L + 1.0) *
200 (-ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + iDim] +
201 UnitNormal[iDim] / (Density_i * SoundSpeed_i));
202 dM_LP[nSpecies + nDim] = 0.5 * (M_L + 1.0) * (-ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + nDim]);
203 dM_LP[nSpecies + nDim + 1] =
204 0.5 * (M_L + 1.0) * (-ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + nDim + 1]);
205
206 /*--- Pressure ---*/
207 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
208 dP_LP[iSpecies] = 0.25 * (M_L + 1.0) *
209 (dPdU_i[iSpecies] * (M_L + 1.0) * (2.0 - M_L) +
210 Pressure_i *
211 (-ProjVelocity_i / (Density_i * SoundSpeed_i) -
212 ProjVelocity_i * da_L[iSpecies] / (pow(SoundSpeed_i, 2))) *
213 (3.0 - 3.0 * M_L));
214 for (auto iDim = 0ul; iDim < nDim; iDim++)
215 dP_LP[nSpecies + iDim] = 0.25 * (M_L + 1.0) *
216 (-Velocity_i[iDim] * dPdU_i[nSpecies + nDim] * (M_L + 1.0) * (2.0 - M_L) +
217 Pressure_i *
218 (-ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + iDim] +
219 UnitNormal[iDim] / (Density_i * SoundSpeed_i)) *
220 (3.0 - 3.0 * M_L));
221 dP_LP[nSpecies + nDim] =
222 0.25 * (M_L + 1.0) *
223 (dPdU_i[nSpecies + nDim] * (M_L + 1.0) * (2.0 - M_L) +
224 Pressure_i * (-ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + nDim]) * (3.0 - 3.0 * M_L));
225 dP_LP[nSpecies + nDim + 1] =
226 0.25 * (M_L + 1.0) *
227 (dPdU_i[nSpecies + nDim + 1] * (M_L + 1.0) * (2.0 - M_L) +
228 Pressure_i * (-ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + nDim + 1]) * (3.0 - 3.0 * M_L));
229 } else {
230 /*--- Mach number ---*/
231 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
232 dM_LP[iSpecies] =
233 -ProjVelocity_i / (Density_i * SoundSpeed_i) - ProjVelocity_i * da_L[iSpecies] / (pow(SoundSpeed_i, 2));
234 for (auto iDim = 0ul; iDim < nDim; iDim++)
235 dM_LP[nSpecies + iDim] = -ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + iDim] +
236 UnitNormal[iDim] / (Density_i * SoundSpeed_i);
237 dM_LP[nSpecies + nDim] = -ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + nDim];
238 dM_LP[nSpecies + nDim + 1] = -ProjVelocity_i / (pow(SoundSpeed_i, 2)) * da_L[nSpecies + nDim + 1];
239
240 /*--- Pressure ---*/
241 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) dP_LP[iSpecies] = dPdU_i[iSpecies];
242 for (auto iDim = 0ul; iDim < nDim; iDim++)
243 dP_LP[nSpecies + iDim] = (-Velocity_i[iDim] * dPdU_i[nSpecies + nDim]);
244 dP_LP[nSpecies + nDim] = dPdU_i[nSpecies + nDim];
245 dP_LP[nSpecies + nDim + 1] = dPdU_i[nSpecies + nDim + 1];
246 }
247
248 /*--- dM contribution ---*/
249 for (auto iVar = 0ul; iVar < nVar; iVar++) {
250 for (auto jVar = 0ul; jVar < nVar; jVar++) {
251 Jacobian_i[iVar][jVar] += dM_LP[jVar] * Fc_LR[iVar] * A_LR;
252 }
253 }
254
255 /*--- Jacobian contribution: dP terms ---*/
256 for (auto iDim = 0ul; iDim < nDim; iDim++) {
257 for (auto iVar = 0ul; iVar < nVar; iVar++) {
258 Jacobian_i[nSpecies + iDim][iVar] += dP_LP[iVar] * UnitNormal[iDim];
259 }
260 }
261 }
262
263 /*--- Right state Jacobian ---*/
264 if (M_F < 0) {
265 /*--- Jacobian contribution: dFc terms ---*/
266 for (auto iVar = 0ul; iVar < nSpecies + nDim; iVar++) {
267 for (auto jVar = 0ul; jVar < nVar; jVar++) {
268 Jacobian_j[iVar][jVar] += M_F * Fc_R[iVar] * da_R[jVar];
269 }
270 Jacobian_j[iVar][iVar] += M_F * SoundSpeed_j;
271 }
272 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
273 Jacobian_j[nSpecies + nDim][iSpecies] +=
274 M_F * (dPdU_j[iSpecies] * SoundSpeed_j + Density_j * Enthalpy_j * da_R[iSpecies]);
275 }
276 for (auto iDim = 0ul; iDim < nDim; iDim++) {
277 Jacobian_j[nSpecies + nDim][nSpecies + iDim] +=
278 M_F *
279 (-dPdU_j[nSpecies + nDim] * Velocity_j[iDim] * SoundSpeed_j + Density_j * Enthalpy_j * da_R[nSpecies + iDim]);
280 }
281 Jacobian_j[nSpecies + nDim][nSpecies + nDim] +=
282 M_F * ((1.0 + dPdU_j[nSpecies + nDim]) * SoundSpeed_j + Density_j * Enthalpy_j * da_R[nSpecies + nDim]);
283 Jacobian_j[nSpecies + nDim][nSpecies + nDim + 1] +=
284 M_F * (dPdU_j[nSpecies + nDim + 1] * SoundSpeed_j + Density_j * Enthalpy_j * da_R[nSpecies + nDim + 1]);
285 for (auto jVar = 0ul; jVar < nVar; jVar++) {
286 Jacobian_j[nSpecies + nDim + 1][jVar] += M_F * Fc_R[nSpecies + nDim + 1] * da_R[jVar];
287 }
288 Jacobian_j[nSpecies + nDim + 1][nSpecies + nDim + 1] += M_F * SoundSpeed_j;
289 }
290
291 /*--- Calculate derivatives of the split pressure flux ---*/
292 if ((M_F < 0) || ((M_F >= 0) && (fabs(M_F) <= 1.0))) {
293 if (fabs(M_R) <= 1.0) {
294 /*--- Mach ---*/
295 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
296 dM_RM[iSpecies] =
297 -0.5 * (M_R - 1.0) *
298 (-ProjVelocity_j / (Density_j * SoundSpeed_j) - ProjVelocity_j * da_R[iSpecies] / (pow(SoundSpeed_j, 2)));
299 for (auto iDim = 0ul; iDim < nDim; iDim++)
300 dM_RM[nSpecies + iDim] = -0.5 * (M_R - 1.0) *
301 (-ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + iDim] +
302 UnitNormal[iDim] / (Density_j * SoundSpeed_j));
303 dM_RM[nSpecies + nDim] = -0.5 * (M_R - 1.0) * (-ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + nDim]);
304 dM_RM[nSpecies + nDim + 1] =
305 -0.5 * (M_R - 1.0) * (-ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + nDim + 1]);
306
307 /*--- Pressure ---*/
308 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
309 dP_RM[iSpecies] = 0.25 * (M_R - 1.0) *
310 (dPdU_j[iSpecies] * (M_R - 1.0) * (2.0 + M_R) +
311 Pressure_j *
312 (-ProjVelocity_j / (Density_j * SoundSpeed_j) -
313 ProjVelocity_j * da_R[iSpecies] / (pow(SoundSpeed_j, 2))) *
314 (3.0 + 3.0 * M_R));
315 for (auto iDim = 0ul; iDim < nDim; iDim++)
316 dP_RM[nSpecies + iDim] = 0.25 * (M_R - 1.0) *
317 ((-Velocity_j[iDim] * dPdU_j[nSpecies + nDim]) * (M_R - 1.0) * (2.0 + M_R) +
318 Pressure_j *
319 (-ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + iDim] +
320 UnitNormal[iDim] / (Density_j * SoundSpeed_j)) *
321 (3.0 + 3.0 * M_R));
322 dP_RM[nSpecies + nDim] =
323 0.25 * (M_R - 1.0) *
324 (dPdU_j[nSpecies + nDim] * (M_R - 1.0) * (2.0 + M_R) +
325 Pressure_j * (-ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + nDim]) * (3.0 + 3.0 * M_R));
326 dP_RM[nSpecies + nDim + 1] =
327 0.25 * (M_R - 1.0) *
328 (dPdU_j[nSpecies + nDim + 1] * (M_R - 1.0) * (2.0 + M_R) +
329 Pressure_j * (-ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + nDim + 1]) * (3.0 + 3.0 * M_R));
330
331 } else {
332 /*--- Mach ---*/
333 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
334 dM_RM[iSpecies] =
335 -ProjVelocity_j / (Density_j * SoundSpeed_j) - ProjVelocity_j * da_R[iSpecies] / (pow(SoundSpeed_j, 2));
336 for (auto iDim = 0ul; iDim < nDim; iDim++)
337 dM_RM[nSpecies + iDim] = -ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + iDim] +
338 UnitNormal[iDim] / (Density_j * SoundSpeed_j);
339 dM_RM[nSpecies + nDim] = -ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + nDim];
340 dM_RM[nSpecies + nDim + 1] = -ProjVelocity_j / (pow(SoundSpeed_j, 2)) * da_R[nSpecies + nDim + 1];
341
342 /*--- Pressure ---*/
343 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) dP_RM[iSpecies] = dPdU_j[iSpecies];
344 for (auto iDim = 0ul; iDim < nDim; iDim++)
345 dP_RM[nSpecies + iDim] = -Velocity_j[iDim] * dPdU_j[nSpecies + nDim];
346 dP_RM[nSpecies + nDim] = dPdU_j[nSpecies + nDim];
347 dP_RM[nSpecies + nDim + 1] = dPdU_j[nSpecies + nDim + 1];
348 }
349
350 /*--- Jacobian contribution: dM terms ---*/
351 for (auto iVar = 0ul; iVar < nVar; iVar++) {
352 for (auto jVar = 0ul; jVar < nVar; jVar++) {
353 Jacobian_j[iVar][jVar] += dM_RM[jVar] * Fc_LR[iVar] * A_LR;
354 }
355 }
356
357 /*--- Jacobian contribution: dP terms ---*/
358 for (auto iDim = 0ul; iDim < nDim; iDim++) {
359 for (auto iVar = 0ul; iVar < nVar; iVar++) {
360 Jacobian_j[nSpecies + iDim][iVar] += dP_RM[iVar] * UnitNormal[iDim];
361 }
362 }
363 }
364
365 /*--- Integrate over dual-face area ---*/
366 for (auto iVar = 0ul; iVar < nVar; iVar++) {
367 for (auto jVar = 0ul; jVar < nVar; jVar++) {
368 Jacobian_i[iVar][jVar] *= Area;
369 Jacobian_j[iVar][jVar] *= Area;
370 }
371 }
372}

374CNumerics::ResidualType<> CUpwAUSM_SLAU_Base_NEMO::ComputeResidual(const CConfig* config) {
375 /*--- Compute geometric quantities ---*/
376 Area = GeometryToolbox::Norm(nDim, Normal);
377
378 for (auto iDim = 0ul; iDim < nDim; iDim++) UnitNormal[iDim] = Normal[iDim] / Area;
379
380 /*--- Pull stored primitive variables ---*/
381 // Primitives: [rho1,...,rhoNs, T, Tve, u, v, w, P, rho, h, a, c]
382 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
383 rhos_i[iSpecies] = V_i[RHOS_INDEX + iSpecies];
384 rhos_j[iSpecies] = V_j[RHOS_INDEX + iSpecies];
385 }
386 for (auto iDim = 0ul; iDim < nDim; iDim++) {
387 Velocity_i[iDim] = V_i[VEL_INDEX + iDim];
388 Velocity_j[iDim] = V_j[VEL_INDEX + iDim];
389 }
390
391 Pressure_i = V_i[P_INDEX];
392 Enthalpy_i = V_i[H_INDEX];
393 Density_i = V_i[RHO_INDEX];
394 SoundSpeed_i = V_i[A_INDEX];
395
396 Pressure_j = V_j[P_INDEX];
397 Enthalpy_j = V_j[H_INDEX];
398 Density_j = V_j[RHO_INDEX];
399 SoundSpeed_j = V_j[A_INDEX];
400
401 e_ve_i = e_ve_j = 0;
402 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
403 e_ve_i += (V_i[RHOS_INDEX + iSpecies] * eve_i[iSpecies]) / Density_i;
404 e_ve_j += (V_j[RHOS_INDEX + iSpecies] * eve_j[iSpecies]) / Density_j;
405 }
406
407 /*--- Projected velocities ---*/
408 ProjVelocity_i = GeometryToolbox::DotProduct(nDim, Velocity_i, UnitNormal);
409 ProjVelocity_j = GeometryToolbox::DotProduct(nDim, Velocity_j, UnitNormal);
410
411 /*--- Compute mass and pressure fluxes of specific scheme ---*/
412 ComputeInterfaceQuantities(config, PressureFlux, M_F, A_F);
413
414 const su2double MassFlux_i = M_F * A_F[0];
415 const su2double MassFlux_j = M_F * A_F[1];
416
417 const su2double DissFlux_i = fabs(MassFlux_i);
418 const su2double DissFlux_j = fabs(MassFlux_j);
419
420 /*--- Assign left & right convective flux vectors ---*/
421 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
422 Fc_L[iSpecies] = rhos_i[iSpecies];
423 Fc_R[iSpecies] = rhos_j[iSpecies];
424 }
425 for (auto iDim = 0ul; iDim < nDim; iDim++) {
426 Fc_L[nSpecies + iDim] = Density_i * Velocity_i[iDim];
427 Fc_R[nSpecies + iDim] = Density_j * Velocity_j[iDim];
428 }
429 Fc_L[nSpecies + nDim] = Density_i * Enthalpy_i;
430 Fc_R[nSpecies + nDim] = Density_j * Enthalpy_j;
431 Fc_L[nSpecies + nDim + 1] = Density_i * e_ve_i;
432 Fc_R[nSpecies + nDim + 1] = Density_j * e_ve_j;
433
434 /*--- Compute numerical flux ---*/
435 for (auto iVar = 0ul; iVar < nVar; iVar++)
436 Flux[iVar] = 0.5 * ((MassFlux_i + DissFlux_i) * Fc_L[iVar] + (MassFlux_j - DissFlux_j) * Fc_R[iVar]) * Area;
437
438 for (auto iDim = 0ul; iDim < nDim; iDim++) Flux[nSpecies + iDim] += PressureFlux[iDim] * Area;
439
440 /*--- If required, compute Jacobians (approximated using AUSM) ---*/
441 if (implicit) ComputeJacobian(Jacobian_i, Jacobian_j);
442
443 return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
444}

450void CUpwAUSM_NEMO::ComputeInterfaceQuantities(const CConfig* config, su2double* pressure, su2double& interface_mach,
451 su2double* interface_soundspeed) {
452 /*--- Calculate L/R Mach numbers ---*/
453 interface_soundspeed[0] = SoundSpeed_i;
454 interface_soundspeed[1] = SoundSpeed_j;
455 M_L = ProjVelocity_i / interface_soundspeed[0];
456 M_R = ProjVelocity_j / interface_soundspeed[1];
457
458 /*--- Calculate split numerical fluxes ---*/
459 su2double M_LP, M_RM, P_LP, P_RM;
460
461 if (fabs(M_L) <= 1.0) {
462 M_LP = 0.25 * (M_L + 1.0) * (M_L + 1.0);
463 P_LP = 0.25 * Pressure_i * (M_L + 1.0) * (M_L + 1.0) * (2.0 - M_L);
464 } else {
465 M_LP = 0.5 * (M_L + fabs(M_L));
466 P_LP = 0.5 * Pressure_i * (M_L + fabs(M_L)) / M_L;
467 }
468
469 if (fabs(M_R) <= 1.0) {
470 M_RM = -0.25 * (M_R - 1.0) * (M_R - 1.0);
471 P_RM = 0.25 * Pressure_j * (M_R - 1.0) * (M_R - 1.0) * (2.0 + M_R);
472 } else {
473 M_RM = 0.5 * (M_R - fabs(M_R));
474 P_RM = 0.5 * Pressure_j * (M_R - fabs(M_R)) / M_R;
475 }
476
477 // M_Lnferface = M(1/2) = (M_LP + M_RM)
478 interface_mach = (M_LP + M_RM);
479
480 // Split pressure P(1/2)
481 // Note: only a single pressure flux in AUSM
482 for (auto iDim = 0ul; iDim < nDim; iDim++) pressure[iDim] = (P_LP + P_RM) * UnitNormal[iDim];
483}

497void CUpwAUSMPLUSUP2_NEMO::ComputeInterfaceQuantities(const CConfig* config, su2double* pressure,
498 su2double& interface_mach, su2double* interface_soundspeed) {
499 const su2double sq_veli = GeometryToolbox::SquaredNorm(nDim, Velocity_i);
500 const su2double sq_velj = GeometryToolbox::SquaredNorm(nDim, Velocity_j);
501
502 /*--- Compute C* ---*/
503 const su2double CstarL = sqrt(2.0 * (Gamma_i - 1.0) / (Gamma_i + 1.0) * Enthalpy_i);
504 const su2double CstarR = sqrt(2.0 * (Gamma_j - 1.0) / (Gamma_j + 1.0) * Enthalpy_j);
505
506 /*--- Compute C^ ---*/
507 const su2double ChatL = CstarL * CstarL / max(CstarL, ProjVelocity_i);
508 const su2double ChatR = CstarR * CstarR / max(CstarR, -ProjVelocity_j);
509
510 /*--- Interface speed of sound ---*/
511 const su2double aF = min(ChatL, ChatR);
512 interface_soundspeed[0] = interface_soundspeed[1] = aF;
513
514 const su2double M_L = ProjVelocity_i / aF;
515 const su2double M_R = ProjVelocity_j / aF;
516
517 const su2double rhoF = 0.5 * (Density_i + Density_j);
518 const su2double MFsq = 0.5 * (M_L * M_L + M_R * M_R);
519
520 const su2double param1 = max(MFsq, Minf * Minf);
521 const su2double Mrefsq = (min(1.0, param1));
522 const su2double fa = 2.0 * sqrt(Mrefsq) - Mrefsq;
523
524 const su2double alpha = 3.0 / 16.0 * (-4.0 + 5.0 * fa * fa);
525 const su2double beta = 1.0 / 8.0;
526
527 /*--- Pressure diffusion term ---*/
528 const su2double Mp = -(Kp / fa) * max((1.0 - sigma * MFsq), 0.0) * (Pressure_j - Pressure_i) / (rhoF * aF * aF);
529
530 su2double M_LP, P_LP, M_RM, P_RM;
531
532 if (fabs(M_L) <= 1.0) {
533 M_LP = 0.25 * (M_L + 1.0) * (M_L + 1.0) + beta * (M_L * M_L - 1.0) * (M_L * M_L - 1.0);
534 P_LP = (0.25 * (M_L + 1.0) * (M_L + 1.0) * (2.0 - M_L) + alpha * M_L * (M_L * M_L - 1.0) * (M_L * M_L - 1.0));
535 } else {
536 M_LP = 0.5 * (M_L + fabs(M_L));
537 P_LP = 0.5 * (M_L + fabs(M_L)) / M_L;
538 }
539
540 if (fabs(M_R) <= 1.0) {
541 M_RM = -0.25 * (M_R - 1.0) * (M_R - 1.0) - beta * (M_R * M_R - 1.0) * (M_R * M_R - 1.0);
542 P_RM = (0.25 * (M_R - 1.0) * (M_R - 1.0) * (2.0 + M_R) - alpha * M_R * (M_R * M_R - 1.0) * (M_R * M_R - 1.0));
543 } else {
544 M_RM = 0.5 * (M_R - fabs(M_R));
545 P_RM = 0.5 * (M_R - fabs(M_R)) / M_R;
546 }
547
548 /*--- Interface Mach number ---*/
549 interface_mach = (M_LP + M_RM + Mp);
550
551 /*--- Modified pressure flux ---*/
552 const su2double pFi = sqrt(0.5 * (sq_veli + sq_velj)) * (P_LP + P_RM - 1.0) * 0.5 * (Density_j + Density_i) * aF;
553
554 for (auto iDim = 0ul; iDim < nDim; iDim++) {
555 pressure[iDim] =
556 (0.5 * (Pressure_j + Pressure_i) + 0.5 * (P_LP - P_RM) * (Pressure_i - Pressure_j) + pFi) * UnitNormal[iDim];
557 }
558}

570void CUpwAUSMPLUSM_NEMO::ComputeInterfaceQuantities(const CConfig* config, su2double* pressure,
571 su2double& interface_mach, su2double* interface_soundspeed) {
572 const su2double sq_veli = GeometryToolbox::SquaredNorm(nDim, Velocity_i);
573 const su2double sq_velj = GeometryToolbox::SquaredNorm(nDim, Velocity_j);
574
575 /*--- Calculate interface numerical gammas and speed of sound ---*/
576 const su2double Enthalpy_norm = 0.5 * (Enthalpy_i + Enthalpy_j);
577 const su2double Gamma_ij = 0.5 * (Gamma_i + Gamma_j);
578 const su2double A_s = sqrt(2.0 * Enthalpy_norm * (Gamma_ij - 1.0) / (Gamma_ij + 1.0));
579
580 su2double A_F;
581 if (0.5 * (ProjVelocity_i + ProjVelocity_j) >= 0.0)
582 A_F = A_s * A_s / max(fabs(ProjVelocity_i), A_s);
583 else
584 A_F = A_s * A_s / max(fabs(ProjVelocity_j), A_s);
585
586 /*--- Compute L/R Mach numbers ---*/
587 interface_soundspeed[0] = interface_soundspeed[1] = A_F;
588 M_L = ProjVelocity_i / interface_soundspeed[0];
589 M_R = ProjVelocity_j / interface_soundspeed[1];
590
591 /*--- Interface mach number w/ pressure diffusion term (M_p)---*/
592 const su2double Density_F = 0.5 * (Density_i + Density_j);
593 const su2double MF_sq = 0.5 * (sq_veli + sq_velj) / (A_F * A_F);
594
595 const su2double param1 = max(MF_sq, Minf * Minf);
596 const su2double Mrefsq = (min(1.0, param1));
597 const su2double fa = 2.0 * sqrt(Mrefsq) - Mrefsq;
598
599 const su2double alpha = 3.0 / 16.0 * (-4.0 + 5.0 * fa * fa);
600 const su2double f = 0.5 * (1 - cos(PI_NUMBER * min(1.0, max(abs(M_L), abs(M_R)))));
601
602 /*--- Pressure sensor terms ---*/
603 const su2double h = min(Sensor_i, Sensor_j);
604 const su2double g = 0.5 * (1 + cos(PI_NUMBER * h));
605 const su2double f0 = min(1.0, max(f, Minf * Minf));
606
607 /*--- Pressure diffusion term ---*/
608 const su2double M_p = -0.5 * (1.0 - f) * (Pressure_j - Pressure_i) / (Density_F * A_F * A_F) * (1.0 - g);
609
610 /*--- Compute base split mach and pressure fluxes ---*/
611 su2double M_LP, M_RM, P_LP, P_RM;
612
613 if (fabs(M_L) <= 1.0) {
614 M_LP = 0.25 * (M_L + 1.0) * (M_L + 1.0) + beta * (M_L * M_L - 1.0) * (M_L * M_L - 1.0);
615 P_LP = (0.25 * (M_L + 1.0) * (M_L + 1.0) * (2.0 - M_L) + alpha * M_L * (M_L * M_L - 1.0) * (M_L * M_L - 1.0));
616 } else {
617 M_LP = 0.5 * (M_L + fabs(M_L));
618 P_LP = 0.5 * (M_L + fabs(M_L)) / M_L;
619 }
620
621 if (fabs(M_R) <= 1.0) {
622 M_RM = -0.25 * (M_R - 1.0) * (M_R - 1.0) - beta * (M_R * M_R - 1.0) * (M_R * M_R - 1.0);
623 P_RM = (0.25 * (M_R - 1.0) * (M_R - 1.0) * (2.0 + M_R) - alpha * M_R * (M_R * M_R - 1.0) * (M_R * M_R - 1.0));
624 } else {
625 M_RM = 0.5 * (M_R - fabs(M_R));
626 P_RM = 0.5 * (M_R - fabs(M_R)) / M_R;
627 }
628
629 interface_mach = M_LP + M_RM + M_p;
630
631 /*--- Compute and add pressure sensor term to pressure flux ---*/
632 const su2double pFi = f0 * (P_LP + P_RM - 1.0) * 0.5 * (Pressure_i + Pressure_j);
633
634 /*--- Velocity diffusion term---*/
635 su2double P_un[MAXNDIM] = {0.0};
636
637 for (auto iDim = 0ul; iDim < nDim; iDim++) {
638 su2double Vel_L = (Velocity_i[iDim] + ProjVelocity_i * UnitNormal[iDim]);
639 su2double Vel_R = (Velocity_j[iDim] - ProjVelocity_j * UnitNormal[iDim]);
640
641 P_un[iDim] = -g * Gamma_ij * 0.5 * (Pressure_i + Pressure_j) / A_F * P_LP * P_RM * (Vel_R - Vel_L);
642 }
643
644 /*--- Pressure flux ---*/
645 for (auto iDim = 0ul; iDim < nDim; iDim++) {
646 pressure[iDim] =
647 (0.5 * (Pressure_j + Pressure_i) + 0.5 * (P_LP - P_RM) * (Pressure_i - Pressure_j) + pFi) * UnitNormal[iDim] +
648 P_un[iDim];
649 }
650}
