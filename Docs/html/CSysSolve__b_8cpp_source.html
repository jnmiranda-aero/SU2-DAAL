56 /*--- Initialize the right-hand side with the gradient of the solution of the primal linear system ---*/
57
58 SU2_OMP_BARRIER
59 SU2_OMP_FOR_STAT(roundUpDiv(n, omp_get_num_threads()))
60 for (unsigned long i = 0; i < n; i++) {

69 for (unsigned long i = 0; i < n; i++) {
70 x_b[i] = SU2_TYPE::GetValue((*LinSysSol_b)[i]);
71 }
72 END_SU2_OMP_FOR
