36/*!
37 * \class CMatrixVectorProduct
38 * \ingroup SpLinSys
39 * \brief Abstract base class for defining matrix-vector products
40 * \author J. Hicken.
41 *
42 * The Krylov-subspace solvers require only matrix-vector products and
43 * not the actual matrix/Jacobian. We need some way to indicate which
44 * function will perform the product. However, sometimes the
45 * functions that define the product will require different numbers
46 * and types of inputs. For example, the forward-difference
47 * approximation to a Jacobian-vector product requires the vector that
48 * defines the Jacobian and a perturbation parameter. The
49 * CMatrixVectorProduct class is used to derive child classes that can
50 * handle the different types of matrix-vector products and still be
51 * passed to a single implementation of the Krylov solvers.
52 * This abstraction may also be used to define matrix-free products.
53 */
54template <class ScalarType>
55class CMatrixVectorProduct {
56 public:
57 virtual ~CMatrixVectorProduct() = 0;
58 virtual void operator()(const CSysVector<ScalarType>& u, CSysVector<ScalarType>& v) const = 0;
59};
60template <class ScalarType>
61CMatrixVectorProduct<ScalarType>::~CMatrixVectorProduct() {}

69class CSysMatrixVectorProduct final : public CMatrixVectorProduct<ScalarType> {

74
75 public:
76 /*!
77 * \brief constructor of the class
78 * \param[in] matrix_ref - matrix reference that will be used to define the products
79 * \param[in] geometry_ref - geometry associated with the problem
80 * \param[in] config_ref - config of the problem
81 */
82 inline CSysMatrixVectorProduct(const CSysMatrix<ScalarType>& matrix_ref, CGeometry* geometry_ref,
83 const CConfig* config_ref)
84 : matrix(matrix_ref), geometry(geometry_ref), config(config_ref) {}
85
86 /*!
87 * \note This class cannot be default constructed as that would leave us with invalid pointers.
88 */
89 CSysMatrixVectorProduct() = delete;
90
91 /*!
92 * \brief operator that defines the CSysMatrix-CSysVector product
93 * \param[in] u - CSysVector that is being multiplied by the sparse matrix
94 * \param[out] v - CSysVector that is the result of the product
95 */
96 inline void operator()(const CSysVector<ScalarType>& u, CSysVector<ScalarType>& v) const override {
97 matrix.MatrixVectorProduct(u, v, geometry, config);
98 }
99};
