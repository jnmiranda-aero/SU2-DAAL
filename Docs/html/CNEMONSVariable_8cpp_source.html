79 bool CNEMONSVariable::SetPrimVar(unsigned long iPoint, CFluidModel *FluidModel) {
80 
81  fluidmodel = static_cast<CNEMOGas*>(FluidModel);
82 
83  /*--- Convert conserved to primitive variables ---*/
84  bool nonPhys = Cons2PrimVar(Solution[iPoint], Primitive[iPoint], dPdU[iPoint], dTdU[iPoint], dTvedU[iPoint], eves[iPoint], Cvves[iPoint]);
85 
86  /*--- Reset solution to previous one, if nonphys ---*/
87  if (nonPhys) {
88  for (auto iVar = 0u; iVar < nVar; iVar++)
89  Solution(iPoint,iVar) = Solution_Old(iPoint,iVar);
90 
91  /*--- Recompute Primitive from previous solution ---*/
92  Cons2PrimVar(Solution[iPoint], Primitive[iPoint], dPdU[iPoint], dTdU[iPoint], dTvedU[iPoint], eves[iPoint], Cvves[iPoint]);
93  }
94 
95  /*--- Set additional point quantities ---*/
96  Gamma(iPoint) = fluidmodel->ComputeGamma();
97 
98  SetVelocity2(iPoint);
99 
100  const auto& Ds = fluidmodel->GetDiffusionCoeff();
101  for (auto iSpecies = 0u; iSpecies < nSpecies; iSpecies++)
102  DiffusionCoeff(iPoint, iSpecies) = Ds[iSpecies];
103 
104  su2double T = Primitive(iPoint,nSpecies);
105  su2double Tve = Primitive(iPoint,nSpecies+1);
106 
107  su2double* val_eves = GetEve(iPoint);
108  const auto& hs = fluidmodel->ComputeSpeciesEnthalpy(T, Tve, val_eves);
109  for (auto iSpecies = 0u; iSpecies < nSpecies; iSpecies++)
110  Enthalpys(iPoint, iSpecies) = hs[iSpecies];
111 
112  LaminarViscosity(iPoint) = fluidmodel->GetViscosity();
113 
114  const auto& thermalconductivities = fluidmodel->GetThermalConductivities();
115  ThermalCond(iPoint) = thermalconductivities[0];
116  ThermalCond_ve(iPoint) = thermalconductivities[1];
117 
118  Primitive(iPoint, LAM_VISC_INDEX) = LaminarViscosity(iPoint);
119 
120  return nonPhys;
121 }
