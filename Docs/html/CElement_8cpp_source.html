30 CElement::CElement(unsigned short ngauss, unsigned short nnodes, unsigned short ndim) {
31  nGaussPoints = ngauss;
32  nNodes = nnodes;
33  nDim = ndim;
34 
35  /*--- Allocate structures. ---*/
36 
37  /*--- Always allocate MAXNDIM to have enough space in all cases.
38  * e.g. elements embedded on a curved surfaces. ---*/
39  CurrentCoord.resize(nNodes, MAXNDIM) = su2double(0.0);
40  RefCoord.resize(nNodes, MAXNDIM) = su2double(0.0);
41 
42  GaussPoint.reserve(nGaussPoints);
43  for (unsigned short iGauss = 0; iGauss < nGaussPoints; ++iGauss) GaussPoint.emplace_back(iGauss, nDim, nNodes);
44 
45  GaussWeight.resize(nGaussPoints) = su2double(0.0);
46  NodalExtrap.resize(nNodes, nGaussPoints) = su2double(0.0);
47 
48  NodalStress.resize(nNodes, 6) = su2double(0.0);
49 
50  Mab.resize(nNodes, nNodes);
51  Ks_ab.resize(nNodes, nNodes);
52  Kab.resize(nNodes);
53  for (auto& kab_i : Kab) kab_i.resize(nNodes, nDim * nDim);
54 
55  Kt_a.resize(nNodes, nDim);
56  FDL_a.resize(nNodes, nDim);
57 
58  HiHj.resize(nNodes, nNodes);
59  DHiDHj.resize(nNodes);
60  for (auto& DHiDHj_a : DHiDHj) {
61  DHiDHj_a.resize(nNodes);
62  for (auto& DHiDHj_ab : DHiDHj_a) DHiDHj_ab.resize(nDim, nDim);
63  }
64 
65  ClearElement();
66 }
