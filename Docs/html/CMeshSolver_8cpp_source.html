36CMeshSolver::CMeshSolver(CGeometry *geometry, CConfig *config) : CFEASolver(LINEAR_SOLVER_MODE::MESH_DEFORM) {
37
38 /*--- Initialize some booleans that determine the kind of problem at hand. ---*/
39
40 time_domain = config->GetTime_Domain();
41 multizone = config->GetMultizone_Problem();
42
43 /*--- Determine if the stiffness per-element is set ---*/
44 switch (config->GetDeform_Stiffness_Type()) {
45 case INVERSE_VOLUME:
46 case SOLID_WALL_DISTANCE:
47 stiffness_set = false;
48 break;
49 case CONSTANT_STIFFNESS:
50 stiffness_set = true;
51 break;
52 }
53
54 /*--- Initialize the number of spatial dimensions, length of the state
55 vector (same as spatial dimensions for grid deformation), and grid nodes. ---*/
56
57 unsigned short iDim;
58 unsigned long iPoint, iElem;
59
60 nDim = geometry->GetnDim();
61 nVar = geometry->GetnDim();
62 nPoint = geometry->GetnPoint();
63 nPointDomain = geometry->GetnPointDomain();
64 nElement = geometry->GetnElem();
65
66 MinVolume_Ref = 0.0;
67 MinVolume_Curr = 0.0;
68
69 MaxVolume_Ref = 0.0;
70 MaxVolume_Curr = 0.0;
71
72 /*--- Initialize the node structure ---*/
73
74 nodes = new CMeshBoundVariable(nPoint, nDim, config);
75 SetBaseClassPointerToNodes();
76
77 /*--- Set which points are vertices and allocate boundary data. ---*/
78
79 for (iPoint = 0; iPoint < nPoint; iPoint++) {
80
81 for (iDim = 0; iDim < nDim; ++iDim)
82 nodes->SetMesh_Coord(iPoint, iDim, geometry->nodes->GetCoord(iPoint, iDim));
83
84 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
85 long iVertex = geometry->nodes->GetVertex(iPoint, iMarker);
86 if (iVertex >= 0) {
87 nodes->Set_isVertex(iPoint,true);
88 break;
89 }
90 }
91 }
92 static_cast<CMeshBoundVariable*>(nodes)->AllocateBoundaryVariables(config);
93
94 /*--- Initialize the element structure ---*/
95
96 element.resize(nElement);
97
98 /*--- Initialize matrix, solution, and r.h.s. structures for the linear solver. ---*/
99
100 if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (Mesh Deformation)." << endl;
101
102 LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
103 LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
104 Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, false, geometry, config);
105
106 /*--- Initialize structures for hybrid-parallel mode. ---*/
107
108 HybridParallelInitialization(geometry);
109
110 /*--- Element container structure. ---*/
111
112 if (nDim == 2) {
113 for(int thread = 0; thread < omp_get_max_threads(); ++thread) {
114
115 const int offset = thread*MAX_FE_KINDS;
116 element_container[FEA_TERM][EL_TRIA+offset] = new CTRIA1();
117 element_container[FEA_TERM][EL_QUAD+offset] = new CQUAD4();
118 }
119 }
120 else {
121 for(int thread = 0; thread < omp_get_max_threads(); ++thread) {
122
123 const int offset = thread*MAX_FE_KINDS;
124 element_container[FEA_TERM][EL_TETRA+offset] = new CTETRA1();
125 element_container[FEA_TERM][EL_HEXA +offset] = new CHEXA8 ();
126 element_container[FEA_TERM][EL_PYRAM+offset] = new CPYRAM5();
127 element_container[FEA_TERM][EL_PRISM+offset] = new CPRISM6();
128 }
129 }
130
131 /*--- Initialize the BGS residuals in multizone problems. ---*/
132 if (config->GetMultizone_Residual()){
133 Residual_BGS.resize(nVar,0.0);
134 Residual_Max_BGS.resize(nVar,0.0);
135 Point_Max_BGS.resize(nVar,0);
136 Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
137 }
138
139
140 /*--- Allocate element properties - only the index, to allow further integration with CFEASolver on a later stage ---*/
141 element_properties = new CProperty*[nElement];
142 for (iElem = 0; iElem < nElement; iElem++){
143 element_properties[iElem] = new CProperty(iElem);
144 }
145
146 /*--- Compute the element volumes using the reference coordinates ---*/
147 SU2_OMP_PARALLEL {
148 SetMinMaxVolume(geometry, config, false);
149 }
150 END_SU2_OMP_PARALLEL
151
152 /*--- Compute the wall distance using the reference coordinates ---*/
153 SetWallDistance(geometry, config);
154
155 if (size != SINGLE_NODE) {
156 vector<unsigned short> essentialMarkers;
157 /*--- Markers types covered in SetBoundaryDisplacements. ---*/
158 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
159 if (((config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) &&
160 (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) ||
161 (config->GetMarker_All_Deform_Mesh(iMarker) == YES) ||
162 (config->GetMarker_All_Moving(iMarker) == YES)) {
163 essentialMarkers.push_back(iMarker);
164 }
165 }
166 Set_VertexEliminationSchedule(geometry, essentialMarkers);
167 }
168
169 SolverName = "MESH";

172void CMeshSolver::SetMinMaxVolume(CGeometry *geometry, CConfig *config, bool updated) {
173
174 /*--- This routine is for post processing, it does not need to be recorded. ---*/
175 const bool wasActive = AD::BeginPassive();
176
177 /*--- Initialize shared reduction variables. ---*/
178 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
179 MaxVolume = -1E22; MinVolume = 1E22;
180 ElemCounter = 0;
181 }
182 END_SU2_OMP_SAFE_GLOBAL_ACCESS
183
184 /*--- Local min/max, final reduction outside loop. ---*/
185 su2double maxVol = -1E22, minVol = 1E22;
186 unsigned long elCount = 0;
187
188 /*--- Loop over the elements in the domain. ---*/
189
190 SU2_OMP_FOR_DYN(omp_chunk_size)
191 for (unsigned long iElem = 0; iElem < nElement; iElem++) {
192
193 int thread = omp_get_thread_num();
194
195 int EL_KIND;
196 unsigned short iNode, nNodes, iDim;
197
198 GetElemKindAndNumNodes(geometry->elem[iElem]->GetVTK_Type(), EL_KIND, nNodes);
199
200 CElement* fea_elem = element_container[FEA_TERM][EL_KIND + thread*MAX_FE_KINDS];
201
202 /*--- For the number of nodes, we get the coordinates from
203 * the connectivity matrix and the geometry structure. ---*/
204
205 for (iNode = 0; iNode < nNodes; iNode++) {
206
207 auto indexNode = geometry->elem[iElem]->GetNode(iNode);
208
209 /*--- Compute the volume with the reference or current coordinates. ---*/
210 for (iDim = 0; iDim < nDim; iDim++) {
211 su2double val_Coord = nodes->GetMesh_Coord(indexNode,iDim);
212 if (updated)
213 val_Coord += nodes->GetSolution(indexNode,iDim);
214
215 fea_elem->SetRef_Coord(iNode, iDim, val_Coord);
216 }
217 }
218
219 /*--- Compute the volume of the element (or the area in 2D cases ). ---*/
220
221 su2double ElemVolume;
222 if (nDim == 2) ElemVolume = fea_elem->ComputeArea();
223 else ElemVolume = fea_elem->ComputeVolume();
224
225 maxVol = max(maxVol, ElemVolume);
226 minVol = min(minVol, ElemVolume);
227
228 if (updated) element[iElem].SetCurr_Volume(ElemVolume);
229 else element[iElem].SetRef_Volume(ElemVolume);
230
231 /*--- Count distorted elements. ---*/
232 if (ElemVolume <= 0.0) elCount++;
233 }
234 END_SU2_OMP_FOR
235 SU2_OMP_CRITICAL
236 {
237 MaxVolume = max(MaxVolume, maxVol);
238 MinVolume = min(MinVolume, minVol);
239 ElemCounter += elCount;
240 }
241 END_SU2_OMP_CRITICAL
242
243 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
244 {
245 elCount = ElemCounter; maxVol = MaxVolume; minVol = MinVolume;
246 SU2_MPI::Allreduce(&elCount, &ElemCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
247 SU2_MPI::Allreduce(&maxVol, &MaxVolume, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
248 SU2_MPI::Allreduce(&minVol, &MinVolume, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
249 }
250 END_SU2_OMP_SAFE_GLOBAL_ACCESS
251
252 /*--- Volume from 0 to 1 ---*/
253
254 SU2_OMP_FOR_STAT(omp_chunk_size)
255 for (unsigned long iElem = 0; iElem < nElement; iElem++) {
256 if (updated) {
257 su2double ElemVolume = element[iElem].GetCurr_Volume()/MaxVolume;
258 element[iElem].SetCurr_Volume(ElemVolume);
259 }
260 else {
261 su2double ElemVolume = element[iElem].GetRef_Volume()/MaxVolume;
262 element[iElem].SetRef_Volume(ElemVolume);
263 }
264 }
265 END_SU2_OMP_FOR
266
267 /*--- Store the maximum and minimum volume. ---*/
268 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
269 if (updated) {
270 MaxVolume_Curr = MaxVolume;
271 MinVolume_Curr = MinVolume;
272 }
273 else {
274 MaxVolume_Ref = MaxVolume;
275 MinVolume_Ref = MinVolume;
276 }
277
278 if ((ElemCounter != 0) && (rank == MASTER_NODE))
279 cout <<"There are " << ElemCounter << " elements with negative volume.\n" << endl;
280
281 }
282 END_SU2_OMP_SAFE_GLOBAL_ACCESS
283
284 AD::EndPassive(wasActive);

288void CMeshSolver::SetWallDistance(CGeometry *geometry, CConfig *config) {
289
290 /*--- Initialize min and max distance ---*/
291
292 MaxDistance = -1E22; MinDistance = 1E22;
293
294 /*--- Compute the total number of nodes on no-slip boundaries ---*/
295
296 unsigned long nVertex_SolidWall = 0;
297 for(auto iMarker=0u; iMarker<config->GetnMarker_All(); ++iMarker) {
298 if(config->GetSolid_Wall(iMarker) && !config->GetMarker_All_Deform_Mesh_Sym_Plane(iMarker)) {
299 nVertex_SolidWall += geometry->GetnVertex(iMarker);
300 }
301 }
302
303 /*--- Allocate the vectors to hold boundary node coordinates
304 and its local ID. ---*/
305
306 vector<su2double> Coord_bound(nDim*nVertex_SolidWall);
307 vector<unsigned long> PointIDs(nVertex_SolidWall);
308
309 /*--- Retrieve and store the coordinates of the no-slip boundary nodes
310 and their local point IDs. ---*/
311
312
313 for (unsigned long iMarker=0, ii=0, jj=0; iMarker<config->GetnMarker_All(); ++iMarker) {
314
315 if (!config->GetSolid_Wall(iMarker) || config->GetMarker_All_Deform_Mesh_Sym_Plane(iMarker)) continue;
316
317 for (auto iVertex=0u; iVertex<geometry->GetnVertex(iMarker); ++iVertex) {
318 auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
319 PointIDs[jj++] = iPoint;
320 for (auto iDim=0u; iDim<nDim; ++iDim){
321 Coord_bound[ii++] = nodes->GetMesh_Coord(iPoint,iDim);
322 }
323 }
324 }
325
326 /*--- Build the ADT of the boundary nodes. ---*/
327
328 CADTPointsOnlyClass WallADT(nDim, nVertex_SolidWall, Coord_bound.data(),
329 PointIDs.data(), true);
330
331 SU2_OMP_PARALLEL
332 {
333 /*--- Loop over all interior mesh nodes and compute the distances to each
334 of the no-slip boundary nodes. Store the minimum distance to the wall
335 for each interior mesh node. ---*/
336
337 if( WallADT.IsEmpty() ) {
338
339 /*--- No solid wall boundary nodes in the entire mesh. Set the
340 wall distance to MaxDistance so we get stiffness of 1. ---*/
341
342 SU2_OMP_FOR_STAT(omp_chunk_size)
343 for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint) {
344 nodes->SetWallDistance(iPoint, MaxDistance);
345 }
346 END_SU2_OMP_FOR
347 }
348 else {
349 su2double MaxDistance_Local = -1E22, MinDistance_Local = 1E22;
350
351 /*--- Solid wall boundary nodes are present. Compute the wall
352 distance for all nodes. ---*/
353 SU2_OMP_FOR_DYN(omp_chunk_size)
354 for(auto iPoint = 0ul; iPoint < nPoint; ++iPoint) {
355 su2double dist;
356 unsigned long pointID;
357 int rankID;
358 WallADT.DetermineNearestNode(nodes->GetMesh_Coord(iPoint), dist,
359 pointID, rankID);
360 nodes->SetWallDistance(iPoint,dist);
361
362 MaxDistance_Local = max(MaxDistance_Local, dist);
363
364 /*--- To discard points on the surface we use > EPS ---*/
365
366 if (dist > EPS) MinDistance_Local = min(MinDistance_Local, dist);
367
368 }
369 END_SU2_OMP_FOR
370 SU2_OMP_CRITICAL
371 {
372 MaxDistance = max(MaxDistance, MaxDistance_Local);
373 MinDistance = min(MinDistance, MinDistance_Local);
374 }
375 END_SU2_OMP_CRITICAL
376
377 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
378 {
379 MaxDistance_Local = MaxDistance;
380 MinDistance_Local = MinDistance;
381 SU2_MPI::Allreduce(&MaxDistance_Local, &MaxDistance, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
382 SU2_MPI::Allreduce(&MinDistance_Local, &MinDistance, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
383 }
384 END_SU2_OMP_SAFE_GLOBAL_ACCESS
385 }
386
387 /*--- Normalize distance from 0 to 1 ---*/
388 SU2_OMP_FOR_STAT(omp_chunk_size)
389 for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint) {
390 su2double nodeDist = nodes->GetWallDistance(iPoint)/MaxDistance;
391 nodes->SetWallDistance(iPoint,nodeDist);
392 }
393 END_SU2_OMP_FOR
394
395 /*--- Compute the element distances ---*/
396 SU2_OMP_FOR_STAT(omp_chunk_size)
397 for (auto iElem = 0ul; iElem < nElement; iElem++) {
398
399 int EL_KIND;
400 unsigned short nNodes = 0;
401 GetElemKindAndNumNodes(geometry->elem[iElem]->GetVTK_Type(), EL_KIND, nNodes);
402
403 /*--- Average the distance of the nodes in the element ---*/
404
405 su2double ElemDist = 0.0;
406 for (auto iNode = 0u; iNode < nNodes; iNode++) {
407 auto iPoint = geometry->elem[iElem]->GetNode(iNode);
408 ElemDist += nodes->GetWallDistance(iPoint);
409 }
410 ElemDist = ElemDist/su2double(nNodes);
411
412 element[iElem].SetWallDistance(ElemDist);
413 }
414 END_SU2_OMP_FOR
415
416 }
417 END_SU2_OMP_PARALLEL
418}

420void CMeshSolver::SetMesh_Stiffness(CNumerics **numerics, CConfig *config){
421
422 if (stiffness_set) return;
423
424 /*--- Use the config option as an upper bound on elasticity modulus.
425 * For RANS meshes the range of element volume or wall distance is
426 * very large and leads to an ill-conditioned stiffness matrix.
427 * Absolute values of elasticity modulus are not important for
428 * mesh deformation, since linear elasticity is used and all
429 * boundary conditions are essential (Dirichlet). ---*/
430 const su2double maxE = config->GetDeform_ElasticityMod();
431
432 /*--- All threads must execute the entire loop (no worksharing),
433 * each sets the stiffnesses for its numerics instance. ---*/
434 SU2_OMP_PARALLEL
435 {
436 CNumerics* myNumerics = numerics[FEA_TERM + omp_get_thread_num()*MAX_TERMS];
437
438 switch (config->GetDeform_Stiffness_Type()) {
439
440 /*--- Stiffness inverse of the volume of the element. ---*/
441 case INVERSE_VOLUME:
442 for (unsigned long iElem = 0; iElem < nElement; iElem++) {
443 su2double E = 1.0 / element[iElem].GetRef_Volume();
444 myNumerics->SetMeshElasticProperties(iElem, min(E,maxE));
445 }
446 break;
447
448 /*--- Stiffness inverse of the distance of the element to the closest wall. ---*/
449 case SOLID_WALL_DISTANCE: {
450 const su2double offset = config->GetDeform_StiffLayerSize();
451 if (fabs(offset) > 0.0) {
452 /*--- With prescribed layer of maximum stiffness (reaches max and holds). ---*/
453 su2double d0 = offset / MaxDistance;
454 su2double dmin = 1.0 / maxE;
455 su2double scale = 1.0 / (1.0 - d0);
456 for (unsigned long iElem = 0; iElem < nElement; iElem++) {
457 su2double E = 1.0 / max(dmin, (element[iElem].GetWallDistance() - d0)*scale);
458 myNumerics->SetMeshElasticProperties(iElem, E);
459 }
460 } else {
461 /*--- Without prescribed layer of maximum stiffness (may not reach max). ---*/
462 for (unsigned long iElem = 0; iElem < nElement; iElem++) {
463 su2double E = 1.0 / element[iElem].GetWallDistance();
464 myNumerics->SetMeshElasticProperties(iElem, min(E,maxE));
465 }
466 }
467 }
468 break;
469 }
470 }
471 END_SU2_OMP_PARALLEL
472
473 stiffness_set = true;
474

477void CMeshSolver::DeformMesh(CGeometry **geometry, CNumerics **numerics, CConfig *config){
478
479 if (multizone) nodes->Set_BGSSolution_k();
480
481 /*--- Capture a few MPI dependencies for AD. ---*/
482 geometry[MESH_0]->InitiateComms(geometry[MESH_0], config, COORDINATES);
483 geometry[MESH_0]->CompleteComms(geometry[MESH_0], config, COORDINATES);
484
485 InitiateComms(geometry[MESH_0], config, SOLUTION);
486 CompleteComms(geometry[MESH_0], config, SOLUTION);
487
488 InitiateComms(geometry[MESH_0], config, MESH_DISPLACEMENTS);
489 CompleteComms(geometry[MESH_0], config, MESH_DISPLACEMENTS);
490
491 /*--- Compute the stiffness matrix, no point recording because we clear the residual. ---*/
492
493 const bool wasActive = AD::BeginPassive();
494
495 Compute_StiffMatrix(geometry[MESH_0], numerics, config);
496
497 AD::EndPassive(wasActive);
498
499 /*--- Clear residual (loses AD info), we do not want an incremental solution. ---*/
500 SU2_OMP_PARALLEL {
501 LinSysRes.SetValZero();
502
503 if (time_domain && config->GetFSI_Simulation()) {
504 SU2_OMP_FOR_STAT(omp_chunk_size)
505 for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint)
506 for (unsigned short iDim = 0; iDim < nDim; ++iDim)
507 LinSysSol(iPoint, iDim) = nodes->GetSolution(iPoint, iDim);
508 END_SU2_OMP_FOR
509 }
510 }
511 END_SU2_OMP_PARALLEL
512
513 /*--- Impose boundary conditions (all of them are ESSENTIAL BC's - displacements). ---*/
514 SetBoundaryDisplacements(geometry[MESH_0], config, false);
515
516 /*--- Solve the linear system. ---*/
517 Solve_System(geometry[MESH_0], config);
518
519 SU2_OMP_PARALLEL {
520
521 /*--- Update the grid coordinates and cell volumes using the solution
522 of the linear system (usol contains the x, y, z displacements). ---*/
523 UpdateGridCoord(geometry[MESH_0], config);
524
525 /*--- Update the dual grid. ---*/
526 CGeometry::UpdateGeometry(geometry, config);
527
528 /*--- Check for failed deformation (negative volumes). ---*/
529 SetMinMaxVolume(geometry[MESH_0], config, true);
530
531 /*--- The Grid Velocity is only computed if the problem is time domain ---*/
532 if (time_domain && !config->GetFSI_Simulation())
533 ComputeGridVelocity(geometry, config);
534
535 }
536 END_SU2_OMP_PARALLEL
537
538 if (time_domain && config->GetFSI_Simulation()) {
539 ComputeGridVelocity_FromBoundary(geometry, numerics, config);
540 }

544void CMeshSolver::UpdateGridCoord(CGeometry *geometry, const CConfig *config){
545
546 /*--- Update the grid coordinates using the solution of the linear system ---*/
547
548 /*--- LinSysSol contains the absolute x, y, z displacements. ---*/
549 SU2_OMP_FOR_STAT(omp_chunk_size)
550 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++){
551 for (unsigned short iDim = 0; iDim < nDim; iDim++) {
552 /*--- Retrieve the displacement from the solution of the linear system ---*/
553 su2double val_disp = LinSysSol(iPoint, iDim);
554 /*--- Store the displacement of the mesh node ---*/
555 nodes->SetSolution(iPoint, iDim, val_disp);
556 /*--- Compute the current coordinate as Mesh_Coord + Displacement ---*/
557 su2double val_coord = nodes->GetMesh_Coord(iPoint,iDim) + val_disp;
558 /*--- Update the geometry container ---*/
559 geometry->nodes->SetCoord(iPoint, iDim, val_coord);
560 }
561 }
562 END_SU2_OMP_FOR
563
564 /*--- Communicate the updated displacements and mesh coordinates. ---*/
565 geometry->InitiateComms(geometry, config, COORDINATES);
566 geometry->CompleteComms(geometry, config, COORDINATES);
567
568}

570void CMeshSolver::ComputeGridVelocity_FromBoundary(CGeometry **geometry, CNumerics **numerics, CConfig *config){
571
572 if (config->GetnZone() == 1)
573 SU2_MPI::Error("It is not possible to compute grid velocity from boundary velocity for single zone problems.\n"
574 "MARKER_FLUID_LOAD should only be used for structural boundaries.", CURRENT_FUNCTION);
575
576 /*--- Compute the stiffness matrix, no point recording because we clear the residual. ---*/
577
578 const bool wasActive = AD::BeginPassive();
579
580 Compute_StiffMatrix(geometry[MESH_0], numerics, config);
581
582 AD::EndPassive(wasActive);
583
584 const su2double velRef = config->GetVelocity_Ref();
585 const su2double invVelRef = 1.0 / velRef;
586
587 /*--- Clear residual (loses AD info), we do not want an incremental solution. ---*/
588 SU2_OMP_PARALLEL {
589 LinSysRes.SetValZero();
590
591 SU2_OMP_FOR_STAT(omp_chunk_size)
592 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++)
593 for (unsigned short iDim = 0; iDim < nDim; iDim++)
594 LinSysSol(iPoint, iDim) = geometry[MESH_0]->nodes->GetGridVel(iPoint)[iDim] * velRef;
595 END_SU2_OMP_FOR
596 }
597 END_SU2_OMP_PARALLEL
598
599 /*--- Impose boundary conditions including boundary velocity ---*/
600 SetBoundaryDisplacements(geometry[MESH_0], config, true);
601
602 /*--- Solve the linear system. ---*/
603 Solve_System(geometry[MESH_0], config);
604
605 SU2_OMP_PARALLEL {
606 SU2_OMP_FOR_STAT(omp_chunk_size)
607 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++)
608 for (unsigned short iDim = 0; iDim < nDim; iDim++)
609 geometry[MESH_0]->nodes->SetGridVel(iPoint, iDim, LinSysSol(iPoint,iDim)*invVelRef);
610 END_SU2_OMP_FOR
611
612 for (auto iMGlevel = 1u; iMGlevel <= config->GetnMGLevels(); iMGlevel++)
613 geometry[iMGlevel]->SetRestricted_GridVelocity(geometry[iMGlevel-1]);
614 }
615 END_SU2_OMP_PARALLEL

619void CMeshSolver::ComputeGridVelocity(CGeometry **geometry, const CConfig *config) const {
620
621 /*--- Compute the velocity of each node. ---*/
622
623 const bool firstOrder = config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST;
624 const bool secondOrder = config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND;
625 const su2double invTimeStep = 1.0 / config->GetDelta_UnstTimeND();
626
627 SU2_OMP_FOR_STAT(omp_chunk_size)
628 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
629
630 /*--- Coordinates of the current point at n+1, n, & n-1 time levels. ---*/
631
632 const su2double* Disp_nM1 = nodes->GetSolution_time_n1(iPoint);
633 const su2double* Disp_n = nodes->GetSolution_time_n(iPoint);
634 const su2double* Disp_nP1 = nodes->GetSolution(iPoint);
635
636 /*--- Compute mesh velocity for this point with 1st or 2nd-order approximation. ---*/
637

639

643 else if (secondOrder)

645
646 geometry[MESH_0]->nodes->SetGridVel(iPoint, iDim, GridVel);
647 }
648 }
649 END_SU2_OMP_FOR
650
651 for (auto iMGlevel = 1u; iMGlevel <= config->GetnMGLevels(); iMGlevel++)
652 geometry[iMGlevel]->SetRestricted_GridVelocity(geometry[iMGlevel-1]);

672void CMeshSolver::SetBoundaryDisplacements(CGeometry *geometry, CConfig *config, bool velocity_transfer){
673
674 /* Surface motions are not applied during discrete adjoint runs as the corresponding
675 * boundary displacements are computed when loading the primal solution, and it
676 * would be complex to account for the incremental nature of these motions.
677 * The derivatives are still correct since the motion does not depend on the solution,
678 * but this means that (for now) we cannot get derivatives w.r.t. motion parameters. */
679
680 if (config->GetSurface_Movement(DEFORMING) && !config->GetDiscrete_Adjoint()) {
681 if (velocity_transfer)
682 SU2_MPI::Error("Forced motions are not compatible with FSI simulations.", CURRENT_FUNCTION);
683
684 if (rank == MASTER_NODE)
685 cout << endl << " Updating surface positions." << endl;
686
687 Surface_Translating(geometry, config, config->GetTimeIter());
688 Surface_Plunging(geometry, config, config->GetTimeIter());
689 Surface_Pitching(geometry, config, config->GetTimeIter());
690 Surface_Rotating(geometry, config, config->GetTimeIter());
691 }
692
693 unsigned short iMarker;
694
695 /*--- Impose zero displacements of all non-moving surfaces that are not MARKER_DEFORM_SYM_PLANE. ---*/
696 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
697 if ((config->GetMarker_All_Deform_Mesh(iMarker) == NO) &&
698 (config->GetMarker_All_Deform_Mesh_Sym_Plane(iMarker) == NO) &&
699 (config->GetMarker_All_Moving(iMarker) == NO) &&
700 (config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
701 (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE)) {
702
703 BC_Clamped(geometry, config, iMarker);
704 }
705 }
706
707 /*--- Impose displacement boundary conditions and symmetry. ---*/
708 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
709 if ((config->GetMarker_All_Deform_Mesh(iMarker) == YES) ||
710 (config->GetMarker_All_Moving(iMarker) == YES)) {
711
712 BC_Deforming(geometry, config, iMarker, velocity_transfer);
713 }
714 else if (config->GetMarker_All_Deform_Mesh_Sym_Plane(iMarker) == YES) {
715
716 BC_Sym_Plane(geometry, config, iMarker);
717 }
718 }
719
720 /*--- Clamp far away nodes according to deform limit. ---*/
721 if ((config->GetDeform_Stiffness_Type() == SOLID_WALL_DISTANCE) &&
722 (config->GetDeform_Limit() < MaxDistance)) {
723
724 const su2double limit = config->GetDeform_Limit() / MaxDistance;
725
726 for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint) {
727 if (nodes->GetWallDistance(iPoint) <= limit) continue;
728
729 su2double zeros[MAXNVAR] = {0.0};
730 nodes->SetSolution(iPoint, zeros);
731 LinSysSol.SetBlock(iPoint, zeros);
732 Jacobian.EnforceSolutionAtNode(iPoint, zeros, LinSysRes);
733 }
734 }
735
736 /*--- Clamp nodes outside of a given area. ---*/
737 if (config->GetHold_GridFixed()) {
738
739 auto MinCoordValues = config->GetHold_GridFixed_Coord();
740 auto MaxCoordValues = &config->GetHold_GridFixed_Coord()[3];
741
742 for (auto iPoint = 0ul; iPoint < geometry->GetnPoint(); iPoint++) {
743 auto Coord = geometry->nodes->GetCoord(iPoint);
744 for (auto iDim = 0; iDim < nDim; iDim++) {
745 if ((Coord[iDim] < MinCoordValues[iDim]) || (Coord[iDim] > MaxCoordValues[iDim])) {
746 su2double zeros[MAXNVAR] = {0.0};
747 nodes->SetSolution(iPoint, zeros);
748 LinSysSol.SetBlock(iPoint, zeros);
749 Jacobian.EnforceSolutionAtNode(iPoint, zeros, LinSysRes);
750 break;
751 }
752 }
753 }
754 }

764void CMeshSolver::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter, bool val_update_geo) {
765
766 /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
767
768 string filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
769
770 if (config->GetRead_Binary_Restart()) {
771 Read_SU2_Restart_Binary(geometry[MESH_0], config, filename);
772 } else {
773 Read_SU2_Restart_ASCII(geometry[MESH_0], config, filename);
774 }
775
776 /*--- Load data from the restart into correct containers. ---*/
777
778 unsigned long iPoint_Global, counter = 0;
779
780 for (iPoint_Global = 0; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
781
782 /*--- Retrieve local index. If this node from the restart file lives
783 on the current processor, we will load and instantiate the vars. ---*/
784
785 auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
786
787 if (iPoint_Local >= 0) {
788
789 /*--- We need to store this point's data, so jump to the correct
790 offset in the buffer of data from the restart file and load it. ---*/
791
792 auto index = counter*Restart_Vars[1];
793
794 for (unsigned short iDim = 0; iDim < nDim; iDim++){
795 /*--- Update the coordinates of the mesh ---*/
796 su2double curr_coord = Restart_Data[index+iDim];
797 /// TODO: "Double deformation" in multizone adjoint if this is set here?
798 /// In any case it should not be needed as deformation is called before other solvers
799 ///geometry[MESH_0]->nodes->SetCoord(iPoint_Local, iDim, curr_coord);
800
801 /*--- Store the displacements computed as the current coordinates
802 minus the coordinates of the reference mesh file ---*/
803 su2double displ = curr_coord - nodes->GetMesh_Coord(iPoint_Local, iDim);
804 nodes->SetSolution(iPoint_Local, iDim, displ);
805 }
806
807 /*--- Increment the overall counter for how many points have been loaded. ---*/
808 counter++;
809 }
810
811 }
812
813 /*--- Detect a wrong solution file ---*/
814
815 if (counter != nPointDomain) {
816 SU2_MPI::Error(string("The solution file ") + filename + string(" doesn't match with the mesh file!\n") +
817 string("It could be empty lines at the end of the file."), CURRENT_FUNCTION);
818 }
819
820 /*--- Communicate the loaded displacements. ---*/
821 solver[MESH_0][MESH_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
822 solver[MESH_0][MESH_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
823
824 /*--- Init the linear system solution. ---*/
825 for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
826 for (unsigned short iDim = 0; iDim < nDim; ++iDim) {
827 LinSysSol(iPoint, iDim) = nodes->GetSolution(iPoint, iDim);
828 }
829 }
830
831 /*--- For time-domain problems, we need to compute the grid velocities. ---*/
832 if (time_domain && !config->GetFSI_Simulation()) {
833 /*--- Update the old geometry (coordinates n and n-1) ---*/
834 RestartOldGeometry(geometry[MESH_0], config);
835
836 /*--- Once Displacement_n and Displacement_n1 are filled we can compute the Grid Velocity ---*/
837 ComputeGridVelocity(geometry, config);
838 }
839
840 /*--- Store the boundary displacements at the Bound_Disp variable. ---*/
841
842 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
843
844 if ((config->GetMarker_All_Deform_Mesh(iMarker) == YES) ||
845 (config->GetMarker_All_Moving(iMarker) == YES)) {
846
847 for (unsigned long iVertex = 0; iVertex < geometry[MESH_0]->nVertex[iMarker]; iVertex++) {
848
849 /*--- Get node index. ---*/
850 auto iNode = geometry[MESH_0]->vertex[iMarker][iVertex]->GetNode();
851
852 /*--- Set boundary solution. ---*/
853 nodes->SetBound_Disp(iNode, nodes->GetSolution(iNode));
854 }
855 }
856 }
857
858 /*--- Delete the class memory that is used to load the restart. ---*/
859
860 Restart_Vars = decltype(Restart_Vars){};
861 Restart_Data = decltype(Restart_Data){};
862}

968void CMeshSolver::Surface_Pitching(CGeometry *geometry, CConfig *config, unsigned long iter) {
969
970 su2double deltaT, time_new, time_old, Lref;
971 su2double Center[3] = {0.0}, VarCoord[3] = {0.0}, Omega[3] = {0.0}, Ampl[3] = {0.0}, Phase[3] = {0.0};
972 su2double VarCoordAbs[3] = {0.0};
973 su2double rotCoord[3] = {0.0}, r[3] = {0.0};
974 su2double rotMatrix[3][3] = {{0.0}};
975 su2double dtheta, dphi, dpsi;
976 const su2double DEG2RAD = PI_NUMBER/180.0;
977 unsigned short iMarker, jMarker, iDim;
978 unsigned long iPoint, iVertex;
979 string Marker_Tag, Moving_Tag;
980
981 /*--- Keep track of points that have been moved to avoid double
982 * deformation on points that appear on multiple markers. ---*/
983
984 const auto& VertexMap = static_cast<const CMeshBoundVariable*>(nodes)->GetVertexMap();
985 std::vector<uint8_t> iPointMoved(VertexMap.GetnVertex(), false);
986
987 /*--- Retrieve values from the config file ---*/
988
989 deltaT = config->GetDelta_UnstTimeND();
990 Lref = config->GetLength_Ref();
991
992 /*--- Compute delta time based on physical time step ---*/
993
994 time_new = iter*deltaT;
995 if (iter == 0) time_old = time_new;
996 else time_old = (iter-1)*deltaT;
997
998 /*--- Store displacement of each node on the pitching surface ---*/
999 /*--- Loop over markers and find the particular marker(s) (surface) to pitch ---*/
1000
1001 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
1002 if (config->GetMarker_All_Moving(iMarker) != YES) continue;
1003
1004 Marker_Tag = config->GetMarker_All_TagBound(iMarker);
1005
1006 for (jMarker = 0; jMarker < config->GetnMarker_Moving(); jMarker++) {
1007
1008 Moving_Tag = config->GetMarker_Moving_TagBound(jMarker);
1009
1010 if ((Marker_Tag != Moving_Tag) || (config->GetKind_SurfaceMovement(jMarker) != DEFORMING)) {
1011 continue;
1012 }
1013
1014 /*--- Pitching origin, frequency, and amplitude from config. ---*/
1015
1016 for (iDim = 0; iDim < 3; iDim++){
1017 Ampl[iDim] = config->GetMarkerPitching_Ampl(jMarker, iDim)*DEG2RAD;
1018 Omega[iDim] = config->GetMarkerPitching_Omega(jMarker, iDim)/config->GetOmega_Ref();
1019 Phase[iDim] = config->GetMarkerPitching_Phase(jMarker, iDim)*DEG2RAD;
1020 Center[iDim] = config->GetMarkerMotion_Origin(jMarker, iDim);
1021 }
1022 /*--- Print some information to the console. Be verbose at the first
1023 iteration only (mostly for debugging purposes). ---*/
1024 // Note that the MASTER_NODE might not contain all the markers being moved.
1025
1026 if (rank == MASTER_NODE) {
1027 cout << " Storing pitching displacement for marker: ";
1028 cout << Marker_Tag << "." << endl;
1029 if (iter == 0) {
1030 cout << " Pitching frequency: (" << Omega[0] << ", " << Omega[1];
1031 cout << ", " << Omega[2] << ") rad/s about origin: (" << Center[0];
1032 cout << ", " << Center[1] << ", " << Center[2] << ")." << endl;
1033 cout << " Pitching amplitude about origin: (" << Ampl[0]/DEG2RAD;
1034 cout << ", " << Ampl[1]/DEG2RAD << ", " << Ampl[2]/DEG2RAD;
1035 cout << ") degrees."<< endl;
1036 cout << " Pitching phase lag about origin: (" << Phase[0]/DEG2RAD;
1037 cout << ", " << Phase[1]/DEG2RAD <<", "<< Phase[2]/DEG2RAD;
1038 cout << ") degrees."<< endl;
1039 }
1040 }
1041
1042 /*--- Compute delta change in the angle about the x, y, & z axes. ---*/
1043
1044 dtheta = -Ampl[0]*(sin(Omega[0]*time_new + Phase[0])
1045 - sin(Omega[0]*time_old + Phase[0]));
1046 dphi = -Ampl[1]*(sin(Omega[1]*time_new + Phase[1])
1047 - sin(Omega[1]*time_old + Phase[1]));
1048 dpsi = -Ampl[2]*(sin(Omega[2]*time_new + Phase[2])
1049 - sin(Omega[2]*time_old + Phase[2]));
1050
1051 /*--- Compute rotation matrix. ---*/
1052
1053 RotationMatrix(dtheta, dphi, dpsi, rotMatrix);
1054
1055 /*--- Apply rotation to the vertices. ---*/
1056
1057 for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
1058
1059 /*--- Index and coordinates of the current point accounting for other
1060 * motions that may be applied, e.g. plunging. ---*/
1061
1062 iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
1063
1064 /*--- Avoid moving points twice. ---*/
1065 auto vertexIndex = iPoint;
1066 VertexMap.GetVertexIndex(vertexIndex);
1067 if (iPointMoved[vertexIndex]) continue;
1068 iPointMoved[vertexIndex] = true;
1069
1070 su2double Coord[3] = {0.0};
1071 for (iDim = 0; iDim < nDim; ++iDim) {
1072 Coord[iDim] = nodes->GetMesh_Coord(iPoint, iDim) + nodes->GetBound_Disp(iPoint, iDim);
1073 }
1074
1075 /*--- Calculate non-dim. position from rotation center ---*/
1076
1077 for (iDim = 0; iDim < nDim; iDim++)
1078 r[iDim] = (Coord[iDim]-Center[iDim])/Lref;
1079
1080 /*--- Compute transformed point coordinates ---*/
1081
1082 Rotate(rotMatrix, Center, r, rotCoord);
1083
1084 /*--- Calculate delta change in the x, y, & z directions ---*/
1085 for (iDim = 0; iDim < nDim; iDim++)
1086 VarCoord[iDim] = (rotCoord[iDim]-Coord[iDim])/Lref;
1087
1088 /*--- Set node displacement for volume deformation ---*/
1089
1090 for (iDim = 0; iDim < nDim; iDim++)
1091 VarCoordAbs[iDim] = nodes->GetBound_Disp(iPoint, iDim) + VarCoord[iDim];
1092
1093 nodes->SetBound_Disp(iPoint, VarCoordAbs);
1094 }
1095 }
1096 }
1097 /*--- For pitching we don't update the motion origin and moment reference origin. ---*/
1098
