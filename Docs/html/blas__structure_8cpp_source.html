55/* Dense matrix multiplication, gemm functionality. */
56void CBlasStructure::gemm(const int M, const int N, const int K, const su2double* A, const su2double* B, su2double* C,
57 const CConfig* config) {
58 /* Initialize the variable for the timing, if profiling is active. */
59#ifdef PROFILE
60 double timeGemm;
61 if (config) config->GEMM_Tick(&timeGemm);
62#endif
63
64#if (defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE)) || \
65 !(defined(HAVE_LIBXSMM) || defined(HAVE_MKL) || defined(HAVE_BLAS))
66 /* Native implementation of the matrix product. This optimized implementation
67 assumes that the matrices are in column major order. This can be
68 accomplished by swapping N and M and A and B. This implementation is based
69 on https://github.com/flame/how-to-optimize-gemm. */
70 gemm_imp(N, M, K, B, A, C);
71
72#else
73#ifdef HAVE_LIBXSMM
74
75 /* The gemm function of libxsmm is used to carry out the multiplication.
76 Note that libxsmm_gemm expects the matrices in column major order. That's
77 why the in the calling sequence A and B and M and N are reversed. */
78 su2double alpha = 1.0;
79 su2double beta = 0.0;
80 char trans = 'N';
81
82 libxsmm_dgemm(&trans, &trans, &N, &M, &K, &alpha, B, &N, A, &K, &beta, C, &N);
83
84#else // MKL and BLAS
85
86 /* The standard blas routine dgemm is used for the multiplication.
87 Call dgemm without transposing the matrices. In that case dgemm expects
88 the matrices in column major order, see the comments for libxsmm. */
89 su2double alpha = 1.0;
90 su2double beta = 0.0;
91 char trans = 'N';
92
93 dgemm_(&trans, &trans, &N, &M, &K, &alpha, B, &N, A, &K, &beta, C, &N);
94
95#endif
96#endif
97
98 /* Store the profiling information, if needed. */
99#ifdef PROFILE
100 if (config) config->GEMM_Tock(timeGemm, M, N, K);
101#endif
102}

104/* Dense matrix vector multiplication, gemv functionality. */
105void CBlasStructure::gemv(const int M, const int N, const su2double* A, const su2double* x, su2double* y) {
106#if (defined(HAVE_BLAS) || defined(HAVE_MKL)) && !(defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE))
107
108 /* The standard blas routine dgemv is used for the multiplication.
109 Note that dgemv expects the matrices in column major order, while
110 A is in row major order. This can be solved by using the transpose
111 and switching M and N. */
112 su2double alpha = 1.0;
113 su2double beta = 0.0;
114 int inc = 1;
115 char trans = 'T';
116
117 dgemv_(&trans, &N, &M, &alpha, A, &N, x, &inc, &beta, y, &inc);
118
119#else
120
121 /* Native implementation of the matix vector product.
122 Initialize the elements of y to zero. */
123 for (int i = 0; i < M; ++i) y[i] = 0.0;
124
125 /* Carry out the matrix vector product. */
126 for (int k = 0; k < M; ++k) {
127 const su2double* AA = A + k * N;
128 for (int l = 0; l < N; ++l) y[k] += AA[l] * x[l];
129 }
130
131#endif
132}

138#define A(i, j) a[(j)*lda + (i)]
139#define B(i, j) b[(j)*ldb + (i)]
140#define C(i, j) c[(j)*ldc + (i)]

142/* Function, which perform the implementation of the gemm functionality. */
143void CBlasStructure::gemm_imp(const int m, const int n, const int k, const su2double* a, const su2double* b,
144 su2double* c) {
145 /* Initialize the elements of c to zero. */
146 for (int i = 0; i < (m * n); ++i) c[i] = 0.0;
147
148 /* Set the leading dimensions of the three matrices. */
149 const int lda = m;
150 const int ldb = k;
151 const int ldc = m;
152
153 /* The full matrix multiplication is split in several blocks.
154 Loop over these blocks. */
155 for (int p = 0; p < k; p += kc) {
156 int pb = min(k - p, kc);
157 for (int j = 0; j < n; j += nc) {
158 int jb = min(n - j, nc);
159 for (int i = 0; i < m; i += mc) {
160 int ib = min(m - i, mc);
161
162 /* Carry out the multiplication for this block. */
163 gemm_inner(ib, jb, pb, &A(i, p), lda, &B(p, j), ldb, &C(i, j), ldc);
164 }
165 }
166 }
167}

171void CBlasStructure::gemm_inner(int m, int n, int k, const su2double* a, int lda, const su2double* b, int ldb,

173 /* Carry out the multiplication for this block. At the

175 gemm_arbitrary(m, n, k, a, lda, b, ldb, c, ldc);

178/* Naive gemm implementation to handle arbitrary sized matrices. */
179void CBlasStructure::gemm_arbitrary(int m, int n, int k, const su2double* a, int lda, const su2double* b, int ldb,
180 su2double* c, int ldc) {
181 /* The order of these loops is tuned for column-major matrices. */
182 for (int p = 0; p < k; p++) {
183 for (int j = 0; j < n; j++) {
184 for (int i = 0; i < m; i++) {
185 C(i, j) += A(i, p) * B(p, j);
186 }
187 }
188 }
189}
