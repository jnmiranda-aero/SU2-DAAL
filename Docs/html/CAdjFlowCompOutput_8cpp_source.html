34 CAdjFlowCompOutput::CAdjFlowCompOutput(CConfig *config, unsigned short nDim) : CAdjFlowOutput(config, nDim) {
35 
36  /*--- Set the default history fields if nothing is set in the config file ---*/
37 
38  if (nRequestedHistoryFields == 0) {
39  RequestCommonHistory(config->GetTime_Domain());
40  requestedHistoryFields.emplace_back("SENSITIVITY");
41  nRequestedHistoryFields = requestedHistoryFields.size();
42  }
43 
44  if (nRequestedScreenFields == 0) {
45  if (config->GetTime_Domain()) requestedScreenFields.emplace_back("TIME_ITER");
46  if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
47  requestedScreenFields.emplace_back("INNER_ITER");
48  requestedScreenFields.emplace_back("RMS_ADJ_DENSITY");
49  requestedScreenFields.emplace_back("RMS_ADJ_MOMENTUM-X");
50  requestedScreenFields.emplace_back("SENS_GEO");
51  requestedScreenFields.emplace_back("SENS_AOA");
52  nRequestedScreenFields = requestedScreenFields.size();
53  }
54 
55  if (nRequestedVolumeFields == 0) {
56  requestedVolumeFields.emplace_back("COORDINATES");
57  requestedVolumeFields.emplace_back("SOLUTION");
58  requestedVolumeFields.emplace_back("SENSITIVITY");
59  nRequestedVolumeFields = requestedVolumeFields.size();
60  }
61 
62  if (find(requestedVolumeFields.begin(), requestedVolumeFields.end(), string("SENSITIVITY")) == requestedVolumeFields.end()) {
63  requestedVolumeFields.emplace_back("SENSITIVITY");
64  nRequestedVolumeFields ++;
65  }
66 
67  stringstream ss;
68  ss << "Zone " << config->GetiZone() << " (Adj. Comp. Fluid)";
69  multiZoneHeaderString = ss.str();
70 
71  /*--- Set the volume filename --- */
72 
73  volumeFilename = config->GetAdj_FileName();
74 
75  /*--- Set the surface filename --- */
76 
77  surfaceFilename = config->GetSurfAdjCoeff_FileName();
78 
79  /*--- Set the restart filename --- */
80 
81  restartFilename = config->GetRestart_AdjFileName();
82 
83  /*--- Add the obj. function extension --- */
84 
85  restartFilename = config->GetObjFunc_Extension(restartFilename);
86 
87  /*--- Set the default convergence field --- */
88 
89  if (convFields.empty() ) convFields.emplace_back("RMS_ADJ_DENSITY");
90 
91 }

175 void CAdjFlowCompOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
176 
177  CSolver* adjflow_solver = solver[ADJFLOW_SOL];
178  CSolver* mesh_solver = solver[MESH_SOL];
179 
180  SetHistoryOutputValue("RMS_ADJ_DENSITY", log10(adjflow_solver->GetRes_RMS(0)));
181  SetHistoryOutputValue("RMS_ADJ_MOMENTUM-X", log10(adjflow_solver->GetRes_RMS(1)));
182  SetHistoryOutputValue("RMS_ADJ_MOMENTUM-Y", log10(adjflow_solver->GetRes_RMS(2)));
183  if (geometry->GetnDim() == 3) {
184  SetHistoryOutputValue("RMS_ADJ_MOMENTUM-Z", log10(adjflow_solver->GetRes_RMS(3)));
185  SetHistoryOutputValue("RMS_ADJ_ENERGY", log10(adjflow_solver->GetRes_RMS(4)));
186  } else {
187  SetHistoryOutputValue("RMS_ADJ_ENERGY", log10(adjflow_solver->GetRes_RMS(3)));
188  }
189 
190  SetHistoryOutputValue("MAX_ADJ_DENSITY", log10(adjflow_solver->GetRes_Max(0)));
191  SetHistoryOutputValue("MAX_ADJ_MOMENTUM-X", log10(adjflow_solver->GetRes_Max(1)));
192  SetHistoryOutputValue("MAX_ADJ_MOMENTUM-Y", log10(adjflow_solver->GetRes_Max(2)));
193  if (geometry->GetnDim() == 3) {
194  SetHistoryOutputValue("MAX_ADJ_MOMENTUM-Z", log10(adjflow_solver->GetRes_Max(3)));
195  SetHistoryOutputValue("MAX_ADJ_ENERGY", log10(adjflow_solver->GetRes_Max(4)));
196  } else {
197  SetHistoryOutputValue("MAX_ADJ_ENERGY", log10(adjflow_solver->GetRes_Max(3)));
198  }
199 
200  if (multiZone) {
201  SetHistoryOutputValue("BGS_ADJ_DENSITY", log10(adjflow_solver->GetRes_BGS(0)));
202  SetHistoryOutputValue("BGS_ADJ_MOMENTUM-X", log10(adjflow_solver->GetRes_BGS(1)));
203  SetHistoryOutputValue("BGS_ADJ_MOMENTUM-Y", log10(adjflow_solver->GetRes_BGS(2)));
204  if (geometry->GetnDim() == 3) {
205  SetHistoryOutputValue("BGS_ADJ_MOMENTUM-Z", log10(adjflow_solver->GetRes_BGS(3)));
206  SetHistoryOutputValue("BGS_ADJ_ENERGY", log10(adjflow_solver->GetRes_BGS(4)));
207  } else {
208  SetHistoryOutputValue("BGS_ADJ_ENERGY", log10(adjflow_solver->GetRes_BGS(3)));
209  }
210  }
211 
212  SetHistoryOutputValue("SENS_GEO", adjflow_solver->GetTotal_Sens_Geo());
213  SetHistoryOutputValue("SENS_AOA", adjflow_solver->GetTotal_Sens_AoA() * PI_NUMBER / 180.0);
214  SetHistoryOutputValue("SENS_MACH", adjflow_solver->GetTotal_Sens_Mach());
215  SetHistoryOutputValue("SENS_PRESS", adjflow_solver->GetTotal_Sens_Press());
216  SetHistoryOutputValue("SENS_TEMP", adjflow_solver->GetTotal_Sens_Temp());
217 
218  SetHistoryOutputValue("LINSOL_ITER", adjflow_solver->GetIterLinSolver());
219  SetHistoryOutputValue("LINSOL_RESIDUAL", log10(adjflow_solver->GetResLinSolver()));
220 
221  if (config->GetDeform_Mesh()) {
222  SetHistoryOutputValue("DEFORM_ITER", mesh_solver->System.GetIterations());
223  SetHistoryOutputValue("DEFORM_RESIDUAL", log10(mesh_solver->System.GetResidual()));
224  }
225 
226  LoadHistoryDataAdjScalar(config, solver);
227 
228  ComputeSimpleCustomOutputs(config);
229 }

231 void CAdjFlowCompOutput::SetVolumeOutputFields(CConfig *config) {
232 
233  // Grid coordinates
234  AddVolumeOutput("COORD-X", "x", "COORDINATES", "x-component of the coordinate vector");
235  AddVolumeOutput("COORD-Y", "y", "COORDINATES", "y-component of the coordinate vector");
236  if (nDim == 3)
237  AddVolumeOutput("COORD-Z", "z", "COORDINATES", "z-component of the coordinate vector");
238 
239  /// BEGIN_GROUP: SOLUTION, DESCRIPTION: The SOLUTION variables of the adjoint solver.
240  /// DESCRIPTION: Adjoint density.
241  AddVolumeOutput("ADJ_DENSITY", "Adjoint_Density", "SOLUTION", "Adjoint density");
242  /// DESCRIPTION: Adjoint momentum x-component.
243  AddVolumeOutput("ADJ_MOMENTUM-X", "Adjoint_Momentum_x", "SOLUTION", "x-component of the adjoint momentum vector");
244  /// DESCRIPTION: Adjoint momentum y-component.
245  AddVolumeOutput("ADJ_MOMENTUM-Y", "Adjoint_Momentum_y", "SOLUTION", "y-component of the adjoint momentum vector");
246  if (nDim == 3)

252  SetVolumeOutputFieldsAdjScalarSolution(config);
253  /// END_GROUP
254 
255  /// BEGIN_GROUP: RESIDUAL, DESCRIPTION: Residuals of the SOLUTION variables.
256  /// DESCRIPTION: Residual of the adjoint density.

258  /// DESCRIPTION: Residual of the adjoint momentum x-component.

260  /// DESCRIPTION: Residual of the adjoint momentum y-component.

262  if (nDim == 3)
263  /// DESCRIPTION: Residual of the adjoint momentum z-component.
264  AddVolumeOutput("RES_ADJ_MOMENTUM-Z", "Residual_Adjoint_Momentum_z", "RESIDUAL", "Residual of the adjoint z-momentum");
265  /// DESCRIPTION: Residual of the adjoint energy.
266  AddVolumeOutput("RES_ADJ_ENERGY", "Residual_Adjoint_Energy", "RESIDUAL", "Residual of the adjoint energy");
267 
268  SetVolumeOutputFieldsAdjScalarResidual(config);
269  /// END_GROUP
270 
271  /// BEGIN_GROUP: SENSITIVITY, DESCRIPTION: Geometrical sensitivities of the current objective function.
272  /// DESCRIPTION: Sensitivity x-component.
273  AddVolumeOutput("SENSITIVITY-X", "Sensitivity_x", "SENSITIVITY", "x-component of the sensitivity vector");
274  /// DESCRIPTION: Sensitivity y-component.
275  AddVolumeOutput("SENSITIVITY-Y", "Sensitivity_y", "SENSITIVITY", "y-component of the sensitivity vector");
276  if (nDim == 3)
277  /// DESCRIPTION: Sensitivity z-component.
278  AddVolumeOutput("SENSITIVITY-Z", "Sensitivity_z", "SENSITIVITY", "z-component of the sensitivity vector");
279  /// DESCRIPTION: Sensitivity in normal direction.
280  AddVolumeOutput("SENSITIVITY", "Surface_Sensitivity", "SENSITIVITY", "sensitivity in normal direction");
281  /// END_GROUP

285 void CAdjFlowCompOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint) {
286 
287  CVariable* Node_AdjFlow = solver[ADJFLOW_SOL]->GetNodes();
288  CPoint* Node_Geo = geometry->nodes;
289 
290  SetVolumeOutputValue("COORD-X", iPoint, Node_Geo->GetCoord(iPoint, 0));
291  SetVolumeOutputValue("COORD-Y", iPoint, Node_Geo->GetCoord(iPoint, 1));
292  if (nDim == 3)
293  SetVolumeOutputValue("COORD-Z", iPoint, Node_Geo->GetCoord(iPoint, 2));
294 
295  SetVolumeOutputValue("ADJ_DENSITY", iPoint, Node_AdjFlow->GetSolution(iPoint, 0));
296  SetVolumeOutputValue("ADJ_MOMENTUM-X", iPoint, Node_AdjFlow->GetSolution(iPoint, 1));
297  SetVolumeOutputValue("ADJ_MOMENTUM-Y", iPoint, Node_AdjFlow->GetSolution(iPoint, 2));
298  if (nDim == 3) {
299  SetVolumeOutputValue("ADJ_MOMENTUM-Z", iPoint, Node_AdjFlow->GetSolution(iPoint, 3));
300  SetVolumeOutputValue("ADJ_ENERGY", iPoint, Node_AdjFlow->GetSolution(iPoint, 4));
301  } else {
302  SetVolumeOutputValue("ADJ_ENERGY", iPoint, Node_AdjFlow->GetSolution(iPoint, 3));
303  }
304 
305  // Residuals
306  SetVolumeOutputValue("RES_ADJ_DENSITY", iPoint, Node_AdjFlow->GetSolution(iPoint, 0) - Node_AdjFlow->GetSolution_Old(iPoint, 0));
307  SetVolumeOutputValue("RES_ADJ_MOMENTUM-X", iPoint, Node_AdjFlow->GetSolution(iPoint, 1) - Node_AdjFlow->GetSolution_Old(iPoint, 1));
308  SetVolumeOutputValue("RES_ADJ_MOMENTUM-Y", iPoint, Node_AdjFlow->GetSolution(iPoint, 2) - Node_AdjFlow->GetSolution_Old(iPoint, 2));
309  if (nDim == 3) {
310  SetVolumeOutputValue("RES_ADJ_MOMENTUM-Z", iPoint, Node_AdjFlow->GetSolution(iPoint, 3) - Node_AdjFlow->GetSolution_Old(iPoint, 3));
311  SetVolumeOutputValue("RES_ADJ_ENERGY", iPoint, Node_AdjFlow->GetSolution(iPoint, 4) - Node_AdjFlow->GetSolution_Old(iPoint, 4));
312  } else {
313  SetVolumeOutputValue("RES_ADJ_ENERGY", iPoint, Node_AdjFlow->GetSolution(iPoint, 3) - Node_AdjFlow->GetSolution_Old(iPoint, 3));
314  }
315 
316  SetVolumeOutputValue("SENSITIVITY-X", iPoint, Node_AdjFlow->GetSensitivity(iPoint, 0));
317  SetVolumeOutputValue("SENSITIVITY-Y", iPoint, Node_AdjFlow->GetSensitivity(iPoint, 1));
318  if (nDim == 3)
319  SetVolumeOutputValue("SENSITIVITY-Z", iPoint, Node_AdjFlow->GetSensitivity(iPoint, 2));
320 
321  LoadVolumeDataAdjScalar(config, solver, iPoint);
322 }
