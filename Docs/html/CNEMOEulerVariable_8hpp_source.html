253 /*!
254 * \brief Get the projected velocity in a unitary vector direction (compressible solver).
255 * \param[in] val_vector - Direction of projection.
256 * \return Value of the projected velocity.
257 */
258 inline su2double GetProjVel(unsigned long iPoint, const su2double *val_vector) const final {
259 su2double ProjVel = 0.0;
260 for (unsigned long iDim = 0; iDim < nDim; iDim++)
261 ProjVel += Primitive(iPoint,VEL_INDEX+iDim)*val_vector[iDim];
262 return ProjVel;
263 }

265 /*!
266 * \brief Set the velocity vector from the solution.
267 * \param[in] val_velocity - Pointer to the velocity.
268 */
269 inline void SetVelocity(unsigned long iPoint) final {
270 Velocity2(iPoint) = 0.0;
271 for (unsigned long iDim = 0; iDim < nDim; iDim++) {
272 Primitive(iPoint,VEL_INDEX+iDim) = Solution(iPoint,nSpecies+iDim) / Primitive(iPoint,RHO_INDEX);
273 Velocity2(iPoint) += pow(Primitive(iPoint,VEL_INDEX+iDim),2);
274 }
275 }
