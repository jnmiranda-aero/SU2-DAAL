42/*!
43 * \brief Base vector expression class.
44 * \ingroup BLAS
45 * \param[in] Derived - The class that inherits from this one to use the expressions.
46 * \param[in] Scalar - Associated scalar type, prevents implicit conversions between exprs.
47 * \note Derived classes must implement operator[], and at least operator= with
48 * expressions (that is when they are evaluated). They must also contain a constexpr
49 * boolean "StoreAsRef", indicating whether to store them by value (false) or by
50 * reference (true), when composing expressions.
51 * Expression classes need to be stored by value to allow nested expressions to
52 * propagate correctly (i.e. if "Scalar" has its own expression templates).
53 * Vector classes should be stored by reference to avoid copies, especially if they
54 * allocate memory dynamically.
55 */
56template <class Derived, class Scalar>
57class CVecExpr {
58 public:
59 /*!
60 * \brief Cast the expression to Derived, usually to allow evaluation via operator[].
61 */
62 FORCEINLINE const Derived& derived() const { return static_cast<const Derived&>(*this); }
63
64 // Allowed from C++14, allows nested expression propagation without
65 // manually calling derived() on the expression being evaluated.
66 // FORCEINLINE auto operator[] (size_t i) const { return derived()[i]; }
67};

69/*!
70 * \brief Expression class to broadcast a scalar value. Allows implementing
71 * "vector-scalar" operations re-using "vector-vector" expressions.
72 */
73template <class Scalar>
74class Bcast : public CVecExpr<Bcast<Scalar>, Scalar> {
75 Scalar x;
76
77 public:
78 static constexpr bool StoreAsRef = false;
79 FORCEINLINE Bcast(const Scalar& x_) : x(x_) {}
80 FORCEINLINE const Scalar& operator[](size_t) const { return x; }
81};

83/*!
84 * \brief std::decay_t from C++14, used to allow implicit conversions
85 * between scalar types, e.g. "CVecExpr<U,double>" + "int/double/etc.".
86 */
87template <class T>
88using decay_t = typename std::decay<T>::type;

90/*! \brief std::remove_reference_t from C++14, removes references from some type. */
91template <class T>
92using remove_reference_t = typename std::remove_reference<T>::type;

94/*! \brief Mechanism to conditionally (based on "StoreAsRef") add lvalue reference to a type. */
95template <class T, bool>
96struct add_lref_if {
97 using type = remove_reference_t<T>;
98};
99template <class T>
100struct add_lref_if<T, true> {
101 using type = remove_reference_t<T>&;
102};
103template <class T>
104using store_t = typename add_lref_if<T, T::StoreAsRef>::type;

120/*--- Macro to create expression classes (EXPR) and overloads (FUN) for unary
121 * functions, based on their coefficient-wise implementation (IMPL). ---*/
122
123#define MAKE_UNARY_FUN(FUN, EXPR, IMPL) \
124 /*!--- Expression class. ---*/ \
125 template <class U, class Scalar> \
126 class EXPR : public CVecExpr<EXPR<U, Scalar>, Scalar> { \
127 store_t<const U> u; \
128 \
129 public: \
130 static constexpr bool StoreAsRef = false; \
131 FORCEINLINE EXPR(const U& u_) : u(u_) {} \
132 FORCEINLINE auto operator[](size_t i) const RETURNS(IMPL(u[i])) \
133 }; \
134 /*!--- Function overload, returns an expression object. ---*/ \
135 template <class U, class S> \
136 FORCEINLINE auto FUN(const CVecExpr<U, S>& u) RETURNS(EXPR<U, S>(u.derived()))
137
138#define sign_impl(x) Scalar(1 - 2 * (x < 0))
139MAKE_UNARY_FUN(operator-, minus_, -)
140MAKE_UNARY_FUN(abs, abs_, math::abs)
141MAKE_UNARY_FUN(sqrt, sqrt_, math::sqrt)
142MAKE_UNARY_FUN(sign, sign_, sign_impl)
143#undef sign_impl

145#undef MAKE_UNARY_FUN

151 /*!--- Expression class. ---*/ \
152 template <class U, class V, class Scalar> \
153 class EXPR : public CVecExpr<EXPR<U, V, Scalar>, Scalar> { \

157 public: \

174/*--- std::max/min have issues (because they return by reference).
175 * fmin and fmax return by value and thus are fine, but they would force
176 * conversions to double, to avoid that we provide integer overloads.
177 * We use int32/64 instead of int/long to avoid issues with Windows,
178 * where long is 32 bits (instead of 64 bits). ---*/
179
180#define MAKE_FMINMAX_OVERLOADS(TYPE) \
181 FORCEINLINE TYPE fmax(TYPE a, TYPE b) { return a < b ? b : a; } \
182 FORCEINLINE TYPE fmin(TYPE a, TYPE b) { return a < b ? a : b; }
183MAKE_FMINMAX_OVERLOADS(int32_t)
184MAKE_FMINMAX_OVERLOADS(int64_t)
185MAKE_FMINMAX_OVERLOADS(uint32_t)
186MAKE_FMINMAX_OVERLOADS(uint64_t)
187/*--- Make the float and double versions of fmin/max available in this
188 * namespace to avoid ambiguous overloads. ---*/
189using std::fmax;
190using std::fmin;
191#undef MAKE_FMINMAX_OVERLOADS

214/*--- Relational operators need to be cast to the scalar type to allow vectorization.
215 * TO_PASSIVE is used to convert active scalars to passive, which CoDi will then capture
216 * by value in its expressions, and thus dangling references are avoided. No AD info
217 * is lost since these operators are non-differentiable. ---*/
218
219#define TO_PASSIVE(IMPL) SU2_TYPE::Passive<Scalar>::Value(IMPL)
220#define le_impl(a, b) TO_PASSIVE(a <= b)
221#define ge_impl(a, b) TO_PASSIVE(a >= b)
222#define eq_impl(a, b) TO_PASSIVE(a == b)
223#define ne_impl(a, b) TO_PASSIVE(a != b)
224#define lt_impl(a, b) TO_PASSIVE(a < b)
225#define gt_impl(a, b) TO_PASSIVE(a > b)
226MAKE_BINARY_FUN(operator<=, le_, le_impl)
227MAKE_BINARY_FUN(operator>=, ge_, ge_impl)
228MAKE_BINARY_FUN(operator==, eq_, eq_impl)
229MAKE_BINARY_FUN(operator!=, ne_, ne_impl)
230MAKE_BINARY_FUN(operator<, lt_, lt_impl)
231MAKE_BINARY_FUN(operator>, gt_, gt_impl)

234#undef ge_impl
235#undef eq_impl
236#undef ne_impl
237#undef lt_impl
