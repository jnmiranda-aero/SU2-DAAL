487  }

531  }

656 void CAdjEulerSolver::SetForceProj_Vector(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
657 
658  su2double *ForceProj_Vector, x = 0.0, y = 0.0, z = 0.0, *Normal, CD, CL, Cp, CpTarget,
659  CT, CQ, x_origin, y_origin, z_origin, WDrag, Area, invCD, CLCD2, invCQ, CTRCQ2;
660  unsigned short iMarker,jMarker,iMarker_Monitoring, iDim;
661  unsigned long iVertex, iPoint;
662  string Marker_Tag, Monitoring_Tag;
663  su2double Weight_ObjFunc=1.0;
664  su2double *ForceProj_Vector2;
665 
666  su2double Alpha = (config->GetAoA()*PI_NUMBER)/180.0;
667  su2double Beta = (config->GetAoS()*PI_NUMBER)/180.0;
668  su2double RefLength = config->GetRefLength();
669  const auto RefOriginMoment = config->GetRefOriginMoment(0);
670  su2double dCD_dCL = config->GetdCD_dCL();
671  su2double dCMx_dCL = config->GetdCMx_dCL();
672  su2double dCMy_dCL = config->GetdCMy_dCL();
673  su2double dCMz_dCL = config->GetdCMz_dCL();
674  bool Fixed_CL = config->GetFixed_CL_Mode();
675 
676  ForceProj_Vector = new su2double[nDim];
677 
678  /*--- Compute coefficients needed for objective function evaluation. ---*/
679 
680  CD = solver_container[FLOW_SOL]->GetTotal_CD();
681  CL = solver_container[FLOW_SOL]->GetTotal_CL();
682  CT = solver_container[FLOW_SOL]->GetTotal_CT();
683  CQ = solver_container[FLOW_SOL]->GetTotal_CQ();
684  invCD = 1.0/CD; CLCD2 = CL/(CD*CD);
685  invCQ = 1.0/CQ; CTRCQ2 = CT/(RefLength*CQ*CQ);
686 
687  x_origin = RefOriginMoment[0]; y_origin = RefOriginMoment[1]; z_origin = RefOriginMoment[2];
688 
689  /*--- Evaluate the boundary condition coefficients,
690  Since there may be more than one objective per marker, first we have to set all Force projection vectors to 0 ---*/
691 
692  for (iMarker = 0; iMarker<nMarker; iMarker++) {
693  if ((config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) &&
694  (config->GetMarker_All_Monitoring(iMarker) == YES))
695  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
696  for (iDim=0; iDim<nDim; iDim++)
697  ForceProj_Vector[iDim]=0.0;
698  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
699  nodes->SetForceProj_Vector(iPoint,ForceProj_Vector);
700  }
701  }
702 
703  /*--- Find the matching iMarker ---*/
704 
705  for (iMarker_Monitoring = 0; iMarker_Monitoring < config->GetnMarker_Monitoring(); iMarker_Monitoring++) {
706  Weight_ObjFunc = config->GetWeight_ObjFunc(iMarker_Monitoring);
707  Monitoring_Tag = config->GetMarker_Monitoring_TagBound(iMarker_Monitoring);
708  for (jMarker=0; jMarker<nMarker; jMarker++) {
709  Marker_Tag = config->GetMarker_All_TagBound(jMarker);
710  if (Monitoring_Tag==Marker_Tag)
711  iMarker = jMarker;
712  }
713 
714 
715  if ((iMarker<nMarker) && (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) &&
716  (config->GetMarker_All_Monitoring(iMarker) == YES)) {
717  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
718 
719  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
720 
721  x = geometry->nodes->GetCoord(iPoint, 0);
722  y = geometry->nodes->GetCoord(iPoint, 1);
723  if (nDim == 3) z = geometry->nodes->GetCoord(iPoint, 2);
724 
725  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
726  ForceProj_Vector2 = nodes->GetForceProj_Vector(iPoint);
727  for (iDim=0; iDim<nDim;iDim++)
728  ForceProj_Vector[iDim]=ForceProj_Vector2[iDim];
729 
730  switch (config->GetKind_ObjFunc(iMarker_Monitoring)) {
731  case DRAG_COEFFICIENT :
732  if (nDim == 2) {
733 
734  ForceProj_Vector[0] += Weight_ObjFunc*cos(Alpha);
735  ForceProj_Vector[1] += Weight_ObjFunc*sin(Alpha);
736 
737  /*--- Modification to run at a fixed CL and CM value ---*/
738 
739  if (Fixed_CL) { ForceProj_Vector[0] += dCD_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= dCD_dCL*Weight_ObjFunc*cos(Alpha); }
740 
741  }
742  if (nDim == 3) {
743 
744  ForceProj_Vector[0] += Weight_ObjFunc*cos(Alpha)*cos(Beta);
745  ForceProj_Vector[1] += Weight_ObjFunc*sin(Beta);
746  ForceProj_Vector[2] += Weight_ObjFunc*sin(Alpha)*cos(Beta);
747 
748  /*--- Modification to run at a fixed CL value ---*/
749 
750  if (Fixed_CL) { ForceProj_Vector[0] += dCD_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= 0.0; ForceProj_Vector[2] -= dCD_dCL*Weight_ObjFunc*cos(Alpha); }
751 
752  }
753  break;
754  case LIFT_COEFFICIENT :
755  if (nDim == 2) { ForceProj_Vector[0] += -Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] += Weight_ObjFunc*cos(Alpha); }
756  if (nDim == 3) { ForceProj_Vector[0] += -Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] = 0.0; ForceProj_Vector[2] += Weight_ObjFunc*cos(Alpha); }
757  break;
758  case SIDEFORCE_COEFFICIENT :
759  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
760  if (nDim == 3) { ForceProj_Vector[0] += -Weight_ObjFunc*sin(Beta) * cos(Alpha); ForceProj_Vector[1] += Weight_ObjFunc*cos(Beta); ForceProj_Vector[2] += -Weight_ObjFunc*sin(Beta) * sin(Alpha); }
761  break;
762  case INVERSE_DESIGN_PRESSURE :
763  Cp = solver_container[FLOW_SOL]->GetCPressure(iMarker, iVertex);
764  CpTarget = solver_container[FLOW_SOL]->GetCPressureTarget(iMarker, iVertex);
765  Area = sqrt(Normal[0]*Normal[0] + Normal[1]*Normal[1]);
766  if (nDim == 3) Area = sqrt(Normal[0]*Normal[0] + Normal[1]*Normal[1] + Normal[2]*Normal[2]);
767  ForceProj_Vector[0] += -Weight_ObjFunc*2.0*(Cp-CpTarget)*Normal[0]/Area; ForceProj_Vector[1] += -Weight_ObjFunc*2.0*(Cp-CpTarget)*Normal[1]/Area;
768  if (nDim == 3) ForceProj_Vector[2] += -Weight_ObjFunc*2.0*(Cp-CpTarget)*Normal[2]/Area;
769  break;
770  case MOMENT_X_COEFFICIENT :
771  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
772  if (nDim == 3) {
773 
774  ForceProj_Vector[0] += 0.0;
775  ForceProj_Vector[1] += -Weight_ObjFunc*(z - z_origin)/RefLength;
776  ForceProj_Vector[2] += Weight_ObjFunc*(y - y_origin)/RefLength;
777 
778  /*--- Modification to run at a fixed CL value ---*/
779 
780  if (Fixed_CL) { ForceProj_Vector[0] += dCMx_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= 0.0; ForceProj_Vector[2] -= dCMx_dCL*Weight_ObjFunc*cos(Alpha); }
781 
782  }
783  break;
784  case MOMENT_Y_COEFFICIENT :
785  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
786  if (nDim == 3) {
787 
788  ForceProj_Vector[0] += Weight_ObjFunc*(z - z_origin)/RefLength;
789  ForceProj_Vector[1] += 0.0;
790  ForceProj_Vector[2] += -Weight_ObjFunc*(x - x_origin)/RefLength;
791 
792  /*--- Modification to run at a fixed CL value ---*/
793 
794  if (Fixed_CL) { ForceProj_Vector[0] += dCMy_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= 0.0; ForceProj_Vector[2] -= dCMy_dCL*Weight_ObjFunc*cos(Alpha); }
795 
796  }
797  break;
798  case MOMENT_Z_COEFFICIENT :
799  if (nDim == 2) {
800 
801  ForceProj_Vector[0] += -Weight_ObjFunc*(y - y_origin)/RefLength;
802  ForceProj_Vector[1] += Weight_ObjFunc*(x - x_origin)/RefLength;
803 
804  /*--- Modification to run at a fixed CL and CM value ---*/
805 
806  if (Fixed_CL) { ForceProj_Vector[0] += dCMz_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= dCMz_dCL*Weight_ObjFunc*cos(Alpha); }
807 
808  }
809  if (nDim == 3) {
810 
811  ForceProj_Vector[0] += -Weight_ObjFunc*(y - y_origin)/RefLength;
812  ForceProj_Vector[1] += Weight_ObjFunc*(x - x_origin)/RefLength;
813  ForceProj_Vector[2] += 0;
814 
815  /*--- Modification to run at a fixed CL value ---*/
816 
817  if (Fixed_CL) { ForceProj_Vector[0] += dCMz_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= 0.0; ForceProj_Vector[2] -= dCMz_dCL*Weight_ObjFunc*cos(Alpha); }
818 
819  }
820  break;
821  case EFFICIENCY :
822  if (nDim == 2) { ForceProj_Vector[0] += -Weight_ObjFunc*(invCD*sin(Alpha)+CLCD2*cos(Alpha)); ForceProj_Vector[1] += Weight_ObjFunc*(invCD*cos(Alpha)-CLCD2*sin(Alpha)); }
823  if (nDim == 3) { ForceProj_Vector[0] += -Weight_ObjFunc*(invCD*sin(Alpha)+CLCD2*cos(Alpha)*cos(Beta)); ForceProj_Vector[1] += -Weight_ObjFunc*CLCD2*sin(Beta); ForceProj_Vector[2] += Weight_ObjFunc*(invCD*cos(Alpha)-CLCD2*sin(Alpha)*cos(Beta)); }
824  break;
825  case EQUIVALENT_AREA :
826  WDrag = Weight_ObjFunc*config->GetWeightCd();
827  if (nDim == 2) { ForceProj_Vector[0] = cos(Alpha)*WDrag; ForceProj_Vector[1] = sin(Alpha)*WDrag; }
828  if (nDim == 3) { ForceProj_Vector[0] = cos(Alpha)*cos(Beta)*WDrag; ForceProj_Vector[1] = sin(Beta)*WDrag; ForceProj_Vector[2] = sin(Alpha)*cos(Beta)*WDrag; }
829  break;
830  case NEARFIELD_PRESSURE :
831  WDrag = Weight_ObjFunc*config->GetWeightCd();
832  if (nDim == 2) { ForceProj_Vector[0] = cos(Alpha)*WDrag; ForceProj_Vector[1] = sin(Alpha)*WDrag; }
833  if (nDim == 3) { ForceProj_Vector[0] = cos(Alpha)*cos(Beta)*WDrag; ForceProj_Vector[1] = sin(Beta)*WDrag; ForceProj_Vector[2] = sin(Alpha)*cos(Beta)*WDrag; }
834  break;
835  case FORCE_X_COEFFICIENT :
836  if (nDim == 2) { ForceProj_Vector[0] += Weight_ObjFunc*1.0; ForceProj_Vector[1] += 0.0; }
837  if (nDim == 3) { ForceProj_Vector[0] += Weight_ObjFunc*1.0; ForceProj_Vector[1] += 0.0; ForceProj_Vector[2] += 0.0; }
838  break;
839  case FORCE_Y_COEFFICIENT :
840  if (nDim == 2) { ForceProj_Vector[0] += 0.0; ForceProj_Vector[1] += Weight_ObjFunc*1.0; }
841  if (nDim == 3) { ForceProj_Vector[0] += 0.0; ForceProj_Vector[1] += Weight_ObjFunc*1.0; ForceProj_Vector[2] += 0.0; }
842  break;
843  case FORCE_Z_COEFFICIENT :
844  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
845  if (nDim == 3) { ForceProj_Vector[0] += 0.0; ForceProj_Vector[1] += 0.0; ForceProj_Vector[2] += Weight_ObjFunc*1.0; }
846  break;
847  case THRUST_COEFFICIENT :
848  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
849  if (nDim == 3) { ForceProj_Vector[0] += 0.0; ForceProj_Vector[1] += 0.0; ForceProj_Vector[2] += Weight_ObjFunc*1.0; }
850  break;
851  case TORQUE_COEFFICIENT :
852  if (nDim == 2) { ForceProj_Vector[0] += Weight_ObjFunc*(y - y_origin)/RefLength; ForceProj_Vector[1] += -Weight_ObjFunc*(x - x_origin)/RefLength; }
853  if (nDim == 3) { ForceProj_Vector[0] += Weight_ObjFunc*(y - y_origin)/RefLength; ForceProj_Vector[1] += -Weight_ObjFunc*(x - x_origin)/RefLength; ForceProj_Vector[2] += 0; }
854  break;
855  case FIGURE_OF_MERIT :
856  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
857  if (nDim == 3) {
858  ForceProj_Vector[0] += -Weight_ObjFunc*invCQ;
859  ForceProj_Vector[1] += -Weight_ObjFunc*CTRCQ2*(z - z_origin);
860  ForceProj_Vector[2] += Weight_ObjFunc*CTRCQ2*(y - y_origin);
861  }
862  break;
863 
864  default :
865  break;
866  }
867 
868  /*--- Store the force projection vector at this node ---*/
869 
870  nodes->SetForceProj_Vector(iPoint,ForceProj_Vector);
871 
872  }
873  }
874  }
875  delete [] ForceProj_Vector;

1614

1780  Jacobian_j[4][0] = 0.0;

1789  Jacobian_j[1][2] = (rv/r)*Area*UnitNormal[0] + (dp_dru)*Area*UnitNormal[1];
1790  Jacobian_j[2][2] = (ru/r)*Area*UnitNormal[0] + (2*rv/r + dp_drv)*Area*UnitNormal[1] + (rw/r)*Area*UnitNormal[2];
1791  Jacobian_j[3][2] = (dp_drw)*Area*UnitNormal[1] + (rv/r)*Area*UnitNormal[2];
1792  Jacobian_j[4][2] = (dp_drE)*Area*UnitNormal[1];
1793

1808

1847  for (jPos = 0; jPos < nVar; jPos++) {

1849  }
1850  }
1851  }
1852  }

1855  Total_Sens_AoA -= Sens_AoA[iMarker] * scale * factor;

1858 
1859  }
1860  }

1865

1867
