98 void CIncTGVSolution::GetSolution(const su2double* val_coords, const su2double val_t, su2double* val_solution) const {
99  /* The exact solution is set for the incompressible Taylor-Green
100  vortex case. This is the classic solution from the original work
101  of Taylor and Green for the specific 2D situation where the
102  exact solution can be derived for an incompressible flow. */
103 
104  /* Store the termporal term more easily (Taylor expansion). */
105  su2double F = 1.0 - 2.0 * (tgvViscosity / tgvDensity) * val_t;
106 
107  /* Compute the primitive variables. */
108  su2double u = tgvVelocity * F * (sin(val_coords[0] / tgvLength) * cos(val_coords[1] / tgvLength));
109  su2double v = -tgvVelocity * F * (cos(val_coords[0] / tgvLength) * sin(val_coords[1] / tgvLength));
110 
111  su2double B = (cos(2.0 * val_coords[0] / tgvLength) + cos(2.0 * val_coords[1] / tgvLength));
112  su2double p = -(tgvDensity / 4.0) * B * F * F;
113 
114  /* Compute the conservative variables. Note that both 2D and 3D
115  cases are treated correctly. */
116  val_solution[0] = p;
117  val_solution[1] = u;
118  val_solution[2] = v;
119  val_solution[nVar - 1] = Temperature;
120 }
