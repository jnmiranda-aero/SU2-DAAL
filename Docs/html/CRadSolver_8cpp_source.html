44void CRadSolver::SetVolumetricHeatSource(CGeometry *geometry, CConfig *config) {
45
46 unsigned long iPoint;
47 unsigned short iDim;
48
49 su2double CP[3]={0.0,0.0,0.0};
50 su2double alpha = config->GetHeatSource_Rot_Z() * PI_NUMBER/180.0;
51 su2double OP_rot[3]={0.0,0.0,0.0};
52 const su2double *OP;
53 const su2double *OC = config->GetHeatSource_Center();
54 const su2double *Axes = config->GetHeatSource_Axes();
55 su2double check;
56 // Reset the boolean for all points
57 nodes->ResetVol_HeatSource();
58 // Loop over all points and determine whether they are inside
59 for (iPoint = 0; iPoint < nPoint; iPoint ++) {
60 check = 0;
61 OP = geometry->nodes->GetCoord(iPoint);
62 // Reference point with respect to center of the ellipse
63 for (iDim = 0; iDim < nDim; iDim++) CP[iDim] = OP[iDim]-OC[iDim];
64 // Rotate point with respect to Z axis
65 OP_rot[0] = OC[0] + CP[0]*cos(alpha) + CP[1]*sin(alpha);
66 OP_rot[1] = OC[1] - CP[0]*sin(alpha) + CP[1]*cos(alpha);
67 // Check if rotated point is inside the ellipse
68 for (iDim = 0; iDim < nDim; iDim++) check += pow(OP_rot[iDim]-OC[iDim],2.0)/pow(Axes[iDim], 2.0);
69 if (check <=1) nodes->SetVol_HeatSource(iPoint);
70 }
71
72}

74void CRadSolver::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter, bool val_update_geo) {
75
76 /*--- Restart the solution from file information ---*/
77
78 unsigned short iVar;
79 unsigned long index;
80
81 bool rans = ((config->GetKind_Solver()== MAIN_SOLVER::INC_RANS) ||
82 (config->GetKind_Solver()== MAIN_SOLVER::DISC_ADJ_INC_RANS));
83
84 string UnstExt, text_line;
85 ifstream restart_file;
86
87 string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
88 /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
89
90 if (config->GetRead_Binary_Restart()) {
91 Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
92 } else {
93 Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
94 }
95
96 int counter = 0;
97 long iPoint_Local = 0; unsigned long iPoint_Global = 0;
98 unsigned long iPoint_Global_Local = 0;
99
100 /*--- Skip flow variables ---*/
101
102 unsigned short skipVars = 0;
103
104 if (nDim == 2) skipVars += 6;
105 if (nDim == 3) skipVars += 8;
106
107 bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
108 bool energy = config->GetEnergy_Equation();
109 bool weakly_coupled_heat = config->GetWeakly_Coupled_Heat();
110
111 if (incompressible && ((!energy) && (!weakly_coupled_heat))) skipVars--;
112
113 /*--- Skip turbulent variables ---*/
114
115 if (rans) skipVars += solver[MESH_0][TURB_SOL]->GetnVar();
116
117 /*--- Load data from the restart into correct containers. ---*/
118
119 counter = 0;
120 for (iPoint_Global = 0; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++ ) {
121
122
123 /*--- Retrieve local index. If this node from the restart file lives
124 on the current processor, we will load and instantiate the vars. ---*/
125
126 iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
127
128 if (iPoint_Local > -1) {
129
130 /*--- We need to store this point's data, so jump to the correct
131 offset in the buffer of data from the restart file and load it. ---*/
132
133 index = counter*Restart_Vars[1] + skipVars;
134 for (iVar = 0; iVar < nVar; iVar++) Solution[iVar] = Restart_Data[index+iVar];
135 nodes->SetSolution(iPoint_Local, Solution);
136 iPoint_Global_Local++;
137
138 /*--- Increment the overall counter for how many points have been loaded. ---*/
139 counter++;
140 }
141
142 }
143
144 /*--- Detect a wrong solution file ---*/
145
146 if (iPoint_Global_Local < nPointDomain) {
147 SU2_MPI::Error(string("The solution file ") + restart_filename + string(" doesn't match with the mesh file!\n") +
148 string("It could be empty lines at the end of the file."), CURRENT_FUNCTION);
149 }
150
151 /*--- MPI communication ---*/
152 solver[MESH_0][RAD_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
153 solver[MESH_0][RAD_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
154
155 /*--- Preprocess the fluid solver to compute the primitive variables ---*/
156 solver[MESH_0][FLOW_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
157
158 /*--- Postprocess the radiation solver to compute the source term that goes into the fluid equations ---*/
159 solver[MESH_0][RAD_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
160
161 /*--- Delete the class memory that is used to load the restart. ---*/
162
163 Restart_Vars = decltype(Restart_Vars){};
164 Restart_Data = decltype(Restart_Data){};
165}
