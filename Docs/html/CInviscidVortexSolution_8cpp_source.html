32CInviscidVortexSolution::CInviscidVortexSolution(unsigned short val_nDim, unsigned short val_nVar,
33 unsigned short val_iMesh, CConfig* config)
34 : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
35 /*--- Write a message that the solution is initialized for the
36 inviscid vortex test case. ---*/
37 if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
38 cout << endl;
39 cout << "Warning: Fluid properties and solution are being " << endl;
40 cout << " initialized for the inviscid vortex case!!!" << endl;
41 cout << endl << flush;
42 }
43
44 /*--- Store the inviscid vortex specific parameters here. ---*/
45 x0Vortex = -0.5; // Initial x-coordinate of the vortex center.
46 y0Vortex = 0.0; // Initial y-coordinate of the vortex center.
47 RVortex = 0.1; // Radius of the vortex.
48 epsVortex = 1.0; // Strength of the vortex.
49
50 /* Get the Mach number and advection angle (in degrees). */
51 MachVortex = config->GetMach();
52 thetaVortex = config->GetAoA();
53
54 /*--- Useful coefficients in which Gamma is present. ---*/
55 Gamma = config->GetGamma();
56 Gm1 = Gamma - 1.0;
57 ovGm1 = 1.0 / Gm1;
58 gamOvGm1 = ovGm1 * Gamma;
59
60 /*--- Perform some sanity and error checks for this solution here. ---*/
61 if ((config->GetTime_Marching() != TIME_MARCHING::TIME_STEPPING) &&
62 (config->GetTime_Marching() != TIME_MARCHING::DT_STEPPING_1ST) &&
63 (config->GetTime_Marching() != TIME_MARCHING::DT_STEPPING_2ND))
64 SU2_MPI::Error("Unsteady mode must be selected for the inviscid vortex", CURRENT_FUNCTION);
65
66 if (Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES &&
67 Kind_Solver != MAIN_SOLVER::RANS && Kind_Solver != MAIN_SOLVER::FEM_EULER &&
68 Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
69 Kind_Solver != MAIN_SOLVER::FEM_LES)
70 SU2_MPI::Error("Compressible flow equations must be selected for the inviscid vortex", CURRENT_FUNCTION);
71
72 if ((Kind_Solver != MAIN_SOLVER::EULER) && (Kind_Solver != MAIN_SOLVER::FEM_EULER))
73 SU2_MPI::Error("Euler equations must be selected for the inviscid vortex", CURRENT_FUNCTION);
74
75 if ((config->GetKind_FluidModel() != STANDARD_AIR) && (config->GetKind_FluidModel() != IDEAL_GAS))
76 SU2_MPI::Error("Standard air or ideal gas must be selected for the inviscid vortex", CURRENT_FUNCTION);
77
78 if (fabs(config->GetPressure_FreeStreamND() - 1.0) > 1.e-8)
79 SU2_MPI::Error("Free-stream pressure must be 1.0 for the inviscid vortex", CURRENT_FUNCTION);
80
81 if (fabs(config->GetDensity_FreeStreamND() - 1.0) > 1.e-8)
82 SU2_MPI::Error("Free-stream density must be 1.0 for the inviscid vortex", CURRENT_FUNCTION);
83}

95void CInviscidVortexSolution::GetSolution(const su2double* val_coords, const su2double val_t,
96 su2double* val_solution) const {
97 /* Compute the free stream velocities in x- and y-direction. */
98 const su2double VelInf = MachVortex * sqrt(Gamma);
99 const su2double uInf = VelInf * cos(thetaVortex * PI_NUMBER / 180.0);
100 const su2double vInf = VelInf * sin(thetaVortex * PI_NUMBER / 180.0);
101
102 /* Compute the coordinates relative to the center of the vortex. */
103 const su2double dx = val_coords[0] - (x0Vortex + val_t * uInf);
104 const su2double dy = val_coords[1] - (y0Vortex + val_t * vInf);
105
106 /* Compute the components of the velocity. */
107 su2double f = 1.0 - (dx * dx + dy * dy) / (RVortex * RVortex);
108 su2double t1 = epsVortex * dy * exp(0.5 * f) / (2.0 * PI_NUMBER * RVortex);
109 su2double u = uInf - VelInf * t1;
110
111 t1 = epsVortex * dx * exp(0.5 * f) / (2.0 * PI_NUMBER * RVortex);
112 su2double v = vInf + VelInf * t1;
113
114 /* Compute the density and the pressure. */
115 t1 = 1.0 - epsVortex * epsVortex * Gm1 * MachVortex * MachVortex * exp(f) / (8.0 * PI_NUMBER * PI_NUMBER);
116
117 su2double rho = pow(t1, ovGm1);
118 su2double p = pow(t1, gamOvGm1);
119
120 /* Compute the conservative variables. Note that both 2D and 3D
121 cases are treated correctly. */
122 val_solution[0] = rho;
123 val_solution[1] = rho * u;
124 val_solution[2] = rho * v;
125 val_solution[3] = 0.0;
126 val_solution[nVar - 1] = p * ovGm1 + 0.5 * rho * (u * u + v * v);
127}
