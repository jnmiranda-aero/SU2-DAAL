251  /*!
252  * \brief Compute value of the Q criteration for vortex idenfitication
253  * \param[in] VelocityGradient - Velocity gradients
254  * \return Value of the Q criteration at the node
255  */
256  template<class T>
257  su2double GetQCriterion(const T& VelocityGradient) const {
258 
259  /*--- Make a 3D copy of the gradient so we do not have worry about nDim ---*/
260 
261  su2double Grad_Vel[3][3] = {{0.0}};
262 
263  for (unsigned short iDim = 0; iDim < nDim; iDim++)
264  for (unsigned short jDim = 0 ; jDim < nDim; jDim++)
265  Grad_Vel[iDim][jDim] = VelocityGradient[iDim][jDim];
266 
267  /*--- Q Criterion Eq 1.2 of HALLER, G. (2005). An objective definition of a vortex.
268  Journal of Fluid Mechanics, 525, 1-26. doi:10.1017/S0022112004002526 ---*/
269 
270  /*--- Components of the strain rate tensor (symmetric) ---*/
271  su2double s11 = Grad_Vel[0][0];
272  su2double s12 = 0.5 * (Grad_Vel[0][1] + Grad_Vel[1][0]);
273  su2double s13 = 0.5 * (Grad_Vel[0][2] + Grad_Vel[2][0]);
274  su2double s22 = Grad_Vel[1][1];
275  su2double s23 = 0.5 * (Grad_Vel[1][2] + Grad_Vel[2][1]);
276  su2double s33 = Grad_Vel[2][2];
277 
278  /*--- Components of the spin tensor (skew-symmetric) ---*/
279  su2double omega12 = 0.5 * (Grad_Vel[0][1] - Grad_Vel[1][0]);
280  su2double omega13 = 0.5 * (Grad_Vel[0][2] - Grad_Vel[2][0]);
281  su2double omega23 = 0.5 * (Grad_Vel[1][2] - Grad_Vel[2][1]);
282 
283  /*--- Q = ||Omega|| - ||Strain|| ---*/
284  su2double Q = 2*(pow(omega12,2) + pow(omega13,2) + pow(omega23,2)) -
285  (pow(s11,2) + pow(s22,2) + pow(s33,2) + 2*(pow(s12,2) + pow(s13,2) + pow(s23,2)));
286 
287  return Q;
288  }

290  /*!
291  * \brief Returns the axisymmetric factor for a point on a marker.
292  */
293  template <class GeoNodes>
294  inline su2double GetAxiFactor(bool axisymmetric, const GeoNodes& nodes, unsigned long iPoint,
295  unsigned short iMarker) {
296  if (!axisymmetric) return 1.0;
297 
298  if (nodes.GetCoord(iPoint, 1) > EPS) return 2 * PI_NUMBER * nodes.GetCoord(iPoint, 1);
299 
300  for (const auto jPoint : nodes.GetPoints(iPoint)) {
301  if (nodes.GetVertex(jPoint, iMarker) >= 0) {
302  /*--- Not multiplied by two since we need to half the y coordinate. ---*/
303  return PI_NUMBER * nodes.GetCoord(jPoint, 1);
304  }
305  }
306  return 0.0;
307  }
