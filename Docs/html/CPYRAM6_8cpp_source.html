30CPYRAM6::CPYRAM6() : CElementWithKnownSizes<NGAUSS, NNODE, NDIM>() {
31 /*--- Gauss coordinates and weights ---*/
32
33 GaussCoord[0][0] = sqrt(4.0 / 27.0);
34 GaussCoord[0][1] = sqrt(4.0 / 27.0);
35 GaussCoord[0][2] = 1.0 / 6.0;
36 GaussWeight(0) = 9.0 / 20.0;
37 GaussCoord[0][0] = sqrt(4.0 / 27.0);
38 GaussCoord[0][1] = sqrt(4.0 / 27.0);
39 GaussCoord[0][2] = 1.0 / 6.0;
40 GaussWeight(1) = 9.0 / 20.0;
41 GaussCoord[0][0] = sqrt(4.0 / 27.0);
42 GaussCoord[0][1] = sqrt(4.0 / 27.0);
43 GaussCoord[0][2] = 1.0 / 6.0;
44 GaussWeight(2) = 9.0 / 20.0;
45 GaussCoord[0][0] = sqrt(4.0 / 27.0);
46 GaussCoord[0][1] = sqrt(4.0 / 27.0);
47 GaussCoord[0][2] = 1.0 / 6.0;
48 GaussWeight(3) = 9.0 / 20.0;
49 GaussCoord[4][0] = 0.0;
50 GaussCoord[4][1] = 0.0;
51 GaussCoord[4][2] = 0.5;
52 GaussWeight(4) = 3.0 / 5.0;
53 GaussCoord[5][0] = 0.0;
54 GaussCoord[5][1] = 0.0;
55 GaussCoord[5][2] = 0.25;
56 GaussWeight(5) = -16.0 / 15.0;
57
58 /*--- Store the values of the shape functions and their derivatives ---*/
59
60 unsigned short iNode, iGauss;
61 su2double Xi, Eta, Zeta, val_Ni;
62
63 for (iGauss = 0; iGauss < nGaussPoints; iGauss++) {
64 Xi = GaussCoord[iGauss][0];
65 Eta = GaussCoord[iGauss][1];
66 Zeta = GaussCoord[iGauss][2];
67
68 val_Ni = 0.25 * (-Xi + Eta + Zeta - 1.0) * (-Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
69 GaussPoint[iGauss].SetNi(val_Ni, 0);
70 val_Ni = 0.25 * (-Xi - Eta + Zeta - 1.0) * (Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
71 GaussPoint[iGauss].SetNi(val_Ni, 1);
72 val_Ni = 0.25 * (Xi + Eta + Zeta - 1.0) * (Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
73 GaussPoint[iGauss].SetNi(val_Ni, 2);
74 val_Ni = 0.25 * (Xi + Eta + Zeta - 1.0) * (-Xi + Eta + Zeta - 1.0) / (1.0 - Zeta);
75 GaussPoint[iGauss].SetNi(val_Ni, 3);
76 val_Ni = Zeta;
77 GaussPoint[iGauss].SetNi(val_Ni, 4);
78
79 /*--- dN/d xi ---*/
80
81 dNiXj[iGauss][0][0] = 0.5 * (Zeta - Xi - 1.0) / (Zeta - 1.0);
82 dNiXj[iGauss][1][0] = 0.5 * Xi / (Zeta - 1.0);
83 dNiXj[iGauss][2][0] = 0.5 * (1.0 - Zeta - Xi) / (Zeta - 1.0);
84 dNiXj[iGauss][3][0] = dNiXj[iGauss][1][0];
85 dNiXj[iGauss][4][0] = 0.0;
86
87 /*--- dN/d eta ---*/
88
89 dNiXj[iGauss][0][1] = 0.5 * Eta / (Zeta - 1.0);
90 dNiXj[iGauss][1][1] = 0.5 * (Zeta - Eta - 1.0) / (Zeta - 1.0);
91 dNiXj[iGauss][2][1] = dNiXj[iGauss][0][1];
92 dNiXj[iGauss][3][1] = 0.5 * (1.0 - Zeta - Eta) / (Zeta - 1.0);
93 dNiXj[iGauss][4][1] = 0.0;
94
95 /*--- dN/d zeta ---*/
96
97 dNiXj[iGauss][0][2] =
98 0.25 * (-1.0 + 2.0 * Zeta - Zeta * Zeta - Eta * Eta + Xi * Xi) / ((1.0 - Zeta) * (1.0 - Zeta));
99 dNiXj[iGauss][1][2] =
100 0.25 * (-1.0 + 2.0 * Zeta - Zeta * Zeta + Eta * Eta - Xi * Xi) / ((1.0 - Zeta) * (1.0 - Zeta));
101 dNiXj[iGauss][2][2] = dNiXj[iGauss][0][2];
102 dNiXj[iGauss][3][2] = dNiXj[iGauss][1][2];
103 dNiXj[iGauss][4][2] = 1.0;
104 }
105
106 /*--- Store the extrapolation functions ---*/
107
108 su2double ExtrapCoord[5][3];
109
110 ExtrapCoord[0][0] = 2.0;
111 ExtrapCoord[0][1] = 0.0;
112 ExtrapCoord[0][2] = -0.316397779494322;
113 ExtrapCoord[1][0] = 0.0;
114 ExtrapCoord[1][1] = 2.0;
115 ExtrapCoord[1][2] = -0.316397779494322;
116 ExtrapCoord[2][0] = -2.0;
117 ExtrapCoord[2][1] = 0.0;
118 ExtrapCoord[2][2] = -0.316397779494322;
119 ExtrapCoord[3][0] = 0.0;
120 ExtrapCoord[3][1] = -2.0;
121 ExtrapCoord[3][2] = -0.316397779494322;
122 ExtrapCoord[4][0] = 0.0;
123 ExtrapCoord[4][1] = 0.0;
124 ExtrapCoord[4][2] = 1.749193338482970;
125
126 for (iNode = 0; iNode < nNodes; iNode++) {
127 Xi = ExtrapCoord[iNode][0];
128 Eta = ExtrapCoord[iNode][1];
129 Zeta = ExtrapCoord[iNode][2];
130
131 NodalExtrap[iNode][0] = 0.25 * (-Xi + Eta + Zeta - 1.0) * (-Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
132 NodalExtrap[iNode][1] = 0.25 * (-Xi - Eta + Zeta - 1.0) * (Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
133 NodalExtrap[iNode][2] = 0.25 * (Xi + Eta + Zeta - 1.0) * (Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
134 NodalExtrap[iNode][3] = 0.25 * (Xi + Eta + Zeta - 1.0) * (-Xi + Eta + Zeta - 1.0) / (1.0 - Zeta);
135 NodalExtrap[iNode][4] = Zeta;
136 }
137}

139su2double CPYRAM6::ComputeVolume(const FrameType mode) const {
140 unsigned short iDim;
141 su2double r1[3] = {0.0, 0.0, 0.0}, r2[3] = {0.0, 0.0, 0.0}, r3[3] = {0.0, 0.0, 0.0},
142 CrossProduct[3] = {0.0, 0.0, 0.0};
143 su2double Volume = 0.0;
144
145 /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
146 for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed)---*/
147 const su2activematrix& Coord = (mode == REFERENCE) ? RefCoord : CurrentCoord;
148
149 for (iDim = 0; iDim < NDIM; iDim++) {
150 r1[iDim] = Coord[1][iDim] - Coord[0][iDim];
151 r2[iDim] = Coord[2][iDim] - Coord[0][iDim];
152 r3[iDim] = Coord[4][iDim] - Coord[0][iDim];
153 }
154
155 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
156 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
157 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
158
159 Volume = fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
160
161 for (iDim = 0; iDim < NDIM; iDim++) {
162 r1[iDim] = Coord[2][iDim] - Coord[0][iDim];
163 r2[iDim] = Coord[3][iDim] - Coord[0][iDim];
164 r3[iDim] = Coord[4][iDim] - Coord[0][iDim];
165 }
166
167 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
168 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
169 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
170
171 Volume += fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
172
173 return Volume;
174}
