62 void CInterface::BroadcastData(const CInterpolator& interpolator,
63  CSolver *donor_solution, CSolver *target_solution,
64  CGeometry *donor_geometry, CGeometry *target_geometry,
65  const CConfig *donor_config, const CConfig *target_config) {
66  static_assert(su2activematrix::Storage == StorageType::RowMajor,"");
67 
68  GetPhysical_Constants(donor_solution, target_solution, donor_geometry, target_geometry,
69  donor_config, target_config);
70 
71  /*--- Loop over interface markers. ---*/
72 
73  for (auto iMarkerInt = 0u; iMarkerInt < donor_config->GetMarker_n_ZoneInterface()/2; iMarkerInt++) {
74 
75  /*--- Check if this interface connects the two zones, if not continue. ---*/
76 
77  const auto markDonor = donor_config->FindInterfaceMarker(iMarkerInt);
78  const auto markTarget = target_config->FindInterfaceMarker(iMarkerInt);
79 
80  if(!CInterpolator::CheckInterfaceBoundary(markDonor, markTarget)) continue;
81 
82  /*--- Count donor vertices on this rank. ---*/
83 
84  int nLocalVertexDonor = 0;
85  if (markDonor >= 0) {
86  for (auto iVertex = 0ul; iVertex < donor_geometry->GetnVertex(markDonor); iVertex++) {
87  auto Point_Donor = donor_geometry->vertex[markDonor][iVertex]->GetNode();
88  /*--- Only domain points are donors. ---*/
89  nLocalVertexDonor += donor_geometry->nodes->GetDomain(Point_Donor);
90  }
91  }
92 
93  /*--- Gather donor counts and compute total sizes, and displacements (cumulative
94  * sums) to perform an Allgatherv of donor indices and variables. ---*/
95 
96  vector<int> nAllVertexDonor(size), nAllVarCounts(size), displIdx(size,0), displVar(size);
97  SU2_MPI::Allgather(&nLocalVertexDonor, 1, MPI_INT, nAllVertexDonor.data(), 1, MPI_INT, SU2_MPI::GetComm());
98 
99  for (int i = 0; i < size; ++i) {
100  nAllVarCounts[i] = nAllVertexDonor[i] * nVar;
101  if(i) displIdx[i] = displIdx[i-1] + nAllVertexDonor[i-1];
102  displVar[i] = displIdx[i] * nVar;
103  }
104 
105  /*--- Fill send buffers. ---*/
106 
107  vector<unsigned long> sendDonorIdx(nLocalVertexDonor);
108  su2activematrix sendDonorVar(nLocalVertexDonor, nVar);
109 
110  if (markDonor >= 0) {
111 
112  /*--- Apply contact resistance if specified. ---*/
113 
114  SetContactResistance(donor_config->GetContactResistance(iMarkerInt));
115 
116  for (auto iVertex = 0ul, iSend = 0ul; iVertex < donor_geometry->GetnVertex(markDonor); iVertex++) {
117  const auto iPoint = donor_geometry->vertex[markDonor][iVertex]->GetNode();
118 
119  /*--- If this processor owns the node. ---*/
120  if (donor_geometry->nodes->GetDomain(iPoint)) {
121 
122  GetDonor_Variable(donor_solution, donor_geometry, donor_config, markDonor, iVertex, iPoint);
123  for (auto iVar = 0u; iVar < nVar; iVar++) sendDonorVar(iSend, iVar) = Donor_Variable[iVar];
124 
125  sendDonorIdx[iSend] = donor_geometry->nodes->GetGlobalIndex(iPoint);
126  ++iSend;
127  }
128  }
129  }
130 
131  /*--- Gather data. ---*/
132 
133  const auto nGlobalVertexDonor = displIdx.back() + nAllVertexDonor.back();
134 
135  vector<unsigned long> donorIdx(nGlobalVertexDonor);
136  su2activematrix donorVar(nGlobalVertexDonor, nVar);
137 
138  SU2_MPI::Allgatherv(sendDonorIdx.data(), sendDonorIdx.size(), MPI_UNSIGNED_LONG, donorIdx.data(),
139  nAllVertexDonor.data(), displIdx.data(), MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
140 
141  SU2_MPI::Allgatherv(sendDonorVar.data(), sendDonorVar.size(), MPI_DOUBLE, donorVar.data(),
142  nAllVarCounts.data(), displVar.data(), MPI_DOUBLE, SU2_MPI::GetComm());
143 
144  /*--- This rank does not need to do more work. ---*/
145  if (markTarget < 0) continue;
146 
147  /*--- Sort the donor information by index to then use binary searches. ---*/
148 
149  vector<size_t> order(donorIdx.size());
150  iota(order.begin(), order.end(), 0ul);
151  sort(order.begin(), order.end(), [&donorIdx](size_t i, size_t j) {return donorIdx[i] < donorIdx[j];} );
152 
153  /*--- inplace permutation. ---*/
154  for (size_t i = 0; i < order.size(); ++i) {
155  auto j = order[i];
156  while (j < i) j = order[j];
157  if (i == j) continue;
158  swap(donorIdx[i], donorIdx[j]);
159  for (auto iVar = 0u; iVar < nVar; ++iVar)
160  swap(donorVar(i,iVar), donorVar(j,iVar));
161  }
162 
163  /*--- Loop over target vertices. ---*/
164 
165  for (auto iVertex = 0ul; iVertex < target_geometry->GetnVertex(markTarget); iVertex++) {
166  const auto iPoint = target_geometry->vertex[markTarget][iVertex]->GetNode();
167 
168  if (!target_geometry->nodes->GetDomain(iPoint)) continue;
169 
170  auto& targetVertex = interpolator.targetVertices[markTarget][iVertex];
171  const auto nDonorPoints = targetVertex.nDonor();
172 
173  InitializeTarget_Variable(target_solution, markTarget, iVertex, nDonorPoints);
174 
175  /*--- For the number of donor points. ---*/
176  for (auto iDonorPoint = 0ul; iDonorPoint < nDonorPoints; iDonorPoint++) {
177 
178  /*--- Get the global index of the donor and the interpolation coefficient. ---*/
179 
180  const auto donorGlobalIndex = targetVertex.globalPoint[iDonorPoint];
181  const auto donorCoeff = targetVertex.coefficient[iDonorPoint];
182 
183  /*--- Find the index of the global donor point in the donor data. ---*/
184 
185  const auto idx = lower_bound(donorIdx.begin(), donorIdx.end(), donorGlobalIndex) - donorIdx.begin();
186  assert(idx < static_cast<long>(donorIdx.size()));
187 
188  /*--- Recover the Target_Variable from the buffer of variables. ---*/
189  RecoverTarget_Variable(donorVar[idx], donorCoeff);
190 
191  /*--- If the value is not directly aggregated in the previous function. ---*/
192  if (!valAggregated)
193  SetTarget_Variable(target_solution, target_geometry, target_config, markTarget, iVertex, iPoint);
194  }
195 
196  /*--- If we have aggregated the values in the function RecoverTarget_Variable, the set is outside the loop. ---*/
197  if (valAggregated)
198  SetTarget_Variable(target_solution, target_geometry, target_config, markTarget, iVertex, iPoint);
199  }
200  }
201 }

203 void CInterface::PreprocessAverage(CGeometry *donor_geometry, CGeometry *target_geometry,
204  const CConfig *donor_config, const CConfig *target_config,
205  unsigned short iMarkerInt){
206 
207  unsigned short nMarkerDonor, nMarkerTarget; // Number of markers on the interface, donor and target side
208  unsigned short iMarkerDonor, iMarkerTarget; // Variables for iteration over markers
209  unsigned short iSpan,jSpan, tSpan = 0, kSpan = 0, nSpanDonor, nSpanTarget, Donor_Flag = 0, Target_Flag = 0;
210  int Marker_Donor = -1, Marker_Target = -1;
211 
212  const su2double *SpanValuesDonor, *SpanValuesTarget;
213  su2double dist, test, dist2, test2;
214 
215  nMarkerDonor = donor_geometry->GetnMarker();
216  nMarkerTarget = target_geometry->GetnMarker();
217  //TODO turbo this approach only works if all the turboamchinery marker
218  // of all zones have the same amount of span wise sections.
219  //TODO turbo initialization needed for the MPI routine should be place somewhere else.
220  nSpanDonor = donor_config->GetnSpanWiseSections();
221  nSpanTarget = target_config->GetnSpanWiseSections();
222 
223  /*--- On the donor side ---*/
224  for (iMarkerDonor = 0; iMarkerDonor < nMarkerDonor; iMarkerDonor++){
225  /*--- If the tag GetMarker_All_MixingPlaneInterface equals the index we are looping at ---*/
226  if ( donor_config->GetMarker_All_MixingPlaneInterface(iMarkerDonor) == iMarkerInt ){
227  /*--- We have identified the local index of the Donor marker ---*/
228  /*--- Now we are going to store the average values that belong to Marker_Donor on each processor ---*/
229  /*--- Store the identifier for the structural marker ---*/
230  Marker_Donor = iMarkerDonor;
231  Donor_Flag = donor_config->GetMarker_All_TurbomachineryFlag(iMarkerDonor);
232  /*--- Exit the for loop: we have found the local index for Mixing-Plane interface ---*/
233  break;
234  }
235  /*--- If the tag hasn't matched any tag within the donor markers ---*/
236  Marker_Donor = -1;
237  Donor_Flag = -1;
238 
239  }
240 
241 #ifdef HAVE_MPI
242  auto BuffMarkerDonor = new int[size];
243  auto BuffDonorFlag = new int[size];
244  for (int iSize=0; iSize<size; iSize++){
245  BuffMarkerDonor[iSize] = -1;
246  BuffDonorFlag[iSize] = -1;
247  }
248 
249  SU2_MPI::Allgather(&Marker_Donor, 1 , MPI_INT, BuffMarkerDonor, 1, MPI_INT, SU2_MPI::GetComm());
250  SU2_MPI::Allgather(&Donor_Flag, 1 , MPI_INT, BuffDonorFlag, 1, MPI_INT, SU2_MPI::GetComm());
251 
252  Marker_Donor= -1;
253  Donor_Flag= -1;
254 
255  for (int iSize=0; iSize<size; iSize++){
256  if(BuffMarkerDonor[iSize] >= 0.0){
257  Marker_Donor = BuffMarkerDonor[iSize];
258  Donor_Flag = BuffDonorFlag[iSize];
259  break;
260  }
261  }
262  delete [] BuffMarkerDonor;
263  delete [] BuffDonorFlag;
264 #endif
265 
266  /*--- On the target side we have to identify the marker as well ---*/
267 
268  for (iMarkerTarget = 0; iMarkerTarget < nMarkerTarget; iMarkerTarget++){
269  /*--- If the tag GetMarker_All_MixingPlaneInterface(iMarkerTarget) equals the index we are looping at ---*/
270  if ( target_config->GetMarker_All_MixingPlaneInterface(iMarkerTarget) == iMarkerInt ){
271  /*--- Store the identifier for the fluid marker ---*/
272 
273  // here i should then store it in the target zone
274 
275  Marker_Target = iMarkerTarget;
276  Target_Flag = target_config->GetMarker_All_TurbomachineryFlag(iMarkerTarget);
277  /*--- Exit the for loop: we have found the local index for iMarkerFSI on the FEA side ---*/
278  break;
279  }
280  /*--- If the tag hasn't matched any tag within the Flow markers ---*/
281  Marker_Target = -1;
282  Target_Flag = -1;
283  }
284 
285  if (Marker_Target != -1 && Marker_Donor != -1){
286 
287  SpanValuesDonor = donor_geometry->GetSpanWiseValue(Donor_Flag);
288  SpanValuesTarget = target_geometry->GetSpanWiseValue(Target_Flag);
289 
290 
291  for(iSpan = 1; iSpan <nSpanTarget-1; iSpan++){
292  dist = 10E+06;
293  dist2 = 10E+06;
294  for (jSpan = 1; jSpan < nSpanDonor - 2; jSpan++) {
295  test = abs(SpanValuesTarget[iSpan] - SpanValuesDonor[jSpan]);
296  test2 = abs(SpanValuesTarget[iSpan] - SpanValuesDonor[jSpan]);
297  if(test < dist && SpanValuesTarget[iSpan] > SpanValuesDonor[jSpan]){
298  dist = test;
299  kSpan = jSpan;
300  }
301  if(test2 < dist2){
302  dist2 = test2;
303  tSpan = jSpan;
304  }
305 
306  }
307  switch(donor_config->GetKind_MixingPlaneInterface()){
308  case MATCHING:
309  SpanLevelDonor[iSpan] = iSpan;
310  SpanValueCoeffTarget[iSpan] = 0.0;
311  break;
312  case NEAREST_SPAN:
313  SpanLevelDonor[iSpan] = tSpan;
314  SpanValueCoeffTarget[iSpan] = 0.0;
315  break;
316  case LINEAR_INTERPOLATION:
317  SpanLevelDonor[iSpan] = kSpan;
318  SpanValueCoeffTarget[iSpan] = (SpanValuesTarget[iSpan] - SpanValuesDonor[kSpan])
319  /(SpanValuesDonor[kSpan + 1] - SpanValuesDonor[kSpan]);
320  break;
321  default:
322  SU2_MPI::Error("MixingPlane interface option not implemented yet", CURRENT_FUNCTION);
323  break;
324 
325  }
326  }
327  }
