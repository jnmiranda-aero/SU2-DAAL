41CIncEulerSolver::CIncEulerSolver(CGeometry *geometry, CConfig *config, unsigned short iMesh,
42 const bool navier_stokes) :
43 CFVMFlowSolverBase<CIncEulerVariable, ENUM_REGIME::INCOMPRESSIBLE>(*geometry, *config) {
44
45 /*--- Based on the navier_stokes boolean, determine if this constructor is
46 * being called by itself, or by its derived class CIncNSSolver. ---*/
47 const string description = navier_stokes? "Navier-Stokes" : "Euler";
48
49 unsigned short iMarker;
50 ifstream restart_file;
51 unsigned short nZone = geometry->GetnZone();
52 bool restart = (config->GetRestart() || config->GetRestart_Flow());
53 int Unst_RestartIter = 0;
54 unsigned short iZone = config->GetiZone();
55 bool dual_time = ((config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
56 (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND));
57 bool time_stepping = config->GetTime_Marching() == TIME_MARCHING::TIME_STEPPING;
58 bool adjoint = (config->GetContinuous_Adjoint()) || (config->GetDiscrete_Adjoint());
59
60 /* A grid is defined as dynamic if there's rigid grid movement or grid deformation AND the problem is time domain */
61 dynamic_grid = config->GetDynamic_Grid();
62
63 /*--- Store the multigrid level. ---*/
64 MGLevel = iMesh;
65
66 /*--- Check for a restart file to evaluate if there is a change in the angle of attack
67 before computing all the non-dimesional quantities. ---*/
68
69 if (restart && (iMesh == MESH_0) && nZone <= 1) {
70
71 /*--- Multizone problems require the number of the zone to be appended. ---*/
72
73 auto filename_ = config->GetSolution_FileName();
74
75 if (nZone > 1) filename_ = config->GetMultizone_FileName(filename_, iZone, ".dat");
76
77 /*--- Modify file name for a dual-time unsteady restart ---*/
78
79 if (dual_time) {
80 if (adjoint) Unst_RestartIter = SU2_TYPE::Int(config->GetUnst_AdjointIter())-1;
81 else if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST)
82 Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-1;
83 else Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-2;
84 filename_ = config->GetUnsteady_FileName(filename_, Unst_RestartIter, ".dat");
85 }
86
87 /*--- Modify file name for a time stepping unsteady restart ---*/
88
89 if (time_stepping) {
90 if (adjoint) Unst_RestartIter = SU2_TYPE::Int(config->GetUnst_AdjointIter())-1;
91 else Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-1;
92 filename_ = config->GetUnsteady_FileName(filename_, Unst_RestartIter, ".dat");
93 }
94
95 /*--- Read and store the restart metadata. ---*/
96
97 filename_ = "flow";
98 filename_ = config->GetFilename(filename_, ".meta", Unst_RestartIter);
99 Read_SU2_Restart_Metadata(geometry, config, adjoint, filename_);
100
101 }
102 if (restart && (config->GetKind_Streamwise_Periodic() == ENUM_STREAMWISE_PERIODIC::MASSFLOW)) {
103 string filename_ = "flow";
104 filename_ = config->GetFilename(filename_, ".meta", Unst_RestartIter);
105 Read_SU2_Restart_Metadata(geometry, config, adjoint, filename_);
106 if (rank==MASTER_NODE) cout << "Setting streamwise periodic pressure drop from restart metadata file." << endl;
107 }
108
109 /*--- Set the gamma value ---*/
110
111 Gamma = config->GetGamma();
112 Gamma_Minus_One = Gamma - 1.0;
113
114 /*--- Define geometry constants in the solver structure.
115 * Incompressible flow, primitive variables (P, vx, vy, vz, T, rho, beta, lamMu, EddyMu, Kt_eff, Cp, Cv) ---*/
116
117 nDim = geometry->GetnDim();
118
119 /*--- Make sure to align the sizes with the constructor of CIncEulerVariable. ---*/
120 nVar = nDim+2; nPrimVar = nDim+9; nPrimVarGrad = nDim+4;
121
122 /*--- Initialize nVarGrad for deallocation ---*/
123
124 nVarGrad = nPrimVarGrad;
125
126 nMarker = config->GetnMarker_All();
127 nPoint = geometry->GetnPoint();
128 nPointDomain = geometry->GetnPointDomain();
129
130 /*--- Store the number of vertices on each marker for deallocation later ---*/
131
132 nVertex.resize(nMarker);
133 for (iMarker = 0; iMarker < nMarker; iMarker++)
134 nVertex[iMarker] = geometry->nVertex[iMarker];
135
136 /*--- Perform the non-dimensionalization for the flow equations using the
137 specified reference values. ---*/
138
139 SetNondimensionalization(config, iMesh);
140
141 /*--- Check if we are executing a verification case. If so, the
142 VerificationSolution object will be instantiated for a particular
143 option from the available library of verification solutions. Note
144 that this is done after SetNondim(), as problem-specific initial
145 parameters are needed by the solution constructors. ---*/
146
147 SetVerificationSolution(nDim, nVar, config);
148
149 /*--- Allocate base class members. ---*/
150
151 Allocate(*config);
152
153 /*--- MPI + OpenMP initialization. ---*/
154
155 HybridParallelInitialization(*config, *geometry);
156
157 /*--- Jacobians and vector structures for implicit computations ---*/
158
159 if (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT) {
160
161 if (rank == MASTER_NODE)
162 cout << "Initialize Jacobian structure (" << description << "). MG level: " << iMesh <<"." << endl;
163
164 Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config, ReducerStrategy);
165 }
166 else {
167 if (rank == MASTER_NODE)
168 cout << "Explicit scheme. No Jacobian structure (" << description << "). MG level: " << iMesh <<"." << endl;
169 }
170
171 /*--- Read farfield conditions ---*/
172
173 Density_Inf = config->GetDensity_FreeStreamND();
174 Pressure_Inf = config->GetPressure_FreeStreamND();
175 Velocity_Inf = config->GetVelocity_FreeStreamND();
176 Temperature_Inf = config->GetTemperature_FreeStreamND();
177
178 /*--- Initialize the secondary values for direct derivative approxiations ---*/
179
180 switch (config->GetDirectDiff()) {
181 case NO_DERIVATIVE:
182 /*--- Default ---*/
183 break;
184 case D_DENSITY:
185 SU2_TYPE::SetDerivative(Density_Inf, 1.0);
186 break;
187 case D_PRESSURE:
188 SU2_TYPE::SetDerivative(Pressure_Inf, 1.0);
189 break;
190 case D_TEMPERATURE:
191 SU2_TYPE::SetDerivative(Temperature_Inf, 1.0);
192 break;
193 case D_MACH: case D_AOA:
194 case D_SIDESLIP: case D_REYNOLDS:
195 case D_TURB2LAM: case D_DESIGN:
196 /*--- Already done in postprocessing of config ---*/
197 break;
198 default:
199 break;
200 }
201
202 SetReferenceValues(*config);
203
204 /*--- Initialize the solution to the far-field state everywhere. ---*/
205
206 if (navier_stokes) {
207 nodes = new CIncNSVariable(Pressure_Inf, Velocity_Inf, Temperature_Inf, nPoint, nDim, nVar, config);
208 } else {
209 nodes = new CIncEulerVariable(Pressure_Inf, Velocity_Inf, Temperature_Inf, nPoint, nDim, nVar, config);
210 }
211 SetBaseClassPointerToNodes();
212
213 if (iMesh == MESH_0) {
214 nodes->NonPhysicalEdgeCounter.resize(geometry->GetnEdge()) = 0;
215 }
216
217 /*--- Initial comms. ---*/
218
219 CommunicateInitialState(geometry, config);
220
221 /*--- Sizing edge mass flux array ---*/
222 if (config->GetBounded_Scalar())
223 EdgeMassFluxes.resize(geometry->GetnEdge()) = su2double(0.0);
224
225 /*--- Add the solver name. ---*/
226 SolverName = "INC.FLOW";
227
228 /*--- Finally, check that the static arrays will be large enough (keep this
229 * check at the bottom to make sure we consider the "final" values). ---*/
230 if((nDim > MAXNDIM) || (nPrimVar > MAXNVAR))
231 SU2_MPI::Error("Oops! The CIncEulerSolver static array sizes are not large enough.", CURRENT_FUNCTION);
232}

235
236 for(auto& model : FluidModel) delete model;
237}

239void CIncEulerSolver::SetNondimensionalization(CConfig *config, unsigned short iMesh) {
240
241 su2double Temperature_FreeStream = 0.0, ModVel_FreeStream = 0.0,Energy_FreeStream = 0.0,
242 ModVel_FreeStreamND = 0.0, Omega_FreeStream = 0.0, Omega_FreeStreamND = 0.0, Viscosity_FreeStream = 0.0,
243 Density_FreeStream = 0.0, Pressure_FreeStream = 0.0, Pressure_Thermodynamic = 0.0, Tke_FreeStream = 0.0,
244 Length_Ref = 0.0, Density_Ref = 0.0, Pressure_Ref = 0.0, Temperature_Ref = 0.0, Velocity_Ref = 0.0, Time_Ref = 0.0,
245 Gas_Constant_Ref = 0.0, Omega_Ref = 0.0, Force_Ref = 0.0, Viscosity_Ref = 0.0, Conductivity_Ref = 0.0, Heat_Flux_Ref = 0.0, Energy_Ref= 0.0, Pressure_FreeStreamND = 0.0, Pressure_ThermodynamicND = 0.0, Density_FreeStreamND = 0.0,
246 Temperature_FreeStreamND = 0.0, Gas_ConstantND = 0.0, Specific_Heat_CpND = 0.0, Thermal_Expansion_CoeffND = 0.0,
247 Velocity_FreeStreamND[3] = {0.0, 0.0, 0.0}, Viscosity_FreeStreamND = 0.0,
248 Tke_FreeStreamND = 0.0, Energy_FreeStreamND = 0.0,
249 Total_UnstTimeND = 0.0, Delta_UnstTimeND = 0.0;
250
251 unsigned short iDim, iVar;
252
253 /*--- Local variables ---*/
254
255 su2double Mach = config->GetMach();
256 su2double Reynolds = config->GetReynolds();
257
258 bool unsteady = (config->GetTime_Marching() != TIME_MARCHING::STEADY);
259 bool viscous = config->GetViscous();
260 bool turbulent = ((config->GetKind_Solver() == MAIN_SOLVER::INC_RANS) ||
261 (config->GetKind_Solver() == MAIN_SOLVER::DISC_ADJ_INC_RANS));
262 bool tkeNeeded = ((turbulent) && ((config->GetKind_Turb_Model() == TURB_MODEL::SST)));
263 bool energy = config->GetEnergy_Equation();
264 bool boussinesq = (config->GetKind_DensityModel() == INC_DENSITYMODEL::BOUSSINESQ);
265
266 /*--- Compute dimensional free-stream values. ---*/
267
268 Density_FreeStream = config->GetInc_Density_Init(); config->SetDensity_FreeStream(Density_FreeStream);
269 Temperature_FreeStream = config->GetInc_Temperature_Init(); config->SetTemperature_FreeStream(Temperature_FreeStream);
270 Pressure_FreeStream = 0.0; config->SetPressure_FreeStream(Pressure_FreeStream);
271
272 /*--- The dimensional viscosity is needed to determine the free-stream conditions.
273 To accomplish this, simply set the non-dimensional coefficients to the
274 dimensional ones. This will be overruled later.---*/
275
276 config->SetTemperature_Ref(1.0);
277 config->SetViscosity_Ref(1.0);
278 config->SetConductivity_Ref(1.0);
279 config->SetGas_Constant_Ref(1.0);
280
281 ModVel_FreeStream = 0.0;
282 for (iDim = 0; iDim < nDim; iDim++) {
283 ModVel_FreeStream += config->GetInc_Velocity_Init()[iDim]*config->GetInc_Velocity_Init()[iDim];
284 config->SetVelocity_FreeStream(config->GetInc_Velocity_Init()[iDim],iDim);
285 }
286 ModVel_FreeStream = sqrt(ModVel_FreeStream); config->SetModVel_FreeStream(ModVel_FreeStream);
287
288 CFluidModel* auxFluidModel = nullptr;
289
290 switch (config->GetKind_FluidModel()) {
291
292 case CONSTANT_DENSITY:
293
294 auxFluidModel = new CConstantDensity(Density_FreeStream, config->GetSpecific_Heat_Cp());
295 auxFluidModel->SetTDState_T(Temperature_FreeStream);
296 break;
297
298 case INC_IDEAL_GAS:
299
300 config->SetGas_Constant(UNIVERSAL_GAS_CONSTANT/(config->GetMolecular_Weight()/1000.0));
301 Pressure_Thermodynamic = Density_FreeStream*Temperature_FreeStream*config->GetGas_Constant();
302 auxFluidModel = new CIncIdealGas(config->GetSpecific_Heat_Cp(), config->GetGas_Constant(), Pressure_Thermodynamic);
303 auxFluidModel->SetTDState_T(Temperature_FreeStream);
304 Pressure_Thermodynamic = auxFluidModel->GetPressure();
305 config->SetPressure_Thermodynamic(Pressure_Thermodynamic);
306 break;
307
308 case INC_IDEAL_GAS_POLY:
309
310 config->SetGas_Constant(UNIVERSAL_GAS_CONSTANT/(config->GetMolecular_Weight()/1000.0));
311 Pressure_Thermodynamic = Density_FreeStream*Temperature_FreeStream*config->GetGas_Constant();
312 auxFluidModel = new CIncIdealGasPolynomial<N_POLY_COEFFS>(config->GetGas_Constant(), Pressure_Thermodynamic);
313 if (viscous) {
314 /*--- Variable Cp model via polynomial. ---*/
315 for (iVar = 0; iVar < config->GetnPolyCoeffs(); iVar++)
316 config->SetCp_PolyCoeffND(config->GetCp_PolyCoeff(iVar), iVar);
317 auxFluidModel->SetCpModel(config);
318 }
319 auxFluidModel->SetTDState_T(Temperature_FreeStream);
320 Pressure_Thermodynamic = auxFluidModel->GetPressure();
321 config->SetPressure_Thermodynamic(Pressure_Thermodynamic);
322 break;
323
324 case FLUID_MIXTURE:
325
326 config->SetGas_Constant(UNIVERSAL_GAS_CONSTANT / (config->GetMolecular_Weight() / 1000.0));
327 Pressure_Thermodynamic = config->GetPressure_Thermodynamic();
328 auxFluidModel = new CFluidScalar(config->GetSpecific_Heat_Cp(), config->GetGas_Constant(), Pressure_Thermodynamic, config);
329 auxFluidModel->SetTDState_T(Temperature_FreeStream, config->GetSpecies_Init());
330 break;
331
332 case FLUID_FLAMELET:
333
334 config->SetGas_Constant(UNIVERSAL_GAS_CONSTANT / (config->GetMolecular_Weight() / 1000.0));
335 Pressure_Thermodynamic = config->GetPressure_Thermodynamic();
336 auxFluidModel = new CFluidFlamelet(config, Pressure_Thermodynamic);
337 config->SetPressure_Thermodynamic(Pressure_Thermodynamic);
338 auxFluidModel->SetTDState_T(Temperature_FreeStream, config->GetSpecies_Init());
339 break;
340
341 default:
342
343 SU2_MPI::Error("Fluid model not implemented for incompressible solver.", CURRENT_FUNCTION);
344 break;
345 }
346
347 if (viscous) {
348
349 for (iVar = 0; iVar < config->GetnPolyCoeffs(); iVar++)
350 config->SetMu_PolyCoeffND(config->GetMu_PolyCoeff(iVar), iVar);
351
352 /*--- Use the fluid model to compute the dimensional viscosity/conductivity. ---*/
353
354 auxFluidModel->SetLaminarViscosityModel(config);
355 Viscosity_FreeStream = auxFluidModel->GetLaminarViscosity();
356 config->SetViscosity_FreeStream(Viscosity_FreeStream);
357
358 Reynolds = Density_FreeStream*ModVel_FreeStream/Viscosity_FreeStream; config->SetReynolds(Reynolds);
359
360 /*--- Turbulence kinetic energy ---*/
361
362 Tke_FreeStream = 3.0/2.0*(ModVel_FreeStream*ModVel_FreeStream*config->GetTurbulenceIntensity_FreeStream()*config->GetTurbulenceIntensity_FreeStream());
363
364 }
365
366 /*--- The non-dim. scheme for incompressible flows uses the following ref. values:
367 Reference length = 1 m (fixed by default, grid in meters)
368 Reference density = liquid density or freestream (input)
369 Reference velocity = liquid velocity or freestream (input)
370 Reference temperature = liquid temperature or freestream (input)
371 Reference pressure = Reference density * Reference velocity * Reference velocity
372 Reference viscosity = Reference Density * Reference velocity * Reference length
373 This is the same non-dim. scheme as in the compressible solver.
374 Note that the Re and Re Length are not used as part of initialization. ---*/
375
376 if (config->GetRef_Inc_NonDim() == DIMENSIONAL) {
377 Density_Ref = 1.0;
378 Velocity_Ref = 1.0;
379 Temperature_Ref = 1.0;
380 Pressure_Ref = 1.0;
381 }
382 else if (config->GetRef_Inc_NonDim() == INITIAL_VALUES) {
383 Density_Ref = Density_FreeStream;
384 Velocity_Ref = ModVel_FreeStream;
385 Temperature_Ref = Temperature_FreeStream;
386 Pressure_Ref = Density_Ref*Velocity_Ref*Velocity_Ref;
387 }
388 else if (config->GetRef_Inc_NonDim() == REFERENCE_VALUES) {
389 Density_Ref = config->GetInc_Density_Ref();
390 Velocity_Ref = config->GetInc_Velocity_Ref();
391 Temperature_Ref = config->GetInc_Temperature_Ref();
392 Pressure_Ref = Density_Ref*Velocity_Ref*Velocity_Ref;
393 }
394 config->SetDensity_Ref(Density_Ref);
395 config->SetVelocity_Ref(Velocity_Ref);
396 config->SetTemperature_Ref(Temperature_Ref);
397 config->SetPressure_Ref(Pressure_Ref);
398
399 /*--- More derived reference values ---*/
400
401 Length_Ref = 1.0; config->SetLength_Ref(Length_Ref);
402 Time_Ref = Length_Ref/Velocity_Ref; config->SetTime_Ref(Time_Ref);
403 Omega_Ref = Velocity_Ref/Length_Ref; config->SetOmega_Ref(Omega_Ref);
404 Force_Ref = Velocity_Ref*Velocity_Ref/Length_Ref; config->SetForce_Ref(Force_Ref);
405 Heat_Flux_Ref = Density_Ref*Velocity_Ref*Velocity_Ref*Velocity_Ref; config->SetHeat_Flux_Ref(Heat_Flux_Ref);
406 Gas_Constant_Ref = Velocity_Ref*Velocity_Ref/Temperature_Ref; config->SetGas_Constant_Ref(Gas_Constant_Ref);
407 Viscosity_Ref = Density_Ref*Velocity_Ref*Length_Ref; config->SetViscosity_Ref(Viscosity_Ref);
408 Conductivity_Ref = Viscosity_Ref*Gas_Constant_Ref; config->SetConductivity_Ref(Conductivity_Ref);
409
410 /*--- Get the freestream energy. Only useful if energy equation is active. ---*/
411
412 Energy_FreeStream = auxFluidModel->GetStaticEnergy() + 0.5*ModVel_FreeStream*ModVel_FreeStream;
413 if (tkeNeeded) { Energy_FreeStream += Tke_FreeStream; };
414 config->SetEnergy_FreeStream(Energy_FreeStream);
415
416 /*--- Auxilary (dimensional) FluidModel no longer needed. ---*/
417 delete auxFluidModel;
418
419 /*--- Compute Mach number ---*/
420
421 if (config->GetKind_FluidModel() == CONSTANT_DENSITY) {
422 Mach = ModVel_FreeStream / sqrt(config->GetBulk_Modulus()/Density_FreeStream);
423 } else {
424 Mach = 0.0;
425 }
426 config->SetMach(Mach);
427
428 /*--- Divide by reference values, to compute the non-dimensional free-stream values ---*/
429
430 Pressure_FreeStreamND = Pressure_FreeStream/config->GetPressure_Ref(); config->SetPressure_FreeStreamND(Pressure_FreeStreamND);
431 Pressure_ThermodynamicND = Pressure_Thermodynamic/config->GetPressure_Ref(); config->SetPressure_ThermodynamicND(Pressure_ThermodynamicND);
432 Density_FreeStreamND = Density_FreeStream/config->GetDensity_Ref(); config->SetDensity_FreeStreamND(Density_FreeStreamND);
433
434 for (iDim = 0; iDim < nDim; iDim++) {
435 Velocity_FreeStreamND[iDim] = config->GetVelocity_FreeStream()[iDim]/Velocity_Ref; config->SetVelocity_FreeStreamND(Velocity_FreeStreamND[iDim], iDim);
436 }
437
438 Temperature_FreeStreamND = Temperature_FreeStream/config->GetTemperature_Ref(); config->SetTemperature_FreeStreamND(Temperature_FreeStreamND);
439 Gas_ConstantND = config->GetGas_Constant()/Gas_Constant_Ref; config->SetGas_ConstantND(Gas_ConstantND);
440 Specific_Heat_CpND = config->GetSpecific_Heat_CpND();
441
442 Thermal_Expansion_CoeffND = config->GetThermal_Expansion_Coeff()*config->GetTemperature_Ref(); config->SetThermal_Expansion_CoeffND(Thermal_Expansion_CoeffND);
443
444 ModVel_FreeStreamND = 0.0;
445 for (iDim = 0; iDim < nDim; iDim++) ModVel_FreeStreamND += Velocity_FreeStreamND[iDim]*Velocity_FreeStreamND[iDim];
446 ModVel_FreeStreamND = sqrt(ModVel_FreeStreamND); config->SetModVel_FreeStreamND(ModVel_FreeStreamND);
447
448 Viscosity_FreeStreamND = Viscosity_FreeStream / Viscosity_Ref; config->SetViscosity_FreeStreamND(Viscosity_FreeStreamND);
449
450 Tke_FreeStream = 3.0/2.0*(ModVel_FreeStream*ModVel_FreeStream*config->GetTurbulenceIntensity_FreeStream()*config->GetTurbulenceIntensity_FreeStream());
451 config->SetTke_FreeStream(Tke_FreeStream);
452
453 Tke_FreeStreamND = 3.0/2.0*(ModVel_FreeStreamND*ModVel_FreeStreamND*config->GetTurbulenceIntensity_FreeStream()*config->GetTurbulenceIntensity_FreeStream());
454 config->SetTke_FreeStreamND(Tke_FreeStreamND);
455
456 Omega_FreeStream = Density_FreeStream*Tke_FreeStream/(Viscosity_FreeStream*config->GetTurb2LamViscRatio_FreeStream());
457 config->SetOmega_FreeStream(Omega_FreeStream);
458
459 Omega_FreeStreamND = Density_FreeStreamND*Tke_FreeStreamND/(Viscosity_FreeStreamND*config->GetTurb2LamViscRatio_FreeStream());
460 config->SetOmega_FreeStreamND(Omega_FreeStreamND);
461
462 const su2double MassDiffusivityND = config->GetDiffusivity_Constant() / (Velocity_Ref * Length_Ref);
463 config->SetDiffusivity_ConstantND(MassDiffusivityND);
464
465 /*--- Create one final fluid model object per OpenMP thread to be able to use them in parallel.
466 * GetFluidModel() should be used to automatically access the "right" object of each thread. ---*/
467
468 assert(FluidModel.empty() && "Potential memory leak!");
469 FluidModel.resize(omp_get_max_threads());
470
471 for (auto& fluidModel : FluidModel) {
472
473 switch (config->GetKind_FluidModel()) {
474
475 case CONSTANT_DENSITY:
476 fluidModel = new CConstantDensity(Density_FreeStreamND, Specific_Heat_CpND);
477 break;
478
479 case INC_IDEAL_GAS:
480 fluidModel = new CIncIdealGas(Specific_Heat_CpND, Gas_ConstantND, Pressure_ThermodynamicND);
481 fluidModel->SetTDState_T(Temperature_FreeStreamND);
482 break;
483
484 case FLUID_MIXTURE:
485 fluidModel = new CFluidScalar(Specific_Heat_CpND, Gas_ConstantND, Pressure_ThermodynamicND, config);
486 fluidModel->SetTDState_T(Temperature_FreeStreamND, config->GetSpecies_Init());
487 break;
488
489 case FLUID_FLAMELET:
490 fluidModel = new CFluidFlamelet(config, Pressure_Thermodynamic);
491 fluidModel->SetTDState_T(Temperature_FreeStreamND, config->GetSpecies_Init());
492 break;
493
494 case INC_IDEAL_GAS_POLY:
495 fluidModel = new CIncIdealGasPolynomial<N_POLY_COEFFS>(Gas_ConstantND, Pressure_ThermodynamicND);
496 if (viscous) {
497 /*--- Variable Cp model via polynomial. ---*/
498 config->SetCp_PolyCoeffND(config->GetCp_PolyCoeff(0)/Gas_Constant_Ref, 0);
499 for (iVar = 1; iVar < config->GetnPolyCoeffs(); iVar++)
500 config->SetCp_PolyCoeffND(config->GetCp_PolyCoeff(iVar)*pow(Temperature_Ref,iVar)/Gas_Constant_Ref, iVar);
501 fluidModel->SetCpModel(config);
502 }
503 fluidModel->SetTDState_T(Temperature_FreeStreamND);
504 break;
505
506 }
507
508 if (viscous) {
509
510 /*--- Viscosity model via polynomial. ---*/
511
512 config->SetMu_PolyCoeffND(config->GetMu_PolyCoeff(0)/Viscosity_Ref, 0);
513 for (iVar = 1; iVar < config->GetnPolyCoeffs(); iVar++)
514 config->SetMu_PolyCoeffND(config->GetMu_PolyCoeff(iVar)*pow(Temperature_Ref,iVar)/Viscosity_Ref, iVar);
515
516 /*--- Conductivity model via polynomial. ---*/
517

922void CIncEulerSolver::SetReferenceValues(const CConfig& config) {
923
924 /*--- Evaluate reference values for non-dimensionalization. For dimensional or non-dim
925 based on initial values, use the far-field state (inf). For a custom non-dim based
926 on user-provided reference values, use the ref values to compute the forces. ---*/
927
928 su2double RefDensity, RefVel2;
929
930 if ((config.GetRef_Inc_NonDim() == DIMENSIONAL) ||
931 (config.GetRef_Inc_NonDim() == INITIAL_VALUES)) {
932 RefDensity = Density_Inf;
933 RefVel2 = GeometryToolbox::SquaredNorm(nDim, Velocity_Inf);
934 }
935 else {
936 RefDensity = config.GetInc_Density_Ref();
937 RefVel2 = pow(config.GetInc_Velocity_Ref(), 2);
938 }
939
940 DynamicPressureRef = 0.5 * RefDensity * RefVel2;
941 AeroCoeffForceRef = DynamicPressureRef * config.GetRefArea();
942
943}

945void CIncEulerSolver::CommonPreprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh,
946 unsigned short iRKStep, unsigned short RunTime_EqSystem, bool Output) {
947
948 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
949 const bool center = (config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED);
950 const bool center_jst = (config->GetKind_Centered_Flow() == CENTERED::JST) && (iMesh == MESH_0);
951 const bool outlet = (config->GetnMarker_Outlet() != 0);
952
953 /*--- Set the primitive variables ---*/
954
955 ompMasterAssignBarrier(ErrorCounter, 0);
956
957 SU2_OMP_ATOMIC
958 ErrorCounter += SetPrimitive_Variables(solver_container, config);
959
960 if ((iMesh == MESH_0) && (config->GetComm_Level() == COMM_FULL)) {
961 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
962 {
963 unsigned long tmp = ErrorCounter;
964 SU2_MPI::Allreduce(&tmp, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
965 config->SetNonphysical_Points(ErrorCounter);
966 }
967 END_SU2_OMP_SAFE_GLOBAL_ACCESS
968 }
969
970 /*--- Artificial dissipation ---*/
971
972 if (center && !Output) {
973 SetMax_Eigenvalue(geometry, config);
974 if (center_jst) {
975 SetCentered_Dissipation_Sensor(geometry, config);
976 SetUndivided_Laplacian(geometry, config);
977 }
978 }
979
980 /*--- Update the beta value based on the maximum velocity. ---*/
981
982 SetBeta_Parameter(geometry, solver_container, config, iMesh);
983
984 /*--- Compute properties needed for mass flow BCs. ---*/
985
986 if (outlet) {
987 SU2_OMP_SAFE_GLOBAL_ACCESS(GetOutlet_Properties(geometry, config, iMesh, Output);)
988 }
989
990 /*--- Initialize the Jacobian matrix and residual, not needed for the reducer strategy
991 * as we set blocks (including diagonal ones) and completely overwrite. ---*/
992
993 if(!ReducerStrategy && !Output) {
994 LinSysRes.SetValZero();
995 if (implicit) Jacobian.SetValZero();
996 else {SU2_OMP_BARRIER} // because of "nowait" in LinSysRes
997 }

1057void CIncEulerSolver::SetTime_Step(CGeometry *geometry, CSolver **solver_container, CConfig *config,
1058 unsigned short iMesh, unsigned long Iteration) {
1059
1060 /*--- Define an object to compute the speed of sound. ---*/
1061 struct SoundSpeed {
1062 FORCEINLINE su2double operator() (const CIncEulerVariable& nodes, unsigned long iPoint, unsigned long jPoint) const {
1063 return sqrt(0.5 * (nodes.GetBetaInc2(iPoint) + nodes.GetBetaInc2(jPoint)));
1064 }
1065
1066 FORCEINLINE su2double operator() (const CIncEulerVariable& nodes, unsigned long iPoint) const {
1067 return sqrt(nodes.GetBetaInc2(iPoint));
1068 }
1069
1070 } soundSpeed;
1071
1072 /*--- Define an object to compute the viscous eigenvalue. ---*/
1073 struct LambdaVisc {
1074 const bool energy;
1075
1076 LambdaVisc(bool e) : energy(e) {}
1077
1078 FORCEINLINE su2double lambda(su2double lamVisc, su2double eddyVisc, su2double rho, su2double k, su2double cv) const {
1079 su2double Lambda_1 = (4.0/3.0)*(lamVisc + eddyVisc);
1080 su2double Lambda_2 = 0.0;
1081 if (energy) Lambda_2 = k / cv;
1082 return (Lambda_1 + Lambda_2) / rho;
1083 }
1084
1085 FORCEINLINE su2double operator() (const CIncEulerVariable& nodes, unsigned long iPoint, unsigned long jPoint) const {
1086 su2double thermalCond = 0.5*(nodes.GetThermalConductivity(iPoint) + nodes.GetThermalConductivity(jPoint));
1087 su2double laminarVisc = 0.5*(nodes.GetLaminarViscosity(iPoint) + nodes.GetLaminarViscosity(jPoint));
1088 su2double eddyVisc = 0.5*(nodes.GetEddyViscosity(iPoint) + nodes.GetEddyViscosity(jPoint));
1089 su2double density = 0.5*(nodes.GetDensity(iPoint) + nodes.GetDensity(jPoint));
1090 su2double cv = 0.5*(nodes.GetSpecificHeatCv(iPoint) + nodes.GetSpecificHeatCv(jPoint));
1091 return lambda(laminarVisc, eddyVisc, density, thermalCond, cv);
1092 }
1093
1094 FORCEINLINE su2double operator() (const CIncEulerVariable& nodes, unsigned long iPoint) const {
1095 su2double thermalCond = nodes.GetThermalConductivity(iPoint);
1096 su2double laminarVisc = nodes.GetLaminarViscosity(iPoint);
1097 su2double eddyVisc = nodes.GetEddyViscosity(iPoint);
1098 su2double density = nodes.GetDensity(iPoint);
1099 su2double cv = nodes.GetSpecificHeatCv(iPoint);
1100 return lambda(laminarVisc, eddyVisc, density, thermalCond, cv);
1101 }
1102
1103 } lambdaVisc(config->GetEnergy_Equation());
1104
1105 /*--- Now instantiate the generic implementation with the two functors above. ---*/
1106
1107 SetTime_Step_impl(soundSpeed, lambdaVisc, geometry, solver_container, config, iMesh, Iteration);
1108
1109}

1206void CIncEulerSolver::Upwind_Residual(CGeometry *geometry, CSolver **solver_container,
1207 CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
1208
1209 CNumerics* numerics = numerics_container[CONV_TERM + omp_get_thread_num()*MAX_TERMS];
1210
1211 /*--- Static arrays of MUSCL-reconstructed primitives and secondaries (thread safety). ---*/
1212 su2double Primitive_i[MAXNVAR] = {0.0}, Primitive_j[MAXNVAR] = {0.0};
1213
1214 unsigned long iPoint, jPoint, counter_local = 0;
1215 unsigned short iDim, iVar;
1216
1217 SU2_OMP_MASTER
1218 ErrorCounter = 0;
1219 END_SU2_OMP_MASTER
1220
1221 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1222 const bool muscl = (config->GetMUSCL_Flow() && (iMesh == MESH_0));
1223 const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE);
1224 const bool van_albada = (config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE);
1225 const bool bounded_scalar = config->GetBounded_Scalar();
1226
1227 /*--- For hybrid parallel AD, pause preaccumulation if there is shared reading of
1228 * variables, otherwise switch to the faster adjoint evaluation mode. ---*/
1229 bool pausePreacc = false;
1230 if (ReducerStrategy) pausePreacc = AD::PausePreaccumulation();
1231 else AD::StartNoSharedReading();
1232
1233 /*--- Loop over edge colors. ---*/
1234 for (auto color : EdgeColoring)
1235 {
1236 /*--- Chunk size is at least OMP_MIN_SIZE and a multiple of the color group size. ---*/
1237 SU2_OMP_FOR_DYN(nextMultiple(OMP_MIN_SIZE, color.groupSize))
1238 for(auto k = 0ul; k < color.size; ++k) {
1239
1240 auto iEdge = color.indices[k];
1241
1242 /*--- Points in edge and normal vectors ---*/
1243
1244 iPoint = geometry->edges->GetNode(iEdge,0); jPoint = geometry->edges->GetNode(iEdge,1);
1245 numerics->SetNormal(geometry->edges->GetNormal(iEdge));
1246
1247 /*--- Grid movement ---*/
1248
1249 if (dynamic_grid)
1250 numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(jPoint));
1251
1252 /*--- Get primitive variables ---*/
1253
1254 auto V_i = nodes->GetPrimitive(iPoint);
1255 auto V_j = nodes->GetPrimitive(jPoint);
1256
1257 /*--- High order reconstruction using MUSCL strategy ---*/
1258
1259 if (muscl) {
1260
1261 auto Coord_i = geometry->nodes->GetCoord(iPoint);
1262 auto Coord_j = geometry->nodes->GetCoord(jPoint);
1263
1264 su2double Vector_ij[MAXNDIM] = {0.0};
1265 for (iDim = 0; iDim < nDim; iDim++) {
1266 Vector_ij[iDim] = 0.5*(Coord_j[iDim] - Coord_i[iDim]);
1267 }
1268
1269 auto Gradient_i = nodes->GetGradient_Reconstruction(iPoint);
1270 auto Gradient_j = nodes->GetGradient_Reconstruction(jPoint);
1271
1272 for (iVar = 0; iVar < nPrimVarGrad; iVar++) {
1273
1274 su2double Project_Grad_i = 0.0;
1275 su2double Project_Grad_j = 0.0;
1276
1277 for (iDim = 0; iDim < nDim; iDim++) {
1278 Project_Grad_i += Vector_ij[iDim]*Gradient_i[iVar][iDim];
1279 Project_Grad_j -= Vector_ij[iDim]*Gradient_j[iVar][iDim];
1280 }
1281
1282 su2double lim_i = 1.0;
1283 su2double lim_j = 1.0;
1284
1285 if (van_albada) {
1286 su2double V_ij = V_j[iVar] - V_i[iVar];
1287 lim_i = LimiterHelpers<>::vanAlbadaFunction(Project_Grad_i, V_ij, EPS);
1288 lim_j = LimiterHelpers<>::vanAlbadaFunction(-Project_Grad_j, V_ij, EPS);
1289 }
1290 else if (limiter) {
1291 lim_i = nodes->GetLimiter_Primitive(iPoint, iVar);
1292 lim_j = nodes->GetLimiter_Primitive(jPoint, iVar);
1293 }
1294
1295 Primitive_i[iVar] = V_i[iVar] + lim_i * Project_Grad_i;
1296 Primitive_j[iVar] = V_j[iVar] + lim_j * Project_Grad_j;
1297 }
1298
1299 for (iVar = nPrimVarGrad; iVar < nPrimVar; iVar++) {
1300 Primitive_i[iVar] = V_i[iVar];
1301 Primitive_j[iVar] = V_j[iVar];
1302 }
1303
1304 /*--- Check for non-physical solutions after reconstruction. If found,
1305 use the cell-average value of the solution. This results in a locally
1306 first-order approximation, but this is typically only active
1307 during the start-up of a calculation or difficult transients. For
1308 incompressible flow, only the temperature and density need to be
1309 checked. Pressure is the dynamic pressure (can be negative). ---*/
1310
1311 if (config->GetEnergy_Equation()) {
1312 const bool neg_temperature_i = (Primitive_i[prim_idx.Temperature()] < 0.0);
1313 const bool neg_temperature_j = (Primitive_j[prim_idx.Temperature()] < 0.0);
1314
1315 const bool neg_density_i = (Primitive_i[prim_idx.Density()] < 0.0);
1316 const bool neg_density_j = (Primitive_j[prim_idx.Density()] < 0.0);
1317
1318 bool bad_recon = neg_temperature_i || neg_temperature_j || neg_density_i || neg_density_j;
1319 bad_recon = nodes->UpdateNonPhysicalEdgeCounter(iEdge, bad_recon);
1320 counter_local += bad_recon;
1321
1322 if (bad_recon) {
1323 for (iVar = 0; iVar < nPrimVar; iVar++) {
1324 Primitive_i[iVar] = V_i[iVar];
1325 Primitive_j[iVar] = V_j[iVar];
1326 }
1327 }
1328 }
1329
1330 numerics->SetPrimitive(Primitive_i, Primitive_j);
1331
1332 } else {
1333
1334 /*--- Set primitive variables without reconstruction ---*/
1335
1336 numerics->SetPrimitive(V_i, V_j);
1337
1338 }
1339
1340 /*--- Compute the residual ---*/
1341
1342 auto residual = numerics->ComputeResidual(config);
1343
1344 if (bounded_scalar) EdgeMassFluxes[iEdge] = residual[0];
1345
1346 /*--- Update residual value ---*/
1347
1348 if (ReducerStrategy) {
1349 EdgeFluxes.SetBlock(iEdge, residual);
1350 if (implicit)
1351 Jacobian.SetBlocks(iEdge, residual.jacobian_i, residual.jacobian_j);
1352 }
1353 else {
1354 LinSysRes.AddBlock(iPoint, residual);
1355 LinSysRes.SubtractBlock(jPoint, residual);
1356
1357 /*--- Set implicit computation ---*/
1358 if (implicit)
1359 Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
1360 }
1361
1362 /*--- Viscous contribution. ---*/
1363
1364 Viscous_Residual(iEdge, geometry, solver_container,
1365 numerics_container[VISC_TERM + omp_get_thread_num()*MAX_TERMS], config);
1366
1367 }
1368 END_SU2_OMP_FOR
1369 } // end color loop
1370
1371 FinalizeResidualComputation(geometry, pausePreacc, counter_local, config);
1372}

1971void CIncEulerSolver::SetPreconditioner(const CConfig *config, unsigned long iPoint,
1972 su2double delta, su2activematrix& Preconditioner) const {
1973
1974 unsigned short iDim, jDim, iVar, jVar;
1975
1976 su2double BetaInc2, Density, dRhodT, Temperature, oneOverCp, Cp;
1977 su2double Velocity[MAXNDIM] = {0.0};
1978
1979 bool variable_density = (config->GetVariable_Density_Model());
1980 bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1981 bool energy = config->GetEnergy_Equation();
1982
1983 /*--- Access the primitive variables at this node. ---*/
1984
1985 Density = nodes->GetDensity(iPoint);
1986 BetaInc2 = nodes->GetBetaInc2(iPoint);
1987 Cp = nodes->GetSpecificHeatCp(iPoint);
1988 oneOverCp = 1.0/Cp;
1989 Temperature = nodes->GetTemperature(iPoint);
1990
1991 for (iDim = 0; iDim < nDim; iDim++)
1992 Velocity[iDim] = nodes->GetVelocity(iPoint,iDim);
1993
1994 /*--- We need the derivative of the equation of state to build the
1995 preconditioning matrix. For now, the only option is the ideal gas
1996 law, but in the future, dRhodT should be in the fluid model. ---*/
1997
1998 if (variable_density) {
1999 dRhodT = -Density/Temperature;
2000 } else {
2001 dRhodT = 0.0;
2002 }
2003
2004 /*--- Calculating the inverse of the preconditioning matrix
2005 that multiplies the time derivative during time integration. ---*/
2006
2007 if (implicit) {
2008
2009 /*--- For implicit calculations, we multiply the preconditioner
2010 by the cell volume over the time step and add to the Jac diagonal. ---*/
2011
2012 Preconditioner[0][0] = 1.0/BetaInc2;
2013 for (iDim = 0; iDim < nDim; iDim++)
2014 Preconditioner[iDim+1][0] = Velocity[iDim]/BetaInc2;
2015
2016 if (energy) Preconditioner[nDim+1][0] = Cp*Temperature/BetaInc2;
2017 else Preconditioner[nDim+1][0] = 0.0;
2018
2019 for (jDim = 0; jDim < nDim; jDim++) {
2020 Preconditioner[0][jDim+1] = 0.0;
2021 for (iDim = 0; iDim < nDim; iDim++) {
2022 if (iDim == jDim) Preconditioner[iDim+1][jDim+1] = Density;
2023 else Preconditioner[iDim+1][jDim+1] = 0.0;
2024 }
2025 Preconditioner[nDim+1][jDim+1] = 0.0;
2026 }
2027
2028 Preconditioner[0][nDim+1] = dRhodT;
2029 for (iDim = 0; iDim < nDim; iDim++)
2030 Preconditioner[iDim+1][nDim+1] = Velocity[iDim]*dRhodT;
2031
2032 if (energy) Preconditioner[nDim+1][nDim+1] = Cp*(dRhodT*Temperature + Density);
2033 else Preconditioner[nDim+1][nDim+1] = 1.0;
2034
2035 for (iVar = 0; iVar < nVar; iVar ++ )
2036 for (jVar = 0; jVar < nVar; jVar ++ )
2037 Preconditioner[iVar][jVar] = delta*Preconditioner[iVar][jVar];
2038
2039 } else {
2040
2041 /*--- For explicit calculations, we move the residual to the
2042 right-hand side and pre-multiply by the preconditioner inverse.
2043 Therefore, we build inv(Precon) here and multiply by the residual
2044 later in the R-K and Euler Explicit time integration schemes. ---*/
2045
2046 Preconditioner[0][0] = Temperature*BetaInc2*dRhodT/Density + BetaInc2;
2047 for (iDim = 0; iDim < nDim; iDim ++)
2048 Preconditioner[iDim+1][0] = -1.0*Velocity[iDim]/Density;
2049
2050 if (energy) Preconditioner[nDim+1][0] = -1.0*Temperature/Density;
2051 else Preconditioner[nDim+1][0] = 0.0;
2052
2053
2054 for (jDim = 0; jDim < nDim; jDim++) {
2055 Preconditioner[0][jDim+1] = 0.0;
2056 for (iDim = 0; iDim < nDim; iDim++) {
2057 if (iDim == jDim) Preconditioner[iDim+1][jDim+1] = 1.0/Density;
2058 else Preconditioner[iDim+1][jDim+1] = 0.0;
2059 }
2060 Preconditioner[nDim+1][jDim+1] = 0.0;
2061 }
2062
2063 Preconditioner[0][nDim+1] = -1.0*BetaInc2*dRhodT*oneOverCp/Density;
2064 for (iDim = 0; iDim < nDim; iDim ++)
2065 Preconditioner[iDim+1][nDim+1] = 0.0;
2066
2067 if (energy) Preconditioner[nDim+1][nDim+1] = oneOverCp/Density;
2068 else Preconditioner[nDim+1][nDim+1] = 0.0;
2069
2070 }
2071
2072}

2657void CIncEulerSolver::SetResidual_DualTime(CGeometry *geometry, CSolver **solver_container, CConfig *config,
2658 unsigned short iRKStep, unsigned short iMesh, unsigned short RunTime_EqSystem) {
2659
2660 /*--- Local variables ---*/
2661
2662 unsigned short iVar, iMarker, iDim, iNeigh;
2663 unsigned long iPoint, jPoint, iEdge, iVertex;
2664
2665 const su2double *V_time_nM1 = nullptr, *V_time_n = nullptr, *V_time_nP1 = nullptr;
2666 su2double U_time_nM1[MAXNVAR], U_time_n[MAXNVAR], U_time_nP1[MAXNVAR];
2667 su2double Volume_nM1, Volume_nP1, TimeStep;
2668 const su2double *Normal = nullptr, *GridVel_i = nullptr, *GridVel_j = nullptr;
2669 su2double Density, Cp;
2670
2671 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
2672 const bool first_order = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
2673 const bool second_order = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
2674 const bool energy = config->GetEnergy_Equation();
2675
2676 const int ndim = nDim;
2677 auto V2U = [ndim](su2double Density, su2double Cp, const su2double* V, su2double* U) {
2678 U[0] = Density;
2679 for (int iDim = 0; iDim < ndim; iDim++) U[iDim+1] = Density*V[iDim+1];
2680 U[ndim+1] = Density*Cp*V[ndim+1];
2681 };
2682
2683 /*--- Store the physical time step ---*/
2684
2685 TimeStep = config->GetDelta_UnstTimeND();
2686
2687 /*--- Compute the dual time-stepping source term for static meshes ---*/
2688
2689 if (!dynamic_grid) {
2690
2691 /*--- Loop over all nodes (excluding halos) ---*/
2692
2693 AD::StartNoSharedReading();
2694
2695 SU2_OMP_FOR_STAT(omp_chunk_size)
2696 for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
2697
2698 /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
2699 we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
2700 previous solutions that are stored in memory. These are actually
2701 the primitive values, but we will convert to conservatives. ---*/
2702
2703 V_time_nM1 = nodes->GetSolution_time_n1(iPoint);
2704 V_time_n = nodes->GetSolution_time_n(iPoint);
2705 V_time_nP1 = nodes->GetSolution(iPoint);
2706
2707 /*--- Access the density and Cp at this node (constant for now). ---*/
2708
2709 Density = nodes->GetDensity(iPoint);
2710 Cp = nodes->GetSpecificHeatCp(iPoint);
2711
2712 /*--- Compute the conservative variable vector for all time levels. ---*/
2713
2714 V2U(Density, Cp, V_time_nM1, U_time_nM1);
2715 V2U(Density, Cp, V_time_n, U_time_n);
2716 V2U(Density, Cp, V_time_nP1, U_time_nP1);
2717
2718 /*--- CV volume at time n+1. As we are on a static mesh, the volume
2719 of the CV will remained fixed for all time steps. ---*/
2720
2721 Volume_nP1 = geometry->nodes->GetVolume(iPoint);
2722
2723 /*--- Compute the dual time-stepping source term based on the chosen
2724 time discretization scheme (1st- or 2nd-order).---*/
2725
2726 for (iVar = 0; iVar < nVar-!energy; iVar++) {
2727 if (first_order)
2728 LinSysRes(iPoint,iVar) += (U_time_nP1[iVar] - U_time_n[iVar])*Volume_nP1 / TimeStep;
2729 if (second_order)
2730 LinSysRes(iPoint,iVar) += ( 3.0*U_time_nP1[iVar] - 4.0*U_time_n[iVar]
2731 +1.0*U_time_nM1[iVar])*Volume_nP1 / (2.0*TimeStep);
2732 }
2733
2734 /*--- Compute the Jacobian contribution due to the dual time source term. ---*/
2735
2736 if (implicit) {
2737 su2double delta = (second_order? 1.5 : 1.0) * Volume_nP1 * Density / TimeStep;
2738
2739 for (iDim = 0; iDim < nDim; iDim++)
2740 Jacobian.AddVal2Diag(iPoint, iDim+1, delta);
2741
2742 if (energy) delta *= Cp;
2743 Jacobian.AddVal2Diag(iPoint, nDim+1, delta);
2744 }
2745 }
2746 END_SU2_OMP_FOR
2747
2748 AD::EndNoSharedReading();
2749 }
2750
2751 else {
2752
2753 /*--- For unsteady flows on dynamic meshes (rigidly transforming or
2754 dynamically deforming), the Geometric Conservation Law (GCL) should be
2755 satisfied in conjunction with the ALE formulation of the governing
2756 equations. The GCL prevents accuracy issues caused by grid motion, i.e.
2757 a uniform free-stream should be preserved through a moving grid. First,
2758 we will loop over the edges and boundaries to compute the GCL component
2759 of the dual time source term that depends on grid velocities. ---*/
2760
2761 SU2_OMP_FOR_STAT(omp_chunk_size)
2762 for (iPoint = 0; iPoint < nPointDomain; ++iPoint) {
2763
2764 /*--- Compute the conservative variables. ---*/
2765
2766 V_time_n = nodes->GetSolution_time_n(iPoint);
2767 Density = nodes->GetDensity(iPoint);
2768 Cp = nodes->GetSpecificHeatCp(iPoint);
2769 V2U(Density, Cp, V_time_n, U_time_n);
2770
2771 GridVel_i = geometry->nodes->GetGridVel(iPoint);
2772
2773 for (iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); iNeigh++) {
2774
2775 iEdge = geometry->nodes->GetEdge(iPoint, iNeigh);
2776 Normal = geometry->edges->GetNormal(iEdge);
2777
2778 jPoint = geometry->nodes->GetPoint(iPoint, iNeigh);
2779 GridVel_j = geometry->nodes->GetGridVel(jPoint);
2780
2781 /*--- Determine whether to consider the normal outward or inward. ---*/
2782 su2double dir = (iPoint < jPoint)? 0.5 : -0.5;
2783
2784 su2double Residual_GCL = 0.0;
2785 for (iDim = 0; iDim < nDim; iDim++)
2786 Residual_GCL += dir*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
2787
2788 for (iVar = 0; iVar < nVar-!energy; iVar++)
2789 LinSysRes(iPoint,iVar) += U_time_n[iVar]*Residual_GCL;
2790 }
2791 }
2792 END_SU2_OMP_FOR
2793
2794 /*--- Loop over the boundary edges ---*/
