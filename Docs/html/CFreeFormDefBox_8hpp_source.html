460  /*!
461  * \brief Set the original value of the control points.
462  */
463  inline void SetOriginalControlPoints() {
464  for (unsigned short iDegree = 0; iDegree <= lDegree_Copy; iDegree++)
465  for (unsigned short jDegree = 0; jDegree <= mDegree_Copy; jDegree++)
466  for (unsigned short kDegree = 0; kDegree <= nDegree_Copy; kDegree++)
467  for (unsigned short iDim = 0; iDim < nDim; iDim++)
468  Coord_Control_Points[iDegree][jDegree][kDegree][iDim] =
469  Coord_Control_Points_Copy[iDegree][jDegree][kDegree][iDim];
470 
471  lDegree = lDegree_Copy;
472  mDegree = mDegree_Copy;
473  nDegree = nDegree_Copy;
474  lOrder = lOrder_Copy;
475  mOrder = mOrder_Copy;
476  nOrder = nOrder_Copy;
477  nControlPoints = nControlPoints_Copy;
478  }
