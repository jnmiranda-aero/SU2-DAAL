55 /*!
56 * \brief Constructor of the class.
57 * \note This class is not meant to be instantiated directly, it is only a building block.
58 * \param[in] npoint - Number of points/nodes/vertices in the domain.
59 * \param[in] ndim - Number of dimensions.
60 * \param[in] nvar - Number of variables.
61 * \param[in] nprimvar - Number of primitive variables.
62 * \param[in] nprimvargrad - Number of primitive variables for which to compute gradients.
63 * \param[in] config - Definition of the particular problem.

65 CFlowVariable(unsigned long npoint, unsigned long ndim, unsigned long nvar, unsigned long nprimvar,
66 unsigned long nprimvargrad, const CConfig* config);

68 public:
69 mutable su2vector<int8_t> NonPhysicalEdgeCounter; /*!< \brief Non-physical reconstruction counter for each edge. */
70 /*!
71 * \brief Updates the non-physical counter of an edge.
72 * \param[in] iEdge - Edge index.
73 * \param[in] isNonPhys - Should be 1 (true) if a non-physical reconstruction was occurred.
74 * \return Whether the reconstruction should be limited to first order, based on the counter.
75 */
76 template <class T>
77 inline T UpdateNonPhysicalEdgeCounter(unsigned long iEdge, const T& isNonPhys) const {
78 if (isNonPhys != 0) {
79 /*--- Force 1st order for this edge for at least 20 iterations. ---*/
80 NonPhysicalEdgeCounter[iEdge] = 21;
81 }
82 NonPhysicalEdgeCounter[iEdge] = std::max<int8_t>(0, NonPhysicalEdgeCounter[iEdge] - 1);
83 return static_cast<T>(NonPhysicalEdgeCounter[iEdge] > 0);
84 }
85

120 /*!
121 * \brief Set the value of the primitive variables.

126 inline void SetPrimitive(unsigned long iPoint, const su2double* val_prim) final {
