71 /*!
72 * \brief Get the global index of a point.
73 * \param[in] iPoint - the local renumbered point ID.
74 * \return Global index of a specific point.
75 */
76 unsigned long GetGlobalIndex(unsigned long iPoint) const override{
77 if (iPoint > nPoints)
78 SU2_MPI::Error(string("Local renumbered iPoint ID ") + to_string(iPoint) +
79 string(" is larger than max number of nodes ") + to_string(nPoints), CURRENT_FUNCTION);
80
81 return Renumber2Global.at(iPoint);

83
84 /*!
85 * \brief Get the beginning global renumbered node ID of the linear partition owned by a specific processor.
86 * \param[in] rank - the processor rank.
87 * \return The beginning global renumbered node ID.
88 */
89 unsigned long GetNodeBegin(unsigned short rank) const override {
90 return nPoint_Recv[rank];
91 }
92
93 /*!
94 * \brief Get the Processor ID a Point belongs to.
95 * \param[in] iPoint - global renumbered ID of the point
96 * \return The rank/processor number.
97 */
98 unsigned short FindProcessor(unsigned long iPoint) const override {
99
100 for (unsigned short iRank = 1; iRank < size; iRank++){
101 if (nPoint_Recv[iRank] > static_cast<int>(iPoint)){
102 return iRank - 1;
103 }
104 }
105 return size-1;
106 }
107
108 /*!
109 * \brief Get the cumulated number of points
110 * \input rank - the processor rank.
111 * \return The cumulated number of points up to certain processor rank.
112 */
113 unsigned long GetnPointCumulative(unsigned short rank) const override {
114 return GetNodeBegin(rank);
115 }
116
