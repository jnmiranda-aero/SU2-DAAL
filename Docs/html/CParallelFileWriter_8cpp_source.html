57bool CFileWriter::WriteMPIBinaryDataAll(const void *data, unsigned long sizeInBytes,
58 unsigned long totalSizeInBytes, unsigned long offsetInBytes){
59
60#ifdef HAVE_MPI
61
62 startTime = SU2_MPI::Wtime();
63
64 MPI_Datatype filetype;
65
66 /*--- Prepare to write the actual data ---*/
67
68 MPI_Type_contiguous(int(sizeInBytes), MPI_BYTE, &filetype);
69 MPI_Type_commit(&filetype);
70
71 /*--- Set the view for the MPI file write, i.e., describe the
72 location in the file that this rank "sees" for writing its
73 piece of the file. ---*/
74
75 MPI_File_set_view(fhw, disp + offsetInBytes, MPI_BYTE, filetype,
76 (char*)"native", MPI_INFO_NULL);
77
78 /*--- Collective call for all ranks to write simultaneously. ---*/
79
80 int ierr = MPI_File_write_all(fhw, data, int(sizeInBytes), MPI_BYTE, MPI_STATUS_IGNORE);
81
82 MPI_Type_free(&filetype);
83
84 disp += totalSizeInBytes;
85 fileSize += sizeInBytes;
86
87 stopTime = SU2_MPI::Wtime();
88
89 usedTime += stopTime - startTime;
90
91 return (ierr == MPI_SUCCESS);
92#else
93
94 startTime = SU2_MPI::Wtime();
95
96 unsigned long bytesWritten;
97
98 /*--- Write binary data ---*/
99
100 bytesWritten = fwrite(data, sizeof(char), sizeInBytes, fhw);
101 fileSize += bytesWritten;
102
103 stopTime = SU2_MPI::Wtime();
104
105 usedTime += stopTime - startTime;
106
107 return (bytesWritten == sizeInBytes);
108#endif
109
110}

112bool CFileWriter::WriteMPIBinaryData(const void *data, unsigned long sizeInBytes, unsigned short processor){
113
114#ifdef HAVE_MPI
115
116 startTime = SU2_MPI::Wtime();
117
118 int ierr = MPI_SUCCESS;
119
120 /*--- Reset the file view. ---*/
121
122 MPI_File_set_view(fhw, 0, MPI_BYTE, MPI_BYTE,
123 (char*)"native", MPI_INFO_NULL);
124
125 if (rank == processor)
126 ierr = MPI_File_write_at(fhw, disp, data, int(sizeInBytes), MPI_BYTE, MPI_STATUS_IGNORE);
127
128 disp += sizeInBytes;
129 fileSize += sizeInBytes;
130
131 stopTime = SU2_MPI::Wtime();
132
133 usedTime += stopTime - startTime;
134
135 return (ierr == MPI_SUCCESS);
136#else
137
138 startTime = SU2_MPI::Wtime();
139
140 unsigned long bytesWritten = sizeInBytes;
141
142 /*--- Write the total size in bytes at the beginning of the binary data blob ---*/
143
144 bytesWritten = fwrite(data, sizeof(char), sizeInBytes, fhw);
145
146 stopTime = SU2_MPI::Wtime();
147
148 usedTime += stopTime - startTime;
149
150 return (bytesWritten == sizeInBytes);
151
152#endif
153
154}

163

201bool CFileWriter::OpenMPIFile(string val_filename){
202
203 /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
204 val_filename.append(fileExt);
205
206#ifdef HAVE_MPI
207 int ierr;
208 disp = 0.0;
209
210 /*--- All ranks open the file using MPI. Here, we try to open the file with
211 exclusive so that an error is generated if the file exists. We always want
212 to write a fresh output file, so we delete any existing files and create
213 a new one. ---*/
214
215 ierr = MPI_File_open(SU2_MPI::GetComm(), val_filename.c_str(),
216 MPI_MODE_CREATE|MPI_MODE_EXCL|MPI_MODE_WRONLY,
217 MPI_INFO_NULL, &fhw);
218 if (ierr != MPI_SUCCESS) {
219 MPI_File_close(&fhw);
220 if (rank == 0)
221 MPI_File_delete(val_filename.c_str(), MPI_INFO_NULL);
222 ierr = MPI_File_open(SU2_MPI::GetComm(), val_filename.c_str(),
223 MPI_MODE_CREATE|MPI_MODE_EXCL|MPI_MODE_WRONLY,
224 MPI_INFO_NULL, &fhw);
225 }
226
227 /*--- Error check opening the file. ---*/
228
229 if (ierr) {
230 SU2_MPI::Error(string("Unable to open file ") +
231 val_filename, CURRENT_FUNCTION);
232 }
233#else
234 fhw = fopen(val_filename.c_str(), "wb");
235 /*--- Error check for opening the file. ---*/
236
237 if (!fhw) {
238 SU2_MPI::Error(string("Unable to open file ") +
239 val_filename, CURRENT_FUNCTION);
240 }
241#endif
242
243 fileSize = 0.0;
244 usedTime = 0;
245
246 return true;
247}

249bool CFileWriter::CloseMPIFile(){
250
251#ifdef HAVE_MPI
252 /*--- All ranks close the file after writing. ---*/
253
254 MPI_File_close(&fhw);
255#else
256 fclose(fhw);
257#endif
258
259 /*--- Communicate the total file size for the restart ---*/
260
261 su2double my_fileSize = fileSize;
262 SU2_MPI::Allreduce(&my_fileSize, &fileSize, 1,
263 MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
264
265 /*--- Compute and store the bandwidth ---*/
266
267 bandwidth = fileSize/(1.0e6)/usedTime;
268
269 return true;
270}
