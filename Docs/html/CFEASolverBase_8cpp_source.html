74void CFEASolverBase::CommunicateExtraEliminationVertices(const CGeometry* geometry, vector<unsigned long>& myPoints) {
75
76 /*--- communicate the boundary points ---*/
77
78 const unordered_set<unsigned long> markerPoints(myPoints.begin(), myPoints.end());
79
80 vector<unsigned long> numPoints(size);
81 unsigned long num = myPoints.size();
82 SU2_MPI::Allgather(&num, 1, MPI_UNSIGNED_LONG, numPoints.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
83
84 /*--- Global to local map for the halo points of the rank (not covered by the CGeometry map). ---*/
85 unordered_map<unsigned long, unsigned long> Global2Local;
86 for (auto iPoint = nPointDomain; iPoint < nPoint; ++iPoint) {
87 Global2Local[geometry->nodes->GetGlobalIndex(iPoint)] = iPoint;
88 }
89
90 /*--- Populate elimination list. ---*/
91 ExtraVerticesToEliminate.clear();
92
93 for (int i = 0; i < size; ++i) {
94 /*--- Send our point list. ---*/
95 if (rank == i) {
96 SU2_MPI::Bcast(myPoints.data(), numPoints[i], MPI_UNSIGNED_LONG, rank, SU2_MPI::GetComm());
97 continue;
98 }
99
100 /*--- Receive point list. ---*/
101 vector<unsigned long> theirPoints(numPoints[i]);
102 SU2_MPI::Bcast(theirPoints.data(), numPoints[i], MPI_UNSIGNED_LONG, i, SU2_MPI::GetComm());
103
104 for (auto iPointGlobal : theirPoints) {
105 /*--- Check if the rank has the point. ---*/
106 auto it = Global2Local.find(iPointGlobal);
107 if (it == Global2Local.end()) continue;
108
109 /*--- If the point is not covered by this rank's markers, mark it for elimination. ---*/
110 if (markerPoints.count(iPointGlobal) == 0)
111 ExtraVerticesToEliminate.push_back(it->second);
112 }
113 }
114
115}
