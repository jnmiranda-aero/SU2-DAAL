79 template <class ScalarType>
80 void CSysMatrix<ScalarType>::Initialize(unsigned long npoint, unsigned long npointdomain, unsigned short nvar,
81  unsigned short neqn, bool EdgeConnect, CGeometry* geometry,
82  const CConfig* config, bool needTranspPtr, bool grad_mode) {
83  assert(omp_get_thread_num() == 0 && "Only the master thread is allowed to initialize the matrix.");
84 
85  if (npoint == 0) return;
86 
87  if (matrix != nullptr) {
88  SU2_MPI::Error("CSysMatrix can only be initialized once.", CURRENT_FUNCTION);
89  }
90 
91  if (nvar > MAXNVAR) {
92  SU2_MPI::Error("nVar larger than expected, increase MAXNVAR.", CURRENT_FUNCTION);
93  }
94 
95  /*--- Application of this matrix, FVM or FEM. ---*/
96  const auto type = EdgeConnect ? ConnectivityType::FiniteVolume : ConnectivityType::FiniteElement;
97 
98  /*--- Type of preconditioner the matrix will be asked to build. ---*/
99  auto prec = config->GetKind_Linear_Solver_Prec();
100 
101  if ((!EdgeConnect && !config->GetStructuralProblem()) || (config->GetKind_SU2() == SU2_COMPONENT::SU2_DEF) ||
102  (config->GetKind_SU2() == SU2_COMPONENT::SU2_DOT)) {
103  /*--- FEM-type connectivity in non-structural context implies mesh deformation. ---*/
104  prec = config->GetKind_Deform_Linear_Solver_Prec();
105  } else if (config->GetDiscrete_Adjoint() && (prec != ILU)) {
106  /*--- Else "upgrade" primal solver settings. ---*/
107  prec = config->GetKind_DiscAdj_Linear_Prec();
108  }
109 
110  /*--- No else if, but separat if case! ---*/
111  if (config->GetSmoothGradient() && grad_mode) {
112  prec = config->GetKind_Grad_Linear_Solver_Prec();
113  }
114 
115  const bool ilu_needed = (prec == ILU);
116  const bool diag_needed = ilu_needed || (prec == JACOBI) || (prec == LINELET);
117 
118  /*--- Basic dimensions. ---*/
119  nVar = nvar;
120  nEqn = neqn;
121  nPoint = npoint;
122  nPointDomain = npointdomain;
123 
124  /*--- Get sparse structure pointers from geometry,
125  * the data is managed by CGeometry to allow re-use. ---*/
126 
127  const auto& csr = geometry->GetSparsePattern(type, 0);
128 
129  nnz = csr.getNumNonZeros();
130  row_ptr = csr.outerPtr();
131  col_ind = csr.innerIdx();
132  dia_ptr = csr.diagPtr();
133 
134  if (needTranspPtr) col_ptr = geometry->GetTransposeSparsePatternMap(type).data();
135 
136  if (type == ConnectivityType::FiniteVolume) {
137  edge_ptr.ptr = geometry->GetEdgeToSparsePatternMap().data();
138  edge_ptr.nEdge = geometry->GetnEdge();
139  }
140 
141  /*--- Get ILU sparse pattern, if fill is 0 no new data is allocated. --*/
142 
143  if (ilu_needed) {
144  ilu_fill_in = config->GetLinear_Solver_ILU_n();
145 
146  const auto& csr_ilu = geometry->GetSparsePattern(type, ilu_fill_in);
147 
148  row_ptr_ilu = csr_ilu.outerPtr();
149  col_ind_ilu = csr_ilu.innerIdx();
150  dia_ptr_ilu = csr_ilu.diagPtr();
151  nnz_ilu = csr_ilu.getNumNonZeros();
152  }
153 
154  /*--- Allocate data. ---*/
155  auto allocAndInit = [](ScalarType*& ptr, unsigned long num) {
156  ptr = MemoryAllocation::aligned_alloc<ScalarType, true>(64, num * sizeof(ScalarType));
157  };
158 
159  allocAndInit(matrix, nnz * nVar * nEqn);
160 
161  /*--- Preconditioners. ---*/
162 
163  if (ilu_needed) allocAndInit(ILU_matrix, nnz_ilu * nVar * nEqn);
164 
165  if (diag_needed) allocAndInit(invM, nPointDomain * nVar * nEqn);
166 
167  /*--- Thread parallel initialization. ---*/
168 
169  int num_threads = omp_get_max_threads();
170 
171  /*--- Set suitable chunk sizes for light static for loops, and heavy
172  dynamic ones, such that threads are approximately evenly loaded. ---*/
173  omp_light_size = computeStaticChunkSize(nnz * nVar * nEqn, num_threads, OMP_MAX_SIZE_L);
174  omp_heavy_size = computeStaticChunkSize(nPointDomain, num_threads, OMP_MAX_SIZE_H);
175 
176  omp_num_parts = config->GetLinear_Solver_Prec_Threads();
177  if (omp_num_parts == 0) omp_num_parts = num_threads;
178 
179  /*--- This is akin to the row_ptr. ---*/
180  omp_partitions = new unsigned long[omp_num_parts + 1];
181  for (unsigned long i = 0; i <= omp_num_parts; ++i) omp_partitions[i] = nPointDomain;
182 
183  /*--- Work estimate based on non-zeros to produce balanced partitions. ---*/
184 
185  const auto row_ptr_prec = ilu_needed ? row_ptr_ilu : row_ptr;
186  const auto nnz_prec = row_ptr_prec[nPointDomain];
187 
188  const auto nnz_per_part = roundUpDiv(nnz_prec, omp_num_parts);
189 
190  for (auto iPoint = 0ul, part = 0ul; iPoint < nPointDomain; ++iPoint) {
191  if (row_ptr_prec[iPoint] >= part * nnz_per_part) omp_partitions[part++] = iPoint;
192  }
193 
194  for (unsigned long thread = 0; thread < omp_num_parts; ++thread) {
195  const auto begin = omp_partitions[thread];
196  const auto end = omp_partitions[thread + 1];
197  if (begin == end) {
198  cout << "WARNING: Redundant thread has been detected. Performance could be impacted due to low number of nodes "
199  "per thread."
200  << endl;
201  break;
202  }
203  }
204 
205  /*--- Generate MKL Kernels ---*/
206 
207 #ifdef USE_MKL
208  using mkl = mkl_jit_wrapper<ScalarType>;
209  mkl::create_gemm(&MatrixMatrixProductJitter, MKL_ROW_MAJOR, MKL_NOTRANS, MKL_NOTRANS, nVar, nVar, nVar, 1.0, nVar,
210  nVar, 0.0, nVar);
211  MatrixMatrixProductKernel = mkl::get_gemm(MatrixMatrixProductJitter);
212 
213  mkl::create_gemm(&MatrixVectorProductJitterBetaZero, MKL_COL_MAJOR, MKL_NOTRANS, MKL_NOTRANS, 1, nVar, nEqn, 1.0, 1,
214  nEqn, 0.0, 1);
215  MatrixVectorProductKernelBetaZero = mkl::get_gemm(MatrixVectorProductJitterBetaZero);
216 
217  mkl::create_gemm(&MatrixVectorProductJitterBetaOne, MKL_COL_MAJOR, MKL_NOTRANS, MKL_NOTRANS, 1, nVar, nEqn, 1.0, 1,
218  nEqn, 1.0, 1);
219  MatrixVectorProductKernelBetaOne = mkl::get_gemm(MatrixVectorProductJitterBetaOne);
220 
221  mkl::create_gemm(&MatrixVectorProductJitterAlphaMinusOne, MKL_COL_MAJOR, MKL_NOTRANS, MKL_NOTRANS, 1, nVar, nEqn,
222  -1.0, 1, nEqn, 1.0, 1);
223  MatrixVectorProductKernelAlphaMinusOne = mkl::get_gemm(MatrixVectorProductJitterAlphaMinusOne);
224 #endif
225 }

227 template <class T>
228 void CSysMatrixComms::Initiate(const CSysVector<T>& x, CGeometry* geometry, const CConfig* config,
229  MPI_QUANTITIES commType) {
230  if (geometry->nP2PSend == 0) return;
231 
232  /*--- Local variables ---*/
233 
234  const unsigned short COUNT_PER_POINT = x.GetNVar();
235  const unsigned short MPI_TYPE = COMM_TYPE_DOUBLE;
236 
237  /*--- Create a boolean for reversing the order of comms. ---*/
238 
239  const bool reverse = (commType == MPI_QUANTITIES::SOLUTION_MATRIXTRANS);
240 
241  /*--- Set the size of the data packet and type depending on quantity. ---*/
242 
243  switch (commType) {
244  case MPI_QUANTITIES::SOLUTION_MATRIX:
245  case MPI_QUANTITIES::SOLUTION_MATRIXTRANS:
246  break;
247  default:
248  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
249  break;
250  }
251 
252  /*--- Check to make sure we have created a large enough buffer
253  for these comms during preprocessing. This is only for the su2double
254  buffer. It will be reallocated whenever we find a larger count
255  per point. After the first cycle of comms, this should be inactive. ---*/
256 
257  geometry->AllocateP2PComms(COUNT_PER_POINT);
258 
259  /*--- Load the specified quantity from the solver into the generic
260  communication buffer in the geometry class. ---*/
261 
262  /*--- Post all non-blocking recvs first before sends. ---*/
263 
264  geometry->PostP2PRecvs(geometry, config, MPI_TYPE, COUNT_PER_POINT, reverse);
265 
266  for (auto iMessage = 0; iMessage < geometry->nP2PSend; iMessage++) {
267  switch (commType) {
268  case MPI_QUANTITIES::SOLUTION_MATRIX: {
269  su2double* bufDSend = geometry->bufD_P2PSend;
270 
271  /*--- Get the offset for the start of this message. ---*/
272 
273  const auto msg_offset = geometry->nPoint_P2PSend[iMessage];
274 
275  /*--- Total count can include multiple pieces of data per point. ---*/
276 
277  const auto nSend = (geometry->nPoint_P2PSend[iMessage + 1] - geometry->nPoint_P2PSend[iMessage]);
278 
279  SU2_OMP_FOR_STAT(CSysMatrix<T>::OMP_MIN_SIZE)
280  for (auto iSend = 0; iSend < nSend; iSend++) {
281  /*--- Get the local index for this communicated data. ---*/
282 
283  const auto iPoint = geometry->Local_Point_P2PSend[msg_offset + iSend];
284 
285  /*--- Compute the offset in the recv buffer for this point. ---*/
286 
287  const auto buf_offset = (msg_offset + iSend) * COUNT_PER_POINT;
288 
289  /*--- Load the buffer with the data to be sent. ---*/
290 
291  for (auto iVar = 0ul; iVar < x.GetNVar(); iVar++) bufDSend[buf_offset + iVar] = x(iPoint, iVar);
292  }
293  END_SU2_OMP_FOR
294  break;
295  }
296 
297  case MPI_QUANTITIES::SOLUTION_MATRIXTRANS: {
298  /*--- We are going to communicate in reverse, so we use the
299  recv buffer for the send instead. Also, all of the offsets
300  and counts are derived from the recv data structures. ---*/
301 
302  su2double* bufDSend = geometry->bufD_P2PRecv;
303 
304  /*--- Get the offset for the start of this message. ---*/
305 
306  const auto msg_offset = geometry->nPoint_P2PRecv[iMessage];
307 
308  /*--- Total count can include multiple pieces of data per point. ---*/
309 
310  const auto nSend = (geometry->nPoint_P2PRecv[iMessage + 1] - geometry->nPoint_P2PRecv[iMessage]);
311 
312  SU2_OMP_FOR_STAT(CSysMatrix<T>::OMP_MIN_SIZE)
313  for (auto iSend = 0; iSend < nSend; iSend++) {
314  /*--- Get the local index for this communicated data. Here we
315  again use the recv structure to find the send point, since
316  the usual recv points are now the senders in reverse mode. ---*/
317 
318  const auto iPoint = geometry->Local_Point_P2PRecv[msg_offset + iSend];
319 
320  /*--- Compute the offset in the recv buffer for this point. ---*/
321 
322  const auto buf_offset = (msg_offset + iSend) * COUNT_PER_POINT;
323 
324  /*--- Load the buffer with the data to be sent. ---*/
325 
326  for (auto iVar = 0ul; iVar < x.GetNVar(); iVar++) bufDSend[buf_offset + iVar] = x(iPoint, iVar);
327  }
328  END_SU2_OMP_FOR
329  break;
330  }
331 
332  default:
333  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
334  break;
335  }
336 
337  /*--- Launch the point-to-point MPI send for this message. ---*/
338 
339  geometry->PostP2PSends(geometry, config, MPI_TYPE, COUNT_PER_POINT, iMessage, reverse);
340  }
341 }

343 template <class T>
344 void CSysMatrixComms::Complete(CSysVector<T>& x, CGeometry* geometry, const CConfig* config, MPI_QUANTITIES commType) {
345  if (geometry->nP2PRecv == 0) return;
346 
347  /*--- Local variables ---*/
348 
349  const unsigned short COUNT_PER_POINT = x.GetNVar();
350 
351  /*--- Global status so all threads can see the result of Waitany. ---*/
352  static SU2_MPI::Status status;
353  int ind;
354 
355  /*--- Store the data that was communicated into the appropriate
356  location within the local class data structures. ---*/
357 
358  for (auto iMessage = 0; iMessage < geometry->nP2PRecv; iMessage++) {
359  /*--- For efficiency, recv the messages dynamically based on
360  the order they arrive. ---*/
361 
362  SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitany(geometry->nP2PRecv, geometry->req_P2PRecv, &ind, &status);)
363 
364  /*--- Once we have recv'd a message, get the source rank. ---*/
365 
366  const auto source = status.MPI_SOURCE;
367 
368  switch (commType) {
369  case MPI_QUANTITIES::SOLUTION_MATRIX: {
370  const su2double* bufDRecv = geometry->bufD_P2PRecv;
371 
372  /*--- We know the offsets based on the source rank. ---*/
373 
374  const auto jRecv = geometry->P2PRecv2Neighbor[source];
375 
376  /*--- Get the offset for the start of this message. ---*/
377 
378  const auto msg_offset = geometry->nPoint_P2PRecv[jRecv];
379 
380  /*--- Get the number of packets to be received in this message. ---*/
381 
382  const auto nRecv = (geometry->nPoint_P2PRecv[jRecv + 1] - geometry->nPoint_P2PRecv[jRecv]);
383 
384  SU2_OMP_FOR_STAT(CSysMatrix<T>::OMP_MIN_SIZE)
385  for (auto iRecv = 0; iRecv < nRecv; iRecv++) {
386  /*--- Get the local index for this communicated data. ---*/
387 
388  const auto iPoint = geometry->Local_Point_P2PRecv[msg_offset + iRecv];
389 
390  /*--- Compute the offset in the recv buffer for this point. ---*/
391 
392  const auto buf_offset = (msg_offset + iRecv) * COUNT_PER_POINT;
393 
394  /*--- Store the data correctly depending on the quantity. ---*/
395 
396  for (auto iVar = 0ul; iVar < x.GetNVar(); iVar++)
397  x(iPoint, iVar) = CSysMatrix<T>::template ActiveAssign<T>(bufDRecv[buf_offset + iVar]);
398  }
399  END_SU2_OMP_FOR
400  break;
401  }
402 
403  case MPI_QUANTITIES::SOLUTION_MATRIXTRANS: {
404  /*--- We are going to communicate in reverse, so we use the
405  send buffer for the recv instead. Also, all of the offsets
406  and counts are derived from the send data structures. ---*/
407 
408  const su2double* bufDRecv = geometry->bufD_P2PSend;
409 
410  /*--- We know the offsets based on the source rank. ---*/
411 
412  const auto jRecv = geometry->P2PSend2Neighbor[source];
413 
414  /*--- Get the offset for the start of this message. ---*/
415 
416  const auto msg_offset = geometry->nPoint_P2PSend[jRecv];
417 
418  /*--- Get the number of packets to be received in this message. ---*/
419 
420  const auto nRecv = (geometry->nPoint_P2PSend[jRecv + 1] - geometry->nPoint_P2PSend[jRecv]);
421 
422  SU2_OMP_FOR_STAT(CSysMatrix<T>::OMP_MIN_SIZE)
423  for (auto iRecv = 0; iRecv < nRecv; iRecv++) {
424  /*--- Get the local index for this communicated data. ---*/
425 
426  const auto iPoint = geometry->Local_Point_P2PSend[msg_offset + iRecv];
427 
428  /*--- Compute the offset in the recv buffer for this point. ---*/
429 
430  const auto buf_offset = (msg_offset + iRecv) * COUNT_PER_POINT;
431 
432  /*--- Update receiving point. ---*/
433 
434  for (auto iVar = 0ul; iVar < x.GetNVar(); iVar++)
435  x(iPoint, iVar) += CSysMatrix<T>::template ActiveAssign<T>(bufDRecv[buf_offset + iVar]);
436  }
437  END_SU2_OMP_FOR
438  break;
439  }
440 
441  default:
442  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
443  break;
444  }
445  }
446 
447  /*--- Verify that all non-blocking point-to-point sends have finished.
448  Note that this should be satisfied, as we have received all of the
449  data in the loop above at this point. ---*/
450 
451 #ifdef HAVE_MPI
452  SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitall(geometry->nP2PSend, geometry->req_P2PSend, MPI_STATUS_IGNORE);)
453 #endif
454 }

474 template <class ScalarType>
475 void CSysMatrix<ScalarType>::Gauss_Elimination(ScalarType* matrix, ScalarType* vec) const {
476 #ifdef USE_MKL_LAPACK
477  // With MKL_DIRECT_CALL enabled, this is significantly faster than native code on Intel Architectures.
478  lapack_int ipiv[MAXNVAR];
479  LAPACKE_dgetrf(LAPACK_ROW_MAJOR, nVar, nVar, matrix, nVar, ipiv);
480  LAPACKE_dgetrs(LAPACK_ROW_MAJOR, 'N', nVar, 1, matrix, nVar, ipiv, vec, 1);
481 #else
482 #define A(I, J) matrix[(I)*nVar + (J)]
483 
484  /*--- Transform system in Upper Matrix ---*/
485  for (auto iVar = 1ul; iVar < nVar; iVar++) {
486  for (auto jVar = 0ul; jVar < iVar; jVar++) {
487  ScalarType weight = A(iVar, jVar) / A(jVar, jVar);
488  for (auto kVar = jVar; kVar < nVar; kVar++) A(iVar, kVar) -= weight * A(jVar, kVar);
489  vec[iVar] -= weight * vec[jVar];
490  }
491  }
492 
493  /*--- Backwards substitution ---*/
494  for (auto iVar = nVar; iVar > 0ul;) {
495  iVar--; // unsigned type
496  for (auto jVar = iVar + 1; jVar < nVar; jVar++) vec[iVar] -= A(iVar, jVar) * vec[jVar];
497  vec[iVar] /= A(iVar, iVar);
498  }
499 #undef A
500 #endif
501 }

503 template <class ScalarType>
504 void CSysMatrix<ScalarType>::MatrixInverse(ScalarType* matrix, ScalarType* inverse) const {
505  /*--- This is a generalization of Gaussian elimination for multiple rhs' (the basis vectors).
506  We could call "Gauss_Elimination" multiple times or fully generalize it for multiple rhs,
507  the performance of both routines would suffer in both cases without the use of exotic templating.
508  And so it feels reasonable to have some duplication here. ---*/
509 
510  assert((matrix != inverse) && "Output cannot be the same as the input.");
511 
512 #define M(I, J) inverse[(I)*nVar + (J)]
513 
514  /*--- Initialize the inverse with the identity. ---*/
515  for (auto iVar = 0ul; iVar < nVar; iVar++)
516  for (auto jVar = 0ul; jVar < nVar; jVar++) M(iVar, jVar) = ScalarType(iVar == jVar);
517 
518  /*--- Inversion ---*/
519 #ifdef USE_MKL_LAPACK
520  // With MKL_DIRECT_CALL enabled, this is significantly faster than native code on Intel Architectures.
521  lapack_int ipiv[MAXNVAR];
522  LAPACKE_dgetrf(LAPACK_ROW_MAJOR, nVar, nVar, matrix, nVar, ipiv);
523  LAPACKE_dgetrs(LAPACK_ROW_MAJOR, 'N', nVar, nVar, matrix, nVar, ipiv, inverse, nVar);
524 #else
525 #define A(I, J) matrix[(I)*nVar + (J)]
526 
527  /*--- Transform system in Upper Matrix ---*/
528  for (auto iVar = 1ul; iVar < nVar; iVar++) {
529  for (auto jVar = 0ul; jVar < iVar; jVar++) {
530  ScalarType weight = A(iVar, jVar) / A(jVar, jVar);
531 
532  for (auto kVar = jVar; kVar < nVar; kVar++) A(iVar, kVar) -= weight * A(jVar, kVar);
533 
534  /*--- at this stage M is lower triangular so not all cols need updating ---*/
535  for (auto kVar = 0ul; kVar <= jVar; kVar++) M(iVar, kVar) -= weight * M(jVar, kVar);
536  }
537  }
538 
539  /*--- Backwards substitution ---*/
540  for (auto iVar = nVar; iVar > 0ul;) {
541  iVar--; // unsigned type
542  for (auto jVar = iVar + 1; jVar < nVar; jVar++)
543  for (auto kVar = 0ul; kVar < nVar; kVar++) M(iVar, kVar) -= A(iVar, jVar) * M(jVar, kVar);
544 
545  for (auto kVar = 0ul; kVar < nVar; kVar++) M(iVar, kVar) /= A(iVar, iVar);
546  }
547 #undef A
548 #endif
549 #undef M
550 }

552 template <class ScalarType>
553 void CSysMatrix<ScalarType>::DeleteValsRowi(unsigned long i) {
554  const auto block_i = i / nVar;
555  const auto row = i % nVar;
556 
557  for (auto index = row_ptr[block_i]; index < row_ptr[block_i + 1]; index++) {
558  for (auto iVar = 0u; iVar < nVar; iVar++)
559  matrix[index * nVar * nVar + row * nVar + iVar] = 0.0; // Delete row values in the block

621 template <class ScalarType>
622 void CSysMatrix<ScalarType>::BuildILUPreconditioner() {
623  /*--- Copy block matrix to compute factorization in-place. ---*/
624 
625  if (ilu_fill_in == 0) {
626  /*--- ILU0, direct copy. ---*/
627  SU2_OMP_FOR_STAT(omp_light_size)
628  for (auto iVar = 0ul; iVar < nnz * nVar * nVar; ++iVar) ILU_matrix[iVar] = matrix[iVar];
629  END_SU2_OMP_FOR
630  } else {
631  /*--- ILUn clear the ILU matrix first. ---*/
632  SU2_OMP_FOR_STAT(omp_light_size)
633  for (auto iVar = 0ul; iVar < nnz_ilu * nVar * nVar; iVar++) ILU_matrix[iVar] = 0.0;
634  END_SU2_OMP_FOR
635 
636  /*--- ILUn, traverse matrix to access its blocks
637  * sequentially and set them in the ILU matrix. ---*/
638  SU2_OMP_FOR_DYN(omp_heavy_size)
639  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
640  for (auto index = row_ptr[iPoint]; index < row_ptr[iPoint + 1]; index++) {
641  auto jPoint = col_ind[index];
642  SetBlock_ILUMatrix(iPoint, jPoint, &matrix[index * nVar * nVar]);
643  }
644  }
645  END_SU2_OMP_FOR
646  }
647 
648  /*--- Transform system in Upper Matrix ---*/
649 
650  /*--- OpenMP Parallelization, a loop construct is used to ensure
651  * the preconditioner is computed correctly even if called
652  * outside of a parallel section. ---*/
653 
654  SU2_OMP_FOR_STAT(1)
655  for (unsigned long thread = 0; thread < omp_num_parts; ++thread) {
656  const auto begin = omp_partitions[thread];
657  const auto end = omp_partitions[thread + 1];
658  if (begin == end) continue;
659 
660  /*--- Each thread will work on the submatrix defined from row/col "begin"
661  * to row/col "end-1" (i.e. the range [begin,end[). Which is exactly
662  * what the MPI-only implementation does. ---*/
663 
664  ScalarType weight[MAXNVAR * MAXNVAR], aux_block[MAXNVAR * MAXNVAR];
665 
666  for (auto iPoint = begin + 1; iPoint < end; iPoint++) {
667  /*--- Invert and store the previous diagonal block to later compute the weight. ---*/
668 
669  InverseDiagonalBlock_ILUMatrix(iPoint - 1, &invM[(iPoint - 1) * nVar * nVar]);
670 
671  /*--- For this row (unknown), loop over its lower diagonal entries. ---*/
672 
673  for (auto index = row_ptr_ilu[iPoint]; index < dia_ptr_ilu[iPoint]; index++) {
674  /*--- jPoint is the column index (jPoint < iPoint). ---*/
675 
676  auto jPoint = col_ind_ilu[index];
677 
678  /*--- We only care about the sub matrix within "begin" and "end-1". ---*/
679 
680  if (jPoint < begin) continue;
681 
682  /*--- Multiply the block by the inverse of the corresponding diagonal block. ---*/
683 
684  auto Block_ij = &ILU_matrix[index * nVar * nVar];
685  MatrixMatrixProduct(Block_ij, &invM[jPoint * nVar * nVar], weight);
686 
687  /*--- "weight" holds Aij*inv(Ajj). Jump to the upper part of the jPoint row. ---*/
688 
689  for (auto index_ = dia_ptr_ilu[jPoint] + 1; index_ < row_ptr_ilu[jPoint + 1]; index_++) {
690  /*--- Get the column index (kPoint > jPoint). ---*/
691 
692  auto kPoint = col_ind_ilu[index_];
693 
694  if (kPoint >= end) break;
695 
696  /*--- If Aik exists, update it: Aik -= Aij*inv(Ajj)*Ajk ---*/
697 
698  auto Block_ik = GetBlock_ILUMatrix(iPoint, kPoint);
699 
700  if (Block_ik != nullptr) {
701  auto Block_jk = &ILU_matrix[index_ * nVar * nVar];
702  MatrixMatrixProduct(weight, Block_jk, aux_block);
703  MatrixSubtraction(Block_ik, aux_block, Block_ik);
704  }
705  }
706 
707  /*--- Lastly, store "weight" in the lower triangular part, which
708  will be reused during the forward solve in the precon/smoother. ---*/
709 
710  for (auto iVar = 0ul; iVar < nVar * nVar; ++iVar) Block_ij[iVar] = weight[iVar];
711  }
712  }
713  InverseDiagonalBlock_ILUMatrix(end - 1, &invM[(end - 1) * nVar * nVar]);
714  }
715  END_SU2_OMP_FOR
716 }

718 template <class ScalarType>
719 void CSysMatrix<ScalarType>::ComputeILUPreconditioner(const CSysVector<ScalarType>& vec, CSysVector<ScalarType>& prod,
720  CGeometry* geometry, const CConfig* config) const {
721  /*--- Coherent view of vectors. ---*/
722  SU2_OMP_BARRIER
723 
724  /*--- OpenMP Parallelization ---*/
725  SU2_OMP_FOR_STAT(1)
726  for (unsigned long thread = 0; thread < omp_num_parts; ++thread) {
727  const auto begin = omp_partitions[thread];
728  const auto end = omp_partitions[thread + 1];
729  if (begin == end) continue;
730 
731  ScalarType aux_vec[MAXNVAR];
732 
733  /*--- Copy vector to then work on prod in place ---*/
734 
735  for (auto iVar = begin * nVar; iVar < end * nVar; iVar++) prod[iVar] = vec[iVar];
736 
737  /*--- Forward solve the system using the lower matrix entries that
738  were computed and stored during the ILU preprocessing. Note
739  that we are overwriting the residual vector as we go. ---*/
740 
741  for (auto iPoint = begin + 1; iPoint < end; iPoint++) {
742  for (auto index = row_ptr_ilu[iPoint]; index < dia_ptr_ilu[iPoint]; index++) {
743  auto jPoint = col_ind_ilu[index];
744  if (jPoint < begin) continue;
745  auto Block_ij = &ILU_matrix[index * nVar * nVar];
746  MatrixVectorProductSub(Block_ij, &prod[jPoint * nVar], &prod[iPoint * nVar]);
747  }
748  }
749 
750  /*--- Backwards substitution (starts at the last row) ---*/
751 
752  for (auto iPoint = end; iPoint > begin;) {
753  iPoint--; // unsigned type
754  for (auto iVar = 0ul; iVar < nVar; iVar++) aux_vec[iVar] = prod[iPoint * nVar + iVar];
755 
756  for (auto index = dia_ptr_ilu[iPoint] + 1; index < row_ptr_ilu[iPoint + 1]; index++) {
757  auto jPoint = col_ind_ilu[index];
758  if (jPoint >= end) break;
759  auto Block_ij = &ILU_matrix[index * nVar * nVar];
760  MatrixVectorProductSub(Block_ij, &prod[jPoint * nVar], aux_vec);
761  }
762 
763  MatrixVectorProduct(&invM[iPoint * nVar * nVar], aux_vec, &prod[iPoint * nVar]);
764  }
765  }
766  END_SU2_OMP_FOR
767 
768  /*--- MPI Parallelization ---*/
769 
770  CSysMatrixComms::Initiate(prod, geometry, config);
771  CSysMatrixComms::Complete(prod, geometry, config);
772 }

774 template <class ScalarType>
775 void CSysMatrix<ScalarType>::ComputeLU_SGSPreconditioner(const CSysVector<ScalarType>& vec,
776  CSysVector<ScalarType>& prod, CGeometry* geometry,
777  const CConfig* config) const {
778  /*--- First part of the symmetric iteration: (D+L).x* = b ---*/
779 
780  /*--- Coherent view of vectors. ---*/
781  SU2_OMP_BARRIER
782 
783  /*--- OpenMP Parallelization ---*/
784  SU2_OMP_FOR_STAT(1)
785  for (unsigned long thread = 0; thread < omp_num_parts; ++thread) {
786  const auto begin = omp_partitions[thread];
787  const auto end = omp_partitions[thread + 1];
788  if (begin == end) continue;
789 
790  /*--- Each thread will work on the submatrix defined from row/col "begin"
791  * to row/col "end-1", except the last thread that also considers halos.
792  * This is NOT exactly equivalent to the MPI implementation on the same
793  * number of domains, for that we would need to define "thread-halos". ---*/
794 
795  ScalarType low_prod[MAXNVAR];
796 
797  for (auto iPoint = begin; iPoint < end; ++iPoint) {
798  auto idx = iPoint * nVar;
799  LowerProduct(prod, iPoint, begin, low_prod); // Compute L.x*
800  VectorSubtraction(&vec[idx], low_prod, &prod[idx]); // Compute y = b - L.x*
801  Gauss_Elimination(iPoint, &prod[idx]); // Solve D.x* = y
802  }
803  }
804  END_SU2_OMP_FOR
805 
806  /*--- MPI Parallelization ---*/
807 
808  CSysMatrixComms::Initiate(prod, geometry, config);
809  CSysMatrixComms::Complete(prod, geometry, config);
810 
811  /*--- Second part of the symmetric iteration: (D+U).x_(1) = D.x* ---*/
812 
813  /*--- OpenMP Parallelization ---*/
814  SU2_OMP_FOR_STAT(1)
815  for (unsigned long thread = 0; thread < omp_num_parts; ++thread) {
816  const auto begin = omp_partitions[thread];
817  const auto row_end = omp_partitions[thread + 1];
818  if (begin == row_end) continue;
819 
820  ScalarType up_prod[MAXNVAR], dia_prod[MAXNVAR];
821 
822  for (auto iPoint = row_end; iPoint > begin;) {
823  iPoint--; // because of unsigned type
824  auto idx = iPoint * nVar;
825  DiagonalProduct(prod, iPoint, dia_prod); // Compute D.x*
826  UpperProduct(prod, iPoint, row_end, up_prod); // Compute U.x_(n+1)
827  VectorSubtraction(dia_prod, up_prod, &prod[idx]); // Compute y = D.x*-U.x_(n+1)
828  Gauss_Elimination(iPoint, &prod[idx]); // Solve D.x* = y
829  }
830  }
831  END_SU2_OMP_FOR
832 
833  /*--- MPI Parallelization ---*/
834 
835  CSysMatrixComms::Initiate(prod, geometry, config);
836  CSysMatrixComms::Complete(prod, geometry, config);
837 }

868 template <class ScalarType>
869 void CSysMatrix<ScalarType>::ComputeLineletPreconditioner(const CSysVector<ScalarType>& vec,
870  CSysVector<ScalarType>& prod, CGeometry* geometry,
871  const CConfig* config) const {
872  /*--- Coherent view of vectors. ---*/
873  SU2_OMP_BARRIER
874 
875  const auto& li = geometry->GetLineletInfo(config);
876 
877  /*--- Jacobi preconditioning where there are no linelets. ---*/
878 
879  SU2_OMP_FOR_(schedule(dynamic, omp_heavy_size) SU2_NOWAIT)
880  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++)
881  if (li.lineletIdx[iPoint] == CGeometry::CLineletInfo::NO_LINELET)
882  MatrixVectorProduct(&(invM[iPoint * nVar * nVar]), &vec[iPoint * nVar], &prod[iPoint * nVar]);
883  END_SU2_OMP_FOR
884 
885  /*--- Solve the tridiagonal systems for the linelets. ---*/
886 
887  SU2_OMP_FOR_DYN(1)
888  for (auto iLinelet = 0ul; iLinelet < li.linelets.size(); iLinelet++) {
889  /*--- Get references to the working vectors allocated for this thread. ---*/
890 
891  const int thread = omp_get_thread_num();
892  auto& lineletUpper = LineletUpper[thread];
893  auto& lineletInvDiag = LineletInvDiag[thread];
894  auto& lineletVector = LineletVector[thread];
895 
896  /*--- Initialize the solution vector with the rhs ---*/
897 
898  const auto nElem = li.linelets[iLinelet].size();
899 
900  for (auto iElem = 0ul; iElem < nElem; iElem++) {
901  const auto iPoint = li.linelets[iLinelet][iElem];
902  for (auto iVar = 0ul; iVar < nVar; iVar++) lineletVector[iElem * nVar + iVar] = vec[iPoint * nVar + iVar];
903  }
904 
905  /*--- Forward pass, eliminate lower entries, modify diagonal and rhs. ---*/
906 
907  /*--- Small temporaries. ---*/
908  ScalarType aux_block[MAXNVAR * MAXNVAR], aux_vector[MAXNVAR];
909 
910  /*--- Copy diagonal block for first point in this linelet. ---*/
911  MatrixCopy(&matrix[dia_ptr[li.linelets[iLinelet][0]] * nVar * nVar], lineletInvDiag.data());
912 
913  for (auto iElem = 1ul; iElem < nElem; iElem++) {
914  /*--- Setup pointers to required matrices and vectors ---*/
915  const auto im1Point = li.linelets[iLinelet][iElem - 1];
916  const auto iPoint = li.linelets[iLinelet][iElem];
917 
918  const auto* d = &matrix[dia_ptr[iPoint] * nVar * nVar];
919  const auto* l = GetBlock(iPoint, im1Point);
920  const auto* u = GetBlock(im1Point, iPoint);
921 
922  auto* inv_dm1 = &lineletInvDiag[(iElem - 1) * nVar * nVar];
923  auto* d_prime = &lineletInvDiag[iElem * nVar * nVar];
924  auto* b_prime = &lineletVector[iElem * nVar];
925 
926  /*--- Invert previous modified diagonal ---*/
927  MatrixCopy(inv_dm1, aux_block);
928  MatrixInverse(aux_block, inv_dm1);
929 
930  /*--- Left-multiply by lower block to obtain the weight ---*/
931  MatrixMatrixProduct(l, inv_dm1, aux_block);
932 
933  /*--- Multiply weight by upper block to modify current diagonal ---*/
934  MatrixMatrixProduct(aux_block, u, d_prime);
935  MatrixSubtraction(d, d_prime, d_prime);
936 
937  /*--- Update the rhs ---*/
938  MatrixVectorProduct(aux_block, &lineletVector[(iElem - 1) * nVar], aux_vector);
939  VectorSubtraction(b_prime, aux_vector, b_prime);
940 
941  /*--- Cache upper block pointer for the backward substitution phase ---*/
942  lineletUpper[iElem - 1] = u;
943  }
944 
945  /*--- Backwards substitution, LineletVector becomes the solution ---*/
946 
947  /*--- x_n = d_n^{-1} * b_n ---*/
948  Gauss_Elimination(&lineletInvDiag[(nElem - 1) * nVar * nVar], &lineletVector[(nElem - 1) * nVar]);
949 
950  /*--- x_i = d_i^{-1}*(b_i - u_i*x_{i+1}) ---*/
951  for (auto iElem = nElem - 1; iElem > 0; --iElem) {
952  const auto* inv_dm1 = &lineletInvDiag[(iElem - 1) * nVar * nVar];
953  MatrixVectorProduct(lineletUpper[iElem - 1], &lineletVector[iElem * nVar], aux_vector);
954  VectorSubtraction(&lineletVector[(iElem - 1) * nVar], aux_vector, aux_vector);
955  MatrixVectorProduct(inv_dm1, aux_vector, &lineletVector[(iElem - 1) * nVar]);
956  }
957 
958  /*--- Copy results to product vector ---*/
959 
960  for (auto iElem = 0ul; iElem < nElem; iElem++) {
961  const auto iPoint = li.linelets[iLinelet][iElem];
962  for (auto iVar = 0ul; iVar < nVar; iVar++) prod[iPoint * nVar + iVar] = lineletVector[iElem * nVar + iVar];
963  }
964  }
965  END_SU2_OMP_FOR
966 
967  /*--- MPI Parallelization ---*/
968 
969  CSysMatrixComms::Initiate(prod, geometry, config);
970  CSysMatrixComms::Complete(prod, geometry, config);
971 }

986 template <class ScalarType>
987 template <class OtherType>
988 void CSysMatrix<ScalarType>::EnforceSolutionAtNode(const unsigned long node_i, const OtherType* x_i,
989  CSysVector<OtherType>& b) {
990  /*--- Eliminate the row associated with node i (Block_ii = I and all other Block_ij = 0).
991  * To preserve eventual symmetry, also attempt to eliminate the column, if the sparse pattern is not
992  * symmetric the entire column may not be eliminated, the result (matrix and vector) is still correct.
993  * The vector is updated with the product of column i by the known (enforced) solution at node i. ---*/
994 
995  for (auto index = row_ptr[node_i]; index < row_ptr[node_i + 1]; ++index) {
996  auto node_j = col_ind[index];
997 
998  /*--- The diagonal block is handled outside the loop. ---*/
999  if (node_j == node_i) continue;
1000 
1001  /*--- Delete block j on row i (bij) and ATTEMPT to delete block i on row j (bji). ---*/
1002  auto bij = &matrix[index * nVar * nVar];
1003  auto bji = GetBlock(node_j, node_i);
1004 
1005  /*--- The "attempt" part. ---*/
1006  if (bji == nullptr) {
1007  node_j = node_i;
1008  bji = bij;
1009  }
1010 
1011  for (auto iVar = 0ul; iVar < nVar; ++iVar) {
1012  for (auto jVar = 0ul; jVar < nVar; ++jVar) {
1013  /*--- Column product. ---*/
1014  b[node_j * nVar + iVar] -= bji[iVar * nVar + jVar] * x_i[jVar];
1015  /*--- Delete blocks. ---*/
1016  bij[iVar * nVar + jVar] = bji[iVar * nVar + jVar] = 0.0;
1017  }
1018  }
1019  }
1020 
1021  /*--- Set the diagonal block to the identity. ---*/
1022  SetVal2Diag(node_i, 1.0);
1023 
1024  /*--- Set known solution in rhs vector. ---*/
1025  b.SetBlock(node_i, x_i);
1026 }

1029 template <class OtherType>
1030 void CSysMatrix<ScalarType>::EnforceSolutionAtDOF(unsigned long node_i, unsigned long iVar, OtherType x_i,
1031  CSysVector<OtherType>& b) {

1033  const auto node_j = col_ind[index];
1034 
1035  /*--- Delete row iVar of block j on row i (bij) and ATTEMPT
1036  * to delete column iVar block i on row j (bji). ---*/

1043  for (auto jVar = 0ul; jVar < nVar; ++jVar) {
1044  /*--- Column product. ---*/
1045  b[node_j * nVar + jVar] -= bji[jVar * nVar + iVar] * x_i;
1046  /*--- Delete entries. ---*/

1049  }
1050 
1051  /*--- Delete row. ---*/

1056  }
1057 
1058  /*--- Set known solution in rhs vector. ---*/
1059  b(node_i, iVar) = x_i;

1078 template <class ScalarType>
1079 void CSysMatrix<ScalarType>::TransposeInPlace() {
1080  assert(nVar == nEqn && "Cannot transpose with nVar != nEqn.");
1081 
1082  auto swapAndTransp = [](unsigned long n, ScalarType* a, ScalarType* b) {
