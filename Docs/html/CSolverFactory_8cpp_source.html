59CSolver** CSolverFactory::CreateSolverContainer(MAIN_SOLVER kindMainSolver, CConfig *config, CGeometry *geometry, int iMGLevel){
60
61 CSolver** solver;
62
63 solver = new CSolver*[MAX_SOLS]();
64
65 switch (kindMainSolver) {
66 case MAIN_SOLVER::TEMPLATE_SOLVER:
67 solver[FLOW_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::TEMPLATE, solver, geometry, config, iMGLevel);
68 break;
69 case MAIN_SOLVER::INC_EULER:
70 solver[FLOW_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::INC_EULER, solver, geometry, config, iMGLevel);
71 solver[RAD_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::RADIATION, solver, geometry, config, iMGLevel);
72 break;
73 case MAIN_SOLVER::EULER:

75 break;
76 case MAIN_SOLVER::NEMO_EULER:

212CSolver* CSolverFactory::CreateSubSolver(SUB_SOLVER_TYPE kindSolver, CSolver **solver, CGeometry *geometry, CConfig *config, int iMGLevel){
213
214 CSolver *genericSolver = nullptr;
215
216 TURB_MODEL kindTurbModel = config->GetKind_Turb_Model();
217 TURB_TRANS_MODEL kindTransModel = config->GetKind_Trans_Model();
218
219 SolverMetaData metaData;
220
221 metaData.solverType = kindSolver;
222
223 switch (kindSolver) {
224 case SUB_SOLVER_TYPE::CONT_ADJ_EULER:
225 genericSolver = new CAdjEulerSolver(geometry, config, iMGLevel);
226 metaData.integrationType = INTEGRATION_TYPE::MULTIGRID;
227 break;
228 case SUB_SOLVER_TYPE::CONT_ADJ_NAVIER_STOKES:
229 genericSolver = new CAdjNSSolver(geometry, config, iMGLevel);
230 metaData.integrationType = INTEGRATION_TYPE::MULTIGRID;
231 break;
232 case SUB_SOLVER_TYPE::CONT_ADJ_TURB:
233 genericSolver = CreateTurbSolver(kindTurbModel, solver, geometry, config, iMGLevel, true);
234 metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
235 break;
236 case SUB_SOLVER_TYPE::DISC_ADJ_TURB:
237 genericSolver = CreateTurbSolver(kindTurbModel, solver, geometry, config, iMGLevel, true);
238 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
239 break;
240 case SUB_SOLVER_TYPE::BASELINE:
241 genericSolver = new CBaselineSolver(geometry, config);
242 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
243 break;
244 case SUB_SOLVER_TYPE::BASELINE_FEM:
245 genericSolver = new CBaselineSolver_FEM(geometry, config);
246 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
247 break;
248 case SUB_SOLVER_TYPE::DISC_ADJ_FEA:
249 genericSolver = new CDiscAdjFEASolver(geometry, config, solver[FEA_SOL], RUNTIME_FEA_SYS, iMGLevel);
250 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
251 break;
252 case SUB_SOLVER_TYPE::DISC_ADJ_MESH:
253 genericSolver = CreateMeshSolver(solver, geometry, config, iMGLevel, true);
254 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
255 break;
256 case SUB_SOLVER_TYPE::DISC_ADJ_FLOW:
257 genericSolver = new CDiscAdjSolver(geometry, config, solver[FLOW_SOL], RUNTIME_FLOW_SYS, iMGLevel);
258 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
259 break;
260 case SUB_SOLVER_TYPE::EULER:
261 case SUB_SOLVER_TYPE::INC_EULER:
262 case SUB_SOLVER_TYPE::NEMO_EULER:
263 case SUB_SOLVER_TYPE::NAVIER_STOKES:
264 case SUB_SOLVER_TYPE::INC_NAVIER_STOKES:
265 case SUB_SOLVER_TYPE::NEMO_NAVIER_STOKES:
266 genericSolver = CreateFlowSolver(kindSolver, solver, geometry, config, iMGLevel);
267 if (!config->GetNewtonKrylov() || config->GetDiscrete_Adjoint() || config->GetContinuous_Adjoint())
268 metaData.integrationType = INTEGRATION_TYPE::MULTIGRID;
269 else
270 metaData.integrationType = INTEGRATION_TYPE::NEWTON;
271 break;
272 case SUB_SOLVER_TYPE::FEA:
273 genericSolver = new CFEASolver(geometry, config);
274 metaData.integrationType = INTEGRATION_TYPE::STRUCTURAL;
275 break;
276 case SUB_SOLVER_TYPE::MESH:
277 genericSolver = CreateMeshSolver(solver, geometry, config, iMGLevel, false);
278 metaData.integrationType = INTEGRATION_TYPE::NONE;
279 break;
280 case SUB_SOLVER_TYPE::DG_EULER:
281 case SUB_SOLVER_TYPE::DG_NAVIER_STOKES:
282 genericSolver = CreateDGSolver(kindSolver, geometry, config, iMGLevel);
283 metaData.integrationType = INTEGRATION_TYPE::FEM_DG;
284 break;
285 case SUB_SOLVER_TYPE::HEAT:
286 genericSolver = CreateHeatSolver(solver, geometry, config, iMGLevel, false);
287 metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
288 break;
289 case SUB_SOLVER_TYPE::DISC_ADJ_HEAT:
290 genericSolver = CreateHeatSolver(solver, geometry, config, iMGLevel, true);
291 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
292 break;
293 case SUB_SOLVER_TYPE::TRANSITION:
294 genericSolver = CreateTransSolver(kindTransModel, solver, geometry, config, iMGLevel, false);
295 metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
296 break;
297 case SUB_SOLVER_TYPE::SPECIES:
298 genericSolver = CreateSpeciesSolver(solver, geometry, config, iMGLevel, false);
299 metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
300 break;
301 case SUB_SOLVER_TYPE::DISC_ADJ_SPECIES:
302 genericSolver = CreateSpeciesSolver(solver, geometry, config, iMGLevel, true);
303 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
304 break;
305 case SUB_SOLVER_TYPE::TURB:
306 case SUB_SOLVER_TYPE::TURB_SA:
307 case SUB_SOLVER_TYPE::TURB_SST:
308 genericSolver = CreateTurbSolver(kindTurbModel, solver, geometry, config, iMGLevel, false);
309 metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
310 break;
311 case SUB_SOLVER_TYPE::TEMPLATE:
312 genericSolver = new CTemplateSolver(geometry, config);
313 metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
314 break;
315 case SUB_SOLVER_TYPE::RADIATION:
316 if (config->AddRadiation()){
317 genericSolver = new CRadP1Solver(geometry, config);
318 }
319 metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
320 break;
321 case SUB_SOLVER_TYPE::DISC_ADJ_RADIATION:
322 if (config->AddRadiation()){
323 genericSolver = new CDiscAdjSolver(geometry, config, solver[RAD_SOL], RUNTIME_RADIATION_SYS, iMGLevel);
324 }
325 metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
326 break;
327 default:
328 SU2_MPI::Error("No proper allocation found for requested sub solver", CURRENT_FUNCTION);
329 break;
330 }
331
332 if (genericSolver != nullptr)
333 allocatedSolvers[genericSolver] = metaData;
334
335 return genericSolver;
336
337}

339CSolver* CSolverFactory::CreateTurbSolver(TURB_MODEL kindTurbModel, CSolver **solver, CGeometry *geometry, CConfig *config, int iMGLevel, int adjoint){
340
341 CSolver *turbSolver = nullptr;
342
343 if (!adjoint){
344 switch (TurbModelFamily(kindTurbModel)) {
345 case TURB_FAMILY::SA:
346 turbSolver = new CTurbSASolver(geometry, config, iMGLevel, solver[FLOW_SOL]->GetFluidModel());
347 solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
348 turbSolver->Postprocessing(geometry, solver, config, iMGLevel);
349 break;
350 case TURB_FAMILY::KW:
351 turbSolver = new CTurbSSTSolver(geometry, config, iMGLevel);
352 solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
353 turbSolver->Postprocessing(geometry, solver, config, iMGLevel);
354 solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
355 break;
356 case TURB_FAMILY::NONE:
357 SU2_MPI::Error("Trying to create TurbSolver container but TURB_MODEL=NONE.", CURRENT_FUNCTION);
358 break;
359 }
360 } else {
361
362 if (config->GetDiscrete_Adjoint()){
363 if (!config->GetFrozen_Visc_Disc())
364 turbSolver = new CDiscAdjSolver(geometry, config, solver[TURB_SOL], RUNTIME_TURB_SYS, iMGLevel);
365 } else {
366 if (!config->GetFrozen_Visc_Cont())
367 turbSolver = new CAdjTurbSolver(geometry, config, iMGLevel);
368 }
369 }
370
371 return turbSolver;
372}

374CSolver* CSolverFactory::CreateTransSolver(TURB_TRANS_MODEL kindTransModel, CSolver **solver, CGeometry *geometry, CConfig *config, int iMGLevel, int adjoint){
375
376 CSolver *transSolver = nullptr;
377
378 if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
379 switch (kindTransModel) {
380 case TURB_TRANS_MODEL::LM :
381 transSolver = new CTransLMSolver(geometry, config, iMGLevel);
382 solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
383 transSolver->Postprocessing(geometry, solver, config, iMGLevel);
384 solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
385 break;
386 case TURB_TRANS_MODEL::NONE:
387 break;
388 }
389 }
390
391 return transSolver;
392}

394CSolver* CSolverFactory::CreateSpeciesSolver(CSolver **solver, CGeometry *geometry, CConfig *config, int iMGLevel, bool adjoint){
395
396 CSolver *speciesSolver = nullptr;
397
398 if (config->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
399 if (adjoint){
400 speciesSolver = new CDiscAdjSolver(geometry, config, solver[SPECIES_SOL], RUNTIME_SPECIES_SYS, iMGLevel);
401 } else {
402 if (config->GetKind_Species_Model() == SPECIES_MODEL::SPECIES_TRANSPORT)
403 speciesSolver = new CSpeciesSolver(geometry, config, iMGLevel);
404 else if (config->GetKind_Species_Model() == SPECIES_MODEL::FLAMELET)
405 speciesSolver = new CSpeciesFlameletSolver(geometry, config, iMGLevel);
406 }
407 }
408 return speciesSolver;
409}

411CSolver* CSolverFactory::CreateHeatSolver(CSolver **solver, CGeometry *geometry, CConfig *config, int iMGLevel, bool adjoint){
412
413 CSolver *heatSolver = nullptr;
414
