420 /*!
421 * \brief Compute the mean rate of strain matrix.
422 * \details The parameter primvargrad can be e.g. PrimVar_Grad_i or Mean_GradPrimVar.
423 * \param[in] nDim - 2 or 3
424 * \param[out] rateofstrain - Rate of strain matrix
425 * \param[in] velgrad - A velocity gradient matrix.
426 * \tparam Mat1 - any type that supports the [][] interface
427 * \tparam Mat2 - any type that supports the [][] interface
428 */
429 template<class Mat1, class Mat2>
430 FORCEINLINE static void ComputeMeanRateOfStrainMatrix(size_t nDim, Mat1& rateofstrain, const Mat2& velgrad){
431
432 /* --- Calculate the rate of strain tensor, using mean velocity gradients --- */
433
434 if (nDim == 3) {
435 rateofstrain[0][0] = velgrad[0][0];
436 rateofstrain[1][1] = velgrad[1][1];
437 rateofstrain[2][2] = velgrad[2][2];
438 rateofstrain[0][1] = 0.5 * (velgrad[0][1] + velgrad[1][0]);
439 rateofstrain[0][2] = 0.5 * (velgrad[0][2] + velgrad[2][0]);
440 rateofstrain[1][2] = 0.5 * (velgrad[1][2] + velgrad[2][1]);
441 rateofstrain[1][0] = rateofstrain[0][1];
442 rateofstrain[2][1] = rateofstrain[1][2];
443 rateofstrain[2][0] = rateofstrain[0][2];
444 }
445 else { // nDim==2
446 rateofstrain[0][0] = velgrad[0][0];
447 rateofstrain[1][1] = velgrad[1][1];
448 rateofstrain[2][2] = 0.0;
449 rateofstrain[0][1] = 0.5 * (velgrad[0][1] + velgrad[1][0]);
450 rateofstrain[0][2] = 0.0;
451 rateofstrain[1][2] = 0.0;
452 rateofstrain[1][0] = rateofstrain[0][1];
453 rateofstrain[2][1] = rateofstrain[1][2];
454 rateofstrain[2][0] = rateofstrain[0][2];
455 }
456 }

458 /*!
459 * \brief Compute the stress tensor from the velocity gradients.
460 * \details To obtain the Reynolds stress tensor +(u_i' u_j')~, divide the result
461 * of this function by (-rho). The argument density is only used if turb_ke is not 0.
462 * To select the velocity gradient components from a primitive variable gradient PrimVar_Grad_i,
463 * write PrimVar_Grad_i+1.
464 * If <code>nDim==2</code>, we use the same formula but only only access the entries [0][0]..[1][1] of
465 * stress and velgrad. If <code>reynolds3x3</code> is true, the other non-diagonal entries of stress
466 * set to zero, and <code>stress[2][2]</code> to some value.
467 * \param[in] nDim - Dimension of the flow problem, 2 or 3
468 * \param[out] stress - Stress tensor
469 * \param[in] velgrad - A velocity gradient matrix.
470 * \param[in] viscosity - Viscosity
471 * \param[in] density - Density
472 * \param[in] turb_ke - Turbulent kinetic energy, for the turbulent stress tensor
473 * \param[in] reynolds3x3 - If true, write to the third row and column of stress even if nDim==2.
474 * \tparam Mat1 - any type that supports the [][] interface
475 * \tparam Mat2 - any type that supports the [][] interface
476 */
477 template<class Mat1, class Mat2, class Scalar>
478 FORCEINLINE static void ComputeStressTensor(size_t nDim, Mat1& stress, const Mat2& velgrad,
479 Scalar viscosity, Scalar density=0.0,
480 Scalar turb_ke=0.0, bool reynolds3x3=false){
481 Scalar divVel = 0.0;
482 for (size_t iDim = 0; iDim < nDim; iDim++) {
483 divVel += velgrad[iDim][iDim];
484 }
485 Scalar pTerm = 2./3. * (divVel * viscosity + density * turb_ke);
486
487 for (size_t iDim = 0; iDim < nDim; iDim++){
488 for (size_t jDim = 0; jDim < nDim; jDim++){
489 stress[iDim][jDim] = viscosity * (velgrad[iDim][jDim]+velgrad[jDim][iDim]);
490 }
491 stress[iDim][iDim] -= pTerm;
492 }
493
494 if(reynolds3x3 && nDim==2) { // fill the third row and column of Reynolds stress matrix
495 stress[0][2] = stress[1][2] = stress[2][0] = stress[2][1] = 0.0;
496 stress[2][2] = -pTerm;
497 }
498 }

500 /*!
501 * \brief Add a correction using a Quadratic Constitutive Relation to the stress tensor.
502 *
503 * See: Spalart, P. R., "Strategies for Turbulence Modelling and Simulation",
504 * International Journal of Heat and Fluid Flow, Vol. 21, 2000, pp. 252-263
505 *
506 * \param[in] nDim: 2D or 3D.
507 * \param[in] gradvel: Velocity gradients.
508 * \param[in,out] tau: Shear stress tensor.
509 */
510 template <class Mat1, class Mat2>
511 FORCEINLINE static void AddQCR(size_t nDim, const Mat1& gradvel, Mat2& tau) {
512 using Scalar = typename std::decay<decltype(gradvel[0][0])>::type;
513
514 const Scalar c_cr1 = 0.3;
515
516 /*--- Denominator Antisymmetric normalized rotation tensor ---*/
517
518 Scalar factor = 0.0;
519 for (size_t iDim = 0; iDim < nDim; iDim++)
520 for (size_t jDim = 0; jDim < nDim; jDim++)
521 factor += gradvel[iDim][jDim] * gradvel[iDim][jDim];
522 factor = 1.0 / sqrt(max(factor,1E-10));
523
524 /*--- Adding the QCR contribution ---*/
525
526 Scalar tauQCR[MAXNDIM][MAXNDIM] = {{0.0}};
527
528 for (size_t iDim = 0; iDim < nDim; iDim++){
529 for (size_t jDim = 0; jDim < nDim; jDim++){
530 for (size_t kDim = 0; kDim < nDim; kDim++){
531 Scalar O_ik = (gradvel[iDim][kDim] - gradvel[kDim][iDim]) * factor;
532 Scalar O_jk = (gradvel[jDim][kDim] - gradvel[kDim][jDim]) * factor;
533 tauQCR[iDim][jDim] += O_ik * tau[jDim][kDim] + O_jk * tau[iDim][kDim];
534 }
535 }
536 }
537
538 for (size_t iDim = 0; iDim < nDim; iDim++)
539 for (size_t jDim = 0; jDim < nDim; jDim++)
540 tau[iDim][jDim] -= c_cr1 * tauQCR[iDim][jDim];
541 }

543 /*!
544 * \brief Perturb the Reynolds stress tensor based on parameters.
545 * \param[in] nDim - Dimension of the flow problem, 2 or 3.
546 * \param[in] uq_eigval_comp - Component 1C 2C 3C.
547 * \param[in] uq_permute - Whether to swap order of eigen vectors.
548 * \param[in] uq_delta_b - Delta_b parameter.
549 * \param[in] uq_urlx - Relaxation factor.
550 * \param[in] velgrad - A velocity gradient matrix.
551 * \param[in] density - Density.
552 * \param[in] viscosity - Eddy viscosity.
553 * \param[in] turb_ke: Turbulent kinetic energy.
554 * \param[out] MeanPerturbedRSM - Perturbed stress tensor.
555 */
556 template<class Mat1, class Mat2, class Scalar>
557 NEVERINLINE static void ComputePerturbedRSM(size_t nDim, size_t uq_eigval_comp, bool uq_permute, su2double uq_delta_b,
558 su2double uq_urlx, const Mat1& velgrad, Scalar density,
559 Scalar viscosity, Scalar turb_ke, Mat2& MeanPerturbedRSM) {
560 Scalar MeanReynoldsStress[3][3];
561 ComputeStressTensor(nDim, MeanReynoldsStress, velgrad, viscosity, density, turb_ke, true);
562 for (size_t iDim = 0; iDim < 3; iDim++)
563 for (size_t jDim = 0; jDim < 3; jDim++)
564 MeanReynoldsStress[iDim][jDim] /= -density;
565
566 /* --- Calculate anisotropic part of Reynolds Stress tensor --- */
567
568 Scalar A_ij[3][3];
569 for (size_t iDim = 0; iDim < 3; iDim++) {
570 for (size_t jDim = 0; jDim < 3; jDim++) {
571 A_ij[iDim][jDim] = .5 * MeanReynoldsStress[iDim][jDim] / turb_ke;
572 }
573 A_ij[iDim][iDim] -= 1.0/3.0;
574 }
575
576 /* --- Get ordered eigenvectors and eigenvalues of A_ij --- */
577
578 Scalar work[3], Eig_Vec[3][3], Eig_Val[3];
579 CBlasStructure::EigenDecomposition(A_ij, Eig_Vec, Eig_Val, 3, work);
580
581 /* compute convex combination coefficients */
582 Scalar c1c = Eig_Val[2] - Eig_Val[1];
583 Scalar c2c = 2.0 * (Eig_Val[1] - Eig_Val[0]);
584 Scalar c3c = 3.0 * Eig_Val[0] + 1.0;
585
586 /* define barycentric traingle corner points */
587 Scalar Corners[3][2];
588 Corners[0][0] = 1.0;
589 Corners[0][1] = 0.0;
590 Corners[1][0] = 0.0;
591 Corners[1][1] = 0.0;
592 Corners[2][0] = 0.5;
593 Corners[2][1] = 0.866025;
594
595 /* define barycentric coordinates */
596 Scalar Barycentric_Coord[2];
597 Barycentric_Coord[0] = Corners[0][0] * c1c + Corners[1][0] * c2c + Corners[2][0] * c3c;
598 Barycentric_Coord[1] = Corners[0][1] * c1c + Corners[1][1] * c2c + Corners[2][1] * c3c;
599
600 /* component 1C, 2C, 3C, converted to index of the "corners" */
601 Scalar New_Coord[2];
602 New_Coord[0] = Corners[uq_eigval_comp-1][0];
603 New_Coord[1] = Corners[uq_eigval_comp-1][1];
604
605 /* calculate perturbed barycentric coordinates */
606 Barycentric_Coord[0] = Barycentric_Coord[0] + (uq_delta_b) * (New_Coord[0] - Barycentric_Coord[0]);
607 Barycentric_Coord[1] = Barycentric_Coord[1] + (uq_delta_b) * (New_Coord[1] - Barycentric_Coord[1]);
608
609 /* rebuild c1c,c2c,c3c based on perturbed barycentric coordinates */
610 c3c = Barycentric_Coord[1] / Corners[2][1];
611 c1c = Barycentric_Coord[0] - Corners[2][0] * c3c;
612 c2c = 1 - c1c - c3c;
613
614 /* build new anisotropy eigenvalues */
615 Eig_Val[0] = (c3c - 1) / 3.0;
616 Eig_Val[1] = 0.5 *c2c + Eig_Val[0];
617 Eig_Val[2] = c1c + Eig_Val[1];
618
619 /* permute eigenvectors if required */
620 if (uq_permute) {
621 for (size_t jDim = 0; jDim < 3; jDim++)
622 swap(Eig_Vec[0][jDim], Eig_Vec[2][jDim]);
623 }
624
625 CBlasStructure::EigenRecomposition(A_ij, Eig_Vec, Eig_Val, 3);
626
627 /* compute perturbed Reynolds stress matrix; using under-relaxation factor (uq_urlx)*/
628 for (size_t iDim = 0; iDim < 3; iDim++) {
629 for (size_t jDim = 0; jDim < 3; jDim++) {
630 auto delta_ij = (jDim==iDim)? 1.0 : 0.0;
631 MeanPerturbedRSM[iDim][jDim] = 2.0 * turb_ke * (A_ij[iDim][jDim] + 1.0/3.0 * delta_ij);
632 MeanPerturbedRSM[iDim][jDim] = MeanReynoldsStress[iDim][jDim] +
633 uq_urlx*(MeanPerturbedRSM[iDim][jDim] - MeanReynoldsStress[iDim][jDim]);
634 }
635 }
636 }

638 /*!
639 * \brief Project average gradient onto normal (with or w/o correction) for viscous fluxes of scalar quantities.
640 * \param[in] nDim - Dimension of the flow problem, 2 or 3.
641 * \param[in] nVar - Number of variables.
642 * \param[in] normal - Area vector.
643 * \param[in] coord_i - Coordinate of point i.
644 * \param[in] coord_j - Coordinate of point j.
645 * \param[in] grad_i - Gradient at point i.
646 * \param[in] grad_j - Gradient at point j.
647 * \param[in] correct - Correct
648 * \param[in] var_i - Variable at point i.
649 * \param[in] var_j - Variable at point j.
650 * \param[out] projNormal - Average gradient projected on normal.
651 * \param[out] projCorrected - Corrected gradient projected on normal.
652 * \return (Edge_Vector DOT normal) / |Edge_Vector|^2.
653 */
654 template<class Vec1, class Vec2, class Mat>
655 FORCEINLINE static su2double ComputeProjectedGradient(int nDim, int nVar, const Vec1& normal,
656 const Vec1& coord_i, const Vec1& coord_j,
657 const Mat& grad_i, const Mat& grad_j,
658 bool correct,
659 const Vec2& var_i, const Vec2& var_j,
660 su2double* projNormal,
661 su2double* projCorrected) {
662 assert(nDim == 2 || nDim == 3);
663 nDim = (nDim > 2)? 3 : 2;
664 su2double edgeVec[MAXNDIM], dist_ij_2 = 0.0, proj_vector_ij = 0.0;
665
666 for (int iDim = 0; iDim < nDim; iDim++) {
667 edgeVec[iDim] = coord_j[iDim] - coord_i[iDim];
668 dist_ij_2 += pow(edgeVec[iDim], 2);
669 proj_vector_ij += edgeVec[iDim] * normal[iDim];
670 }
671 proj_vector_ij /= max(dist_ij_2,EPS);
672
673 /*--- Mean gradient approximation. ---*/
674 for (int iVar = 0; iVar < nVar; iVar++) {
675 projNormal[iVar] = 0.0;
676 su2double edgeProj = 0.0;
677
678 for (int iDim = 0; iDim < nDim; iDim++) {
679 su2double meanGrad = 0.5 * (grad_i[iVar][iDim] + grad_j[iVar][iDim]);
680 projNormal[iVar] += meanGrad * normal[iDim];
681 if (correct) edgeProj += meanGrad * edgeVec[iDim];
682 }
683
684 projCorrected[iVar] = projNormal[iVar];
685 if (correct) projCorrected[iVar] -= (edgeProj - (var_j[iVar]-var_i[iVar])) * proj_vector_ij;
686 }
687
688 return proj_vector_ij;
689 }
