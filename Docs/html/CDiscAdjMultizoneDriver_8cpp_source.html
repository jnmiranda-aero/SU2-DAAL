34CDiscAdjMultizoneDriver::CDiscAdjMultizoneDriver(char* confFile,
35 unsigned short val_nZone,
36 SU2_Comm MPICommunicator)
37
38 : CMultizoneDriver(confFile, val_nZone, MPICommunicator) {
39
40 direct_nInst.resize(nZone,1);
41 nInnerIter.resize(nZone);
42
43 for (iZone = 0; iZone < nZone; iZone++)
44 nInnerIter[iZone] = config_container[iZone]->GetnInner_Iter();
45
46 Has_Deformation.resize(nZone) = false;
47
48
49 FixPtCorrector.resize(nZone);
50 LinSolver.resize(nZone);
51 AdjRHS.resize(nZone);
52 AdjSol.resize(nZone);
53
54 direct_iteration = new CIteration**[nZone];
55 direct_output = new COutput*[nZone];
56
57 for (iZone = 0; iZone < nZone; iZone++) {
58
59 /*--- Instantiate a direct iteration for each zone. ---*/
60
61 direct_iteration[iZone] = new CIteration*[direct_nInst[iZone]];
62
63 for(iInst = 0; iInst < direct_nInst[iZone]; iInst++) {
64
65 switch (config_container[iZone]->GetKind_Solver()) {
66
67 case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
68 direct_iteration[iZone][iInst] = CIterationFactory::CreateIteration(MAIN_SOLVER::EULER, config_container[iZone]);
69 break;
70 case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
71 direct_iteration[iZone][iInst] = CIterationFactory::CreateIteration(MAIN_SOLVER::INC_EULER, config_container[iZone]);
72 break;
73 case MAIN_SOLVER::DISC_ADJ_HEAT:
74 direct_iteration[iZone][iInst] = CIterationFactory::CreateIteration(MAIN_SOLVER::HEAT_EQUATION, config_container[iZone]);
75 break;
76 case MAIN_SOLVER::DISC_ADJ_FEM:
77 direct_iteration[iZone][iInst] = CIterationFactory::CreateIteration(MAIN_SOLVER::FEM_ELASTICITY, config_container[iZone]);
78 break;
79 default:
80 SU2_MPI::Error("There is no discrete adjoint functionality for one of the specified solvers yet.",
81 CURRENT_FUNCTION);
82 }
83 }
84
85 /*--- Instantiate a direct output to get the results of each direct zone. ---*/
86
87 switch (config_container[iZone]->GetKind_Solver()) {
88
89 case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
90 direct_output[iZone] = COutputFactory::CreateOutput(MAIN_SOLVER::EULER, config_container[iZone], nDim);
91 break;
92 case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
93 direct_output[iZone] = COutputFactory::CreateOutput(MAIN_SOLVER::INC_EULER, config_container[iZone], nDim);
94 break;
95 case MAIN_SOLVER::DISC_ADJ_HEAT:
96 direct_output[iZone] = COutputFactory::CreateOutput(MAIN_SOLVER::HEAT_EQUATION, config_container[iZone], nDim);
97 break;
98 case MAIN_SOLVER::DISC_ADJ_FEM:
99 direct_output[iZone] = COutputFactory::CreateOutput(MAIN_SOLVER::FEM_ELASTICITY, config_container[iZone], nDim);
100 break;
101 default:
102 direct_output[iZone] = nullptr;
103 break;
104 }
105
106 direct_output[iZone]->PreprocessHistoryOutput(config_container[iZone], false);
107
108 }

127void CDiscAdjMultizoneDriver::Preprocess(unsigned long TimeIter) {
128
129 const bool time_domain = driver_config->GetTime_Domain();
130
131 for (iZone = 0; iZone < nZone; iZone++) {
132 /*--- Set current time iteration ---*/
133 config_container[iZone]->SetTimeIter(TimeIter);
134
135 if (time_domain)
136 config_container[iZone]->SetPhysicalTime(static_cast<su2double>(TimeIter)*config_container[iZone]->GetDelta_UnstTimeND());
137 else
138 config_container[iZone]->SetPhysicalTime(0.0);
139
140 /*--- Preprocess the iteration of each zone. ---*/
141
142 iteration_container[iZone][INST_0]->Preprocess(output_container[iZone], integration_container, geometry_container,
143 solver_container, numerics_container, config_container, surface_movement,
144 grid_movement, FFDBox, iZone, INST_0);
145 }
146
147 if (TimeIter) {
148 /*--- Reset cross-terms before new time iterations. ---*/
149 for (auto& matOfMat : Cross_Terms)
150 for (auto& vecOfMat : matOfMat)
151 for (auto& mat : vecOfMat)
152 mat = 0.0;
153
154 /*--- Initialize external with dynamic contributions. ---*/
155 SetExternalToDualTimeDer();
156 }

160void CDiscAdjMultizoneDriver::StartSolver() {
161
162 const bool time_domain = driver_config->GetTime_Domain();
163
164 /*--- Main external loop of the solver. Runs for the number of time steps required. ---*/
165
166 if (rank == MASTER_NODE) {
167 cout <<"\n------------------------------ Begin Solver -----------------------------" << endl;
168
169 cout << "\nSimulation Run using the Discrete Adjoint Multizone Driver" << endl;
170
171 if (time_domain)
172 cout << "The simulation will run for " << driver_config->GetnTime_Iter() << " time steps." << endl;
173 }
174
175 /*--- General setup. ---*/
176
177 for (iZone = 0; iZone < nZone; iZone++) {
178
179 wrt_sol_freq = min(wrt_sol_freq, config_container[iZone]->GetVolumeOutputFrequency(0));
180
181 /*--- Set BGS_Solution_k to Solution, this is needed to restart
182 * correctly as the first OF gradient will overwrite the solution. ---*/
183
184 Set_BGSSolution_k_To_Solution(iZone);
185
186 /*--- Prepare Krylov or quasi-Newton methods. ---*/
187
188 const auto nPoint = geometry_container[iZone][INST_0][MESH_0]->GetnPoint();
189 const auto nPointDomain = geometry_container[iZone][INST_0][MESH_0]->GetnPointDomain();
190 const auto nVar = GetTotalNumberOfVariables(iZone, true);
191
192 if (config_container[iZone]->GetNewtonKrylov() &&
193 config_container[iZone]->GetnQuasiNewtonSamples() >= KrylovMinIters) {
194 AdjRHS[iZone].Initialize(nPoint, nPointDomain, nVar, nullptr);
195 AdjSol[iZone].Initialize(nPoint, nPointDomain, nVar, nullptr);
196 LinSolver[iZone].SetToleranceType(LinearToleranceType::RELATIVE);
197 }
198 else if (config_container[iZone]->GetnQuasiNewtonSamples() > 1) {
199 FixPtCorrector[iZone].resize(config_container[iZone]->GetnQuasiNewtonSamples(), nPoint, nVar, nPointDomain);
200 }
201 }
202
203 /*--- Size and initialize the matrix of cross-terms. ---*/
204
205 InitializeCrossTerms();
206
207 /*--- Run time iterations. ---*/
208
209 while (TimeIter < driver_config->GetnTime_Iter()) {
210
211 Preprocess(TimeIter);
212
213 Run();
214
215 TimeIter++;
216 }

220bool CDiscAdjMultizoneDriver::Iterate(unsigned short iZone, unsigned long iInnerIter, bool KrylovMode) {
221
222 config_container[iZone]->SetInnerIter(iInnerIter);
223
224 /*--- Evaluate the tape section belonging to solvers in iZone.
225 * Only evaluate TRANSFER terms on the last iteration or after convergence. ---*/
226
227 eval_transfer = (eval_transfer || (iInnerIter == nInnerIter[iZone]-1)) && !KrylovMode;
228
229 ComputeAdjoints(iZone, eval_transfer);
230
231 /*--- Extracting adjoints for solvers in iZone w.r.t. to outputs in iZone (diagonal part). ---*/
232
233 iteration_container[iZone][INST_0]->IterateDiscAdj(geometry_container, solver_container,
234 config_container, iZone, INST_0, false);
235
236 /*--- Use QN driver to improve the solution. ---*/
237
238 if (FixPtCorrector[iZone].size()) {
239 GetAllSolutions(iZone, true, FixPtCorrector[iZone].FPresult());
240 FixPtCorrector[iZone].compute();
241 if(iInnerIter) SetAllSolutions(iZone, true, FixPtCorrector[iZone]);
242 }
243
244 /*--- This is done explicitly here for multizone cases, only in inner iterations and not when
245 * extracting cross terms so that the adjoint residuals in each zone still make sense. ---*/
246
247 if (!KrylovMode) SetSolutionOldToSolution(iZone);
248
249 /*--- Print out the convergence data to screen and history file. ---*/
250
251 return iteration_container[iZone][INST_0]->Monitor(output_container[iZone], integration_container, geometry_container,
252 solver_container, numerics_container, config_container,
253 surface_movement, grid_movement, FFDBox, iZone, INST_0);
254}

256void CDiscAdjMultizoneDriver::KrylovInnerIters(unsigned short iZone) {
257
258 /*--- Use FGMRES to solve the adjoint system, the RHS is -External,
259 * the solution are the iZone adjoint variables + External,
260 * Recall that External also contains the OF gradient. ---*/
261
262 GetAdjointRHS(iZone, AdjRHS[iZone]);
263
264 AddExternalToSolution(iZone);
265
266 GetAllSolutions(iZone, true, AdjSol[iZone]);
267
268 const bool monitor = config_container[iZone]->GetWrt_ZoneConv();
269 const auto product = AdjointProduct(this, iZone);
270
271 /*--- Manipulate the screen output frequency to avoid printing garbage. ---*/
272 const auto wrtFreq = config_container[iZone]->GetScreen_Wrt_Freq(2);
273 config_container[iZone]->SetScreen_Wrt_Freq(2, nInnerIter[iZone]);
274 LinSolver[iZone].SetMonitoringFrequency(wrtFreq);
275
276 Scalar eps = 1.0;
277 for (auto totalIter = nInnerIter[iZone]; totalIter >= KrylovMinIters && eps > KrylovTol;) {
278 Scalar eps_l = 0.0;
279 Scalar tol_l = KrylovTol / eps;
280 auto iter = min(totalIter-2ul, config_container[iZone]->GetnQuasiNewtonSamples()-2ul);
281 iter = LinSolver[iZone].FGMRES_LinSolver(AdjRHS[iZone], AdjSol[iZone], product, Identity(),
282 tol_l, iter, eps_l, monitor, config_container[iZone]);
283 totalIter -= iter+1;
284 eps *= eps_l;
285 }
286
287 /*--- Store the solution and restore user settings. ---*/
288 SetAllSolutions(iZone, true, AdjSol[iZone]);
289 config_container[iZone]->SetScreen_Wrt_Freq(2, wrtFreq);
290
291 /*--- Set the old solution such that iterating gives meaningful residuals. ---*/
292 AdjSol[iZone] += AdjRHS[iZone];
293 SetAllSolutionsOld(iZone, true, AdjSol[iZone]);
294
295 /*--- Iterate to evaluate cross terms and residuals, this cannot happen within GMRES
296 * because the vectors it multiplies by the Jacobian are not the actual solution. ---*/
297 eval_transfer = true;
298 Iterate(iZone, product.iInnerIter);
299
300 /*--- Set the solution as obtained from GMRES, otherwise it would be GMRES+Iterate once.
301 * This is set without the "External" (by adding RHS above) so that it can be added
302 * again in the next outer iteration with new contributions from other zones. ---*/
303 SetAllSolutions(iZone, true, AdjSol[iZone]);
304
305}

307void CDiscAdjMultizoneDriver::Run() {
308
309 const unsigned long nOuterIter = driver_config->GetnOuter_Iter();
310 const bool time_domain = driver_config->GetTime_Domain();
311 driver_config->Set_StartTime(SU2_MPI::Wtime());
312
313 /*--- If the gradient of the objective function is 0 so are the adjoint variables.
314 * Unless in unsteady problems where there are other contributions to the RHS. ---*/
315
316 const auto zeroGrad = EvaluateObjectiveFunctionGradient();
317
318 if (zeroGrad && !time_domain) {
319 if (rank == MASTER_NODE) {
320 cout << "\nThe gradient of the objective function is numerically 0.";
321 cout << "\nThis implies that the adjoint variables are also 0.\n\n";
322 }
323 EvaluateSensitivities(0, true);
324 return;
325 }
326
327 /*--- Loop over the number of outer iterations. ---*/
328
329 for (unsigned long iOuterIter = 0, StopCalc = false; !StopCalc; iOuterIter++) {
330
331 driver_config->SetOuterIter(iOuterIter);
332
333 for (iZone = 0; iZone < nZone; iZone++)
334 config_container[iZone]->SetOuterIter(iOuterIter);
335
336 /*--- For the adjoint iteration we need the derivatives of the iteration function with
337 * respect to the state (and possibly the mesh coordinate) variables.
338 * Since these derivatives do not change in the steady state case we only have to record
339 * if the current recording is different from them.
340 *
341 * To set the tape appropriately, the following recording methods are provided:
342 * (1) CLEAR_INDICES: All information from a previous recording is removed.
343 * (2) SOLUTION_VARIABLES: State variables of all solvers in a zone as input.
344 * (3) MESH_COORDS / MESH_DEFORM: Mesh coordinates as input.
345 * (4) SOLUTION_AND_MESH: Mesh coordinates and state variables as input.
346 *
347 * By default, all (state and mesh coordinate variables) will be declared as output,
348 * since it does not change the computational effort, just the memory consumption of the tape. ---*/
349
350
351 /*--- If we want to set up zone-specific tapes (retape), we do not need to record
352 * here. Otherwise, the whole tape of a coupled run will be created. ---*/
353
354 if (RecordingState != RECORDING::SOLUTION_VARIABLES) {
355 SetRecording(RECORDING::CLEAR_INDICES, Kind_Tape::FULL_TAPE, ZONE_0);
356 SetRecording(RECORDING::SOLUTION_VARIABLES, Kind_Tape::FULL_TAPE, ZONE_0);
357 }
358
359 /*-- Start loop over zones. ---*/
360
361 for (iZone = 0; iZone < nZone; iZone++) {
362
363 config_container[iZone]->Set_StartTime(SU2_MPI::Wtime());
364
365 /*--- Start inner iterations from where we stopped in previous outer iteration. ---*/
366
367 Set_Solution_To_BGSSolution_k(iZone);
368
369 /*--- Inner loop to allow for multiple adjoint updates with respect to solvers in iZone. ---*/
370
371 const bool restart = config_container[iZone]->GetRestart();
372 const bool no_restart = (iOuterIter > 0) || !restart;
373 eval_transfer = false;
374
375 /*--- Reset QN driver for new inner iterations. ---*/
376
377 if (FixPtCorrector[iZone].size()) {
378 FixPtCorrector[iZone].reset();
379 if(restart && (iOuterIter==1)) GetAllSolutions(iZone, true, FixPtCorrector[iZone]);
380 }
381
382 if (!config_container[iZone]->GetNewtonKrylov() || !no_restart || nInnerIter[iZone] < KrylovMinIters) {
383
384 /*--- Regular fixed-point, possibly with quasi-Newton method. ---*/
385
386 for (unsigned long iInnerIter = 0; iInnerIter < nInnerIter[iZone]; iInnerIter++) {
387
388 /*--- Add off-diagonal contribution (including the OF gradient) to Solution. ---*/
389
390 if (no_restart || (iInnerIter > 0)) {
391 AddExternalToSolution(iZone);
392 }
393 else {
394 /*--- If we restarted, Solution already has all contributions,
395 * we run only one inner iter to compute the cross terms. ---*/
396 eval_transfer = true;
397 }
398
399 const bool converged = Iterate(iZone, iInnerIter);
400
401 if (eval_transfer) break;
402
403 eval_transfer = converged;
404 }
405 }
406 else {
407 /*--- Use Krylov method to drive inner iterations of this zone. ---*/
408
409 KrylovInnerIters(iZone);
410 }
411
412 /*--- Off-diagonal (coupling term) BGS update. ---*/
413
414 for (unsigned short jZone = 0; jZone < nZone; jZone++) {
415
416 if (jZone != iZone && interface_container[jZone][iZone] != nullptr) {
417
418 /*--- Extracting adjoints for solvers in jZone w.r.t. to the output of all solvers in iZone,
419 * that is, for the cases iZone != jZone we are evaluating cross derivatives between zones. ---*/
420
421 config_container[jZone]->SetInnerIter(0);
422 iteration_container[jZone][INST_0]->IterateDiscAdj(geometry_container, solver_container,
423 config_container, jZone, INST_0, true);
424
425 /*--- Extract the cross-term performing a relaxed update of it and of the sum (External) for jZone. ---*/
426
427 UpdateCrossTerm(iZone, jZone);
428 }
429 }
430
431 /*--- Compute residual from Solution and Solution_BGS_k and update the latter. ---*/
432
433 SetResidual_BGS(iZone);
434 Set_BGSSolution_k_To_Solution(iZone);
435 }
436
437 /*--- Set the multizone output. ---*/
438
439 driver_output->SetMultizoneHistoryOutput(output_container, config_container, driver_config, TimeIter, iOuterIter);
440
441 /*--- Check for convergence. ---*/
442
443 StopCalc = driver_output->GetConvergence() || (iOuterIter == nOuterIter-1) ||
444 ((nZone==1) && output_container[ZONE_0]->GetConvergence());
445
446 /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
447
448 AD::ClearAdjoints();
449
450 /*--- Compute the geometrical sensitivities and write them to file, except for time_domain. ---*/
451
452 if (time_domain) continue;
453
454 if (StopCalc || ((iOuterIter % wrt_sol_freq == 0) && (iOuterIter != 0)))
455 EvaluateSensitivities(iOuterIter, StopCalc);
456 }
457
458 if (time_domain) {
459 EvaluateSensitivities(TimeIter, (TimeIter+1) == driver_config->GetnTime_Iter());
460 }
461
462}

464bool CDiscAdjMultizoneDriver::EvaluateObjectiveFunctionGradient() {
465
466 /*--- Evaluate the objective function gradient w.r.t. the solutions of all zones. ---*/
467
468 SetRecording(RECORDING::CLEAR_INDICES, Kind_Tape::OBJECTIVE_FUNCTION_TAPE, ZONE_0);
469 SetRecording(RECORDING::SOLUTION_VARIABLES, Kind_Tape::OBJECTIVE_FUNCTION_TAPE, ZONE_0);
470 RecordingState = RECORDING::CLEAR_INDICES;
471
472 AD::ClearAdjoints();
473 SetAdjObjFunction();
474 AD::ComputeAdjoint(OBJECTIVE_FUNCTION, START);
475
476 /*--- Initialize External with the objective function gradient. ---*/
477
478 su2double rhs_norm = 0.0;
479
480 for (iZone = 0; iZone < nZone; iZone++) {
481
482 iteration_container[iZone][INST_0]->IterateDiscAdj(geometry_container, solver_container,
483 config_container, iZone, INST_0, false);
484 AddSolutionToExternal(iZone);
485
486 for (unsigned short iSol=0; iSol < MAX_SOLS; iSol++) {
487 auto solver = solver_container[iZone][INST_0][MESH_0][iSol];
488 if (solver && solver->GetAdjoint())
489 for (unsigned short iVar=0; iVar < solver->GetnVar(); ++iVar)
490 rhs_norm += solver->GetRes_RMS(iVar);
491 }
492 }
493
494 return rhs_norm < EPS;
495}

497void CDiscAdjMultizoneDriver::EvaluateSensitivities(unsigned long Iter, bool force_writing) {
498
499 /*--- SetRecording stores the computational graph on one iteration of the direct problem. Calling it with NONE
500 * as argument ensures that all information from a previous recording is removed. ---*/
501
502 SetRecording(RECORDING::CLEAR_INDICES, Kind_Tape::FULL_TAPE, ZONE_0);
503
504 /*--- Store the computational graph of one direct iteration with the mesh coordinates as input. ---*/
505
506 SetRecording(RECORDING::MESH_COORDS, Kind_Tape::FULL_TAPE, ZONE_0);
507
508 /*--- Initialize the adjoint of the output variables of the iteration with the adjoint solution
509 * of the current iteration. The values are passed to the AD tool. ---*/
510
511 for (iZone = 0; iZone < nZone; iZone++) {
512
513 Set_Solution_To_BGSSolution_k(iZone);
514
515 AddExternalToSolution(iZone);
516
517 iteration_container[iZone][INST_0]->InitializeAdjoint(solver_container, geometry_container,
518 config_container, iZone, INST_0);
519 }
520
521 /*--- Initialize the adjoint of the objective function with 1.0. ---*/
522
523 SetAdjObjFunction();
524
525 /*--- Interpret the stored information by calling the corresponding routine of the AD tool. ---*/
526
527 AD::ComputeAdjoint();
528
529 /*--- Extract the computed sensitivity values. ---*/
530
531 for (iZone = 0; iZone < nZone; iZone++) {
532
533 auto config = config_container[iZone];
534 auto solvers = solver_container[iZone][INST_0][MESH_0];
535 auto geometry = geometry_container[iZone][INST_0][MESH_0];
536
537 int IDX_SOL = -1;
538 if (config->GetFluidProblem()) IDX_SOL = ADJFLOW_SOL;
539 else if (config->GetHeatProblem()) IDX_SOL = ADJHEAT_SOL;
540 else if (config->GetStructuralProblem()) IDX_SOL = ADJFEA_SOL;
541 else {
542 if (rank == MASTER_NODE)
543 cout << "WARNING: Sensitivities not set for one of the specified discrete adjoint solvers!" << endl;
544 continue;
545 }
546
547 if (Has_Deformation(iZone))
548 solvers[ADJMESH_SOL]->SetSensitivity(geometry, config, solvers[IDX_SOL]);
549 else
550 solvers[IDX_SOL]->SetSensitivity(geometry, config);
551
552 iteration_container[iZone][INST_0]->Postprocess(output_container[iZone], integration_container, geometry_container,
553 solver_container, numerics_container, config_container,
554 surface_movement, grid_movement, FFDBox, iZone, INST_0);
555 }
556
557 /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
558
559 AD::ClearAdjoints();
560
561 /*--- Output files (CMultizoneDriver::Output uses StopCalc to force file output). ---*/
562
563 swap(StopCalc, force_writing);
564 Output(Iter);
565 swap(StopCalc, force_writing);
566
567}

569void CDiscAdjMultizoneDriver::SetRecording(RECORDING kind_recording, Kind_Tape tape_type, unsigned short record_zone) {
570
571 AD::Reset();
572
573 /*--- Prepare for recording by resetting the solution to the initial converged solution. ---*/
574
575 for(iZone = 0; iZone < nZone; iZone++) {
576 for (unsigned short iSol = 0; iSol < MAX_SOLS; iSol++) {
577 for (unsigned short iMesh = 0; iMesh <= config_container[iZone]->GetnMGLevels(); iMesh++) {
578 auto solver = solver_container[iZone][INST_0][iMesh][iSol];
579 if (solver && solver->GetAdjoint()) {
580 solver->SetRecording(geometry_container[iZone][INST_0][iMesh], config_container[iZone]);
581 }
582 }
583 }
584 }
585
586 if (rank == MASTER_NODE) {
587 cout << "\n-------------------------------------------------------------------------\n";
588 switch(kind_recording) {
589 case RECORDING::CLEAR_INDICES: cout << "Clearing the computational graph." << endl; break;
590 case RECORDING::MESH_COORDS: cout << "Storing computational graph wrt MESH COORDINATES." << endl; break;
591 case RECORDING::SOLUTION_VARIABLES: cout << "Storing computational graph wrt CONSERVATIVE VARIABLES." << endl; break;
592 default: break;
593 }
594 }
595
596 /*--- Enable recording and register input of the flow iteration (conservative variables or node coordinates) --- */
597
598 if(kind_recording != RECORDING::CLEAR_INDICES) {
599
600 AD::StartRecording();
601
602 AD::Push_TapePosition(); /// START
603
604 for (iZone = 0; iZone < nZone; iZone++) {
605
606 /*--- In multi-physics, MESH_COORDS is an umbrella term for "geometric sensitivities",
607 * if a zone has mesh deformation its recording type needs to change to MESH_DEFORM
608 * as those sensitivities are managed by the adjoint mesh solver instead. ---*/
609
610 RECORDING type_recording = kind_recording;
611
612 if (Has_Deformation(iZone) && (kind_recording == RECORDING::MESH_COORDS)) {
613 type_recording = RECORDING::MESH_DEFORM;
614 }
615
616 iteration_container[iZone][INST_0]->RegisterInput(solver_container, geometry_container,
617 config_container, iZone, INST_0, type_recording);
618 }
619 }
620

622
623 for (iZone = 0; iZone < nZone; iZone++) {
624 iteration_container[iZone][INST_0]->SetDependencies(solver_container, geometry_container, numerics_container,
625 config_container, iZone, INST_0, kind_recording);
626 }
627
628 AD::Push_TapePosition(); /// DEPENDENCIES
629
630 /*--- Extract the objective function and store it.
631 * It is necessary to include data transfer and mesh updates in this section as some functions
632 * computed in one zone depend explicitly on the variables of others through that path. --- */
633
634 if ((tape_type == Kind_Tape::OBJECTIVE_FUNCTION_TAPE) || (kind_recording == RECORDING::MESH_COORDS)) {
635 HandleDataTransfer();
636 for (iZone = 0; iZone < nZone; iZone++) {
637 if (Has_Deformation(iZone)) {
638 iteration_container[iZone][INST_0]->SetDependencies(solver_container, geometry_container, numerics_container,
639 config_container, iZone, INST_0, kind_recording);
640 }
641 }
642 SetObjFunction(kind_recording);
643 }
644
645 AD::Push_TapePosition(); /// OBJECTIVE_FUNCTION
646
647 if (tape_type != Kind_Tape::OBJECTIVE_FUNCTION_TAPE) {
648
649 /*--- We do the communication here to not differentiate wrt updated boundary data.
650 * For recording w.r.t. mesh coordinates the transfer was included before the
651 * objective function, so we do not repeat it here. ---*/
652
653 if (kind_recording != RECORDING::MESH_COORDS) {
654 HandleDataTransfer();
655 }
656
657 AD::Push_TapePosition(); /// TRANSFER
658
659 for(iZone = 0; iZone < nZone; iZone++) {
660
661 AD::Push_TapePosition(); /// enter_zone
662
663 DirectIteration(iZone, kind_recording);
664
665 iteration_container[iZone][INST_0]->RegisterOutput(solver_container, geometry_container,
666 config_container, iZone, INST_0);
667 AD::Push_TapePosition(); /// leave_zone
668 }
669 PrintDirectResidual(kind_recording);
670 }
671
672 if (kind_recording != RECORDING::CLEAR_INDICES && driver_config->GetWrt_AD_Statistics()) {
673 if (rank == MASTER_NODE) AD::PrintStatistics();
674#ifdef CODI_REVERSE_TYPE
675 if (size > SINGLE_NODE) {
676 su2double myMem = AD::getTape().getTapeValues().getUsedMemorySize(), totMem = 0.0;
677 SU2_MPI::Allreduce(&myMem, &totMem, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
678 if (rank == MASTER_NODE) {
679 cout << "MPI\n";
680 cout << "-------------------------------------\n";
681 cout << " Total memory used : " << totMem << " MB\n";
682 cout << "-------------------------------------\n" << endl;
683 }
684 }
685#endif
686 }
687
688 AD::StopRecording();
689
690 RecordingState = kind_recording;
691}

791

817 AD::ComputeAdjoint(TRANSFER, OBJECTIVE_FUNCTION);

820 * are extracted (e.g. AoA, Mach, etc.) ---*/
821

826void CDiscAdjMultizoneDriver::InitializeCrossTerms() {
827
828 if (Cross_Terms.empty()) {
829 Cross_Terms.resize(nZone, vector<vector<su2passivematrix> >(nZone));
830 }
831
832 for(auto iZone = 0u; iZone < nZone; iZone++) {
833 for (auto jZone = 0u; jZone < nZone; jZone++) {
834 if (iZone != jZone || interface_container[jZone][iZone] != nullptr) {
835
836 /*--- If jZone contributes to iZone in the primal problem, then
837 * iZone contributes to jZone in the adjoint problem. ---*/
838
839 Cross_Terms[iZone][jZone].resize(MAX_SOLS);
840
841 for (auto iSol = 0u; iSol < MAX_SOLS; iSol++) {
842
843 auto solver = solver_container[jZone][INST_0][MESH_0][iSol];
844 if (solver && solver->GetAdjoint()) {
845
846 const auto nPoint = geometry_container[jZone][INST_0][MESH_0]->GetnPoint();
847 const auto nVar = solver->GetnVar();
848 Cross_Terms[iZone][jZone][iSol].resize(nPoint,nVar) = 0.0;
849 }
850 }
851 }
852 }

856void CDiscAdjMultizoneDriver::HandleDataTransfer() {
857
858 for(iZone = 0; iZone < nZone; iZone++) {
859
860 /*--- In principle, the mesh does not need to be updated ---*/
861 bool DeformMesh = false;
862
863 /*--- Transfer from all the remaining zones ---*/
864 for (unsigned short jZone = 0; jZone < nZone; jZone++){
865 /*--- The target zone is iZone ---*/
866 if (jZone != iZone && interface_container[jZone][iZone] != nullptr) {
867 DeformMesh |= TransferData(jZone, iZone);
868 }
869 }
870
871 /*--- If a mesh update is required due to the transfer of data ---*/
872 const unsigned long ExtIter = 0;
873 if (DeformMesh) DynamicMeshUpdate(iZone, ExtIter);
874
875 Has_Deformation(iZone) = DeformMesh;
876 }
