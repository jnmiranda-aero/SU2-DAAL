158 void CFaceOfElement::CreateUniqueNumberingWithOrientation() {
159  /*--- Determine the element type and create the unique numbering accordingly. ---*/
160  bool swapElements = false;
161  switch (nCornerPoints) {
162  case 2: {
163  /* Element is a line. Check if the node numbering must be swapped. If so
164  also the element information must be swapped, because element 0 is to
165  the left of the face and element 1 to the right. */
166  if (cornerPoints[1] < cornerPoints[0]) {
167  swap(cornerPoints[0], cornerPoints[1]);
168  swapElements = true;
169  }
170  break;
171  }
172 
173  case 3: {
174  /* Element is a triangle. The vertices are sorted in increasing order.
175  If the sequence of the new numbering is opposite to the current
176  numbering, the element information must be exchanged, because
177  element 0 is to the left of the face and element 1 to the right. */
178  unsigned long nn[] = {cornerPoints[0], cornerPoints[1], cornerPoints[2]};
179  unsigned short ind = 0;
180  if (nn[1] < nn[ind]) ind = 1;
181  if (nn[2] < nn[ind]) ind = 2;
182 
183  unsigned short indm1 = ind == 0 ? 2 : ind - 1; // Next lower index.
184  unsigned short indp1 = ind == 2 ? 0 : ind + 1; // Next upper index.
185 
186  if (nn[indp1] < nn[indm1]) {
187  /* The orientation of the triangle remains the same.
188  Store the new sorted node numbering. */
189  cornerPoints[0] = nn[ind];
190  cornerPoints[1] = nn[indp1];
191  cornerPoints[2] = nn[indm1];
192  } else {
193  /* The orientation of the triangle changes. Store the new
194  sorted node numbering and set swapElements to true. */
195  cornerPoints[0] = nn[ind];
196  cornerPoints[1] = nn[indm1];
197  cornerPoints[2] = nn[indp1];
198  swapElements = true;
199  }
200 
201  break;
202  }
203 
204  case 4: {
205  /* Element is a quadrilateral. The vertices are sorted in increasing order
206  under the condition neighboring vertices remain neighbors. If the
207  sequence of the new numbering is opposite to the current
208  numbering, the element information must be exchanged, because
209  element 0 is to the left of the face and element 1 to the right. */
210  unsigned long nn[] = {cornerPoints[0], cornerPoints[1], cornerPoints[2], cornerPoints[3]};
211  unsigned short ind = 0;
212  if (nn[1] < nn[ind]) ind = 1;
213  if (nn[2] < nn[ind]) ind = 2;
214  if (nn[3] < nn[ind]) ind = 3;
215 
216  unsigned short indm1 = ind == 0 ? 3 : ind - 1; // Next lower index.
217  unsigned short indp1 = ind == 3 ? 0 : ind + 1; // Next upper index.
218  unsigned short indp2 = ind >= 2 ? ind - 2 : ind + 2; // Opposite index.
219 
220  if (nn[indp1] < nn[indm1]) {
221  /* The orientation of the quadrilateral remains the same.
222  Store the new sorted node numbering. */
223  cornerPoints[0] = nn[ind];
224  cornerPoints[1] = nn[indp1];
225  cornerPoints[2] = nn[indp2];
226  cornerPoints[3] = nn[indm1];
227  } else {
228  /* The orientation of the quadrilateral changes. Store the new
229  sorted node numbering and set swapElements to true. */
230  cornerPoints[0] = nn[ind];
231  cornerPoints[1] = nn[indm1];
232  cornerPoints[2] = nn[indp2];
233  cornerPoints[3] = nn[indp1];
234  swapElements = true;
235  }
236 
237  break;
238  }
239 
240  default: {
241  ostringstream message;
242  message << "Unknown surface element type with " << nCornerPoints << " corners." << endl;
243  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
244  }
245  }
246 
247  /* Swap the element information, if needed. */
248  if (swapElements) {
249  swap(elemID0, elemID1);
250  swap(nPolyGrid0, nPolyGrid1);
251  swap(nPolySol0, nPolySol1);
252  swap(nDOFsElem0, nDOFsElem1);
253  swap(elemType0, elemType1);
254  swap(faceID0, faceID1);
255  }
256 }

283 bool CMatchingFace::operator<(const CMatchingFace& other) const {
284  /* First compare the number of corner points. ---*/
285  if (nCornerPoints != other.nCornerPoints) return nCornerPoints < other.nCornerPoints;
286 
287  /*--- Determine the tolerance for comparing both objects. ---*/
288  const su2double tol = min(tolForMatching, other.tolForMatching);
289 
290  /*--- Loop over the number of corner points and dimensions and compare the
291  coordinates. If considered different, return true if the current face
292  is considered smaller and false otherwise. ---*/
293  for (unsigned short k = 0; k < nCornerPoints; ++k) {
294  for (unsigned short l = 0; l < nDim; ++l) {
295  if (fabs(cornerCoor[k][l] - other.cornerCoor[k][l]) > tol) return cornerCoor[k][l] < other.cornerCoor[k][l];
296  }
297  }
298 
299  /*--- Both objects are considered the same. Return false. ---*/
300  return false;

303 void CMatchingFace::SortFaceCoordinates() {
304  /*--- Determine the tolerance for a matching point for this face. This is
305  accomplished by computing the minimum distance between the points of
306  the face, multiplied by a relative tolerance. ---*/
307  for (unsigned short k = 0; k < nCornerPoints; ++k) {
308  for (unsigned short j = (k + 1); j < nCornerPoints; ++j) {
309  su2double dist = 0.0;
310  for (unsigned short l = 0; l < nDim; ++l) {
311  su2double ds = cornerCoor[k][l] - cornerCoor[j][l];
312  dist += ds * ds;
313  }
314  dist = sqrt(dist);
315 
316  if (k == 0 && j == 1)
317  tolForMatching = dist;
318  else
319  tolForMatching = min(tolForMatching, dist);
320  }
321  }
322 
323  tolForMatching *= 1.e-6;
324 
325  /*--- Sort the points in increasing order based on the coordinates.
326  An insertion sort algorithm is used, which is quite efficient
327  for at most four corner points. ---*/
328  for (unsigned short k = 1; k < nCornerPoints; ++k) {
329  for (unsigned short j = k; j > 0; --j) {
330  /* Check if cornerCoor[j] is considered less than cornerCoor[j-1]. */
331  bool lessThan = false;
332  for (unsigned short l = 0; l < nDim; ++l) {
333  if (fabs(cornerCoor[j][l] - cornerCoor[j - 1][l]) > tolForMatching) {
334  lessThan = cornerCoor[j][l] < cornerCoor[j - 1][l];
335  break;
336  }
337  }
338 
339  /* If cornerCoor[j] is less than cornerCoor[j-1] they must be swapped.
340  Otherwise an exit can be made from the j-loop. */
341  if (lessThan) {
342  for (unsigned short l = 0; l < nDim; ++l) swap(cornerCoor[j][l], cornerCoor[j - 1][l]);
343  } else
344  break;
345  }
346  }
347 }

407  /*--- Check the grid ---*/
408 
409  if (mesh_file.fail())
410  SU2_MPI::Error(string("There is no mesh file (CPhysicalGeometry)!! ") + val_mesh_filename, CURRENT_FUNCTION);
411 
412  /*--- If more than one, find the zone in the mesh file ---*/
413

415  if (time_spectral) {
416  if (rank == MASTER_NODE) cout << "Reading time spectral instance " << val_iZone + 1 << ":" << endl;

419  /*--- Search for the current domain ---*/
420  position = text_line.find("IZONE=", 0);
421  if (position != string::npos) {

431  }
432 
433  /*--- Read grid file with format SU2 ---*/
434 
435  while (getline(mesh_file, text_line)) {
436  /*--- Read the dimension of the problem ---*/
437 
438  position = text_line.find("NDIME=", 0);
439  if (position != string::npos) {
440  if (!domain_flag) {
441  text_line.erase(0, 6);
442  nDim = atoi(text_line.c_str());

445  if (nDim == 3) cout << "Three dimensional problem." << endl;

448  } else {

450  }
451  }
452 
453  /*--- Read the information about inner elements ---*/
454 
455  position = text_line.find("NELEM=", 0);
456  if (position != string::npos) {
457  text_line.erase(0, 6);
458  stringstream stream_line(text_line);
459  stream_line >> Global_nElem;
460 
461  if ((rank == MASTER_NODE) && (size > SINGLE_NODE))
462  cout << Global_nElem << " interior elements before parallel partitioning." << endl;
463 
464  /*--- Check if the number of cores used is larger than the number of
465  elements. Terminate if this is the case, because it does not make
466  sense to do this. ---*/
467  unsigned long nCores = size; // Correct for the number of cores per rank.
468  if (nCores > Global_nElem) {
469  ostringstream message;
470  message << "The number of cores, " << nCores;

472  message << "This is not an efficient use of the resources and therefore "
473  << "SU2 will terminate.";

475  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
476  }
477 
478  /*--- Compute the number of elements that will be on each processor.
479  This is a linear partitioning with the addition of a simple load
480  balancing for any remainder elements. ---*/
481  unsigned long total_elem_accounted = 0;
482  for (unsigned long i = 0; i < (unsigned long)size; i++) {
483  nPointLinear[i] = Global_nElem / size;
484  total_elem_accounted = total_elem_accounted + nPointLinear[i];

534

542  }
543

545  nDOFs_tot += nDOFsSol;
546  }
547

549  break;
550  }
551  }

554

556  vector<unsigned long> nodeIDsElemLoc;
557  nodeIDsElemLoc.reserve(nDOFsGrid_Local);
558  for (unsigned long i = 0; i < loc_element_count; i++) {

561  }

564  vector<unsigned long>::iterator lastNode;

599  getline(mesh_file, text_line);
600 
601  if (binary_search(nodeIDsElemLoc.begin(), nodeIDsElemLoc.end(), i)) {
602  istringstream point_line(text_line);
603  su2double Coord[3] = {0.0};
604  point_line >> Coord[0];
605  point_line >> Coord[1];
606  if (nDim == 3) point_line >> Coord[2];
607  nodes->SetCoord(ii, Coord);
608  nodes->SetGlobalIndex(ii, i);
609  ++ii;
610  }
611  }
612 
613  break;
614  }
615  }
616 
617  mesh_file.close();
618 
619  /*--- Determine the faces of the local elements. --- */
620  vector<CFaceOfElement> localFaces;
621  for (unsigned long k = 0; k < loc_element_count; k++) {
622  /*--- Get the global IDs of the corner points of all the faces of this elements. ---*/
623  unsigned short nFaces;
624  unsigned short nPointsPerFace[6];
625  unsigned long faceConn[6][4];
626 
627  elem[k]->GetCornerPointsAllFaces(nFaces, nPointsPerFace, faceConn);
628 
629  /*--- Loop over the faces and add them to localFaces. ---*/
630  for (unsigned short i = 0; i < nFaces; ++i) {
631  CFaceOfElement thisFace;
632  thisFace.nCornerPoints = nPointsPerFace[i];
633  for (unsigned short j = 0; j < nPointsPerFace[i]; ++j) thisFace.cornerPoints[j] = faceConn[i][j];
634  thisFace.elemID0 = k + beg_node[rank];
635 
636  thisFace.CreateUniqueNumbering();
637  localFaces.push_back(thisFace);
638  }
639  }
640 
641  /*--- Sort localFaces in increasing order and remove the double entities,
642  such that the binary search later on is a bit more efficient. ---*/
643  sort(localFaces.begin(), localFaces.end());
644  vector<CFaceOfElement>::iterator lastFace;
645  lastFace = unique(localFaces.begin(), localFaces.end());
646  localFaces.erase(lastFace, localFaces.end());
647 
648  /*--- Open the grid file again and go to the position where
649  the correct zone is stored. ---*/
650  mesh_file.open(val_mesh_filename.c_str(), ios::in);
651 
652  if (val_nZone > 1 && !time_spectral) {
653  while (getline(mesh_file, text_line)) {
654  position = text_line.find("IZONE=", 0);
655  if (position != string::npos) {
656  text_line.erase(0, 6);
657  unsigned short jDomain = atoi(text_line.c_str());
658  if (jDomain == val_iZone + 1) break;
659  }
660  }
661  }
662 
663  /*--- While loop to read the boundary information. ---*/
664  while (getline(mesh_file, text_line)) {
665  /*--- Read number of markers ---*/
666  position = text_line.find("NMARK=", 0);
667  if (position != string::npos) {
668  text_line.erase(0, 6);
669  istringstream stream_line(text_line);
670  stream_line >> nMarker;
671 
672  if (rank == MASTER_NODE) cout << nMarker << " surface markers." << endl;
673  config->SetnMarker_All(nMarker);
674  bound = new CPrimalGrid**[nMarker];
675  nElem_Bound = new unsigned long[nMarker];
676  Tag_to_Marker = new string[nMarker_Max];
677 
678  for (unsigned short iMarker = 0; iMarker < nMarker; iMarker++) {
679  getline(mesh_file, text_line);
680  text_line.erase(0, 11);
681  text_line.erase(remove(text_line.begin(), text_line.end(), ' '), text_line.end());
682  text_line.erase(remove(text_line.begin(), text_line.end(), '\r'), text_line.end());
683  text_line.erase(remove(text_line.begin(), text_line.end(), '\n'), text_line.end());
684 
685  Marker_Tag = text_line;
686 
687  /*--- Read the number of elements for this marker. ---*/
688  getline(mesh_file, text_line);
689  text_line.erase(0, 13);
690  istringstream nmark_line(text_line);
691  unsigned long nElem_Bound_Global;
692  nmark_line >> nElem_Bound_Global;
693  if (rank == MASTER_NODE)
694  cout << nElem_Bound_Global << " boundary elements in index " << iMarker << " (Marker = " << Marker_Tag << ")."
695  << endl;
696 
697  /*--- Define a vector of FEM boundary elements to store the local boundary faces. ---*/
698  vector<CBoundaryFace> boundElems;
699 
700  /*--- Loop over the global boundary faces. ---*/
701  for (unsigned long i = 0; i < nElem_Bound_Global; ++i) {
702  getline(mesh_file, text_line);
703  istringstream bound_line(text_line);
704 
705  /*--- Determine the element type, its number of DOFs and read
706  its node IDs. ---*/
707  unsigned long typeRead;
708  bound_line >> typeRead;
709  unsigned short nPolyGrid = typeRead / 100 + 1;
710  unsigned short VTK_Type = typeRead % 100;
711  unsigned short nDOFEdgeGrid = nPolyGrid + 1;
712 
713  unsigned short nDOFsGrid = 0;
714  CFaceOfElement thisFace;
715  thisFace.cornerPoints[0] = 0;
716  thisFace.cornerPoints[1] = nPolyGrid;
717  switch (VTK_Type) {
718  case LINE:
719  nDOFsGrid = nDOFEdgeGrid;
720  thisFace.nCornerPoints = 2;
721  break;
722 
723  case TRIANGLE:
724  nDOFsGrid = nDOFEdgeGrid * (nDOFEdgeGrid + 1) / 2;
725  thisFace.nCornerPoints = 3;
726  thisFace.cornerPoints[2] = nDOFsGrid - 1;
727  break;
728 
729  case QUADRILATERAL:
730  nDOFsGrid = nDOFEdgeGrid * nDOFEdgeGrid;
731  thisFace.nCornerPoints = 4;
732  thisFace.cornerPoints[2] = nPolyGrid * nDOFEdgeGrid;
733  thisFace.cornerPoints[3] = nDOFsGrid - 1;
734  break;
735 
736  default:
737  ostringstream message;
738  message << "Unknown FEM boundary element value, " << typeRead << ", in " << val_mesh_filename;
739  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
740  }
741 
742  vector<unsigned long> nodeIDs(nDOFsGrid);
743  for (unsigned short j = 0; j < nDOFsGrid; ++j) bound_line >> nodeIDs[j];
744 
745  /*--- Convert the local numbering of thisFace to global numbering
746  and create a unique numbering of these nodes. ---*/
747  for (unsigned short j = 0; j < thisFace.nCornerPoints; ++j)
748  thisFace.cornerPoints[j] = nodeIDs[thisFace.cornerPoints[j]];
749  thisFace.CreateUniqueNumbering();
750 
751  /*--- Check if this boundary face must be stored on this rank.
752  If so, create an object of CBoundaryFace and add it
753  to boundElems. ---*/
754  vector<CFaceOfElement>::iterator low;
755  low = lower_bound(localFaces.begin(), localFaces.end(), thisFace);
756  if (low != localFaces.end()) {
757  if (!(thisFace < *low)) {
758  CBoundaryFace thisBoundFace;
759  thisBoundFace.VTK_Type = VTK_Type;
760  thisBoundFace.nPolyGrid = nPolyGrid;
761  thisBoundFace.nDOFsGrid = nDOFsGrid;
762  thisBoundFace.globalBoundElemID = i;
763  thisBoundFace.domainElementID = low->elemID0;
764  thisBoundFace.Nodes = nodeIDs;
765 
766  boundElems.push_back(thisBoundFace);
767  }
768  }
769  }
770 
771  /*--- Allocate space for the boundary elements and store the ones
772  whose parent element is stored on this rank. ---*/
773  nElem_Bound[iMarker] = boundElems.size();
774  bound[iMarker] = new CPrimalGrid*[nElem_Bound[iMarker]];
775 
776  for (unsigned long i = 0; i < nElem_Bound[iMarker]; ++i)
777  bound[iMarker][i] = new CPrimalGridBoundFEM(boundElems[i].globalBoundElemID, boundElems[i].domainElementID,
778  boundElems[i].VTK_Type, boundElems[i].nPolyGrid,
779  boundElems[i].nDOFsGrid, boundElems[i].Nodes);
780 
781  /*--- Update config information storing the boundary information in the right place ---*/
782  Tag_to_Marker[config->GetMarker_CfgFile_TagBound(Marker_Tag)] = Marker_Tag;
783  config->SetMarker_All_TagBound(iMarker, Marker_Tag);
784  config->SetMarker_All_KindBC(iMarker, config->GetMarker_CfgFile_KindBC(Marker_Tag));
785  config->SetMarker_All_Monitoring(iMarker, config->GetMarker_CfgFile_Monitoring(Marker_Tag));
786  config->SetMarker_All_GeoEval(iMarker, config->GetMarker_CfgFile_GeoEval(Marker_Tag));
787  config->SetMarker_All_Designing(iMarker, config->GetMarker_CfgFile_Designing(Marker_Tag));
788  config->SetMarker_All_Plotting(iMarker, config->GetMarker_CfgFile_Plotting(Marker_Tag));
789  config->SetMarker_All_Analyze(iMarker, config->GetMarker_CfgFile_Analyze(Marker_Tag));
790  config->SetMarker_All_ZoneInterface(iMarker, config->GetMarker_CfgFile_ZoneInterface(Marker_Tag));
791  config->SetMarker_All_DV(iMarker, config->GetMarker_CfgFile_DV(Marker_Tag));
792  config->SetMarker_All_Moving(iMarker, config->GetMarker_CfgFile_Moving(Marker_Tag));
793  config->SetMarker_All_SobolevBC(iMarker, config->GetMarker_CfgFile_SobolevBC(Marker_Tag));
794  config->SetMarker_All_PerBound(iMarker, config->GetMarker_CfgFile_PerBound(Marker_Tag));
795  config->SetMarker_All_SendRecv(iMarker, NONE);
796  }
797 
798  break;
799  }
800  }
801 
802  mesh_file.close();
803 }

805 void CPhysicalGeometry::Read_CGNS_Format_Parallel_FEM(CConfig* config, const string& val_mesh_filename,
806  unsigned short val_iZone, unsigned short val_nZone) {
807 #ifdef HAVE_CGNS
808 
809  /*--- For proper support of the high order elements, at least version 3.3
810  of CGNS must be used. Check this. ---*/
811 #if CGNS_VERSION >= 3300
812 
813  /*--- Check whether the supplied file is truly a CGNS file. ---*/
814  int file_type;
815  if (cg_is_cgns(val_mesh_filename.c_str(), &file_type) != CG_OK)
816  SU2_MPI::Error(val_mesh_filename + string(" is not a CGNS file that can be read."), CURRENT_FUNCTION);
817 
818  /*--- Initialize counters for local/global points & elements ---*/
819  Global_nPoint = 0;
820  Global_nPointDomain = 0;
821  Global_nElem = 0;
822  nelem_edge = 0;
823  Global_nelem_edge = 0;
824  nelem_triangle = 0;
825  Global_nelem_triangle = 0;
826  nelem_quad = 0;
827  Global_nelem_quad = 0;
828  nelem_tetra = 0;
829  Global_nelem_tetra = 0;
830  nelem_hexa = 0;
831  Global_nelem_hexa = 0;
832  nelem_prism = 0;
833  Global_nelem_prism = 0;
834  nelem_pyramid = 0;
835  Global_nelem_pyramid = 0;
836 
837  /*--------------------------------------------------------------------------*/
838  /*--- Checking of the file, determine the dimensions, etc. ---*/
839  /*--------------------------------------------------------------------------*/
840 
841  /*--- Allocate memory for the linear partition of the elements of the mesh.
842  These arrays are the size of the number of ranks. ---*/
843  beg_node = new unsigned long[size];
844  end_node = new unsigned long[size];
845  nPointLinear = new unsigned long[size];
846 
847  /* Open the CGNS file for reading and check if it went OK. */
848  int fn;
849  if (cg_open(val_mesh_filename.c_str(), CG_MODE_READ, &fn) != CG_OK) cg_error_exit();
850  if (rank == MASTER_NODE) {
851  cout << "Reading the CGNS file: " << val_mesh_filename << "." << endl;
852  }
853 
854  /* Get the number of databases. This is the highest node in the CGNS heirarchy.
855  The current implementation assumes that there is only one database. */
856  int nbases;
857  if (cg_nbases(fn, &nbases) != CG_OK) cg_error_exit();
858  if (nbases > 1) {
859  ostringstream message;
860  message << "CGNS file contains " << nbases << " databases." << endl;
861  message << "CGNS reader can handle only 1 at the moment." << endl;
862 
863  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
864  }
865 
866  /* Read the information of the base, especially the number of dimensions. */
867  char cgnsname[CGNS_STRING_SIZE];
868  int cellDim, physDim;
869  const int iBase = 1;
870  if (cg_base_read(fn, iBase, cgnsname, &cellDim, &physDim)) cg_error_exit();
871  nDim = physDim;
872 
873  if (cellDim != physDim) {
874  ostringstream message;
875  message << "The element dimension, " << cellDim << ", differs from the physical dimension, " << physDim << "."
876  << endl;
877  message << "These should be the same for the DG-FEM solver." << endl;
878 
879  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
880  }
881 
882  /* Write the info about the number of dimensions. */
883  if (rank == MASTER_NODE) {
884  if (nDim == 2) cout << "Two dimensional problem." << endl;
885  if (nDim == 3) cout << "Three dimensional problem." << endl;
886  }
887 
888  /* Set the zone ID for CGNS, CGNS indices starts at 1, and check
889  if the requested zone ID is a valid one. */
890  const int iZone = val_iZone + 1;
891 
892  int nzones;
893  if (cg_nzones(fn, iBase, &nzones) != CG_OK) cg_error_exit();
894  if (iZone < 1 || iZone > nzones) {
895  ostringstream message;
896  message << "Zone " << iZone << " requested for reading, but there are only " << nzones
897  << " zones present in the CGNS file." << endl;
898 
899  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
900  }
901 
902  /* Determine the zone type for the requested zone and check if it is
903  unstructured. */
904  ZoneType_t zoneType;
905  if (cg_zone_type(fn, iBase, iZone, &zoneType) != CG_OK) cg_error_exit();
906  if (zoneType != Unstructured)
907  SU2_MPI::Error("Structured CGNS zone found while unstructured expected.", CURRENT_FUNCTION);
908 
909  /* Determine the number of sections for the connectivities in this zone. */
910  int nsections;
911  if (cg_nsections(fn, iBase, iZone, &nsections) != CG_OK) cg_error_exit();
912 
913  /*--------------------------------------------------------------------------*/
914  /*--- Reading and distributing the volume elements. ---*/
915  /*--------------------------------------------------------------------------*/
916 
917  /* Loop over the sections to store the meta data in CGNSElemTypes
918  and to determine the number of volume elements.
919  Note that the indices start at 1 in CGNS. */
920  vector<CCGNSElementType> CGNSElemTypes(nsections);
921 
922  for (int iConn = 1; iConn <= nsections; ++iConn) {
923  CGNSElemTypes[iConn - 1].DetermineMetaData(nDim, fn, iBase, iZone, iConn);
924  if (CGNSElemTypes[iConn - 1].volumeConn) Global_nElem += CGNSElemTypes[iConn - 1].nElem;
925  }
926 
927  if ((rank == MASTER_NODE) && (size > SINGLE_NODE))
928  cout << Global_nElem << " interior elements before parallel partitioning." << endl;
929 
930  /*--- Check if the number of cores used is larger than the number of
931  elements. Terminate if this is the case, because it does not make
932  sense to do this. ---*/
933  unsigned long nCores = size; // Correct for the number of cores per rank.
934  if (nCores > Global_nElem) {
935  ostringstream message;
936  message << "The number of cores, " << nCores;
937  message << ", is larger than the number of elements, " << Global_nElem << "." << endl;
938  message << "This is not an efficient use of the resources and therefore "
939  << "SU2 will terminate.";
940 
941  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
942  }
943 
944  /*--- Compute the number of elements that will be stored on each rank.
945  This is a linear partitioning with the addition of a simple load
946  balancing for any remainder elements. ---*/
947  unsigned long total_elem_accounted = 0;
948  for (unsigned long i = 0; i < (unsigned long)size; i++) {
949  nPointLinear[i] = Global_nElem / size;
950  total_elem_accounted = total_elem_accounted + nPointLinear[i];
951  }
952 
953  /*--- Get the number of remainder elements after the even division ---*/
954  const unsigned long rem_elem = Global_nElem - total_elem_accounted;
955  for (unsigned long i = 0; i < rem_elem; i++) {
956  ++nPointLinear[i];
957  }
958 
959  /*--- Store the local number of elements and the beginning/end index ---*/
960  nElem = nPointLinear[rank];
961  beg_node[0] = 0;
962  end_node[0] = beg_node[0] + nPointLinear[0];
963  for (int i = 1; i < size; i++) {
964  beg_node[i] = end_node[i - 1];
965  end_node[i] = beg_node[i] + nPointLinear[i];
966  }
967 
968  /*--- Allocate space for elements ---*/
969  elem = new CPrimalGrid*[nElem]();
970 
971  /*--- Loop over over the connectivities and read the elements to be stored on
972  this rank. Furthermore, determine the local amount of DOFs for the
973  solution (which may differ from the number of DOFS for the grid). ---*/
974  unsigned long nDOFsLoc = 0, elemCount = 0, locElemCount = 0;
975 
976  for (int iConn = 0; iConn < nsections; ++iConn) {
977  if (CGNSElemTypes[iConn].volumeConn) {
978  /* Determine the global volume element range for this connectivity. */
979  const unsigned long elemCountOld = elemCount;
980  elemCount += CGNSElemTypes[iConn].nElem;
981 
982  /* Check for overlap with the element range this rank is responsible for. */
983  const unsigned long indBegOverlap = max(elemCountOld, beg_node[rank]);
984  const unsigned long indEndOverlap = min(elemCount, end_node[rank]);
985 
986  if (indEndOverlap > indBegOverlap) {
987  /* This rank must read element data from this connectivity section.
988  Determine the offset relative to the start of this section and the
989  number of elements to be read by this rank. */
990  const unsigned long offsetRank = indBegOverlap - elemCountOld;
991  const unsigned long nElemRank = indEndOverlap - indBegOverlap;
992 
993  /* Read the connectivity range determined above. */
994  CGNSElemTypes[iConn].ReadConnectivityRange(fn, iBase, iZone, offsetRank, nElemRank, beg_node[rank], elem,
995  locElemCount, nDOFsLoc);
996  }
997  }
998  }
999 
1000 #ifdef HAVE_MPI
1001  /* The global offset of the DOFs must be corrected when running in
1002  parallel. Therefore gather the number of DOFs of all the ranks. */
1003  vector<unsigned long> nDOFsPerRank(size);
1004  SU2_MPI::Allgather(&nDOFsLoc, 1, MPI_UNSIGNED_LONG, nDOFsPerRank.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
1005 
1006  /* Determine the offset for the DOFs on this rank. */
1007  unsigned long offsetRank = 0;
1008  for (int i = 0; i < rank; ++i) offsetRank += nDOFsPerRank[i];
1009 
1010  /* Loop over the local elements to correct the global offset of the DOFs. */
1011  for (unsigned long i = 0; i < nElem; ++i) elem[i]->AddOffsetGlobalDOFs(offsetRank);
1012 #endif
1013 
1014  /*--------------------------------------------------------------------------*/
1015  /*--- Reading and distributing the coordinates. ---*/
1016  /*--------------------------------------------------------------------------*/
1017 
1018  /* Determine the global number of vertices in the requested zone.
1019  The other size information is not used. */
1020  cgsize_t sizes[3];
1021  if (cg_zone_read(fn, iBase, iZone, cgnsname, sizes) != CG_OK) cg_error_exit();
1022  Global_nPoint = sizes[0];
1023 
1024  /*--- Determine the number of points per rank in cumulative storage format.
1025  This is done to avoid that every rank reads all the coordinates.
1026  The required coordinates for each rank are later obtained via
1027  communication. ---*/
1028  unsigned long totalPointsAccounted = 0;
1029  vector<cgsize_t> nPointsPerRank(size + 1);
1030  for (int i = 1; i <= size; ++i) {
1031  nPointsPerRank[i] = Global_nPoint / size;
1032  totalPointsAccounted += nPointsPerRank[i];
1033  }
1034 
1035  const unsigned long nPointsRem = Global_nPoint - totalPointsAccounted;
1036  for (unsigned long i = 1; i <= nPointsRem; ++i) ++nPointsPerRank[i];
1037 
1038  nPointsPerRank[0] = 0;
1039  for (int i = 0; i < size; ++i) nPointsPerRank[i + 1] += nPointsPerRank[i];
1040 
1041  /* Determine the number of points that must be read by this rank and
1042  allocate the memory for the coordinate buffers. */
1043  const cgsize_t nPointsRead = nPointsPerRank[rank + 1] - nPointsPerRank[rank];
1044  vector<vector<su2double> > coorBuf(nDim, vector<su2double>(nPointsRead));
1045 
1046  /* Loop over the number of dimensions to read the coordinates. Note that
1047  the loop starts at 1 and ends at nDim because CGNS requires this. */
1048  for (unsigned short iDim = 1; iDim <= nDim; ++iDim) {
1049  /* Determine the data type and name of the coordinate. Copy the name
1050  of the coordinate in a string for easier comparison. */
1051  DataType_t datatype;
1052  if (cg_coord_info(fn, iBase, iZone, iDim, &datatype, cgnsname) != CG_OK) cg_error_exit();
1053  string coorname = cgnsname;
1054 
1055  /* Check the name of the coordinate and determine the index in coorBuf
1056  where to store this coordinate. Normally this should be iDim-1. */
1057  unsigned short indC = 0;
1058  if (coorname == "CoordinateX")
1059  indC = 0;
1060  else if (coorname == "CoordinateY")
1061  indC = 1;
1062  else if (coorname == "CoordinateZ")
1063  indC = 2;
1064  else
1065  SU2_MPI::Error(string("Unknown coordinate name, ") + coorname + string(", encountered in the CGNS file."),
1066  CURRENT_FUNCTION);
1067 
1068  /* Easier storage of the range in the CGNS file. */
1069  cgsize_t range_min = nPointsPerRank[rank] + 1;
1070  cgsize_t range_max = nPointsPerRank[rank + 1];
1071 
1072  /*--- Read the coordinate with the required precision and copy
1073  this data to the correct index in coorBuf. ---*/
1074  switch (datatype) {
1075  case RealSingle: {
1076  /* Single precision used. */
1077  vector<float> buf(nPointsRead);
1078  if (cg_coord_read(fn, iBase, iZone, cgnsname, datatype, &range_min, &range_max, buf.data()) != CG_OK)
1079  cg_error_exit();
1080 
1081  for (cgsize_t i = 0; i < nPointsRead; ++i) coorBuf[indC][i] = buf[i];
1082  break;
1083  }
1084 
1085  case RealDouble: {
1086  /* Double precision used. */
1087  vector<double> buf(nPointsRead);
1088  if (cg_coord_read(fn, iBase, iZone, cgnsname, datatype, &range_min, &range_max, buf.data()) != CG_OK)
1089  cg_error_exit();
1090 
1091  for (cgsize_t i = 0; i < nPointsRead; ++i) coorBuf[indC][i] = buf[i];
1092  break;
1093  }
1094 
1095  default: {
1096  ostringstream message;
1097  message << "Datatype for coordinates must be RealSingle or RealDouble, "
1098  << "not " << datatype << endl;
1099  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
1100  }
1101  }
1102  }
1103 
1104  /* Make a distinction between sequential and parallel mode. */
1105 #ifdef HAVE_MPI
1106  /*--- Parallel mode. Create a vector, which contains the global
1107  node IDs of the local elements. ---*/
1108  vector<unsigned long> nodeIDsElemLoc;
1109  nodeIDsElemLoc.reserve(nDOFsLoc);
1110  for (unsigned long i = 0; i < locElemCount; ++i) {
1111  unsigned short nDOFsElem = elem[i]->GetnNodes();
1112  for (unsigned short j = 0; j < nDOFsElem; ++j) nodeIDsElemLoc.push_back(elem[i]->GetNode(j));
1113  }
1114 
1115  sort(nodeIDsElemLoc.begin(), nodeIDsElemLoc.end());
1116  vector<unsigned long>::iterator lastNode;
1117  lastNode = unique(nodeIDsElemLoc.begin(), nodeIDsElemLoc.end());
1118  nodeIDsElemLoc.erase(lastNode, nodeIDsElemLoc.end());
1119 
1120  /*--- Allocate the memory for the coordinates to be stored on this rank. ---*/
1121  nPoint = nodeIDsElemLoc.size();
1122  nodes = new CPoint(nPoint, nDim);
1123 
1124  /*--- Store the global ID's of the nodes in such a way that they can
1125  be sent to the rank that actually stores the coordinates.. ---*/
1126  vector<vector<unsigned long> > nodeBuf(size, vector<unsigned long>(0));
1127  for (unsigned long i = 0; i < nodeIDsElemLoc.size(); ++i) {
1128  const cgsize_t nodeID = nodeIDsElemLoc[i];
1129  vector<cgsize_t>::iterator low;
