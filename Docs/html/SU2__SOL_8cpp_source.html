545 output[iZone]->PreprocessVolumeOutput(config_container[iZone]);
546 output[iZone]->PreprocessHistoryOutput(config_container[iZone], false);
547
548 SolutionInstantiated[iZone] = true;
549 }
550 config_container[iZone]->SetiInst(INST_0);
551 solver_container[iZone][INST_0]->LoadRestart(geometry_container[iZone], &solver_container[iZone],
552 config_container[iZone], TimeIter, true);
553 }
554
555 if (rank == MASTER_NODE) cout << "Writing the volume solution for time step " << TimeIter << "." << endl;
556
557 for (iZone = 0; iZone < nZone; iZone++) {
558 WriteFiles(config_container[iZone], geometry_container[iZone][INST_0], &solver_container[iZone][INST_0],
559 output[iZone], TimeIter);
560 }
561 }
562
563 TimeIter++;
564 if (StopCalc) break;
565 }
566
567 }
568
569 else if (config_container[ZONE_0]->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE) {
570 /*--- Read in the restart file for this time step ---*/
571 for (iZone = 0; iZone < nZone; iZone++) {
572 for (iInst = 0; iInst < nInst[iZone]; iInst++) {
573 config_container[iZone]->SetiInst(iInst);
574 config_container[iZone]->SetTimeIter(iInst);
575
576 /*--- Either instantiate the solution class or load a restart file. ---*/
577 solver_container[iZone][iInst] =
578 new CBaselineSolver(geometry_container[iZone][iInst], config_container[iZone]);
579 solver_container[iZone][iInst]->LoadRestart(geometry_container[iZone], &solver_container[iZone],
580 config_container[iZone], iInst, true);
581 output[iZone] = new CBaselineOutput(config_container[iZone], geometry_container[iZone][iInst]->GetnDim(),
582 solver_container[iZone][iInst]);
583 output[iZone]->PreprocessVolumeOutput(config_container[iZone]);
584 output[iZone]->PreprocessHistoryOutput(config_container[iZone], false);
585
586 /*--- Print progress in solution writing to the screen. ---*/
587 if (rank == MASTER_NODE) {
588 cout << "Storing the volume solution for time instance " << iInst << "." << endl;
589 }
590
591 WriteFiles(config_container[iZone], geometry_container[iZone][iInst], &solver_container[iZone][iInst],
592 output[iZone], iInst);
593 }
594 }
595
596 }
597
598 else if (config_container[ZONE_0]->GetTime_Domain()) {
599 /*--- Dynamic simulation: merge all unsteady time steps. First,
600 find the frequency and total number of files to write. ---*/
601
602 su2double Physical_dt, Physical_t;
603 unsigned long TimeIter = 0;
604 bool StopCalc = false;
605 bool SolutionInstantiated = false;
606
607 /*--- Check for an dynamic restart (structural analysis). Update ExtIter if necessary. ---*/
608 if (config_container[ZONE_0]->GetKind_Solver() == MAIN_SOLVER::FEM_ELASTICITY &&
609 config_container[ZONE_0]->GetRestart())
610 TimeIter = config_container[ZONE_0]->GetRestart_Iter();
611
612 while (TimeIter < config_container[ZONE_0]->GetnTime_Iter()) {
613 /*--- Check several conditions in order to merge the correct time step files. ---*/
614 /*--- If the solver is structural, the total and delta_t are obtained from different functions. ---*/
615
616 Physical_dt = config_container[ZONE_0]->GetTime_Step();
617 Physical_t = (TimeIter + 1) * Physical_dt;
618 if (Physical_t >= config_container[ZONE_0]->GetMax_Time()) StopCalc = true;
619
620 if ((TimeIter + 1 == config_container[ZONE_0]->GetnTime_Iter()) || (StopCalc) ||
621 ((config_container[ZONE_0]->GetTime_Domain()) &&
622 ((TimeIter == 0) || (TimeIter % config_container[ZONE_0]->GetVolumeOutputFrequency(0) == 0)))) {
623 /*--- Set the current iteration number in the config class. ---*/
624 config_container[ZONE_0]->SetTimeIter(TimeIter);
625
626 /*--- Read in the restart file for this time step ---*/
627 for (iZone = 0; iZone < nZone; iZone++) {
628 /*--- Either instantiate the solution class or load a restart file. ---*/
629 if (!SolutionInstantiated &&
630 (TimeIter == 0 ||
631 ((config_container[ZONE_0]->GetRestart() &&
632 (SU2_TYPE::Int(TimeIter) == SU2_TYPE::Int(config_container[ZONE_0]->GetRestart_Iter()))) ||
633 TimeIter % config_container[ZONE_0]->GetVolumeOutputFrequency(0) == 0 ||
634 TimeIter + 1 == config_container[ZONE_0]->GetnTime_Iter()))) {
635 solver_container[iZone][INST_0] =
636 new CBaselineSolver(geometry_container[iZone][INST_0], config_container[iZone]);
637 output[iZone] = new CBaselineOutput(config_container[iZone], geometry_container[iZone][INST_0]->GetnDim(),
638 solver_container[iZone][INST_0]);
639 output[iZone]->PreprocessVolumeOutput(config_container[iZone]);
640 output[iZone]->PreprocessHistoryOutput(config_container[iZone], false);
641
642 SolutionInstantiated = true;
643 }
644 config_container[iZone]->SetiInst(INST_0);
645 solver_container[iZone][INST_0]->LoadRestart(geometry_container[iZone], &solver_container[iZone],
646 config_container[iZone], TimeIter, true);
647 }
648
649 if (rank == MASTER_NODE) cout << "Writing the volume solution for time step " << TimeIter << "." << endl;
650 for (iZone = 0; iZone < nZone; iZone++) {
651 WriteFiles(config_container[iZone], geometry_container[iZone][INST_0], &solver_container[iZone][INST_0],
652 output[iZone], TimeIter);
653 }
654 }
655
656 TimeIter++;
657 if (StopCalc) break;
658 }
659
660 }
661
662 else {
663 /*--- Steady simulation: merge the single solution file. ---*/
664
665 for (iZone = 0; iZone < nZone; iZone++) {
666 config_container[iZone]->SetiInst(INST_0);
667 /*--- Definition of the solution class ---*/
668 solver_container[iZone][INST_0] =
669 new CBaselineSolver(geometry_container[iZone][INST_0], config_container[iZone]);
670 solver_container[iZone][INST_0]->LoadRestart(geometry_container[iZone], &solver_container[iZone],
671 config_container[iZone], 0, true);
672 output[iZone] = new CBaselineOutput(config_container[iZone], geometry_container[iZone][INST_0]->GetnDim(),
673 solver_container[iZone][INST_0]);
674 output[iZone]->PreprocessVolumeOutput(config_container[iZone]);
675 output[iZone]->PreprocessHistoryOutput(config_container[iZone], false);
676 }
677 for (iZone = 0; iZone < nZone; iZone++) {
678 WriteFiles(config_container[iZone], geometry_container[iZone][INST_0], &solver_container[iZone][INST_0],
679 output[iZone], 0);
680 }
681 }
682 }
683
684 delete config;
685 config = nullptr;
686
687 if (rank == MASTER_NODE)
688 cout << endl << "------------------------- Finalize Solver -------------------------" << endl;
689
690 if (geometry_container != nullptr) {
691 for (iZone = 0; iZone < nZone; iZone++) {
692 for (iInst = 0; iInst < nInst[iZone]; iInst++) {
693 if (geometry_container[iZone][iInst] != nullptr) {
694 delete geometry_container[iZone][iInst];
695 }
696 }
697 if (geometry_container[iZone] != nullptr) delete[] geometry_container[iZone];
698 }
699 delete[] geometry_container;
700 }
701 if (rank == MASTER_NODE) cout << "Deleted CGeometry container." << endl;
702
703 if (solver_container != nullptr) {
704 for (iZone = 0; iZone < nZone; iZone++) {
705 for (iInst = 0; iInst < nInst[iZone]; iInst++) {
706 if (solver_container[iZone][iInst] != nullptr) {
707 delete solver_container[iZone][iInst];
708 }
709 }
710 if (solver_container[iZone] != nullptr) delete[] solver_container[iZone];
711 }
712 delete[] solver_container;
713 }
714 if (rank == MASTER_NODE) cout << "Deleted CSolver class." << endl;
715
716 if (config_container != nullptr) {
717 for (iZone = 0; iZone < nZone; iZone++) {
718 if (config_container[iZone] != nullptr) {
719 delete config_container[iZone];
720 }
721 }
722 delete[] config_container;
723 }
724 if (rank == MASTER_NODE) cout << "Deleted CConfig container." << endl;
725
726 if (output != nullptr) {
727 for (iZone = 0; iZone < nZone; iZone++) {
728 if (output[iZone] != nullptr) {
729 delete output[iZone];
730 }
731 }
732 delete[] output;
733 }
734 if (rank == MASTER_NODE) cout << "Deleted COutput class." << endl;
735
736 /*--- Synchronization point after a single solver iteration. Compute the
737 wall clock time required. ---*/
738
739 StopTime = SU2_MPI::Wtime();
740
741 /*--- Compute/print the total time for performance benchmarking. ---*/
742
743 UsedTime = StopTime - StartTime;
744 if (rank == MASTER_NODE) {
745 cout << "\nCompleted in " << fixed << UsedTime << " seconds on " << size;
746 if (size == 1)
747 cout << " core." << endl;
748 else
749 cout << " cores." << endl;
750 }
751
752 /*--- Exit the solver cleanly ---*/
753
754 if (rank == MASTER_NODE)
755 cout << endl << "------------------------- Exit Success (SU2_SOL) ------------------------" << endl << endl;
756
757 /*--- Finalize MPI parallelization ---*/
758 SU2_MPI::Finalize();
759
760 return EXIT_SUCCESS;
761}
