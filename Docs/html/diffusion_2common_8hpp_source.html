35/*!
36 * \brief Average gradients at i/j points.
37 */
38template<size_t nVar, size_t nDim, class GradientType>
39FORCEINLINE MatrixDbl<nVar,nDim> averageGradient(Int iPoint, Int jPoint,
40 const GradientType& gradient) {
41 auto avgGrad = gatherVariables<nVar,nDim>(iPoint, gradient);
42 auto grad_j = gatherVariables<nVar,nDim>(jPoint, gradient);
43 for (size_t iVar = 0; iVar < nVar; ++iVar) {
44 for (size_t iDim = 0; iDim < nDim; ++iDim) {
45 avgGrad(iVar,iDim) *= 0.5;
46 avgGrad(iVar,iDim) += 0.5 * grad_j(iVar,iDim);
47 }
48 }
49 return avgGrad;
50}

52/*!
53 * \brief Correct average gradient with the directional derivative to avoid decoupling.
54 */
55template<size_t nVar, size_t nDim, class PrimitiveType>
56FORCEINLINE void correctGradient(const PrimitiveType& V,
57 const VectorDbl<nDim>& vector_ij,
58 Double dist2_ij,
59 MatrixDbl<nVar,nDim>& avgGrad) {
60 for (size_t iVar = 0; iVar < nVar; ++iVar) {
61 Double corr = (dot(avgGrad[iVar],vector_ij) - V.j.all(iVar) + V.i.all(iVar)) / dist2_ij;
62 for (size_t iDim = 0; iDim < nDim; ++iDim) {
63 avgGrad(iVar,iDim) -= corr * vector_ij(iDim);
64 }
65 }
66}

68/*!
69 * \brief Compute the stress tensor.
70 * \note Second viscosity term ignored.
71 */
72template<size_t nVar, size_t nDim>
73FORCEINLINE MatrixDbl<nDim> stressTensor(Double viscosity,
74 const MatrixDbl<nVar,nDim>& grad) {
75 /*--- Hydrostatic term. ---*/
76 Double velDiv = 0.0;
77 for (size_t iDim = 0; iDim < nDim; ++iDim) {
78 velDiv += grad(iDim+1,iDim);
79 }
80 Double pTerm = 2.0/3.0 * viscosity * velDiv;
81
82 MatrixDbl<nDim> tau;
83 for (size_t iDim = 0; iDim < nDim; ++iDim) {
84 /*--- Deviatoric term. ---*/
85 for (size_t jDim = 0; jDim < nDim; ++jDim) {
86 tau(iDim,jDim) = viscosity * (grad(jDim+1,iDim) + grad(iDim+1,jDim));
87 }
88 tau(iDim,iDim) -= pTerm;
89 }
90 return tau;
91}

93/*!
94 * \brief Add perturbed stress tensor.
95 * \note Not inlined because it is not easy to vectorize properly, due to tred2 and tql2.
96 */
97template<class PrimitiveType, class MatrixType, size_t nDim, class... Ts>
98NEVERINLINE void addPerturbedRSM(const PrimitiveType& V,
99 const MatrixType& grad,
100 const Double& turb_ke,
101 MatrixDbl<nDim,nDim>& tau,
102 Ts... uq_args) {
103 /*--- Handle SIMD dimensions 1 by 1. ---*/
104 for (size_t k = 0; k < Double::Size; ++k) {
105 su2double velgrad[nDim][nDim];
106 for (size_t iVar = 0; iVar < nDim; ++iVar)
107 for (size_t iDim = 0; iDim < nDim; ++iDim)
108 velgrad[iVar][iDim] = grad(iVar+1,iDim)[k];
109
110 su2double rsm[3][3];
111 CNumerics::ComputePerturbedRSM(nDim, uq_args..., velgrad, V.density()[k],
112 V.eddyVisc()[k], turb_ke[k], rsm);
113
114 for (size_t iDim = 0; iDim < nDim; ++iDim)
115 for (size_t jDim = 0; jDim < nDim; ++jDim)
116 tau(iDim,jDim)[k] -= V.density()[k] * rsm[iDim][jDim];
117 }
118}

120/*!
121 * \brief SA-QCR2000 modification of the stress tensor.
122 */
123template<class MatrixType, size_t nDim>
124FORCEINLINE void addQCR(const MatrixType& grad, MatrixDbl<nDim>& tau) {
125 constexpr passivedouble c_cr1 = 0.3;
126
127 /*--- Denominator, antisymmetric normalized rotation tensor. ---*/
128 Double denom = 0.0;
129 for (size_t iDim = 0; iDim < nDim; ++iDim)
130 for (size_t jDim = 0; jDim < nDim; ++jDim)
131 denom += grad(iDim+1,jDim) * grad(iDim+1,jDim);
132
133 const Double factor = 1 / sqrt(fmax(denom,1e-10));
134
135 /*--- Compute the QCR term, and update the stress tensor. ---*/
136 MatrixDbl<nDim> qcr;
137 for (size_t iDim = 0; iDim < nDim; ++iDim) {
138 for (size_t jDim = 0; jDim < nDim; ++jDim) {
139 qcr(iDim,jDim) = 0.0;
140 for (size_t kDim = 0; kDim < nDim; ++kDim) {
141 Double O_ik = (grad(iDim+1,kDim) - grad(kDim+1,iDim)) * factor;
142 Double O_jk = (grad(jDim+1,kDim) - grad(kDim+1,jDim)) * factor;
143 qcr(iDim,jDim) += O_ik*tau(jDim,kDim) + O_jk*tau(iDim,kDim);
144 }
145 }
146 }
147 for (size_t iDim = 0; iDim < nDim; ++iDim)
148 for (size_t jDim = 0; jDim < nDim; ++jDim)
149 tau(iDim,jDim) -= c_cr1 * qcr(iDim,jDim);
150}

152/*!
153 * \brief Scale the stress tensor according to the target (from a
154 * wall function) magnitude in the tangential direction.
155 */
156template<class Container, size_t nDim>
157FORCEINLINE void addTauWall(Int iPoint, Int jPoint,
158 const Container& tauWall,
159 const VectorDbl<nDim>& unitNormal,
160 MatrixDbl<nDim>& tau) {
161
162 Double tauWall_i = fmax(gatherVariables(iPoint, tauWall), 0.0);
163 Double tauWall_j = fmax(gatherVariables(jPoint, tauWall), 0.0);
164
165 Double isWall_i = tauWall_i > 0.0;
166 Double isWall_j = tauWall_j > 0.0;
167 /*--- Arithmetic xor. ---*/
168 Double isNormalEdge = isWall_i+isWall_j - 2*isWall_i*isWall_j;
169
170 /*--- Tau wall is 0 for edges that are not normal to walls. ---*/
171 Double tauWall_ij = (tauWall_i+tauWall_j) * isNormalEdge;
172
173 /*--- Scale is 1 for those edges, i.e. tau is not changed. ---*/
174 Double scale =
175 tauWall_ij / fmax(norm(tangentProjection(tau,unitNormal)), EPS) + (1.0-isNormalEdge);
176
177 for (size_t iDim = 0; iDim < nDim; ++iDim)
178 for (size_t jDim = 0; jDim < nDim; ++jDim)
179 tau(iDim,jDim) *= scale;
180}

182/*!
183 * \brief Jacobian of the stress tensor (compressible flow).
184 */
185template<size_t nVar, size_t nDim, class PrimitiveType>
186FORCEINLINE MatrixDbl<nDim,nVar> stressTensorJacobian(const PrimitiveType& V,
187 const VectorDbl<nDim>& normal,
188 Double dist_ij) {
189 Double viscosity = V.laminarVisc() + V.eddyVisc();
190 Double xi = viscosity / (V.density() * dist_ij);
191 MatrixDbl<nDim,nVar> jac;
192 for (size_t iDim = 0; iDim < nDim; ++iDim) {
193 /*--- Momentum. ---*/
194 for (size_t jDim = 0; jDim < nDim; ++jDim) {
195 jac(iDim,jDim+1) = (-1/3.0) * xi * normal(iDim) * normal(jDim);
196 }
197 jac(iDim,iDim+1) -= xi;
198 /*--- Density. ---*/
199 jac(iDim,0) = -dot<nDim>(&jac(iDim,1), V.velocity());
200 /*--- Energy. ---*/
201 jac(iDim,nDim+1) = 0.0;
202 }
203 return jac;
204}

206/*!
207 * \brief Viscous flux for compressible flows.
208 */
209template<size_t nVar, size_t nDim, class PrimitiveType>
210FORCEINLINE VectorDbl<nVar> viscousFlux(const PrimitiveType& V,
211 const MatrixDbl<nDim>& tau,
212 const VectorDbl<nDim>& heatFlux,
213 const VectorDbl<nDim>& normal) {
214 VectorDbl<nVar> flux;
215 flux(0) = 0.0;
216 for (size_t iDim = 0; iDim < nDim; ++iDim) {
217 /*--- Using the symmetry of the tensor. ---*/
218 flux(iDim+1) = dot(tau[iDim], normal);
219 }
220 flux(nDim+1) = 0.0;
221 for (size_t iDim = 0; iDim < nDim; ++iDim) {
222 auto viscWork = dot<nDim>(tau[iDim], V.velocity());
223 flux(nDim+1) += normal(iDim) * (heatFlux(iDim) + viscWork);
224 }
225 return flux;
226}
