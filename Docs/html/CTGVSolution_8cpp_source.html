32 CTGVSolution::CTGVSolution(unsigned short val_nDim, unsigned short val_nVar, unsigned short val_iMesh, CConfig* config)
33  : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
34  /*--- Write a message that the solution is initialized for the
35  Taylor-Green vortex test case. ---*/
36 
37  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
38  cout << endl;
39  cout << "Warning: Fluid properties and solution are being " << endl;
40  cout << " initialized for the Taylor-Green vortex case!!!" << endl;
41  cout << endl << flush;
42  }
43 
44  /*--- Store TGV specific parameters here. ---*/
45 
46  tgvLength = 1.0; // Taylor-Green length scale.
47  tgvVelocity = 1.0; // Taylor-Green velocity.
48  tgvDensity = 1.0; // Taylor-Green density.
49  tgvPressure = 100.0; // Taylor-Green pressure.
50 
51  /*--- Useful coefficient in which Gamma is present. ---*/
52 
53  ovGm1 = 1.0 / (config->GetGamma() - 1.0);
54 
55  /*--- Perform some sanity and error checks for this solution here. ---*/
56 
57  if ((config->GetTime_Marching() != TIME_MARCHING::TIME_STEPPING) &&
58  (config->GetTime_Marching() != TIME_MARCHING::DT_STEPPING_1ST) &&
59  (config->GetTime_Marching() != TIME_MARCHING::DT_STEPPING_2ND))
60  SU2_MPI::Error("Unsteady mode must be selected for the Taylor Green Vortex", CURRENT_FUNCTION);
61 
62  if (Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES &&
63  Kind_Solver != MAIN_SOLVER::RANS && Kind_Solver != MAIN_SOLVER::FEM_EULER &&
64  Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
65  Kind_Solver != MAIN_SOLVER::FEM_LES)
66  SU2_MPI::Error("Compressible flow equations must be selected for the Taylor Green Vortex", CURRENT_FUNCTION);
67 
68  if ((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) && (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
69  SU2_MPI::Error("Navier Stokes equations must be selected for the Taylor Green Vortex", CURRENT_FUNCTION);
70 
71  if ((config->GetKind_FluidModel() != STANDARD_AIR) && (config->GetKind_FluidModel() != IDEAL_GAS))
72  SU2_MPI::Error("Standard air or ideal gas must be selected for the Taylor Green Vortex", CURRENT_FUNCTION);
73 
74  if (config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
75  SU2_MPI::Error("Constant viscosity must be selected for the Taylor Green Vortex", CURRENT_FUNCTION);
76 
77  if (config->GetKind_ConductivityModel() != CONDUCTIVITYMODEL::CONSTANT_PRANDTL)
78  SU2_MPI::Error("Constant Prandtl number must be selected for the Taylor Green Vortex", CURRENT_FUNCTION);
79 }

83 void CTGVSolution::GetSolution(const su2double* val_coords, const su2double val_t, su2double* val_solution) const {
84  /* The initial conditions are set for the Taylor-Green vortex case, which
85  is a DNS case that features vortex breakdown into turbulence. These
86  particular settings are for the typical Re = 1600 case (M = 0.08) with
87  an initial temperature of 300 K. Note that this condition works in both
88  2D and 3D. */
89 
90  su2double val_coordsZ = 0.0;
91  if (nDim == 3) val_coordsZ = val_coords[2];
92 
93  /* Compute the primitive variables. */
94 
95  su2double rho = tgvDensity;
96  su2double u =
97  tgvVelocity * (sin(val_coords[0] / tgvLength) * cos(val_coords[1] / tgvLength) * cos(val_coordsZ / tgvLength));
98  su2double v =
99  -tgvVelocity * (cos(val_coords[0] / tgvLength) * sin(val_coords[1] / tgvLength) * cos(val_coordsZ / tgvLength));
100 
101  su2double factorA = cos(2.0 * val_coordsZ / tgvLength) + 2.0;
102  su2double factorB = (cos(2.0 * val_coords[0] / tgvLength) + cos(2.0 * val_coords[1] / tgvLength));
103 
104  su2double p = (tgvPressure + tgvDensity * (pow(tgvVelocity, 2.0) / 16.0) * factorA * factorB);
105 
106  /* Compute the conservative variables. Note that both 2D and 3D
107  cases are treated correctly. */
108 
109  val_solution[0] = rho;
110  val_solution[1] = rho * u;
111  val_solution[2] = rho * v;
112  val_solution[3] = 0.0;
113  val_solution[nVar - 1] = p * ovGm1 + 0.5 * rho * (u * u + v * v);
114 }
