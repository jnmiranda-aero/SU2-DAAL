36/*!
37 * \brief Class to represent a matrix (without owning the data, this just wraps a pointer).
38 */
39template <class T>
40class CMatrixView {
41 public:
42 using Scalar = typename std::remove_const<T>::type;
43 using Index = unsigned long;
44
45 private:
46 T* m_ptr;
47 Index m_cols;
48
49 public:
50 CMatrixView(T* ptr = nullptr, Index cols = 0) : m_ptr(ptr), m_cols(cols) {}
51
52 template <class U>
53 friend class CMatrixView;
54 template <class U>
55 CMatrixView(const CMatrixView<U>& other) : m_ptr(other.m_ptr), m_cols(other.m_cols) {}
56
57 explicit CMatrixView(su2matrix<Scalar>& mat) : m_ptr(mat.data()), m_cols(mat.cols()) {}
58
59 template <class U = T, su2enable_if<std::is_const<U>::value> = 0>
60 explicit CMatrixView(const su2matrix<Scalar>& mat) : m_ptr(mat.data()), m_cols(mat.cols()) {}
61
62 const Scalar* operator[](Index i) const noexcept { return &m_ptr[i * m_cols]; }
63 const Scalar& operator()(Index i, Index j) const noexcept { return m_ptr[i * m_cols + j]; }
64
65 template <class U = T, su2enable_if<!std::is_const<U>::value> = 0>
66 Scalar* operator[](Index i) noexcept {
67 return &m_ptr[i * m_cols];
68 }
69
70 template <class U = T, su2enable_if<!std::is_const<U>::value> = 0>
71 Scalar& operator()(Index i, Index j) noexcept {
72 return m_ptr[i * m_cols + j];
73 }
74
75 friend CMatrixView operator+(CMatrixView mv, Index incr) { return CMatrixView(mv[incr], mv.m_cols); }
76};

78/*!
79 * \class C3DContainerDecorator
80 * \brief Decorate a matrix type (Storage) with 3 dimensions.
81 */
82template <class Storage>
83class C3DContainerDecorator {
84 static_assert(!Storage::IsVector, "Storage type must be a matrix.");
85 static_assert(Storage::IsRowMajor, "Storage type must be row major.");
86
87 public:
88 using Scalar = typename Storage::Scalar;
89 using Index = typename Storage::Index;
90 static constexpr bool IsRowMajor = true;
91 static constexpr bool IsColumnMajor = false;
92
93 using Matrix = CMatrixView<Scalar>;
94 using ConstMatrix = CMatrixView<const Scalar>;
95
96 using CInnerIter = typename Storage::CInnerIter;
97 template <class T, size_t N>
98 using CInnerIterGather = typename Storage::template CInnerIterGather<simd::Array<T, N> >;
99
100 private:
101 Storage m_storage;
102 Index m_innerSz;
103
104 public:
105 C3DContainerDecorator() = default;
106
107 C3DContainerDecorator(Index length, Index rows, Index cols, Scalar value = 0) noexcept {
108 resize(length, rows, cols, value);
109 }
110
111 void resize(Index length, Index rows, Index cols, Scalar value = 0) noexcept {
112 m_innerSz = cols;
113 m_storage.resize(length, rows * cols) = value;
114 }
115
116 /*!
117 * \brief Container sizes.
118 */
119 Index size() const noexcept { return m_storage.size(); }
120 Index length() const noexcept { return m_storage.rows(); }
121 Index rows() const noexcept { return m_storage.cols() / m_innerSz; }
122 Index cols() const noexcept { return m_innerSz; }
123
124 /*!
125 * \brief Element-wise access.
126 */
127 Scalar& operator()(Index i, Index j, Index k) noexcept { return m_storage(i, j * m_innerSz + k); }
128 const Scalar& operator()(Index i, Index j, Index k) const noexcept { return m_storage(i, j * m_innerSz + k); }
129
130 /*!
131 * \brief Matrix access.
132 */
133 Matrix operator[](Index i) noexcept { return Matrix(m_storage[i], m_innerSz); }
134 ConstMatrix operator[](Index i) const noexcept { return ConstMatrix(m_storage[i], m_innerSz); }
135
136 /*!
137 * \brief Matrix access with an offset.
138 */
139 Matrix operator()(Index i, Index j) noexcept { return Matrix(m_storage[i] + j * m_innerSz, m_innerSz); }
140 ConstMatrix operator()(Index i, Index j) const noexcept {
141 return ConstMatrix(m_storage[i] + j * m_innerSz, m_innerSz);
142 }
143
144 /*!
145 * \brief Get a scalar iterator to the inner-most dimension of the container.
146 */
147 FORCEINLINE CInnerIter innerIter(Index i, Index j) const noexcept {
148 return CInnerIter(&m_storage(i, j * m_innerSz), 1);
149 }
150
151 /*!
152 * \brief Get a SIMD gather iterator to the inner-most dimension of the container.
153 */
154 template <class T, size_t N>
155 FORCEINLINE CInnerIterGather<T, N> innerIter(simd::Array<T, N> i, Index j) const noexcept {
156 return CInnerIterGather<T, N>(m_storage.data(), 1, i * m_storage.cols() + j * m_innerSz);
157 }
158
159 /*!
160 * \brief Return copy of data in a static size container (see C2DContainer::get).
161 * \param[in] i - Outer index.
162 * \param[in] j - Starting middle index for the copy (amount determined by container size).
163 */
164 template <class StaticContainer, class Int>
165 FORCEINLINE StaticContainer get(Int i, Index j = 0) const noexcept {
166 return m_storage.template get<StaticContainer>(i, j * m_innerSz);
167 }
168};

177/*!

179 * \brief Helper class, adds dummy trailing dimension to a reference of a
180 * vector object making it a dummy matrix.
181 * \note The constness of the object is derived from the template type, but
182 * we allways keep a reference, never a copy of the associated vector.
183 */
184template <class T>
185struct C2DDummyLastView {
186 static_assert(T::IsVector, "This class decorates vectors.");
187 using Index = typename T::Index;
188 using Scalar = typename T::Scalar;
189
190 T& data;
191
192 C2DDummyLastView() = delete;
193
194 C2DDummyLastView(T& ref) : data(ref) {}

197 Scalar& operator()(Index i, Index) noexcept {

200

233/*!
234 * \brief Allocate a vector of varying-size vectors and initialize with some value.
235 * \param[in] M - the first index is >=0 and < M
236 * \param[in] N - the second index is >=0 and < N[first index]
237 * \param[in,out] X - the vector of vectors
238 * \param[in] val - the value for initialization, default is 0
239 * \tparam IndexVector - type of N
240 * \tparam VectorOfVector - type of X
241 */
242template <class IndexVector, class VectorOfVector, class Scalar = int>
243inline void AllocVectorOfVectors(size_t M, const IndexVector& N, VectorOfVector& X, Scalar val = 0) {
244 X.resize(M);
245 for (size_t i = 0; i < M; ++i) {
246 X[i].resize(N[i]);
247 for (auto& x : X[i]) x = val;
248 }
249}

251/*!
252 * \overload Deduce outer size from index vector.
253 */
254template <class IndexVector, class VectorOfVector, class Scalar = int>
255inline void AllocVectorOfVectors(const IndexVector& N, VectorOfVector& X, Scalar val = 0) {
256 auto M = N.size();
257 AllocVectorOfVectors(M, N, X, val);
258}

260/*!
261 * \brief Allocate a vector of matrices with varying row count, and initialize with some value.
262 * \param[in] M - the first index is >=0 and < M
263 * \param[in] N - the second index is >=0 and < N[first index]
264 * \param[in] P - the third index is >=0 and < P
265 * \param[in,out] X - the vector of matrices
266 * \param[in] val - the value for initialization, default is 0
267 * \tparam IndexVector - type of N
268 * \tparam VectorOfMatrix - type of X
269 */
270template <class IndexVector, class VectorOfMatrix, class Scalar = int>
271inline void AllocVectorOfMatrices(size_t M, const IndexVector& N, size_t P, VectorOfMatrix& X, Scalar val = 0) {
272 X.resize(M);
273 for (size_t i = 0; i < M; ++i) {
274 X[i].resize(N[i], P);
275 for (auto& x : X[i]) x = val;
276 }
277}
