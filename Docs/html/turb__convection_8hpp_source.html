37 * \author A. Bueno.

39template <class FlowIndices>
40class CUpwSca_TurbSA final : public CUpwScalar<FlowIndices> {
41private:
42 using Base = CUpwScalar<FlowIndices>;
43 using Base::a0;
44 using Base::a1;
45 using Base::Flux;
46 using Base::Jacobian_i;
47 using Base::Jacobian_j;
48 using Base::ScalarVar_i;
49 using Base::ScalarVar_j;
50 using Base::bounded_scalar;
51
52 /*!
53 * \brief Adds any extra variables to AD.
54 */
55 void ExtraADPreaccIn() override {}
56
57 /*!
58 * \brief SA specific steps in the ComputeResidual method
59 * \param[in] config - Definition of the particular problem.
60 */
61 void FinishResidualCalc(const CConfig* config) override {
62 Flux[0] = a0*ScalarVar_i[0] + a1*ScalarVar_j[0];
63 Jacobian_i[0][0] = a0;
64 Jacobian_j[0][0] = a1;
65 }
66
67public:
68 /*!
69 * \brief Constructor of the class.

74 CUpwSca_TurbSA(unsigned short val_nDim, unsigned short val_nVar, const CConfig* config)
75 : CUpwScalar<FlowIndices>(val_nDim, val_nVar, config) { bounded_scalar = config->GetBounded_Turb(); }
76};

78/*!

80 * \brief Class for doing a scalar upwind solver for the Menter SST turbulence model equations.
81 * \ingroup ConvDiscr
82 * \author A. Campos.

84template <class FlowIndices>
85class CUpwSca_TurbSST final : public CUpwScalar<FlowIndices> {
86private:
87 using Base = CUpwScalar<FlowIndices>;
88 using Base::nDim;
89 using Base::V_i;
90 using Base::V_j;
91 using Base::a0;
92 using Base::a1;
93 using Base::Flux;
94 using Base::Jacobian_i;
95 using Base::Jacobian_j;
96 using Base::ScalarVar_i;
97 using Base::ScalarVar_j;
98 using Base::idx;
99 using Base::bounded_scalar;
100
101 /*!
102 * \brief Adds any extra variables to AD
103 */
104 void ExtraADPreaccIn() override {}
105
106 /*!
107 * \brief SST specific steps in the ComputeResidual method
108 * \param[in] config - Definition of the particular problem.
109 */
110 void FinishResidualCalc(const CConfig* config) override {
111 Flux[0] = a0*V_i[idx.Density()]*ScalarVar_i[0] + a1*V_j[idx.Density()]*ScalarVar_j[0];
112 Flux[1] = a0*V_i[idx.Density()]*ScalarVar_i[1] + a1*V_j[idx.Density()]*ScalarVar_j[1];
113
114 Jacobian_i[0][0] = a0; Jacobian_i[0][1] = 0.0;
115 Jacobian_i[1][0] = 0.0; Jacobian_i[1][1] = a0;
116
117 Jacobian_j[0][0] = a1; Jacobian_j[0][1] = 0.0;
118 Jacobian_j[1][0] = 0.0; Jacobian_j[1][1] = a1;
119 }
120
121public:
122 /*!
123 * \brief Constructor of the class.
124 * \param[in] val_nDim - Number of dimensions of the problem.

130};
