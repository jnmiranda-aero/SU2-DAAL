108  * \param[in] config - Definition of the particular problem.
109  */
110  void FinishResidualCalc(const CConfig* config) override {
111  Flux[0] = a0*V_i[idx.Density()]*ScalarVar_i[0] + a1*V_j[idx.Density()]*ScalarVar_j[0];
112  Flux[1] = a0*V_i[idx.Density()]*ScalarVar_i[1] + a1*V_j[idx.Density()]*ScalarVar_j[1];
113 
114  Jacobian_i[0][0] = a0; Jacobian_i[0][1] = 0.0;
115  Jacobian_i[1][0] = 0.0; Jacobian_i[1][1] = a0;
116 
117  Jacobian_j[0][0] = a1; Jacobian_j[0][1] = 0.0;
118  Jacobian_j[1][0] = 0.0; Jacobian_j[1][1] = a1;
