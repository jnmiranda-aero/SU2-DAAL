36CNEMOEulerSolver::CNEMOEulerSolver(CGeometry *geometry, CConfig *config,
37 unsigned short iMesh, const bool navier_stokes) :
38 CFVMFlowSolverBase<CNEMOEulerVariable, ENUM_REGIME::COMPRESSIBLE>(*geometry, *config) {
39
40 /*--- Based on the navier_stokes boolean, determine if this constructor is
41 * being called by itself, or by its derived class CNEMONSSolver. ---*/
42 string description;
43 if (navier_stokes) {
44 description = "Navier-Stokes";
45 }
46 else {
47 description = "Euler";
48 }
49
50 const auto nZone = geometry->GetnZone();
51 const bool restart = (config->GetRestart() || config->GetRestart_Flow());
52 const auto direct_diff = config->GetDirectDiff();
53 const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
54 (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
55 const bool time_stepping = (config->GetTime_Marching() == TIME_MARCHING::TIME_STEPPING);
56 const bool adjoint = config->GetContinuous_Adjoint() || config->GetDiscrete_Adjoint();
57
58 int Unst_RestartIter = 0;
59
60 /*--- A grid is defined as dynamic if there's rigid grid movement or grid deformation AND the problem is time domain ---*/
61 dynamic_grid = config->GetDynamic_Grid();
62
63 /*--- Store the multigrid level. ---*/
64 MGLevel = iMesh;
65
66 /*--- Check for a restart file to evaluate if there is a change in the AoA
67 before non-dimensionalizing ---*/
68 if (restart && (iMesh == MESH_0) && nZone <=1 ) {
69
70 /*--- Modify file name for a dual-time unsteady restart ---*/
71 if (dual_time) {
72 if (adjoint) Unst_RestartIter = SU2_TYPE::Int(config->GetUnst_AdjointIter())-1;
73 else if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST)
74 Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-1;
75 else Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-2;
76 }
77
78 /*--- Modify file name for a time stepping unsteady restart ---*/
79 if (time_stepping) {
80 if (adjoint) Unst_RestartIter = SU2_TYPE::Int(config->GetUnst_AdjointIter())-1;
81 else Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-1;
82 }
83
84 string filename_ = "flow";
85 filename_ = config->GetFilename(filename_, ".meta", Unst_RestartIter);
86
87 /*--- Read and store the restart metadata ---*/
88 Read_SU2_Restart_Metadata(geometry, config, false, filename_);
89
90 }
91
92 /*--- Define geometric constants in the solver structure ---*/
93 nSpecies = config->GetnSpecies();
94 nMarker = config->GetnMarker_All();
95 nDim = geometry->GetnDim();
96 nPoint = geometry->GetnPoint();
97 nPointDomain = geometry->GetnPointDomain();
98
99 /*--- Set size of the conserved and primitive vectors ---*/
100 // U: [rho1, ..., rhoNs, rhou, rhov, rhow, rhoe, rhoeve]^T
101 // V: [rho1, ..., rhoNs, T, Tve, u, v, w, P, rho, h, a, rhoCvtr, rhoCvve]^T
102 // GradV: [rho1, ..., rhoNs, T, Tve, u, v, w, P, rho, h, a, rhoCvtr, rhoCvve]^T
103 // Viscous: append [mu, mu_t]^T
104 nVar = nSpecies + nDim + 2;
105 if (navier_stokes) { nPrimVar = nSpecies + nDim + 10; }
106 else { nPrimVar = nSpecies +nDim +8; }
107 nPrimVarGrad = nSpecies + nDim + 8;
108
109 /*--- Initialize nVarGrad for deallocation ---*/
110 nVarGrad = nPrimVarGrad;
111
112 /*--- Store the number of vertices on each marker for deallocation ---*/
113 nVertex.resize(nMarker);
114 for (auto iMarker = 0ul; iMarker < nMarker; iMarker++)
115 nVertex[iMarker] = geometry->nVertex[iMarker];
116
117 MassFrac_Inf = config->GetGas_Composition();
118
119 /*--- Perform the non-dimensionalization for the flow equations using the
120 specified reference values. ---*/
121 SetNondimensionalization(config, iMesh);
122
123 /// TODO: This type of variables will be replaced.
124
125 AllocateTerribleLegacyTemporaryVariables();
126
127 /*--- Allocate base class members. ---*/
128
129 Allocate(*config);
130
131 /*--- Allocate Jacobians for implicit time-stepping ---*/
132 if (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT) {
133
134 /*--- Jacobians and vector structures for implicit computations ---*/
135 if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (" << description << "). MG level: " << iMesh <<"." << endl;
136 Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config);
137 }
138 else {
139 if (rank == MASTER_NODE) cout<< "Explicit Scheme. No Jacobian structure (" << description << "). MG level: " << iMesh <<"."<<endl;
140 }
141
142 /*--- Read farfield conditions from the config file ---*/
143 Mach_Inf = config->GetMach();
144 Density_Inf = config->GetDensity_FreeStreamND();
145 Pressure_Inf = config->GetPressure_FreeStreamND();
146 Velocity_Inf = config->GetVelocity_FreeStreamND();
147 Temperature_Inf = config->GetTemperature_FreeStreamND();
148 Temperature_ve_Inf = config->GetTemperature_ve_FreeStreamND();
149
150 /*--- Initialize the secondary values for direct derivative approxiations ---*/
151 switch(direct_diff) {
152 case NO_DERIVATIVE:
153 /*--- Default ---*/
154 break;
155 case D_DENSITY:
156 SU2_TYPE::SetDerivative(Density_Inf, 1.0);
157 break;
158 case D_PRESSURE:
159 SU2_TYPE::SetDerivative(Pressure_Inf, 1.0);
160 break;
161 case D_TEMPERATURE:
162 SU2_TYPE::SetDerivative(Temperature_Inf, 1.0);
163 break;
164 case D_MACH: case D_AOA:
165 case D_SIDESLIP: case D_REYNOLDS:
166 case D_TURB2LAM: case D_DESIGN:
167 /*--- Already done in postprocessing of config ---*/
168 break;
169 default:
170 break;
171 }
172
173 SetReferenceValues(*config);
174
175 /*--- Vectorize free stream Mach number based on AoA & AoS ---*/
176 su2double *Mvec_Inf;
177 Mvec_Inf = new su2double[nDim];
178 const su2double Alpha = config->GetAoA()*PI_NUMBER/180.0;
179 const su2double Beta = config->GetAoS()*PI_NUMBER/180.0;
180 if (nDim == 2) {
181 Mvec_Inf[0] = cos(Alpha)*Mach_Inf;
182 Mvec_Inf[1] = sin(Alpha)*Mach_Inf;
183 }
184 if (nDim == 3) {
185 Mvec_Inf[0] = cos(Alpha)*cos(Beta)*Mach_Inf;
186 Mvec_Inf[1] = sin(Beta)*Mach_Inf;
187 Mvec_Inf[2] = sin(Alpha)*cos(Beta)*Mach_Inf;
188 }
189
190 /*--- Initialize the solution to the far-field state everywhere. ---*/
191
192 if (navier_stokes) {
193 nodes = new CNEMONSVariable (Pressure_Inf, MassFrac_Inf, Mvec_Inf,
194 Temperature_Inf, Temperature_ve_Inf,
195 nPoint, nDim, nVar, nPrimVar, nPrimVarGrad,
196 config, FluidModel);
197 node_infty = new CNEMONSVariable (Pressure_Inf, MassFrac_Inf, Mvec_Inf,
198 Temperature_Inf, Temperature_ve_Inf,
199 1, nDim, nVar, nPrimVar, nPrimVarGrad,
200 config, FluidModel);
201 } else {
202 nodes = new CNEMOEulerVariable(Pressure_Inf, MassFrac_Inf, Mvec_Inf,
203 Temperature_Inf, Temperature_ve_Inf,
204 nPoint, nDim, nVar, nPrimVar, nPrimVarGrad,
205 config, FluidModel);
206 node_infty = new CNEMOEulerVariable(Pressure_Inf, MassFrac_Inf, Mvec_Inf,
207 Temperature_Inf, Temperature_ve_Inf,
208 1, nDim, nVar, nPrimVar, nPrimVarGrad,
209 config, FluidModel);
210 }
211 SetBaseClassPointerToNodes();
212
213 node_infty->SetPrimVar(0, FluidModel);
214
215 /*--- Initial comms. ---*/
216
217 CommunicateInitialState(geometry, config);
218
219 /*--- Add the solver name. ---*/
220 SolverName = "NEMO.FLOW";
221
222 /*--- Finally, check that the static arrays will be large enough (keep this
223 * check at the bottom to make sure we consider the "final" values). ---*/
224 if((nDim > MAXNDIM) || (nPrimVar > MAXNVAR))
225 SU2_MPI::Error("Oops! The CNEMOEulerSolver static array sizes are not large enough.",CURRENT_FUNCTION);
226
227 /*--- Deallocate arrays ---*/
228 delete [] Mvec_Inf;
229
230}

239void CNEMOEulerSolver::CommonPreprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh,
240 unsigned short iRKStep, unsigned short RunTime_EqSystem, bool Output) {
241
242 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
243 const bool center = (config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED);
244 const bool center_jst = (config->GetKind_Centered_Flow() == CENTERED::JST) && (iMesh == MESH_0);
245 const bool center_jst_ke = (config->GetKind_Centered_Flow() == CENTERED::JST_KE) && (iMesh == MESH_0);
246
247 /*--- Set the primitive variables ---*/
248 ompMasterAssignBarrier(ErrorCounter,0);
249 SU2_OMP_ATOMIC
250 ErrorCounter += SetPrimitive_Variables(solver_container, config, Output);
251
252 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
253 { /*--- Ops that are not OpenMP parallel go in this block. ---*/
254
255 if ((iMesh == MESH_0) && (config->GetComm_Level() == COMM_FULL)) {
256 unsigned long tmp = ErrorCounter;
257 SU2_MPI::Allreduce(&tmp, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
258 config->SetNonphysical_Points(ErrorCounter);
259
260 if ((rank == MASTER_NODE) && (ErrorCounter != 0))
261 cout << "Warning. The initial solution contains "<< ErrorCounter << " points that are not physical." << endl;
262 }
263 }
264 END_SU2_OMP_SAFE_GLOBAL_ACCESS
265
266 /*--- Artificial dissipation ---*/
267
268 if (center && !Output) {
269 SetMax_Eigenvalue(geometry, config);
270 if (center_jst) SetUndivided_Laplacian(geometry, config);
271 if (center_jst || center_jst_ke) SetCentered_Dissipation_Sensor(geometry, config);
272 }
273
274 /*--- Set Pressure diffusion sensor ---*/
275 if (config->GetKind_Upwind_Flow() == UPWIND::AUSMPLUSM)
276 SetPressureDiffusionSensor(geometry, config);
277
278 /*--- Initialize the Jacobian matrix and residual, not needed for the reducer strategy
279 * as we set blocks (including diagonal ones) and completely overwrite. ---*/
280
281 if(!ReducerStrategy && !Output) {
282 LinSysRes.SetValZero();
283 if (implicit) Jacobian.SetValZero();
284 }
285}

287void CNEMOEulerSolver::Preprocessing(CGeometry *geometry, CSolver **solver_container,
288 CConfig *config, unsigned short iMesh,
289 unsigned short iRKStep,
290 unsigned short RunTime_EqSystem, bool Output) {
291 const unsigned long InnerIter = config->GetInnerIter();
292 const bool muscl = config->GetMUSCL_Flow() && (iMesh == MESH_0);
293 const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE) && (InnerIter <= config->GetLimiterIter());
294 const bool center = config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED;
295 const bool van_albada = config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE;
296
297 /*--- Common preprocessing steps ---*/
298 CommonPreprocessing(geometry, solver_container, config, iMesh, iRKStep, RunTime_EqSystem, Output);
299
300 /*--- Upwind second order reconstruction ---*/
301 if (muscl && !center && !Output) {
302
303 /*--- Calculate the gradients ---*/
304 if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
305 SetPrimitive_Gradient_GG(geometry, config, true);
306 }
307 if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
308 SetPrimitive_Gradient_LS(geometry, config, true);
309 }
310
311 /*--- Limiter computation ---*/
312 if (limiter && !van_albada) {
313 SetPrimitive_Limiter(geometry, config);
314 }
315 }

318unsigned long CNEMOEulerSolver::SetPrimitive_Variables(CSolver **solver_container, CConfig *config, bool Output) {
319
320 unsigned long nonPhysicalPoints = 0;
321 bool nonphysical = true;
322
323 for (auto iPoint = 0ul; iPoint < nPoint; iPoint ++) {
324
325 /*--- Incompressible flow, primitive variables ---*/
326
327 nonphysical = nodes->SetPrimVar(iPoint,FluidModel);
328
329 /* Check for non-realizable states for reporting. */
330
331 if (nonphysical) nonPhysicalPoints++;
332
333 /*--- Initialize the convective, source and viscous residual vector ---*/
334
335 if (!Output) LinSysRes.SetBlock_Zero(iPoint);
336
337 }
338
339 return nonPhysicalPoints;

342void CNEMOEulerSolver::SetTime_Step(CGeometry *geometry, CSolver **solver_container, CConfig *config,
343 unsigned short iMesh, unsigned long Iteration) {
344
345 /*--- Define an object to compute the speed of sound. ---*/
346 struct SoundSpeed {
347 FORCEINLINE su2double operator() (const CNEMOEulerVariable& nodes, unsigned long iPoint, unsigned long jPoint) const {
348 return 0.5 * (nodes.GetSoundSpeed(iPoint) + nodes.GetSoundSpeed(jPoint));
349 }
350
351 FORCEINLINE su2double operator() (const CNEMOEulerVariable& nodes, unsigned long iPoint) const {
352 return nodes.GetSoundSpeed(iPoint);
353 }
354
355 } soundSpeed;
356
357 /*--- Define an object to compute the viscous eigenvalue. ---*/
358 struct LambdaVisc {
359 FORCEINLINE su2double lambda(su2double lamVisc, su2double eddyVisc, su2double rho, su2double k, su2double cv) const {
360 /*--- Determine the viscous spectral radius and apply it to the control volume ---*/
361 const su2double Lambda_1 = (4.0/3.0)*(lamVisc + eddyVisc);
362 return (Lambda_1 + k/cv)/rho;
363 }
364
365 FORCEINLINE su2double operator() (const CNEMOEulerVariable& nodes, unsigned long iPoint, unsigned long jPoint) const {
366 const su2double lamVisc = 0.5*(nodes.GetLaminarViscosity(iPoint) + nodes.GetLaminarViscosity(jPoint));
367 const su2double eddyVisc = 0.5*(nodes.GetEddyViscosity(iPoint) + nodes.GetEddyViscosity(jPoint));
368 const su2double thermalCond = 0.5*(nodes.GetThermalConductivity(iPoint) + nodes.GetThermalConductivity(jPoint) +
369 nodes.GetThermalConductivity_ve(iPoint) + nodes.GetThermalConductivity_ve(jPoint));
370 const su2double density = 0.5*(nodes.GetDensity(iPoint) + nodes.GetDensity(jPoint));
371 const su2double cv = 0.5*(nodes.GetRhoCv_tr(iPoint) + nodes.GetRhoCv_ve(iPoint) +
372 nodes.GetRhoCv_tr(jPoint) + nodes.GetRhoCv_ve(jPoint))/ density;
373 return lambda(lamVisc, eddyVisc, density, thermalCond, cv);
374 }
375
376 FORCEINLINE su2double operator() (const CNEMOEulerVariable& nodes, unsigned long iPoint) const {
377 const su2double lamVisc = nodes.GetLaminarViscosity(iPoint);
378 const su2double eddyVisc = nodes.GetEddyViscosity(iPoint);
379 const su2double thermalCond = nodes.GetThermalConductivity(iPoint) + nodes.GetThermalConductivity_ve(iPoint);
380 const su2double density = nodes.GetDensity(iPoint);
381 const su2double cv = (nodes.GetRhoCv_tr(iPoint) + nodes.GetRhoCv_ve(iPoint))/ density;
382 return lambda(lamVisc, eddyVisc, density, thermalCond, cv);
383 }
384
385 } lambdaVisc;
386
387 /*--- Now instantiate the generic implementation with the two functors above. ---*/
388
389 SetTime_Step_impl(soundSpeed, lambdaVisc, geometry, solver_container, config, iMesh, Iteration);
390
391}

413void CNEMOEulerSolver::Centered_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
414 CConfig *config, unsigned short iMesh, unsigned short iRKStep) {
415
416 CNumerics* numerics = numerics_container[CONV_TERM];
417
418 /*--- Set booleans based on config settings ---*/
419 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
420
421 for (auto iEdge = 0ul; iEdge < geometry->GetnEdge(); iEdge++) {
422
423 /*--- Points in edge, set normal vectors, and number of neighbors ---*/
424 const auto iPoint = geometry->edges->GetNode(iEdge, 0);
425 const auto jPoint = geometry->edges->GetNode(iEdge, 1);
426 numerics->SetNormal(geometry->edges->GetNormal(iEdge));
427 numerics->SetNeighbor(geometry->nodes->GetnNeighbor(iPoint),
428 geometry->nodes->GetnNeighbor(jPoint));
429
430 /*--- Pass conservative & primitive variables w/o reconstruction to CNumerics ---*/
431 numerics->SetConservative(nodes->GetSolution(iPoint), nodes->GetSolution(jPoint));
432 numerics->SetPrimitive (nodes->GetPrimitive(iPoint), nodes->GetPrimitive(jPoint));
433
434 /*--- Pass supplementary information to CNumerics ---*/
435 numerics->SetdPdU ( nodes->GetdPdU(iPoint), nodes->GetdPdU(jPoint));
436 numerics->SetdTdU ( nodes->GetdTdU(iPoint), nodes->GetdTdU(jPoint));
437 numerics->SetdTvedU( nodes->GetdTvedU(iPoint), nodes->GetdTvedU(jPoint));
438 numerics->SetEve ( nodes->GetEve(iPoint), nodes->GetEve(jPoint));
439 numerics->SetCvve ( nodes->GetCvve(iPoint), nodes->GetCvve(jPoint));
440 numerics->SetGamma ( nodes->GetGamma(iPoint), nodes->GetGamma(jPoint));
441
442 /*--- Set the largest convective eigenvalue ---*/
443 numerics->SetLambda(nodes->GetLambda(iPoint), nodes->GetLambda(jPoint));
444
445 /*--- Compute residuals, and Jacobians ---*/
446 auto residual = numerics->ComputeResidual(config);
447
448 /*--- Check residuals/Jacobians --*/
449 bool err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
450
451 /*--- Update the residual and Jacobian ---*/
452 if (!err) {
453 LinSysRes.AddBlock(iPoint, residual);
454 LinSysRes.SubtractBlock(jPoint, residual);
455 if (implicit) {
456 Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
457 }
458 }
459 }

462void CNEMOEulerSolver::Upwind_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
463 CConfig *config, unsigned short iMesh) {
464
465 /*--- Set booleans based on config settings ---*/
466 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
467 const bool muscl = (config->GetMUSCL_Flow() && (iMesh == MESH_0));
468 const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE);
469 const bool van_albada = (config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE);
470
471 /*--- Non-physical counter. ---*/
472 unsigned long counter_local = 0;
473 SU2_OMP_MASTER
474 ErrorCounter = 0;
475 END_SU2_OMP_MASTER
476
477 /*--- Pick one numerics object per thread. ---*/
478 CNumerics* numerics = numerics_container[CONV_TERM];
479
480 /*--- Static arrays for MUSCL reconstructed variables ---*/
481 su2double Primitive_i[MAXNVAR] = {0.0}, Primitive_j[MAXNVAR] = {0.0};
482 su2double Conserved_i[MAXNVAR] = {0.0}, Conserved_j[MAXNVAR] = {0.0};
483 su2double dPdU_i[MAXNVAR] = {0.0}, dPdU_j[MAXNVAR] = {0.0};
484 su2double dTdU_i[MAXNVAR] = {0.0}, dTdU_j[MAXNVAR] = {0.0};
485 su2double dTvedU_i[MAXNVAR] = {0.0}, dTvedU_j[MAXNVAR] = {0.0};
486 su2double Eve_i[MAXNVAR] = {0.0}, Eve_j[MAXNVAR] = {0.0};
487 su2double Cvve_i[MAXNVAR] = {0.0}, Cvve_j[MAXNVAR] = {0.0};
488 su2double Project_Grad_i[MAXNVAR] = {0.0}, Project_Grad_j[MAXNVAR] = {0.0};
489 su2double Gamma_i = 0.0, Gamma_j = 0.0;
490
491 /*--- Loop over edges and calculate convective fluxes ---*/
492 for(auto iEdge = 0ul; iEdge < geometry->GetnEdge(); iEdge++) {
493
494 /*--- Retrieve node numbers and pass edge normal to CNumerics ---*/
495 const auto iPoint = geometry->edges->GetNode(iEdge, 0);
496 const auto jPoint = geometry->edges->GetNode(iEdge, 1);
497
498 numerics->SetNormal(geometry->edges->GetNormal(iEdge));
499
500 const auto Coord_i = geometry->nodes->GetCoord(iPoint);
501 const auto Coord_j = geometry->nodes->GetCoord(jPoint);
502
503 /*--- Get conserved & primitive variables from CVariable ---*/
504 auto U_i = nodes->GetSolution(iPoint); auto U_j = nodes->GetSolution(jPoint);
505 auto V_i = nodes->GetPrimitive(iPoint); auto V_j = nodes->GetPrimitive(jPoint);
506
507 /*--- Set them with or without high order reconstruction using MUSCL strategy. ---*/
508 if (!muscl) {
509
510 numerics->SetPrimitive (V_i, V_j);
511 numerics->SetConservative(U_i, U_j);
512 numerics->SetdPdU (nodes->GetdPdU(iPoint), nodes->GetdPdU(jPoint));
513 numerics->SetdTdU (nodes->GetdTdU(iPoint), nodes->GetdTdU(jPoint));
514 numerics->SetdTvedU(nodes->GetdTvedU(iPoint), nodes->GetdTvedU(jPoint));
515 numerics->SetEve (nodes->GetEve(iPoint), nodes->GetEve(jPoint));
516 numerics->SetCvve (nodes->GetCvve(iPoint), nodes->GetCvve(jPoint));
517 numerics->SetGamma (nodes->GetGamma(iPoint), nodes->GetGamma(jPoint));
518
519 } else {
520
521 /*--- High order reconstruction using MUSCL strategy ---*/
522 su2double Vector_ij[MAXNDIM] = {0.0};
523 for (auto iDim = 0ul; iDim < nDim; iDim++) {
524 Vector_ij[iDim] = 0.5*(Coord_j[iDim] - Coord_i[iDim]);
525 }
526
527 /*--- Retrieve gradient information ---*/
528 auto Gradient_i = nodes->GetGradient_Reconstruction(iPoint);
529 auto Gradient_j = nodes->GetGradient_Reconstruction(jPoint);
530
531 /*--- Set and extract limiters ---*/
532 su2double *Limiter_i = nullptr, *Limiter_j = nullptr;
533
534 if (limiter && !van_albada){
535 Limiter_i = nodes->GetLimiter_Primitive(iPoint);
536 Limiter_j = nodes->GetLimiter_Primitive(jPoint);
537 }
538
539 su2double lim_i = 2.0;
540 su2double lim_j = 2.0;
541
542 for (auto iVar = 0ul; iVar < nPrimVarGrad; iVar++) {
543 Project_Grad_i[iVar] = 0.0; Project_Grad_j[iVar] = 0.0;
544
545 for (auto iDim = 0ul; iDim < nDim; iDim++) {
546 Project_Grad_i[iVar] += Vector_ij[iDim]*Gradient_i[iVar][iDim];
547 Project_Grad_j[iVar] -= Vector_ij[iDim]*Gradient_j[iVar][iDim];
548 }
549
550 if (limiter) {
551 if (van_albada) {
552 su2double V_ij = V_j[iVar] - V_i[iVar];
553 su2double va_lim_i = LimiterHelpers<>::vanAlbadaFunction(Project_Grad_i[iVar], V_ij, EPS);
554 su2double va_lim_j = LimiterHelpers<>::vanAlbadaFunction(-Project_Grad_j[iVar], V_ij, EPS);
555 lim_i = min(lim_i, va_lim_i);
556 lim_j = min(lim_j, va_lim_j);
557 } else {
558 lim_i = min(lim_i, Limiter_i[iVar]);
559 lim_j = min(lim_j, Limiter_j[iVar]);
560 }
561 } else {
562 lim_i = lim_j = 1.0;
563 }
564 }
565 su2double lim_ij = min(lim_i, lim_j);
566
567 for (auto iVar = 0ul; iVar < nPrimVarGrad; iVar++) {
568 Primitive_i[iVar] = V_i[iVar] + lim_ij*Project_Grad_i[iVar];
569 Primitive_j[iVar] = V_j[iVar] + lim_ij*Project_Grad_j[iVar];
570 }
571
572 /*--- Check for non-physical solutions after reconstruction. If found, use the
573 cell-average value of the solution. This is a locally 1st order approximation,
574 which is typically only active during the start-up of a calculation. ---*/
575 bool chk_err_i = CheckNonPhys(Primitive_i);
576 bool chk_err_j = CheckNonPhys(Primitive_j);
577
578 nodes->SetNon_Physical(iPoint, chk_err_i);
579 nodes->SetNon_Physical(jPoint, chk_err_j);
580
581 /*--- Get updated state, in case the point recovered after the set. ---*/
582 chk_err_i = nodes->GetNon_Physical(iPoint);
583 chk_err_j = nodes->GetNon_Physical(jPoint);
584
585 counter_local += chk_err_i + chk_err_j;
586
587 /*--- Compute Secondary variables in a thermaodynamically consistent way. ---*/
588 if (!chk_err_i) Gamma_i = ComputeConsistentExtrapolation(GetFluidModel(), nSpecies, Primitive_i, dPdU_i, dTdU_i, dTvedU_i, Eve_i, Cvve_i);
589 if (!chk_err_j) Gamma_j = ComputeConsistentExtrapolation(GetFluidModel(), nSpecies, Primitive_j, dPdU_j, dTdU_j, dTvedU_j, Eve_j, Cvve_j);
590
591 /*--- Recompute Conserved variables if Roe or MSW scheme ---*/
592 if ((config->GetKind_Upwind_Flow() == UPWIND::ROE) || (config->GetKind_Upwind_Flow() == UPWIND::MSW)){
593 if (!chk_err_i) RecomputeConservativeVector(Conserved_i, Primitive_i);
594 if (!chk_err_j) RecomputeConservativeVector(Conserved_j, Primitive_j);
595 }
596
597 /*--- If non-physical, revert to first order ---*/
598 numerics->SetConservative(chk_err_i ? U_i : Conserved_i, chk_err_j ? U_j : Conserved_j);
599 numerics->SetPrimitive (chk_err_i ? V_i : Primitive_i, chk_err_j ? V_j : Primitive_j);
600 numerics->SetdPdU (chk_err_i ? nodes->GetdPdU (iPoint) : dPdU_i, chk_err_j ? nodes->GetdPdU (jPoint) : dPdU_j);
601 numerics->SetdTdU (chk_err_i ? nodes->GetdTdU (iPoint) : dTdU_i, chk_err_j ? nodes->GetdTdU (jPoint) : dTdU_j);
602 numerics->SetdTvedU(chk_err_i ? nodes->GetdTvedU(iPoint) : dTvedU_i, chk_err_j ? nodes->GetdTvedU(jPoint) : dTvedU_j);
603 numerics->SetEve (chk_err_i ? nodes->GetEve (iPoint) : Eve_i, chk_err_j ? nodes->GetEve (jPoint) : Eve_j);
604 numerics->SetCvve (chk_err_i ? nodes->GetCvve (iPoint) : Cvve_i, chk_err_j ? nodes->GetCvve (jPoint) : Cvve_j);
605 numerics->SetGamma (chk_err_i ? nodes->GetGamma (iPoint) : Gamma_i, chk_err_j ? nodes->GetGamma (jPoint) : Gamma_j);
606
607 }
608 if (config->GetKind_Upwind_Flow() == UPWIND::AUSMPLUSM)
609 numerics->SetSensor (nodes->GetSensor(iPoint), nodes->GetSensor(jPoint));
610
611 /*--- Compute the residual ---*/
612 auto residual = numerics->ComputeResidual(config);
613
614 /*--- Check for NaNs before applying the residual to the linear system ---*/
615 bool err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
616
617 /*--- Update the residual and Jacobian ---*/
618 if (!err) {
619 LinSysRes.AddBlock(iPoint, residual);
620 LinSysRes.SubtractBlock(jPoint, residual);
621 if (implicit) {
622 Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
623 }
624 }
625 }
626
627 /*--- Warning message about non-physical reconstructions. ---*/
628 if ((iMesh == MESH_0) && (config->GetComm_Level() == COMM_FULL)) {
629 /*--- Add counter results for all threads. ---*/
630 SU2_OMP_ATOMIC
631 ErrorCounter += counter_local;
632
633 /*--- Add counter results for all ranks. ---*/
634 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
635 {
636 counter_local = ErrorCounter;
637 SU2_MPI::Reduce(&counter_local, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE, SU2_MPI::GetComm());
638 config->SetNonphysical_Reconstr(ErrorCounter);
639 }
640 END_SU2_OMP_SAFE_GLOBAL_ACCESS
641 }
642}

677void CNEMOEulerSolver::RecomputeConservativeVector(su2double *U, const su2double *V) const {
678
679 /*---Useful variables ---*/
680 vector<su2double> rhos;
681 rhos.resize(nSpecies,0.0);
682
683 /*--- Set Indices ---*/
684 //TODO: Move these to a general location
685 const unsigned short RHO_INDEX = nodes->GetRhoIndex();
686 const unsigned short T_INDEX = nodes->GetTIndex();
687 const unsigned short TVE_INDEX = nodes->GetTveIndex();

690 /*--- Set densities and mass fraction ---*/
691 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++){
692 U[iSpecies] = V[iSpecies];
693 rhos[iSpecies] = V[iSpecies];
694 }
695
696 /*--- Set momentum and compute v^2 ---*/
697 su2double sqvel = 0.0;
698 for (auto iDim = 0ul; iDim < nDim; iDim++){
699 U[nSpecies+iDim] = V[RHO_INDEX]*V[VEL_INDEX+iDim];
700 sqvel += V[VEL_INDEX+iDim]*V[VEL_INDEX+iDim];
701 }
702
703 /*--- Set the fluidmodel and recompute energies ---*/
704 FluidModel->SetTDStateRhosTTv( rhos, V[T_INDEX], V[TVE_INDEX]);
705 const auto& Energies = FluidModel->ComputeMixtureEnergies();
706
707 /*--- Set conservative energies ---*/
708 U[nSpecies+nDim] = V[RHO_INDEX]*(Energies[0]+0.5*sqvel);
709 U[nSpecies+nDim+1] = V[RHO_INDEX]*(Energies[1]);
710
711}

746void CNEMOEulerSolver::Source_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
747
748 /*--- Assign booleans ---*/
749 bool err = false;
750 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
751 const bool frozen = config->GetFrozen();
752 const bool monoatomic = config->GetMonoatomic();
753 const bool axisymm = config->GetAxisymmetric();
754 const bool viscous = config->GetViscous();
755 const bool rans = (config->GetKind_Turb_Model() != TURB_MODEL::NONE);
756
757 CNumerics* numerics = numerics_container[SOURCE_FIRST_TERM];
758
759 /*--- Initialize the error counter ---*/
760 unsigned long eAxi_local = 0;
761 unsigned long eChm_local = 0;
762 unsigned long eVib_local = 0;
763
764 /*--- Initialize the source residual to zero ---*/
765 for (auto iVar = 0ul; iVar < nVar; iVar++) Residual[iVar] = 0.0;
766
767 /*--- Preprocess viscous axisymm variables (if necessary) ---*/
768 if (axisymm && viscous) {
769 ComputeAxisymmetricAuxGradients(geometry,config);
770 }
771
772 AD::StartNoSharedReading();
773
774 /*--- loop over interior points ---*/
775 SU2_OMP_FOR_DYN(omp_chunk_size)
776 for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
777
778 /*--- Set conserved & primitive variables ---*/
779 numerics->SetConservative(nodes->GetSolution(iPoint), nullptr);
780 numerics->SetPrimitive (nodes->GetPrimitive(iPoint), nullptr);
781
782 /*--- Pass supplementary information to CNumerics ---*/
783 numerics->SetdPdU(nodes->GetdPdU(iPoint), nullptr);
784 numerics->SetdTdU(nodes->GetdTdU(iPoint), nullptr);
785 numerics->SetdTvedU(nodes->GetdTvedU(iPoint), nullptr);
786 numerics->SetEve(nodes->GetEve(iPoint), nullptr);
787 numerics->SetCvve(nodes->GetCvve(iPoint), nullptr);

790 numerics->SetVolume(geometry->nodes->GetVolume(iPoint));
791 numerics->SetCoord(geometry->nodes->GetCoord(iPoint), nullptr);
792
793 /*--- Compute finite rate chemistry ---*/
794
795 if(!monoatomic){
796 if(!frozen){
797 /*--- Compute the non-equilibrium chemistry ---*/
798 auto residual = numerics->ComputeChemistry(config);
799
800 /*--- Check for errors before applying source to the linear system ---*/
801 err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
802
803 /*--- Apply the chemical sources to the linear system ---*/
804 if (!err) {
805 LinSysRes.SubtractBlock(iPoint, residual);
806 if (implicit)
807 Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
808 } else
809 eChm_local++;
810 }
811 }
812
813 /*--- Compute vibrational energy relaxation ---*/
814 /// NOTE: Jacobians don't account for relaxation time derivatives
815
816 if (!monoatomic){
817 auto residual = numerics->ComputeVibRelaxation(config);
818
819 /*--- Check for errors before applying source to the linear system ---*/
820 err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
821
822 /*--- Apply the vibrational relaxation terms to the linear system ---*/
823 if (!err) {
824 LinSysRes.SubtractBlock(iPoint, residual);
825 if (implicit)
826 Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
827 } else
828 eVib_local++;
829 }
830
831 /*--- Compute axisymmetric source terms (if needed) ---*/
832 if (axisymm) {
833
834 /*--- If necessary, set variables needed for viscous computation ---*/
835 if (viscous) {
836
837 /*--- Set gradient of primitive variables ---*/
838 numerics->SetPrimVarGradient(nodes->GetGradient_Primitive(iPoint), nullptr);
839
840 /*--- Set gradient of auxillary variables ---*/
841 numerics->SetAuxVarGrad(nodes->GetAuxVarGradient(iPoint), nullptr);
842
843 /*--- Set diffusion coefficient ---*/
844 numerics->SetDiffusionCoeff(nodes->GetDiffusionCoeff(iPoint), nullptr);
845
846 /*--- Laminar viscosity ---*/
847 numerics->SetLaminarViscosity(nodes->GetLaminarViscosity(iPoint), 0.0);
848
849 /*--- Eddy viscosity ---*/
850 numerics->SetEddyViscosity(nodes->GetEddyViscosity(iPoint), 0.0);
851
852 /*--- Thermal conductivity ---*/
853 numerics->SetThermalConductivity(nodes->GetThermalConductivity(iPoint), 0.0);
854
855 /*--- Vib-el. thermal conductivity ---*/
856 numerics->SetThermalConductivity_ve(nodes->GetThermalConductivity_ve(iPoint), 0.0);
857
858 /*--- Set turbulence kinetic energy ---*/
859 if (rans){
860 CVariable* turbNodes = solver_container[TURB_SOL]->GetNodes();
861 numerics->SetTurbKineticEnergy(turbNodes->GetSolution(iPoint,0), 0.0);
862 }
863 }
864
865 auto residual = numerics->ComputeAxisymmetric(config);
866
867 /*--- Check for errors before applying source to the linear system ---*/
868 err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
869
870 /*--- Apply the update to the linear system ---*/
871 if (!err) {
872 LinSysRes.AddBlock(iPoint, residual);
873 if (implicit)
874 Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
875 } else
876 eAxi_local++;
877 }
878 }
879 END_SU2_OMP_FOR
880
881 AD::EndNoSharedReading();
882
883 /*--- Checking for NaN ---*/
884 unsigned long eAxi_global = eAxi_local;
885 unsigned long eChm_global = eChm_local;
886 unsigned long eVib_global = eVib_local;
887
888 //THIS IS NO FUN
889 if ((eAxi_global != 0) ||
890 (eChm_global != 0) ||
891 (eVib_global != 0)) {
892 cout << "Warning!! Instances of NaN in the following source terms: " << endl;
893 cout << "Axisymmetry: " << eAxi_global << endl;
894 cout << "Chemical: " << eChm_global << endl;
895 cout << "Vib. Relax: " << eVib_global << endl;
896 }

931void CNEMOEulerSolver::ComputeUnderRelaxationFactor(const CConfig *config) {
932
933 /* Loop over the solution update given by relaxing the linear
934 system for this nonlinear iteration. */
935
936 const su2double allowableRatio = 0.2;
937
938 SU2_OMP_FOR_STAT(omp_chunk_size)
939 for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
940 su2double localUnderRelaxation = 1.0;
941
942 su2double num = 0.0;
943 su2double denom = 0.0;
944
945 for (auto iVar = 0; iVar < nVar; iVar++) {
946 /* We impose a limit on the maximum percentage that the
947 density (sum of all species) and energy can change over a nonlinear iteration. */
948
949 const unsigned long index = iPoint * nVar + iVar;
950 if (iVar < config->GetnSpecies()) {
951 num += fabs(LinSysSol[index]);
952 denom += fabs(nodes->GetSolution(iPoint, iVar));
953
954 /*--- If final density/species, compute Under-relaxation ---*/
955 if (iVar == (config ->GetnSpecies()-1)){
956 su2double ratio = (num/(denom+EPS));
957 if (ratio > allowableRatio) {
958 localUnderRelaxation = min(allowableRatio / ratio, localUnderRelaxation);
959 }
960 }
961
962 /*--- Energy ---*/
963 if (iVar == (nVar-2)){
964 su2double ratio = fabs(LinSysSol[index]) / (fabs(nodes->GetSolution(iPoint, iVar)) + EPS);
965 if (ratio > allowableRatio) {
966 localUnderRelaxation = min(allowableRatio / ratio, localUnderRelaxation);
967 }
968 }
969 }
970 }
971
972 /* Threshold the relaxation factor in the event that there is
973 a very small value. This helps avoid catastrophic crashes due
974 to non-realizable states by canceling the update. */
975
976 if (localUnderRelaxation < 1e-10) localUnderRelaxation = 0.0;
977
978 /* Store the under-relaxation factor for this point. */
979
980 nodes->SetUnderRelaxation(iPoint, localUnderRelaxation);
981 }
982 END_SU2_OMP_FOR
983}

1047 /*--- ---*/

1397void CNEMOEulerSolver::BC_Sym_Plane(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
1398 CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
1399
1400 /*--- Allocate the necessary vector structures ---*/
1401 su2double Normal[MAXNDIM] = {0.0}, UnitNormal[MAXNDIM] = {0.0};
1402
1403 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1404
1405 /*--- Loop over all the vertices on this boundary (val_marker) ---*/
1406 for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
1407 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
1408
1409 /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
1410 if (geometry->nodes->GetDomain(iPoint)) {
1411
1412 /*--- Normal vector for this vertex (negative for outward convention) ---*/
1413 geometry->vertex[val_marker][iVertex]->GetNormal(Normal);
1414
1415 /*--- Calculate parameters from the geometry ---*/
1416 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
1417
1418 for (auto iDim = 0ul; iDim < nDim; iDim++){
1419 UnitNormal[iDim] = -Normal[iDim]/Area;
1420 }
1421
1422 /*--- Retrieve the pressure on the vertex ---*/
1423 const su2double P = nodes->GetPressure(iPoint);
1424
1425 /*--- Apply the flow-tangency b.c. to the convective flux ---*/
1426 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
1427 Residual[iSpecies] = 0.0;
1428 for (auto iDim = 0ul; iDim < nDim; iDim++){
1429 Residual[nSpecies+iDim] = P * UnitNormal[iDim] * Area;
1430 }
1431 Residual[nSpecies+nDim] = 0.0;
1432 Residual[nSpecies+nDim+1] = 0.0;
1433
1434 /*--- Add value to the residual ---*/
1435 LinSysRes.AddBlock(iPoint, Residual);
1436
1437 /*--- If using implicit time-stepping, calculate b.c. contribution to Jacobian ---*/
1438 if (implicit) {
1439
1440 /*--- Allocate arrays needed for implicit solver ---*/
1441 su2double Velocity[MAXNDIM] = {0.0};
1442
1443 /*--- Get species molar mass ---*/
1444 auto& Ms = FluidModel->GetSpeciesMolarMass();
1445
1446 /*--- Initialize Jacobian ---*/
1447 for (auto iVar = 0ul; iVar < nVar; iVar++)
1448 for (auto jVar = 0ul; jVar < nVar; jVar++)
1449 Jacobian_i[iVar][jVar] = 0.0;
1450
1451 /*--- Calculate state i ---*/
1452 const su2double rho = nodes->GetDensity(iPoint);
1453 const su2double rhoE = nodes->GetSolution(iPoint,nSpecies+nDim);
1454 const su2double rhoEve = nodes->GetSolution(iPoint,nSpecies+nDim+1);
1455 const auto dPdU = nodes->GetdPdU(iPoint);
1456 for (auto iDim = 0ul; iDim < nDim; iDim++)
1457 Velocity[iDim] = nodes->GetVelocity(iPoint,iDim);
1458
1459 su2double conc = 0.0;
1460 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
1461 const su2double cs = nodes->GetMassFraction(iPoint,iSpecies);
1462 conc += cs * rho/Ms[iSpecies];
1463
1464 for (auto iDim = 0ul; iDim < nDim; iDim++) {
1465 Jacobian_i[nSpecies+iDim][iSpecies] = dPdU[iSpecies] * UnitNormal[iDim];
1466 Jacobian_i[iSpecies][nSpecies+iDim] = cs * UnitNormal[iDim];
1467 }
1468 }
1469
1470 for (auto iDim = 0ul; iDim < nDim; iDim++) {
1471 for (auto jDim = 0ul; jDim < nDim; jDim++) {
1472 Jacobian_i[nSpecies+iDim][nSpecies+jDim] = Velocity[iDim]*UnitNormal[jDim]
1473 + dPdU[nSpecies+jDim]*UnitNormal[iDim];
1474 }
1475 Jacobian_i[nSpecies+iDim][nSpecies+nDim] = dPdU[nSpecies+nDim] *UnitNormal[iDim];
1476 Jacobian_i[nSpecies+iDim][nSpecies+nDim+1] = dPdU[nSpecies+nDim+1]*UnitNormal[iDim];
1477
1478 Jacobian_i[nSpecies+nDim][nSpecies+iDim] = (rhoE+P)/rho * UnitNormal[iDim];
1479 Jacobian_i[nSpecies+nDim+1][nSpecies+iDim] = rhoEve/rho * UnitNormal[iDim];
1480 }
1481
1482 /*--- Integrate over the dual-grid area ---*/
1483 for (auto iVar = 0ul; iVar < nVar; iVar++)
1484 for (auto jVar = 0ul; jVar < nVar; jVar++)
1485 Jacobian_i[iVar][jVar] = Jacobian_i[iVar][jVar] * Area;
1486
1487 /*--- Apply the contribution to the system ---*/
1488 Jacobian.AddBlock2Diag(iPoint, Jacobian_i);
1489
1490 }
1491 }
1492 }

1611void CNEMOEulerSolver::BC_Inlet(CGeometry *geometry, CSolver **solver_container,
1612 CNumerics *conv_numerics, CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
1613 SU2_MPI::Error("BC_INLET: Not operational in NEMO.", CURRENT_FUNCTION);
1614
1615 unsigned short RHO_INDEX, nSpecies;
1616
1617 su2double T_Total, P_Total, Velocity[3], Velocity2, H_Total, Temperature, Riemann,
1618 Pressure, Density, Energy, Mach2, SoundSpeed2, SoundSpeed_Total2, Vel_Mag,
1619 alpha, aa, bb, cc, dd, Area, UnitNormal[3] = {0.0};
1620
1621 const su2double *Flow_Dir;
1622 bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1623
1624 bool dynamic_grid = config->GetGrid_Movement();
1625 su2double Two_Gamma_M1 = 2.0/Gamma_Minus_One;
1626 su2double Gas_Constant = config->GetGas_ConstantND();
1627 INLET_TYPE Kind_Inlet = config->GetKind_Inlet();
1628 string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
1629
1630 auto *U_domain = new su2double[nVar]; auto *U_inlet = new su2double[nVar];
1631 auto *V_domain = new su2double[nPrimVar]; auto *V_inlet = new su2double[nPrimVar];
1632 auto *Normal = new su2double[nDim];
1633
1634 nSpecies = config->GetnSpecies();
1635 auto *Spec_Density = new su2double[nSpecies];
1636 for (auto iSpecies = 0ul; iSpecies<nSpecies; iSpecies++)
1637 Spec_Density[iSpecies] = 0.0; /*--- To avoid a compiler warning. ---*/
1638
1639 RHO_INDEX = nodes->GetRhoIndex();
1640
1641 /*--- Loop over all the vertices on this boundary marker ---*/
1642 for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
1643 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
1644
1645 /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
1646 if (geometry->nodes->GetDomain(iPoint)) {
1647
1648 /*--- Normal vector for this vertex (negate for outward convention) ---*/
1649 geometry->vertex[val_marker][iVertex]->GetNormal(Normal);
1650 for (auto iDim = 0ul; iDim < nDim; iDim++) Normal[iDim] = -Normal[iDim];
1651 conv_numerics->SetNormal(Normal);
1652
1653 Area = GeometryToolbox::Norm(nDim, Normal);
1654
1655 for (auto iDim = 0ul; iDim < nDim; iDim++)
1656 UnitNormal[iDim] = Normal[iDim]/Area;
1657
1658 /*--- Retrieve solution at this boundary node ---*/
1659 for (auto iVar = 0ul; iVar < nVar; iVar++) U_domain[iVar] = nodes->GetSolution(iPoint, iVar);
1660 for (auto iVar = 0ul; iVar < nPrimVar; iVar++) V_domain[iVar] = nodes->GetPrimitive(iPoint,iVar);
1661
1662 /*--- Build the fictitious intlet state based on characteristics ---*/
1663
1664 /*--- Subsonic inflow: there is one outgoing characteristic (u-c),
1665 therefore we can specify all but one state variable at the inlet.
1666 The outgoing Riemann invariant provides the final piece of info.
1667 Adapted from an original implementation in the Stanford University
1668 multi-block (SUmb) solver in the routine bcSubsonicInflow.f90
1669 written by Edwin van der Weide, last modified 04-20-2009. ---*/
1670
1671 switch (Kind_Inlet) {
1672
1673 /*--- Total properties have been specified at the inlet. ---*/
1674 case INLET_TYPE::TOTAL_CONDITIONS:
1675
1676 /*--- Retrieve the specified total conditions for this inlet. ---*/
1677 P_Total = config->GetInlet_Ptotal(Marker_Tag);
1678 T_Total = config->GetInlet_Ttotal(Marker_Tag);
1679 Flow_Dir = config->GetInlet_FlowDir(Marker_Tag);
1680
1681 /*--- Non-dim. the inputs if necessary. ---*/
1682 P_Total /= config->GetPressure_Ref();
1683 T_Total /= config->GetTemperature_Ref();
1684
1685 /*--- Store primitives and set some variables for clarity. ---*/
1686 Density = V_domain[RHO_INDEX];
1687 for (auto iDim = 0ul; iDim < nDim; iDim++)
1688 Velocity[iDim] = U_domain[nSpecies+iDim]/Density;
1689
1690 Velocity2 = GeometryToolbox::SquaredNorm(nDim, Velocity);
1691 Energy = U_domain[nVar-2]/Density;
1692 Pressure = Gamma_Minus_One*Density*(Energy-0.5*Velocity2);
1693 H_Total = (Gamma*Gas_Constant/Gamma_Minus_One)*T_Total;
1694 SoundSpeed2 = Gamma*Pressure/Density;
1695
1696 /*--- Compute the acoustic Riemann invariant that is extrapolated
1697 from the domain interior. ---*/
1698 Riemann = 2.0*sqrt(SoundSpeed2)/Gamma_Minus_One;
1699 for (auto iDim = 0ul; iDim < nDim; iDim++)
1700 Riemann += Velocity[iDim]*UnitNormal[iDim];
1701
1702 /*--- Total speed of sound ---*/
1703 SoundSpeed_Total2 = Gamma_Minus_One*(H_Total - (Energy + Pressure/Density)+0.5*Velocity2) + SoundSpeed2;
1704
1705 /*--- Dot product of normal and flow direction. This should
1706 be negative due to outward facing boundary normal convention. ---*/
1707 alpha = GeometryToolbox::DotProduct(nDim, UnitNormal, Flow_Dir);
1708
1709 /*--- Coefficients in the quadratic equation for the velocity ---*/
1710 aa = 1.0 + 0.5*Gamma_Minus_One*alpha*alpha;
1711 bb = -1.0*Gamma_Minus_One*alpha*Riemann;
1712 cc = 0.5*Gamma_Minus_One*Riemann*Riemann
1713 -2.0*SoundSpeed_Total2/Gamma_Minus_One;
1714
1715 /*--- Solve quadratic equation for velocity magnitude. Value must
1716 be positive, so the choice of root is clear. ---*/
1717 dd = bb*bb - 4.0*aa*cc;
1718 dd = sqrt(max(0.0,dd));
1719 Vel_Mag = (-bb + dd)/(2.0*aa);
1720 Vel_Mag = max(0.0,Vel_Mag);
1721 Velocity2 = Vel_Mag*Vel_Mag;
1722
1723 /*--- Compute speed of sound from total speed of sound eqn. ---*/
1724 SoundSpeed2 = SoundSpeed_Total2 - 0.5*Gamma_Minus_One*Velocity2;
1725
1726 /*--- Mach squared (cut between 0-1), use to adapt velocity ---*/
1727 Mach2 = Velocity2/SoundSpeed2;
1728 Mach2 = min(1.0,Mach2);
1729 Velocity2 = Mach2*SoundSpeed2;
1730 Vel_Mag = sqrt(Velocity2);
1731 SoundSpeed2 = SoundSpeed_Total2 - 0.5*Gamma_Minus_One*Velocity2;
1732
1733 /*--- Compute new velocity vector at the inlet ---*/
1734 for (auto iDim = 0ul; iDim < nDim; iDim++)
1735 Velocity[iDim] = Vel_Mag*Flow_Dir[iDim];
1736
1737 /*--- Static temperature from the speed of sound relation ---*/
1738 Temperature = SoundSpeed2/(Gamma*Gas_Constant);
1739 //NEED TVE AS WELL
1740
1741 /*--- Static pressure using isentropic relation at a point ---*/
1742 Pressure = P_Total*pow((Temperature/T_Total),Gamma/Gamma_Minus_One);
1743
1744 /*--- Density at the inlet from the gas law ---*/
1745 Density = Pressure/(Gas_Constant*Temperature);
1746 //NEED SPECIES DENSITIES
1747
1748 /*--- Using pressure, density, & velocity, compute the energy ---*/
1749 Energy = Pressure/(Density*Gamma_Minus_One)+0.5*Velocity2;
1750 //NEED EVE AS WELL
1751
1752 /*--- Conservative variables, using the derived quantities ---*/
1753 for (auto iSpecies = 0ul; iSpecies<nSpecies; iSpecies++)
1754 U_inlet[iSpecies] = Spec_Density[iSpecies];
1755 for (auto iDim = 0ul; iDim < nDim; iDim++)
1756 U_inlet[nSpecies+iDim] = Velocity[iDim]*Density;
1757 U_inlet[nVar-2] = Energy*Density;
1758 //U_inlet[nVar-1]=Eve
1759
1760 /*--- Primitive variables, using the derived quantities ---*/
1761 for (auto iSpecies = 0ul; iSpecies<nSpecies; iSpecies++)
1762 V_inlet[iSpecies] = Spec_Density[iSpecies];
1763 V_inlet[nSpecies] = Temperature;
1764 //V_inlet[nSpecies+1] = Tve
1765 for (auto iDim = 0ul; iDim < nDim; iDim++)
1766 V_inlet[nSpecies+2] = Velocity[iDim];
1767 V_inlet[nSpecies+nDim+2] = Pressure;
1768 V_inlet[RHO_INDEX] = Density;
1769 //V_inlet[H_INDEX] = H;
1770 //V_inlet[A_INDEX] = A;
1771 //V_inlet[RHO_CVTR_INDEX] = rcvtr;
1772 //V_inlet[RHO_CVVE_INDEX] = rcvve;
1773
1774 break;
1775
1776 /*--- Mass flow has been specified at the inlet. ---*/
1777 case INLET_TYPE::MASS_FLOW:
1778
1779 /*--- Retrieve the specified mass flow for the inlet. ---*/
1780 Density = config->GetInlet_Ttotal(Marker_Tag);
1781 Vel_Mag = config->GetInlet_Ptotal(Marker_Tag);
1782 Flow_Dir = config->GetInlet_FlowDir(Marker_Tag);
1783
1784 /*--- Non-dim. the inputs if necessary. ---*/
1785 Density /= config->GetDensity_Ref();
1786 Vel_Mag /= config->GetVelocity_Ref();
1787
1788 /*--- Get primitives from current inlet state. ---*/
1789 for (auto iDim = 0ul; iDim < nDim; iDim++)
1790 Velocity[iDim] = nodes->GetVelocity(iPoint, iDim);
1791 Pressure = nodes->GetPressure(iPoint);
1792 SoundSpeed2 = Gamma*Pressure/U_domain[0];
1793
1794 /*--- Compute the acoustic Riemann invariant that is extrapolated
1795 from the domain interior. ---*/
1796 Riemann = Two_Gamma_M1*sqrt(SoundSpeed2);
1797 for (auto iDim = 0ul; iDim < nDim; iDim++)
1798 Riemann += Velocity[iDim]*UnitNormal[iDim];
1799
1800 /*--- Speed of sound squared for fictitious inlet state ---*/
1801 SoundSpeed2 = Riemann;
1802 for (auto iDim = 0ul; iDim < nDim; iDim++)
1803 SoundSpeed2 -= Vel_Mag*Flow_Dir[iDim]*UnitNormal[iDim];
1804
1805 SoundSpeed2 = max(0.0,0.5*Gamma_Minus_One*SoundSpeed2);
1806 SoundSpeed2 = SoundSpeed2*SoundSpeed2;
1807
1808 /*--- Pressure for the fictitious inlet state ---*/
1809 Pressure = SoundSpeed2*Density/Gamma;
1810
1811 /*--- Energy for the fictitious inlet state ---*/
1812 Energy = Pressure/(Density*Gamma_Minus_One)+0.5*Vel_Mag*Vel_Mag;
1813
1814 /*--- Conservative variables, using the derived quantities ---*/
1815 U_inlet[0] = Density;
1816 for (auto iDim = 0ul; iDim < nDim; iDim++)
1817 U_inlet[iDim+1] = Vel_Mag*Flow_Dir[iDim]*Density;
1818 U_inlet[nDim+1] = Energy*Density;
1819
1820 /*--- Primitive variables, using the derived quantities ---*/
1821 V_inlet[0] = Pressure / ( Gas_Constant * Density);
1822 for (auto iDim = 0ul; iDim < nDim; iDim++)
1823 V_inlet[iDim+1] = Vel_Mag*Flow_Dir[iDim];
1824 V_inlet[nDim+1] = Pressure;
1825 V_inlet[nDim+2] = Density;
1826
1827 break;
1828
1829 default:
1830 SU2_MPI::Error("Unsupported INLET_TYPE.", CURRENT_FUNCTION);
1831 break;
1832 }
1833
1834 /*--- Set various quantities in the solver class ---*/
1835 conv_numerics->SetConservative(U_domain, U_inlet);
1836
1837 if (dynamic_grid)
1838 conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
1839
1840 /*--- Compute the residual using an upwind scheme ---*/
1841 auto residual = conv_numerics->ComputeResidual(config);
1842
1843 LinSysRes.AddBlock(iPoint, residual);
1844
1845 /*--- Jacobian contribution for implicit integration ---*/
1846 if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
1847

1849// if (viscous) {
1850
1851// /*--- Set the normal vector and the coordinates ---*/
1852// visc_numerics->SetNormal(Normal);
1853// su2double Coord_Reflected[MAXNDIM];
1854// GeometryToolbox::PointPointReflect(nDim, geometry->nodes->GetCoord(Point_Normal),
1855// geometry->nodes->GetCoord(iPoint), Coord_Reflected);
1856// visc_numerics->SetCoord(geometry->nodes->GetCoord(), Coord_Reflected);
1857
1858// /*--- Primitive variables, and gradient ---*/
1859// visc_numerics->SetPrimitive(V_domain, V_inlet);
1860// visc_numerics->SetPrimVarGradient(nodes->GetGradient_Primitive(), nodes->GetGradient_Primitive());
1861
1862// /*--- Laminar viscosity ---*/
1863// visc_numerics->SetLaminarViscosity(nodes->GetLaminarViscosity(), nodes->GetLaminarViscosity());
1864
1865// /*--- Compute and update residual ---*/
1866// visc_numerics->ComputeResidual(Residual, Jacobian_i, Jacobian_j, config);

1873 }
1874 }

1880 delete [] V_inlet;
1881 delete [] Normal;
1882 delete [] Spec_Density;

1885void CNEMOEulerSolver::BC_Outlet(CGeometry *geometry, CSolver **solver_container,
1886 CNumerics *conv_numerics, CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
1887
1888 su2double UnitNormal[MAXNDIM] = {0.0};
1889 vector<su2double> rhos;
1890 rhos.resize(nSpecies,0.0);
1891
1892 string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
1893 const bool dynamic_grid = config->GetGrid_Movement();
1894 const bool gravity = config->GetGravityForce();
1895 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1896
1897 su2double *U_domain;
1898 auto *U_outlet = new su2double[nVar];
1899 su2double *V_domain, *V_outlet;
1900 auto *Normal = new su2double[nDim];
1901 auto *Ys = new su2double[nSpecies];
1902
1903 const unsigned short T_INDEX = nodes->GetTIndex();
1904 const unsigned short TVE_INDEX = nodes->GetTveIndex();
1905 const unsigned short VEL_INDEX = nodes->GetVelIndex();
1906 const unsigned short P_INDEX = nodes->GetPIndex();
1907 const unsigned short RHO_INDEX = nodes->GetRhoIndex();
1908 const unsigned short H_INDEX = nodes->GetHIndex();
1909 const unsigned short A_INDEX = nodes->GetAIndex();
1910 const unsigned short RHOCVTR_INDEX = nodes->GetRhoCvtrIndex();
1911 const unsigned short RHOCVVE_INDEX = nodes->GetRhoCvveIndex();
1912
1913 /*--- Loop over all the vertices on this boundary marker ---*/
1914 for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
1915 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
1916
1917 /*--- Allocate the value at the outlet ---*/
1918 V_outlet = GetCharacPrimVar(val_marker, iVertex);
1919
1920 /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
1921 if (geometry->nodes->GetDomain(iPoint)) {
1922
1923 /*--- Normal vector for this vertex (negate for outward convention) ---*/
1924 geometry->vertex[val_marker][iVertex]->GetNormal(Normal);
1925 for (auto iDim = 0ul; iDim < nDim; iDim++) Normal[iDim] = -Normal[iDim];
1926 conv_numerics->SetNormal(Normal);
1927
1928 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
1929
1930 for (auto iDim = 0ul; iDim < nDim; iDim++)
1931 UnitNormal[iDim] = Normal[iDim]/Area;
1932
1933 /*--- Current solution at this boundary node ---*/
1934 U_domain = nodes->GetSolution(iPoint);
1935 V_domain = nodes->GetPrimitive(iPoint);
1936
1937 /*--- Initialize solution at outlet ---*/
1938 for (auto iVar = 0ul; iVar < nVar; iVar++) U_outlet[iVar] = 0.0;
1939 for (auto iVar = 0ul; iVar < nPrimVar; iVar++) V_outlet[iVar] = 0.0;
1940
1941 /*--- Build the fictitious intlet state based on characteristics ---*/
1942
1943 /*--- Compute Gamma using domain state ---*/
1944 Gamma = nodes->GetGamma(iPoint);
1945 Gamma_Minus_One = Gamma - 1.0;
1946
1947 /*--- Retrieve the specified back pressure for this outlet. ---*/
1948 su2double P_Exit;
1949 if (gravity) P_Exit = config->GetOutlet_Pressure(Marker_Tag) - geometry->nodes->GetCoord(iPoint, nDim-1)*STANDARD_GRAVITY;
1950 else P_Exit = config->GetOutlet_Pressure(Marker_Tag);
1951
1952 /*--- Non-dim. the inputs if necessary. ---*/
1953 P_Exit = P_Exit/config->GetPressure_Ref();
1954
1955 /*--- Check whether the flow is supersonic at the exit. The type
1956 of boundary update depends on this. ---*/
1957 su2double Density = V_domain[RHO_INDEX];
1958 su2double Velocity2 = 0.0;
1959 su2double Vn = 0.0;
1960 su2double Velocity[MAXNDIM] = {0.0};
1961 for (auto iDim = 0ul; iDim < nDim; iDim++) {
1962 Velocity[iDim] = V_domain[VEL_INDEX+iDim];
1963 Velocity2 += Velocity[iDim]*Velocity[iDim];
1964 Vn += Velocity[iDim]*UnitNormal[iDim];
1965 }
1966 su2double Temperature = V_domain[T_INDEX];
1967 su2double Tve = V_domain[TVE_INDEX];
1968 su2double Pressure = V_domain[P_INDEX];
1969 su2double SoundSpeed = V_domain[A_INDEX];
1970 su2double Mach_Exit = sqrt(Velocity2)/SoundSpeed;
1971
1972 /*--- Compute Species Concentrations ---*/
1973 for (auto iSpecies = 0ul; iSpecies<nSpecies;iSpecies++){
1974 Ys[iSpecies] = V_domain[iSpecies]/Density;
1975 }
1976
1977 /*--- Recompute boundary state depending Mach number ---*/
1978 if (Mach_Exit >= 1.0) {
1979
1980 /*--- Supersonic exit flow: there are no incoming characteristics,
1981 so no boundary condition is necessary. Set outlet state to current
1982 state so that upwinding handles the direction of propagation. ---*/
1983 for (auto iVar = 0ul; iVar < nVar; iVar++) U_outlet[iVar] = U_domain[iVar];
1984 for (auto iVar = 0ul; iVar < nPrimVar; iVar++) V_outlet[iVar] = V_domain[iVar];
1985
1986 } else {
1987
1988 /*--- Subsonic exit flow: there is one incoming characteristic,
1989 therefore one variable can be specified (back pressure) and is used
1990 to update the conservative variables. Compute the entropy and the
1991 acoustic Riemann variable. These invariants, as well as the
1992 tangential velocity components, are extrapolated. The Temperatures
1993 (T and Tve) and species concentraition are also assumed to be extrapolated.
1994 ---*/
1995
1996 const su2double Entropy = Pressure*pow(1.0/Density,Gamma);
1997 const su2double Riemann = Vn + 2.0*SoundSpeed/Gamma_Minus_One;
1998
1999 /*--- Compute the new fictious state at the outlet ---*/
2000 // U: [rho1, ..., rhoNs, rhou, rhov, rhow, rhoe, rhoeve]^T
2001 // V: [rho1, ..., rhoNs, T, Tve, u, v, w, P, rho, h, a, rhoCvtr, rhoCvve]^T
2002 Density = pow(P_Exit/Entropy,1.0/Gamma);
2003 Pressure = P_Exit;
2004 SoundSpeed = sqrt(Gamma*P_Exit/Density);
2005 const su2double Vn_Exit = Riemann - 2.0*SoundSpeed/Gamma_Minus_One;
2006 Velocity2 = 0.0;
2007 for (auto iDim = 0ul; iDim < nDim; iDim++) {
2008 Velocity[iDim] = Velocity[iDim] + (Vn_Exit-Vn)*UnitNormal[iDim];
2009 Velocity2 += Velocity[iDim]*Velocity[iDim];
2010 }
2011
2012 /*--- Primitive variables, using the derived quantities ---*/
2013 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies ++) {
2014 V_outlet[iSpecies] = Ys[iSpecies]*Density;
2015 rhos[iSpecies] = V_outlet[iSpecies];
2016 }
2017
2018 V_outlet[T_INDEX] = V_domain[T_INDEX];
2019 V_outlet[TVE_INDEX] = V_domain[TVE_INDEX];
2020
2021 for (auto iDim = 0ul; iDim < nDim; iDim++){
2022 V_outlet[VEL_INDEX+iDim] = Velocity[iDim];
2023 }
2024
2025 V_outlet[P_INDEX] = Pressure;
2026 V_outlet[RHO_INDEX] = Density;
2027 V_outlet[A_INDEX] = SoundSpeed;
2028
2029 /*--- Set mixture state and compute quantities ---*/
2030 FluidModel->SetTDStateRhosTTv(rhos, Temperature, Tve);
2031 V_outlet[RHOCVTR_INDEX] = FluidModel->ComputerhoCvtr();
2032 V_outlet[RHOCVVE_INDEX] = FluidModel->ComputerhoCvve();
2033
2034 const auto& energies = FluidModel->ComputeMixtureEnergies();
2035
2036 /*--- Conservative variables, using the derived quantities ---*/
2037 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies ++){
2038 U_outlet[iSpecies] = V_outlet[iSpecies];
2039 }
2040
2041 for (auto iDim = 0ul; iDim < nDim; iDim++)
2042 U_outlet[nSpecies+iDim] = Velocity[iDim]*Density;
2043
2044 U_outlet[nVar-2] = (energies[0] + 0.5*Velocity2) * Density;
2045 U_outlet[nVar-1] = energies[1] * Density;
2046
2047 }
2048
2049 /*--- Setting Last remaining variables ---*/
2050 V_outlet[H_INDEX]= (U_outlet[nVar-2]+Pressure)/Density;
2051
2052 /*--- Set various quantities in the solver class ---*/
2053 conv_numerics->SetConservative(U_domain, U_outlet);
2054 conv_numerics->SetPrimitive(V_domain,V_outlet);
2055
2056 if (dynamic_grid)
2057 conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
2058
2059 /*--- Passing supplementary information to CNumerics ---*/
