37void CIntegration::Space_Integration(CGeometry *geometry,
38 CSolver **solver_container,
39 CNumerics **numerics,
40 CConfig *config, unsigned short iMesh,
41 unsigned short iRKStep,
42 unsigned short RunTime_EqSystem) {
43 unsigned short iMarker, KindBC;
44
45 unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
46 bool dual_time = ((config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
47 (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND));
48
49 /*--- Compute inviscid residuals ---*/
50
51 switch (config->GetKind_ConvNumScheme()) {
52 case SPACE_CENTERED:
53 solver_container[MainSolver]->Centered_Residual(geometry, solver_container, numerics, config, iMesh, iRKStep);
54 break;
55 case SPACE_UPWIND:
56 solver_container[MainSolver]->Upwind_Residual(geometry, solver_container, numerics, config, iMesh);
57 break;
58 }
59
60 /*--- Compute viscous residuals ---*/
61 solver_container[MainSolver]->Viscous_Residual(geometry, solver_container, numerics, config, iMesh, iRKStep);
62
63 /*--- Compute source term residuals ---*/
64 solver_container[MainSolver]->Source_Residual(geometry, solver_container, numerics, config, iMesh);
65
66 /*--- Add viscous and convective residuals, and compute the Dual Time Source term ---*/
67
68 if (dual_time)
69 solver_container[MainSolver]->SetResidual_DualTime(geometry, solver_container, config, iRKStep, iMesh, RunTime_EqSystem);
70
71 /*--- Pick convective and viscous numerics objects for the current thread. ---*/
72
73 CNumerics* conv_bound_numerics = numerics[CONV_BOUND_TERM + omp_get_thread_num()*MAX_TERMS];
74 CNumerics* visc_bound_numerics = numerics[VISC_BOUND_TERM + omp_get_thread_num()*MAX_TERMS];
75
76 /*--- Pause preaccumulation in boundary conditions for hybrid parallel AD. ---*/
77 /// TODO: Check if this is really needed.
78 //const auto pausePreacc = (omp_get_num_threads() > 1) && AD::PausePreaccumulation();
79
80 /*--- Boundary conditions that depend on other boundaries (they require MPI sincronization)---*/
81
82 solver_container[MainSolver]->BC_Fluid_Interface(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config);
83
84 /*--- Compute Fourier Transformations for markers where NRBC_BOUNDARY is applied---*/
85
86 if (config->GetBoolGiles() && config->GetSpatialFourier()){
87 solver_container[MainSolver]->PreprocessBC_Giles(geometry, config, conv_bound_numerics, INFLOW);
88
89 solver_container[MainSolver]->PreprocessBC_Giles(geometry, config, conv_bound_numerics, OUTFLOW);
90 }
91
92 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
93 if (config->GetBoolTurbomachinery()){
94 /*--- Average quantities at the inflow and outflow boundaries ---*/
95 solver_container[MainSolver]->TurboAverageProcess(solver_container, geometry,config,INFLOW);
96 solver_container[MainSolver]->TurboAverageProcess(solver_container, geometry, config, OUTFLOW);
97 }
98 }
99 END_SU2_OMP_SAFE_GLOBAL_ACCESS
100
101 /*--- Weak boundary conditions ---*/
102
103 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
104 KindBC = config->GetMarker_All_KindBC(iMarker);
105 switch (KindBC) {
106 case EULER_WALL:
107 solver_container[MainSolver]->BC_Euler_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
108 break;
109 case ACTDISK_INLET:
110 solver_container[MainSolver]->BC_ActDisk_Inlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
111 break;
112 case ENGINE_INFLOW:
113 solver_container[MainSolver]->BC_Engine_Inflow(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
114 break;
115 case INLET_FLOW:
116 solver_container[MainSolver]->BC_Inlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
117 break;
118 case ACTDISK_OUTLET:
119 solver_container[MainSolver]->BC_ActDisk_Outlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
120 break;
121 case ENGINE_EXHAUST:
122 solver_container[MainSolver]->BC_Engine_Exhaust(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
123 break;
124 case SUPERSONIC_INLET:
125 solver_container[MainSolver]->BC_Supersonic_Inlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
126 break;
127 case OUTLET_FLOW:
128 solver_container[MainSolver]->BC_Outlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
129 break;
130 case SUPERSONIC_OUTLET:
131 solver_container[MainSolver]->BC_Supersonic_Outlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
132 break;
133 case GILES_BOUNDARY:
134 solver_container[MainSolver]->BC_Giles(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
135 break;
136 case RIEMANN_BOUNDARY:
137 if (config->GetBoolTurbomachinery()){
138 solver_container[MainSolver]->BC_TurboRiemann(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
139 }
140 else{
141 solver_container[MainSolver]->BC_Riemann(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
142 }
143 break;
144 case FAR_FIELD:
145 solver_container[MainSolver]->BC_Far_Field(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
146 break;
147 case SYMMETRY_PLANE:
148 solver_container[MainSolver]->BC_Sym_Plane(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
149 break;
150 }
151 }
152
153 /*--- Modification of the system on the whole domain, like for a strong BC. ---*/
154 solver_container[MainSolver]->Impose_Fixed_Values(geometry, config);
155
156 /*--- Strong boundary conditions (Navier-Stokes and Dirichlet type BCs) ---*/
157
158 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++)
159 switch (config->GetMarker_All_KindBC(iMarker)) {
160 case ISOTHERMAL:
161 solver_container[MainSolver]->BC_Isothermal_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
162 break;
163 case HEAT_FLUX:
164 solver_container[MainSolver]->BC_HeatFlux_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
165 break;
166 case HEAT_TRANSFER:
167 solver_container[MainSolver]->BC_HeatTransfer_Wall(geometry, config, iMarker);
168 break;
169 case CUSTOM_BOUNDARY:
170 solver_container[MainSolver]->BC_Custom(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
171 break;
172 case CHT_WALL_INTERFACE:
173
174 if ( MainSolver == FLOW_SOL && (config->GetKind_FluidModel() == FLUID_FLAMELET) ){
175 solver_container[MainSolver]->BC_Isothermal_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
176 break;
177 }
178 if ((MainSolver == SPECIES_SOL) || (MainSolver == HEAT_SOL) || ((MainSolver == FLOW_SOL) && ((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) || config->GetEnergy_Equation()))) {
179 solver_container[MainSolver]->BC_ConjugateHeat_Interface(geometry, solver_container, conv_bound_numerics, config, iMarker);
180 }
181 else {
182 solver_container[MainSolver]->BC_HeatFlux_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
183 }
184 break;
185 case SMOLUCHOWSKI_MAXWELL:
186 solver_container[MainSolver]->BC_Smoluchowski_Maxwell(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
187 break;
188 }
189
190 /*--- Complete residuals for periodic boundary conditions. We loop over
191 the periodic BCs in matching pairs so that, in the event that there are
192 adjacent periodic markers, the repeated points will have their residuals
193 accumulated correctly during the communications. ---*/
194
195 if (config->GetnMarker_Periodic() > 0) {
196 solver_container[MainSolver]->BC_Periodic(geometry, solver_container, conv_bound_numerics, config);
197 }
198
199 //AD::ResumePreaccumulation(pausePreacc);
200
201}

203void CIntegration::Time_Integration(CGeometry *geometry, CSolver **solver_container, CConfig *config,
204 unsigned short iRKStep, unsigned short RunTime_EqSystem) {
205
206 unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
207
208 switch (config->GetKind_TimeIntScheme()) {
209 case (RUNGE_KUTTA_EXPLICIT):
210 solver_container[MainSolver]->ExplicitRK_Iteration(geometry, solver_container, config, iRKStep);
211 break;
212 case (CLASSICAL_RK4_EXPLICIT):
213 solver_container[MainSolver]->ClassicalRK4_Iteration(geometry, solver_container, config, iRKStep);
214 break;
215 case (EULER_EXPLICIT):
216 solver_container[MainSolver]->ExplicitEuler_Iteration(geometry, solver_container, config);
217 break;
218 case (EULER_IMPLICIT):
219 solver_container[MainSolver]->ImplicitEuler_Iteration(geometry, solver_container, config);
220 break;
221 }

244void CIntegration::SetDualTime_Solver(const CGeometry *geometry, CSolver *solver, const CConfig *config, unsigned short iMesh) {
245
246 SU2_OMP_PARALLEL
247 {
248 /*--- Store old solution ---*/
249 solver->GetNodes()->Set_Solution_time_n1();
250 solver->GetNodes()->Set_Solution_time_n();
251
252 SU2_OMP_SAFE_GLOBAL_ACCESS(solver->ResetCFLAdapt();)
253
254 SU2_OMP_FOR_STAT(roundUpDiv(geometry->GetnPoint(), omp_get_num_threads()))
255 for (auto iPoint = 0ul; iPoint < geometry->GetnPoint(); iPoint++) {
256
257 /*--- Initialize the underrelaxation ---*/
258 solver->GetNodes()->SetUnderRelaxation(iPoint, 1.0);
259
260 /*--- Initialize the local CFL number ---*/
261 solver->GetNodes()->SetLocalCFL(iPoint, config->GetCFL(iMesh));
262 }
263 END_SU2_OMP_FOR
264
265 }
266 END_SU2_OMP_PARALLEL
