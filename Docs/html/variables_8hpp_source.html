86 /*!
87  * \brief Primitive to conservative conversion.
88  */
89 template<size_t nDim, size_t N>
90 FORCEINLINE CCompressibleConservatives<nDim> compressibleConservatives(const CCompressiblePrimitives<nDim,N>& V) {
91  CCompressibleConservatives<nDim> U;
92  U.density() = V.density();
93  for (size_t iDim = 0; iDim < nDim; ++iDim) {
94  U.momentum(iDim) = V.density() * V.velocity(iDim);
95  }
96  U.rhoEnergy() = V.density() * V.enthalpy() - V.pressure();
97  return U;
98 }

112 /*!
113  * \brief Compute Roe-averaged variables from pair of primitive variables.
114  */
115 template<size_t nDim, class PrimVarType>
116 FORCEINLINE CRoeVariables<nDim> roeAveragedVariables(Double gamma,
117  const CPair<PrimVarType>& V,
118  const VectorDbl<nDim>& normal) {
119  CRoeVariables<nDim> roeAvg;
120  Double R = sqrt(V.j.density() / V.i.density());
121  Double D = 1 / (R+1);
122  roeAvg.density = R * V.i.density();
123  for (size_t iDim = 0; iDim < nDim; ++iDim) {
124  roeAvg.velocity(iDim) = (R*V.j.velocity(iDim) + V.i.velocity(iDim)) * D;
125  }
126  roeAvg.enthalpy = (R*V.j.enthalpy() + V.i.enthalpy()) * D;
127  roeAvg.speedSound = sqrt((gamma-1) * (roeAvg.enthalpy - 0.5*squaredNorm(roeAvg.velocity)));
128  roeAvg.projVel = dot(roeAvg.velocity, normal);
129  return roeAvg;
130 }
