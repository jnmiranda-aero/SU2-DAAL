66/*!
67 * \brief Type to store compressible conservative (i.e. solution) variables.
68 */
69template<size_t nDim_>
70struct CCompressibleConservatives {
71 static constexpr size_t nDim = nDim_;
72 static constexpr size_t nVar = nDim+2;
73 VectorDbl<nVar> all;
74
75 FORCEINLINE Double& density() { return all(0); }
76 FORCEINLINE Double& rhoEnergy() { return all(nDim+1); }
77 FORCEINLINE Double& momentum(size_t iDim) { return all(iDim+1); }
78 FORCEINLINE const Double& density() const { return all(0); }
79 FORCEINLINE const Double& rhoEnergy() const { return all(nDim+1); }
80 FORCEINLINE const Double& momentum(size_t iDim) const { return all(iDim+1); }
81
82 FORCEINLINE Double energy() const { return rhoEnergy() / density(); }
83 FORCEINLINE const Double* momentum() const { return &momentum(0); }
84};

86/*!
87 * \brief Primitive to conservative conversion.
88 */
89template<size_t nDim, size_t N>
90FORCEINLINE CCompressibleConservatives<nDim> compressibleConservatives(const CCompressiblePrimitives<nDim,N>& V) {
91 CCompressibleConservatives<nDim> U;
92 U.density() = V.density();
93 for (size_t iDim = 0; iDim < nDim; ++iDim) {
94 U.momentum(iDim) = V.density() * V.velocity(iDim);
95 }
96 U.rhoEnergy() = V.density() * V.enthalpy() - V.pressure();
97 return U;
98}

103template<size_t nDim>

112/*!
113 * \brief Compute Roe-averaged variables from pair of primitive variables.
114 */
115template<size_t nDim, class PrimVarType>
116FORCEINLINE CRoeVariables<nDim> roeAveragedVariables(Double gamma,
117 const CPair<PrimVarType>& V,
118 const VectorDbl<nDim>& normal) {
119 CRoeVariables<nDim> roeAvg;
120 Double R = sqrt(V.j.density() / V.i.density());
121 Double D = 1 / (R+1);
122 roeAvg.density = R * V.i.density();
123 for (size_t iDim = 0; iDim < nDim; ++iDim) {
124 roeAvg.velocity(iDim) = (R*V.j.velocity(iDim) + V.i.velocity(iDim)) * D;
125 }
126 roeAvg.enthalpy = (R*V.j.enthalpy() + V.i.enthalpy()) * D;
127 roeAvg.speedSound = sqrt((gamma-1) * (roeAvg.enthalpy - 0.5*squaredNorm(roeAvg.velocity)));
128 roeAvg.projVel = dot(roeAvg.velocity, normal);
129 return roeAvg;
130}
