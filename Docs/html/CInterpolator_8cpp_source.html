45bool CInterpolator::CheckInterfaceBoundary(int markDonor, int markTarget) {
46 /*--- Determine whether the boundary is not on the rank because of
47 * the partition or because it is not part of the zone. ---*/
48 int donorCheck = -1, targetCheck = -1;
49 SU2_MPI::Allreduce(&markDonor, &donorCheck, 1, MPI_INT, MPI_MAX, SU2_MPI::GetComm());
50 SU2_MPI::Allreduce(&markTarget, &targetCheck, 1, MPI_INT, MPI_MAX, SU2_MPI::GetComm());
51 return (donorCheck != -1) && (targetCheck != -1);
52}

54bool CInterpolator::CheckZonesInterface(const CConfig* donor, const CConfig* target) {
55 /*--- Loop over all interface markers to find if the 2 zones share any interface boundary. ---*/
56 for (auto iInter = 0; iInter < (donor->GetMarker_n_ZoneInterface() / 2); iInter++) {
57 if (CheckInterfaceBoundary(donor->FindInterfaceMarker(iInter), target->FindInterfaceMarker(iInter))) return true;
58 }
59 return false;
60}

62void CInterpolator::Determine_ArraySize(int markDonor, int markTarget, unsigned long nVertexDonor,
63 unsigned short nDim) {
64 /*--- Count donor vertices. ---*/
65 auto nLocalVertex_Donor = 0ul;
66 for (auto iVertex = 0ul; iVertex < nVertexDonor; iVertex++) {
67 auto iPointDonor = donor_geometry->vertex[markDonor][iVertex]->GetNode();
68 nLocalVertex_Donor += donor_geometry->nodes->GetDomain(iPointDonor);
69 }
70
71 Buffer_Send_nVertex_Donor[0] = nLocalVertex_Donor;
72
73 /*--- Send Interface vertex information --*/
74 SU2_MPI::Allreduce(&nLocalVertex_Donor, &MaxLocalVertex_Donor, 1, MPI_UNSIGNED_LONG, MPI_MAX, SU2_MPI::GetComm());
75 SU2_MPI::Allgather(Buffer_Send_nVertex_Donor, 1, MPI_UNSIGNED_LONG, Buffer_Receive_nVertex_Donor, 1,
76 MPI_UNSIGNED_LONG, SU2_MPI::GetComm());

79void CInterpolator::Collect_VertexInfo(int markDonor, int markTarget, unsigned long nVertexDonor, unsigned short nDim) {
80 Buffer_Send_Coord.resize(MaxLocalVertex_Donor, nDim);
81 Buffer_Send_GlobalPoint.resize(MaxLocalVertex_Donor) = -1;
82
83 /*--- Copy coordinates and point to the auxiliar vector --*/
84 unsigned long iLocalVertexDonor = 0;
85
86 for (unsigned long iVertex = 0; iVertex < nVertexDonor; iVertex++) {
87 auto iPointDonor = donor_geometry->vertex[markDonor][iVertex]->GetNode();
88 if (donor_geometry->nodes->GetDomain(iPointDonor)) {
89 Buffer_Send_GlobalPoint[iLocalVertexDonor] = donor_geometry->nodes->GetGlobalIndex(iPointDonor);
90 for (unsigned short iDim = 0; iDim < nDim; iDim++)
91 Buffer_Send_Coord(iLocalVertexDonor, iDim) = donor_geometry->nodes->GetCoord(iPointDonor, iDim);
92 iLocalVertexDonor++;
93 }
94 }
95
96 SU2_MPI::Allgather(Buffer_Send_Coord.data(), MaxLocalVertex_Donor * nDim, MPI_DOUBLE, Buffer_Receive_Coord.data(),
97 MaxLocalVertex_Donor * nDim, MPI_DOUBLE, SU2_MPI::GetComm());
98 SU2_MPI::Allgather(Buffer_Send_GlobalPoint.data(), MaxLocalVertex_Donor, MPI_UNSIGNED_LONG,
99 Buffer_Receive_GlobalPoint.data(), MaxLocalVertex_Donor, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
100}

102unsigned long CInterpolator::Collect_ElementInfo(int markDonor, unsigned short nDim, bool compress,
103 vector<unsigned long>& allNumElem, vector<unsigned short>& numNodes,
104 su2matrix<long>& idxNodes) const {
105 const auto maxElemNodes = (nDim == 2u) ? 2u : 4u; // line and quad respectively
106
107 unsigned long nElemDonor = 0;
108 if (markDonor != -1) nElemDonor = donor_geometry->GetnElem_Bound(markDonor);
109
110 allNumElem.resize(size);
111 SU2_MPI::Allgather(&nElemDonor, 1, MPI_UNSIGNED_LONG, allNumElem.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
112
113 auto nMaxElemDonor = *max_element(allNumElem.begin(), allNumElem.end());
114
115 vector<unsigned short> bufferSendNum(nMaxElemDonor);
116 su2matrix<long> bufferSendIdx(nMaxElemDonor, maxElemNodes);
117
118 numNodes.resize(nMaxElemDonor * size);
119 idxNodes.resize(nMaxElemDonor * size, maxElemNodes);
120
121 for (auto iElem = 0ul; iElem < nElemDonor; ++iElem) {
122 const auto nNode = donor_geometry->bound[markDonor][iElem]->GetnNodes();
123 bufferSendNum[iElem] = nNode;
124 assert(nNode <= maxElemNodes && "Donor element has too many nodes.");
125
126 for (auto iNode = 0u; iNode < nNode; ++iNode) {
127 auto iPoint = donor_geometry->bound[markDonor][iElem]->GetNode(iNode);
128 auto iPointGlobal = donor_geometry->nodes->GetGlobalIndex(iPoint);
129 bufferSendIdx(iElem, iNode) = iPointGlobal;
130 }
131 }
132
133 SU2_MPI::Allgather(bufferSendNum.data(), bufferSendNum.size(), MPI_UNSIGNED_SHORT, numNodes.data(),
134 bufferSendNum.size(), MPI_UNSIGNED_SHORT, SU2_MPI::GetComm());
135 SU2_MPI::Allgather(bufferSendIdx.data(), bufferSendIdx.size(), MPI_LONG, idxNodes.data(), bufferSendIdx.size(),
136 MPI_LONG, SU2_MPI::GetComm());
137
138 if (!compress) return accumulate(allNumElem.begin(), allNumElem.end(), 0ul);
139
140 /*--- Compress the information (overlapping copy do not use memcpy). ---*/
141
142 unsigned long dstIdx = 0;
143 for (int iProcessor = 0; iProcessor < size; ++iProcessor) {
144 auto srcOffset = iProcessor * nMaxElemDonor;
145 for (auto idx = 0u; idx < allNumElem[iProcessor]; ++idx) {
146 numNodes[dstIdx] = numNodes[srcOffset + idx];
147 for (auto iNode = 0u; iNode < maxElemNodes; ++iNode) idxNodes(dstIdx, iNode) = idxNodes(srcOffset + idx, iNode);
148 ++dstIdx;
149 }
150 }
151
152 return dstIdx;
153}

155void CInterpolator::ReconstructBoundary(unsigned long val_zone, int val_marker) {
156 const CGeometry* geom = Geometry[val_zone][INST_0][MESH_0];
157 const auto nDim = geom->GetnDim();
158
159 /*--- If this zone has no parts of the marker, it will not participate
160 * in the first part of this function (collection). ---*/
161 unsigned long nVertex, nElems;
162 if (val_marker != -1) {
163 nVertex = geom->GetnVertex(val_marker);
164 nElems = geom->GetnElem_Bound(val_marker);
165 } else {
166 nVertex = 0;
167 nElems = 0;
168 }
169
170 /*--- Get the number of domain vertices on the marker, and a mapping
171 * (iVertex) -> (iLocalVertex, non-domain points being ignored). ---*/
172 su2vector<unsigned long> iVertex_to_iLocalVertex(nVertex);
173 unsigned long nLocalVertex = 0;
174 for (unsigned long iVertex = 0; iVertex < nVertex; iVertex++) {
175 const auto iPoint = geom->vertex[val_marker][iVertex]->GetNode();
176 if (geom->nodes->GetDomain(iPoint)) {
177 iVertex_to_iLocalVertex[iVertex] = nLocalVertex;
178 nLocalVertex++;
179 } else {
180 iVertex_to_iLocalVertex[iVertex] = numeric_limits<unsigned long>::max();
181 }
182 }
183
184 // coordinates of all domain vertices on the marker
185 su2activematrix Buffer_Send_Coord(nLocalVertex, nDim);
186 // global point IDs of all domain vertices on the marker
187 su2vector<unsigned long> Buffer_Send_GlobalPoint(nVertex);
188
189 /*--- Assign to each domain vertex on the marker, identified by local point ID,
190 * a set of surface-neighbor vertices on the marker, identified by global point ID. ---*/
191 unordered_map<unsigned long, set<unsigned long> > neighbors;
192
193 /*--- Define or initialize them. ---*/
194 for (unsigned long iVertex = 0; iVertex < nVertex; iVertex++) {
195 const auto iPoint = geom->vertex[val_marker][iVertex]->GetNode();
196 if (geom->nodes->GetDomain(iPoint)) {
197 const auto iLocalVertex = iVertex_to_iLocalVertex[iVertex];
198 Buffer_Send_GlobalPoint[iLocalVertex] = geom->nodes->GetGlobalIndex(iPoint);
199 for (unsigned long iDim = 0; iDim < nDim; iDim++)
200 Buffer_Send_Coord(iLocalVertex, iDim) = geom->nodes->GetCoord(iPoint, iDim);
201 neighbors.insert(pair<unsigned long, set<unsigned long> >(iPoint, set<unsigned long>()));
202 }
203 }
204
205 /*--- Define the neighbors map. ---*/
206 for (unsigned long iElem = 0; iElem < nElems; iElem++) {
207 const CPrimalGrid* elem = geom->bound[val_marker][iElem];
208 for (unsigned short iNode = 0; iNode < elem->GetnNodes(); iNode++) {
209 const auto iPoint = elem->GetNode(iNode);
210 if (geom->nodes->GetDomain(iPoint)) {
211 set<unsigned long>& neighb = neighbors.at(iPoint);
212 for (unsigned short iNeighbor = 0; iNeighbor < elem->GetnNeighbor_Nodes(iNode); iNeighbor++) {
213 unsigned long jPoint = elem->GetNode(elem->GetNeighbor_Nodes(iNode, iNeighbor));
214 unsigned long jPoint_global = geom->nodes->GetGlobalIndex(jPoint);
215 neighb.insert(jPoint_global);
216 }
217 }
218 }
219 }
220
221 // numbers of surface-neighbors of all domain vertices on the marker
222 su2vector<unsigned long> Buffer_Send_nLinkedNodes(nLocalVertex);
223 // cumsum of Buffer_Send_nLinkedNodes
224 su2vector<unsigned long> Buffer_Send_StartLinkedNodes(nLocalVertex);
225 unsigned long nLocalLinkedNodes = 0;
226 for (unsigned long iVertex = 0; iVertex < nVertex; iVertex++) {
227 const auto iPoint = geom->vertex[val_marker][iVertex]->GetNode();
228 if (geom->nodes->GetDomain(iPoint)) {
229 const auto iLocalVertex = iVertex_to_iLocalVertex[iVertex];
230 Buffer_Send_nLinkedNodes[iLocalVertex] = neighbors.at(iPoint).size();
231 Buffer_Send_StartLinkedNodes[iLocalVertex] = nLocalLinkedNodes;
232 nLocalLinkedNodes += Buffer_Send_nLinkedNodes[iLocalVertex];
233 }
234 }
235 // global point IDs of surface-neighbors of all domain vertices on the marker
236 su2vector<unsigned long> Buffer_Send_LinkedNodes(nLocalLinkedNodes);
237 unsigned long index = 0;
238 for (unsigned long iVertex = 0; iVertex < nVertex; iVertex++) {
239 const auto iPoint = geom->vertex[val_marker][iVertex]->GetNode();
240 if (geom->nodes->GetDomain(iPoint)) {
241 for (unsigned long jPoint_global : neighbors.at(iPoint)) {
242 Buffer_Send_LinkedNodes[index] = jPoint_global;
243 index++;
244 }
245 }
246 }
247
248 /*--- Now these arrays of all processes must be joined to a single/global arrays. For this,
249 * the entries of StartLinkedNodes must be shifted.
250 * Furthermore, the global point IDs in LinkedNodes are replaced by global vertex IDs.
251 * For this, the master process collects the data from all processes, joins them and broadcasts them again. ---*/
252
253 /*--- Allocate global arrays. ---*/
254 unsigned long nGlobalLinkedNodes;
255 SU2_MPI::Allreduce(&nLocalVertex, &nGlobalVertex, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
256 SU2_MPI::Allreduce(&nLocalLinkedNodes, &nGlobalLinkedNodes, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
257
258 Buffer_Receive_Coord.resize(nGlobalVertex, nDim);
259 Buffer_Receive_GlobalPoint.resize(nGlobalVertex);
260 Buffer_Receive_Proc.resize(nGlobalVertex);
261
262 Buffer_Receive_nLinkedNodes.resize(nGlobalVertex);
263 Buffer_Receive_LinkedNodes.resize(nGlobalLinkedNodes);
264 Buffer_Receive_StartLinkedNodes.resize(nGlobalVertex);
265
266 /*--- Master process gathers data from all ranks ---*/
267#ifdef HAVE_MPI
268 int nProcessor = size, iRank;
269 unsigned long received_nLocalLinkedNodes, received_nLocalVertex;
270
271 if (rank == MASTER_NODE) {
272 /*--- "Receive" from master process, i.e. copy. ---*/
273 for (unsigned long iVertex = 0; iVertex < nLocalVertex; iVertex++)
274 for (unsigned long iDim = 0; iDim < nDim; iDim++)
275 Buffer_Receive_Coord(iVertex, iDim) = Buffer_Send_Coord(iVertex, iDim);
276
277 for (unsigned long iVertex = 0; iVertex < nLocalVertex; iVertex++) {
278 Buffer_Receive_GlobalPoint[iVertex] = Buffer_Send_GlobalPoint[iVertex];
279 Buffer_Receive_Proc[iVertex] = MASTER_NODE;
280 Buffer_Receive_nLinkedNodes[iVertex] = Buffer_Send_nLinkedNodes[iVertex];
281 Buffer_Receive_StartLinkedNodes[iVertex] = Buffer_Send_StartLinkedNodes[iVertex];
282 }
283
284 for (unsigned long iVertex = 0; iVertex < nLocalLinkedNodes; iVertex++)
285 Buffer_Receive_LinkedNodes[iVertex] = Buffer_Send_LinkedNodes[iVertex];
286
287 unsigned long received_nLocalVertex_sum = nLocalVertex;
288 unsigned long received_nLocalLinkedNodes_sum = nLocalLinkedNodes;
289
290 /*--- Receive from other processes to the appropriate position in the buffers and shift StartLinkedNodes indices.
291 * ---*/
292 for (iRank = 1; iRank < nProcessor; iRank++) {
293 SU2_MPI::Recv(&received_nLocalLinkedNodes, 1, MPI_UNSIGNED_LONG, iRank, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
294 SU2_MPI::Recv(&Buffer_Receive_LinkedNodes[received_nLocalLinkedNodes_sum], received_nLocalLinkedNodes,
295 MPI_UNSIGNED_LONG, iRank, 1, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
296
297 SU2_MPI::Recv(&received_nLocalVertex, 1, MPI_UNSIGNED_LONG, iRank, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
298 SU2_MPI::Recv(Buffer_Receive_Coord[received_nLocalVertex_sum], nDim * received_nLocalVertex, MPI_DOUBLE, iRank, 1,
299 SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
300
301 SU2_MPI::Recv(&Buffer_Receive_GlobalPoint[received_nLocalVertex_sum], received_nLocalVertex, MPI_UNSIGNED_LONG,
302 iRank, 1, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
303 SU2_MPI::Recv(&Buffer_Receive_nLinkedNodes[received_nLocalVertex_sum], received_nLocalVertex, MPI_UNSIGNED_LONG,
304 iRank, 1, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
305 SU2_MPI::Recv(&Buffer_Receive_StartLinkedNodes[received_nLocalVertex_sum], received_nLocalVertex,
306 MPI_UNSIGNED_LONG, iRank, 1, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
307
308 for (unsigned long iVertex = 0; iVertex < received_nLocalVertex; iVertex++) {
309 Buffer_Receive_Proc[received_nLocalVertex_sum + iVertex] = iRank;
310 Buffer_Receive_StartLinkedNodes[received_nLocalVertex_sum + iVertex] += received_nLocalLinkedNodes_sum;
311 }
312
313 received_nLocalVertex_sum += received_nLocalVertex;
314 received_nLocalLinkedNodes_sum += received_nLocalLinkedNodes;
315 }
316 } else {
317 SU2_MPI::Send(&nLocalLinkedNodes, 1, MPI_UNSIGNED_LONG, 0, 0, SU2_MPI::GetComm());
318 SU2_MPI::Send(Buffer_Send_LinkedNodes.data(), nLocalLinkedNodes, MPI_UNSIGNED_LONG, 0, 1, SU2_MPI::GetComm());
319
320 SU2_MPI::Send(&nLocalVertex, 1, MPI_UNSIGNED_LONG, 0, 0, SU2_MPI::GetComm());
321 SU2_MPI::Send(Buffer_Send_Coord.data(), Buffer_Send_Coord.size(), MPI_DOUBLE, 0, 1, SU2_MPI::GetComm());
322
323 SU2_MPI::Send(Buffer_Send_GlobalPoint.data(), nLocalVertex, MPI_UNSIGNED_LONG, 0, 1, SU2_MPI::GetComm());
324 SU2_MPI::Send(Buffer_Send_nLinkedNodes.data(), nLocalVertex, MPI_UNSIGNED_LONG, 0, 1, SU2_MPI::GetComm());
325 SU2_MPI::Send(Buffer_Send_StartLinkedNodes.data(), nLocalVertex, MPI_UNSIGNED_LONG, 0, 1, SU2_MPI::GetComm());
326 }
327#else
328 for (unsigned long iVertex = 0; iVertex < nLocalVertex; iVertex++)
329 for (unsigned long iDim = 0; iDim < nDim; iDim++)
330 Buffer_Receive_Coord(iVertex, iDim) = Buffer_Send_Coord(iVertex, iDim);
331
332 for (unsigned long iVertex = 0; iVertex < nGlobalVertex; iVertex++) {
333 Buffer_Receive_GlobalPoint[iVertex] = Buffer_Send_GlobalPoint[iVertex];
334 Buffer_Receive_Proc[iVertex] = MASTER_NODE;
335 Buffer_Receive_nLinkedNodes[iVertex] = Buffer_Send_nLinkedNodes[iVertex];
336 Buffer_Receive_StartLinkedNodes[iVertex] = Buffer_Send_StartLinkedNodes[iVertex];
337 }
338
339 for (unsigned long iVertex = 0; iVertex < nGlobalLinkedNodes; iVertex++)
340 Buffer_Receive_LinkedNodes[iVertex] = Buffer_Send_LinkedNodes[iVertex];
341#endif
342
343 /*--- Master process replaces global point indices in Buffer_Receive_LinkedNodes by their indices in
344 * Buffer_Receive_GlobalPoint, Buffer_Receive_nLinkedNodes etc. ---*/
345 if (rank == MASTER_NODE) {
346 for (unsigned long iVertex = 0; iVertex < nGlobalVertex; iVertex++) {
347 unsigned long* uptr = &Buffer_Receive_LinkedNodes[Buffer_Receive_StartLinkedNodes[iVertex]];
348
349 for (unsigned long jLinkedNode = 0; jLinkedNode < Buffer_Receive_nLinkedNodes[iVertex]; jLinkedNode++) {
350 const auto jPoint = uptr[jLinkedNode];
351 bool found = false; // Global point index has been found
352 for (unsigned long kVertex = 0; kVertex < nGlobalVertex; kVertex++) {
353 if (Buffer_Receive_GlobalPoint[kVertex] == jPoint) {
354 uptr[jLinkedNode] = kVertex;
355 found = true;
356 break;
357 }
358 }
359
360 if (!found) { // remove from list
361 for (unsigned long kVertex = jLinkedNode; kVertex < Buffer_Receive_nLinkedNodes[iVertex] - 1; kVertex++) {
362 uptr[kVertex] = uptr[kVertex + 1];
363 }
364 Buffer_Receive_nLinkedNodes[iVertex]--;
365 jLinkedNode--;
366 }
367 }
368 }
369 }
370
371 SU2_MPI::Bcast(Buffer_Receive_GlobalPoint.data(), nGlobalVertex, MPI_UNSIGNED_LONG, 0, SU2_MPI::GetComm());
372 SU2_MPI::Bcast(Buffer_Receive_Coord.data(), Buffer_Receive_Coord.size(), MPI_DOUBLE, 0, SU2_MPI::GetComm());
373 SU2_MPI::Bcast(Buffer_Receive_Proc.data(), nGlobalVertex, MPI_UNSIGNED_LONG, 0, SU2_MPI::GetComm());
374
375 SU2_MPI::Bcast(Buffer_Receive_nLinkedNodes.data(), nGlobalVertex, MPI_UNSIGNED_LONG, 0, SU2_MPI::GetComm());
376 SU2_MPI::Bcast(Buffer_Receive_StartLinkedNodes.data(), nGlobalVertex, MPI_UNSIGNED_LONG, 0, SU2_MPI::GetComm());
377 SU2_MPI::Bcast(Buffer_Receive_LinkedNodes.data(), nGlobalLinkedNodes, MPI_UNSIGNED_LONG, 0, SU2_MPI::GetComm());
378}
