105CNumerics::ResidualType<> CAvgGrad_NEMO::ComputeResidual(const CConfig *config) {
106
107 /*--- Normalized normal vector ---*/
108 Area = GeometryToolbox::Norm(nDim, Normal);
109
110 for (auto iDim = 0ul; iDim < nDim; iDim++)
111 UnitNormal[iDim] = Normal[iDim]/Area;
112
113 /*--- Mean transport coefficients ---*/
114 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
115 Mean_Diffusion_Coeff[iSpecies] = 0.5*(Diffusion_Coeff_i[iSpecies] +
116 Diffusion_Coeff_j[iSpecies]);
117 Mean_Laminar_Viscosity = 0.5*(Laminar_Viscosity_i +
118 Laminar_Viscosity_j);
119 Mean_Eddy_Viscosity = 0.5*(Eddy_Viscosity_i +
120 Eddy_Viscosity_j);
121 Mean_Thermal_Conductivity = 0.5*(Thermal_Conductivity_i +
122 Thermal_Conductivity_j);
123 Mean_Thermal_Conductivity_ve = 0.5*(Thermal_Conductivity_ve_i +
124 Thermal_Conductivity_ve_j);
125
126 /*--- Mean gradient approximation ---*/
127 // Mass fraction
128 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
129 PrimVar_i[iSpecies] = V_i[iSpecies]/V_i[RHO_INDEX];
130 PrimVar_j[iSpecies] = V_j[iSpecies]/V_j[RHO_INDEX];
131 Mean_PrimVar[iSpecies] = 0.5*(PrimVar_i[iSpecies] + PrimVar_j[iSpecies]);
132 for (auto iDim = 0ul; iDim < nDim; iDim++) {
133 Mean_GradPrimVar[iSpecies][iDim] = 0.5*(1.0/V_i[RHO_INDEX] * (PrimVar_Grad_i[iSpecies][iDim] -
134 PrimVar_i[iSpecies] * PrimVar_Grad_i[RHO_INDEX][iDim]) +
135 1.0/V_j[RHO_INDEX] * (PrimVar_Grad_j[iSpecies][iDim] -
136 PrimVar_j[iSpecies] * PrimVar_Grad_j[RHO_INDEX][iDim]));
137 }
138 }
139
140 for (auto iVar = nSpecies; iVar < nPrimVar; iVar++) {
141 PrimVar_i[iVar] = V_i[iVar];
142 PrimVar_j[iVar] = V_j[iVar];
143 Mean_PrimVar[iVar] = 0.5*(PrimVar_i[iVar]+PrimVar_j[iVar]);
144 }
145 for (auto iVar = nSpecies; iVar < nPrimVarGrad; iVar++) {
146 for (auto iDim = 0; iDim < nDim; iDim++) {
147 Mean_GradPrimVar[iVar][iDim] = 0.5*(PrimVar_Grad_i[iVar][iDim] +
148 PrimVar_Grad_j[iVar][iDim]);
149 }
150 }
151 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
152 Mean_Eve[iSpecies] = 0.5*(eve_i[iSpecies] + eve_j[iSpecies]);
153 Mean_Cvve[iSpecies] = 0.5*(Cvve_i[iSpecies] + Cvve_j[iSpecies]);
154 }
155
156 /*--- Get projected flux tensor ---*/
157 GetViscousProjFlux(Mean_PrimVar, Mean_GradPrimVar, Mean_Eve, Normal,
158 Mean_Diffusion_Coeff, Mean_Laminar_Viscosity, Mean_Eddy_Viscosity,
159 Mean_Thermal_Conductivity, Mean_Thermal_Conductivity_ve,
160 config);
161
162
163 /*--- Update viscous residual ---*/
164 for (auto iVar = 0ul; iVar < nVar; iVar++)
165 Flux[iVar] = Proj_Flux_Tensor[iVar];
166
167 /*--- Compute the implicit part ---*/
168 if (implicit) {
169
170 su2double dist_ij_2[MAXNDIM] = {0.0};
171 GeometryToolbox::Distance(nDim, Coord_j, Coord_i, dist_ij_2);
172 dist_ij = GeometryToolbox::SquaredNorm(nDim, dist_ij_2);
173
174 for (auto iVar = 0ul; iVar < nVar; iVar++) {
175 for (auto jVar = 0ul; jVar < nVar; jVar++) {
176 Jacobian_i[iVar][jVar] = 0.0;
177 Jacobian_j[iVar][jVar] = 0.0;
178 }
179 }
180 GetViscousProjJacs(Mean_PrimVar, Mean_Eve, Mean_Cvve,
181 Mean_Diffusion_Coeff, Mean_Laminar_Viscosity,Mean_Eddy_Viscosity,
182 Mean_Thermal_Conductivity, Mean_Thermal_Conductivity_ve,
183 dist_ij, UnitNormal, Area, Proj_Flux_Tensor,
184 Jacobian_i, Jacobian_j, config);
185 }
186
187 return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
188
189}

263CNumerics::ResidualType<> CAvgGradCorrected_NEMO::ComputeResidual(const CConfig *config) {
264
265 /*--- Normalized normal vector ---*/
266 Area = GeometryToolbox::Norm(nDim, Normal);
267
268 for (auto iDim = 0ul; iDim < nDim; iDim++)
269 UnitNormal[iDim] = Normal[iDim]/Area;
270
271 /*--- Compute vector going from iPoint to jPoint ---*/
272 for (auto iDim = 0ul; iDim < nDim; iDim++) {
273 Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
274 }
275 const su2double dist_ij_2 = GeometryToolbox::SquaredNorm(nDim, Edge_Vector);
276
277 /*--- Make a local copy of the primitive variables ---*/
278 // NOTE: We are transforming the species density terms to species mass fractions
279 // Mass fraction
280 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
281 PrimVar_i[iSpecies] = V_i[iSpecies]/V_i[RHO_INDEX];
282 PrimVar_j[iSpecies] = V_j[iSpecies]/V_j[RHO_INDEX];
283 Mean_PrimVar[iSpecies] = 0.5*(PrimVar_i[iSpecies] + PrimVar_j[iSpecies]);
284 for (auto iDim = 0ul; iDim < nDim; iDim++) {
285 Mean_GradPrimVar[iSpecies][iDim] = 0.5*(1.0/V_i[RHO_INDEX] *
286 (PrimVar_Grad_i[iSpecies][iDim] -
287 PrimVar_i[iSpecies] *
288 PrimVar_Grad_i[RHO_INDEX][iDim]) +
289 1.0/V_j[RHO_INDEX] *
290 (PrimVar_Grad_j[iSpecies][iDim] -
291 PrimVar_j[iSpecies] *
292 PrimVar_Grad_j[RHO_INDEX][iDim]));
293 }
294 }
295 for (auto iVar = nSpecies; iVar < nPrimVar; iVar++) {
296 PrimVar_i[iVar] = V_i[iVar];
297 PrimVar_j[iVar] = V_j[iVar];
298 Mean_PrimVar[iVar] = 0.5*(PrimVar_i[iVar]+PrimVar_j[iVar]);
299 }
300 for (auto iVar = nSpecies; iVar < nPrimVarGrad; iVar++) {
301 for (auto iDim = 0ul; iDim < nDim; iDim++) {
302 Mean_GradPrimVar[iVar][iDim] = 0.5*(PrimVar_Grad_i[iVar][iDim] +
303 PrimVar_Grad_j[iVar][iDim]);
304 }
305 }
306 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
307 Mean_Eve[iSpecies] = 0.5*(eve_i[iSpecies] + eve_j[iSpecies]);
308 Mean_Cvve[iSpecies] = 0.5*(Cvve_i[iSpecies] + Cvve_j[iSpecies]);
309 }
310
311 /*--- Mean transport coefficients ---*/
312 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
313 Mean_Diffusion_Coeff[iSpecies] = 0.5*(Diffusion_Coeff_i[iSpecies] +
314 Diffusion_Coeff_j[iSpecies]);
315 Mean_Laminar_Viscosity = 0.5*(Laminar_Viscosity_i +
316 Laminar_Viscosity_j);
317 Mean_Eddy_Viscosity = 0.5*(Eddy_Viscosity_i +
318 Eddy_Viscosity_j);
319 Mean_Thermal_Conductivity = 0.5*(Thermal_Conductivity_i +
320 Thermal_Conductivity_j);
321 Mean_Thermal_Conductivity_ve = 0.5*(Thermal_Conductivity_ve_i +
322 Thermal_Conductivity_ve_j);
323
324 /*--- Projection of the mean gradient in the direction of the edge ---*/
325 for (auto iVar = 0ul; iVar < nPrimVarGrad; iVar++) {
326 Proj_Mean_GradPrimVar_Edge[iVar] = GeometryToolbox::DotProduct(nDim, Mean_GradPrimVar[iVar], Edge_Vector);
327 for (auto iDim = 0ul; iDim < nDim; iDim++) {
328 Mean_GradPrimVar[iVar][iDim] -= (Proj_Mean_GradPrimVar_Edge[iVar] -
329 (PrimVar_j[iVar]-PrimVar_i[iVar]))*Edge_Vector[iDim] / dist_ij_2;
330 }
331 }
332
333 /*--- Get projected flux tensor ---*/
334 GetViscousProjFlux(Mean_PrimVar, Mean_GradPrimVar, Mean_Eve,
335 Normal, Mean_Diffusion_Coeff,
336 Mean_Laminar_Viscosity,
337 Mean_Eddy_Viscosity,
338 Mean_Thermal_Conductivity,
339 Mean_Thermal_Conductivity_ve,
340 config);
341
342 /*--- Update viscous residual ---*/
343 for (auto iVar = 0ul; iVar < nVar; iVar++)
344 Flux[iVar] = Proj_Flux_Tensor[iVar];
345
346 /*--- Compute the implicit part ---*/
347 if (implicit) {
348 dist_ij = sqrt(dist_ij_2);
349
350 for (auto iVar = 0ul; iVar < nVar; iVar++) {
351 for (unsigned short jVar = 0; jVar < nVar; jVar++) {
352 Jacobian_i[iVar][jVar] = 0.0;
353 Jacobian_j[iVar][jVar] = 0.0;
354 }
355 }
356 GetViscousProjJacs(Mean_PrimVar, Mean_Eve, Mean_Cvve,
357 Mean_Diffusion_Coeff, Mean_Laminar_Viscosity, Mean_Eddy_Viscosity,
358 Mean_Thermal_Conductivity, Mean_Thermal_Conductivity_ve,
359 dist_ij, UnitNormal, Area, Proj_Flux_Tensor,
360 Jacobian_i, Jacobian_j, config);
361
362 }
363
364 return ResidualType<>(Flux, Jacobian_j, Jacobian_j);
365
366}
