32CFVMDataSorter::CFVMDataSorter(CConfig *config, CGeometry *geometry, const vector<string> &valFieldNames) :
33 CParallelDataSorter(config, valFieldNames){
34
35 nDim = geometry->GetnDim();
36
37 std::vector<unsigned long> globalID;
38
39 nGlobalPointBeforeSort = geometry->GetGlobal_nPointDomain();
40 nLocalPointsBeforeSort = geometry->GetnPointDomain();
41
42 Local_Halo.resize(geometry->GetnPoint());
43
44 for (unsigned long iPoint = 0; iPoint < geometry->GetnPoint(); iPoint++){
45
46 /*--- Store the global IDs ---*/
47
48 globalID.push_back(geometry->nodes->GetGlobalIndex(iPoint));
49
50 Local_Halo[iPoint] = !geometry->nodes->GetDomain(iPoint);
51 }
52
53
54 /*--- Search all send/recv boundaries on this partition for halo cells. In
55 particular, consider only the recv conditions (these are the true halo
56 nodes). Check the ranks of the processors that are communicating and
57 choose to keep only the halo cells from the higher rank processor. ---*/
58
59 SetHaloPoints(geometry, config);
60
61 /*--- Create the linear partitioner --- */
62
63 linearPartitioner.Initialize(nGlobalPointBeforeSort, 0);
64
65 /*--- Prepare the send buffers ---*/
66
67 PrepareSendBuffers(globalID);
68
69}

71void CFVMDataSorter::SetHaloPoints(CGeometry *geometry, CConfig *config){
72
73 unsigned long iPoint, iVertex;
74 unsigned short iMarker;
75 int SendRecv, RecvFrom;
76 bool notHalo;
77
78 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
79 if (config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) {
80 SendRecv = config->GetMarker_All_SendRecv(iMarker);
81 RecvFrom = abs(SendRecv)-1;
82
83 for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
84 iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
85
86 /*--- We need to keep one copy of overlapping halo cells. ---*/
87
88 notHalo = ((SendRecv < 0) && (rank > RecvFrom));
89
90 /*--- If we found either of these types of nodes, flag them to be kept. ---*/
91
92 if (notHalo) {
93 Local_Halo[iPoint] = false;
94 }
95 }
96 }
97 }

121void CFVMDataSorter::SortVolumetricConnectivity(CConfig *config,
122 CGeometry *geometry,
123 unsigned short Elem_Type,
124 bool val_sort) {
125
126 unsigned long iProcessor;
127 unsigned short NODES_PER_ELEMENT = 0;
128 unsigned long iPoint, jPoint;
129 unsigned long nElem_Total = 0, Global_Index;
130
131 int *Conn_Elem = nullptr;
132
133#ifdef HAVE_MPI
134 SU2_MPI::Request *send_req, *recv_req;
135 SU2_MPI::Status status;
136 int ind;
137#endif
138
139 /*--- Store the local number of this element type and the number of nodes
140 per this element type. In serial, this will be the total number of this
141 element type in the entire mesh. In parallel, it is the number on only
142 the current partition. ---*/
143
144 switch (Elem_Type) {
145 case TRIANGLE:
146 NODES_PER_ELEMENT = N_POINTS_TRIANGLE;
147 break;
148 case QUADRILATERAL:
149 NODES_PER_ELEMENT = N_POINTS_QUADRILATERAL;
150 break;
151 case TETRAHEDRON:
152 NODES_PER_ELEMENT = N_POINTS_TETRAHEDRON;
153 break;
154 case HEXAHEDRON:
155 NODES_PER_ELEMENT = N_POINTS_HEXAHEDRON;
156 break;
157 case PRISM:
158 NODES_PER_ELEMENT = N_POINTS_PRISM;
159 break;
160 case PYRAMID:
161 NODES_PER_ELEMENT = N_POINTS_PYRAMID;
162 break;
163 default:
164 SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
165 }
166
167 /*--- We start with the connectivity distributed across all procs with
168 no particular ordering assumed. We need to loop through our local partition
169 and decide how many elements we must send to each other rank in order to
170 have all elements sorted according to a linear partitioning of the grid
171 nodes, i.e., rank 0 holds the first nPoint()/nProcessors nodes.
172 First, initialize a counter and flag. ---*/
173
174 int *nElem_Flag = new int[size]();
175
176 for (int ii=0; ii < size; ii++) {
177 nElem_Send[ii] = 0;
178 nElem_Cum[ii] = 0;
179 nElem_Flag[ii]= -1;
180 }
181 nElem_Send[size] = 0; nElem_Cum[size] = 0;
182
183 for (int ii = 0; ii < (int)geometry->GetnElem(); ii++ ) {
184 if (geometry->elem[ii]->GetVTK_Type() == Elem_Type) {
185 for ( int jj = 0; jj < NODES_PER_ELEMENT; jj++ ) {
186
187 /*--- Get the index of the current point. ---*/
188
189 iPoint = geometry->elem[ii]->GetNode(jj);
190 Global_Index = geometry->nodes->GetGlobalIndex(iPoint);
191
192 /*--- Search for the lowest global index in this element. We
193 send the element to the processor owning the range that includes
194 the lowest global index value. ---*/
195
196 for (int kk = 0; kk < NODES_PER_ELEMENT; kk++) {
197 jPoint = geometry->elem[ii]->GetNode(kk);
198 unsigned long newID = geometry->nodes->GetGlobalIndex(jPoint);
199 if (newID < Global_Index) Global_Index = newID;
200 }
201
202 /*--- Search for the processor that owns this point. If we are
203 sorting the elements, we use the linear partitioning to find
204 the rank, otherwise, we simply have the current rank load its
205 own elements into the connectivity data structure. ---*/
206
207 if (val_sort) {
208 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
209 } else {
210 iProcessor = rank;
211 }
212
213
214 /*--- If we have not visited this element yet, increment our
215 number of elements that must be sent to a particular proc. ---*/
216
217 if ((nElem_Flag[iProcessor] != ii)) {
218 nElem_Flag[iProcessor] = ii;
219 nElem_Send[iProcessor+1]++;
220 }
221
222 }
223 }
224 }
225
226 /*--- Communicate the number of cells to be sent/recv'd amongst
227 all processors. After this communication, each proc knows how
228 many cells it will receive from each other processor. ---*/
229
230 SU2_MPI::Alltoall(&(nElem_Send[1]), 1, MPI_INT,
231 &(nElem_Cum[1]), 1, MPI_INT, SU2_MPI::GetComm());
232
233 /*--- Prepare to send connectivities. First check how many
234 messages we will be sending and receiving. Here we also put
235 the counters into cumulative storage format to make the
236 communications simpler. ---*/
237
238 int nSends = 0, nRecvs = 0;
239 for (int ii=0; ii < size; ii++) nElem_Flag[ii] = -1;
240
241 for (int ii = 0; ii < size; ii++) {
242 if ((ii != rank) && (nElem_Send[ii+1] > 0)) nSends++;
243 if ((ii != rank) && (nElem_Cum[ii+1] > 0)) nRecvs++;
244
245 nElem_Send[ii+1] += nElem_Send[ii];
246 nElem_Cum[ii+1] += nElem_Cum[ii];
247 }
248
249 /*--- Allocate memory to hold the connectivity that we are
250 sending. ---*/
251
252 auto connSend = new unsigned long[NODES_PER_ELEMENT*nElem_Send[size]]();
253
254 /*--- Allocate arrays for storing halo flags. ---*/
255
256 auto haloSend = new unsigned short[nElem_Send[size]]();
257
258 /*--- Create an index variable to keep track of our index
259 position as we load up the send buffer. ---*/
260
261 auto *index = new unsigned long[size]();
262 for (int ii=0; ii < size; ii++) index[ii] = NODES_PER_ELEMENT*nElem_Send[ii];
263
264 auto *haloIndex = new unsigned long[size]();
265 for (int ii=0; ii < size; ii++) haloIndex[ii] = nElem_Send[ii];
266
267 /*--- Loop through our elements and load the elems and their
268 additional data that we will send to the other procs. ---*/
269
270 for (int ii = 0; ii < (int)geometry->GetnElem(); ii++) {
271 if (geometry->elem[ii]->GetVTK_Type() == Elem_Type) {
272 for ( int jj = 0; jj < NODES_PER_ELEMENT; jj++ ) {
273
274 /*--- Get the index of the current point. ---*/
275
276 iPoint = geometry->elem[ii]->GetNode(jj);
277 Global_Index = geometry->nodes->GetGlobalIndex(iPoint);
278
279 /*--- Search for the lowest global index in this element. We
280 send the element to the processor owning the range that includes
281 the lowest global index value. ---*/
282
283 for (int kk = 0; kk < NODES_PER_ELEMENT; kk++) {
284 jPoint = geometry->elem[ii]->GetNode(kk);
285 unsigned long newID = geometry->nodes->GetGlobalIndex(jPoint);
286 if (newID < Global_Index) Global_Index = newID;
287 }
288
289 /*--- Search for the processor that owns this point. If we are
290 sorting the elements, we use the linear partitioning to find
291 the rank, otherwise, we simply have the current rank load its
292 own elements into the connectivity data structure. ---*/
293
294 if (val_sort) {
295 iProcessor = linearPartitioner.GetRankContainingIndex(Global_Index);
296 } else {
297 iProcessor = rank;
298 }
299
300 /*--- Load connectivity into the buffer for sending ---*/
301
302 if (nElem_Flag[iProcessor] != ii) {
303
304 nElem_Flag[iProcessor] = ii;
305 unsigned long nn = index[iProcessor];
306 unsigned long mm = haloIndex[iProcessor];
307
308 /*--- Load the connectivity values. ---*/
309
310 for (int kk = 0; kk < NODES_PER_ELEMENT; kk++) {
311 iPoint = geometry->elem[ii]->GetNode(kk);
312 connSend[nn] = geometry->nodes->GetGlobalIndex(iPoint); nn++;
313
314 /*--- Check if this is a halo node. If so, flag this element
315 as a halo cell. We will use this later to sort and remove
316 any duplicates from the connectivity list. Note that just checking
317 whether the point is a halo point is not enough, since we want to keep
318 elements on one side of the send receive boundary. ---*/
319
320 if (Local_Halo[iPoint]) haloSend[mm] = true;
321 }
322
323 /*--- Increment the index by the message length ---*/
324
325 index[iProcessor] += NODES_PER_ELEMENT;
326 haloIndex[iProcessor]++;
327
328 }
329 }
330 }
331 }
332
333 /*--- Free memory after loading up the send buffer. ---*/
334
335 delete [] index;
336 delete [] haloIndex;
337
338 /*--- Allocate the memory that we need for receiving the conn
339 values and then cue up the non-blocking receives. Note that
340 we do not include our own rank in the communications. We will
341 directly copy our own data later. ---*/
342
343 auto connRecv = new unsigned long[NODES_PER_ELEMENT*nElem_Cum[size]]();
344
345 auto haloRecv = new unsigned short[nElem_Cum[size]]();
346
347#ifdef HAVE_MPI
348
349 send_req = new SU2_MPI::Request[2*nSends];
350 recv_req = new SU2_MPI::Request[2*nRecvs];
351
352 /*--- Launch the non-blocking recv's for the connectivity. ---*/
353
354 unsigned long iMessage = 0;
355 for (int ii=0; ii<size; ii++) {
356 if ((ii != rank) && (nElem_Cum[ii+1] > nElem_Cum[ii])) {
357 int ll = NODES_PER_ELEMENT*nElem_Cum[ii];
358 int kk = nElem_Cum[ii+1] - nElem_Cum[ii];
359 int count = NODES_PER_ELEMENT*kk;
360 int source = ii;
361 int tag = ii + 1;
362 SU2_MPI::Irecv(&(connRecv[ll]), count, MPI_UNSIGNED_LONG, source, tag,
363 SU2_MPI::GetComm(), &(recv_req[iMessage]));
364 iMessage++;
365 }
366 }
367
368 /*--- Launch the non-blocking sends of the connectivity. ---*/
369
370 iMessage = 0;
371 for (int ii=0; ii<size; ii++) {
372 if ((ii != rank) && (nElem_Send[ii+1] > nElem_Send[ii])) {
373 int ll = NODES_PER_ELEMENT*nElem_Send[ii];
374 int kk = nElem_Send[ii+1] - nElem_Send[ii];
375 int count = NODES_PER_ELEMENT*kk;
376 int dest = ii;
377 int tag = rank + 1;
378 SU2_MPI::Isend(&(connSend[ll]), count, MPI_UNSIGNED_LONG, dest, tag,
379 SU2_MPI::GetComm(), &(send_req[iMessage]));
380 iMessage++;
381 }
382 }
383
384 /*--- Repeat the process to communicate the halo flags. ---*/
385
386 iMessage = 0;
387 for (int ii=0; ii<size; ii++) {
388 if ((ii != rank) && (nElem_Cum[ii+1] > nElem_Cum[ii])) {
389 int ll = nElem_Cum[ii];
390 int kk = nElem_Cum[ii+1] - nElem_Cum[ii];
391 int count = kk;
392 int source = ii;
393 int tag = ii + 1;
394 SU2_MPI::Irecv(&(haloRecv[ll]), count, MPI_UNSIGNED_SHORT, source, tag,
395 SU2_MPI::GetComm(), &(recv_req[iMessage+nRecvs]));
396 iMessage++;
397 }
398 }
399
400 /*--- Launch the non-blocking sends of the halo flags. ---*/
401
402 iMessage = 0;
403 for (int ii=0; ii<size; ii++) {
404 if ((ii != rank) && (nElem_Send[ii+1] > nElem_Send[ii])) {
405 int ll = nElem_Send[ii];
406 int kk = nElem_Send[ii+1] - nElem_Send[ii];
407 int count = kk;
408 int dest = ii;
409 int tag = rank + 1;
410 SU2_MPI::Isend(&(haloSend[ll]), count, MPI_UNSIGNED_SHORT, dest, tag,
411 SU2_MPI::GetComm(), &(send_req[iMessage+nSends]));
412 iMessage++;
413 }
414 }
415#endif
416
417 /*--- Copy my own rank's data into the recv buffer directly. ---*/
418
419 int mm = NODES_PER_ELEMENT*nElem_Cum[rank];
420 int ll = NODES_PER_ELEMENT*nElem_Send[rank];
421 int kk = NODES_PER_ELEMENT*nElem_Send[rank+1];
422
423 for (int nn=ll; nn<kk; nn++, mm++) connRecv[mm] = connSend[nn];
424
425 mm = nElem_Cum[rank];
426 ll = nElem_Send[rank];
427 kk = nElem_Send[rank+1];
428
429 for (int nn=ll; nn<kk; nn++, mm++) haloRecv[mm] = haloSend[nn];
430
431 /*--- Wait for the non-blocking sends and recvs to complete. ---*/
432
433#ifdef HAVE_MPI
434 int number = 2*nSends;
435 for (int ii = 0; ii < number; ii++)
436 SU2_MPI::Waitany(number, send_req, &ind, &status);
437
438 number = 2*nRecvs;
439 for (int ii = 0; ii < number; ii++)
440 SU2_MPI::Waitany(number, recv_req, &ind, &status);
441
442 delete [] send_req;
443 delete [] recv_req;
444#endif
445
446 /*--- Store the connectivity for this rank in the proper data
447 structure before post-processing below. Note that we add 1 here
448 to the connectivity for vizualization packages. First, allocate
449 appropriate amount of memory for this section. ---*/
450
451 if (nElem_Cum[size] > 0) Conn_Elem = new int[NODES_PER_ELEMENT*nElem_Cum[size]]();
452 int count = 0; nElem_Total = 0;
453 for (int ii = 0; ii < nElem_Cum[size]; ii++) {
454 if (!haloRecv[ii]) {
455 nElem_Total++;
456 for (int jj = 0; jj < NODES_PER_ELEMENT; jj++) {
457 Conn_Elem[count] = (int)connRecv[ii*NODES_PER_ELEMENT+jj] + 1;
458 count++;
459 }
460 }
461 }
462
463 nElemPerType[TypeMap.at(Elem_Type)] = nElem_Total;
464
465 /*--- Store the particular global element count in the class data,
466 and set the class data pointer to the connectivity array. ---*/
467
468 switch (Elem_Type) {
469 case TRIANGLE:
470 delete [] Conn_Tria_Par;
471 Conn_Tria_Par = Conn_Elem;
472 break;
473 case QUADRILATERAL:
474 delete [] Conn_Quad_Par;
475 Conn_Quad_Par = Conn_Elem;
476 break;
477 case TETRAHEDRON:
478 delete [] Conn_Tetr_Par;
479 Conn_Tetr_Par = Conn_Elem;
480 break;
481 case HEXAHEDRON:
482 delete [] Conn_Hexa_Par;
483 Conn_Hexa_Par = Conn_Elem;
484 break;
485 case PRISM:
486 delete [] Conn_Pris_Par;
487 Conn_Pris_Par = Conn_Elem;
488 break;
489 case PYRAMID:
490 delete [] Conn_Pyra_Par;
491 Conn_Pyra_Par = Conn_Elem;
492 break;
493 default:
494 SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
495 break;
496 }
497
498 /*--- Free temporary memory from communications ---*/
499
500 delete [] connSend;
501 delete [] connRecv;
502 delete [] haloSend;
503 delete [] haloRecv;
504 delete [] nElem_Flag;
505
506}
