37/*!
38 * \class CRoeBase
39 * \ingroup ConvDiscr
40 * \brief Base class for Roe schemes, derived classes implement
41 * the dissipation term in a const "finalizeFlux" method.
42 * A base class implementing "viscousTerms" is accepted as template parameter.
43 * Similarly to derived, that method should update the flux and Jacobians, but
44 * whereas "finalizeFlux" is passed data prepared by CRoeBase, "viscousTerms"
45 * takes the same input arguments as "ComputeFlux", i.e. it can fetch more
46 * data from CVariable. Derived is meant to implement small details,
47 * Base is meant to do heavy lifting.
48 */
49template<class Derived, class Base>
50class CRoeBase : public Base {
51protected:
52 using Base::nDim;
53 static constexpr size_t nVar = CCompressibleConservatives<nDim>::nVar;
54 static constexpr size_t nPrimVarGrad = nDim+4;
55 static constexpr size_t nPrimVar = Max(Base::nPrimVar, nPrimVarGrad);
56
57 const su2double kappa;
58 const su2double gamma;
59 const su2double entropyFix;
60 const bool finestGrid;
61 const bool dynamicGrid;
62 const bool muscl;
63 const LIMITER typeLimiter;
64
65 /*!
66 * \brief Constructor, store some constants and forward args to base.
67 */
68 template<class... Ts>
69 CRoeBase(const CConfig& config, unsigned iMesh, Ts&... args) : Base(config, iMesh, args...),
70 kappa(config.GetRoe_Kappa()),
71 gamma(config.GetGamma()),
72 entropyFix(config.GetEntropyFix_Coeff()),
73 finestGrid(iMesh == MESH_0),
74 dynamicGrid(config.GetDynamic_Grid()),
75 muscl(finestGrid && config.GetMUSCL_Flow()),
76 typeLimiter(config.GetKind_SlopeLimit_Flow()) {
77 }
78
79public:
80 /*!
81 * \brief Implementation of the base Roe flux.
82 */
83 void ComputeFlux(Int iEdge,
84 const CConfig& config,
85 const CGeometry& geometry,
86 const CVariable& solution_,
87 UpdateType updateType,
88 Double updateMask,
89 CSysVector<su2double>& vector,
90 SparseMatrixType& matrix) const final {
91
92 /*--- Start preaccumulation, inputs are registered
93 * automatically in "gatherVariables". ---*/
94 AD::StartPreacc();
95
96 const bool implicit = (config.GetKind_TimeIntScheme() == EULER_IMPLICIT);
97 const auto& solution = static_cast<const CEulerVariable&>(solution_);
98
99 const auto iPoint = geometry.edges->GetNode(iEdge,0);
100 const auto jPoint = geometry.edges->GetNode(iEdge,1);
101
102 /*--- Geometric properties. ---*/
103
104 const auto vector_ij = distanceVector<nDim>(iPoint, jPoint, geometry.nodes->GetCoord());
105
106 const auto normal = gatherVariables<nDim>(iEdge, geometry.edges->GetNormal());
107 const auto area = norm(normal);
108 VectorDbl<nDim> unitNormal;
109 for (size_t iDim = 0; iDim < nDim; ++iDim) {
110 unitNormal(iDim) = normal(iDim) / area;
111 }
112
113 /*--- Reconstructed primitives. ---*/
114
115 CPair<CCompressiblePrimitives<nDim,nPrimVar> > V1st;
116 V1st.i.all = gatherVariables<nPrimVar>(iPoint, solution.GetPrimitive());
117 V1st.j.all = gatherVariables<nPrimVar>(jPoint, solution.GetPrimitive());
118
119 auto V = reconstructPrimitives<CCompressiblePrimitives<nDim,nPrimVarGrad> >(
120 iEdge, iPoint, jPoint, muscl, typeLimiter, V1st, vector_ij, solution);
121
122 /*--- Compute conservative variables. ---*/
123
124 CPair<CCompressibleConservatives<nDim> > U;
125 U.i = compressibleConservatives(V.i);
126 U.j = compressibleConservatives(V.j);
127
128 /*--- Roe averaged variables. ---*/
129
130 auto roeAvg = roeAveragedVariables(gamma, V, unitNormal);
131
132 /*--- P tensor. ---*/
133
134 auto pMat = pMatrix(gamma, roeAvg.density, roeAvg.velocity,
135 roeAvg.projVel, roeAvg.speedSound, unitNormal);
136
137 /*--- Grid motion. ---*/
138
139 Double projGridVel = 0.0, projVel = roeAvg.projVel;
140 if (dynamicGrid) {
141 const auto& gridVel = geometry.nodes->GetGridVel();
142 projGridVel = 0.5*(dot(gatherVariables<nDim>(iPoint,gridVel), unitNormal)+
143 dot(gatherVariables<nDim>(jPoint,gridVel), unitNormal));
144 projVel -= projGridVel;
145 }
146
147 /*--- Convective eigenvalues. ---*/
148
149 VectorDbl<nVar> lambda;
150 for (size_t iDim = 0; iDim < nDim; ++iDim) {
151 lambda(iDim) = projVel;
152 }
153 lambda(nDim) = projVel + roeAvg.speedSound;
154 lambda(nDim+1) = projVel - roeAvg.speedSound;
155
156 /*--- Apply Mavriplis' entropy correction to eigenvalues. ---*/
157
158 Double maxLambda = abs(projVel) + roeAvg.speedSound;
159
160 for (size_t iVar = 0; iVar < nVar; ++iVar) {
161 lambda(iVar) = fmax(abs(lambda(iVar)), entropyFix*maxLambda);
162 }
163
164 /*--- Inviscid fluxes and Jacobians. ---*/
165
166 auto flux_i = inviscidProjFlux(V.i, U.i, normal);
167 auto flux_j = inviscidProjFlux(V.j, U.j, normal);
168
169 VectorDbl<nVar> flux;
170 for (size_t iVar = 0; iVar < nVar; ++iVar) {
171 flux(iVar) = 0.5 * (flux_i(iVar) + flux_j(iVar));
172 }
173
174 MatrixDbl<nVar> jac_i, jac_j;
175 if (implicit) {
176 jac_i = inviscidProjJac(gamma, V.i.velocity(), U.i.energy(), normal, kappa);
177 jac_j = inviscidProjJac(gamma, V.j.velocity(), U.j.energy(), normal, kappa);
178 }
179
180 /*--- Correct for grid motion. ---*/
181
182 if (dynamicGrid) {
183 for (size_t iVar = 0; iVar < nVar; ++iVar) {
184 Double dFdU = projGridVel * area * 0.5;
185 flux(iVar) -= dFdU * (U.i.all(iVar) + U.j.all(iVar));
186
187 if (implicit) {
188 jac_i(iVar,iVar) -= dFdU;
189 jac_j(iVar,iVar) -= dFdU;
190 }
191 }
192 }
193
194 /*--- Finalize in derived class (static polymorphism). ---*/
195
196 const auto derived = static_cast<const Derived*>(this);
197
198 derived->finalizeFlux(flux, jac_i, jac_j, implicit, area, unitNormal, V,
199 U, roeAvg, lambda, pMat, iPoint, jPoint, solution);
200
201 /*--- Add the contributions from the base class (static decorator). ---*/
202
203 Base::viscousTerms(iEdge, iPoint, jPoint, V1st, solution_, vector_ij, geometry,
204 config, area, unitNormal, implicit, flux, jac_i, jac_j);
205
206 /*--- Stop preaccumulation. ---*/
207
208 stopPreacc(flux);
209
210 /*--- Update the vector and system matrix. ---*/
211
212 updateLinearSystem(iEdge, iPoint, jPoint, implicit, updateType,
213 updateMask, flux, jac_i, jac_j, vector, matrix);
214 }
215};

221 */
222template<class Decorator>
223class CRoeScheme : public CRoeBase<CRoeScheme<Decorator>,Decorator> {
224private:
225 using Base = CRoeBase<CRoeScheme<Decorator>,Decorator>;
226 using Base::nDim;
227 using Base::nVar;
228 using Base::gamma;
229 using Base::kappa;
230 const ENUM_ROELOWDISS typeDissip;
231
232public:
233 /*!
234 * \brief Constructor, store some constants and forward to base.
235 */
236 template<class... Ts>
237 CRoeScheme(const CConfig& config, Ts&... args) : Base(config, args...),
238 typeDissip(static_cast<ENUM_ROELOWDISS>(config.GetKind_RoeLowDiss())) {
239 }
240
241 /*!
242 * \brief Updates flux and Jacobians with standard Roe dissipation.
243 * \note "Ts" is here just in case other schemes in the family need extra args.
244 */
245 template<class PrimVarType, class ConsVarType, class... Ts>
246 FORCEINLINE void finalizeFlux(VectorDbl<nVar>& flux,
247 MatrixDbl<nVar>& jac_i,
248 MatrixDbl<nVar>& jac_j,
249 bool implicit,
250 Double area,
251 const VectorDbl<nDim>& unitNormal,
252 const CPair<PrimVarType>& V,
253 const CPair<ConsVarType>& U,
254 const CRoeVariables<nDim>& roeAvg,
255 const VectorDbl<nVar>& lambda,
256 const MatrixDbl<nVar>& pMat,
257 Int iPoint,
258 Int jPoint,
259 const CEulerVariable& solution,
260 Ts&...) const {
261 /*--- Inverse P tensor. ---*/
262
263 auto pMatInv = pMatrixInv(gamma, roeAvg.density, roeAvg.velocity,
264 roeAvg.projVel, roeAvg.speedSound, unitNormal);
265
266 /*--- Diference between conservative variables at jPoint and iPoint. ---*/
267
268 VectorDbl<nVar> deltaU;
269 for (size_t iVar = 0; iVar < nVar; ++iVar) {
270 deltaU(iVar) = U.j.all(iVar) - U.i.all(iVar);
271 }
272
273 /*--- Dissipation terms. ---*/
274
275 Double dissipation = roeDissipation(iPoint, jPoint, typeDissip, solution);
276
277 for (size_t iVar = 0; iVar < nVar; ++iVar) {
278 for (size_t jVar = 0; jVar < nVar; ++jVar) {
279 /*--- Compute |projModJacTensor| = P x |Lambda| x P^-1. ---*/
280
281 Double projModJacTensor = 0.0;
282 for (size_t kVar = 0; kVar < nVar; ++kVar) {
283 projModJacTensor += pMat(iVar,kVar) * lambda(kVar) * pMatInv(kVar,jVar);
284 }
285
286 Double dDdU = projModJacTensor * (1-kappa) * area * dissipation;
287
288 /*--- Update flux and Jacobians. ---*/
289
290 flux(iVar) -= dDdU * deltaU(jVar);
291
292 if(implicit) {
293 jac_i(iVar,jVar) += dDdU;
294 jac_j(iVar,jVar) -= dDdU;
295 }
296 }
297 }
298 }
