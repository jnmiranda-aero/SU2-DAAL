49 unique_ptr<CViscosityModel> CFluidModel::MakeLaminarViscosityModel(const CConfig* config, unsigned short iSpecies) {
50  switch (config->GetKind_ViscosityModel()) {
51  case VISCOSITYMODEL::CONSTANT:
52  return unique_ptr<CConstantViscosity>(new CConstantViscosity(config->GetMu_ConstantND(iSpecies)));
53  case VISCOSITYMODEL::COOLPROP:
54  return unique_ptr<CCoolPropViscosity>(new CCoolPropViscosity(config->GetFluid_Name()));
55  case VISCOSITYMODEL::SUTHERLAND:
56  return unique_ptr<CSutherland>(new CSutherland(config->GetMu_RefND(iSpecies),
57  config->GetMu_Temperature_RefND(iSpecies),
58  config->GetMu_SND(iSpecies)));
59  case VISCOSITYMODEL::POLYNOMIAL:
60  return unique_ptr<CPolynomialViscosity<N_POLY_COEFFS>>(
61  new CPolynomialViscosity<N_POLY_COEFFS>(config->GetMu_PolyCoeffND()));
62  case VISCOSITYMODEL::FLAMELET:
63  /*--- Viscosity is obtained from the LUT ---*/
64  return nullptr;
65  default:
66  SU2_MPI::Error("Viscosity model not available.", CURRENT_FUNCTION);
67  return nullptr;
68  }
69 }

75 unique_ptr<CConductivityModel> CFluidModel::MakeThermalConductivityModel(const CConfig* config,
76  unsigned short iSpecies) {
77  switch (config->GetKind_ConductivityModel()) {
78  case CONDUCTIVITYMODEL::CONSTANT:
79  if (config->GetKind_ConductivityModel_Turb() == CONDUCTIVITYMODEL_TURB::CONSTANT_PRANDTL) {
80  return unique_ptr<CConstantConductivityRANS>(
81  new CConstantConductivityRANS(config->GetThermal_Conductivity_ConstantND(iSpecies),
82  config->GetPrandtl_Turb(iSpecies)));
83  } else {
84  return unique_ptr<CConstantConductivity>(
85  new CConstantConductivity(config->GetThermal_Conductivity_ConstantND(iSpecies)));
86  }
87  break;
88  case CONDUCTIVITYMODEL::COOLPROP:
89  if (config->GetKind_ConductivityModel_Turb() == CONDUCTIVITYMODEL_TURB::CONSTANT_PRANDTL) {
90  return unique_ptr<CConstantConductivityRANS>(
91  new CConstantConductivityRANS(config->GetThermal_Conductivity_ConstantND(iSpecies),
92  config->GetPrandtl_Turb(iSpecies)));
93  } else {
94  return unique_ptr<CCoolPropConductivity>(
95  new CCoolPropConductivity(config->GetFluid_Name()));
96  }
97  break;
98  case CONDUCTIVITYMODEL::CONSTANT_PRANDTL:
99  if (config->GetKind_ConductivityModel_Turb() == CONDUCTIVITYMODEL_TURB::CONSTANT_PRANDTL) {
100  return unique_ptr<CConstantPrandtlRANS>(
101  new CConstantPrandtlRANS(config->GetPrandtl_Lam(iSpecies), config->GetPrandtl_Turb(iSpecies)));
102  } else {
103  return unique_ptr<CConstantPrandtl>(new CConstantPrandtl(config->GetPrandtl_Lam(iSpecies)));
104  }
105  break;
106  case CONDUCTIVITYMODEL::POLYNOMIAL:
107  if (config->GetKind_ConductivityModel_Turb() == CONDUCTIVITYMODEL_TURB::CONSTANT_PRANDTL) {
108  return unique_ptr<CPolynomialConductivityRANS<N_POLY_COEFFS>>(
109  new CPolynomialConductivityRANS<N_POLY_COEFFS>(config->GetKt_PolyCoeffND(), config->GetPrandtl_Turb()));
110  } else {
111  return unique_ptr<CPolynomialConductivity<N_POLY_COEFFS>>(
112  new CPolynomialConductivity<N_POLY_COEFFS>(config->GetKt_PolyCoeffND()));
113  }
114  break;
115  case CONDUCTIVITYMODEL::FLAMELET:
116  /*--- Conductivity is obtained from the LUT ---*/
117  return nullptr;
118  default:
119  SU2_MPI::Error("Conductivity model not available.", CURRENT_FUNCTION);
120  return nullptr;
121  }
122 }

128 unique_ptr<CDiffusivityModel> CFluidModel::MakeMassDiffusivityModel(const CConfig* config, unsigned short iSpecies) {
129  switch (config->GetKind_Diffusivity_Model()) {
130  case DIFFUSIVITYMODEL::CONSTANT_DIFFUSIVITY:
131  return unique_ptr<CConstantDiffusivity>(new CConstantDiffusivity(config->GetDiffusivity_ConstantND()));
132  break;
133  case DIFFUSIVITYMODEL::CONSTANT_SCHMIDT:
134  return unique_ptr<CConstantSchmidt>(new CConstantSchmidt(config->GetSchmidt_Number_Laminar()));
135  break;
136  case DIFFUSIVITYMODEL::UNITY_LEWIS:
137  return unique_ptr<CConstantLewisDiffusivity>(new CConstantLewisDiffusivity(1.0));
138  break;
139  case DIFFUSIVITYMODEL::CONSTANT_LEWIS:
140  return unique_ptr<CConstantLewisDiffusivity>(
141  new CConstantLewisDiffusivity(config->GetConstant_Lewis_Number(iSpecies)));
142  break;
143  case DIFFUSIVITYMODEL::FLAMELET:
144  /*--- Diffusivity is obtained from the LUT ---*/
145  return nullptr;
146  break;
147  default:
148  SU2_MPI::Error("Diffusivity model not available.", CURRENT_FUNCTION);
149  return nullptr;
150  }
151 }
