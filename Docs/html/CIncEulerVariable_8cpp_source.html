31CIncEulerVariable::CIncEulerVariable(su2double pressure, const su2double *velocity, su2double temperature,
32 unsigned long npoint, unsigned long ndim, unsigned long nvar, const CConfig *config)
33 : CFlowVariable(npoint, ndim, nvar, ndim + 9, ndim + 4, config),
34 indices(ndim, 0) {
35
36 const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
37 (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
38 const bool classical_rk4 = (config->GetKind_TimeIntScheme_Flow() == CLASSICAL_RK4_EXPLICIT);
39
40 /*--- Solution initialization ---*/
41
42 su2double val_solution[5] = {pressure, velocity[0], velocity[1], temperature, temperature};
43 if(nDim==3) val_solution[3] = velocity[2];
44
45 for(unsigned long iPoint=0; iPoint<nPoint; ++iPoint)
46 for (unsigned long iVar = 0; iVar < nVar; iVar++)
47 Solution(iPoint,iVar) = val_solution[iVar];
48
49 Solution_Old = Solution;
50
51 if (classical_rk4) Solution_New = Solution;
52
53 /*--- Allocate and initialize solution for dual time strategy ---*/
54
55 if (dual_time) {
56 Solution_time_n = Solution;
57 Solution_time_n1 = Solution;
58 }
59
60 if (config->GetKind_Streamwise_Periodic() != ENUM_STREAMWISE_PERIODIC::NONE) {
61 Streamwise_Periodic_RecoveredPressure.resize(nPoint) = su2double(0.0);
62 if (config->GetStreamwise_Periodic_Temperature())
63 Streamwise_Periodic_RecoveredTemperature.resize(nPoint) = su2double(0.0);
64 }
65}

67bool CIncEulerVariable::SetPrimVar(unsigned long iPoint, CFluidModel *FluidModel) {
68
69 bool physical = true;
70
71 /*--- Set the value of the pressure ---*/
72
73 SetPressure(iPoint);
74
75 /*--- Set the value of the temperature directly ---*/
76
77 su2double Temperature = Solution(iPoint, nDim+1);
78 const auto check_temp = SetTemperature(iPoint, Temperature);
79
80 /*--- Use the fluid model to compute the new value of density.
81 Note that the thermodynamic pressure is constant and decoupled
82 from the dynamic pressure being iterated. ---*/
83
84 /*--- Use the fluid model to compute the new value of density. ---*/
85
86 FluidModel->SetTDState_T(Temperature);
87
88 /*--- Set the value of the density ---*/
89
90 const auto check_dens = SetDensity(iPoint, FluidModel->GetDensity());
91
92 /*--- Non-physical solution found. Revert to old values. ---*/
93
94 if (check_dens || check_temp) {
95
96 /*--- Copy the old solution ---*/
97
98 for (auto iVar = 0ul; iVar < nVar; iVar++)
99 Solution(iPoint, iVar) = Solution_Old(iPoint, iVar);
100
101 /*--- Recompute the primitive variables ---*/
102
103 Temperature = Solution(iPoint, nDim+1);
104 SetTemperature(iPoint, Temperature);
105 FluidModel->SetTDState_T(Temperature);
106 SetDensity(iPoint, FluidModel->GetDensity());
107
108 /*--- Flag this point as non-physical. ---*/
109
110 physical = false;
111
112 }
113
114 /*--- Set the value of the velocity and velocity^2 (requires density) ---*/
115
116 SetVelocity(iPoint);
117
118 /*--- Set specific heats (only necessary for consistency with preconditioning). ---*/
119
120 SetSpecificHeatCp(iPoint, FluidModel->GetCp());
121 SetSpecificHeatCv(iPoint, FluidModel->GetCv());
122
123 return physical;
