94void CFluidScalar::MassToMoleFractions(const su2double* val_scalars) {
95
96 su2double val_scalars_sum{0.0};
97 for (int i_scalar = 0; i_scalar < n_species_mixture - 1; i_scalar++) {
98 massFractions[i_scalar] = val_scalars[i_scalar];
99 val_scalars_sum += val_scalars[i_scalar];
100 }
101 massFractions[n_species_mixture - 1] = 1 - val_scalars_sum;
102
103 su2double mixtureMolarMass{0.0};
104 for (int iVar = 0; iVar < n_species_mixture; iVar++) {
105 mixtureMolarMass += massFractions[iVar] / molarMasses[iVar];
106 }
107
108 for (int iVar = 0; iVar < n_species_mixture; iVar++) {
109 moleFractions[iVar] = (massFractions[iVar] / molarMasses[iVar]) / mixtureMolarMass;
110 }
111}

113su2double CFluidScalar::WilkeViscosity(const su2double* val_scalars) {
114
115 /* Fill laminarViscosity with n_species_mixture viscosity values. */
116 for (int iVar = 0; iVar < n_species_mixture; iVar++) {
117 LaminarViscosityPointers[iVar]->SetViscosity(Temperature, Density);
118 laminarViscosity[iVar] = LaminarViscosityPointers[iVar]->GetViscosity();
119 }
120
121 su2double viscosityMixture = 0.0;
122
123 for (int i = 0; i < n_species_mixture; i++) {
124 su2double wilkeDenumerator = 0.0;
125 for (int j = 0; j < n_species_mixture; j++) {
126 if (j != i) {
127 const su2double phi =
128 pow(1 + sqrt(laminarViscosity[i] / laminarViscosity[j]) * pow(molarMasses[j] / molarMasses[i], 0.25), 2) /
129 sqrt(8 * (1 + molarMasses[i] / molarMasses[j]));
130 wilkeDenumerator += moleFractions[j] * phi;
131 } else {
132 wilkeDenumerator += moleFractions[j];
133 }
134 }
135 const su2double wilkeNumerator = moleFractions[i] * laminarViscosity[i];
136 viscosityMixture += wilkeNumerator / wilkeDenumerator;
137 }
138 return viscosityMixture;
139}

141su2double CFluidScalar::DavidsonViscosity(const su2double* val_scalars) {
142 const su2double A = 0.375;
143
144 for (int iVar = 0; iVar < n_species_mixture; iVar++) {
145 LaminarViscosityPointers[iVar]->SetViscosity(Temperature, Density);
146 laminarViscosity[iVar] = LaminarViscosityPointers[iVar]->GetViscosity();
147 }
148
149 su2double mixtureFractionDenumerator = 0.0;
150 for (int i = 0; i < n_species_mixture; i++) {
151 mixtureFractionDenumerator += moleFractions[i] * sqrt(molarMasses[i]);
152 }
153
154 std::array<su2double, ARRAYSIZE> mixtureFractions;
155 for (int j = 0; j < n_species_mixture; j++) {
156 mixtureFractions[j] = (moleFractions[j] * sqrt(molarMasses[j])) / mixtureFractionDenumerator;
157 }
158
159 su2double fluidity = 0.0;
160 for (int i = 0; i < n_species_mixture; i++) {
161 for (int j = 0; j < n_species_mixture; j++) {
162 const su2double E = (2 * sqrt(molarMasses[i]) * sqrt(molarMasses[j])) / (molarMasses[i] + molarMasses[j]);
163 fluidity +=
164 ((mixtureFractions[i] * mixtureFractions[j]) / (sqrt(laminarViscosity[i]) * sqrt(laminarViscosity[j]))) *
165 pow(E, A);
166 }
167 }
168 return 1.0 / fluidity;
169}

171su2double CFluidScalar::WilkeConductivity(const su2double* val_scalars) {
172
173 for (int iVar = 0; iVar < n_species_mixture; iVar++) {
174 ThermalConductivityPointers[iVar]->SetConductivity(Temperature, Density, Mu, 0.0, 0.0, 0.0, 0.0);
175 laminarThermalConductivity[iVar] = ThermalConductivityPointers[iVar]->GetConductivity();
176 }
177
178 su2double conductivityMixture = 0.0;
179
180 for (int i = 0; i < n_species_mixture; i++) {
181 su2double wilkeDenumerator = 0.0;
182 for (int j = 0; j < n_species_mixture; j++) {
183 if (j != i) {
184 const su2double phi =
185 pow(1 + sqrt(laminarViscosity[i] / laminarViscosity[j]) * pow(molarMasses[j] / molarMasses[i], 0.25), 2) /
186 sqrt(8 * (1 + molarMasses[i] / molarMasses[j]));
187 wilkeDenumerator += moleFractions[j] * phi;
188 } else {
189 wilkeDenumerator += moleFractions[j];
190 }

193 conductivityMixture += wilkeNumerator / wilkeDenumerator;
194 }

198su2double CFluidScalar::ComputeGasConstant() {
199 su2double MeanMolecularWeight = 0.0;
200
201 for (int i = 0; i < n_species_mixture; i++) {
202 MeanMolecularWeight += moleFractions[i] * molarMasses[i] / 1000;
203 }
204 Gas_Constant = UNIVERSAL_GAS_CONSTANT / (GasConstant_Ref * MeanMolecularWeight);
205
206 return Gas_Constant;
207}

209su2double CFluidScalar::ComputeMeanSpecificHeatCp(const su2double* val_scalars) {
210 su2double mean_cp = 0.0;
211
212 for (int i_scalar = 0; i_scalar < n_species_mixture; i_scalar++) {
213 mean_cp += specificHeat[i_scalar] * massFractions[i_scalar];
214 }
215 return mean_cp;

218void CFluidScalar::SetTDState_T(const su2double val_temperature, const su2double* val_scalars) {
219 MassToMoleFractions(val_scalars);
220 ComputeGasConstant();
221 Temperature = val_temperature;
222 Density = Pressure_Thermodynamic / (Temperature * Gas_Constant);
223 Cp = ComputeMeanSpecificHeatCp(val_scalars);
224 Cv = Cp - Gas_Constant;
225
226 if (wilke) {
227 Mu = WilkeViscosity(val_scalars);
228 } else if (davidson) {
229 Mu = DavidsonViscosity(val_scalars);
230 }
231
232 Kt = WilkeConductivity(val_scalars);
233 ComputeMassDiffusivity();
234}
