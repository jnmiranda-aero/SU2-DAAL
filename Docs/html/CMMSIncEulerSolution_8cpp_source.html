32 CMMSIncEulerSolution::CMMSIncEulerSolution(unsigned short val_nDim, unsigned short val_nVar, unsigned short val_iMesh,
33  CConfig* config)
34  : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
35  /*--- Write a message that the solution is initialized for the manufactured
36  solution for the incompressible Navier-Stokes equations. ---*/
37  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
38  cout << endl;
39  cout << "Warning: Fluid properties and solution are being " << endl;
40  cout << " initialized for the manufactured solution " << endl;
41  cout << " of the incompressible Euler equations!!!" << endl;
42  cout << endl << flush;
43  }
44 
45  /*--- Coefficients, needed to determine the solution. ---*/
46  Density = config->GetDensity_FreeStreamND();
47  Temperature = config->GetTemperature_FreeStreamND();
48 
49  /*--- Constants, which describe this manufactured solution. This is a
50  solution where the primitive variables vary as a combination
51  of sine and cosine functions. The solution is from Salari K, and
52  Knupp P, "Code verification by the method of manufactured solutions,"
53  SAND 2000-1444, Sandia National Laboratories, Albuquerque, NM, 2000. ---*/
54 
55  P_0 = 1.0;
56  u_0 = 1.0;
57  v_0 = 1.0;
58  epsilon = 0.001;
59 
60  /*--- Perform some sanity and error checks for this solution here. ---*/
61  if (config->GetTime_Marching() != TIME_MARCHING::STEADY)
62  SU2_MPI::Error("Steady mode must be selected for the MMS incompressible Euler case", CURRENT_FUNCTION);
63 
64  if (Kind_Solver != MAIN_SOLVER::INC_EULER && Kind_Solver != MAIN_SOLVER::INC_NAVIER_STOKES &&
65  Kind_Solver != MAIN_SOLVER::INC_RANS)
66  SU2_MPI::Error("Incompressible flow equations must be selected for the MMS incompressible Euler case",
67  CURRENT_FUNCTION);
68 
69  if (Kind_Solver != MAIN_SOLVER::INC_EULER)
70  SU2_MPI::Error("Euler equations must be selected for the MMS incompressible Euler case", CURRENT_FUNCTION);
71 
72  if (config->GetKind_FluidModel() != CONSTANT_DENSITY)
73  SU2_MPI::Error("Constant density fluid model must be selected for the MMS incompressible Euler case",
74  CURRENT_FUNCTION);
75 
76  if (config->GetEnergy_Equation())
77  SU2_MPI::Error("Energy equation must be disabled (isothermal) for the MMS incompressible Euler case",
78  CURRENT_FUNCTION);
79 }

89 void CMMSIncEulerSolution::GetSolution(const su2double* val_coords, const su2double val_t,
90  su2double* val_solution) const {
91  /* Easier storage of the x- and y-coordinates. */
92  const su2double x = val_coords[0];
93  const su2double y = val_coords[1];
94 
95  /* Compute the primitives from the defined solution. */
96  const su2double u = u_0 * (sin(x * x + y * y) + epsilon);
97  const su2double v = v_0 * (cos(x * x + y * y) + epsilon);
98  const su2double p = P_0 * (sin(x * x + y * y) + 2.0);
99 
100  /* For the incompressible solver, we return the primitive variables
101  directly, as they are used for the working variables in the solver.
102  Note that the implementation below is valid for both 2D and 3D. */
103  val_solution[0] = p;
104  val_solution[1] = u;
105  val_solution[2] = v;
106  val_solution[3] = 0.0;
107  val_solution[nVar - 1] = Temperature;
108 }

110 void CMMSIncEulerSolution::GetMMSSourceTerm(const su2double* val_coords, const su2double val_t,
111  su2double* val_source) const {
112  /*--- Easier storage of the x- and y-coordinates. ---*/
113  const su2double x = val_coords[0];
114  const su2double y = val_coords[1];
115 
116  /*--- The expressions for the source terms are generated
117  automatically by the sympy package in python.---*/
118  val_source[0] = 2 * Density * (u_0 * x * cos(pow(x, 2) + pow(y, 2)) - v_0 * y * sin(pow(x, 2) + pow(y, 2)));
119  val_source[1] = 4 * Density * pow(u_0, 2) * x * (epsilon + sin(pow(x, 2) + pow(y, 2))) * cos(pow(x, 2) + pow(y, 2)) -
120  2 * Density * u_0 * v_0 * y * (epsilon + sin(pow(x, 2) + pow(y, 2))) * sin(pow(x, 2) + pow(y, 2)) +
121  2 * Density * u_0 * v_0 * y * (epsilon + cos(pow(x, 2) + pow(y, 2))) * cos(pow(x, 2) + pow(y, 2)) +
122  2 * P_0 * x * cos(pow(x, 2) + pow(y, 2));
123  val_source[2] = -2 * Density * u_0 * v_0 * x * (epsilon + sin(pow(x, 2) + pow(y, 2))) * sin(pow(x, 2) + pow(y, 2)) +
124  2 * Density * u_0 * v_0 * x * (epsilon + cos(pow(x, 2) + pow(y, 2))) * cos(pow(x, 2) + pow(y, 2)) -
125  4 * Density * pow(v_0, 2) * y * (epsilon + cos(pow(x, 2) + pow(y, 2))) * sin(pow(x, 2) + pow(y, 2)) +
126  2 * P_0 * y * cos(pow(x, 2) + pow(y, 2));
127  val_source[3] = 0.0;
128  val_source[nVar - 1] = 0.0;
129 }
