104 /*!
105 * \brief Helper function, prunes (by setting to zero) small interpolation coefficients,
106 * i.e. <= tolerance*max(abs(coeffs)). The vector is re-scaled such that sum(coeffs)==1.
107 * \param[in] tolerance - Relative pruning tolerance.
108 * \param[in] size - Size of the coefficient vector.
109 * \param[in,out] coeffs - Iterator to start of vector of interpolation coefficients.
110 * \return Number of non-zero coefficients after pruning and correction factor.
111 */
112 template <typename Float, typename Int, class ForwardIt>
113 static pair<Int, Float> PruneSmallCoefficients(Float tolerance, Int size, ForwardIt coeffs) {
114 /*--- Determine the pruning threshold. ---*/
115 Float thresh = 0.0;
116 auto end = coeffs;
117 for (Int i = 0; i < size; ++i) thresh = max(thresh, fabs(*(end++)));
118 thresh *= tolerance;
119
120 /*--- Prune and count non-zeros. ---*/
121 Int numNonZeros = 0;
122 Float coeffSum = 0.0;
123 for (auto it = coeffs; it != end; ++it) {
124 if (fabs(*it) > thresh) { // keep
125 coeffSum += *it;
126 ++numNonZeros;
127 } else {
128 *it = 0.0;
129 } // prune
130 }
131
132 /*--- Correct remaining coefficients, sum must be 1 for conservation. ---*/
133 Float correction = 1.0 / coeffSum;
134 while (coeffs != end) *(coeffs++) *= correction;
135
136 return make_pair(numNonZeros, correction);
137 }
