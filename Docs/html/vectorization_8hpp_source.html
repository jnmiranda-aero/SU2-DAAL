45/*--- Detect preferred SIMD size (bytes). This only covers x86 architectures. ---*/
46#if defined(__AVX512F__)
47constexpr size_t PREFERRED_SIZE = 64;
48#elif defined(__AVX__)
49constexpr size_t PREFERRED_SIZE = 32;
50#elif defined(__SSE2__)
51constexpr size_t PREFERRED_SIZE = 16;
52#else
53constexpr size_t PREFERRED_SIZE = 8;
54#endif
55
56/*!
57 * \brief Convert the SIMD size (bytes) to a lenght (num elems).
58 */
59template <class T>
60constexpr size_t preferredLen() {
61 return PREFERRED_SIZE / sizeof(T);
62}
63
64template <>
65constexpr size_t preferredLen<su2double>() {
66#ifdef CODI_REVERSE_TYPE
67 /*--- Use a SIMD size of 1 for reverse AD, larger sizes increase
68 * the pre-accumulation time with no performance benefit. ---*/
69 return 1;
70#else
71 /*--- For forward AD there is a performance benefit. This covers
72 * forward AD and primal mode (su2double == passivedouble). ---*/
73 return PREFERRED_SIZE / sizeof(passivedouble);
74#endif
75}

77/*!
78 * \class Array
79 * \brief A simple SIMD type relying on implicit vectorization, i.e. done by
80 * the compiler, explicitly vectorized specializations are defined after.
81 * \note This class gets its math operator overloads from CVecExpr, the
82 * specializations do not use expression templates, IF YOU NEED A NEW FUNCTION,
83 * define it both in vector_expressions.hpp and in special_vectorization.hpp.
84 */
85template <class Scalar_t, size_t N = preferredLen<Scalar_t>()>
86class Array : public CVecExpr<Array<Scalar_t, N>, Scalar_t> {
87#define FOREACH for (size_t k = 0; k < N; ++k)
88 static_assert(N > 0, "Invalid SIMD size");
89
90 public:
91 using Scalar = Scalar_t;
92 enum : size_t { Size = N };
93 enum : size_t { Align = Size * sizeof(Scalar) };
94 static constexpr bool StoreAsRef = true;
95
96 private:
97 alignas(Size * sizeof(Scalar)) Scalar x_[N];
98
99 public:
100#define ARRAY_BOILERPLATE \
101 /*!--- Access elements ---*/ \
102 FORCEINLINE Scalar& operator[](size_t k) { return x_[k]; } \
103 FORCEINLINE const Scalar& operator[](size_t k) const { return x_[k]; } \
104 /*!--- Constructors ---*/ \
105 FORCEINLINE Array() = default; \
106 FORCEINLINE Array(Scalar x) { bcast(x); } \
107 FORCEINLINE Array(std::initializer_list<Scalar> vals) { \
108 auto it = vals.begin(); \
109 FOREACH { \
110 x_[k] = *it; \
111 ++it; \
112 } \
113 } \
114 FORCEINLINE Array(Scalar x0, Scalar dx) { FOREACH x_[k] = x0 + k * dx; } \
115 FORCEINLINE Array(const Scalar* ptr) { load(ptr); } \
116 template <class T> \
117 FORCEINLINE Array(const Scalar* beg, const T& off) { \
118 gather(beg, off); \
119 } \
120 /*!--- Reduction operations ---*/ \
121 FORCEINLINE Scalar sum() const { \
122 Scalar s(0); \
123 FOREACH { s += x_[k]; } \
124 return s; \
125 } \
126 FORCEINLINE Scalar dot(const Array& other) const { \
127 Scalar s(0); \
128 FOREACH { s += x_[k] * other[k]; } \
129 return s; \
130 }
131
132#if defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE)
133 /*--- These are not very nice but without them it would not be
134 * possible to assign literals to Arrays of active types. ---*/
135 template <class U = Scalar, su2enable_if<std::is_same<U, su2double>::value> = 0>
136 FORCEINLINE Array(passivedouble x) {
137 bcast(x);
138 }
139 template <class U = Scalar, su2enable_if<std::is_same<U, su2double>::value> = 0>
140 FORCEINLINE Array& operator=(passivedouble x) {
141 bcast(x);
142 return *this;
143 }
144#endif
145
146 ARRAY_BOILERPLATE
147
148 /*! \brief Copy construct from expression. */
149 template <class U>
150 FORCEINLINE Array(const CVecExpr<U, Scalar>& expr) {
151 FOREACH x_[k] = expr.derived()[k];
152 }
153
154 /*--- Implementation of the construction primitives. ---*/
155
156 FORCEINLINE void bcast(Scalar x) { FOREACH x_[k] = x; }
157 FORCEINLINE void load(const Scalar* ptr) { FOREACH x_[k] = ptr[k]; }
158 FORCEINLINE void loada(const Scalar* ptr) { load(ptr); }
159 FORCEINLINE void store(Scalar* ptr) const { FOREACH ptr[k] = x_[k]; }
160 FORCEINLINE void storea(Scalar* ptr) const { store(ptr); }
161 FORCEINLINE void stream(Scalar* ptr) const { store(ptr); }
162 template <class T>
163 FORCEINLINE void gather(const Scalar* begin, const T& offsets) {
164 FOREACH x_[k] = begin[offsets[k]];
165 }
166
167 /*--- Compound assignment operators. ---*/
168
169#define MAKE_COMPOUND(OP) \
170 FORCEINLINE Array& operator OP(Scalar x) { \
171 FOREACH { x_[k] OP x; } \
172 return *this; \
173 } \
174 template <class U> \
175 FORCEINLINE Array& operator OP(const CVecExpr<U, Scalar>& expr) { \
176 FOREACH { x_[k] OP expr.derived()[k]; } \
177 return *this; \
178 }
179 MAKE_COMPOUND(=)
180 MAKE_COMPOUND(+=)
181 MAKE_COMPOUND(-=)
182 MAKE_COMPOUND(*=)
183 MAKE_COMPOUND(/=)
184#undef MAKE_COMPOUND
185
186#undef FOREACH
187};
