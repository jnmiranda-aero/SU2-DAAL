43su2double CEdge::GetVolume(const su2double* coord_Edge_CG, const su2double* coord_FaceElem_CG,
44 const su2double* coord_Elem_CG, const su2double* coord_Point) {
45 constexpr unsigned long nDim = 3;
46
47 su2double vec_a[nDim] = {0.0}, vec_b[nDim] = {0.0}, vec_c[nDim] = {0.0}, vec_d[nDim] = {0.0};
48
49 Distance(nDim, coord_Edge_CG, coord_Point, vec_a);
50 Distance(nDim, coord_FaceElem_CG, coord_Point, vec_b);
51 Distance(nDim, coord_Elem_CG, coord_Point, vec_c);
52
53 CrossProduct(vec_a, vec_b, vec_d);
54
55 return fabs(DotProduct(nDim, vec_c, vec_d)) / 6.0;
56}

58su2double CEdge::GetVolume(const su2double* coord_Edge_CG, const su2double* coord_Elem_CG,
59 const su2double* coord_Point) {
60 constexpr unsigned long nDim = 2;
61
62 su2double vec_a[nDim] = {0.0}, vec_b[nDim] = {0.0};
63
64 Distance(nDim, coord_Elem_CG, coord_Point, vec_a);
65 Distance(nDim, coord_Edge_CG, coord_Point, vec_b);
66
67 return 0.5 * fabs(vec_a[0] * vec_b[1] - vec_a[1] * vec_b[0]);
68}

70void CEdge::SetNodes_Coord(unsigned long iEdge, const su2double* coord_Edge_CG, const su2double* coord_FaceElem_CG,
71 const su2double* coord_Elem_CG) {
72 constexpr unsigned long nDim = 3;
73
74 su2double vec_a[nDim] = {0.0}, vec_b[nDim] = {0.0}, Dim_Normal[nDim];
75
76 Distance(nDim, coord_Elem_CG, coord_Edge_CG, vec_a);
77 Distance(nDim, coord_FaceElem_CG, coord_Edge_CG, vec_b);
78
79 CrossProduct(vec_a, vec_b, Dim_Normal);
80
81 for (auto iDim = 0ul; iDim < nDim; ++iDim) Normal(iEdge, iDim) += 0.5 * Dim_Normal[iDim];
82}
