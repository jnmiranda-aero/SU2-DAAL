32/*!
33 * \class CAvgGrad_Base
34 * \brief A base class for computing viscous terms using an average of gradients.
35 * \details This is the base class for the numerics classes that compute the
36 * viscous fluxes for the flow solvers (i.e. compressible or incompressible
37 * Navier Stokes). The actual numerics classes derive from this class.
38 * This class is used to share functions and variables that are common to all
39 * of the flow viscous numerics. For example, the turbulent stress tensor
40 * is computed identically for all three derived classes.
41 * \ingroup ViscDiscr
42 * \author C. Pederson, A. Bueno, F. Palacios, T. Economon
43 */
44class CAvgGrad_Base : public CNumerics {
45protected:
46 const unsigned short nPrimVar; /*!< \brief The size of the primitive variable array used in the numerics class. */
47 const bool correct_gradient; /*!< \brief Apply a correction to the gradient term */
48 bool implicit = false; /*!< \brief Implicit calculus. */
49 su2double
50 heat_flux_vector[MAXNDIM] = {0.0}, /*!< \brief Flux of total energy due to molecular and turbulent diffusion */
51 *heat_flux_jac_i = nullptr, /*!< \brief Jacobian of the molecular + turbulent heat flux vector, projected onto the normal vector. */
52 **tau_jacobian_i = nullptr; /*!< \brief Jacobian of the viscous + turbulent stress tensor, projected onto the normal vector. */
53 su2double *Mean_PrimVar = nullptr; /*!< \brief Mean primitive variables. */
54 const su2double
55 *PrimVar_i = nullptr,
56 *PrimVar_j = nullptr; /*!< \brief Primitives variables at point i and j. */
57 su2double **Mean_GradPrimVar = nullptr, /*!< \brief Mean value of the gradient. */
58 Mean_Laminar_Viscosity, /*!< \brief Mean value of the viscosity. */
59 Mean_Eddy_Viscosity, /*!< \brief Mean value of the eddy viscosity. */
60 Mean_turb_ke, /*!< \brief Mean value of the turbulent kinetic energy. */
61 Mean_TauWall, /*!< \brief Mean wall shear stress (wall functions). */
62 TauWall_i, TauWall_j, /*!< \brief Wall shear stress at point i and j (wall functions). */
63 dist_ij_2, /*!< \brief Length of the edge and face, squared */
64 Edge_Vector[MAXNDIM] = {0.0}, /*!< \brief Vector from point i to point j. */
65 *Proj_Mean_GradPrimVar_Edge = nullptr; /*!< \brief Inner product of the Mean gradient and the edge vector. */
66
67 su2double** Jacobian_i = nullptr; /*!< \brief The Jacobian w.r.t. point i after computation. */
68 su2double** Jacobian_j = nullptr; /*!< \brief The Jacobian w.r.t. point j after computation. */
69
70 /*!
71 * \brief Scale the stress tensor using a predefined wall stress.
72 *
73 * This function requires that the stress tensor already be
74 * computed using \ref GetStressTensor
75 *
76 * \param[in] UnitNormal - Unit normal vector.
77 * \param[in] TauWall - The wall stress.
78 */
79 void AddTauWall(const su2double *UnitNormal,
80 su2double TauWall);
81
82 /**
83 * \brief Calculate the Jacobian of the viscous + turbulent stress tensor
84 *
85 * This function is intended only for the compressible flow solver.
86 * This Jacobian is projected onto the normal vector, so it is of dimension
87 * [nDim][nVar]
88 *
89 * \param[in] val_Mean_PrimVar - Mean value of the primitive variables.
90 * \param[in] val_laminar_viscosity - Value of the laminar viscosity.
91 * \param[in] val_eddy_viscosity - Value of the eddy viscosity.
92 * \param[in] val_dist_ij - Distance between the points.
93 * \param[in] val_normal - Normal vector, the norm of the vector is the area of the face.
94 */
95 void SetTauJacobian(const su2double* val_Mean_PrimVar,
96 su2double val_laminar_viscosity,
97 su2double val_eddy_viscosity,
98 su2double val_dist_ij,
99 const su2double *val_normal);
100
101
102 /**
103 * \brief Calculate the Jacobian of the viscous and turbulent stress tensor
104 *
105 * This function is intended only for the incompressible flow solver.
106 * This Jacobian is projected onto the normal vector, so it is of dimension
107 * [nDim][nVar]
108 *
109 * \param[in] val_laminar_viscosity - Value of the laminar viscosity.
110 * \param[in] val_eddy_viscosity - Value of the eddy viscosity.

114 void SetIncTauJacobian(su2double val_laminar_viscosity,

116 su2double val_dist_ij,
117 const su2double *val_normal);

119 /*!

128 void GetViscousProjFlux(const su2double *val_primvar,

131 /*!
132 * \brief TSL-Approximation of Viscous NS Jacobians.

143 void GetViscousProjJacs(const su2double *val_Mean_PrimVar,
144 su2double val_dS,

147 su2double **val_Proj_Jac_Tensor_j);

149 /*!
150 * \brief Apply a correction to the gradient to reduce the truncation error
151 *
152 * \param[in] val_PrimVar_i - Primitive variables at point i
153 * \param[in] val_PrimVar_j - Primitive variables at point j
154 * \param[in] val_edge_vector - The vector between points i and j
155 * \param[in] val_dist_ij_2 - The distance between points i and j, squared
156 * \param[in] val_nPrimVar - The number of primitive variables
157 */
158 void CorrectGradient(su2double** GradPrimVar,
159 const su2double* val_PrimVar_i,
160 const su2double* val_PrimVar_j,

162 su2double val_dist_ij_2,
163 const unsigned short val_nPrimVar);
164
165public:
166
167 /*!
168 * \brief Constructor of the class.
169 * \param[in] val_nDim - Number of dimension of the problem.
170 * \param[in] val_nVar - Number of variables of the problem.
171 * \param[in] val_nPrimVar - Number of primitive variables to use.
172 * \param[in] val_correct_grad - Apply a correction to the gradient
173 * \param[in] config - Definition of the particular problem.
174 */
175 CAvgGrad_Base(unsigned short val_nDim, unsigned short val_nVar,
176 unsigned short val_nPrimVar,
177 bool val_correct_grad, const CConfig* config);
178
179 /*!
180 * \brief Destructor of the class.

182 ~CAvgGrad_Base() override;
183
184 /*!
185 * \brief Set the value of the wall shear stress at point i and j (wall functions).
186 * \param[in] val_tauwall_i - Value of the wall shear stress at point i.
187 * \param[in] val_tauwall_j - Value of the wall shear stress at point j.
188 */
189 inline void SetTau_Wall(su2double val_tauwall_i, su2double val_tauwall_j) override {
190 TauWall_i = val_tauwall_i;
191 TauWall_j = val_tauwall_j;
192 }
193
194 /*!
195 * \brief Calculate the viscous + turbulent stress tensor
196 * \param[in] val_primvar - Primitive variables.
197 * \param[in] val_gradprimvar - Gradient of the primitive variables.
198 * \param[in] val_turb_ke - Turbulent kinetic energy
199 * \param[in] val_laminar_viscosity - Laminar viscosity.
200 * \param[in] val_eddy_viscosity - Eddy viscosity.
201 */
202 void SetStressTensor(const su2double *val_primvar,
203 const su2double* const *val_gradprimvar,
204 su2double val_turb_ke,
205 su2double val_laminar_viscosity,
206 su2double val_eddy_viscosity);
207
208 /*!
209 * \brief Get a component of the viscous stress tensor.
210 *
211 * \param[in] iDim - The first index
212 * \param[in] jDim - The second index
213 * \return The component of the viscous stress tensor at iDim, jDim
214 */
215 inline su2double GetStressTensor(unsigned short iDim, unsigned short jDim) const { return tau[iDim][jDim];}
216
217 /*!
218 * \brief Get a component of the heat flux vector.
219 * \param[in] iDim - The index of the component
220 * \return The component of the heat flux vector at iDim
221 */
222 inline su2double GetHeatFluxVector(unsigned short iDim) const { return heat_flux_vector[iDim]; }
223
224};
