34 CFlowIncOutput::CFlowIncOutput(CConfig *config, unsigned short nDim) : CFlowOutput(config, nDim, false) {
35 
36  turb_model = config->GetKind_Turb_Model();
37 
38  heat = config->GetEnergy_Equation();
39 
40  weakly_coupled_heat = config->GetWeakly_Coupled_Heat();
41  flamelet = (config->GetKind_Species_Model() == SPECIES_MODEL::FLAMELET);
42 
43  streamwisePeriodic = (config->GetKind_Streamwise_Periodic() != ENUM_STREAMWISE_PERIODIC::NONE);
44  streamwisePeriodic_temperature = config->GetStreamwise_Periodic_Temperature();
45 
46  /*--- Set the default history fields if nothing is set in the config file ---*/
47 
48  if (nRequestedHistoryFields == 0){
49  RequestCommonHistory(config->GetTime_Domain());
50  nRequestedHistoryFields = requestedHistoryFields.size();
51  }
52 
53  if (nRequestedScreenFields == 0){
54  if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
55  requestedScreenFields.emplace_back("INNER_ITER");
56  requestedScreenFields.emplace_back("RMS_PRESSURE");
57  requestedScreenFields.emplace_back("RMS_VELOCITY-X");
58  requestedScreenFields.emplace_back("RMS_VELOCITY-Y");
59  nRequestedScreenFields = requestedScreenFields.size();
60  }
61 
62  if (nRequestedVolumeFields == 0){
63  requestedVolumeFields.emplace_back("COORDINATES");
64  requestedVolumeFields.emplace_back("SOLUTION");
65  requestedVolumeFields.emplace_back("PRIMITIVE");
66  nRequestedVolumeFields = requestedVolumeFields.size();
67  }
68 
69  if (gridMovement) {
70  auto notFound = requestedVolumeFields.end();
71  if (find(requestedVolumeFields.begin(), notFound, string("GRID_VELOCITY")) == notFound) {
72  requestedVolumeFields.emplace_back("GRID_VELOCITY");
73  nRequestedVolumeFields ++;
74  }
75  }
76 
77  stringstream ss;
78  ss << "Zone " << config->GetiZone() << " (Incomp. Fluid)";
79  multiZoneHeaderString = ss.str();
80 
81  /*--- Set the volume filename --- */
82 
83  volumeFilename = config->GetVolume_FileName();
84 
85  /*--- Set the surface filename --- */
86 
87  surfaceFilename = config->GetSurfCoeff_FileName();
88 
89  /*--- Set the restart filename --- */
90 
91  restartFilename = config->GetRestart_FileName();
92 
93  /*--- Set the default convergence field --- */
94 
95  if (convFields.empty()) convFields.emplace_back("RMS_PRESSURE");
96 
97 }

196 void CFlowIncOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
197 
198  CSolver* flow_solver = solver[FLOW_SOL];
199  CSolver* heat_solver = solver[HEAT_SOL];
200  CSolver* rad_solver = solver[RAD_SOL];
201  CSolver* mesh_solver = solver[MESH_SOL];
202 
203  SetHistoryOutputValue("RMS_PRESSURE", log10(flow_solver->GetRes_RMS(0)));
204  SetHistoryOutputValue("RMS_VELOCITY-X", log10(flow_solver->GetRes_RMS(1)));
205  SetHistoryOutputValue("RMS_VELOCITY-Y", log10(flow_solver->GetRes_RMS(2)));
206  if (nDim == 3) SetHistoryOutputValue("RMS_VELOCITY-Z", log10(flow_solver->GetRes_RMS(3)));
207 
208  if (config->AddRadiation())
209  SetHistoryOutputValue("RMS_RAD_ENERGY", log10(rad_solver->GetRes_RMS(0)));
210 
211  SetHistoryOutputValue("MAX_PRESSURE", log10(flow_solver->GetRes_Max(0)));
212  SetHistoryOutputValue("MAX_VELOCITY-X", log10(flow_solver->GetRes_Max(1)));
213  SetHistoryOutputValue("MAX_VELOCITY-Y", log10(flow_solver->GetRes_Max(2)));
214  if (nDim == 3) SetHistoryOutputValue("RMS_VELOCITY-Z", log10(flow_solver->GetRes_Max(3)));
215 
216  if (multiZone){
217  SetHistoryOutputValue("BGS_PRESSURE", log10(flow_solver->GetRes_BGS(0)));
218  SetHistoryOutputValue("BGS_VELOCITY-X", log10(flow_solver->GetRes_BGS(1)));
219  SetHistoryOutputValue("BGS_VELOCITY-Y", log10(flow_solver->GetRes_BGS(2)));
220  if (nDim == 3) SetHistoryOutputValue("BGS_VELOCITY-Z", log10(flow_solver->GetRes_BGS(3)));
221  if (config->AddRadiation())
222  SetHistoryOutputValue("BGS_RAD_ENERGY", log10(rad_solver->GetRes_BGS(0)));
223  }
224 
225  if (weakly_coupled_heat){
226  SetHeatCoefficients(config, heat_solver);
227  SetHistoryOutputValue("AVG_TEMPERATURE", heat_solver->GetTotal_AvgTemperature());
228  SetHistoryOutputValue("RMS_TEMPERATURE", log10(heat_solver->GetRes_RMS(0)));
229  SetHistoryOutputValue("MAX_TEMPERATURE", log10(heat_solver->GetRes_Max(0)));
230  if (multiZone) SetHistoryOutputValue("BGS_TEMPERATURE", log10(heat_solver->GetRes_BGS(0)));
231  }
232  if (heat) {
233  SetHeatCoefficients(config, flow_solver);
234  SetHistoryOutputValue("AVG_TEMPERATURE", flow_solver->GetTotal_AvgTemperature());
235  SetHistoryOutputValue("RMS_TEMPERATURE", log10(flow_solver->GetRes_RMS(nDim + 1)));
236  SetHistoryOutputValue("MAX_TEMPERATURE", log10(flow_solver->GetRes_Max(nDim + 1)));
237  if (multiZone) {
238  SetHistoryOutputValue("BGS_TEMPERATURE", log10(flow_solver->GetRes_BGS(nDim + 1)));
239  }
240  }
241 
242  SetHistoryOutputValue("LINSOL_ITER", flow_solver->GetIterLinSolver());
243  SetHistoryOutputValue("LINSOL_RESIDUAL", log10(flow_solver->GetResLinSolver()));
244 
245  if (config->GetDeform_Mesh()){
246  SetHistoryOutputValue("DEFORM_MIN_VOLUME", mesh_solver->GetMinimum_Volume());
247  SetHistoryOutputValue("DEFORM_MAX_VOLUME", mesh_solver->GetMaximum_Volume());
248  SetHistoryOutputValue("DEFORM_ITER", mesh_solver->GetIterLinSolver());
249  SetHistoryOutputValue("DEFORM_RESIDUAL", log10(mesh_solver->GetResLinSolver()));
250  }
251 
252  SetHistoryOutputValue("MIN_DELTA_TIME", flow_solver->GetMin_Delta_Time());
253  SetHistoryOutputValue("MAX_DELTA_TIME", flow_solver->GetMax_Delta_Time());
254 
255  SetHistoryOutputValue("MIN_CFL", flow_solver->GetMin_CFL_Local());
256  SetHistoryOutputValue("MAX_CFL", flow_solver->GetMax_CFL_Local());
257  SetHistoryOutputValue("AVG_CFL", flow_solver->GetAvg_CFL_Local());
258 
259  LoadHistoryDataScalar(config, solver);
260 
261  if(streamwisePeriodic) {
262  SetHistoryOutputValue("STREAMWISE_MASSFLOW", flow_solver->GetStreamwisePeriodicValues().Streamwise_Periodic_MassFlow);
263  SetHistoryOutputValue("STREAMWISE_DP", flow_solver->GetStreamwisePeriodicValues().Streamwise_Periodic_PressureDrop);
264  SetHistoryOutputValue("STREAMWISE_HEAT", flow_solver->GetStreamwisePeriodicValues().Streamwise_Periodic_IntegratedHeatFlow);
265  }
266 
267  /*--- Set the analyse surface history values --- */
268 
269  SetAnalyzeSurface(solver, geometry, config, false);
270 
271  /*--- Set aeroydnamic coefficients --- */
272 
273  SetAerodynamicCoefficients(config, flow_solver);
274 
275  /*--- Set rotating frame coefficients --- */
276 
277  SetRotatingFrameCoefficients(flow_solver);
278 
279  /*--- Keep this as last, since it uses the history values that were set. ---*/
280 
281  SetCustomOutputs(solver, geometry, config);
282 
283  SetCustomAndComboObjectives(FLOW_SOL, config, solver);
284 
285 }

383 void CFlowIncOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
384 
385  const auto* Node_Flow = solver[FLOW_SOL]->GetNodes();
386  const CVariable* Node_Heat = nullptr;
387  const CVariable* Node_Rad = nullptr;
388  auto* Node_Geo = geometry->nodes;
389 
390  if (weakly_coupled_heat){
391  Node_Heat = solver[HEAT_SOL]->GetNodes();
392  }
393 
394  LoadCoordinates(Node_Geo->GetCoord(iPoint), iPoint);
395 
396  SetVolumeOutputValue("PRESSURE", iPoint, Node_Flow->GetSolution(iPoint, 0));
397  SetVolumeOutputValue("VELOCITY-X", iPoint, Node_Flow->GetSolution(iPoint, 1));
398  SetVolumeOutputValue("VELOCITY-Y", iPoint, Node_Flow->GetSolution(iPoint, 2));
399  if (nDim == 3)
400  SetVolumeOutputValue("VELOCITY-Z", iPoint, Node_Flow->GetSolution(iPoint, 3));
401 
402  if (heat || flamelet) SetVolumeOutputValue("TEMPERATURE", iPoint, Node_Flow->GetSolution(iPoint, nDim+1));
403  if (weakly_coupled_heat) SetVolumeOutputValue("TEMPERATURE", iPoint, Node_Heat->GetSolution(iPoint, 0));
404 
405  // Radiation solver
406  if (config->AddRadiation()){
407  Node_Rad = solver[RAD_SOL]->GetNodes();
408  SetVolumeOutputValue("P1-RAD", iPoint, Node_Rad->GetSolution(iPoint,0));
409  }
410 
411  if (gridMovement){
412  SetVolumeOutputValue("GRID_VELOCITY-X", iPoint, Node_Geo->GetGridVel(iPoint)[0]);
413  SetVolumeOutputValue("GRID_VELOCITY-Y", iPoint, Node_Geo->GetGridVel(iPoint)[1]);
414  if (nDim == 3)
415  SetVolumeOutputValue("GRID_VELOCITY-Z", iPoint, Node_Geo->GetGridVel(iPoint)[2]);
416  }
417 
418  const su2double factor = solver[FLOW_SOL]->GetReferenceDynamicPressure();
419  SetVolumeOutputValue("PRESSURE_COEFF", iPoint, (Node_Flow->GetPressure(iPoint) - solver[FLOW_SOL]->GetPressure_Inf())/factor);
420  SetVolumeOutputValue("DENSITY", iPoint, Node_Flow->GetDensity(iPoint));
421 
422  if (config->GetKind_Solver() == MAIN_SOLVER::INC_RANS || config->GetKind_Solver() == MAIN_SOLVER::INC_NAVIER_STOKES){
423  SetVolumeOutputValue("LAMINAR_VISCOSITY", iPoint, Node_Flow->GetLaminarViscosity(iPoint));
424  SetVolumeOutputValue("HEAT_CAPACITY", iPoint, Node_Flow->GetSpecificHeatCp(iPoint));
425  SetVolumeOutputValue("THERMAL_CONDUCTIVITY", iPoint, Node_Flow->GetThermalConductivity(iPoint));
426  }
427 
428  SetVolumeOutputValue("RES_PRESSURE", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 0));
429  SetVolumeOutputValue("RES_VELOCITY-X", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 1));
430  SetVolumeOutputValue("RES_VELOCITY-Y", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 2));
431  if (nDim == 3)
432  SetVolumeOutputValue("RES_VELOCITY-Z", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 3));
433  if (config->GetEnergy_Equation())
434  SetVolumeOutputValue("RES_TEMPERATURE", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nDim+1));
435 
436  if (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE && config->GetKind_SlopeLimit_Flow() != LIMITER::VAN_ALBADA_EDGE) {
437  SetVolumeOutputValue("LIMITER_PRESSURE", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 0));
