37 class CADTComparePointClass {
38  private:
39  const su2double* pointCoor; /*!< \brief Pointer to the coordinates of the points. */
40  const unsigned short splitDirection; /*!< \brief Split direction used in the sorting. */
41  const unsigned short nDim; /*!< \brief Number of spatial dimensions stored in the coordinates. */
42 
43  public:
44  /*!
45  * \brief Constructor of the class. The member variables are initialized.
46  * \param[in] coor Pointer to the coordinates of the points.
47  * \param[in] splitDir Direction that must be used to sort the coordinates.
48  * \param[in] nDimADT Number of spatial dimensions of the ADT and coordinates.
49  */
50  CADTComparePointClass(const su2double* coor, const unsigned short splitDir, const unsigned short nDimADT)
51  : pointCoor(coor), splitDirection(splitDir), nDim(nDimADT) {}
52 
53  /*!
54  * \brief Operator used for the sorting of the points.
55  * \param[in] p0 Index of the first point to be compared.
56  * \param[in] p1 Index of the second point to be compared.
57  */
58  inline bool operator()(const unsigned long p0, const unsigned long p1) const {
59  return pointCoor[nDim * p0 + splitDirection] < pointCoor[nDim * p1 + splitDirection];
60  }
61

65  CADTComparePointClass() = delete;
