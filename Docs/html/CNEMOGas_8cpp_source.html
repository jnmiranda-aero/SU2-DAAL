59void CNEMOGas::SetTDStatePTTv(su2double val_pressure, const su2double *val_massfrac,
60 su2double val_temperature, su2double val_temperature_ve){
61
62 for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
63 MassFrac[iSpecies] = val_massfrac[iSpecies];
64 Pressure = val_pressure;
65 T = val_temperature;
66 Tve = val_temperature_ve;
67
68 su2double denom = 0.0;
69
70 /*--- Calculate mixture density from supplied primitive quantities ---*/
71 for (iSpecies = 0; iSpecies < nEl; iSpecies++)
72 denom += MassFrac[iSpecies] * (Ru/MolarMass[iSpecies]) * Tve;
73 for (iSpecies = nEl; iSpecies < nSpecies; iSpecies++)
74 denom += MassFrac[iSpecies] * (Ru/MolarMass[iSpecies]) * T;
75 Density = Pressure / denom;
76
77 for (iSpecies = 0; iSpecies < nSpecies; iSpecies++){
78 rhos[iSpecies] = MassFrac[iSpecies]*Density;
79 MassFrac[iSpecies] = rhos[iSpecies]/Density;
80 }
81}

83su2double CNEMOGas::ComputeSoundSpeed(){
84
85 su2double conc = 0.0;
86 su2double rhoCvtr = 0.0;
87 Density = 0.0;
88
89 auto& Cvtrs = GetSpeciesCvTraRot();
90
91 for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
92 Density+=rhos[iSpecies];
93
94 for (iSpecies = nEl; iSpecies < nSpecies; iSpecies++){
95 conc += rhos[iSpecies]/MolarMass[iSpecies];
96 rhoCvtr += rhos[iSpecies] * Cvtrs[iSpecies];
97 }
98 SoundSpeed2 = (1.0 + Ru/rhoCvtr*conc) * Pressure/Density;
99
100 return(sqrt(SoundSpeed2));
101}

104
105 su2double P = 0.0;
106
107 for (iSpecies = 0; iSpecies < nEl; iSpecies++)
108 P += rhos[iSpecies] * Ru/MolarMass[iSpecies] * Tve;
109 for (iSpecies = nEl; iSpecies < nSpecies; iSpecies++)
110 P += rhos[iSpecies] * Ru/MolarMass[iSpecies] * T;
111
112 Pressure = P;
113

129su2double CNEMOGas::ComputeGamma(){
130
131 /*--- Extract Values ---*/
132 rhoCvtr = ComputerhoCvtr();
133
134 /*--- Gamma Computation ---*/
135 su2double rhoR = 0.0;
136 for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
137 rhoR += rhos[iSpecies]*Ru/MolarMass[iSpecies];
138
139 gamma = rhoR/rhoCvtr+1;

156void CNEMOGas::ComputedPdU(const su2double *V, const vector<su2double>& val_eves, su2double *val_dPdU){
157
158 // Note: Electron energy not included properly.
159
160 if (val_dPdU == nullptr) {
161 SU2_MPI::Error("Array dPdU not allocated!", CURRENT_FUNCTION);
162 }
163
164 /*--- Determine the electron density (if ionized) ---*/
165 su2double rho_el = 0.0;
166 if (ionization) { rho_el = rhos[0]; }
167
168 /*--- Necessary indexes to assess primitive variables ---*/
169 unsigned long RHOS_INDEX = 0;
170 unsigned long T_INDEX = nSpecies;
171 unsigned long VEL_INDEX = nSpecies+2;
172 unsigned long RHOCVTR_INDEX = nSpecies+nDim+6;
173 unsigned long RHOCVVE_INDEX = nSpecies+nDim+7;
174
175 /*--- Extract variables ---*/
176 for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
177 rhos[iSpecies] = V[RHOS_INDEX+iSpecies];
178
179 Cvtrs = GetSpeciesCvTraRot();
180 Enthalpy_Formation = GetSpeciesFormationEnthalpy();
181 Ref_Temperature = GetRefTemperature();
182
183 /*--- Rename for convenience ---*/
184 su2double rhoCvtr = V[RHOCVTR_INDEX];
185 su2double rhoCvve = V[RHOCVVE_INDEX];
186 T = V[T_INDEX];
187
188 /*--- Pre-compute useful quantities ---*/
189 su2double CvtrBAR = 0.0;
190 su2double sqvel = 0.0;
191 su2double conc = 0.0;
192 for (iDim = 0; iDim < nDim; iDim++)
193 sqvel += V[VEL_INDEX+iDim] * V[VEL_INDEX+iDim];
194 for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
195 CvtrBAR += rhos[iSpecies]*Cvtrs[iSpecies];
196 conc += rhos[iSpecies]/MolarMass[iSpecies];
197 }
198
199 /*--- Species density derivatives ---*/
200 su2double ef = 0.0;
201 for (iSpecies = nEl; iSpecies < nHeavy; iSpecies++) {
202 ef = Enthalpy_Formation[iSpecies] - Ru/MolarMass[iSpecies]*Ref_Temperature[iSpecies];
203 val_dPdU[iSpecies] = T*Ru/MolarMass[iSpecies] + Ru*conc/rhoCvtr *
204 (-Cvtrs[iSpecies]*(T-Ref_Temperature[iSpecies]) -
205 ef + 0.5*sqvel);
206 }
207
208 if (ionization) {
209 for (iSpecies = nEl; iSpecies < nSpecies; iSpecies++) {
210 // evibs = Ru/MolarMass[iSpecies] * thetav[iSpecies]/(exp(thetav[iSpecies]/Tve)-1.0);
211 // num = 0.0;
212 // denom = g[iSpecies][0] * exp(-thetae[iSpecies][0]/Tve);
213 // for (iEl = 1; iEl < nElStates[iSpecies]; iEl++) {
214 // num += g[iSpecies][iEl] * thetae[iSpecies][iEl] * exp(-thetae[iSpecies][iEl]/Tve);
215 // denom += g[iSpecies][iEl] * exp(-thetae[iSpecies][iEl]/Tve);
216 // }
217 // eels = Ru/MolarMass[iSpecies] * (num/denom);
218
219 val_dPdU[iSpecies] -= rho_el * Ru/MolarMass[0] * (val_eves[iSpecies])/rhoCvve;
220 }
221 ef = Enthalpy_Formation[0] - Ru/MolarMass[0]*Ref_Temperature[0];
222 val_dPdU[0] = Ru*conc/rhoCvtr * (-ef + 0.5*sqvel) + Ru/MolarMass[0]*Tve
223 - rho_el*Ru/MolarMass[0] * (-3.0/2.0*Ru/MolarMass[0]*Tve)/rhoCvve;
224 }
225
226 /*--- Momentum derivatives ---*/
227 for (iDim = 0; iDim < nDim; iDim++)
228 val_dPdU[nSpecies+iDim] = -conc*Ru*V[VEL_INDEX+iDim]/rhoCvtr;
229
230 /*--- Total energy derivative ---*/
231 val_dPdU[nSpecies+nDim] = conc*Ru / rhoCvtr;
232
233 /*--- Vib.-el energy derivative ---*/
234 val_dPdU[nSpecies+nDim+1] = -val_dPdU[nSpecies+nDim] +
235 rho_el*Ru/MolarMass[0]*1.0/rhoCvve;
236
237}

239void CNEMOGas::ComputedTdU(const su2double *V, su2double *val_dTdU){
240
241 /*--- Necessary indexes to assess primitive variables ---*/
242 const unsigned long T_INDEX = nSpecies;
243 const unsigned long VEL_INDEX = nSpecies+2;
244 const unsigned long RHOCVTR_INDEX = nSpecies+nDim+6;
245
246 /*--- Rename for convenience ---*/
247 T = V[T_INDEX];
248 su2double rhoCvtr = V[RHOCVTR_INDEX];
249
250 Cvtrs = GetSpeciesCvTraRot();
251 Enthalpy_Formation = GetSpeciesFormationEnthalpy();
252 Ref_Temperature = GetRefTemperature();
253
254 /*--- Calculate supporting quantities ---*/
255 const su2double v2 = GeometryToolbox::SquaredNorm(nDim, &V[VEL_INDEX]);
256
257 /*--- Species density derivatives ---*/
258 for (iSpecies = nEl; iSpecies < nSpecies; iSpecies++) {
259 su2double ef = Enthalpy_Formation[iSpecies] - Ru/MolarMass[iSpecies]*Ref_Temperature[iSpecies];
260 val_dTdU[iSpecies] = (-ef + 0.5*v2 + Cvtrs[iSpecies]*(Ref_Temperature[iSpecies]-T)) / rhoCvtr;
261 }
262
263 /*--- Momentum derivatives ---*/
264 for (iDim = 0; iDim < nDim; iDim++)
265 val_dTdU[nSpecies+iDim] = -V[VEL_INDEX+iDim] / V[RHOCVTR_INDEX];
266
267 /*--- Energy derivatives ---*/
268 val_dTdU[nSpecies+nDim] = 1.0 / V[RHOCVTR_INDEX];
269 val_dTdU[nSpecies+nDim+1] = -1.0 / V[RHOCVTR_INDEX];
270
271}

273void CNEMOGas::ComputedTvedU(const su2double *V, const vector<su2double>& val_eves, su2double *val_dTvedU){
274
275 /*--- Necessary indexes to assess primitive variables ---*/
276 unsigned long RHOCVVE_INDEX = nSpecies+nDim+7;
277
278 /*--- Rename for convenience ---*/
279 su2double rhoCvve = V[RHOCVVE_INDEX];
280
281 /*--- Species density derivatives ---*/
282 for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
283 val_dTvedU[iSpecies] = -val_eves[iSpecies]/rhoCvve;
284 }
285 /*--- Momentum derivatives ---*/
286 for (iDim = 0; iDim < nDim; iDim++)
287 val_dTvedU[nSpecies+iDim] = 0.0;
288
289 /*--- Energy derivatives ---*/
290 val_dTvedU[nSpecies+nDim] = 0.0;
291 val_dTvedU[nSpecies+nDim+1] = 1.0 / rhoCvve;
