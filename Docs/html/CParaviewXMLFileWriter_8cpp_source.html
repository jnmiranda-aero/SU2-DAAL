49void CParaviewXMLFileWriter::WriteData(string val_filename){
50
51 if (!dataSorter->GetConnectivitySorted()){
52 SU2_MPI::Error("Connectivity must be sorted.", CURRENT_FUNCTION);
53 }
54
55 /*--- We always have 3 coords, independent of the actual value of nDim ---*/
56
57 const int NCOORDS = 3;
58 const unsigned short nDim = dataSorter->GetnDim();
59 unsigned short iDim = 0;
60
61 /*--- Array containing the field names we want to output ---*/
62
63 const vector<string>& fieldNames = dataSorter->GetFieldNames();
64
65 unsigned long iPoint, iElem;
66
67 char str_buf[255];
68
69 OpenMPIFile(val_filename);
70
71 dataOffset = 0;
72
73 /*--- Communicate the number of total points that will be
74 written by each rank. After this communication, each proc knows how
75 many poinnts will be written before its location in the file and the
76 offsets can be correctly set. ---*/
77
78 unsigned long myPoint, GlobalPoint;
79
80 GlobalPoint = dataSorter->GetnPointsGlobal();
81 myPoint = dataSorter->GetnPoints();
82
83 /*--- Compute our local number of elements, the required storage,
84 and reduce the total number of elements and storage globally. ---*/
85
86 unsigned long myElem, myElemStorage, GlobalElem, GlobalElemStorage;
87
88 unsigned long nParallel_Line = dataSorter->GetnElem(LINE),
89 nParallel_Tria = dataSorter->GetnElem(TRIANGLE),
90 nParallel_Quad = dataSorter->GetnElem(QUADRILATERAL),
91 nParallel_Tetr = dataSorter->GetnElem(TETRAHEDRON),
92 nParallel_Hexa = dataSorter->GetnElem(HEXAHEDRON),
93 nParallel_Pris = dataSorter->GetnElem(PRISM),
94 nParallel_Pyra = dataSorter->GetnElem(PYRAMID);
95
96 myElem = dataSorter->GetnElem();
97 myElemStorage = dataSorter->GetnConn();
98 GlobalElem = dataSorter->GetnElemGlobal();
99 GlobalElemStorage = dataSorter->GetnConnGlobal();
100
101 /* Write the ASCII XML header. Note that we use the appended format for the data,
102 * which means that all data is appended at the end of the file in one binary blob.
103 */
104
105 if (!bigEndian){
106 WriteMPIString("<VTKFile type=\"UnstructuredGrid\" version=\"1.0\" byte_order=\"LittleEndian\" header_type=\"UInt64\">\n", MASTER_NODE);
107 } else {
108 WriteMPIString("<VTKFile type=\"UnstructuredGrid\" version=\"1.0\" byte_order=\"BigEndian\" header_type=\"UInt64\">\n", MASTER_NODE);
109 }
110
111 WriteMPIString("<UnstructuredGrid>\n", MASTER_NODE);
112
113 SPRINTF(str_buf, "<Piece NumberOfPoints=\"%i\" NumberOfCells=\"%i\">\n",
114 SU2_TYPE::Int(GlobalPoint), SU2_TYPE::Int(GlobalElem));
115
116 WriteMPIString(std::string(str_buf), MASTER_NODE);
117 WriteMPIString("<Points>\n", MASTER_NODE);
118 AddDataArray(VTKDatatype::FLOAT32, "", NCOORDS, myPoint*NCOORDS, GlobalPoint*NCOORDS);
119 WriteMPIString("</Points>\n", MASTER_NODE);
120 WriteMPIString("<Cells>\n", MASTER_NODE);
121 AddDataArray(VTKDatatype::INT32, "connectivity", 1, myElemStorage, GlobalElemStorage);
122 AddDataArray(VTKDatatype::INT32, "offsets", 1, myElem, GlobalElem);
123 AddDataArray(VTKDatatype::UINT8, "types", 1, myElem, GlobalElem);
124 WriteMPIString("</Cells>\n", MASTER_NODE);
125
126 WriteMPIString("<PointData>\n", MASTER_NODE);
127
128 /*--- Adjust container start location to avoid point coords. ---*/
129
130 unsigned short varStart = 2;
131 if (nDim == 3) varStart++;
132
133 /*--- Loop over all variables that have been registered in the output. ---*/
134
135 unsigned short iField, VarCounter = varStart;
136 for (iField = varStart; iField < fieldNames.size(); iField++) {
137
138 string fieldname = fieldNames[iField];
139 fieldname.erase(remove(fieldname.begin(), fieldname.end(), '"'),
140 fieldname.end());
141
142 /*--- Check whether this field is a vector or scalar. ---*/
143
144 bool output_variable = true, isVector = false;
145 size_t found = fieldNames[iField].find("_x");
146 if (found!=string::npos) {
147 output_variable = true;
148 isVector = true;
149 }
150 found = fieldNames[iField].find("_y");
151 if (found!=string::npos) {
152 /*--- We have found a vector, so skip the Y component. ---*/
153 output_variable = false;
154 VarCounter++;
155 }
156 found = fieldNames[iField].find("_z");
157 if (found!=string::npos) {
158 /*--- We have found a vector, so skip the Z component. ---*/
159 output_variable = false;
160 VarCounter++;
161 }
162
163 /*--- Write the point data as an <X,Y,Z> vector or a scalar. ---*/
164
165 if (output_variable && isVector) {
166
167 /*--- Adjust the string name to remove the leading "X-" ---*/
168
169 fieldname.erase(fieldname.end()-2,fieldname.end());
170
171 AddDataArray(VTKDatatype::FLOAT32, fieldname, NCOORDS, myPoint*NCOORDS, GlobalPoint*NCOORDS);
172
173 } else if (output_variable) {
174
175 AddDataArray(VTKDatatype::FLOAT32, fieldname, 1, myPoint, GlobalPoint);
176
177 }
178
179 }
180 WriteMPIString("</PointData>\n", MASTER_NODE);
181 WriteMPIString("</Piece>\n", MASTER_NODE);
182 WriteMPIString("</UnstructuredGrid>\n", MASTER_NODE);
183
184 /*--- Now write all the data we have previously defined into the binary section of the file ---*/
185
186 WriteMPIString("<AppendedData encoding=\"raw\">\n_", MASTER_NODE);
187
188 /*--- Load/write the 1D buffer of point coordinates. Note that we
189 always have 3 coordinate dimensions, even for 2D problems. ---*/
190
191 vector<float> dataBufferFloat(myPoint*NCOORDS);
192 for (iPoint = 0; iPoint < myPoint; iPoint++) {
193 for (iDim = 0; iDim < NCOORDS; iDim++) {
194 if (nDim == 2 && iDim == 2) {
195 dataBufferFloat[iPoint*NCOORDS + iDim] = 0.0;
196 } else {
197 auto val = (float)dataSorter->GetData(iDim, iPoint);
198 dataBufferFloat[iPoint*NCOORDS + iDim] = val;
199 }
200 }
201 }
202
203 WriteDataArray(dataBufferFloat.data(), VTKDatatype::FLOAT32, NCOORDS*myPoint, GlobalPoint*NCOORDS,
204 dataSorter->GetnPointCumulative(rank)*NCOORDS);
205
206 /*--- Load/write 1D buffers for the connectivity of each element type. ---*/
207
208 vector<int> connBuf(myElemStorage);
209 vector<int> offsetBuf(myElem);
210 unsigned long iStorage = 0, iElemID = 0;
211 unsigned short iNode = 0;
212
213 auto copyToBuffer = [&](GEO_TYPE type, unsigned long nElem, unsigned short nPoints){
214 for (iElem = 0; iElem < nElem; iElem++) {
215 for (iNode = 0; iNode < nPoints; iNode++){
216 connBuf[iStorage+iNode] = int(dataSorter->GetElemConnectivity(type, iElem, iNode)-1);
217 }
218 iStorage += nPoints;
219 offsetBuf[iElemID++] = int(iStorage + dataSorter->GetnElemConnCumulative(rank));
220 }
221 };
222
223 copyToBuffer(LINE, nParallel_Line, N_POINTS_LINE);
224 copyToBuffer(TRIANGLE, nParallel_Tria, N_POINTS_TRIANGLE);
225 copyToBuffer(QUADRILATERAL, nParallel_Quad, N_POINTS_QUADRILATERAL);
226 copyToBuffer(TETRAHEDRON, nParallel_Tetr, N_POINTS_TETRAHEDRON);
227 copyToBuffer(HEXAHEDRON, nParallel_Hexa, N_POINTS_HEXAHEDRON);
228 copyToBuffer(PRISM, nParallel_Pris, N_POINTS_PRISM);
229 copyToBuffer(PYRAMID, nParallel_Pyra, N_POINTS_PYRAMID);
230
231 WriteDataArray(connBuf.data(), VTKDatatype::INT32, myElemStorage, GlobalElemStorage,
232 dataSorter->GetnElemConnCumulative(rank));
233 WriteDataArray(offsetBuf.data(), VTKDatatype::INT32, myElem, GlobalElem, dataSorter->GetnElemCumulative(rank));
234
235 /*--- Load/write the cell type for all elements in the file. ---*/
236
237 vector<uint8_t> typeBuf(myElem);
238 auto typeIter = typeBuf.begin();
239
240 std::fill(typeIter, typeIter+nParallel_Line, LINE); typeIter += nParallel_Line;
241 std::fill(typeIter, typeIter+nParallel_Tria, TRIANGLE); typeIter += nParallel_Tria;
242 std::fill(typeIter, typeIter+nParallel_Quad, QUADRILATERAL); typeIter += nParallel_Quad;
243 std::fill(typeIter, typeIter+nParallel_Tetr, TETRAHEDRON); typeIter += nParallel_Tetr;
244 std::fill(typeIter, typeIter+nParallel_Hexa, HEXAHEDRON); typeIter += nParallel_Hexa;
245 std::fill(typeIter, typeIter+nParallel_Pris, PRISM); typeIter += nParallel_Pris;
246 std::fill(typeIter, typeIter+nParallel_Pyra, PYRAMID); typeIter += nParallel_Pyra;
247
248 WriteDataArray(typeBuf.data(), VTKDatatype::UINT8, myElem, GlobalElem, dataSorter->GetnElemCumulative(rank));
249
250 /*--- Loop over all variables that have been registered in the output. ---*/
251
252 VarCounter = varStart;
253 for (iField = varStart; iField < fieldNames.size(); iField++) {
254
255 /*--- Check whether this field is a vector or scalar. ---*/
256
257 bool output_variable = true, isVector = false;
258 size_t found = fieldNames[iField].find("_x");
259 if (found!=string::npos) {
260 output_variable = true;
261 isVector = true;
262 }
263 found = fieldNames[iField].find("_y");
264 if (found!=string::npos) {
265 /*--- We have found a vector, so skip the Y component. ---*/
266 output_variable = false;
267 VarCounter++;
268 }
269 found = fieldNames[iField].find("_z");
270 if (found!=string::npos) {
271 /*--- We have found a vector, so skip the Z component. ---*/
272 output_variable = false;
273 VarCounter++;
274 }
275
276 /*--- Write the point data as an <X,Y,Z> vector or a scalar. ---*/
277
278 if (output_variable && isVector) {
279
280 /*--- Load up the buffer for writing this rank's vector data. ---*/
281
282 float val = 0.0;
283 for (iPoint = 0; iPoint < myPoint; iPoint++) {
284 for (iDim = 0; iDim < NCOORDS; iDim++) {
285 if (nDim == 2 && iDim == 2) {
286 dataBufferFloat[iPoint*NCOORDS + iDim] = 0.0;
287 } else {
288 val = (float)dataSorter->GetData(VarCounter+iDim,iPoint);
289 dataBufferFloat[iPoint*NCOORDS + iDim] = val;
290 }
291 }
292 }
293
294 WriteDataArray(dataBufferFloat.data(), VTKDatatype::FLOAT32, myPoint*NCOORDS, GlobalPoint*NCOORDS,
295 dataSorter->GetnPointCumulative(rank)*NCOORDS);
296
297 VarCounter++;
298
299 } else if (output_variable) {
300
301
302 /*--- For now, create a temp 1D buffer to load up the data for writing.
303 This will be replaced with a derived data type most likely. ---*/
304
305 for (iPoint = 0; iPoint < myPoint; iPoint++) {
306 auto val = (float)dataSorter->GetData(VarCounter,iPoint);
307 dataBufferFloat[iPoint] = val;
308 }
309
310 WriteDataArray(dataBufferFloat.data(), VTKDatatype::FLOAT32, myPoint, GlobalPoint,
311 dataSorter->GetnPointCumulative(rank));
312
313 VarCounter++;
314 }
315
316 }
317
318 WriteMPIString("</AppendedData>\n", MASTER_NODE);
319 WriteMPIString("</VTKFile>\n", MASTER_NODE);
320
321 CloseMPIFile();
322
323}

325void CParaviewXMLFileWriter::WriteDataArray(void* data, VTKDatatype type, unsigned long arraySize,
326 unsigned long globalSize, unsigned long offset){
327
328 std::string typeStr;
329 unsigned long typeSize = 0;
330
331 GetTypeInfo(type, typeStr, typeSize);
332
333 /*--- Compute the size of the data to write in bytes ---*/
334
335 size_t byteSize = arraySize*typeSize;
336
337 /*--- The total data size ---*/
338 size_t totalByteSize = globalSize*typeSize;
339
340 /*--- Only the master node writes the total size in bytes as unsigned long in front of the array data ---*/
341
342 if (!WriteMPIBinaryData(&totalByteSize, sizeof(size_t), MASTER_NODE)){
343 SU2_MPI::Error("Writing array size failed", CURRENT_FUNCTION);
344 }
345
346 /*--- Collectively write all the data ---*/
347
348 if (!WriteMPIBinaryDataAll(data, byteSize, totalByteSize, offset*typeSize)){
349 SU2_MPI::Error("Writing data array failed", CURRENT_FUNCTION);
350 }
351}

353void CParaviewXMLFileWriter::AddDataArray(VTKDatatype type, string name,
354 unsigned short nComponents, unsigned long arraySize, unsigned long globalSize){
355
356 /*--- Add quotation marks around the arguments ---*/
357
358 name = "\"" + name + "\"";
359
360 string nComp ="\"" + PrintingToolbox::to_string(nComponents) + "\"";
361
362 /*--- Full precision ASCII output of offset for 32 bit integer ---*/
363
364 stringstream ss;
365 ss.precision(10); ss.setf(std::ios::fixed, std::ios::floatfield);
366 ss << "\"" << dataOffset << "\"";
367 string offsetStr = ss.str();
368
369 std::string typeStr;
370 unsigned long typeSize = 0;
371
372 GetTypeInfo(type, typeStr, typeSize);
373
374 /*--- Total data size ---*/
375
376 size_t totalByteSize = globalSize*typeSize;
377
378 /*--- Write the ASCII XML header information for this array ---*/
379
380 WriteMPIString(string("<DataArray type=") + typeStr +
381 string(" Name=") + name +
382 string(" NumberOfComponents= ") + nComp +
383 string(" offset=") + offsetStr +
384 string(" format=\"appended\"/>\n"), MASTER_NODE);
385
386 dataOffset += totalByteSize + sizeof(size_t);
387
388}
