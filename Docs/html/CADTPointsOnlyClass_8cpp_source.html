32 CADTPointsOnlyClass::CADTPointsOnlyClass(unsigned short nDim, unsigned long nPoints, const su2double* coor,
33  const unsigned long* pointID, const bool globalTree) {
34  /* Allocate some thread-safe working variables if required. */
35 #ifdef HAVE_OMP
36  FrontLeaves.resize(omp_get_max_threads());
37  FrontLeavesNew.resize(omp_get_max_threads());
38 #endif
39 
40  /*--- Make a distinction between parallel and sequential mode. ---*/
41 
42 #ifdef HAVE_MPI
43 
44  /* Parallel mode. Check whether a global or a local tree must be built. */
45  if (globalTree) {
46  /*--- A global tree must be built. All points are gathered on all ranks.
47  First determine the number of points per rank and store them in such
48  a way that the info can be used directly in Allgatherv. ---*/
49  int rank, size;
50  SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &rank);
51  SU2_MPI::Comm_size(SU2_MPI::GetComm(), &size);
52 
53  vector<int> recvCounts(size), displs(size);
54  int sizeLocal = (int)nPoints;
55 
56  SU2_MPI::Allgather(&sizeLocal, 1, MPI_INT, recvCounts.data(), 1, MPI_INT, SU2_MPI::GetComm());
57  displs[0] = 0;
58  for (int i = 1; i < size; ++i) displs[i] = displs[i - 1] + recvCounts[i - 1];
59 
60  int sizeGlobal = displs.back() + recvCounts.back();
61 
62  /*--- Gather the local pointID's and the ranks of the nodes on all ranks. ---*/
63  localPointIDs.resize(sizeGlobal);
64  SU2_MPI::Allgatherv(pointID, sizeLocal, MPI_UNSIGNED_LONG, localPointIDs.data(), recvCounts.data(), displs.data(),
65  MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
66 
67  ranksOfPoints.resize(sizeGlobal);
68  vector<int> rankLocal(sizeLocal, rank);
69  SU2_MPI::Allgatherv(rankLocal.data(), sizeLocal, MPI_INT, ranksOfPoints.data(), recvCounts.data(), displs.data(),
70  MPI_INT, SU2_MPI::GetComm());
71 
72  /*--- Gather the coordinates of the points on all ranks. ---*/
73  for (int i = 0; i < size; ++i) {
74  recvCounts[i] *= nDim;
75  displs[i] *= nDim;
76  }
77 
78  coorPoints.resize(nDim * sizeGlobal);
79  SU2_MPI::Allgatherv(coor, nDim * sizeLocal, MPI_DOUBLE, coorPoints.data(), recvCounts.data(), displs.data(),
80  MPI_DOUBLE, SU2_MPI::GetComm());
81  } else {
82  /*--- A local tree must be built. Copy the coordinates and point IDs and
83  set the ranks to the rank of this processor. ---*/
84  int rank;
85  SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &rank);
86 
87  coorPoints.assign(coor, coor + nDim * nPoints);
88  localPointIDs.assign(pointID, pointID + nPoints);
89  ranksOfPoints.assign(nPoints, rank);
90  }
91 
92 #else
93 
94  /*--- Sequential mode. Copy the coordinates and point IDs and
95  set the ranks to MASTER_NODE. ---*/
96  coorPoints.assign(coor, coor + nDim * nPoints);
97  localPointIDs.assign(pointID, pointID + nPoints);
98  ranksOfPoints.assign(nPoints, MASTER_NODE);
99 
100 #endif
101 
102  /*--- Build the tree. ---*/
103  BuildADT(nDim, localPointIDs.size(), coorPoints.data());
104 
105  /*--- Reserve the memory for frontLeaves and frontLeavesNew,
106  which are needed during the tree search. ---*/
107  for (auto& vec : FrontLeaves) vec.reserve(200);
108  for (auto& vec : FrontLeavesNew) vec.reserve(200);
109 }

111 void CADTPointsOnlyClass::DetermineNearestNode_impl(vector<unsigned long>& frontLeaves,
112  vector<unsigned long>& frontLeavesNew, const su2double* coor,
113  su2double& dist, unsigned long& pointID, int& rankID) const {
114  const bool wasActive = AD::BeginPassive();
115 
116  /*--------------------------------------------------------------------------*/
117  /*--- Step 1: Initialize the nearest node to the central node of the ---*/
118  /*--- root leaf. Note that the distance is the distance squared ---*/
119  /*--- to avoid a sqrt. ---*/
120  /*--------------------------------------------------------------------------*/
121 
122  unsigned long kk = leaves[0].centralNodeID, minIndex;
123  const su2double* coorTarget = coorPoints.data() + nDimADT * kk;
124 
125  pointID = localPointIDs[kk];
126  rankID = ranksOfPoints[kk];
127  minIndex = kk;
128  dist = 0.0;
129  for (unsigned short l = 0; l < nDimADT; ++l) {
130  const su2double ds = coor[l] - coorTarget[l];
131  dist += ds * ds;
132  }
133 
134  /*--------------------------------------------------------------------------*/
135  /*--- Step 2: Traverse the tree and search for the nearest node. ---*/
136  /*--- During the tree traversal the currently stored distance ---*/
137  /*--- squared is modified, because the guaranteed minimum ---*/
138  /*--- distance squared of the children could be smaller. ---*/
139  /*--------------------------------------------------------------------------*/
140 
141  /* Start at the root leaf of the ADT, i.e. initialize frontLeaves such that
142  it only contains the root leaf. Make sure to wipe out any data from a
143  previous search. */
144  frontLeaves.clear();
145  frontLeaves.push_back(0);
146 
147  /* Infinite loop of the tree traversal. */
148  for (;;) {
149  /* Initialize the new front, i.e. the front for the next round, to empty. */
150  frontLeavesNew.clear();
151 
152  /* Loop over the leaves of the current front. */
153  for (unsigned long i = 0; i < frontLeaves.size(); ++i) {
154  /* Store the current leaf a bit easier in ll and loop over its children. */
155  const unsigned long ll = frontLeaves[i];
156  for (unsigned short mm = 0; mm < 2; ++mm) {
157  /* Determine whether this child contains a node or a leaf
158  of the next level of the ADT. */
159  kk = leaves[ll].children[mm];
160  if (leaves[ll].childrenAreTerminal[mm]) {
161  /*--- Child contains a node. Compute the distance squared to this node
162  and store it if this distance squared is less than the currently
163  stored value. ---*/
164  coorTarget = coorPoints.data() + nDimADT * kk;
165  su2double distTarget = 0;
166  for (unsigned short l = 0; l < nDimADT; ++l) {
167  const su2double ds = coor[l] - coorTarget[l];
168  distTarget += ds * ds;
169  }
170 
171  if (distTarget < dist) {
172  dist = distTarget;
173  pointID = localPointIDs[kk];
174  rankID = ranksOfPoints[kk];
175  minIndex = kk;
176  }
177  } else {
178  /*--- Child contains a leaf. Determine the possible minimum distance
179  squared to that leaf. ---*/
180  su2double posDist = 0.0;
181  for (unsigned short l = 0; l < nDimADT; ++l) {
182  su2double ds = 0.0;
183  if (coor[l] < leaves[kk].xMin[l])
184  ds = coor[l] - leaves[kk].xMin[l];
185  else if (coor[l] > leaves[kk].xMax[l])
186  ds = coor[l] - leaves[kk].xMax[l];
187 
188  posDist += ds * ds;
189  }
190 
191  /*--- Check if the possible minimum distance is less than the currently
192  stored minimum distance. If so this leaf must be stored for the
193  next round. In that case the distance squared to the central node is
194  determined, which is used to update the currently stored value. ---*/
195  if (posDist < dist) {
196  frontLeavesNew.push_back(kk);
197 
198  const unsigned long jj = leaves[kk].centralNodeID;
199 
200  coorTarget = coorPoints.data() + nDimADT * jj;
201  su2double distTarget = 0;
202  for (unsigned short l = 0; l < nDimADT; ++l) {
203  const su2double ds = coor[l] - coorTarget[l];
204  distTarget += ds * ds;
205  }
206 
207  if (distTarget < dist) {
208  dist = distTarget;
209  pointID = localPointIDs[jj];
210  rankID = ranksOfPoints[jj];
211  minIndex = jj;
212  }
213  }
214  }
215  }
216  }
217 
218  /*--- End of the loop over the current front. Copy the data from
219  frontLeavesNew to frontLeaves for the next round. If the new front
220  is empty the entire tree has been traversed and a break can be made
221  from the infinite loop. ---*/
222  frontLeaves = frontLeavesNew;
223  if (frontLeaves.empty()) break;
224  }
225 
226  AD::EndPassive(wasActive);
227 
228  /* Recompute the distance to get the correct dependency if we use AD */
229  coorTarget = coorPoints.data() + nDimADT * minIndex;
230  dist = 0.0;
231  for (unsigned short l = 0; l < nDimADT; ++l) {
232  const su2double ds = coor[l] - coorTarget[l];
233  dist += ds * ds;
234  }
235 
236  /* At the moment the distance squared to the nearest node is stored.
237  Take the sqrt to obtain the correct value. */
238  dist = sqrt(dist);
239 }
