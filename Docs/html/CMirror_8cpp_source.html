33CMirror::CMirror(CGeometry**** geometry_container, const CConfig* const* config, const CInterpolator* interpolator,
34 unsigned int iZone, unsigned int jZone)
35 : CInterpolator(geometry_container, config, iZone, jZone), transpInterpolator(interpolator) {
36 using PrintingToolbox::to_string;
37 if (jZone < iZone) {
38 SU2_MPI::Error(string("The order of the zones does not allow conservative interpolation to be setup.\n"
39 "Swap zones ") +

41 CURRENT_FUNCTION);
42 }
43 SetTransferCoeff(config);

46void CMirror::SetTransferCoeff(const CConfig* const* config) {
47 const int nProcessor = size;
48
49 vector<unsigned long> allNumVertexTarget(nProcessor);
50 vector<unsigned long> allNumVertexDonor(nProcessor);
51 vector<unsigned long> allNumNodeDonor(nProcessor);
52
53 /*--- The target vertex information of the transpose interpolator. ---*/
54 const auto& donorVertices = transpInterpolator->targetVertices;
55
56 targetVertices.resize(config[targetZone]->GetnMarker_All());
57
58 /*--- Number of markers on the interface ---*/
59 const auto nMarkerInt = (config[targetZone]->GetMarker_n_ZoneInterface()) / 2;
60
61 /*--- For the number of markers on the interface... ---*/
62 for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
63 /* High level procedure:
64 * - Gather the interpolation matrix of the donor geometry;
65 * - Set the interpolation matrix of the target as the transpose.
66 */
67
68 /*--- On the donor side: find the tag of the boundary sharing the interface ---*/
69 const auto markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
70
71 /*--- On the target side: find the tag of the boundary sharing the interface ---*/
72 const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
73
74 /*--- Checks if the zone contains the interface, if not continue to the next step ---*/
75 if (!CheckInterfaceBoundary(markDonor, markTarget)) continue;
76
77 unsigned long nVertexDonor = 0, nVertexTarget = 0;
78 if (markDonor != -1) nVertexDonor = donor_geometry->GetnVertex(markDonor);
79 if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex(markTarget);
80
81 /*--- Count the number of donor nodes on the donor geometry. ---*/
82 unsigned long nVertexDonorLocal = 0;
83 unsigned long nNodeDonorLocal = 0;
84 for (auto iVertex = 0ul; iVertex < nVertexDonor; iVertex++) {
85 const auto iPoint = donor_geometry->vertex[markDonor][iVertex]->GetNode();
86 if (donor_geometry->nodes->GetDomain(iPoint)) {
87 nNodeDonorLocal += donorVertices[markDonor][iVertex].nDonor();
88 nVertexDonorLocal++;
89 }
90 }
91
92 /*--- Communicate vertex and donor node counts. ---*/
93 SU2_MPI::Allgather(&nVertexTarget, 1, MPI_UNSIGNED_LONG, allNumVertexTarget.data(), 1, MPI_UNSIGNED_LONG,
94 SU2_MPI::GetComm());
95 SU2_MPI::Allgather(&nVertexDonorLocal, 1, MPI_UNSIGNED_LONG, allNumVertexDonor.data(), 1, MPI_UNSIGNED_LONG,
96 SU2_MPI::GetComm());
97 SU2_MPI::Allgather(&nNodeDonorLocal, 1, MPI_UNSIGNED_LONG, allNumNodeDonor.data(), 1, MPI_UNSIGNED_LONG,
98 SU2_MPI::GetComm());
99
100 /*--- Copy donor interpolation matrix (triplet format). ---*/
101 vector<long> sendGlobalIndex(nNodeDonorLocal);
102 vector<long> sendDonorIndex(nNodeDonorLocal);
103 vector<su2double> sendDonorCoeff(nNodeDonorLocal);
104
105 for (auto iVertex = 0ul, iDonor = 0ul; iVertex < nVertexDonor; ++iVertex) {
106 auto& donor_vertex = donorVertices[markDonor][iVertex];
107 const auto iPoint = donor_geometry->vertex[markDonor][iVertex]->GetNode();
108
109 if (!donor_geometry->nodes->GetDomain(iPoint)) continue;
110
111 const auto nDonor = donor_vertex.nDonor();
112 const auto donorGlobalIndex = donor_geometry->nodes->GetGlobalIndex(iPoint);
113
114 for (auto i = 0u; i < nDonor; ++i) {
115 sendGlobalIndex[iDonor] = donorGlobalIndex;
116 sendDonorIndex[iDonor] = donor_vertex.globalPoint[i];
117 sendDonorCoeff[iDonor] = donor_vertex.coefficient[i];
118 ++iDonor;
119 }
120 }
121
122 /*--- Sort the matrix by donor index, effectively transposing the triplets. ---*/
123 vector<int> order(nNodeDonorLocal);
124 iota(order.begin(), order.end(), 0);
125 sort(order.begin(), order.end(), [&sendDonorIndex](int i, int j) { return sendDonorIndex[i] < sendDonorIndex[j]; });
126 for (int i = 0; i < int(nNodeDonorLocal); ++i) {
127 int j = order[i];
128 while (j < i) j = order[j];
129 if (i == j) continue;
130 swap(sendGlobalIndex[i], sendGlobalIndex[j]);
131 swap(sendDonorIndex[i], sendDonorIndex[j]);
132 swap(sendDonorCoeff[i], sendDonorCoeff[j]);
133 }
134 vector<int>().swap(order); // no longer needed
135
136 /*--- Communicate donor interpolation matrix and info. We only gather the
137 * matrix in ranks that need it, i.e. have target vertices, to avoid
138 * potentially huge memory usage when running in parallel. This is an
139 * in-place Gatherv (done manually due to AD issues in the past). ---*/
140
141 vector<int> iSendProcessor;
142 for (int iProcessor = 0; iProcessor < nProcessor; ++iProcessor)
143 if (allNumVertexDonor[iProcessor] != 0) iSendProcessor.push_back(iProcessor);
144
145 const int nSend = iSendProcessor.size();
146
147 vector<long*> GlobalIndex(nSend, nullptr), DonorIndex(nSend, nullptr);
148 vector<su2double*> DonorCoeff(nSend, nullptr);
149
150 /*--- For each "target processor" that needs the interpolation matrix. ---*/
151 for (int iProcessor = 0; iProcessor < nProcessor; ++iProcessor) {
152 if (allNumVertexTarget[iProcessor] == 0) continue;
153
154 /*--- For each "donor processor that contain a part of the matrix. ---*/
155 for (int iSend = 0; iSend < nSend; ++iSend) {
156 const auto jProcessor = iSendProcessor[iSend];
157 const auto numCoeff = allNumNodeDonor[jProcessor];
158
159 if ((rank == iProcessor) && (rank == jProcessor)) {
160 /*--- "Self" communication. ---*/
161 GlobalIndex[iSend] = sendGlobalIndex.data();
162 DonorIndex[iSend] = sendDonorIndex.data();
163 DonorCoeff[iSend] = sendDonorCoeff.data();
164 } else if (rank == iProcessor) {
165 /*--- "I'm" the target, allocate and receive. ---*/
166 GlobalIndex[iSend] = new long[numCoeff];
167 DonorIndex[iSend] = new long[numCoeff];
168 DonorCoeff[iSend] = new su2double[numCoeff];
169 SU2_MPI::Recv(GlobalIndex[iSend], numCoeff, MPI_LONG, jProcessor, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
170 SU2_MPI::Recv(DonorIndex[iSend], numCoeff, MPI_LONG, jProcessor, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
171 SU2_MPI::Recv(DonorCoeff[iSend], numCoeff, MPI_DOUBLE, jProcessor, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
172 } else if (rank == jProcessor) {
173 /*--- "I'm" the donor, send. ---*/
174 SU2_MPI::Send(sendGlobalIndex.data(), numCoeff, MPI_LONG, iProcessor, 0, SU2_MPI::GetComm());
175 SU2_MPI::Send(sendDonorIndex.data(), numCoeff, MPI_LONG, iProcessor, 0, SU2_MPI::GetComm());
176 SU2_MPI::Send(sendDonorCoeff.data(), numCoeff, MPI_DOUBLE, iProcessor, 0, SU2_MPI::GetComm());
177 }
178 }
179 }
180
181 if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
182
183 /*--- Loop over the vertices on the target marker, define one row of the transpose matrix. ---*/
184
185 SU2_OMP_PARALLEL {
186 SU2_OMP_FOR_DYN(roundUpDiv(nVertexTarget, 2 * omp_get_max_threads()))
187 for (auto iVertex = 0ul; iVertex < nVertexTarget; ++iVertex) {
188 auto& target_vertex = targetVertices[markTarget][iVertex];
189 const auto iPoint = target_geometry->vertex[markTarget][iVertex]->GetNode();
190
191 if (!target_geometry->nodes->GetDomain(iPoint)) continue;
192
193 /*--- Any point of the donor geometry, that has this target point as a donor, becomes a donor. ---*/
194 const long targetGlobalIndex = target_geometry->nodes->GetGlobalIndex(iPoint);
195
196 /*--- Count donors and safe the binary search results (this is why we sorted the matrix). ---*/
197 auto nDonor = 0ul;
198 vector<pair<long*, long*> > ranges(nSend);
199 for (int iSend = 0; iSend < nSend; ++iSend) {
200 const auto iProcessor = iSendProcessor[iSend];
201 const auto numCoeff = allNumNodeDonor[iProcessor];
202 auto p = equal_range(DonorIndex[iSend], DonorIndex[iSend] + numCoeff, targetGlobalIndex);
203 nDonor += (p.second - p.first);
204 ranges[iSend] = p;
205 }
206
207 target_vertex.resize(nDonor);
208
209 /*--- Use the search results to set the interpolation coefficients. ---*/
210 for (int iSend = 0, iDonor = 0; iSend < nSend; ++iSend) {
211 const auto iProcessor = iSendProcessor[iSend];
212
213 const auto first = ranges[iSend].first - DonorIndex[iSend];
214 const auto last = ranges[iSend].second - DonorIndex[iSend];
215
216 for (auto iCoeff = first; iCoeff < last; ++iCoeff) {
217 target_vertex.processor[iDonor] = iProcessor;
218 target_vertex.coefficient[iDonor] = DonorCoeff[iSend][iCoeff];
219 target_vertex.globalPoint[iDonor] = GlobalIndex[iSend][iCoeff];
220 ++iDonor;
221 }
222 }
223 }
224 END_SU2_OMP_FOR
225 }
226 END_SU2_OMP_PARALLEL
227
228 /*--- Free the heap allocations. ---*/
229 for (auto ptr : GlobalIndex)
230 if (ptr != sendGlobalIndex.data()) delete[] ptr;
231 for (auto ptr : DonorIndex)
232 if (ptr != sendDonorIndex.data()) delete[] ptr;
233 for (auto ptr : DonorCoeff)
234 if (ptr != sendDonorCoeff.data()) delete[] ptr;
235
236 } // end marker loop
237}
