237 void CFEASolver::HybridParallelInitialization(CGeometry* geometry) {
238 #ifdef HAVE_OMP
239  /*--- Get the element coloring. ---*/
240 
241  su2double parallelEff = 1.0;
242  const auto& coloring = geometry->GetElementColoring(&parallelEff);
243 
244  /*--- If the coloring is too bad use lock-guarded accesses
245  * to CSysMatrix/Vector in element loops instead. ---*/
246  LockStrategy = parallelEff < COLORING_EFF_THRESH;
247 
248  /*--- When using locks force a single color to reduce the color loop overhead. ---*/
249  if (LockStrategy && (coloring.getOuterSize()>1))
250  geometry->SetNaturalElementColoring();
251 
252  if (!coloring.empty()) {
253  /*--- We are not constrained by the color group size when using locks. ---*/
254  auto groupSize = LockStrategy? 1ul : geometry->GetElementColorGroupSize();
255  auto nColor = coloring.getOuterSize();
256  ElemColoring.reserve(nColor);
257 
258  for(auto iColor = 0ul; iColor < nColor; ++iColor)
259  ElemColoring.emplace_back(coloring.innerIdx(iColor), coloring.getNumNonZeros(iColor), groupSize);
260  }
261 
262  su2double minEff = 1.0;
263  SU2_MPI::Reduce(&parallelEff, &minEff, 1, MPI_DOUBLE, MPI_MIN, MASTER_NODE, SU2_MPI::GetComm());
264 
265  if (minEff < COLORING_EFF_THRESH) {
266  cout << "WARNING: The element coloring efficiency was " << minEff << ", a fallback strategy is in use.\n"
267  << " Better performance may be possible by reducing the number of threads per rank." << endl;
268  }
269 
270  if (LockStrategy) {
271  UpdateLocks.resize(nPoint);
272  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++)
273  omp_init_lock(&UpdateLocks[iPoint]);
274  }
275 
276  omp_chunk_size = computeStaticChunkSize(nPointDomain, omp_get_max_threads(), OMP_MAX_SIZE);
277 #else
278  ElemColoring[0] = DummyGridColor<>(nElement);
279 #endif
280 }

282 void CFEASolver::Set_ElementProperties(CGeometry *geometry, CConfig *config) {
283 
284  const auto iZone = config->GetiZone();
285  const auto nZone = geometry->GetnZone();
286 
287  const bool topology_mode = config->GetTopology_Optimization();
288 
289  element_properties = new CProperty*[nElement];
290 
291  /*--- Restart the solution from file information ---*/
292 
293  auto filename = config->GetFEA_FileName();
294 
295  /*--- If multizone, append zone name ---*/
296  if (nZone > 1)
297  filename = config->GetMultizone_FileName(filename, iZone, ".dat");
298 
299  if (rank == MASTER_NODE) cout << "Filename: " << filename << "." << endl;
300 
301  ifstream properties_file;
302  properties_file.open(filename);
303 
304  /*--- In case there is no file, all elements get the same property (0) ---*/
305 
306  if (properties_file.fail()) {
307  if (rank == MASTER_NODE){
308  cout << "There is no element-based properties file." << endl;
309  cout << "The structural domain has uniform properties." << endl;
310 
311  if (topology_mode)
312  SU2_MPI::Error("Topology mode requires an element-based properties file.",CURRENT_FUNCTION);
313  }
314 
315  for (auto iElem = 0ul; iElem < nElement; iElem++){
316  element_properties[iElem] = new CElementProperty(FEA_TERM, 0, 0, 0);
317  }
318 
319  element_based = false;
320 
321  }
322  else{
323 
324  element_based = true;
325 
326  /*--- In case this is a parallel simulation, we need to perform the Global2Local index transformation first. ---*/
327 
328  unordered_map<unsigned long, unsigned long> Global2Local;
329 
330  for (auto iElem = 0ul; iElem < nElement; iElem++)
331  Global2Local[geometry->elem[iElem]->GetGlobalIndex()] = iElem;
332 
333  /*--- Read all lines in the restart file ---*/
334 
335  unsigned long iElem_Global_Local = 0, iElem_Global = 0; string text_line;
336 
337  /*--- The first line is the header ---*/
338 
339  getline (properties_file, text_line);
340 
341  for (iElem_Global = 0; iElem_Global < geometry->GetGlobal_nElemDomain(); iElem_Global++ ) {
342 
343  getline (properties_file, text_line);
344 
345  istringstream point_line(text_line);
346 
347  /*--- Retrieve local index. If this element from the restart file lives
348  only on a different processor, the value of iPoint_Local will be -1.
349  Otherwise, the local index for this node on the current processor
350  will be returned and used to instantiate the vars. ---*/
351 
352  auto it = Global2Local.find(iElem_Global);
353 
354  if (it != Global2Local.end()) {
355 
356  auto iElem_Local = it->second;
357 
358  unsigned long elProperties[4], index;
359 
360  if (config->GetAdvanced_FEAElementBased() || topology_mode){
361  point_line >> index >> elProperties[0] >> elProperties[1] >> elProperties[2] >> elProperties[3];
362  element_properties[iElem_Local] = new CElementProperty(
363  elProperties[0], elProperties[1], elProperties[2], elProperties[3]);
364  }
365  else{
366  point_line >> index >> elProperties[0];
367  element_properties[iElem_Local] = new CElementProperty(0, elProperties[0], 0, 0);
368  }
369 
370 
371  /*--- For backwards compatibility we only read a fifth column in topology mode ---*/
372  if (topology_mode) {
373  su2double elDensity;
374  point_line >> elDensity;
375  element_properties[iElem_Local]->SetDesignDensity(elDensity);
376  }
377 
378  iElem_Global_Local++;
379  }
380 
381  }
382 
383  /*--- Detect a wrong solution file ---*/
384 
385  if (iElem_Global_Local != nElement) {
386  SU2_MPI::Error("The properties file " + filename + " doesn't match with the mesh file!\n"
387  "It could be empty lines at the end of the file.", CURRENT_FUNCTION);
388  }
389 
390  }

557 void CFEASolver::Preprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, CNumerics **numerics,
558  unsigned short iMesh, unsigned long Iteration, unsigned short RunTime_EqSystem, bool Output) {
559 
560  const bool dynamic = config->GetTime_Domain();
561  const bool disc_adj_fem = (config->GetKind_Solver() == MAIN_SOLVER::DISC_ADJ_FEM);
562  const bool body_forces = config->GetDeadLoad();
563  const bool topology_mode = config->GetTopology_Optimization();
564 
565  /*
566  * For topology optimization we apply a filter on the design density field to avoid
567  * numerical issues (checkerboards), ensure mesh independence, and impose a length scale.
568  * This has to be done before computing the mass matrix and the dead load terms.
569  * This filter, and the volume fraction objective function, require the element volumes,
570  * so we ask "geometry" to compute them.
571  * This only needs to be done for the undeformed (initial) shape.
572  */
573  if (topology_mode && !topol_filter_applied) {
574  geometry->SetElemVolume();
575  FilterElementDensities(geometry, config);
576  topol_filter_applied = true;
577  }
578 
579  /*
580  * If the problem is dynamic we need a mass matrix, which will be constant along the calculation
581  * both for linear and nonlinear analysis, i.e. only computed once on the first time step.
582  *
583  * The same with the integration constants, as for now we consider the time step to be constant.
584  */
585  if (dynamic && (initial_calc || disc_adj_fem)) {
586  Compute_IntegrationConstants(config);
587  Compute_MassMatrix(geometry, numerics, config);
588  }
589 
590  /*
591  * If body forces are taken into account, we need to compute the term that goes into the residual,
592  * which will be constant along the calculation both for linear and nonlinear analysis.
593  *
594  * Only initialized once, at the first iteration of the first time step.
595  */
596  if (body_forces && (initial_calc || disc_adj_fem))
597  Compute_DeadLoad(geometry, numerics, config);
598 
599  /*--- Clear the linear system solution. ---*/
600  SU2_OMP_PARALLEL
601  {
602  LinSysSol.SetValZero();
603  }
604  END_SU2_OMP_PARALLEL
605 
606  /*--- Clear external forces. ---*/
607  nodes->Clear_SurfaceLoad_Res();
608 
609  /*--- Next call to Preprocessing will not be "initial_calc" and linear operations will not be repeated. ---*/
610  initial_calc = false;
611 
612 }
