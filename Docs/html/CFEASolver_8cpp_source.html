50CFEASolver::CFEASolver(CGeometry *geometry, CConfig *config) : CFEASolverBase(geometry, config) {
51
52 bool dynamic = (config->GetTime_Domain());
53 config->SetDelta_UnstTimeND(config->GetDelta_UnstTime());
54
55 /*--- Test whether we consider dielectric elastomers ---*/
56 bool de_effects = config->GetDE_Effects();
57
58 /*--- A priori we don't have an element-based input file (most of the applications will be like this) ---*/
59 element_based = false;
60 topol_filter_applied = false;
61 initial_calc = true;
62
63 /*--- Here is where we assign the kind of each element ---*/
64
65 SU2_OMP_PARALLEL
66 {
67 const int offset = omp_get_thread_num()*MAX_FE_KINDS;
68
69 if (nDim == 2) {
70 /*--- Basic terms ---*/
71 element_container[FEA_TERM][EL_TRIA+offset] = new CTRIA1();
72 element_container[FEA_TERM][EL_QUAD+offset] = new CQUAD4();
73
74 if (de_effects) {
75 element_container[DE_TERM][EL_TRIA+offset] = new CTRIA1();
76 element_container[DE_TERM][EL_QUAD+offset] = new CQUAD4();
77 }
78 }
79 else {
80 element_container[FEA_TERM][EL_TETRA+offset] = new CTETRA1();
81 element_container[FEA_TERM][EL_HEXA +offset] = new CHEXA8 ();
82 element_container[FEA_TERM][EL_PYRAM+offset] = new CPYRAM5();
83 element_container[FEA_TERM][EL_PRISM+offset] = new CPRISM6();
84
85 if (de_effects) {
86 element_container[DE_TERM][EL_TETRA+offset] = new CTETRA1();
87 element_container[DE_TERM][EL_HEXA +offset] = new CHEXA8 ();
88 element_container[DE_TERM][EL_PYRAM+offset] = new CPYRAM5();
89 element_container[DE_TERM][EL_PRISM+offset] = new CPRISM6();
90 }
91 }
92 }
93 END_SU2_OMP_PARALLEL
94
95 /*--- Set element properties ---*/
96 Set_ElementProperties(geometry, config);
97
98 Total_CFEA = 0.0;
99 WAitken_Dyn = 0.0;
100 WAitken_Dyn_tn1 = 0.0;
101
102 SetFSI_ConvValue(0,0.0);
103 SetFSI_ConvValue(1,0.0);
104
105 nVar = nDim;
106
107 /*--- Define some auxiliary vectors related to the residual ---*/
108
109 Residual_RMS.resize(nVar,0.0);
110 Residual_Max.resize(nVar,0.0);
111 Point_Max.resize(nVar,0);
112 Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
113
114 /*--- The length of the solution vector depends on whether the problem is static or dynamic ---*/
115
116 unsigned short nSolVar;
117 string text_line, filename;
118 ifstream restart_file;
119
120 if (dynamic) nSolVar = 3 * nVar;
121 else nSolVar = nVar;
122
123 auto* SolInit = new su2double[nSolVar]();
124
125 /*--- Initialize from zero everywhere ---*/
126
127 nodes = new CFEABoundVariable(SolInit, nPoint, nDim, nVar, config);
128 SetBaseClassPointerToNodes();
129
130 delete [] SolInit;
131
132 /*--- Set which points are vertices and allocate boundary data. ---*/
133
134 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++)
135 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
136 long iVertex = geometry->nodes->GetVertex(iPoint, iMarker);
137 if (iVertex >= 0) {
138 nodes->Set_isVertex(iPoint,true);
139 break;
140 }
141 }
142 static_cast<CFEABoundVariable*>(nodes)->AllocateBoundaryVariables(config);
143
144
145 if (config->GetRefGeom()) Set_ReferenceGeometry(geometry, config);
146 if (config->GetPrestretch()) Set_Prestretch(geometry, config);
147
148 /*--- Initialization of matrix structures ---*/
149 if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (Non-Linear Elasticity)." << endl;
150
151 Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, false, geometry, config);
152
153 if (dynamic) {
154 MassMatrix.Initialize(nPoint, nPointDomain, nVar, nVar, false, geometry, config);
155 TimeRes_Aux.Initialize(nPoint, nPointDomain, nVar, 0.0);
156 TimeRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
157 }
158
159 /*--- Initialization of linear solver structures ---*/
160 LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
161 LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
162 LinSysReact.Initialize(nPoint, nPointDomain, nVar, 0.0);
163
164 /*--- Initialize structures for hybrid-parallel mode. ---*/
165 HybridParallelInitialization(geometry);
166
167 /*--- Initialize the value of the total objective function ---*/
168 Total_OFRefGeom = 0.0;
169 Total_OFRefNode = 0.0;
170 Total_OFVolFrac = 1.0;
171 Total_OFDiscreteness = 0.0;
172 Total_OFCompliance = 0.0;
173
174 /*--- Initialize the value of the global objective function ---*/
175 Global_OFRefGeom = 0.0;
176 Global_OFRefNode = 0.0;
177
178 /*--- Initialize the value of the total gradient for the forward mode ---*/
179 Total_ForwardGradient = 0.0;
180
181 OutputForwardModeGradient(config, true, 0.0, 0.0, 0.0, 0.0);
182
183 /*--- Initialize the BGS residuals in FSI problems. ---*/
184 if (config->GetMultizone_Residual()){
185
186 FSI_Residual = 0.0;
187 RelaxCoeff = 1.0;
188 ForceCoeff = 1.0;
189
190 Residual_BGS.resize(nVar,1.0);
191 Residual_Max_BGS.resize(nVar,1.0);
192 Point_Max_BGS.resize(nVar,0);
193 Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
194 }
195 else {
196 ForceCoeff = 1.0;
197 }
198
199 /*--- Penalty value - to maintain constant the stiffness in optimization problems - TODO: this has to be improved ---*/
200 PenaltyValue = 0.0;
201
202 if (size != SINGLE_NODE) {
203 vector<unsigned short> essentialMarkers;
204 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
205 const auto kindBnd = config->GetMarker_All_KindBC(iMarker);
206 if ((kindBnd == CLAMPED_BOUNDARY) ||
207 (kindBnd == DISP_DIR_BOUNDARY) ||
208 (kindBnd == DISPLACEMENT_BOUNDARY)) {
209 essentialMarkers.push_back(iMarker);
210 }
211 }
212 Set_VertexEliminationSchedule(geometry, essentialMarkers);
213 }
214
215 /*--- Add the solver name. ---*/
216 SolverName = "FEA";
217}

237void CFEASolver::HybridParallelInitialization(CGeometry* geometry) {
238#ifdef HAVE_OMP
239 /*--- Get the element coloring. ---*/
240
241 su2double parallelEff = 1.0;
242 const auto& coloring = geometry->GetElementColoring(&parallelEff);
243
244 /*--- If the coloring is too bad use lock-guarded accesses
245 * to CSysMatrix/Vector in element loops instead. ---*/
246 LockStrategy = parallelEff < COLORING_EFF_THRESH;
247
248 /*--- When using locks force a single color to reduce the color loop overhead. ---*/
249 if (LockStrategy && (coloring.getOuterSize()>1))
250 geometry->SetNaturalElementColoring();
251
252 if (!coloring.empty()) {
253 /*--- We are not constrained by the color group size when using locks. ---*/
254 auto groupSize = LockStrategy? 1ul : geometry->GetElementColorGroupSize();
255 auto nColor = coloring.getOuterSize();
256 ElemColoring.reserve(nColor);
257
258 for(auto iColor = 0ul; iColor < nColor; ++iColor)
259 ElemColoring.emplace_back(coloring.innerIdx(iColor), coloring.getNumNonZeros(iColor), groupSize);
260 }
261
262 su2double minEff = 1.0;
263 SU2_MPI::Reduce(&parallelEff, &minEff, 1, MPI_DOUBLE, MPI_MIN, MASTER_NODE, SU2_MPI::GetComm());
264
265 if (minEff < COLORING_EFF_THRESH) {
266 cout << "WARNING: The element coloring efficiency was " << minEff << ", a fallback strategy is in use.\n"
267 << " Better performance may be possible by reducing the number of threads per rank." << endl;
268 }
269
270 if (LockStrategy) {
271 UpdateLocks.resize(nPoint);
272 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++)
273 omp_init_lock(&UpdateLocks[iPoint]);
274 }
275
276 omp_chunk_size = computeStaticChunkSize(nPointDomain, omp_get_max_threads(), OMP_MAX_SIZE);
277#else
278 ElemColoring[0] = DummyGridColor<>(nElement);
279#endif
280}

282void CFEASolver::Set_ElementProperties(CGeometry *geometry, CConfig *config) {
283
284 const auto iZone = config->GetiZone();
285 const auto nZone = geometry->GetnZone();
286
287 const bool topology_mode = config->GetTopology_Optimization();
288
289 element_properties = new CProperty*[nElement];
290
291 /*--- Restart the solution from file information ---*/
292
293 auto filename = config->GetFEA_FileName();
294
295 /*--- If multizone, append zone name ---*/
296 if (nZone > 1)
297 filename = config->GetMultizone_FileName(filename, iZone, ".dat");
298
299 if (rank == MASTER_NODE) cout << "Filename: " << filename << "." << endl;
300
301 ifstream properties_file;
302 properties_file.open(filename);
303
304 /*--- In case there is no file, all elements get the same property (0) ---*/
305
306 if (properties_file.fail()) {
307 if (rank == MASTER_NODE){
308 cout << "There is no element-based properties file." << endl;
309 cout << "The structural domain has uniform properties." << endl;
310
311 if (topology_mode)
312 SU2_MPI::Error("Topology mode requires an element-based properties file.",CURRENT_FUNCTION);
313 }
314
315 for (auto iElem = 0ul; iElem < nElement; iElem++){
316 element_properties[iElem] = new CElementProperty(FEA_TERM, 0, 0, 0);
317 }
318
319 element_based = false;
320
321 }
322 else{
323
324 element_based = true;
325
326 /*--- In case this is a parallel simulation, we need to perform the Global2Local index transformation first. ---*/
327
328 unordered_map<unsigned long, unsigned long> Global2Local;
329
330 for (auto iElem = 0ul; iElem < nElement; iElem++)
331 Global2Local[geometry->elem[iElem]->GetGlobalIndex()] = iElem;
332
333 /*--- Read all lines in the restart file ---*/
334
335 unsigned long iElem_Global_Local = 0, iElem_Global = 0; string text_line;
336
337 /*--- The first line is the header ---*/
338
339 getline (properties_file, text_line);
340
341 for (iElem_Global = 0; iElem_Global < geometry->GetGlobal_nElemDomain(); iElem_Global++ ) {
342
343 getline (properties_file, text_line);
344
345 istringstream point_line(text_line);
346
347 /*--- Retrieve local index. If this element from the restart file lives
348 only on a different processor, the value of iPoint_Local will be -1.
349 Otherwise, the local index for this node on the current processor
350 will be returned and used to instantiate the vars. ---*/
351
352 auto it = Global2Local.find(iElem_Global);
353
354 if (it != Global2Local.end()) {
355
356 auto iElem_Local = it->second;
357
358 unsigned long elProperties[4], index;
359
360 if (config->GetAdvanced_FEAElementBased() || topology_mode){
361 point_line >> index >> elProperties[0] >> elProperties[1] >> elProperties[2] >> elProperties[3];
362 element_properties[iElem_Local] = new CElementProperty(
363 elProperties[0], elProperties[1], elProperties[2], elProperties[3]);
364 }
365 else{
366 point_line >> index >> elProperties[0];
367 element_properties[iElem_Local] = new CElementProperty(0, elProperties[0], 0, 0);
368 }
369
370
371 /*--- For backwards compatibility we only read a fifth column in topology mode ---*/
372 if (topology_mode) {
373 su2double elDensity;
374 point_line >> elDensity;
375 element_properties[iElem_Local]->SetDesignDensity(elDensity);
376 }
377
378 iElem_Global_Local++;
379 }
380
381 }
382
383 /*--- Detect a wrong solution file ---*/
384
385 if (iElem_Global_Local != nElement) {
386 SU2_MPI::Error("The properties file " + filename + " doesn't match with the mesh file!\n"
387 "It could be empty lines at the end of the file.", CURRENT_FUNCTION);
388 }
389
390 }

394void CFEASolver::Set_Prestretch(CGeometry *geometry, CConfig *config) {
395
396 const auto iZone = config->GetiZone();
397 const auto nZone = geometry->GetnZone();
398
399 /*--- Restart the solution from file information ---*/
400
401 auto filename = config->GetPrestretch_FEMFileName();
402
403 /*--- If multizone, append zone name ---*/
404 if (nZone > 1)
405 filename = config->GetMultizone_FileName(filename, iZone, ".dat");
406
407 if (rank == MASTER_NODE) cout << "Filename: " << filename << "." << endl;
408
409 ifstream prestretch_file;
410 prestretch_file.open(filename);
411
412 /*--- In case there is no file ---*/
413
414 if (prestretch_file.fail()) {
415 SU2_MPI::Error(string("There is no FEM prestretch reference file ") + filename, CURRENT_FUNCTION);
416 }
417
418 /*--- Make a global to local map that also covers halo nodes (the one in geometry does not). ---*/
419
420 unordered_map<unsigned long, unsigned long> Global2Local;
421
422 for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint)
423 Global2Local[geometry->nodes->GetGlobalIndex(iPoint)] = iPoint;
424
425 /*--- Read all lines in the restart file ---*/
426
427 unsigned long iPoint_Global_Local = 0, iPoint_Global = 0; string text_line;
428
429 /*--- The first line is the header ---*/
430
431 getline (prestretch_file, text_line);
432
433 while (getline (prestretch_file, text_line)) {
434 istringstream point_line(text_line);
435
436 /*--- Retrieve local index. If this node from the restart file lives
437 on the current processor, we will load and instantiate the vars. ---*/
438
439 auto it = Global2Local.find(iPoint_Global);
440
441 if (it != Global2Local.end()) {
442
443 auto iPoint_Local = it->second;
444
445 su2double Sol[MAXNVAR] = {0.0};
446 unsigned long index;
447
448 if (nDim == 2) point_line >> Sol[0] >> Sol[1] >> index;
449 if (nDim == 3) point_line >> Sol[0] >> Sol[1] >> Sol[2] >> index;
450
451 for (unsigned short iVar = 0; iVar < nVar; iVar++)
452 nodes->SetPrestretch(iPoint_Local, iVar, Sol[iVar]);
453
454 iPoint_Global_Local++;
455 }
456 iPoint_Global++;
457 }
458
459 /*--- Detect a wrong solution file ---*/
460
461 if (iPoint_Global_Local != nPoint) {
462 SU2_MPI::Error(string("The solution file ") + filename + string(" doesn't match with the mesh file!\n") +
463 string("It could be empty lines at the end of the file."), CURRENT_FUNCTION);
464 }

557void CFEASolver::Preprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, CNumerics **numerics,
558 unsigned short iMesh, unsigned long Iteration, unsigned short RunTime_EqSystem, bool Output) {
559
560 const bool dynamic = config->GetTime_Domain();
561 const bool disc_adj_fem = (config->GetKind_Solver() == MAIN_SOLVER::DISC_ADJ_FEM);
562 const bool body_forces = config->GetDeadLoad();
563 const bool topology_mode = config->GetTopology_Optimization();
564
565 /*
566 * For topology optimization we apply a filter on the design density field to avoid
567 * numerical issues (checkerboards), ensure mesh independence, and impose a length scale.
568 * This has to be done before computing the mass matrix and the dead load terms.
569 * This filter, and the volume fraction objective function, require the element volumes,
570 * so we ask "geometry" to compute them.
571 * This only needs to be done for the undeformed (initial) shape.
572 */
573 if (topology_mode && !topol_filter_applied) {
574 geometry->SetElemVolume();
575 FilterElementDensities(geometry, config);
576 topol_filter_applied = true;
577 }
578
579 /*
580 * If the problem is dynamic we need a mass matrix, which will be constant along the calculation
581 * both for linear and nonlinear analysis, i.e. only computed once on the first time step.
582 *
583 * The same with the integration constants, as for now we consider the time step to be constant.
584 */
585 if (dynamic && (initial_calc || disc_adj_fem)) {
586 Compute_IntegrationConstants(config);
587 Compute_MassMatrix(geometry, numerics, config);
588 }
589
590 /*
591 * If body forces are taken into account, we need to compute the term that goes into the residual,
592 * which will be constant along the calculation both for linear and nonlinear analysis.
593 *
594 * Only initialized once, at the first iteration of the first time step.
595 */
596 if (body_forces && (initial_calc || disc_adj_fem))
597 Compute_DeadLoad(geometry, numerics, config);
598
599 /*--- Clear the linear system solution. ---*/
600 SU2_OMP_PARALLEL
601 {
602 LinSysSol.SetValZero();
603 }
604 END_SU2_OMP_PARALLEL
605
606 /*--- Clear external forces. ---*/
607 nodes->Clear_SurfaceLoad_Res();
608
609 /*--- Next call to Preprocessing will not be "initial_calc" and linear operations will not be repeated. ---*/
610 initial_calc = false;
611
612}

643void CFEASolver::Compute_StiffMatrix(CGeometry *geometry, CNumerics **numerics, const CConfig *config) {
644
645 const bool topology_mode = config->GetTopology_Optimization();
646 const su2double simp_exponent = config->GetSIMP_Exponent();
647 const su2double simp_minstiff = config->GetSIMP_MinStiffness();
648
649 /*--- Start OpenMP parallel region. ---*/
650
651 SU2_OMP_PARALLEL
652 {
653 /*--- Clear vector and matrix before calculation. ---*/
654 LinSysRes.SetValZero();
655 Jacobian.SetValZero();
656
657 for(auto color : ElemColoring) {
658
659 /*--- Chunk size is at least OMP_MIN_SIZE and a multiple of the color group size. ---*/
660 SU2_OMP_FOR_DYN(nextMultiple(OMP_MIN_SIZE, color.groupSize))
661 for(auto k = 0ul; k < color.size; ++k) {
662
663 auto iElem = color.indices[k];
664
665 unsigned short iNode, jNode, iDim, iVar;
666
667 int thread = omp_get_thread_num();
668
669 /*--- Convert VTK type to index in the element container. ---*/
670 int EL_KIND;
671 unsigned short nNodes;
672 GetElemKindAndNumNodes(geometry->elem[iElem]->GetVTK_Type(), EL_KIND, nNodes);
673
674 /*--- Each thread needs a dedicated element. ---*/
675 CElement* element = element_container[FEA_TERM][EL_KIND+thread*MAX_FE_KINDS];
676

1195

1714void CFEASolver::Postprocessing(CGeometry *geometry, CConfig *config, CNumerics **numerics, bool of_comp_mode) {
1715
1716 /*--- Compute the objective function. ---*/
1717
1718 const auto kindObjFunc = config->GetKind_ObjFunc();
1719 const bool penalty = ((kindObjFunc == REFERENCE_GEOMETRY) || (kindObjFunc == REFERENCE_NODE)) &&
1720 ((config->GetDV_FEA() == YOUNG_MODULUS) || (config->GetDV_FEA() == DENSITY_VAL));
1721
1722 auto computeAllFunctions = [&]() {
1723 /*--- Compute stresses for monitoring and output. ---*/
1724 Compute_NodalStress(geometry, numerics, config);
1725
1726 /*--- Compute functions for monitoring and output. ---*/
1727 Compute_OFRefNode(geometry, config);
1728 Compute_OFCompliance(geometry, config);
1729 if (config->GetRefGeom()) Compute_OFRefGeom(geometry, config);
1730 if (config->GetTopology_Optimization()) Compute_OFVolFrac(geometry, config);
1731 };
1732
1733 if (of_comp_mode) {
1734 if (penalty) Stiffness_Penalty(geometry, numerics, config);
1735
1736 switch (kindObjFunc) {
1737 case REFERENCE_GEOMETRY: Compute_OFRefGeom(geometry, config); break;
1738 case REFERENCE_NODE: Compute_OFRefNode(geometry, config); break;
1739 case VOLUME_FRACTION: Compute_OFVolFrac(geometry, config); break;
1740 case TOPOL_DISCRETENESS: Compute_OFVolFrac(geometry, config); break;
1741 case TOPOL_COMPLIANCE: Compute_OFCompliance(geometry, config); break;
1742 case STRESS_PENALTY:
1743 Compute_NodalStress(geometry, numerics, config);
1744 break;
1745 case CUSTOM_OBJFUNC:
1746 /*--- No easy way to know, so compute everything. ---*/
1747 computeAllFunctions();
1748 break;
1749 }
1750 return;
1751 }
1752
1753 if (!config->GetDiscrete_Adjoint()) {
1754 if (penalty) Stiffness_Penalty(geometry, numerics, config);
1755 computeAllFunctions();
1756 }
1757
1758 /*--- Residuals do not have to be computed while recording. ---*/
1759 if (config->GetDiscrete_Adjoint() && AD::TapeActive()) return;
1760
1761 if (config->GetGeometricConditions() == STRUCT_DEFORMATION::LARGE) {
1762
1763 /*--- For nonlinear analysis we have 3 convergence criteria: ---*/
1764 /*--- UTOL = norm(Delta_U(k)): ABSOLUTE, norm of the incremental displacements ---*/
1765 /*--- RTOL = norm(Residual(k): ABSOLUTE, norm of the residual (T-F) ---*/
1766 /*--- ETOL = Delta_U(k) * Residual(k): ABSOLUTE, energy norm ---*/
1767
1768 SU2_OMP_PARALLEL
1769 {
1770 su2double utol = LinSysSol.norm();
1771 su2double rtol = LinSysRes.norm();
1772 su2double etol = fabs(LinSysSol.dot(LinSysRes));
1773
1774 SU2_OMP_MASTER
1775 {
1776 Conv_Check[0] = utol;
1777 Conv_Check[1] = rtol;
1778 Conv_Check[2] = etol;
1779 }
1780 END_SU2_OMP_MASTER
1781 }
1782 END_SU2_OMP_PARALLEL
1783 }
1784 else {
1785
1786 /*--- If the problem is linear, the only check we do is the RMS of the residuals. ---*/
1787 /*--- Compute the residual Ax-f ---*/
1788
1789 const auto ResidualAux = computeLinearResidual(Jacobian, LinSysSol, LinSysRes);
1790
1791 SU2_OMP_PARALLEL {
1792
1793 /*--- Compute the residual. ---*/
1794 su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
1795 unsigned long idxMax[MAXNVAR] = {0};
1796
1797 SU2_OMP_FOR_STAT(omp_chunk_size)
1798 for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
1799 for (auto iVar = 0ul; iVar < nVar; iVar++) {
1800 /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
1801 ResidualReductions_PerThread(iPoint, iVar, ResidualAux(iPoint, iVar), resRMS, resMax, idxMax);
1802 }
1803 }
1804 END_SU2_OMP_FOR
1805
1806 /*--- "Add" residuals from all threads to global residual variables. ---*/
1807 ResidualReductions_FromAllThreads(geometry, config, resRMS,resMax,idxMax);
1808
1809 }
1810 END_SU2_OMP_PARALLEL
1811
1812 }

1819

1821
1822 /*--- Retrieve the normal pressure and the application conditions for the considered boundary. ---*/
1823

2028su2double CFEASolver::Compute_LoadCoefficient(su2double CurrentTime, su2double RampTime, const CConfig *config){
2029
2030 su2double LoadCoeff = 1.0;
2031
2032 bool Ramp_Load = config->GetRamp_Load();
2033 bool Sine_Load = config->GetSine_Load();
2034 bool Ramp_And_Release = config->GetRampAndRelease_Load();
2035
2036 bool restart = config->GetRestart(); // Restart analysis
2037 bool fsi = config->GetFSI_Simulation();
2038 bool stat_fsi = !config->GetTime_Domain();
2039
2040 /*--- This offset introduces the ramp load in dynamic cases starting from the restart point. ---*/
2041 bool offset = (restart && fsi && (!stat_fsi));
2042 su2double DeltaT = config->GetDelta_UnstTime();
2043 su2double OffsetTime = offset? DeltaT * (config->GetRestart_Iter()-1) : su2double(0.0);
2044
2045 /*--- Polynomial functions from https://en.wikipedia.org/wiki/Smoothstep ---*/
2046
2047 if ((Ramp_Load) && (RampTime > 0.0)) {
2048
2049 su2double TransferTime = (CurrentTime - OffsetTime) / RampTime;
2050
2051 switch (config->GetDynamic_LoadTransfer()) {
2052 case INSTANTANEOUS:
2053 LoadCoeff = 1.0;
2054 break;
2055 case POL_ORDER_1:
2056 LoadCoeff = TransferTime;
2057 break;
2058 case POL_ORDER_3:
2059 LoadCoeff = -2.0 * pow(TransferTime,3.0) + 3.0 * pow(TransferTime,2.0);
2060 break;
2061 case POL_ORDER_5:
2062 LoadCoeff = 6.0 * pow(TransferTime, 5.0) - 15.0 * pow(TransferTime, 4.0) + 10 * pow(TransferTime, 3.0);
2063 break;
2064 case SIGMOID_10:
2065 LoadCoeff = (1 / (1+exp(-1.0 * 10.0 * (TransferTime - 0.5)) ) );
2066 break;
2067 case SIGMOID_20:
2068 LoadCoeff = (1 / (1+exp(-1.0 * 20.0 * (TransferTime - 0.5)) ) );
2069 break;
2070 }
2071
2072 if (TransferTime > 1.0) LoadCoeff = 1.0;
2073
2074 LoadCoeff = max(LoadCoeff,0.0);
2075 LoadCoeff = min(LoadCoeff,1.0);
2076
2077 }
2078 else if (Sine_Load){
2079
2080 /*--- Retrieve amplitude, frequency (Hz) and phase (rad) ---*/
2081 su2double SineAmp = config->GetLoad_Sine()[0];
2082 su2double SineFreq = config->GetLoad_Sine()[1];
2083 su2double SinePhase = config->GetLoad_Sine()[2];
2084
2085 LoadCoeff = SineAmp * sin(2*PI_NUMBER*SineFreq*CurrentTime + SinePhase);
2086 }
2087
2088 /*--- Add possibility to release the load after the ramp---*/
2089 if ((Ramp_And_Release) && (CurrentTime > RampTime)){
2090 LoadCoeff = 0.0;
2091 }
2092
2093 /*--- Store the force coefficient ---*/
2094
2095 SetForceCoeff(LoadCoeff);
2096
2097 return LoadCoeff;

2418 /*--- Compute solution at t_n+1, and update velocities and accelerations ---*/
2419
2420 SU2_OMP_PARALLEL_(for schedule(static,omp_chunk_size))
2421 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
2422
2423 unsigned short iVar;
2424
2425 for (iVar = 0; iVar < nVar; iVar++) {
2426
2427 /*--- Compute the solution from the previous time step and the solution computed at t+1-alpha_f ---*/
2428 /*--- U(t+dt) = 1/alpha_f*(U(t+1-alpha_f)-alpha_f*U(t)) ---*/
2429
2430 su2double sol = (1/(1-alpha_f)) * (nodes->GetSolution(iPoint,iVar) -
2431 alpha_f * nodes->GetSolution_time_n(iPoint,iVar));
2432
2433 nodes->SetSolution(iPoint, iVar, sol);
2434 }
2435
2436 for (iVar = 0; iVar < nVar; iVar++) {
2437
2438 /*--- Acceleration component of the solution ---*/
2439 /*--- U''(t+dt-alpha_m) = a8*(U(t+dt)-U(t))+a2*(U'(t))+a3*(U''(t)) ---*/
2440
2441 su2double tmp = a_dt[8]*(nodes->GetSolution(iPoint,iVar) -
2442 nodes->GetSolution_time_n(iPoint,iVar)) -
2443 a_dt[2]* nodes->GetSolution_Vel_time_n(iPoint,iVar) -
2444 a_dt[3]* nodes->GetSolution_Accel_time_n(iPoint,iVar);
2445
2446 /*--- Compute the solution from the previous time step and the solution computed at t+1-alpha_f ---*/
2447 /*--- U''(t+dt) = 1/alpha_m*(U''(t+1-alpha_m)-alpha_m*U''(t)) ---*/
2448
2449 su2double sol = (1/(1-alpha_m)) * (tmp - alpha_m*nodes->GetSolution_Accel_time_n(iPoint,iVar));
2450
2451 nodes->SetSolution_Accel(iPoint, iVar, sol);
2452
2453 /*--- Velocity component of the solution ---*/
2454 /*--- U'(t+dt) = U'(t)+ a6*(U''(t)) + a7*(U''(t+dt)) ---*/
2455
2456 sol = nodes->GetSolution_Vel_time_n(iPoint,iVar)+
2457 a_dt[6]* nodes->GetSolution_Accel_time_n(iPoint,iVar) +
2458 a_dt[7]* nodes->GetSolution_Accel(iPoint,iVar);
2459
2460 nodes->SetSolution_Vel(iPoint, iVar, sol);
2461 }
2462
2463 }
2464 END_SU2_OMP_PARALLEL
2465
