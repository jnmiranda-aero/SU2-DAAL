31void CFEAIteration::Iterate(COutput* output, CIntegration**** integration, CGeometry**** geometry, CSolver***** solver,
32 CNumerics****** numerics, CConfig** config, CSurfaceMovement** surface_movement,
33 CVolumetricMovement*** grid_movement, CFreeFormDefBox*** FFDBox, unsigned short val_iZone,
34 unsigned short val_iInst) {
35 bool StopCalc = false;
36 unsigned long IntIter = 0;
37
38 const unsigned long TimeIter = config[val_iZone]->GetTimeIter();
39 const unsigned long nIncrements = config[val_iZone]->GetNumberIncrements();
40
41 const bool nonlinear = (config[val_iZone]->GetGeometricConditions() == STRUCT_DEFORMATION::LARGE);
42 const bool linear = (config[val_iZone]->GetGeometricConditions() == STRUCT_DEFORMATION::SMALL);
43 const bool disc_adj_fem = config[val_iZone]->GetDiscrete_Adjoint();
44
45 /*--- Loads applied in steps (not used for discrete adjoint). ---*/
46 const bool incremental_load = config[val_iZone]->GetIncrementalLoad() && !disc_adj_fem;
47
48 CIntegration* feaIntegration = integration[val_iZone][val_iInst][FEA_SOL];
49 CSolver* feaSolver = solver[val_iZone][val_iInst][MESH_0][FEA_SOL];
50
51 /*--- FEA equations ---*/
52 config[val_iZone]->SetGlobalParam(MAIN_SOLVER::FEM_ELASTICITY, RUNTIME_FEA_SYS);
53
54 if (linear) {
55 /*--- Run the (one) iteration ---*/
56
57 config[val_iZone]->SetInnerIter(0);
58
59 feaIntegration->Structural_Iteration(geometry, solver, numerics, config, RUNTIME_FEA_SYS, val_iZone, val_iInst);
60
61 if (!disc_adj_fem) {
62 Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement, FFDBox,
63 val_iZone, INST_0);
64
65 /*--- Set the convergence monitor to true, to prevent the solver to stop in intermediate FSI subiterations ---*/
66 output->SetConvergence(true);
67 }
68
69 } else if (nonlinear && !incremental_load) {
70 /*--- THIS IS THE DIRECT APPROACH (NO INCREMENTAL LOAD APPLIED) ---*/
71
72 /*--- Keep the current inner iter, we need to restore it in discrete adjoint cases
73 * because file output depends on it. ---*/
74 const auto CurIter = config[val_iZone]->GetInnerIter();
75
76 /*--- Newton-Raphson subiterations ---*/
77
78 for (IntIter = 0; IntIter < config[val_iZone]->GetnInner_Iter(); IntIter++) {
79 config[val_iZone]->SetInnerIter(IntIter);
80
81 feaIntegration->Structural_Iteration(geometry, solver, numerics, config, RUNTIME_FEA_SYS, val_iZone, val_iInst);
82
83 /*--- Limit to only one iteration for the discrete adjoint recording, restore inner iter (see above) ---*/
84 if (disc_adj_fem) {
85 config[val_iZone]->SetInnerIter(CurIter);
86 break;
87 }
88 StopCalc = Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement,
89 FFDBox, val_iZone, INST_0);
90
91 if (StopCalc && (IntIter > 0)) break;
92 }
93
94 } else {
95 /*--- THIS IS THE INCREMENTAL LOAD APPROACH (only makes sense for nonlinear) ---*/
96
97 /*--- Assume the initial load increment as 1.0 ---*/
98
99 feaSolver->SetLoad_Increment(0, 1.0);
100 feaSolver->SetForceCoeff(1.0);
101
102 /*--- Run two nonlinear iterations to check if incremental loading can be skipped ---*/
103
104 auto Iterate = [&](unsigned long IntIter) {
105 config[val_iZone]->SetInnerIter(IntIter);
106 feaIntegration->Structural_Iteration(geometry, solver, numerics, config, RUNTIME_FEA_SYS, val_iZone, val_iInst);
107
108 StopCalc = Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement,
109 FFDBox, val_iZone, INST_0);
110 };
111
112 for (IntIter = 0; IntIter < 2; ++IntIter) {
113 Iterate(IntIter);
114 }
115
116 /*--- Early return if we already meet the convergence criteria. ---*/
117 if (StopCalc) return;
118
119 /*--- Check user-defined criteria to either increment loads or continue with NR iterations. ---*/
120
121 bool meetCriteria = true;
122 for (int i = 0; i < 3; ++i)
123 meetCriteria &= (log10(feaSolver->GetRes_FEM(i)) < config[val_iZone]->GetIncLoad_Criteria(i));
124
125 /*--- If the criteria is met, i.e. the load is not too large, continue the regular calculation. ---*/
126
127 if (meetCriteria) {
128 /*--- Newton-Raphson subiterations ---*/
129
130 for (IntIter = 2; IntIter < config[val_iZone]->GetnInner_Iter(); IntIter++) {
131 Iterate(IntIter);
132 if (StopCalc) break;
133 }
134
135 }
136
137 /*--- If the criteria is not met, a whole set of subiterations for the different loads must be done. ---*/
138
139 else {
140 /*--- Restore solution to initial. Because we ramp the load from zero, in multizone cases it is not
141 * adequate to take "old values" as those will be for maximum loading on the previous outer iteration. ---*/
142
143 feaSolver->SetInitialCondition(geometry[val_iZone][val_iInst], solver[val_iZone][val_iInst], config[val_iZone],
144 TimeIter);
145
146 /*--- For the number of increments ---*/
147 for (auto iIncrement = 1ul; iIncrement <= nIncrements; iIncrement++) {
148 /*--- Set the load increment and the initial condition, and output the
149 * parameters of UTOL, RTOL, ETOL for the previous iteration ---*/
150
151 su2double loadIncrement = su2double(iIncrement) / nIncrements;
152 feaSolver->SetLoad_Increment(iIncrement, loadIncrement);
153
154 /*--- Set the convergence monitor to false, to force the solver to converge every subiteration ---*/
155 output->SetConvergence(false);
156
157 if (rank == MASTER_NODE) cout << "\nIncremental load: increment " << iIncrement << endl;
158
159 /*--- Newton-Raphson subiterations ---*/
160
161 for (IntIter = 0; IntIter < config[val_iZone]->GetnInner_Iter(); IntIter++) {
162 Iterate(IntIter);
163 if (StopCalc && (IntIter > 0)) break;
164 }
165 }
166 /*--- Just to be sure, set default increment settings. ---*/
167 feaSolver->SetLoad_Increment(0, 1.0);
168 }
169 }
170}

172void CFEAIteration::Update(COutput* output, CIntegration**** integration, CGeometry**** geometry, CSolver***** solver,
173 CNumerics****** numerics, CConfig** config, CSurfaceMovement** surface_movement,
174 CVolumetricMovement*** grid_movement, CFreeFormDefBox*** FFDBox, unsigned short val_iZone,
175 unsigned short val_iInst) {
176 CSolver* feaSolver = solver[val_iZone][val_iInst][MESH_0][FEA_SOL];
177
178 /*----------------- Update structural solver ----------------------*/
179
180 if (config[val_iZone]->GetTime_Domain()) {
181 integration[val_iZone][val_iInst][FEA_SOL]->SetDualTime_Solver(geometry[val_iZone][val_iInst][MESH_0], feaSolver,
182 config[val_iZone], MESH_0);
183 } else if (config[val_iZone]->GetFSI_Simulation() && config[val_iZone]->GetRelaxation()) {
184 /*--- For FSI problems with relaxation, output the relaxed result, which is the one transferred into the fluid
185 * domain (for consistent restart purposes). ---*/
186 if (config[val_iZone]->GetKind_TimeIntScheme_FEA() == STRUCT_TIME_INT::NEWMARK_IMPLICIT) {
187 feaSolver->ImplicitNewmark_Relaxation(geometry[val_iZone][val_iInst][MESH_0], config[val_iZone]);
188 }
189 }
190}

219bool CFEAIteration::Monitor(COutput* output, CIntegration**** integration, CGeometry**** geometry, CSolver***** solver,
220 CNumerics****** numerics, CConfig** config, CSurfaceMovement** surface_movement,
221 CVolumetricMovement*** grid_movement, CFreeFormDefBox*** FFDBox, unsigned short val_iZone,
222 unsigned short val_iInst) {

224

226

228 config[val_iZone], config[val_iZone]->GetTimeIter(), config[val_iZone]->GetOuterIter(),
229 config[val_iZone]->GetInnerIter());
230
