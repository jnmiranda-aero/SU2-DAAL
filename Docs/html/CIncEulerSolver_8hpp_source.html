239 /*!
240 * \author H. Kline
241 * \brief Compute weighted-sum "combo" objective output
242 * \param[in] config - Definition of the particular problem.
243 * \param[in] solver - Container vector with all the solutions.
244 */
245 void Evaluate_ObjFunc(const CConfig *config, CSolver **solver) final {
246 Total_ComboObj = EvaluateCommonObjFunc(*config);
247
248 if (config->GetWeakly_Coupled_Heat()) {
249 solver[HEAT_SOL]->Evaluate_ObjFunc(config, solver);
250 Total_ComboObj += solver[HEAT_SOL]->GetTotal_ComboObj();
251 }
252 }
