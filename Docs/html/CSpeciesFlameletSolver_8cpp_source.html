66 void CSpeciesFlameletSolver::Preprocessing(CGeometry* geometry, CSolver** solver_container, CConfig* config,
67  unsigned short iMesh, unsigned short iRKStep,
68  unsigned short RunTime_EqSystem, bool Output) {
69  unsigned long n_not_in_domain_local = 0, n_not_in_domain_global = 0;
70  vector<su2double> scalars_vector(nVar);
71  unsigned long spark_iter_start, spark_duration;
72  bool ignition = false;
73  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
74 
75  /*--- Retrieve spark ignition parameters for spark-type ignition. ---*/
76  if ((config->GetFlameletInitType() == FLAMELET_INIT_TYPE::SPARK) && !config->GetRestart()) {
77  auto spark_init = config->GetFlameInit();
78  spark_iter_start = ceil(spark_init[4]);
79  spark_duration = ceil(spark_init[5]);
80  unsigned long iter = config->GetMultizone_Problem() ? config->GetOuterIter() : config->GetInnerIter();
81  ignition = ((iter >= spark_iter_start) && (iter <= (spark_iter_start + spark_duration)) && !config->GetRestart());
82  }
83 
84  SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetGlobalParam(config->GetKind_Solver(), RunTime_EqSystem);)
85 
86  SU2_OMP_FOR_STAT(omp_chunk_size)
87  for (auto i_point = 0u; i_point < nPoint; i_point++) {
88  CFluidModel* fluid_model_local = solver_container[FLOW_SOL]->GetFluidModel();
89  su2double* scalars = nodes->GetSolution(i_point);
90  for (auto iVar = 0u; iVar < nVar; iVar++) scalars_vector[iVar] = scalars[iVar];
91 
92  /*--- Compute total source terms from the production and consumption. ---*/
93  unsigned long misses = SetScalarSources(config, fluid_model_local, i_point, scalars_vector);
94 
95  if (ignition) {
96  /*--- Apply source terms within spark radius. ---*/
97  su2double dist_from_center = 0,
98  spark_radius = config->GetFlameInit()[3];
99  dist_from_center = GeometryToolbox::SquaredDistance(nDim, geometry->nodes->GetCoord(i_point), config->GetFlameInit());
100  if (dist_from_center < pow(spark_radius,2)) {
101  for (auto iVar = 0u; iVar < nVar; iVar++)
102  nodes->SetScalarSource(i_point, iVar, nodes->GetScalarSources(i_point)[iVar] + config->GetSpark()[iVar]);
103  }
104  }
105 
106  nodes->SetTableMisses(i_point, misses);
107  n_not_in_domain_local += misses;
108  /*--- Obtain passive look-up scalars. ---*/
109  SetScalarLookUps(config, fluid_model_local, i_point, scalars_vector);
110 
111  /*--- Set mass diffusivity based on thermodynamic state. ---*/
112  auto T = flowNodes->GetTemperature(i_point);
113  fluid_model_local->SetTDState_T(T, scalars);
114  /*--- set the diffusivity in the fluid model to the diffusivity obtained from the lookup table ---*/
115  for (auto i_scalar = 0u; i_scalar < nVar; ++i_scalar) {
116  nodes->SetDiffusivity(i_point, fluid_model_local->GetMassDiffusivity(i_scalar), i_scalar);
117  }
118 
119  /*--- Obtain preferential diffusion scalar values. ---*/
120  if (config->GetPreferentialDiffusion())
121  SetPreferentialDiffusionScalars(config, fluid_model_local, i_point, scalars_vector);
122 
123  if (!Output) LinSysRes.SetBlock_Zero(i_point);
124  }
125  END_SU2_OMP_FOR
126  /* --- Sum up some global counters over processes. --- */
127  SU2_MPI::Reduce(&n_not_in_domain_local, &n_not_in_domain_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
128  SU2_MPI::GetComm());
129  if ((rank == MASTER_NODE) && (n_not_in_domain_global > 0))
130  cout << "Number of points outside manifold domain: " << n_not_in_domain_global << endl;
131 
132  /*--- Compute preferential diffusion scalar gradients. ---*/
133  if (config->GetPreferentialDiffusion()) {
134  switch (config->GetKind_Gradient_Method()) {
135  case GREEN_GAUSS:
136  SetAuxVar_Gradient_GG(geometry, config);
137  break;
138  case WEIGHTED_LEAST_SQUARES:
139  SetAuxVar_Gradient_LS(geometry, config);
140  break;
141  default:
142  break;
143  }
144  }
145  /*--- Clear Residual and Jacobian. Upwind second order reconstruction and gradients ---*/
146  CommonPreprocessing(geometry, config, Output);
147 }

149 void CSpeciesFlameletSolver::SetInitialCondition(CGeometry** geometry, CSolver*** solver_container, CConfig* config,
150  unsigned long ExtIter) {
151  const bool Restart = (config->GetRestart() || config->GetRestart_Flow());
152 
153  if ((!Restart) && ExtIter == 0) {
154  if (rank == MASTER_NODE) {
155  cout << "Initializing progress variable and total enthalpy (using temperature)" << endl;
156  }
157 
158  su2double flame_offset[3] = {0, 0, 0}, flame_normal[3] = {0, 0, 0}, flame_thickness = 0, flame_burnt_thickness = 0,
159  flamenorm = 0;
160  bool flame_front_ignition = (config->GetFlameletInitType() == FLAMELET_INIT_TYPE::FLAME_FRONT);
161 
162  if (flame_front_ignition) {
163  /*--- Collect flame front ignition parameters. ---*/
164  auto flame_init = config->GetFlameInit();
165  for (auto iDim = 0u; iDim < 3; ++iDim) {
166  flame_offset[iDim] = flame_init[iDim];
167  flame_normal[iDim] = flame_init[3 + iDim];
168  }
169  flame_thickness = flame_init[6];
170  flame_burnt_thickness = flame_init[7];
171  flamenorm = GeometryToolbox::Norm(nDim, flame_normal);
172  }
173 
174  const su2double temp_inlet = config->GetInc_Temperature_Init();
175  su2double enth_inlet = config->GetSpecies_Init()[I_ENTH];
176 
177  su2double prog_burnt = 0, prog_unburnt, point_loc;
178  su2double scalar_init[MAXNVAR];
179 
180  if (rank == MASTER_NODE) {
181  cout << "initial condition: T = " << temp_inlet << endl;
182  for (auto iCV = 0u; iCV < config->GetNControlVars(); iCV++) {
183  const auto& cv_name = config->GetControllingVariableName(iCV);
184  cout << "initial condition: " << cv_name << " = " << config->GetSpecies_Init()[iCV] << endl;
185  }
186  switch (config->GetFlameletInitType()) {
187  case FLAMELET_INIT_TYPE::FLAME_FRONT:
188  cout << "Ignition with a straight flame front" << endl;
189  break;
190  case FLAMELET_INIT_TYPE::SPARK:
191  cout << "Ignition with an artificial spark" << endl;
192  break;
193  case FLAMELET_INIT_TYPE::NONE:
194  cout << "No solution ignition (cold flow)" << endl;
195  break;
196  default:
197  break;
198  }
199  }
200 
201  CFluidModel* fluid_model_local;
202 
203  unsigned long n_not_iterated_local = 0, n_not_in_domain_local = 0, n_points_unburnt_local = 0,
204  n_points_burnt_local = 0, n_points_flame_local = 0, n_not_iterated_global, n_not_in_domain_global,
205  n_points_burnt_global, n_points_flame_global, n_points_unburnt_global;
206 
207  for (unsigned long i_mesh = 0; i_mesh <= config->GetnMGLevels(); i_mesh++) {
208  fluid_model_local = solver_container[i_mesh][FLOW_SOL]->GetFluidModel();
209 
210  for (auto iVar = 0u; iVar < nVar; iVar++) scalar_init[iVar] = config->GetSpecies_Init()[iVar];
211 
212  /*--- Set enthalpy based on initial temperature and scalars. ---*/
213  n_not_iterated_local += GetEnthFromTemp(fluid_model_local, temp_inlet, config->GetSpecies_Init(), &enth_inlet);
214  scalar_init[I_ENTH] = enth_inlet;
215 
216  prog_unburnt = config->GetSpecies_Init()[I_PROGVAR];
217  prog_burnt = GetBurntProgressVariable(fluid_model_local, scalar_init);
218  SU2_OMP_FOR_STAT(omp_chunk_size)
219  for (unsigned long i_point = 0; i_point < nPoint; i_point++) {
220  auto coords = geometry[i_mesh]->nodes->GetCoord(i_point);
221 
222  if (flame_front_ignition) {
223  /*--- Determine if point is above or below the plane, assuming the normal
224  is pointing towards the burned region. ---*/
225  point_loc = 0.0;
226  for (unsigned short i_dim = 0; i_dim < geometry[i_mesh]->GetnDim(); i_dim++) {
227  point_loc += flame_normal[i_dim] * (coords[i_dim] - flame_offset[i_dim]);
228  }
229 
230  /*--- Compute the exact distance from point to plane. ---*/
231  point_loc = point_loc / flamenorm;
232 
233  /* --- Unburnt region upstream of the flame. --- */
234  if (point_loc <= 0) {
235  scalar_init[I_PROGVAR] = prog_unburnt;
236  n_points_unburnt_local++;
237 
238  /* --- Flame zone where we lineary increase from unburnt to burnt conditions. --- */
239  } else if ((point_loc > 0) && (point_loc <= flame_thickness)) {
240  scalar_init[I_PROGVAR] = prog_unburnt + point_loc * (prog_burnt - prog_unburnt) / flame_thickness;
241  n_points_flame_local++;
242 
243  /* --- Burnt region behind the flame zone. --- */
244  } else if ((point_loc > flame_thickness) && (point_loc <= flame_thickness + flame_burnt_thickness)) {
245  scalar_init[I_PROGVAR] = prog_burnt;
246  n_points_burnt_local++;
247 
248  /* --- Unburnt region downstream of the flame and burnt region. --- */
249  } else {
250  scalar_init[I_PROGVAR] = prog_unburnt;
251  n_points_unburnt_local++;
252  }
253  } else {
254  scalar_init[I_PROGVAR] = prog_unburnt;
255  }
256  /* --- Perform manifold evaluation to check whether initial scalar is out-of-bounds. --- */
257  fluid_model_local->SetTDState_T(temp_inlet, scalar_init);
258  n_not_in_domain_local += fluid_model_local->GetExtrapolation();
259 
260  /* --- Initialize the auxiliary transported scalars (not controlling variables). --- */
261  for (int i_scalar = config->GetNControlVars(); i_scalar < config->GetNScalars(); ++i_scalar) {
262  scalar_init[i_scalar] = config->GetSpecies_Init()[i_scalar];
263  }
264 
265  solver_container[i_mesh][SPECIES_SOL]->GetNodes()->SetSolution(i_point, scalar_init);
266  }
267 
268  solver_container[i_mesh][SPECIES_SOL]->InitiateComms(geometry[i_mesh], config, MPI_QUANTITIES::SOLUTION);
269  solver_container[i_mesh][SPECIES_SOL]->CompleteComms(geometry[i_mesh], config, MPI_QUANTITIES::SOLUTION);
270 
271  solver_container[i_mesh][FLOW_SOL]->InitiateComms(geometry[i_mesh], config, MPI_QUANTITIES::SOLUTION);
272  solver_container[i_mesh][FLOW_SOL]->CompleteComms(geometry[i_mesh], config, MPI_QUANTITIES::SOLUTION);
273 
274  solver_container[i_mesh][FLOW_SOL]->Preprocessing(geometry[i_mesh], solver_container[i_mesh], config, i_mesh,
275  NO_RK_ITER, RUNTIME_FLOW_SYS, false);
276  END_SU2_OMP_FOR
277  }
278 
279  /* --- Sum up some global counters over processes. --- */
280  SU2_MPI::Reduce(&n_not_in_domain_local, &n_not_in_domain_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
281  SU2_MPI::GetComm());
282  SU2_MPI::Reduce(&n_not_iterated_local, &n_not_iterated_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
283  SU2_MPI::GetComm());
284  SU2_MPI::Reduce(&n_points_unburnt_local, &n_points_unburnt_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
285  SU2_MPI::GetComm());
286  SU2_MPI::Reduce(&n_points_burnt_local, &n_points_burnt_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
287  SU2_MPI::GetComm());
288  SU2_MPI::Reduce(&n_points_flame_local, &n_points_flame_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
289  SU2_MPI::GetComm());
290 
291  if (rank == MASTER_NODE) {
292  cout << endl;
293  if (flame_front_ignition) {
294  cout << " Number of points in unburnt region: " << n_points_unburnt_global << "." << endl;
295  cout << " Number of points in burnt region : " << n_points_burnt_global << "." << endl;
296  cout << " Number of points in flame zone : " << n_points_flame_global << "." << endl;
297  }
298 
299  if (n_not_in_domain_global > 0)
300  cout << " Initial condition: Number of points outside of table domain: " << n_not_in_domain_global << " !!!"
301  << endl;
302 
303  if (n_not_iterated_global > 0)
304  cout << " Initial condition: Number of points in which enthalpy could not be iterated: "
305  << n_not_iterated_global << " !!!" << endl;
306  }
307  }

310 void CSpeciesFlameletSolver::SetPreconditioner(CGeometry* geometry, CSolver** solver_container, CConfig* config) {
311  const bool variable_density = (config->GetVariable_Density_Model());
312  const bool implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
313 
314  SU2_OMP_FOR_STAT(omp_chunk_size)
315  for (auto iPoint = 0u; iPoint < nPointDomain; iPoint++) {
316  /*--- Access the primitive variables at this node. ---*/
317 
318  su2double Density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
319  su2double BetaInc2 = solver_container[FLOW_SOL]->GetNodes()->GetBetaInc2(iPoint);
320  su2double Temperature = solver_container[FLOW_SOL]->GetNodes()->GetTemperature(iPoint);
321 
322  su2double SolP = solver_container[FLOW_SOL]->LinSysSol(iPoint, prim_idx.Pressure());
323  su2double SolT = solver_container[FLOW_SOL]->LinSysSol(iPoint, prim_idx.Temperature());
324 
325  /*--- We need the derivative of the equation of state to build the
326  preconditioning matrix. For now, the only option is the ideal gas
327  law, but in the future, dRhodT should be in the fluid model. ---*/
328 
329  su2double dRhodT = 0.0;
330  if (variable_density) {
331  dRhodT = -Density / Temperature;
332  }
333 
334  /*--- Passive scalars have no impact on the density. ---*/
335 
336  su2double dRhodC = 0.0;
337 
338  /*--- Modify matrix diagonal with term including volume and time step. ---*/
339 
340  su2double Vol = geometry->nodes->GetVolume(iPoint);
341  su2double Delta =
342  Vol / (config->GetCFLRedCoeff_Species() * solver_container[FLOW_SOL]->GetNodes()->GetDelta_Time(iPoint));
343 
344  /*--- Calculating the inverse of the preconditioning matrix
345  that multiplies the time derivative during time integration. ---*/
346 
347  if (implicit) {
348  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
349  su2double scalar = nodes->GetSolution(iPoint, iVar);
350 
351  /*--- Compute the lag terms for the decoupled linear system from
352  the mean flow equations and add to the residual for the scalar.
353  In short, we are effectively making these terms explicit. ---*/
354 
355  su2double artcompc1 = SolP * scalar / (Density * BetaInc2);
356  su2double artcompc2 = SolT * dRhodT * scalar / (Density);
357 
358  LinSysRes(iPoint, iVar) += artcompc1 + artcompc2;
359 
360  /*--- Add the extra Jacobian term to the scalar system. ---*/
361 
362  su2double Jaccomp = scalar * dRhodC + Density;
363  su2double JacTerm = Jaccomp * Delta;
364 
365  Jacobian.AddVal2Diag(iPoint, iVar, JacTerm);
366  }
367  }
368  }
369  END_SU2_OMP_FOR
370 }

408 void CSpeciesFlameletSolver::BC_Isothermal_Wall_Generic(CGeometry* geometry, CSolver** solver_container,
409  CNumerics* conv_numerics, CNumerics* visc_numerics,
410  CConfig* config, unsigned short val_marker, bool cht_mode) {
411  const bool implicit = config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT;
412  const string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
413  CFluidModel* fluid_model_local = solver_container[FLOW_SOL]->GetFluidModel();
414  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
415  unsigned long n_not_iterated = 0;
416 
417  /*--- Loop over all the vertices on this boundary marker. ---*/
418  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
419  for (unsigned long iVertex = 0; iVertex < geometry->nVertex[val_marker]; iVertex++) {
420  unsigned long iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
421 
422  su2double temp_wall, enth_wall;
423  if (cht_mode)
424  temp_wall = solver_container[FLOW_SOL]->GetConjugateHeatVariable(val_marker, iVertex, 0);
425  else
426  temp_wall = config->GetIsothermal_Temperature(Marker_Tag);
427 
428  /*--- Check if the node belongs to the domain (i.e., not a halo node). ---*/
429 
430  if (geometry->nodes->GetDomain(iPoint)) {
431  if (config->GetMarker_StrongBC(Marker_Tag) == true) {
432  /*--- Initial guess for enthalpy value. ---*/
433  enth_wall = nodes->GetSolution(iPoint, I_ENTH);
434 
435  /*--- Set enthalpy on the wall. ---*/
436  n_not_iterated += GetEnthFromTemp(fluid_model_local, temp_wall, nodes->GetSolution(iPoint), &enth_wall);
437 
438  /*--- Impose the value of the enthalpy as a strong boundary
439  condition (Dirichlet) and remove any
440  contribution to the residual at this node. ---*/
441 
442  nodes->SetSolution(iPoint, I_ENTH, enth_wall);
443  nodes->SetSolution_Old(iPoint, I_ENTH, enth_wall);
444 
445  LinSysRes(iPoint, I_ENTH) = 0.0;
446 
447  nodes->SetVal_ResTruncError_Zero(iPoint, I_ENTH);
448 
449  if (implicit) {
450  unsigned long total_index = iPoint * nVar + I_ENTH;
451  Jacobian.DeleteValsRowi(total_index);
452  }
453  } else {
454  /*--- Weak BC formulation. ---*/
455  const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
456 
457  const su2double Area = GeometryToolbox::Norm(nDim, Normal);
458 
459  const auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
460 
461  /*--- Get coordinates of i & nearest normal and compute distance. ---*/
462 
463  const auto Coord_i = geometry->nodes->GetCoord(iPoint);
464  const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);
465  su2double Edge_Vector[MAXNDIM];
466  GeometryToolbox::Distance(nDim, Coord_j, Coord_i, Edge_Vector);
467  su2double dist_ij_2 = GeometryToolbox::SquaredNorm(nDim, Edge_Vector);
468  su2double dist_ij = sqrt(dist_ij_2);
469 
470  /*--- Compute the normal gradient in temperature using Twall. ---*/
471  ///TODO: Account for preferential diffusion in computation of the heat flux
472  su2double dTdn = -(flowNodes->GetTemperature(Point_Normal) - temp_wall) / dist_ij;
473 
474  /*--- Get thermal conductivity. ---*/
475 
476  su2double thermal_conductivity = flowNodes->GetThermalConductivity(iPoint);
477 
478  /*--- Apply a weak boundary condition for the energy equation.
479  Compute the residual due to the prescribed heat flux. ---*/
480 
481  LinSysRes(iPoint, I_ENTH) -= thermal_conductivity * dTdn * Area;
482  }
483  }
484  }
485  END_SU2_OMP_FOR
486 
487  if (rank == MASTER_NODE && n_not_iterated > 0) {
488  cout << " !!! Wall bc (" << Marker_Tag
489  << "): Number of points in which enthalpy could not be iterated: " << n_not_iterated << " !!!" << endl;
490  }
491 }

505 unsigned long CSpeciesFlameletSolver::SetScalarSources(const CConfig* config, CFluidModel* fluid_model_local,
506  unsigned long iPoint, const vector<su2double>& scalars) {
507  /*--- Compute total source terms from the production and consumption. ---*/
508 
509  vector<su2double> table_sources(config->GetNControlVars() + 2 * config->GetNUserScalars());
510  unsigned long misses = fluid_model_local->EvaluateDataSet(scalars, FLAMELET_LOOKUP_OPS::SOURCES, table_sources);
511  table_sources[I_PROGVAR] = fmax(0, table_sources[I_PROGVAR]);
512  nodes->SetTableMisses(iPoint, misses);
513 
514  /*--- The source term for progress variable is always positive, we clip from below to makes sure. --- */
515 
516  vector<su2double> source_scalar(config->GetNScalars());
517  for (auto iCV = 0u; iCV < config->GetNControlVars(); iCV++) source_scalar[iCV] = table_sources[iCV];
518 
519  /*--- Source term for the auxiliary species transport equations. ---*/
520  for (size_t i_aux = 0; i_aux < config->GetNUserScalars(); i_aux++) {
521  /*--- The source term for the auxiliary equations consists of a production term and a consumption term:
522  S_TOT = S_PROD + S_CONS * Y ---*/
523  su2double y_aux = scalars[config->GetNControlVars() + i_aux];
524  su2double source_prod = table_sources[config->GetNControlVars() + 2 * i_aux];
525  su2double source_cons = table_sources[config->GetNControlVars() + 2 * i_aux + 1];
526  source_scalar[config->GetNControlVars() + i_aux] = source_prod + source_cons * y_aux;
527  }
528  for (auto i_scalar = 0u; i_scalar < nVar; i_scalar++)
529  nodes->SetScalarSource(iPoint, i_scalar, source_scalar[i_scalar]);
530  return misses;
531 }

565 void CSpeciesFlameletSolver::Viscous_Residual(const unsigned long iEdge, const CGeometry* geometry, CSolver** solver_container,
566  CNumerics* numerics, const CConfig* config) {
567  /*--- Overloaded viscous residual method which accounts for preferential diffusion. ---*/
568  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT),
569  PreferentialDiffusion = config->GetPreferentialDiffusion();
570 
571  /*--- Points in edge ---*/
572  auto iPoint = geometry->edges->GetNode(iEdge, 0);
573  auto jPoint = geometry->edges->GetNode(iEdge, 1);
574 
575  auto SolverSpecificNumerics = [&](unsigned long iPoint, unsigned long jPoint) {
576  /*--- Mass diffusivity coefficients. ---*/
577 
578  numerics->SetDiffusionCoeff(nodes->GetDiffusivity(iPoint), nodes->GetDiffusivity(jPoint));
579  };
580 
581  /*--- Regular viscous scalar residual computation. ---*/
582 
583  Viscous_Residual_impl(SolverSpecificNumerics, iEdge, geometry, solver_container, numerics, config);
584 
585  /*--- Viscous residual due to preferential diffusion ---*/
586  if (PreferentialDiffusion) {
587  CFlowVariable* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
588 
589  su2double scalar_i[MAXNVAR] = {0},
590  scalar_j[MAXNVAR] = {0},
591  diff_coeff_beta_i[MAXNVAR] = {0},
592  diff_coeff_beta_j[MAXNVAR] = {0};
593 
594  // Number of active transport scalars
595  const auto n_CV = config->GetNControlVars();
596 
597  su2activematrix scalar_grad_i(MAXNVAR, MAXNDIM), scalar_grad_j(MAXNVAR, MAXNDIM);
598  /*--- Looping over spatial dimensions to fill in the diffusion scalar gradients. ---*/
599  /*--- The scalar gradient is subtracted to account for regular viscous diffusion. ---*/
600  for (auto iScalar = 0u; iScalar < n_CV; ++iScalar) {
601  for (auto iDim = 0u; iDim < nDim; ++iDim) {
602  switch (iScalar) {
603  case I_PROGVAR:
604  scalar_grad_i[iScalar][iDim] =
605  nodes->GetAuxVarGradient(iPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_PROGVAR, iDim) -
606  nodes->GetGradient(iPoint, I_PROGVAR, iDim);
607  scalar_grad_j[iScalar][iDim] =
608  nodes->GetAuxVarGradient(jPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_PROGVAR, iDim) -
609  nodes->GetGradient(jPoint, I_PROGVAR, iDim);
610  break;
611  case I_ENTH:
612  scalar_grad_i[iScalar][iDim] =
613  nodes->GetAuxVarGradient(iPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH, iDim) -
614  nodes->GetGradient(iPoint, I_ENTH, iDim);
615  scalar_grad_j[iScalar][iDim] =
616  nodes->GetAuxVarGradient(jPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH, iDim) -
617  nodes->GetGradient(jPoint, I_ENTH, iDim);
618  break;
619  case I_MIXFRAC:
620  scalar_grad_i[iScalar][iDim] =
621  nodes->GetAuxVarGradient(iPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_MIXFRAC, iDim) -
622  nodes->GetGradient(iPoint, I_MIXFRAC, iDim);
623  scalar_grad_j[iScalar][iDim] =
624  nodes->GetAuxVarGradient(jPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_MIXFRAC, iDim) -
625  nodes->GetGradient(jPoint, I_MIXFRAC, iDim);
626  break;
627  default:
628  break;
629  }
630  }
631  }
632  /*--- No preferential diffusion modification for passive species. ---*/
633  for (auto iScalar = n_CV; iScalar < nVar; ++iScalar) {
634  for (auto iDim = 0u; iDim < nDim; ++iDim) {
635  scalar_grad_i[iScalar][iDim] = 0;
636  scalar_grad_j[iScalar][iDim] = 0;
637  }
638  }
639 
640  for (auto iScalar = 0u; iScalar < n_CV; ++iScalar) {
641  /*--- Filling in the preferential diffusion scalars (beta_pv, beta_h2, beta_Z). ---*/
642  switch (iScalar) {
643  case I_PROGVAR:
644  scalar_i[iScalar] = nodes->GetAuxVar(iPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_PROGVAR) -
645  nodes->GetSolution(iPoint, iScalar);
646  scalar_j[iScalar] = nodes->GetAuxVar(jPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_PROGVAR) -
647  nodes->GetSolution(jPoint, iScalar);
648  break;
649  case I_ENTH:
650  scalar_i[iScalar] =
651  nodes->GetAuxVar(iPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH) - nodes->GetSolution(iPoint, iScalar);
652  scalar_j[iScalar] =
653  nodes->GetAuxVar(jPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH) - nodes->GetSolution(jPoint, iScalar);
654  break;
655  case I_MIXFRAC:
656  scalar_i[iScalar] = nodes->GetAuxVar(iPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_MIXFRAC) -
657  nodes->GetSolution(iPoint, iScalar);
658  scalar_j[iScalar] = nodes->GetAuxVar(jPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_MIXFRAC) -
659  nodes->GetSolution(jPoint, iScalar);
660  break;
661  default:
662  break;
663  }
664  diff_coeff_beta_i[iScalar] = nodes->GetDiffusivity(iPoint, iScalar);
665  diff_coeff_beta_j[iScalar] = nodes->GetDiffusivity(jPoint, iScalar);
666  }
667 
668  for (auto iScalar = n_CV; iScalar < nVar; ++iScalar) {
669  scalar_i[iScalar] = 0;
670  scalar_j[iScalar] = 0;
671  diff_coeff_beta_i[iScalar] = 0;
672  diff_coeff_beta_j[iScalar] = 0;
673  }
674 
675  numerics->SetScalarVar(scalar_i, scalar_j);
676 
677  numerics->SetScalarVarGradient(CMatrixView<su2double>(scalar_grad_i), CMatrixView<su2double>(scalar_grad_j));
678 
679  numerics->SetDiffusionCoeff(diff_coeff_beta_i, diff_coeff_beta_j);
680 
681  /*--- Computing first preferential residual component. ---*/
682  auto residual_PD = numerics->ComputeResidual(config);
683 
684  if (ReducerStrategy) {
685  EdgeFluxes.SubtractBlock(iEdge, residual_PD);
686 
687  if (implicit) Jacobian.UpdateBlocksSub(iEdge, residual_PD.jacobian_i, residual_PD.jacobian_j);
688  } else {
689  LinSysRes.SubtractBlock(iPoint, residual_PD);
690  LinSysRes.AddBlock(jPoint, residual_PD);
691  /*--- Set implicit computation ---*/
692  if (implicit) Jacobian.UpdateBlocksSub(iEdge, iPoint, jPoint, residual_PD.jacobian_i, residual_PD.jacobian_j);
693  }
694 
695  /* Computing the second preferential diffusion terms due to heat flux */
696  for (auto iScalar = 0u; iScalar < nVar; ++iScalar) {
697  for (auto iDim = 0u; iDim < nDim; ++iDim) {
698  if (iScalar == I_ENTH) {
699  /* Setting the temperature gradient */
700  scalar_grad_i[iScalar][iDim] = flowNodes->GetGradient_Primitive(iPoint, prim_idx.Temperature(), iDim);
701  scalar_grad_j[iScalar][iDim] = flowNodes->GetGradient_Primitive(jPoint, prim_idx.Temperature(), iDim);
702  } else {
703  scalar_grad_i[iScalar][iDim] = 0;
704  scalar_grad_j[iScalar][iDim] = 0;
705  }
706  }
707 
708  if (iScalar == I_ENTH) {
709  scalar_i[iScalar] = flowNodes->GetTemperature(iPoint);
710  scalar_j[iScalar] = flowNodes->GetTemperature(jPoint);
711  diff_coeff_beta_i[iScalar] = nodes->GetAuxVar(iPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH_THERMAL) *
712  nodes->GetDiffusivity(iPoint, iScalar);
713  diff_coeff_beta_j[iScalar] = nodes->GetAuxVar(jPoint, FLAMELET_PREF_DIFF_SCALARS::I_BETA_ENTH_THERMAL) *
714  nodes->GetDiffusivity(jPoint, iScalar);
715  } else {
716  scalar_i[iScalar] = 0;
717  scalar_j[iScalar] = 0;
718  diff_coeff_beta_i[iScalar] = 0;
719  diff_coeff_beta_j[iScalar] = 0;
720  }
721  }
722 
723  numerics->SetScalarVar(scalar_i, scalar_j);
724 
725  numerics->SetScalarVarGradient(CMatrixView<su2double>(scalar_grad_i), CMatrixView<su2double>(scalar_grad_j));
726 
727  numerics->SetDiffusionCoeff(diff_coeff_beta_i, diff_coeff_beta_j);
728 
729  auto residual_thermal = numerics->ComputeResidual(config);
730 
731  if (ReducerStrategy) {
732  EdgeFluxes.SubtractBlock(iEdge, residual_thermal);
733  } else {
734  LinSysRes.SubtractBlock(iPoint, residual_thermal);
735  LinSysRes.AddBlock(jPoint, residual_thermal);
736  /* No implicit part for the preferential diffusion of heat */
737  }
738  }
739 }

742  const su2double* scalar_solution, su2double* val_enth) {

744  su2double delta_temp_final = 0.001;
745  su2double enth_iter = scalar_solution[I_ENTH];
746  su2double delta_enth;

754  for (auto iVar = 0u; iVar < nVar; iVar++) val_scalars[iVar] = scalar_solution[iVar];
755

757  /*--- Add all quantities and their names to the look up vectors. ---*/
758  val_scalars[I_ENTH] = enth_iter;

763 
764  delta_temp_iter = val_temp - Temperature;
765 
766  delta_enth = Cp * delta_temp_iter;
767 
768  enth_iter += delta_enth;
769  }
770 
771  *val_enth = enth_iter;
772 
773  if (counter >= counter_limit) {
774  exit_code = 1;
775  }
