37CSpeciesFlameletSolver::CSpeciesFlameletSolver(CGeometry* geometry, CConfig* config, unsigned short iMesh)
38 : CSpeciesSolver(geometry, config, true) {
39 /*--- Dimension of the problem. ---*/
40 nVar = config->GetNScalars();
41 include_mixture_fraction = (config->GetNControlVars() == 3);
42
43 Initialize(geometry, config, iMesh, nVar);
44
45 /*--- Initialize the solution to the far-field state everywhere. ---*/
46
47 nodes = new CSpeciesFlameletVariable(Solution_Inf, nPoint, nDim, nVar, config);
48 SetBaseClassPointerToNodes();
49
50 /*--- Store the initial CFL number for all grid points. ---*/
51
52 const su2double CFL = config->GetCFL(MGLevel) * config->GetCFLRedCoeff_Species();
53 SU2_OMP_FOR_STAT(omp_chunk_size)
54 for (auto iPoint = 0u; iPoint < nPoint; iPoint++) {
55 nodes->SetLocalCFL(iPoint, CFL);
56 }
57 END_SU2_OMP_FOR
58 Min_CFL_Local = CFL;
59 Max_CFL_Local = CFL;
60 Avg_CFL_Local = CFL;
61
62 /*--- Add the solver name. ---*/
63 SolverName = "FLAMELET";
64}

66void CSpeciesFlameletSolver::Preprocessing(CGeometry* geometry, CSolver** solver_container, CConfig* config,
67 unsigned short iMesh, unsigned short iRKStep,
68 unsigned short RunTime_EqSystem, bool Output) {
69 unsigned long n_not_in_domain_local = 0, n_not_in_domain_global = 0;
70 vector<su2double> scalars_vector(nVar);
71
72 auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
73
74 SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetGlobalParam(config->GetKind_Solver(), RunTime_EqSystem);)
75
76 SU2_OMP_FOR_STAT(omp_chunk_size)
77 for (auto i_point = 0u; i_point < nPoint; i_point++) {
78 CFluidModel* fluid_model_local = solver_container[FLOW_SOL]->GetFluidModel();
79 su2double* scalars = nodes->GetSolution(i_point);
80 for (auto iVar = 0u; iVar < nVar; iVar++) scalars_vector[iVar] = scalars[iVar];
81
82 /*--- Compute total source terms from the production and consumption. ---*/
83 unsigned long misses = SetScalarSources(config, fluid_model_local, i_point, scalars_vector);
84 nodes->SetTableMisses(i_point, misses);
85 n_not_in_domain_local += misses;
86 /*--- Obtain passive look-up scalars. ---*/
87 SetScalarLookUps(config, fluid_model_local, i_point, scalars_vector);
88
89 /*--- Set mass diffusivity based on thermodynamic state. ---*/
90 su2double T = flowNodes->GetTemperature(i_point);
91 fluid_model_local->SetTDState_T(T, scalars);
92 /*--- set the diffusivity in the fluid model to the diffusivity obtained from the lookup table ---*/
93 for (auto i_scalar = 0u; i_scalar < nVar; ++i_scalar) {
94 nodes->SetDiffusivity(i_point, fluid_model_local->GetMassDiffusivity(i_scalar), i_scalar);
95 }
96
97 if (!Output) LinSysRes.SetBlock_Zero(i_point);
98 }
99 END_SU2_OMP_FOR
100 /* --- Sum up some global counters over processes. --- */
101 SU2_MPI::Reduce(&n_not_in_domain_local, &n_not_in_domain_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
102 SU2_MPI::GetComm());
103 if ((rank == MASTER_NODE) && (n_not_in_domain_global > 0))
104 cout << "Number of points outside manifold domain: " << n_not_in_domain_global << endl;
105 /*--- Clear Residual and Jacobian. Upwind second order reconstruction and gradients ---*/
106 CommonPreprocessing(geometry, config, Output);
107}

109void CSpeciesFlameletSolver::SetInitialCondition(CGeometry** geometry, CSolver*** solver_container, CConfig* config,
110 unsigned long ExtIter) {
111 const bool Restart = (config->GetRestart() || config->GetRestart_Flow());
112
113 if ((!Restart) && ExtIter == 0) {
114 if (rank == MASTER_NODE) {
115 cout << "Initializing progress variable and total enthalpy (using temperature)" << endl;
116 }
117
118 const su2double* flame_init = config->GetFlameInit();
119 const su2double flame_offset[3] = {flame_init[0], flame_init[1], flame_init[2]};
120 const su2double flame_normal[3] = {flame_init[3], flame_init[4], flame_init[5]};
121 const su2double flame_thickness = flame_init[6];
122 const su2double flame_burnt_thickness = flame_init[7];
123
124 const su2double flamenorm = GeometryToolbox::Norm(nDim, flame_normal);
125 const su2double temp_inlet = config->GetInc_Temperature_Init();
126 su2double enth_inlet = config->GetSpecies_Init()[I_ENTH];
127
128 su2double prog_burnt = 0, prog_unburnt, point_loc;
129 su2double scalar_init[MAXNVAR];
130
131 if (rank == MASTER_NODE) {
132 cout << "initial condition: T = " << temp_inlet << endl;
133 for (auto iCV = 0u; iCV < config->GetNControlVars(); iCV++) {
134 const auto& cv_name = config->GetControllingVariableName(iCV);
135 cout << "initial condition: " << cv_name << " = " << config->GetSpecies_Init()[iCV] << endl;
136 }
137 }
138
139 CFluidModel* fluid_model_local;
140
141 unsigned long n_not_iterated_local = 0, n_not_in_domain_local = 0, n_points_unburnt_local = 0,
142 n_points_burnt_local = 0, n_points_flame_local = 0, n_not_iterated_global, n_not_in_domain_global,
143 n_points_burnt_global, n_points_flame_global, n_points_unburnt_global;
144
145 for (unsigned long i_mesh = 0; i_mesh <= config->GetnMGLevels(); i_mesh++) {
146 fluid_model_local = solver_container[i_mesh][FLOW_SOL]->GetFluidModel();
147
148 for (auto iVar = 0u; iVar < nVar; iVar++) scalar_init[iVar] = config->GetSpecies_Init()[iVar];
149
150 /*--- Set enthalpy based on initial temperature and scalars. ---*/
151 n_not_iterated_local += GetEnthFromTemp(fluid_model_local, temp_inlet, config->GetSpecies_Init(), &enth_inlet);
152 scalar_init[I_ENTH] = enth_inlet;
153
154 prog_unburnt = config->GetSpecies_Init()[I_PROGVAR];
155 prog_burnt = GetBurntProgressVariable(fluid_model_local, scalar_init);
156 SU2_OMP_FOR_STAT(omp_chunk_size)
157 for (unsigned long i_point = 0; i_point < nPoint; i_point++) {
158 auto coords = geometry[i_mesh]->nodes->GetCoord(i_point);
159
160 /*--- Determine if point is above or below the plane, assuming the normal
161 is pointing towards the burned region. ---*/
162 point_loc = 0.0;
163 for (unsigned short i_dim = 0; i_dim < geometry[i_mesh]->GetnDim(); i_dim++) {
164 point_loc += flame_normal[i_dim] * (coords[i_dim] - flame_offset[i_dim]);
165 }
166
167 /*--- Compute the exact distance from point to plane. ---*/
168 point_loc = point_loc / flamenorm;
169
170 /* --- Unburnt region upstream of the flame. --- */
171 if (point_loc <= 0) {
172 scalar_init[I_PROGVAR] = prog_unburnt;
173 n_points_unburnt_local++;
174
175 /* --- Flame zone where we lineary increase from unburnt to burnt conditions. --- */
176 } else if ((point_loc > 0) && (point_loc <= flame_thickness)) {
177 scalar_init[I_PROGVAR] = prog_unburnt + point_loc * (prog_burnt - prog_unburnt) / flame_thickness;
178 n_points_flame_local++;
179
180 /* --- Burnt region behind the flame zone. --- */
181 } else if ((point_loc > flame_thickness) && (point_loc <= flame_thickness + flame_burnt_thickness)) {
182 scalar_init[I_PROGVAR] = prog_burnt;
183 n_points_burnt_local++;
184
185 /* --- Unburnt region downstream of the flame and burnt region. --- */
186 } else {
187 scalar_init[I_PROGVAR] = prog_unburnt;
188 n_points_unburnt_local++;
189 }
190
191 /* --- Perform manifold evaluation to check whether initial scalar is out-of-bounds. --- */
192 fluid_model_local->SetTDState_T(temp_inlet, scalar_init);
193 n_not_in_domain_local += fluid_model_local->GetExtrapolation();
194
195 /* --- Initialize the auxiliary transported scalars (not controlling variables). --- */
196 for (int i_scalar = config->GetNControlVars(); i_scalar < config->GetNScalars(); ++i_scalar) {
197 scalar_init[i_scalar] = config->GetSpecies_Init()[i_scalar];
198 }
199
200 solver_container[i_mesh][SPECIES_SOL]->GetNodes()->SetSolution(i_point, scalar_init);
201 }
202
203 solver_container[i_mesh][SPECIES_SOL]->InitiateComms(geometry[i_mesh], config, SOLUTION);
204 solver_container[i_mesh][SPECIES_SOL]->CompleteComms(geometry[i_mesh], config, SOLUTION);
205
206 solver_container[i_mesh][FLOW_SOL]->InitiateComms(geometry[i_mesh], config, SOLUTION);
207 solver_container[i_mesh][FLOW_SOL]->CompleteComms(geometry[i_mesh], config, SOLUTION);
208
209 solver_container[i_mesh][FLOW_SOL]->Preprocessing(geometry[i_mesh], solver_container[i_mesh], config, i_mesh,
210 NO_RK_ITER, RUNTIME_FLOW_SYS, false);
211 END_SU2_OMP_FOR
212 }
213
214 /* --- Sum up some global counters over processes. --- */
215 SU2_MPI::Reduce(&n_not_in_domain_local, &n_not_in_domain_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
216 SU2_MPI::GetComm());
217 SU2_MPI::Reduce(&n_not_iterated_local, &n_not_iterated_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
218 SU2_MPI::GetComm());
219 SU2_MPI::Reduce(&n_points_unburnt_local, &n_points_unburnt_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
220 SU2_MPI::GetComm());
221 SU2_MPI::Reduce(&n_points_burnt_local, &n_points_burnt_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
222 SU2_MPI::GetComm());
223 SU2_MPI::Reduce(&n_points_flame_local, &n_points_flame_global, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE,
224 SU2_MPI::GetComm());
225
226 if (rank == MASTER_NODE) {
227 cout << endl;
228 cout << " Number of points in unburnt region: " << n_points_unburnt_global << "." << endl;
229 cout << " Number of points in burnt region : " << n_points_burnt_global << "." << endl;
230 cout << " Number of points in flame zone : " << n_points_flame_global << "." << endl;
231
232 if (n_not_in_domain_global > 0)
233 cout << " Initial condition: Number of points outside of table domain: " << n_not_in_domain_global << " !!!"
234 << endl;
235
236 if (n_not_iterated_global > 0)
237 cout << " Initial condition: Number of points in which enthalpy could not be iterated: "
238 << n_not_iterated_global << " !!!" << endl;
239 }
240 }

243void CSpeciesFlameletSolver::SetPreconditioner(CGeometry* geometry, CSolver** solver_container, CConfig* config) {
244 const bool variable_density = (config->GetVariable_Density_Model());
245 const bool implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
246
247 SU2_OMP_FOR_STAT(omp_chunk_size)
248 for (auto iPoint = 0u; iPoint < nPointDomain; iPoint++) {
249 /*--- Access the primitive variables at this node. ---*/
250
251 su2double Density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
252 su2double BetaInc2 = solver_container[FLOW_SOL]->GetNodes()->GetBetaInc2(iPoint);
253 su2double Temperature = solver_container[FLOW_SOL]->GetNodes()->GetTemperature(iPoint);
254
255 su2double SolP = solver_container[FLOW_SOL]->LinSysSol(iPoint, prim_idx.Pressure());
256 su2double SolT = solver_container[FLOW_SOL]->LinSysSol(iPoint, prim_idx.Temperature());
257
258 /*--- We need the derivative of the equation of state to build the
259 preconditioning matrix. For now, the only option is the ideal gas
260 law, but in the future, dRhodT should be in the fluid model. ---*/
261
262 su2double dRhodT = 0.0;
263 if (variable_density) {
264 dRhodT = -Density / Temperature;
265 }
266
267 /*--- Passive scalars have no impact on the density. ---*/
268
269 su2double dRhodC = 0.0;
270
271 /*--- Modify matrix diagonal with term including volume and time step. ---*/
272
273 su2double Vol = geometry->nodes->GetVolume(iPoint);
274 su2double Delta =
275 Vol / (config->GetCFLRedCoeff_Species() * solver_container[FLOW_SOL]->GetNodes()->GetDelta_Time(iPoint));
276
277 /*--- Calculating the inverse of the preconditioning matrix
278 that multiplies the time derivative during time integration. ---*/
279
280 if (implicit) {
281 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
282 su2double scalar = nodes->GetSolution(iPoint, iVar);
283
284 /*--- Compute the lag terms for the decoupled linear system from
285 the mean flow equations and add to the residual for the scalar.
286 In short, we are effectively making these terms explicit. ---*/
287
288 su2double artcompc1 = SolP * scalar / (Density * BetaInc2);
289 su2double artcompc2 = SolT * dRhodT * scalar / (Density);
290
291 LinSysRes(iPoint, iVar) += artcompc1 + artcompc2;
292
293 /*--- Add the extra Jacobian term to the scalar system. ---*/
294
295 su2double Jaccomp = scalar * dRhodC + Density;
296 su2double JacTerm = Jaccomp * Delta;
297
298 Jacobian.AddVal2Diag(iPoint, iVar, JacTerm);
299 }
300 }
301 }
302 END_SU2_OMP_FOR
303}

305void CSpeciesFlameletSolver::Source_Residual(CGeometry* geometry, CSolver** solver_container,
306 CNumerics** numerics_container, CConfig* config, unsigned short iMesh) {
307 SU2_OMP_FOR_STAT(omp_chunk_size)
308 for (auto i_point = 0u; i_point < nPointDomain; i_point++) {
309 /*--- Add source terms from the lookup table directly to the residual. ---*/
310 for (auto i_var = 0; i_var < nVar; i_var++) {
311 LinSysRes(i_point, i_var) -= nodes->GetScalarSources(i_point)[i_var] * geometry->nodes->GetVolume(i_point);
312 }
313 }
314 END_SU2_OMP_FOR
315
316 /*--- call the species solver for the shared sources (axisymmetric) ---*/
317 CSpeciesSolver::Source_Residual(geometry, solver_container, numerics_container, config, iMesh);
318}

335 }
336
337 /*--- Call the general inlet boundary condition implementation. ---*/
338 CSpeciesSolver::BC_Inlet(geometry, solver_container, conv_numerics, visc_numerics, config, val_marker);

341void CSpeciesFlameletSolver::BC_Isothermal_Wall_Generic(CGeometry* geometry, CSolver** solver_container,
342 CNumerics* conv_numerics, CNumerics* visc_numerics,
343 CConfig* config, unsigned short val_marker, bool cht_mode) {
344 const bool implicit = config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT;
345 const string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
346 su2double temp_wall;
347 CFluidModel* fluid_model_local = solver_container[FLOW_SOL]->GetFluidModel();
348 auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
349 su2double enth_wall;
350 unsigned long n_not_iterated = 0;
351
352 /*--- Loop over all the vertices on this boundary marker. ---*/
353 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
354 for (unsigned long iVertex = 0; iVertex < geometry->nVertex[val_marker]; iVertex++) {
355 unsigned long iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
356
357 if (cht_mode)
358 temp_wall = solver_container[FLOW_SOL]->GetConjugateHeatVariable(val_marker, iVertex, 0);
359 else
360 temp_wall = config->GetIsothermal_Temperature(Marker_Tag);
361
362 /*--- Check if the node belongs to the domain (i.e., not a halo node). ---*/
363
364 if (geometry->nodes->GetDomain(iPoint)) {
365 if (config->GetMarker_StrongBC(Marker_Tag) == true) {
366 /*--- Initial guess for enthalpy value. ---*/
367 enth_wall = nodes->GetSolution(iPoint, I_ENTH);
368
369 /*--- Set enthalpy on the wall. ---*/
370 n_not_iterated += GetEnthFromTemp(fluid_model_local, temp_wall, nodes->GetSolution(iPoint), &enth_wall);
371
372 /*--- Impose the value of the enthalpy as a strong boundary
373 condition (Dirichlet) and remove any
374 contribution to the residual at this node. ---*/
375
376 nodes->SetSolution(iPoint, I_ENTH, enth_wall);
377 nodes->SetSolution_Old(iPoint, I_ENTH, enth_wall);
378
379 LinSysRes(iPoint, I_ENTH) = 0.0;
380
381 nodes->SetVal_ResTruncError_Zero(iPoint, I_ENTH);
382
383 if (implicit) {
384 unsigned long total_index = iPoint * nVar + I_ENTH;
385 Jacobian.DeleteValsRowi(total_index);
386 }
387 } else {
388 /*--- Weak BC formulation. ---*/
389 const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
390
391 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
392
393 const auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
394
395 /*--- Get coordinates of i & nearest normal and compute distance. ---*/
396
397 const auto Coord_i = geometry->nodes->GetCoord(iPoint);
398 const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);
399 su2double Edge_Vector[MAXNDIM];
400 GeometryToolbox::Distance(nDim, Coord_j, Coord_i, Edge_Vector);
401 su2double dist_ij_2 = GeometryToolbox::SquaredNorm(nDim, Edge_Vector);
402 su2double dist_ij = sqrt(dist_ij_2);
403
404 /*--- Compute the normal gradient in temperature using Twall. ---*/
405
406 su2double dTdn = -(flowNodes->GetTemperature(Point_Normal) - temp_wall) / dist_ij;
407
408 /*--- Get thermal conductivity. ---*/
409
410 su2double thermal_conductivity = flowNodes->GetThermalConductivity(iPoint);
411
412 /*--- Apply a weak boundary condition for the energy equation.
413 Compute the residual due to the prescribed heat flux. ---*/
414
415 LinSysRes(iPoint, I_ENTH) -= thermal_conductivity * dTdn * Area;
416 }
417 }
418 }
419 END_SU2_OMP_FOR
420
421 if (rank == MASTER_NODE && n_not_iterated > 0) {
422 cout << " !!! Wall bc (" << Marker_Tag
423 << "): Number of points in which enthalpy could not be iterated: " << n_not_iterated << " !!!" << endl;
424 }
425}

439unsigned long CSpeciesFlameletSolver::SetScalarSources(const CConfig* config, CFluidModel* fluid_model_local,
440 unsigned long iPoint, const vector<su2double>& scalars) {
441 /*--- Compute total source terms from the production and consumption. ---*/
442
443 vector<su2double> table_sources(config->GetNControlVars() + 2 * config->GetNUserScalars());
444 unsigned long misses = fluid_model_local->EvaluateDataSet(scalars, FLAMELET_LOOKUP_OPS::SOURCES, table_sources);
445 nodes->SetTableMisses(iPoint, misses);
446
447 /*--- The source term for progress variable is always positive, we clip from below to makes sure. --- */
448
449 vector<su2double> source_scalar(config->GetNScalars());
450 for (auto iCV = 0u; iCV < config->GetNControlVars(); iCV++) source_scalar[iCV] = table_sources[iCV];
451 source_scalar[I_PROGVAR] = fmax(EPS, source_scalar[I_PROGVAR]);
452
453 /*--- Source term for the auxiliary species transport equations. ---*/
454 for (size_t i_aux = 0; i_aux < config->GetNUserScalars(); i_aux++) {
455 /*--- The source term for the auxiliary equations consists of a production term and a consumption term:
456 S_TOT = S_PROD + S_CONS * Y ---*/
457 su2double y_aux = scalars[config->GetNControlVars() + i_aux];
458 su2double source_prod = table_sources[config->GetNControlVars() + 2 * i_aux];
459 su2double source_cons = table_sources[config->GetNControlVars() + 2 * i_aux + 1];
460 source_scalar[config->GetNControlVars() + i_aux] = source_prod + source_cons * y_aux;
461 }
462 for (auto i_scalar = 0u; i_scalar < nVar; i_scalar++)
463 nodes->SetScalarSource(iPoint, i_scalar, source_scalar[i_scalar]);
464 return misses;
465}

467unsigned long CSpeciesFlameletSolver::SetScalarLookUps(const CConfig* config, CFluidModel* fluid_model_local,
468 unsigned long iPoint, const vector<su2double>& scalars) {
469 /*--- Compute total source terms from the production and consumption. ---*/
470
471 vector<su2double> lookup_scalar(config->GetNLookups());
472 unsigned long misses = fluid_model_local->EvaluateDataSet(scalars, FLAMELET_LOOKUP_OPS::LOOKUP, lookup_scalar);
473
474 for (auto i_lookup = 0u; i_lookup < config->GetNLookups(); i_lookup++) {
475 nodes->SetLookupScalar(iPoint, lookup_scalar[i_lookup], i_lookup);
476 }
477 return misses;
478}

480unsigned long CSpeciesFlameletSolver::GetEnthFromTemp(CFluidModel* fluid_model, su2double const val_temp,
481 const su2double* scalar_solution, su2double* val_enth) {
482 /*--- convergence criterion for temperature in [K], high accuracy needed for restarts. ---*/
483 su2double delta_temp_final = 0.001;
484 su2double enth_iter = scalar_solution[I_ENTH];
485 su2double delta_enth;
486 su2double delta_temp_iter = 1e10;
487 unsigned long exit_code = 0;
488 const int counter_limit = 1000;
489
490 int counter = 0;
491
492 su2double val_scalars[MAXNVAR];
493 for (auto iVar = 0u; iVar < nVar; iVar++) val_scalars[iVar] = scalar_solution[iVar];
494
495 while ((abs(delta_temp_iter) > delta_temp_final) && (counter++ < counter_limit)) {
496 /*--- Add all quantities and their names to the look up vectors. ---*/
497 val_scalars[I_ENTH] = enth_iter;
498 fluid_model->SetTDState_T(val_temp, val_scalars);
499
500 su2double Temperature = fluid_model->GetTemperature();
501 su2double Cp = fluid_model->GetCp();
502
503 delta_temp_iter = val_temp - Temperature;
