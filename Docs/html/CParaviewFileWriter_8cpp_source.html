38void CParaviewFileWriter::WriteData(string val_filename){
39
40 /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
41 val_filename.append(fileExt);
42
43 if (!dataSorter->GetConnectivitySorted()){
44 SU2_MPI::Error("Connectivity must be sorted.", CURRENT_FUNCTION);
45 }
46
47 unsigned short iDim = 0, nDim = dataSorter->GetnDim();
48
49 unsigned long iPoint, iElem;
50
51 unsigned long nGlobal_Elem_Storage;
52
53 ofstream Paraview_File;
54
55 int iProcessor;
56
57 const vector<string> fieldNames = dataSorter->GetFieldNames();
58
59 /*--- Set a timer for the file writing. ---*/
60
61 startTime = SU2_MPI::Wtime();
62
63 /*--- Open Paraview ASCII file and write the header. ---*/
64
65 if (rank == MASTER_NODE) {
66
67 Paraview_File.open(val_filename.c_str(), ios::out);
68 Paraview_File.precision(6);
69 Paraview_File << "# vtk DataFile Version 3.0\n";
70 Paraview_File << "vtk output\n";
71 Paraview_File << "ASCII\n";
72 Paraview_File << "DATASET UNSTRUCTURED_GRID\n";
73
74 /*--- Write the header ---*/
75 Paraview_File << "POINTS "<< dataSorter->GetnPointsGlobal() <<" double\n";
76
77 }
78
79 Paraview_File.close();
80
81#ifdef HAVE_MPI
82 SU2_MPI::Barrier(SU2_MPI::GetComm());
83#endif
84
85 /*--- Each processor opens the file. ---*/
86
87 Paraview_File.open(val_filename.c_str(), ios::out | ios::app);
88
89 /*--- Write surface and volumetric point coordinates. ---*/
90
91 for (iProcessor = 0; iProcessor < size; iProcessor++) {
92 if (rank == iProcessor) {
93
94 /*--- Write the node data from this proc ---*/
95
96
97 for (iPoint = 0; iPoint < dataSorter->GetnPoints(); iPoint++) {
98 for (iDim = 0; iDim < nDim; iDim++)
99 Paraview_File << scientific << dataSorter->GetData(iDim, iPoint) << "\t";
100 if (nDim == 2) Paraview_File << scientific << "0.0" << "\t";
101 }
102 }
103
104 Paraview_File.flush();
105#ifdef HAVE_MPI
106 SU2_MPI::Barrier(SU2_MPI::GetComm());
107#endif
108 }
109
110 /*--- Reduce the total number of each element. ---*/
111
112 unsigned long nParallel_Line = dataSorter->GetnElem(LINE),
113 nParallel_Tria = dataSorter->GetnElem(TRIANGLE),
114 nParallel_Quad = dataSorter->GetnElem(QUADRILATERAL),
115 nParallel_Tetr = dataSorter->GetnElem(TETRAHEDRON),
116 nParallel_Hexa = dataSorter->GetnElem(HEXAHEDRON),
117 nParallel_Pris = dataSorter->GetnElem(PRISM),
118 nParallel_Pyra = dataSorter->GetnElem(PYRAMID);
119
120 if (rank == MASTER_NODE) {
121
122 /*--- Write the header ---*/
123 nGlobal_Elem_Storage = dataSorter->GetnElemGlobal() + dataSorter->GetnConnGlobal();
124
125 Paraview_File << "\nCELLS " << dataSorter->GetnElemGlobal() << "\t" << nGlobal_Elem_Storage << "\n";
126
127 }
128
129 Paraview_File.flush();
130#ifdef HAVE_MPI
131 SU2_MPI::Barrier(SU2_MPI::GetComm());
132#endif
133
134 /*--- Write connectivity data. ---*/
135
136 for (iProcessor = 0; iProcessor < size; iProcessor++) {
137 if (rank == iProcessor) {
138
139
140 for (iElem = 0; iElem < nParallel_Line; iElem++) {
141 Paraview_File << N_POINTS_LINE << "\t";
142 Paraview_File << dataSorter->GetElemConnectivity(LINE, iElem, 0)-1 << "\t";
143 Paraview_File << dataSorter->GetElemConnectivity(LINE, iElem, 1)-1 << "\t";
144 }
145
146 for (iElem = 0; iElem < nParallel_Tria; iElem++) {
147 Paraview_File << N_POINTS_TRIANGLE << "\t";
148 Paraview_File << dataSorter->GetElemConnectivity(TRIANGLE, iElem, 0)-1 << "\t";
149 Paraview_File << dataSorter->GetElemConnectivity(TRIANGLE, iElem, 1)-1 << "\t";
150 Paraview_File << dataSorter->GetElemConnectivity(TRIANGLE, iElem, 2)-1 << "\t";
151 }
152
153 for (iElem = 0; iElem < nParallel_Quad; iElem++) {
154 Paraview_File << N_POINTS_QUADRILATERAL << "\t";
155 Paraview_File << dataSorter->GetElemConnectivity(QUADRILATERAL, iElem, 0)-1 << "\t";
156 Paraview_File << dataSorter->GetElemConnectivity(QUADRILATERAL, iElem, 1)-1 << "\t";
157 Paraview_File << dataSorter->GetElemConnectivity(QUADRILATERAL, iElem, 2)-1 << "\t";
158 Paraview_File << dataSorter->GetElemConnectivity(QUADRILATERAL, iElem, 3)-1 << "\t";
159 }
160
161
162 for (iElem = 0; iElem < nParallel_Tetr; iElem++) {
163 Paraview_File << N_POINTS_TETRAHEDRON << "\t";
164 Paraview_File << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 0)-1 << "\t"
165 << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 1)-1 << "\t";
166 Paraview_File << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 2)-1 << "\t"
167 << dataSorter->GetElemConnectivity(TETRAHEDRON, iElem, 3)-1 << "\t";
168 }
169
170 for (iElem = 0; iElem < nParallel_Hexa; iElem++) {
171 Paraview_File << N_POINTS_HEXAHEDRON << "\t";
172 Paraview_File << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 0)-1 << "\t"
173 << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 1)-1 << "\t";
174 Paraview_File << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 2)-1 << "\t"
175 << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 3)-1 << "\t";
176 Paraview_File << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 4)-1 << "\t"
177 << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 5)-1 << "\t";
178 Paraview_File << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 6)-1 << "\t"
179 << dataSorter->GetElemConnectivity(HEXAHEDRON, iElem, 7)-1 << "\t";
180 }
181
182 for (iElem = 0; iElem < nParallel_Pris; iElem++) {
183 Paraview_File << N_POINTS_PRISM << "\t";
184 Paraview_File << dataSorter->GetElemConnectivity(PRISM, iElem, 0)-1 << "\t"
185 << dataSorter->GetElemConnectivity(PRISM, iElem, 1)-1 << "\t";
186 Paraview_File << dataSorter->GetElemConnectivity(PRISM, iElem, 2)-1 << "\t"
187 << dataSorter->GetElemConnectivity(PRISM, iElem, 3)-1 << "\t";
188 Paraview_File << dataSorter->GetElemConnectivity(PRISM, iElem, 4)-1 << "\t"
189 << dataSorter->GetElemConnectivity(PRISM, iElem, 5)-1 << "\t";
190 }
191
192 for (iElem = 0; iElem < nParallel_Pyra; iElem++) {
193 Paraview_File << N_POINTS_PYRAMID << "\t";
194 Paraview_File << dataSorter->GetElemConnectivity(PYRAMID, iElem, 0)-1 << "\t"
195 << dataSorter->GetElemConnectivity(PYRAMID, iElem, 1)-1 << "\t";
196 Paraview_File << dataSorter->GetElemConnectivity(PYRAMID, iElem, 2)-1 << "\t"
197 << dataSorter->GetElemConnectivity(PYRAMID, iElem, 3)-1 << "\t";
198 Paraview_File << dataSorter->GetElemConnectivity(PYRAMID, iElem, 4)-1 << "\t";
199 }
200
201 } Paraview_File.flush();
202#ifdef HAVE_MPI
203 SU2_MPI::Barrier(SU2_MPI::GetComm());
204#endif
205 }
206
207 if (rank == MASTER_NODE) {
208
209 /*--- Write the header ---*/
210 Paraview_File << "\nCELL_TYPES " << dataSorter->GetnElemGlobal() << "\n";
211
212 }
213
214 Paraview_File.flush();
215#ifdef HAVE_MPI
216 SU2_MPI::Barrier(SU2_MPI::GetComm());
217#endif
218
219 for (iProcessor = 0; iProcessor < size; iProcessor++) {
220 if (rank == iProcessor) {
221 for (iElem = 0; iElem < nParallel_Line; iElem++) Paraview_File << "3\t";
222 for (iElem = 0; iElem < nParallel_Tria; iElem++) Paraview_File << "5\t";
223 for (iElem = 0; iElem < nParallel_Quad; iElem++) Paraview_File << "9\t";
224 for (iElem = 0; iElem < nParallel_Tetr; iElem++) Paraview_File << "10\t";
225 for (iElem = 0; iElem < nParallel_Hexa; iElem++) Paraview_File << "12\t";
226 for (iElem = 0; iElem < nParallel_Pris; iElem++) Paraview_File << "13\t";
227 for (iElem = 0; iElem < nParallel_Pyra; iElem++) Paraview_File << "14\t";
228 }
229 Paraview_File.flush();
230#ifdef HAVE_MPI
231 SU2_MPI::Barrier(SU2_MPI::GetComm());
232#endif
233 }
234
235 if (rank == MASTER_NODE) {
236 /*--- Write the header ---*/
237 Paraview_File << "\nPOINT_DATA "<< dataSorter->GetnPointsGlobal() <<"\n";
238
239 }
240
241 Paraview_File.flush();
242#ifdef HAVE_MPI
243 SU2_MPI::Barrier(SU2_MPI::GetComm());
244#endif
245
246 unsigned short varStart = 2;
247 if (nDim == 3) varStart++;
248
249 /*--- Need to adjust container location to avoid PointID tag and coords. ---*/
250 unsigned short VarCounter = varStart;
251
252 for (unsigned short iField = varStart; iField < fieldNames.size(); iField++) {
253
254 string fieldname = fieldNames[iField];
255
256 fieldname.erase(remove(fieldname.begin(), fieldname.end(), '"'), fieldname.end());
257
258 bool output_variable = true, isVector = false;
259 size_t found = fieldNames[iField].find("_x");
260 if (found!=string::npos) {
261 output_variable = true;
262 isVector = true;
263 }
264 found = fieldNames[iField].find("_y");
265 if (found!=string::npos) {
266 output_variable = false;
267 //skip
268 Paraview_File.flush();
269#ifdef HAVE_MPI
270 SU2_MPI::Barrier(SU2_MPI::GetComm());
271#endif
272 VarCounter++;
273 }
274 found = fieldNames[iField].find("_z");
275 if (found!=string::npos) {
276 output_variable = false;
277 //skip
278 Paraview_File.flush();
279#ifdef HAVE_MPI
280 SU2_MPI::Barrier(SU2_MPI::GetComm());
281#endif
282 VarCounter++;
283 }
284
285 if (output_variable && isVector) {
286
287 fieldname.erase(fieldname.end()-2,fieldname.end());
288
289 if (rank == MASTER_NODE) {
290 Paraview_File << "\nVECTORS " << fieldname << " double\n";
291 }
292
293 Paraview_File.flush();
294#ifdef HAVE_MPI
295 SU2_MPI::Barrier(SU2_MPI::GetComm());
296#endif
297
298 /*--- Write surface and volumetric point coordinates. ---*/
299
300 for (iProcessor = 0; iProcessor < size; iProcessor++) {
301 if (rank == iProcessor) {
302
303 /*--- Write the node data from this proc ---*/
304
305 for (iPoint = 0; iPoint < dataSorter->GetnPoints(); iPoint++) {
306 Paraview_File << scientific << dataSorter->GetData(VarCounter+0, iPoint) << "\t" << dataSorter->GetData(VarCounter+1, iPoint) << "\t";
307 if (nDim == 3) Paraview_File << scientific << dataSorter->GetData(VarCounter+2, iPoint) << "\t";
308 if (nDim == 2) Paraview_File << scientific << "0.0" << "\t";
309 }
310 }
311
312 Paraview_File.flush();
313#ifdef HAVE_MPI
314 SU2_MPI::Barrier(SU2_MPI::GetComm());
315#endif
316 }
317
318 VarCounter++;
319
320 } else if (output_variable) {
321
322 if (rank == MASTER_NODE) {
323
324 Paraview_File << "\nSCALARS " << fieldname << " double 1\n";
325 Paraview_File << "LOOKUP_TABLE default\n";
326 }
327
328 Paraview_File.flush();
329#ifdef HAVE_MPI
330 SU2_MPI::Barrier(SU2_MPI::GetComm());
331#endif
332
333 /*--- Write surface and volumetric point coordinates. ---*/
334
335 for (iProcessor = 0; iProcessor < size; iProcessor++) {
336 if (rank == iProcessor) {
337
338 /*--- Write the node data from this proc ---*/
339
340 for (iPoint = 0; iPoint < dataSorter->GetnPoints(); iPoint++) {
341 Paraview_File << scientific << dataSorter->GetData(VarCounter, iPoint) << "\t";
342 }
343
344 }
345 Paraview_File.flush();
346#ifdef HAVE_MPI
347 SU2_MPI::Barrier(SU2_MPI::GetComm());
348#endif
349 }
350
351 VarCounter++;
352 }
353
354 }
355
356 Paraview_File.close();
357
358
359 /*--- Compute and store the write time. ---*/
360
361 stopTime = SU2_MPI::Wtime();
362
363 usedTime = stopTime-startTime;
364
365 fileSize = DetermineFilesize(val_filename);
366
367 /*--- Compute and store the bandwidth ---*/
368
369 bandwidth = fileSize/(1.0e6)/usedTime;
370}
