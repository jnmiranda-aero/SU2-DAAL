32CMMSNSUnitQuadSolutionWallBC::CMMSNSUnitQuadSolutionWallBC(unsigned short val_nDim, unsigned short val_nVar,
33 unsigned short val_iMesh, CConfig* config)
34 : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
35 /*--- Write a message that the solution is initialized for the manufactured
36 solution for the Navier-Stokes equations on a unit quad with no-slip
37 wall boundary conditions. ---*/
38 if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
39 cout << endl;
40 cout << "Warning: Fluid properties and solution are being " << endl;
41 cout << " initialized for the manufactured solution " << endl;
42 cout << " of the Navier-Stokes equations on a unit quad" << endl;
43 cout << " with no-slip wall boundary conditions!!!" << endl;
44 cout << endl << flush;
45 }
46
47 /*--- Coefficients, needed to determine the solution. ---*/
48 const su2double Prandtl = config->GetPrandtl_Lam();
49
50 RGas = config->GetGas_Constant();
51 Gamma = config->GetGamma();
52 Viscosity = config->GetMu_Constant();
53 Conductivity = Viscosity * Gamma * RGas / (Prandtl * (Gamma - 1.0));
54
55 /*--- Initialize TWall to the default value of 300 K (in case the outer wall
56 is not modelled as an isothermal wall) and try to retrieve the wall
57 temperature from the boundary conditions. ---*/
58 TWall = 300.0;
59 for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
60 if (config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL) {
61 const string Marker_Tag = config->GetMarker_All_TagBound(iMarker);
62 TWall = config->GetIsothermal_Temperature(Marker_Tag);
63 }
64 }
65
66 /*--- Get the reference values for pressure, density and velocity. ---*/
67 Pressure_Ref = config->GetPressure_Ref();
68 Density_Ref = config->GetDensity_Ref();
69 Velocity_Ref = config->GetVelocity_Ref();
70
71 /*--- The constants for the density and velocities. ---*/
72 rho_0 = 1.25;
73 u_0 = 135.78;
74 v_0 = -67.61;
75
76 /*--- The constants for the temperature solution. ---*/
77 a_T1 = 1.05;
78 a_T2 = -0.85;
79
80 /*--- Perform some sanity and error checks for this solution here. ---*/
81 if (config->GetTime_Marching() != TIME_MARCHING::STEADY)
82 SU2_MPI::Error("Steady mode must be selected for the MMS NS Unit Quad case with wall BCs.", CURRENT_FUNCTION);
83
84 if (Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES &&
85 Kind_Solver != MAIN_SOLVER::RANS && Kind_Solver != MAIN_SOLVER::FEM_EULER &&
86 Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
87 Kind_Solver != MAIN_SOLVER::FEM_LES)
88 SU2_MPI::Error("Compressible flow equations must be selected for the MMS NS Unit Quad case with wall BCs.",
89 CURRENT_FUNCTION);
90
91 if ((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) && (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
92 SU2_MPI::Error("Navier Stokes equations must be selected for the MMS NS Unit Quad case with wall BCs.",
93 CURRENT_FUNCTION);
94
95 if ((config->GetKind_FluidModel() != STANDARD_AIR) && (config->GetKind_FluidModel() != IDEAL_GAS))
96 SU2_MPI::Error("Standard air or ideal gas must be selected for the MMS NS Unit Quad case with wall BCs.",
97 CURRENT_FUNCTION);
98
99 if (config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
100 SU2_MPI::Error("Sutherland must be selected for viscosity for the MMS NS Unit Quad case with wall BCs.",
101 CURRENT_FUNCTION);
102
103 if (config->GetKind_ConductivityModel() != CONDUCTIVITYMODEL::CONSTANT_PRANDTL)
104 SU2_MPI::Error("Constant Prandtl number must be selected for the MMS NS Unit Quad case with wall BCs.",
105 CURRENT_FUNCTION);
106}

116void CMMSNSUnitQuadSolutionWallBC::GetSolution(const su2double* val_coords, const su2double val_t,
117 su2double* val_solution) const {
118 /* Easier storage of the y-coordinate. */
119 const su2double y = val_coords[1];
120
121 /* Determine the dimensional solution for the temperature. */
122 const su2double Pi = PI_NUMBER;
123 const su2double fact = y * y / (a_T1 + a_T2);
124
125 su2double T = 0.25 * TWall * (3.0 + fact * (a_T1 * cos(Pi * (y - 1.0)) + a_T2 * cos(Pi * (y - 1.0) * 2.0)));
126
127 /* Determine the dimensional solution for the velocities. */
128 su2double u = u_0 * y * (1.0 - y) * 4.0;
129 su2double v = v_0 * y * (1.0 - y) * 4.0;
130
131 /* Compute the pressure from the density and temperature. */
132 su2double rho = rho_0;
133 su2double p = rho * RGas * T;
134
135 /* Determine the non-dimensional solution. */
136 rho /= Density_Ref;
137 p /= Pressure_Ref;
138 u /= Velocity_Ref;
139 v /= Velocity_Ref;
140
141 /* Compute the conservative variables from the primitive ones.
142 Note that the implementation below is valid for both 2D and 3D. */
143 val_solution[0] = rho;
144 val_solution[1] = rho * u;
145 val_solution[2] = rho * v;
146 val_solution[3] = 0.0;
147 val_solution[nDim + 1] = p / (Gamma - 1.0) + 0.5 * rho * (u * u + v * v);
148}

150void CMMSNSUnitQuadSolutionWallBC::GetMMSSourceTerm(const su2double* val_coords, const su2double val_t,
151 su2double* val_source) const {
152 /*--- Abbreviate Pi and the y-coordinate. ---*/
153 const su2double Pi = PI_NUMBER;
154 const su2double y = val_coords[1];
155
156 /*--- The source code for the source terms is generated in Maple.
157 See the file CMMSNSUnitQuadSolutionWallBC.mw in the directory
158 CreateMMSSourceTerms for the details how to do this. ---*/
159 const su2double t1 = (v_0 * rho_0);
160 const su2double t2 = 1.0 - y;
161 const su2double t6 = rho_0 * u_0;
162 const su2double t7 = t2 * t2;
163 const su2double t8 = t7 * y;
164 const su2double t12 = y * y;
165 const su2double t13 = t2 * t12;
166 const su2double t20 = (v_0 * v_0);
167 const su2double t21 = (t20 * rho_0);
168 const su2double t26 = rho_0 * RGas;
169 const su2double t27 = a_T1 + a_T2;
170 const su2double t28 = 1.0 / t27;
171 const su2double t30 = -t2 * Pi;
172 const su2double t31 = cos(t30);
173 const su2double t32 = t31 * a_T1;
174 const su2double t33 = 2.0 * t30;
175 const su2double t34 = cos(t33);
176 const su2double t36 = t34 * a_T2 + t32;
177 const su2double t39 = t28 * t12;
178 const su2double t41 = sin(t30);
179 const su2double t42 = t41 * Pi * a_T1;
180 const su2double t43 = Pi * a_T2;
181 const su2double t44 = sin(t33);
182 const su2double t50 = (0.2e1 * t36 * t28 * y + (-0.2e1 * t44 * t43 - t42) * t39) * TWall;
183 const su2double t52 = t50 * t26 / 0.4e1;
184 const su2double t57 = 1.0 / (Gamma - 1.0);
185 const su2double t61 = (u_0 * u_0);
186 const su2double t62 = y * t61;
187 const su2double t64 = t12 * t61;
188 const su2double t66 = y * t20;
189 const su2double t68 = t12 * t20;
190 const su2double t75 = t2 * y;
191 const su2double t80 = (t36 * t39 + 0.3e1) * TWall;
192 const su2double t92 = v_0 * (t57 * t80 * t26 / 0.4e1 + (8 * (t7 * t64 + t7 * t68) * rho_0) + t80 * t26 / 0.4e1);
193 const su2double t97 = t41 * t31;
194 const su2double t102 = a_T1 * y;
195 const su2double t104 = t31 * t31;
196 const su2double t111 = 0.1e1 / t27 / 0.4e1;
197 const su2double t115 = Pi * Pi;
198 const su2double t116 = t41 * t41;
199 const su2double t118 = a_T2 * y;
200 const su2double t138 = Viscosity * (-8.0 * y + 4.0);
201 const su2double t149 = Viscosity * (2.0 * y - 1.0);
202 const su2double t155 =
203 0.4e1 * t75 * v_0 *
204 (t57 * t50 * t26 / 0.4e1 + (16.0 * (-t2 * t64 - t2 * t68 + t7 * t62 + t7 * t66) * rho_0) + t52) +
205 0.4e1 * t2 * t92 - 0.4e1 * y * t92 -
206 Conductivity * t111 *
207 (-t102 * Pi * t41 - 0.4e1 * y * t43 * t97 + 0.4e1 * t104 * a_T2 - 0.2e1 * a_T2 + 0.2e1 * t32) * TWall -
208 Conductivity * t111 *
209 (-t102 * t31 * t115 - 0.4e1 * t118 * t115 * t104 + 0.4e1 * t118 * t116 * t115 - 0.12e2 * t43 * t97 -
210 0.3e1 * t42) *
211 TWall * y +
212 (32.0 * t75 * t61 * Viscosity) - (4.0 * t2 * t61 * t138) + (4.0 * t62 * t138) +
213 0.128e3 / 0.3e1 * t75 * t20 * Viscosity + 0.64e2 / 0.3e1 * t2 * t20 * t149 - 0.64e2 / 0.3e1 * t66 * t149;
214
215 /*--- Set the source term, which is valid for both 2D and 3D cases.
216 Note the scaling for the correct non-dimensionalization. ---*/
217 val_source[0] = 4.0 * t2 * t1 - 4.0 * y * t1;
218 val_source[1] = (-0.32e2 * v_0 * t13 * t6 + 0.32e2 * v_0 * t8 * t6 + (8.0 * Viscosity * u_0));
219 val_source[2] = ((32.0 * t8 * t21) - (32.0 * t13 * t21) + t52 + 0.32e2 / 0.3e1 * Viscosity * v_0);
220 val_source[3] = 0.0;
221 val_source[nDim + 1] = t155;
222
223 val_source[0] /= Density_Ref * Velocity_Ref;
224 val_source[1] /= Pressure_Ref;
225 val_source[2] /= Pressure_Ref;
226 val_source[nDim + 1] /= Velocity_Ref * Pressure_Ref;
227}
