34void CPoint::MinimalAllocation(unsigned long npoint) {
35 /*--- Global index a parallel simulation. ---*/
36 GlobalIndex.resize(npoint) = 0;
37
38 /*--- Set the color for mesh partitioning. ---*/
39 Color.resize(npoint) = 0;
40
41 /*--- Coordinates. ---*/
42 Coord.resize(npoint, nDim) = su2double(0.0);
43}

51void CPoint::FullAllocation(unsigned short imesh, const CConfig* config) {
52 const auto npoint = GlobalIndex.size();
53
54 /*--- Volumes ---*/
55
56 Volume.resize(npoint) = su2double(0.0);
57 Periodic_Volume.resize(npoint) = su2double(0.0);
58
59 if (config->GetTime_Marching() != TIME_MARCHING::STEADY) {
60 Volume_n.resize(npoint) = su2double(0.0);
61 Volume_nM1.resize(npoint) = su2double(0.0);
62 if (config->GetDynamic_Grid() && config->GetDiscrete_Adjoint()) {
63 Volume_Old.resize(npoint) = su2double(0.0);
64 Volume_n_Old.resize(npoint) = su2double(0.0);
65 Volume_nM1_Old.resize(npoint) = su2double(0.0);
66 }
67 }
68
69 if (config->GetDiscrete_Adjoint()) {
70 AD_InputIndex.resize(npoint, nDim) = 0;
71 AD_OutputIndex.resize(npoint, nDim) = 0;
72 }
73
74 /*--- Multigrid structures. ---*/
75 if (config->GetnMGLevels() > 0) {
76 Parent_CV.resize(npoint) = 0;
77 Agglomerate.resize(npoint) = false;
78 Agglomerate_Indirect.resize(npoint) = false;
79 /*--- The finest grid does not have children CV's. ---*/
80 if (imesh != MESH_0) {
81 nChildren_CV.resize(npoint) = 0;
82 Children_CV.resize(npoint);
83 }
84 }
85
86 /*--- Identify boundaries, physical boundaries (not send-receive condition), detect if
87 * an element belong to the domain or it must be computed with other processor. ---*/
88 Domain.resize(npoint) = true;
89 Boundary.resize(npoint) = false;
90 SolidBoundary.resize(npoint) = false;
91 ViscousBoundary.resize(npoint) = false;
92 PhysicalBoundary.resize(npoint) = false;
93 PeriodicBoundary.resize(npoint) = false;
94
95 Vertex.resize(npoint);
96
97 /*--- For smoothing the numerical grid coordinates ---*/
98 if (config->GetSmoothNumGrid()) {
99 Coord_Old.resize(npoint, nDim) = su2double(0.0);
100 Coord_Sum.resize(npoint, nDim) = su2double(0.0);
101 }
102
103 /*--- Storage of grid velocities for dynamic meshes. ---*/
104
105 if (config->GetDynamic_Grid()) {
106 GridVel.resize(npoint, nDim) = su2double(0.0);
107
108 /*--- Grid velocity gradients are needed for the continuous adjoint. ---*/
109 if (config->GetContinuous_Adjoint()) GridVel_Grad.resize(npoint, nDim, nDim, 0.0);
110
111 /*--- Structures for storing old node coordinates for computing grid
112 * velocities via finite differencing with dynamically deforming meshes. ---*/
113 /*--- In the case of CMeshSolver, these coordinates are stored as solutions to the mesh problem. ---*/
114 if (config->GetGrid_Movement() && (config->GetTime_Marching() != TIME_MARCHING::STEADY)) {
115 Coord_n.resize(npoint, nDim) = su2double(0.0);
116 Coord_p1.resize(npoint, nDim) = su2double(0.0);
117 Coord_n1.resize(npoint, nDim) = su2double(0.0);
118 if (Coord_Old.empty()) Coord_Old.resize(npoint, nDim) = su2double(0.0);
119 }
120 }
121
122 /*--- Other geometric properties of the CV's required by numerical methods. ---*/
123 nNeighbor.resize(npoint) = 0;
124 MaxLength.resize(npoint) = su2double(0.0);
125 Curvature.resize(npoint) = su2double(0.0);
126
127 Wall_Distance.resize(npoint) = su2double(0.0);
128 ClosestWall_Rank.resize(npoint) = -1;
129 ClosestWall_Zone.resize(npoint) = numeric_limits<unsigned short>::max();
130 ClosestWall_Marker = ClosestWall_Zone;
131 ClosestWall_Elem.resize(npoint) = numeric_limits<unsigned long>::max();
132
133 RoughnessHeight.resize(npoint) = su2double(0.0);
134 SharpEdge_Distance.resize(npoint) = su2double(0.0);
135}
