38CVerificationSolution::CVerificationSolution(unsigned short val_nDim, unsigned short val_nVar, unsigned short val_iMesh,
39 CConfig* config) {
40 /*--- Store the kind of solver ---*/
41
42 Kind_Solver = config->GetKind_Solver();
43
44 /*--- Store the rank and size for the calculation. ---*/
45
46 size = SU2_MPI::GetSize();
47 rank = SU2_MPI::GetRank();
48
49 /*--- Store the dimension and number of variables. ---*/
50
51 nDim = val_nDim;
52 nVar = val_nVar;
53
54 /*--- Allocate space for global error metrics for verification. ---*/
55 Error_RMS = new su2double[nVar];
56 Error_Max = new su2double[nVar];
57
58 Error_Point_Max = new unsigned long[nVar];
59 for (unsigned short iVar = 0; iVar < nVar; iVar++) Error_Point_Max[iVar] = 0;
60
61 Error_Point_Max_Coord = new su2double*[nVar];
62 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
63 Error_Point_Max_Coord[iVar] = new su2double[nDim];
64 for (unsigned short iDim = 0; iDim < nDim; iDim++) Error_Point_Max_Coord[iVar][iDim] = 0.0;
65 }
66}

109void CVerificationSolution::GetLocalError(const su2double* val_coords, const su2double val_t,
110 const su2double* val_solution, su2double* val_error) const {
111 /*--- Get the value of the verification solution first.
112 Use val_error to store this solution. ---*/
113
114 GetSolution(val_coords, val_t, val_error);
115
116 /*--- Compute the local error as the difference between the current
117 numerical solution and the verification solution. ---*/
118
119 for (unsigned short iVar = 0; iVar < nVar; ++iVar) val_error[iVar] = val_solution[iVar] - val_error[iVar];
120}

122void CVerificationSolution::SetVerificationError(unsigned long nDOFsGlobal, CConfig* config) {
123 /* Disable the reduce for the error to avoid overhead if requested. */
124 if (config->GetComm_Level() == COMM_FULL) {
125 /*--- Get the number of ranks and the MPI communicator. ---*/
126 int size = SU2_MPI::GetSize();
127 SU2_MPI::Comm comm = SU2_MPI::GetComm();
128
129 /*--- The local L2 norms must be added to obtain the global value. ---*/
130 vector<su2double> rbufError(nVar, 0.0);
131 SU2_MPI::Allreduce(Error_RMS, rbufError.data(), nVar, MPI_DOUBLE, MPI_SUM, comm);
132
133 for (unsigned short iVar = 0; iVar < nVar; ++iVar) {
134 SetError_RMS(iVar, max(EPS * EPS, sqrt(rbufError[iVar] / nDOFsGlobal)));
135 }
136
137 /*--- The global maximum norms must be obtained. ---*/
138 rbufError.resize(nVar * size, 0.0);
139 SU2_MPI::Allgather(Error_Max, nVar, MPI_DOUBLE, rbufError.data(), nVar, MPI_DOUBLE, comm);
140
141 vector<unsigned long> rbufPoint(nVar * size);
142 SU2_MPI::Allgather(Error_Point_Max, nVar, MPI_UNSIGNED_LONG, rbufPoint.data(), nVar, MPI_UNSIGNED_LONG, comm);
143
144 vector<su2double> sbufCoor(nDim * nVar, 0.0);
145 for (unsigned short iVar = 0; iVar < nVar; ++iVar) {
146 for (unsigned short iDim = 0; iDim < nDim; ++iDim)
147 sbufCoor[iVar * nDim + iDim] = Error_Point_Max_Coord[iVar][iDim];
148 }
149
150 vector<su2double> rbufCoor(nDim * nVar * size, 0.0);
151 SU2_MPI::Allgather(sbufCoor.data(), nVar * nDim, MPI_DOUBLE, rbufCoor.data(), nVar * nDim, MPI_DOUBLE, comm);
152
153 for (unsigned short iVar = 0; iVar < nVar; ++iVar) {
154 for (int proc = 0; proc < size; ++proc)
155 AddError_Max(iVar, rbufError[proc * nVar + iVar], rbufPoint[proc * nVar + iVar],
156 &rbufCoor[proc * nVar * nDim + iVar * nDim]);
157 }
158 }
