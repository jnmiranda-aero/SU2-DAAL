57void CTurbomachineryState::ComputeState(CFluidModel& fluidModel, const CTurbomachineryPrimitiveState& primitiveState) {
58 /*--- Assign new primitive values ---*/
59 Density = primitiveState.GetDensity();
60 Pressure = primitiveState.GetPressure();
61 std::vector<su2double> velocity = primitiveState.GetVelocity();
62 Velocity.assign(velocity.begin(), velocity.end());
63 su2double tangVel = primitiveState.GetTangVelocity();
64
65 /*--- Compute static TD quantities ---*/
66 fluidModel.SetTDState_Prho(Pressure, Density);
67 Entropy = fluidModel.GetEntropy();
68 Enthalpy = fluidModel.GetStaticEnergy() + Pressure / Density;
69 su2double soundSpeed = fluidModel.GetSoundSpeed();
70
71 /*--- Compute total TD quantities ---*/
72 TotalEnthalpy = Enthalpy + 0.5 * GetVelocityValue() * GetVelocityValue();
73 fluidModel.SetTDState_hs(TotalEnthalpy, Entropy);
74 TotalPressure = fluidModel.GetPressure();
75 TotalTemperature = fluidModel.GetTemperature();
76
77 /*--- Compute absolute kinematic quantities---*/
78 MassFlow = Density * Velocity[0] * Area;
79 AbsFlowAngle = atan(Velocity[1] / Velocity[0]);
80 Mach.assign(Velocity.begin(), Velocity.end());
81 std::for_each(Mach.begin(), Mach.end(), [&](su2double& el) { el /= soundSpeed; });
82
83 /*--- Compute relative kinematic quantities ---*/
84 su2double tangVel2 = tangVel * tangVel;
85 RelVelocity.assign(Velocity.begin(), Velocity.end());
86 RelVelocity[1] -= tangVel;
87 su2double relVel2 = GetRelVelocityValue();
88 FlowAngle = atan(RelVelocity[1] / RelVelocity[0]);
89 RelMach.assign(RelVelocity.begin(), RelVelocity.end());
90 std::for_each(RelMach.begin(), RelMach.end(), [&](su2double& el) { el /= soundSpeed; });
91
92 /*--- Compute total relative TD quantities ---*/
93 Rothalpy = Enthalpy + 0.5 * relVel2 - 0.5 * tangVel2;
94 fluidModel.SetTDState_hs(Rothalpy, Entropy);
95 TotalRelPressure = fluidModel.GetPressure();
96
97 /*--- Compute isentropic quantities ---*/
98 fluidModel.SetTDState_Ps(Pressure, Entropy);
99}

113void CTurbineBladePerformance::ComputePerformance(const CTurbomachineryCombinedPrimitiveStates& primitives) {
114 /*--- Compute Inlet and Outlet state ---*/
115 InletState.ComputeState(FluidModel, primitives.GetInletPrimitiveState());
116 OutletState.ComputeState(FluidModel, primitives.GetOutletPrimitiveState());
117
118 /*--- Compute isentropic Outflow quantities ---*/
119 FluidModel.SetTDState_Ps(OutletState.GetPressure(), InletState.GetEntropy());
120 su2double enthalpyOutIs = FluidModel.GetStaticEnergy() + OutletState.GetPressure() / FluidModel.GetDensity();
121 su2double tangVel = primitives.GetOutletPrimitiveState().GetTangVelocity();
122 su2double relVelOutIs2 = 2 * (OutletState.GetRothalpy() - enthalpyOutIs) + tangVel * tangVel;
123
124 /*--- Compute performance ---*/
125 EntropyGen = (OutletState.GetEntropy() - InletState.GetEntropy());
126 EulerianWork = InletState.GetTotalEnthalpy() - OutletState.GetTotalEnthalpy();
127 TotalPressureLoss = (InletState.GetTotalRelPressure() - OutletState.GetTotalRelPressure()) /
128 (OutletState.GetTotalRelPressure() - OutletState.GetPressure());
129 KineticEnergyLoss = 2 * (OutletState.GetEnthalpy() - enthalpyOutIs) / relVelOutIs2;
130}

163CTurboOutput::CTurboOutput(CConfig** config, const CGeometry& geometry, CFluidModel& fluidModel) {
164 unsigned short nBladesRow = config[ZONE_0]->GetnMarker_Turbomachinery();
165 unsigned short nDim = geometry.GetnDim();
166
167 for (unsigned short iBladeRow = 0; iBladeRow < nBladesRow; iBladeRow++) {
168 vector<shared_ptr<CTurbomachineryBladePerformance>> bladeSpanPerformances;
169 unsigned short nSpan = config[iBladeRow]->GetnSpanWiseSections();
170 for (unsigned short iSpan = 0; iSpan < nSpan + 1; iSpan++) {
171 su2double areaIn = geometry.GetSpanAreaIn(iBladeRow, iSpan);
172 su2double areaOut = geometry.GetSpanAreaOut(iBladeRow, iSpan);
173 su2double radiusIn = geometry.GetTurboRadiusIn(iBladeRow, iSpan);
174 su2double radiusOut = geometry.GetTurboRadiusOut(iBladeRow, iSpan);
175
176 /* Switch between the Turbomachinery Performance Kind */
177 switch (config[ZONE_0]->GetKind_TurboPerf(iBladeRow)) {
178 case TURBO_PERF_KIND::TURBINE:
179 bladeSpanPerformances.push_back(
180 make_shared<CTurbineBladePerformance>(fluidModel, nDim, areaIn, radiusIn, areaOut, radiusOut));
181 break;
182
183 case TURBO_PERF_KIND::COMPRESSOR:
184 bladeSpanPerformances.push_back(
185 make_shared<CCompressorBladePerformance>(fluidModel, nDim, areaIn, radiusIn, areaOut, radiusOut));
186 break;
187
188 case TURBO_PERF_KIND::PROPELLOR:
189 bladeSpanPerformances.push_back(
190 make_shared<CPropellorBladePerformance>(fluidModel, nDim, areaIn, radiusIn, areaOut, radiusOut));
191 break;
192
193 default:
194 bladeSpanPerformances.push_back(
195 make_shared<CTurbineBladePerformance>(fluidModel, nDim, areaIn, radiusIn, areaOut, radiusOut));
196 break;
197 }
198 }
199 BladesPerformances.push_back(bladeSpanPerformances);
200 }

241void CTurbomachineryStagePerformance::ComputeTurbineStagePerformance(const CTurbomachineryState& InState,
242 const CTurbomachineryState& OutState) {
243 /*--- Compute isentropic Outflow quantities ---*/
244 fluidModel.SetTDState_Ps(OutState.GetPressure(), InState.GetEntropy());
245 su2double enthalpyOutIs = fluidModel.GetStaticEnergy() + OutState.GetPressure() / fluidModel.GetDensity();
246 su2double totEnthalpyOutIs = enthalpyOutIs + 0.5 * OutState.GetVelocityValue() * OutState.GetVelocityValue();
247
248 /*--- Compute turbine stage performance ---*/
249 NormEntropyGen = (OutState.GetEntropy() - InState.GetEntropy()) / InState.GetEntropy();
250 EulerianWork = InState.GetTotalEnthalpy() - OutState.GetTotalEnthalpy();
251 TotalStaticEfficiency = EulerianWork / (InState.GetTotalEnthalpy() - enthalpyOutIs);
252 TotalTotalEfficiency = EulerianWork / (InState.GetTotalEnthalpy() - totEnthalpyOutIs);
253 TotalStaticPressureRatio = InState.GetTotalPressure() / OutState.GetPressure();
254 TotalTotalPressureRatio = InState.GetTotalPressure() / OutState.GetTotalPressure();
255}

257void CTurbomachineryStagePerformance::ComputeCompressorStagePerformance(const CTurbomachineryState& InState,
258 const CTurbomachineryState& OutState) {
259 /*--- Compute isentropic Outflow quantities ---*/
260 fluidModel.SetTDState_Ps(OutState.GetPressure(), InState.GetEntropy());
261 su2double enthalpyOutIs = fluidModel.GetStaticEnergy() + OutState.GetPressure() / fluidModel.GetDensity();
262 su2double totEnthalpyOutIs = enthalpyOutIs + 0.5 * OutState.GetVelocityValue() * OutState.GetVelocityValue();
263
264 /*--- Compute compressor stage performance ---*/
265 NormEntropyGen = (OutState.GetEntropy() - InState.GetEntropy()) / InState.GetEntropy();
266 EulerianWork = OutState.GetTotalEnthalpy() - InState.GetTotalEnthalpy();
267 TotalStaticEfficiency = (enthalpyOutIs - InState.GetTotalEnthalpy()) / EulerianWork;
268 TotalTotalEfficiency = (totEnthalpyOutIs - InState.GetTotalEnthalpy()) / EulerianWork;
269 TotalStaticPressureRatio = OutState.GetPressure() / InState.GetTotalPressure();
270 TotalTotalPressureRatio = OutState.GetTotalPressure() / InState.GetTotalPressure();
271}
