39 /*!
40  * \brief Compute the gradient of a field using the Green-Gauss theorem.
41  * \ingroup FvmAlgos
42  * \note Template nDim to allow efficient unrolling of inner loops.
43  * \note Gradients can be computed only for a contiguous range of variables, defined
44  * by [varBegin, varEnd[ (e.g. 0,1 computes the gradient of the 1st variable).
45  * This can be used, for example, to compute only velocity gradients.
46  * \note The function uses an optional solver object to perform communications, if
47  * none (nullptr) is provided the function does not fail (the objective of
48  * this is to improve test-ability).
49  * \param[in] solver - Optional, solver associated with the field (used only for MPI).
50  * \param[in] kindMpiComm - Type of MPI communication required.
51  * \param[in] kindPeriodicComm - Type of periodic communication required.
52  * \param[in] geometry - Geometric grid properties.
53  * \param[in] config - Configuration of the problem, used to identify types of boundaries.
54  * \param[in] field - Generic object implementing operator (iPoint, iVar).
55  * \param[in] varBegin - Index of first variable for which to compute the gradient.
56  * \param[in] varEnd - Index of last variable for which to compute the gradient.
57  * \param[in] idxVel - Index of velocity, or -1 if no velocity present.
58  * \param[out] gradient - Generic object implementing operator (iPoint, iVar, iDim).
59  */
60 template <size_t nDim, class FieldType, class GradientType>
61 void computeGradientsGreenGauss(CSolver* solver, MPI_QUANTITIES kindMpiComm, PERIODIC_QUANTITIES kindPeriodicComm,
62  CGeometry& geometry, const CConfig& config, const FieldType& field,
63  const size_t varBegin, const size_t varEnd, const int idxVel, GradientType& gradient) {
64  const size_t nPointDomain = geometry.GetnPointDomain();
65 
66 #ifdef HAVE_OMP
67  constexpr size_t OMP_MAX_CHUNK = 512;
68 
69  const auto chunkSize = computeStaticChunkSize(nPointDomain, omp_get_max_threads(), OMP_MAX_CHUNK);
70 #endif
71 
72  static constexpr size_t MAXNVAR = 20;
73 
74  /*--- For each (non-halo) volume integrate over its faces (edges). ---*/
75 
76  SU2_OMP_FOR_DYN(chunkSize)
77  for (size_t iPoint = 0; iPoint < nPointDomain; ++iPoint) {
78  auto nodes = geometry.nodes;
79 
80  /*--- Cannot preaccumulate if hybrid parallel due to shared reading. ---*/
81  if (omp_get_num_threads() == 1) AD::StartPreacc();
82  AD::SetPreaccIn(nodes->GetVolume(iPoint));
83  AD::SetPreaccIn(nodes->GetPeriodicVolume(iPoint));
84 
85  for (size_t iVar = varBegin; iVar < varEnd; ++iVar) AD::SetPreaccIn(field(iPoint, iVar));
86 
87  /*--- Clear the gradient. --*/
88 
89  for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
90  for (size_t iDim = 0; iDim < nDim; ++iDim) gradient(iPoint, iVar, iDim) = 0.0;
91 
92  /*--- Handle averaging and division by volume in one constant. ---*/
93 
94  su2double halfOnVol = 0.5 / (nodes->GetVolume(iPoint) + nodes->GetPeriodicVolume(iPoint));
95 
96  /*--- Add a contribution due to each neighbor. ---*/
97 
98  for (size_t iNeigh = 0; iNeigh < nodes->GetnPoint(iPoint); ++iNeigh) {
99  size_t iEdge = nodes->GetEdge(iPoint, iNeigh);
100  size_t jPoint = nodes->GetPoint(iPoint, iNeigh);
101 
102  /*--- Determine if edge points inwards or outwards of iPoint.
103  * If inwards we need to flip the area vector. ---*/
104 
105  su2double dir = (iPoint < jPoint) ? 1.0 : -1.0;
106  su2double weight = dir * halfOnVol;
107 
108  const auto area = geometry.edges->GetNormal(iEdge);
109  AD::SetPreaccIn(area, nDim);
110 
111  for (size_t iVar = varBegin; iVar < varEnd; ++iVar) {
112  AD::SetPreaccIn(field(jPoint, iVar));
113  su2double flux = weight * (field(iPoint, iVar) + field(jPoint, iVar));
114 
115  for (size_t iDim = 0; iDim < nDim; ++iDim) gradient(iPoint, iVar, iDim) += flux * area[iDim];
116  }
117  }
118 
119  for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
120  for (size_t iDim = 0; iDim < nDim; ++iDim) AD::SetPreaccOut(gradient(iPoint, iVar, iDim));
121 
122  AD::EndPreacc();
123  }
124  END_SU2_OMP_FOR
125 
126  su2double flux[MAXNVAR] = {0.0};
127 
128  /*--- Add edges of markers that contribute to the gradients ---*/
129  for (size_t iMarker = 0; iMarker < geometry.GetnMarker(); ++iMarker) {
130  if ((config.GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
131  (config.GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
132  (config.GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
133  /*--- Work is shared in inner loop as two markers
134  * may try to update the same point. ---*/
135 
136  SU2_OMP_FOR_STAT(32)
137  for (size_t iVertex = 0; iVertex < geometry.GetnVertex(iMarker); ++iVertex) {
138  size_t iPoint = geometry.vertex[iMarker][iVertex]->GetNode();
139  auto nodes = geometry.nodes;
140 
141  /*--- Halo points do not need to be considered. ---*/
142 
143  if (!nodes->GetDomain(iPoint)) continue;
144 
145  su2double volume = nodes->GetVolume(iPoint) + nodes->GetPeriodicVolume(iPoint);
146  const auto area = geometry.vertex[iMarker][iVertex]->GetNormal();
147 
148  for (size_t iVar = varBegin; iVar < varEnd; iVar++)
149  flux[iVar] = field(iPoint,iVar) / volume;
150 
151  for (size_t iVar = varBegin; iVar < varEnd; iVar++) {
152  for (size_t iDim = 0; iDim < nDim; iDim++) {
153  gradient(iPoint, iVar, iDim) -= flux[iVar] * area[iDim];
154  }
155  } // loop over variables
156  } // vertices
157  END_SU2_OMP_FOR
158  } //found right marker
159  } // iMarkers
160 
161 
162  /*--- Compute the corrections for symmetry planes and Euler walls. ---*/
163 
164  correctGradientsSymmetry<nDim>(geometry, config, varBegin, varEnd, idxVel, gradient);
165 
166  /*--- If no solver was provided we do not communicate ---*/
167 
168  if (solver == nullptr) return;
169 
170  /*--- Account for periodic contributions. ---*/
171 
172  for (size_t iPeriodic = 1; iPeriodic <= config.GetnMarker_Periodic() / 2; ++iPeriodic) {
173  solver->InitiatePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm);
174  solver->CompletePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm);
175  }
176 
177  /*--- Obtain the gradients at halo points from the MPI ranks that own them. ---*/
178 
179  solver->InitiateComms(&geometry, &config, kindMpiComm);
180  solver->CompleteComms(&geometry, &config, kindMpiComm);
181 }
182 } // namespace detail
