34/*!
35 * \brief Compute the gradient of a field using the Green-Gauss theorem.
36 * \ingroup FvmAlgos
37 * \note Template nDim to allow efficient unrolling of inner loops.
38 * \note Gradients can be computed only for a contiguous range of variables, defined
39 * by [varBegin, varEnd[ (e.g. 0,1 computes the gradient of the 1st variable).
40 * This can be used, for example, to compute only velocity gradients.
41 * \note The function uses an optional solver object to perform communications, if
42 * none (nullptr) is provided the function does not fail (the objective of
43 * this is to improve test-ability).
44 * \param[in] solver - Optional, solver associated with the field (used only for MPI).
45 * \param[in] kindMpiComm - Type of MPI communication required.
46 * \param[in] kindPeriodicComm - Type of periodic communication required.
47 * \param[in] geometry - Geometric grid properties.
48 * \param[in] config - Configuration of the problem, used to identify types of boundaries.
49 * \param[in] field - Generic object implementing operator (iPoint, iVar).
50 * \param[in] varBegin - Index of first variable for which to compute the gradient.
51 * \param[in] varEnd - Index of last variable for which to compute the gradient.
52 * \param[out] gradient - Generic object implementing operator (iPoint, iVar, iDim).
53 */
54template<size_t nDim, class FieldType, class GradientType>
55void computeGradientsGreenGauss(CSolver* solver,
56 MPI_QUANTITIES kindMpiComm,
57 PERIODIC_QUANTITIES kindPeriodicComm,
58 CGeometry& geometry,
59 const CConfig& config,
60 const FieldType& field,
61 size_t varBegin,
62 size_t varEnd,
63 GradientType& gradient)
64{
65 const size_t nPointDomain = geometry.GetnPointDomain();
66
67#ifdef HAVE_OMP
68 constexpr size_t OMP_MAX_CHUNK = 512;
69
70 const auto chunkSize = computeStaticChunkSize(nPointDomain, omp_get_max_threads(), OMP_MAX_CHUNK);
71#endif
72
73 /*--- For each (non-halo) volume integrate over its faces (edges). ---*/
74
75 SU2_OMP_FOR_DYN(chunkSize)
76 for (size_t iPoint = 0; iPoint < nPointDomain; ++iPoint)
77 {
78 auto nodes = geometry.nodes;
79
80 /*--- Cannot preaccumulate if hybrid parallel due to shared reading. ---*/
81 if (omp_get_num_threads() == 1) AD::StartPreacc();
82 AD::SetPreaccIn(nodes->GetVolume(iPoint));
83 AD::SetPreaccIn(nodes->GetPeriodicVolume(iPoint));
84
85 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
86 AD::SetPreaccIn(field(iPoint,iVar));
87
88 /*--- Clear the gradient. --*/
89
90 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
91 for (size_t iDim = 0; iDim < nDim; ++iDim)
92 gradient(iPoint, iVar, iDim) = 0.0;
93
94 /*--- Handle averaging and division by volume in one constant. ---*/
95
96 su2double halfOnVol = 0.5 / (nodes->GetVolume(iPoint)+nodes->GetPeriodicVolume(iPoint));
97
98 /*--- Add a contribution due to each neighbor. ---*/
99
100 for (size_t iNeigh = 0; iNeigh < nodes->GetnPoint(iPoint); ++iNeigh)
101 {
102 size_t iEdge = nodes->GetEdge(iPoint,iNeigh);
103 size_t jPoint = nodes->GetPoint(iPoint,iNeigh);
104
105 /*--- Determine if edge points inwards or outwards of iPoint.
106 * If inwards we need to flip the area vector. ---*/
107
108 su2double dir = (iPoint < jPoint)? 1.0 : -1.0;
109 su2double weight = dir * halfOnVol;
110
111 const auto area = geometry.edges->GetNormal(iEdge);
112 AD::SetPreaccIn(area, nDim);
113
114 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
115 {
116 AD::SetPreaccIn(field(jPoint,iVar));
117
118 su2double flux = weight * (field(iPoint,iVar) + field(jPoint,iVar));
119
120 for (size_t iDim = 0; iDim < nDim; ++iDim)
121 gradient(iPoint, iVar, iDim) += flux * area[iDim];
122 }
123
124 }
125
126 for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
127 for (size_t iDim = 0; iDim < nDim; ++iDim)
128 AD::SetPreaccOut(gradient(iPoint,iVar,iDim));
129
130 AD::EndPreacc();
131 }
132 END_SU2_OMP_FOR
133
134 /*--- Add boundary fluxes. ---*/
135
136 for (size_t iMarker = 0; iMarker < geometry.GetnMarker(); ++iMarker)
137 {
138 if ((config.GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
139 (config.GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
140 (config.GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY))
141 {
142 /*--- Work is shared in inner loop as two markers
143 * may try to update the same point. ---*/
144
145 SU2_OMP_FOR_STAT(32)
146 for (size_t iVertex = 0; iVertex < geometry.GetnVertex(iMarker); ++iVertex)
147 {
148 size_t iPoint = geometry.vertex[iMarker][iVertex]->GetNode();
149 auto nodes = geometry.nodes;
150
151 /*--- Halo points do not need to be considered. ---*/
152
153 if (!nodes->GetDomain(iPoint)) continue;
154
155 su2double volume = nodes->GetVolume(iPoint) + nodes->GetPeriodicVolume(iPoint);
156
157 const auto area = geometry.vertex[iMarker][iVertex]->GetNormal();
158
159 for (size_t iVar = varBegin; iVar < varEnd; iVar++)
160 {
161 su2double flux = field(iPoint,iVar) / volume;
162
163 for (size_t iDim = 0; iDim < nDim; iDim++)
164 gradient(iPoint, iVar, iDim) -= flux * area[iDim];
165 }
166 }
167 END_SU2_OMP_FOR
168 }
169 }
170
171 /*--- If no solver was provided we do not communicate ---*/
172
173 if (solver == nullptr) return;
174
175 /*--- Account for periodic contributions. ---*/
176
177 for (size_t iPeriodic = 1; iPeriodic <= config.GetnMarker_Periodic()/2; ++iPeriodic)
178 {
179 solver->InitiatePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm);
180 solver->CompletePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm);
181 }
182
183 /*--- Obtain the gradients at halo points from the MPI ranks that own them. ---*/
184
185 solver->InitiateComms(&geometry, &config, kindMpiComm);
186 solver->CompleteComms(&geometry, &config, kindMpiComm);
187
188}
189} // end namespace

192 * \brief Instantiations for 2D and 3D.
193 * \ingroup FvmAlgos
194 */
195template<class FieldType, class GradientType>
196void computeGradientsGreenGauss(CSolver* solver,
197 MPI_QUANTITIES kindMpiComm,
198 PERIODIC_QUANTITIES kindPeriodicComm,
199 CGeometry& geometry,
200 const CConfig& config,
201 const FieldType& field,
202 size_t varBegin,
203 size_t varEnd,
204 GradientType& gradient) {
205 switch (geometry.GetnDim()) {
206 case 2:
207 detail::computeGradientsGreenGauss<2>(solver, kindMpiComm, kindPeriodicComm, geometry,
208 config, field, varBegin, varEnd, gradient);
209 break;
210 case 3:
211 detail::computeGradientsGreenGauss<3>(solver, kindMpiComm, kindPeriodicComm, geometry,
212 config, field, varBegin, varEnd, gradient);
213 break;
214 default:
215 SU2_MPI::Error("Too many dimensions to compute gradients.", CURRENT_FUNCTION);
216 break;
217 }
