91  /*!
92  * \brief Run the external solver for the task it is currently setup to execute.
93  */
94  void Run() {
95  dpastix(&state, SU2_MPI::GetComm(), nCols, colptr.data(), rowidx.data(), values.data(), loc2glb.data(), perm.data(),
96  NULL, workvec.data(), 1, iparm, dparm);
97  }

99  /*!
100  * \brief Run the "clean" task, releases all memory, leaves object in unusable state.
101  */
102  void Clean() {
103  using namespace PaStiX;
104  if (isfactorized) {
105  iparm[IPARM_VERBOSE] = (verb > 0) ? API_VERBOSE_NO : API_VERBOSE_NOT;
106  iparm[IPARM_START_TASK] = API_TASK_CLEAN;
107  iparm[IPARM_END_TASK] = API_TASK_CLEAN;
108  Run();
109  isfactorized = false;
110  }
111  }

182  /*!
183  * \brief Runs the "solve" task for any rhs/sol with operator []
184  * \param[in] rhs - Right hand side of the linear system.
185  * \param[out] sol - Solution of the system.
186  */
187  template <class T>
188  void Solve(const T& rhs, T& sol) {
189  using namespace PaStiX;
190 
191  if (!isfactorized) SU2_MPI::Error("The factorization has not been computed yet.", CURRENT_FUNCTION);
192 
193  unsigned long i;
194 
195  for (i = 0; i < matrix.size_rhs(); ++i) workvec[i] = rhs[i];
196 
197  iparm[IPARM_VERBOSE] = API_VERBOSE_NOT;
198  iparm[IPARM_START_TASK] = API_TASK_SOLVE;
199  iparm[IPARM_END_TASK] = API_TASK_SOLVE;
200  Run();
201 
202  for (i = 0; i < matrix.size_rhs(); ++i) sol[i] = workvec[i];
203  }
