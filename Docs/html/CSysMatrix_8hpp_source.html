394  /*!
395  * \brief Get a pointer to the start of block "ij"
396  * \param[in] block_i - Row index.
397  * \param[in] block_j - Column index.
398  * \return Pointer to location in memory where the block starts.
399  */
400  FORCEINLINE const ScalarType* GetBlock(unsigned long block_i, unsigned long block_j) const {
401  /*--- The position of the diagonal block is known which allows halving the search space. ---*/
402  const auto end = (block_j < block_i) ? dia_ptr[block_i] : row_ptr[block_i + 1];
403  for (auto index = (block_j < block_i) ? row_ptr[block_i] : dia_ptr[block_i]; index < end; ++index)
404  if (col_ind[index] == block_j) return &matrix[index * nVar * nEqn];
405  return nullptr;
406  }

431  /*!
432  * \brief Set the value of a block (in flat format) in the sparse matrix with scaling.
433  * \note If the template param Overwrite is false we add to the block (bij += alpha*b).
434  * \param[in] block_i - Row index.
435  * \param[in] block_j - Column index.
436  * \param[in] val_block - Block to set to A(i, j).
437  * \param[in] alpha - Scale factor.
438  */
439  template <class OtherType, bool Overwrite = true, su2enable_if<!is_pointer<OtherType>::value> = 0>
440  inline void SetBlock(unsigned long block_i, unsigned long block_j, const OtherType* val_block,
441  OtherType alpha = 1.0) {
442  auto mat_ij = GetBlock(block_i, block_j);
443  if (!mat_ij) return;
444  SU2_OMP_SIMD
445  for (auto iVar = 0ul; iVar < nVar * nEqn; ++iVar) {
446  mat_ij[iVar] = (Overwrite ? ScalarType(0) : mat_ij[iVar]) + PassiveAssign(alpha * val_block[iVar]);
447  }
448  }

463  /*!
464  * \brief Set the value of a scaled block in the sparse matrix.
465  * \note If the template param Overwrite is false we add to the block (bij += alpha*b).
466  * \param[in] block_i - Row index.
467  * \param[in] block_j - Column index.
468  * \param[in] val_block - Block to set to A(i, j).
469  * \param[in] alpha - Scale factor.
470  */
471  template <class OtherType, bool Overwrite = true>
472  inline void SetBlock(unsigned long block_i, unsigned long block_j, const OtherType* const* val_block,
473  OtherType alpha = 1.0) {
474  auto mat_ij = GetBlock(block_i, block_j);
475  if (!mat_ij) return;
476  for (auto iVar = 0ul; iVar < nVar; ++iVar) {
477  for (auto jVar = 0ul; jVar < nEqn; ++jVar) {
478  *mat_ij = (Overwrite ? ScalarType(0) : *mat_ij) + PassiveAssign(alpha * val_block[iVar][jVar]);
479  ++mat_ij;
480  }
481  }
482  }

508  /*!
509  * \brief Set the value of a scaled block in the sparse matrix.
510  * \note This is an templated overload for C2Dcontainer specialization su2matrix.
511  * It assumes that MatrixType supports a member type Scalar and access operator[][].
512  * If the template param Overwrite is false we add to the block (bij += alpha*b).
513  * \param[in] block_i - Row index.
514  * \param[in] block_j - Column index.
515  * \param[in] val_block - Block to set to A(i, j).
516  * \param[in] alpha - Scale factor.
517  */
518  template <class MatrixType, bool Overwrite = true>
519  inline void SetBlock(unsigned long block_i, unsigned long block_j, MatrixType& val_block,
520  typename MatrixType::Scalar alpha = 1.0) {
521  auto mat_ij = GetBlock(block_i, block_j);
522  if (!mat_ij) return;
523  for (auto iVar = 0ul; iVar < nVar; ++iVar) {

584  /*!
585  * \brief SIMD version, does the update for multiple edges and points.
586  * \note Nothing is updated if the mask is 0.
587  */
588  template <class MatTypeSIMD, size_t N, class I, class F = ScalarType>
589  FORCEINLINE void UpdateBlocks(simd::Array<I, N> iEdge, simd::Array<I, N> iPoint, simd::Array<I, N> jPoint,
590  const MatTypeSIMD& block_i, const MatTypeSIMD& block_j, simd::Array<F, N> mask = 1) {
591  static_assert(MatTypeSIMD::StaticSize, "This method requires static size blocks.");
592  static_assert(MatTypeSIMD::IsRowMajor, "Block storage is not compatible with matrix.");
593  constexpr size_t blkSz = MatTypeSIMD::StaticSize;
594  assert(blkSz == nVar * nEqn);
595 
596  /*--- "Transpose" the blocks, scale, and possibly convert types,
597  * giving the compiler the chance to vectorize all of these. ---*/
598  ScalarType blk_i[N][blkSz], blk_j[N][blkSz];
599 
600  for (size_t i = 0; i < blkSz; ++i) {
601  SU2_OMP_SIMD_IF_NOT_AD
602  for (size_t k = 0; k < N; ++k) {
603  blk_i[k][i] = PassiveAssign(-mask[k] * block_i.data()[i][k]);
604  blk_j[k][i] = PassiveAssign(mask[k] * block_j.data()[i][k]);
605  }
606  }
607 
608  /*--- Update one by one skipping if mask is 0. ---*/
609  for (size_t k = 0; k < N; ++k) {
610  if (mask[k] == 0) continue;
611 
612  /*--- Fetch the blocks. ---*/
613  auto bii = &matrix[dia_ptr[iPoint[k]] * blkSz];
614  auto bjj = &matrix[dia_ptr[jPoint[k]] * blkSz];
615  auto bij = &matrix[edge_ptr(iEdge[k], 0) * blkSz];
616  auto bji = &matrix[edge_ptr(iEdge[k], 1) * blkSz];
617 
618  /*--- Update, block i was negated during transpose in the
619  * hope the assignments below become non-temporal stores. ---*/
620  SU2_OMP_SIMD
621  for (size_t i = 0; i < blkSz; ++i) {
622  bii[i] -= blk_i[k][i];
623  bjj[i] -= blk_j[k][i];
624  bij[i] = blk_j[k][i];
625  bji[i] = blk_i[k][i];
626  }
627  }
628  }
