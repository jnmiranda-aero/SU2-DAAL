33CNEMOCompOutput::CNEMOCompOutput(const CConfig *config, unsigned short nDim) : CFlowOutput(config, nDim, false) {
34
35 turb_model = config->GetKind_Turb_Model();
36 nSpecies = config->GetnSpecies();
37
38 /*--- Set the default history fields if nothing is set in the config file ---*/
39
40 if (nRequestedHistoryFields == 0){
41 RequestCommonHistory(config->GetTime_Domain());
42 nRequestedHistoryFields = requestedHistoryFields.size();
43 }
44 if (nRequestedScreenFields == 0){
45 if (config->GetTime_Domain()) requestedScreenFields.emplace_back("TIME_ITER");
46 if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
47 requestedScreenFields.emplace_back("INNER_ITER");
48 for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
49 requestedScreenFields.emplace_back("RMS_DENSITY_" + std::to_string(iSpecies));
50 requestedScreenFields.emplace_back("RMS_MOMENTUM-X");
51 requestedScreenFields.emplace_back("RMS_MOMENTUM-Y");
52 requestedScreenFields.emplace_back("RMS_ENERGY");
53 requestedScreenFields.emplace_back("RMS_ENERGY_VE");
54 nRequestedScreenFields = requestedScreenFields.size();
55 }
56 if (nRequestedVolumeFields == 0){
57 requestedVolumeFields.emplace_back("COORDINATES");
58 requestedVolumeFields.emplace_back("SOLUTION");
59 requestedVolumeFields.emplace_back("PRIMITIVE");
60 requestedVolumeFields.emplace_back("AUXILIARY");
61 nRequestedVolumeFields = requestedVolumeFields.size();
62 }
63
64 if (gridMovement) {
65 auto notFound = requestedVolumeFields.end();
66 if (find(requestedVolumeFields.begin(), notFound, string("GRID_VELOCITY")) == notFound) {
67 requestedVolumeFields.emplace_back("GRID_VELOCITY");
68 nRequestedVolumeFields ++;
69 }
70 }
71
72 stringstream ss;
73 ss << "Zone " << config->GetiZone() << " (Comp. Fluid)";
74 multiZoneHeaderString = ss.str();
75
76 /*--- Set the volume filename --- */
77
78 volumeFilename = config->GetVolume_FileName();
79
80 /*--- Set the surface filename --- */
81
82 surfaceFilename = config->GetSurfCoeff_FileName();
83
84 /*--- Set the restart filename --- */
85
86 restartFilename = config->GetRestart_FileName();
87
88 /*--- Set the default convergence field --- */
89
90 if (convFields.empty() ) convFields.emplace_back("RMS_DENSITY_0");
91
92 if (config->GetFixed_CL_Mode()) {
93 bool found = false;
94 for (unsigned short iField = 0; iField < convFields.size(); iField++)
95 if (convFields[iField] == "LIFT") found = true;
96 if (!found) {
97 if (rank == MASTER_NODE)
98 cout<<" Fixed CL: Adding LIFT as Convergence Field to ensure convergence to target CL"<<endl;
99 convFields.emplace_back("LIFT");
100 newFunc.resize(convFields.size());
101 oldFunc.resize(convFields.size());
102 cauchySerie.resize(convFields.size(), vector<su2double>(nCauchy_Elems, 0.0));
103 }
104 }

306void CNEMOCompOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
307
308 const auto* Node_Flow = solver[FLOW_SOL]->GetNodes();
309 auto* Node_Geo = geometry->nodes;
310 const auto nSpecies = config->GetnSpecies();
311
312 LoadCoordinates(Node_Geo->GetCoord(iPoint), iPoint);
313
314 for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
315 SetVolumeOutputValue("DENSITY_" + std::to_string(iSpecies), iPoint, Node_Flow->GetSolution(iPoint, iSpecies));
316
317 SetVolumeOutputValue("MOMENTUM-X", iPoint, Node_Flow->GetSolution(iPoint, nSpecies));
318 SetVolumeOutputValue("MOMENTUM-Y", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+1));
319 if (nDim == 3){
320 SetVolumeOutputValue("MOMENTUM-Z", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+2));
321 SetVolumeOutputValue("ENERGY", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+3));
322 SetVolumeOutputValue("ENERGY_VE", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+4));
323 } else {
324 SetVolumeOutputValue("ENERGY", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+2));
325 SetVolumeOutputValue("ENERGY_VE", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+3));
326 }
327
328 for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
329 SetVolumeOutputValue("MASSFRAC_" + std::to_string(iSpecies), iPoint, Node_Flow->GetSolution(iPoint, iSpecies)/Node_Flow->GetDensity(iPoint));
330
331 if (gridMovement){
332 SetVolumeOutputValue("GRID_VELOCITY-X", iPoint, Node_Geo->GetGridVel(iPoint)[0]);
333 SetVolumeOutputValue("GRID_VELOCITY-Y", iPoint, Node_Geo->GetGridVel(iPoint)[1]);
334 if (nDim == 3)
335 SetVolumeOutputValue("GRID_VELOCITY-Z", iPoint, Node_Geo->GetGridVel(iPoint)[2]);
336 }
337
338 SetVolumeOutputValue("PRESSURE", iPoint, Node_Flow->GetPressure(iPoint));
339 SetVolumeOutputValue("TEMPERATURE_TR", iPoint, Node_Flow->GetTemperature(iPoint));
340 SetVolumeOutputValue("TEMPERATURE_VE", iPoint, Node_Flow->GetTemperature_ve(iPoint));
341 SetVolumeOutputValue("VELOCITY-X", iPoint, Node_Flow->GetVelocity(iPoint, 0));
342 SetVolumeOutputValue("VELOCITY-Y", iPoint, Node_Flow->GetVelocity(iPoint, 1));
343 if (nDim == 3)
344 SetVolumeOutputValue("VELOCITY-Z", iPoint, Node_Flow->GetVelocity(iPoint, 2));
345
346 SetVolumeOutputValue("MACH", iPoint, sqrt(Node_Flow->GetVelocity2(iPoint))/Node_Flow->GetSoundSpeed(iPoint));
347
348 const su2double factor = solver[FLOW_SOL]->GetReferenceDynamicPressure();
349 SetVolumeOutputValue("PRESSURE_COEFF", iPoint, (Node_Flow->GetPressure(iPoint) - solver[FLOW_SOL]->GetPressure_Inf())/factor);
350
351 if (config->GetViscous()){
352 SetVolumeOutputValue("LAMINAR_VISCOSITY", iPoint, Node_Flow->GetLaminarViscosity(iPoint));
353 SetVolumeOutputValue("THERMAL_CONDUCTIVITY_TR", iPoint, Node_Flow->GetThermalConductivity(iPoint));
354 SetVolumeOutputValue("THERMAL_CONDUCTIVITY_VE", iPoint, Node_Flow->GetThermalConductivity_ve(iPoint));
355 }
356
357 for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
358 SetVolumeOutputValue("RES_DENSITY_" + std::to_string(iSpecies), iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, iSpecies));
359
360 SetVolumeOutputValue("RES_MOMENTUM-X", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies));
361 SetVolumeOutputValue("RES_MOMENTUM-Y", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+1));
362 if (nDim == 3){
363 SetVolumeOutputValue("RES_MOMENTUM-Z", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+2));
364 SetVolumeOutputValue("RES_ENERGY", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+3));
365 SetVolumeOutputValue("RES_ENERGY_VE", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+4));
366 } else {
367 SetVolumeOutputValue("RES_ENERGY", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+2));
368 SetVolumeOutputValue("RES_ENERGY_VE", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+3));
369 }
370
371 if (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE && config->GetKind_SlopeLimit_Flow() != LIMITER::VAN_ALBADA_EDGE) {
372 SetVolumeOutputValue("LIMITER_DENSITY", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 0));
373 SetVolumeOutputValue("LIMITER_MOMENTUM-X", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 1));
374 SetVolumeOutputValue("LIMITER_MOMENTUM-Y", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 2));
375 if (nDim == 3){
376 SetVolumeOutputValue("LIMITER_MOMENTUM-Z", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 3));
377 SetVolumeOutputValue("LIMITER_ENERGY", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 4));
378 } else {
379 SetVolumeOutputValue("LIMITER_ENERGY", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 3));
380 }
381 }
382
383 LoadVolumeDataScalar(config, solver, geometry, iPoint);
384
385 LoadCommonFVMOutputs(config, geometry, iPoint);
386
387 if (config->GetTime_Domain()) {
388 LoadTimeAveragedData(iPoint, Node_Flow);
389 }
390}

392void CNEMOCompOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
393
394 CSolver* NEMO_solver = solver[FLOW_SOL];
395 CSolver* mesh_solver = solver[MESH_SOL];
396 unsigned short nSpecies = config->GetnSpecies();
397
398 for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
399 SetHistoryOutputValue("RMS_DENSITY_" + std::to_string(iSpecies), log10(NEMO_solver->GetRes_RMS(iSpecies)));
400
401 SetHistoryOutputValue("RMS_MOMENTUM-X", log10(NEMO_solver->GetRes_RMS(nSpecies)));
402 SetHistoryOutputValue("RMS_MOMENTUM-Y", log10(NEMO_solver->GetRes_RMS(nSpecies+1)));
403 if (nDim == 2){
404 SetHistoryOutputValue("RMS_ENERGY", log10(NEMO_solver->GetRes_RMS(nSpecies+2)));
405 SetHistoryOutputValue("RMS_ENERGY_VE", log10(NEMO_solver->GetRes_RMS(nSpecies+3)));
406 } else {
407 SetHistoryOutputValue("RMS_MOMENTUM-Z", log10(NEMO_solver->GetRes_RMS(nSpecies+2)));
408 SetHistoryOutputValue("RMS_ENERGY", log10(NEMO_solver->GetRes_RMS(nSpecies+3)));
409 SetHistoryOutputValue("RMS_ENERGY_VE", log10(NEMO_solver->GetRes_RMS(nSpecies+4)));
410 }
411 SetHistoryOutputValue("MAX_DENSITY", log10(NEMO_solver->GetRes_Max(0)));
412 SetHistoryOutputValue("MAX_MOMENTUM-X", log10(NEMO_solver->GetRes_Max(1)));
413 SetHistoryOutputValue("MAX_MOMENTUM-Y", log10(NEMO_solver->GetRes_Max(2)));
414 if (nDim == 2)
