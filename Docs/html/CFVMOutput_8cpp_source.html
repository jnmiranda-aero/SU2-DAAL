44 void CFVMOutput::AddCommonFVMOutputs(const CConfig *config) {
45 
46  // Mesh quality metrics
47  AddVolumeOutput("ORTHOGONALITY", "Orthogonality", "MESH_QUALITY", "Orthogonality Angle (deg.)");
48  AddVolumeOutput("ASPECT_RATIO", "Aspect_Ratio", "MESH_QUALITY", "CV Face Area Aspect Ratio");
49  AddVolumeOutput("VOLUME_RATIO", "Volume_Ratio", "MESH_QUALITY", "CV Sub-Volume Ratio");
50 
51  AddVolumeOutput("RANK", "rank", "MPI", "Rank of the MPI-partition");
52 
53  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); ++iMesh) {
54  stringstream key, name;
55  key << "MG_" << iMesh;
56  name << "Coarse_Grid_" << iMesh;
57  AddVolumeOutput(key.str(), name.str(), "MULTIGRID", "Coarse mesh");
58  }
59 
60  if (config->GetKind_Linear_Solver_Prec() == LINELET) {
61  AddVolumeOutput("LINELET", "Linelet", "LINELET", "Mesh lines built for the line implicit preconditioner");
62  }
63 }

65 void CFVMOutput::LoadCommonFVMOutputs(const CConfig* config, const CGeometry* geometry, unsigned long iPoint) {
66 
67  // Mesh quality metrics, computed in CPhysicalGeometry::ComputeMeshQualityStatistics.
68  if (config->GetWrt_MeshQuality()) {
69  SetVolumeOutputValue("ORTHOGONALITY", iPoint, geometry->Orthogonality[iPoint]);
70  SetVolumeOutputValue("ASPECT_RATIO", iPoint, geometry->Aspect_Ratio[iPoint]);
71  SetVolumeOutputValue("VOLUME_RATIO", iPoint, geometry->Volume_Ratio[iPoint]);
72  }
73 
74  SetVolumeOutputValue("RANK", iPoint, rank);
75 
76  if (config->GetWrt_MultiGrid()) {
77  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); ++iMesh) {
78  stringstream key;
79  key << "MG_" << iMesh;
80  SetVolumeOutputValue(key.str(), iPoint, geometry->CoarseGridColor(iPoint,iMesh-1));
81  }
82  }
83 
84  if (config->GetKind_Linear_Solver_Prec() == LINELET) {
85  SetVolumeOutputValue("LINELET", iPoint, geometry->GetLineletInfo(config).lineletColor[iPoint]);
86  }
