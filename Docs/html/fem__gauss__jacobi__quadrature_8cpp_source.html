100 //****************************************************************************80
101 //
102 // Purpose:
103 //
104 // CDGQF computes a Gauss quadrature formula with default A, B and simple knots.
105 //
106 // Discussion:
107 //
108 // This routine computes all the knots and weights of a Gauss quadrature
109 // formula with a classical weight function with default values for A and B,
110 // and only simple knots.
111 //
112 // There are no moments checks and no printing is done.
113 //
114 // Use routine EIQFS to evaluate a quadrature computed by CGQFS.
115 //
116 // Licensing:
117 //
118 // This code is distributed under the GNU LGPL license.
119 //
120 // Modified:
121 //
122 // 08 January 2010
123 //
124 // Author:
125 //
126 // Original FORTRAN77 version by Sylvan Elhay, Jaroslav Kautsky.
127 // C++ version by John Burkardt.
128 //
129 // Reference:
130 //
131 // Sylvan Elhay, Jaroslav Kautsky,
132 // Algorithm 655: IQPACK, FORTRAN Subroutines for the Weights of
133 // Interpolatory Quadrature,
134 // ACM Transactions on Mathematical Software,
135 // Volume 13, Number 4, December 1987, pages 399-415.
136 //
137 // Parameters:
138 //
139 // Input, int NT, the number of knots.
140 //
141 // Input, int KIND, the rule.
142 // 1, Legendre, (a,b) 1.0
143 // 2, Chebyshev, (a,b) ((b-x)*(x-a))^(-0.5)
144 // 3, Gegenbauer, (a,b) ((b-x)*(x-a))^alpha
145 // 4, Jacobi, (a,b) (b-x)^alpha*(x-a)^beta
146 // 5, Generalized Laguerre, (a,inf) (x-a)^alpha*exp(-b*(x-a))
147 // 6, Generalized Hermite, (-inf,inf) |x-a|^alpha*exp(-b*(x-a)^2)
148 // 7, Exponential, (a,b) |x-(a+b)/2.0|^alpha
149 // 8, Rational, (a,inf) (x-a)^alpha*(x+b)^beta
150 //
151 // Input, passivedouble ALPHA, the value of Alpha, if needed.
152 //
153 // Input, passivedouble BETA, the value of Beta, if needed.
154 //
155 // Output, passivedouble T[NT], the knots.
156 //
157 // Output, passivedouble WTS[NT], the weights.
158 //
159 {
160  passivedouble* aj;
161  passivedouble* bj;
162  passivedouble zemu;
163 
164  parchk(kind, 2 * nt, alpha, beta);
165  //
166  // Get the Jacobi matrix and zero-th moment.
167  //
168  aj = new passivedouble[nt];
169  bj = new passivedouble[nt];
170 
171  zemu = class_matrix(kind, nt, alpha, beta, aj, bj);
172  //
173  // Compute the knots and weights.
174  //
175  sgqf(nt, aj, bj, zemu, t, wts);
176 
177  delete[] aj;
178  delete[] bj;
179 }

185 //****************************************************************************80
186 //
187 // Purpose:
188 //
189 // CGQF computes knots and weights of a Gauss quadrature formula.
190 //
191 // Discussion:
192 //
193 // The user may specify the interval (A,B).
194 //
195 // Only simple knots are produced.
196 //
197 // Use routine EIQFS to evaluate this quadrature formula.
198 //
199 // Licensing:
200 //
201 // This code is distributed under the GNU LGPL license.
202 //
203 // Modified:
204 //
205 // 16 February 2010
206 //
207 // Author:
208 //
209 // Original FORTRAN77 version by Sylvan Elhay, Jaroslav Kautsky.
210 // C++ version by John Burkardt.
211 //
212 // Reference:
213 //
214 // Sylvan Elhay, Jaroslav Kautsky,
215 // Algorithm 655: IQPACK, FORTRAN Subroutines for the Weights of
216 // Interpolatory Quadrature,
217 // ACM Transactions on Mathematical Software,
218 // Volume 13, Number 4, December 1987, pages 399-415.
219 //
220 // Parameters:
221 //
222 // Input, int NT, the number of knots.
223 //
224 // Input, int KIND, the rule.
225 // 1, Legendre, (a,b) 1.0
226 // 2, Chebyshev Type 1, (a,b) ((b-x)*(x-a))^-0.5)
227 // 3, Gegenbauer, (a,b) ((b-x)*(x-a))^alpha
228 // 4, Jacobi, (a,b) (b-x)^alpha*(x-a)^beta
229 // 5, Generalized Laguerre, (a,+oo) (x-a)^alpha*exp(-b*(x-a))
230 // 6, Generalized Hermite, (-oo,+oo) |x-a|^alpha*exp(-b*(x-a)^2)
231 // 7, Exponential, (a,b) |x-(a+b)/2.0|^alpha
232 // 8, Rational, (a,+oo) (x-a)^alpha*(x+b)^beta
233 // 9, Chebyshev Type 2, (a,b) ((b-x)*(x-a))^(+0.5)
234 //
235 // Input, passivedouble ALPHA, the value of Alpha, if needed.
236 //
237 // Input, passivedouble BETA, the value of Beta, if needed.
238 //
239 // Input, passivedouble A, B, the interval endpoints, or
240 // other parameters.
241 //
242 // Output, passivedouble T[NT], the knots.
243 //
244 // Output, passivedouble WTS[NT], the weights.
245 //
246 {
247  int i;
248  int* mlt;
249  int* ndx;
250  //
251  // Compute the Gauss quadrature formula for default values of A and B.
252  //
253  cdgqf(nt, kind, alpha, beta, t, wts);
254  //
255  // Prepare to scale the quadrature formula to other weight function with
256  // valid A and B.
257  //
258  mlt = new int[nt];
259  for (i = 0; i < nt; i++) {
260  mlt[i] = 1;
261  }
262  ndx = new int[nt];
263  for (i = 0; i < nt; i++) {
264  ndx[i] = i + 1;
265  }
266  scqf(nt, t, mlt, wts, nt, ndx, wts, t, kind, alpha, beta, a, b);
267 
268  delete[] mlt;
269  delete[] ndx;
270 }

273 passivedouble CGaussJacobiQuadrature::class_matrix(int kind, int m, passivedouble alpha, passivedouble beta,
274  passivedouble aj[], passivedouble bj[])

276 //****************************************************************************80
277 //
278 // Purpose:
279 //
280 // CLASS_MATRIX computes the Jacobi matrix for a quadrature rule.
281 //
282 // Discussion:
283 //
284 // This routine computes the diagonal AJ and sub-diagonal BJ
285 // elements of the order M tridiagonal symmetric Jacobi matrix
286 // associated with the polynomials orthogonal with respect to
287 // the weight function specified by KIND.
288 //
289 // For weight functions 1-7, M elements are defined in BJ even
290 // though only M-1 are needed. For weight function 8, BJ(M) is
291 // set to zero.
292 //
293 // The zero-th moment of the weight function is returned in ZEMU.
294 //
295 // Licensing:
296 //
297 // This code is distributed under the GNU LGPL license.
298 //
299 // Modified:
300 //
301 // 08 January 2010
302 //
303 // Author:
304 //
305 // Original FORTRAN77 version by Sylvan Elhay, Jaroslav Kautsky.
306 // C++ version by John Burkardt.
307 //
308 // Reference:
309 //
310 // Sylvan Elhay, Jaroslav Kautsky,
311 // Algorithm 655: IQPACK, FORTRAN Subroutines for the Weights of
312 // Interpolatory Quadrature,
313 // ACM Transactions on Mathematical Software,
314 // Volume 13, Number 4, December 1987, pages 399-415.
315 //
316 // Parameters:
317 //
318 // Input, int KIND, the rule.
319 // 1, Legendre, (a,b) 1.0
320 // 2, Chebyshev, (a,b) ((b-x)*(x-a))^(-0.5)
321 // 3, Gegenbauer, (a,b) ((b-x)*(x-a))^alpha
322 // 4, Jacobi, (a,b) (b-x)^alpha*(x-a)^beta
323 // 5, Generalized Laguerre, (a,inf) (x-a)^alpha*exp(-b*(x-a))
324 // 6, Generalized Hermite, (-inf,inf) |x-a|^alpha*exp(-b*(x-a)^2)
325 // 7, Exponential, (a,b) |x-(a+b)/2.0|^alpha
326 // 8, Rational, (a,inf) (x-a)^alpha*(x+b)^beta
327 //
328 // Input, int M, the order of the Jacobi matrix.
329 //
330 // Input, passivedouble ALPHA, the value of Alpha, if needed.
331 //
332 // Input, passivedouble BETA, the value of Beta, if needed.
333 //
334 // Output, passivedouble AJ[M], BJ[M], the diagonal and subdiagonal
335 // of the Jacobi matrix.
336 //
337 // Output, passivedouble CLASS_MATRIX, the zero-th moment.
338 //
339 {
340  passivedouble a2b2;
341  passivedouble ab;
342  passivedouble aba;
343  passivedouble abi;
344  passivedouble abj;
345  passivedouble abti;
346  passivedouble apone;
347  int i;
348  passivedouble pi = 3.14159265358979323846264338327950;
349  passivedouble temp;
350  passivedouble temp2;
351  passivedouble zemu;
352 
353  temp = r8_epsilon();
354 
355  parchk(kind, 2 * m - 1, alpha, beta);
356 
357  temp2 = 0.5;
358 
359  if (500.0 * temp < fabs(pow(tgamma(temp2), 2) - pi)) {
360  cout << "\n";
361  cout << "CLASS_MATRIX - Fatal error!\n";
362  cout << " Gamma function does not match machine parameters.\n";
363  exit(1);
364  }
365 
366  if (kind == 1) {
367  ab = 0.0;
368 
369  zemu = 2.0 / (ab + 1.0);
370 
371  for (i = 0; i < m; i++) {
372  aj[i] = 0.0;
373  }
374 
375  for (i = 1; i <= m; i++) {
376  abi = i + ab * (i % 2);
377  abj = 2 * i + ab;
378  bj[i - 1] = sqrt(abi * abi / (abj * abj - 1.0));
379  }
380  } else if (kind == 2) {
381  zemu = pi;
382 
383  for (i = 0; i < m; i++) {
384  aj[i] = 0.0;
385  }
386 
387  bj[0] = sqrt(0.5);
388  for (i = 1; i < m; i++) {
389  bj[i] = 0.5;
390  }
391  } else if (kind == 3) {
392  ab = alpha * 2.0;
393  zemu = pow(2.0, ab + 1.0) * pow(tgamma(alpha + 1.0), 2) / tgamma(ab + 2.0);
394 
395  for (i = 0; i < m; i++) {
396  aj[i] = 0.0;
397  }
398 
399  bj[0] = sqrt(1.0 / (2.0 * alpha + 3.0));
400  for (i = 2; i <= m; i++) {
401  bj[i - 1] = sqrt(i * (i + ab) / (4.0 * pow(i + alpha, 2) - 1.0));
402  }
403  } else if (kind == 4) {
404  ab = alpha + beta;
405  abi = 2.0 + ab;
406  zemu = pow(2.0, ab + 1.0) * tgamma(alpha + 1.0) * tgamma(beta + 1.0) / tgamma(abi);
407  aj[0] = (beta - alpha) / abi;
408  bj[0] = sqrt(4.0 * (1.0 + alpha) * (1.0 + beta) / ((abi + 1.0) * abi * abi));
409  a2b2 = beta * beta - alpha * alpha;
410 
411  for (i = 2; i <= m; i++) {
412  abi = 2.0 * i + ab;
413  aj[i - 1] = a2b2 / ((abi - 2.0) * abi);
414  abi = abi * abi;
415  bj[i - 1] = sqrt(4.0 * i * (i + alpha) * (i + beta) * (i + ab) / ((abi - 1.0) * abi));
416  }
417  } else if (kind == 5) {
418  zemu = tgamma(alpha + 1.0);
419 
420  for (i = 1; i <= m; i++) {
421  aj[i - 1] = 2.0 * i - 1.0 + alpha;
422  bj[i - 1] = sqrt(i * (i + alpha));
423  }
424  } else if (kind == 6) {
425  zemu = tgamma((alpha + 1.0) / 2.0);
426 
427  for (i = 0; i < m; i++) {
428  aj[i] = 0.0;
429  }
430 
431  for (i = 1; i <= m; i++) {
432  bj[i - 1] = sqrt((i + alpha * (i % 2)) / 2.0);
433  }
434  } else if (kind == 7) {
435  ab = alpha;
436  zemu = 2.0 / (ab + 1.0);
437 
438  for (i = 0; i < m; i++) {
439  aj[i] = 0.0;
440  }
441 
442  for (i = 1; i <= m; i++) {
443  abi = i + ab * (i % 2);
444  abj = 2 * i + ab;
445  bj[i - 1] = sqrt(abi * abi / (abj * abj - 1.0));
446  }
447  } else // if ( kind == 8 )
448  {
449  ab = alpha + beta;
450  zemu = tgamma(alpha + 1.0) * tgamma(-(ab + 1.0)) / tgamma(-beta);
451  apone = alpha + 1.0;
452  aba = ab * apone;
453  aj[0] = -apone / (ab + 2.0);
454  bj[0] = -aj[0] * (beta + 1.0) / (ab + 2.0) / (ab + 3.0);
455  for (i = 2; i <= m; i++) {
456  abti = ab + 2.0 * i;
457  aj[i - 1] = aba + 2.0 * (ab + i) * (i - 1);
458  aj[i - 1] = -aj[i - 1] / abti / (abti - 2.0);
459  }
460 
461  for (i = 2; i <= m - 1; i++) {
462  abti = ab + 2.0 * i;
463  bj[i - 1] = i * (alpha + i) / (abti - 1.0) * (beta + i) / (abti * abti) * (ab + i) / (abti + 1.0);
464  }
465  bj[m - 1] = 0.0;
466  for (i = 0; i < m; i++) {
467  bj[i] = sqrt(bj[i]);
468  }
469  }
470 
471  return zemu;
472 }

475 void CGaussJacobiQuadrature::imtqlx(int n, passivedouble d[], passivedouble e[], passivedouble z[])
476 
477 //****************************************************************************80
478 //
479 // Purpose:
480 //
481 // IMTQLX diagonalizes a symmetric tridiagonal matrix.
482 //
483 // Discussion:
484 //
485 // This routine is a slightly modified version of the EISPACK routine to
486 // perform the implicit QL algorithm on a symmetric tridiagonal matrix.
487 //
488 // The authors thank the authors of EISPACK for permission to use this
489 // routine.
490 //
491 // It has been modified to produce the product Q' * Z, where Z is an input
492 // vector and Q is the orthogonal matrix diagonalizing the input matrix.
493 // The changes consist (essentialy) of applying the orthogonal transformations
494 // directly to Z as they are generated.
495 //
496 // Licensing:
497 //
498 // This code is distributed under the GNU LGPL license.
499 //
500 // Modified:
501 //
502 // 08 January 2010
503 //
504 // Author:
505 //
506 // Original FORTRAN77 version by Sylvan Elhay, Jaroslav Kautsky.
507 // C++ version by John Burkardt.
508 //
509 // Reference:
510 //
511 // Sylvan Elhay, Jaroslav Kautsky,
512 // Algorithm 655: IQPACK, FORTRAN Subroutines for the Weights of
513 // Interpolatory Quadrature,
514 // ACM Transactions on Mathematical Software,
515 // Volume 13, Number 4, December 1987, pages 399-415.
516 //
517 // Roger Martin, James Wilkinson,
518 // The Implicit QL Algorithm,
519 // Numerische Mathematik,
520 // Volume 12, Number 5, December 1968, pages 377-383.
521 //
522 // Parameters:
523 //
524 // Input, int N, the order of the matrix.
525 //
526 // Input/output, passivedouble D(N), the diagonal entries of the matrix.
527 // On output, the information in D has been overwritten.
528 //
529 // Input/output, passivedouble E(N), the subdiagonal entries of the
530 // matrix, in entries E(1) through E(N-1). On output, the information in
531 // E has been overwritten.
532 //
533 // Input/output, passivedouble Z(N). On input, a vector. On output,
534 // the value of Q' * Z, where Q is the matrix that diagonalizes the
535 // input symmetric tridiagonal matrix.
536 //
537 {
538  passivedouble b;
539  passivedouble c;
540  passivedouble f;
541  passivedouble g;
542  int i;
543  int ii;
544  int itn = 30;
545  int j;
546  int k;
547  int l;
548  int m = 0; // To avoid a compiler warning.
549  int mml;
550  passivedouble p;
551  passivedouble prec;
552  passivedouble r;
553  passivedouble s;
554 
555  prec = r8_epsilon();
556 
557  if (n == 1) {
558  return;
559  }
560 
561  e[n - 1] = 0.0;
562 
563  for (l = 1; l <= n; l++) {
564  j = 0;
565  for (;;) {
566  for (m = l; m <= n; m++) {
567  if (m == n) {
568  break;
569  }
570 
571  if (fabs(e[m - 1]) <= prec * (fabs(d[m - 1]) + fabs(d[m]))) {
572  break;
573  }
574  }
575  p = d[l - 1];
576  if (m == l) {
577  break;
578  }
579  if (itn <= j) {
580  cout << "\n";
581  cout << "IMTQLX - Fatal error!\n";
582  cout << " Iteration limit exceeded\n";
583  exit(1);
584  }
585  j = j + 1;
586  g = (d[l] - p) / (2.0 * e[l - 1]);
587  r = sqrt(g * g + 1.0);
588  g = d[m - 1] - p + e[l - 1] / (g + fabs(r) * r8_sign(g));
589  s = 1.0;
590  c = 1.0;
591  p = 0.0;
592  mml = m - l;
593 
594  for (ii = 1; ii <= mml; ii++) {
595  i = m - ii;
596  f = s * e[i - 1];
597  b = c * e[i - 1];
598 
599  if (fabs(g) <= fabs(f)) {
600  c = g / f;
601  r = sqrt(c * c + 1.0);
602  e[i] = f * r;
603  s = 1.0 / r;
604  c = c * s;
605  } else {
606  s = f / g;
607  r = sqrt(s * s + 1.0);
608  e[i] = g * r;
609  c = 1.0 / r;
610  s = s * c;
611  }
612  g = d[i] - p;
613  r = (d[i - 1] - g) * s + 2.0 * c * b;
614  p = s * r;
615  d[i] = g + p;
616  g = c * r - b;
617  f = z[i];
618  z[i] = s * z[i - 1] + c * f;
619  z[i - 1] = c * z[i - 1] - s * f;
620  }
621  d[l - 1] = d[l - 1] - p;
622  e[l - 1] = g;
623  e[m - 1] = 0.0;
624  }
625  }
626  //
627  // Sorting.
628  //
629  for (ii = 2; ii <= m; ii++) {
630  i = ii - 1;
631  k = i;
632  p = d[i - 1];
633 
634  for (j = ii; j <= n; j++) {
635  if (d[j - 1] < p) {
636  k = j;
637  p = d[j - 1];
638  }
639  }
640 
641  if (k != i) {
642  d[k - 1] = d[i - 1];
