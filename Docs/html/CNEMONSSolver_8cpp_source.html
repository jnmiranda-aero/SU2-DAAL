57void CNEMONSSolver::Preprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh,
58 unsigned short iRKStep, unsigned short RunTime_EqSystem, bool Output) {
59
60 const auto InnerIter = config->GetInnerIter();
61 const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE) && (InnerIter <= config->GetLimiterIter());
62 const bool van_albada = config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE;
63 const bool muscl = config->GetMUSCL_Flow() && (iMesh == MESH_0);
64 const bool center = config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED;
65 const bool wall_functions = config->GetWall_Functions();
66
67 /*--- Common preprocessing steps (implemented by CNEMOEulerSolver) ---*/
68
69 CommonPreprocessing(geometry, solver_container, config, iMesh, iRKStep, RunTime_EqSystem, Output);
70
71 /*--- Compute gradient for MUSCL reconstruction. ---*/
72
73 if (config->GetReconstructionGradientRequired() && muscl && !center) {
74 switch (config->GetKind_Gradient_Method_Recon()) {
75 case GREEN_GAUSS:
76 SetPrimitive_Gradient_GG(geometry, config, true); break;
77 case LEAST_SQUARES:
78 case WEIGHTED_LEAST_SQUARES:
79 SetPrimitive_Gradient_LS(geometry, config, true); break;
80 default: break;
81 }
82 }
83
84 /*--- Compute gradient of the primitive variables ---*/
85
86 if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
87 SetPrimitive_Gradient_GG(geometry, config);
88 }
89 else if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
90 SetPrimitive_Gradient_LS(geometry, config);
91 }
92
93 /*--- Compute the limiters ---*/
94
95 if (muscl && !center && limiter && !van_albada && !Output) {
96 SetPrimitive_Limiter(geometry, config);
97 }
98
99 /*--- Compute vorticity and strain mag. ---*/
100
101 ComputeVorticityAndStrainMag(*config, geometry, iMesh);
102
103 /*--- Compute the TauWall from the wall functions ---*/
104
105 if (wall_functions) {
106 SetTau_Wall_WF(geometry, solver_container, config);
107 }

146void CNEMONSSolver::SetPrimitive_Gradient_GG(CGeometry *geometry, const CConfig *config, bool reconstruction) {
147
148 auto& gradient = reconstruction ? nodes->GetGradient_Reconstruction() : nodes->GetGradient_Primitive();
149 const auto comm = reconstruction? PRIMITIVE_GRAD_REC : PRIMITIVE_GRADIENT;
150 const auto commPer = reconstruction? PERIODIC_PRIM_GG_R : PERIODIC_PRIM_GG;
151
152 /*--- Get indices of species & mixture density ---*/
153 const unsigned short RHOS_INDEX = nodes->GetRhosIndex();
154 const unsigned short RHO_INDEX = nodes->GetRhoIndex();
155
156 /*--- Modify species density to mass concentration ---*/
157 for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
158 su2double primitives_aux[MAXNVAR] = {0.0};
159 for (auto iVar = 0ul; iVar < nPrimVar; iVar++)
160 primitives_aux[iVar] = nodes->GetPrimitive(iPoint, iVar);
161 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
162 primitives_aux[RHOS_INDEX+iSpecies] = primitives_aux[RHOS_INDEX+iSpecies]/primitives_aux[RHO_INDEX];
163 for (auto iVar = 0ul; iVar < nPrimVar; iVar++)
164 nodes->SetPrimitive_Aux(iPoint, iVar, primitives_aux[iVar] );
165 }
166
167 const auto& primitives = nodes->GetPrimitive_Aux();
168
169 computeGradientsGreenGauss(this, comm, commPer, *geometry, *config, primitives, 0, nPrimVarGrad, gradient);
170}

172void CNEMONSSolver::Viscous_Residual(CGeometry *geometry,
173 CSolver **solver_container,
174 CNumerics **numerics_container,
175 CConfig *config, unsigned short iMesh,
176 unsigned short iRKStep) {
177
178 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
179 CNumerics* numerics = numerics_container[VISC_TERM];
180
181 for (auto iEdge = 0ul; iEdge < geometry->GetnEdge(); iEdge++) {
182
183 /*--- Points, coordinates and normal vector in edge ---*/
184 const auto iPoint = geometry->edges->GetNode(iEdge, 0);
185 const auto jPoint = geometry->edges->GetNode(iEdge, 1);
186 numerics->SetCoord(geometry->nodes->GetCoord(iPoint),
187 geometry->nodes->GetCoord(jPoint) );
188 numerics->SetNormal(geometry->edges->GetNormal(iEdge));
189
190 /*--- Primitive variables, and gradient ---*/
191 numerics->SetConservative (nodes->GetSolution(iPoint),
192 nodes->GetSolution(jPoint) );
193 numerics->SetPrimitive (nodes->GetPrimitive(iPoint),
194 nodes->GetPrimitive(jPoint) );
195 numerics->SetPrimVarGradient(nodes->GetGradient_Primitive(iPoint),
196 nodes->GetGradient_Primitive(jPoint) );
197
198 /*--- Pass supplementary information to CNumerics ---*/
199 numerics->SetdPdU (nodes->GetdPdU(iPoint), nodes->GetdPdU(jPoint));
200 numerics->SetdTdU (nodes->GetdTdU(iPoint), nodes->GetdTdU(jPoint));
201 numerics->SetdTvedU(nodes->GetdTvedU(iPoint), nodes->GetdTvedU(jPoint));
202 numerics->SetEve (nodes->GetEve(iPoint), nodes->GetEve(jPoint));
203 numerics->SetCvve (nodes->GetCvve(iPoint), nodes->GetCvve(jPoint));
204
205 /*--- Species diffusion coefficients ---*/
206 numerics->SetDiffusionCoeff(nodes->GetDiffusionCoeff(iPoint),
207 nodes->GetDiffusionCoeff(jPoint) );
208
209 /*--- Laminar viscosity ---*/
210 numerics->SetLaminarViscosity(nodes->GetLaminarViscosity(iPoint),
211 nodes->GetLaminarViscosity(jPoint) );
212
213 /*--- Eddy viscosity ---*/
214 numerics->SetEddyViscosity(nodes->GetEddyViscosity(iPoint),
215 nodes->GetEddyViscosity(jPoint) );
216
217 /*--- Thermal conductivity ---*/
218 numerics->SetThermalConductivity(nodes->GetThermalConductivity(iPoint),
219 nodes->GetThermalConductivity(jPoint));
220
221 /*--- Vib-el. thermal conductivity ---*/
222 numerics->SetThermalConductivity_ve(nodes->GetThermalConductivity_ve(iPoint),
223 nodes->GetThermalConductivity_ve(jPoint) );
224
225 /*--- Compute and update residual ---*/
226 auto residual = numerics->ComputeResidual(config);
227
228 /*--- Check for NaNs before applying the residual to the linear system ---*/
229 bool err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
230
231 /*--- Update the residual and Jacobian ---*/
232 if (!err) {
233 LinSysRes.SubtractBlock(iPoint, residual);
234 LinSysRes.AddBlock(jPoint, residual);
235 if (implicit) {
236 Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
237 }
238 }
239 } //iEdge
240}

242void CNEMONSSolver::BC_HeatFluxNonCatalytic_Wall(CGeometry *geometry,
243 CSolver **solver_container,
244 CNumerics *conv_numerics,
245 CNumerics *sour_numerics,
246 CConfig *config,
247 unsigned short val_marker) {
248
249 /*--- Local variables ---*/
250 const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
251 su2double Wall_HeatFlux = config->GetWall_HeatFlux(Marker_Tag) / config->GetHeat_Flux_Ref();
252 if (config->GetIntegrated_HeatFlux()) {
253 Wall_HeatFlux /= geometry->GetSurfaceArea(config, val_marker);
254 }
255 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
256
257 /*--- Set "Proportional control" coefficient ---*/
258 const su2double pcontrol = 1.0;
259
260 /*--- Get the locations of the primitive variables ---*/
261 const unsigned short T_INDEX = nodes->GetTIndex();
262 const unsigned short TVE_INDEX = nodes->GetTveIndex();
263
264 /*--- Loop over all of the vertices on this boundary marker ---*/
265 SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
266 for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
267
268 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
269
270 /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
271 if (!geometry->nodes->GetDomain(iPoint)) continue;
272
273 /*--- Compute dual-grid area and boundary normal ---*/
274 const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
275 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
276
277 /*--- Initialize the convective & viscous residuals to zero ---*/
278 for (auto iVar = 0ul; iVar < nVar; iVar++) {Res_Visc[iVar] = 0.0;}
279
280 /*--- Set the residual on the boundary with the specified heat flux ---*/
281 // TODO: Look into this!
282 // Note: Contributions from qtr and qve are used for proportional control
283 // to drive the solution toward the specified heatflux more quickly.
284 const auto GradV = nodes->GetGradient_Primitive(iPoint);
285 su2double dTdn = 0.0;
286 su2double dTvedn = 0.0;
287 for (auto iDim = 0ul; iDim < nDim; iDim++) {
288 dTdn += GradV[T_INDEX][iDim]*Normal[iDim];
289 dTvedn += GradV[TVE_INDEX][iDim]*Normal[iDim];
290 }
291 const su2double ktr = nodes->GetThermalConductivity(iPoint);
292 const su2double kve = nodes->GetThermalConductivity_ve(iPoint);
293
294 /*--- Compute residual ---*/
295 Res_Visc[nSpecies+nDim] += pcontrol*(ktr*dTdn+kve*dTvedn) +
296 Wall_HeatFlux*Area;
297 Res_Visc[nSpecies+nDim+1] += pcontrol*(kve*dTvedn) +
298 Wall_HeatFlux*Area;
299
300 /*--- Impose the value of the velocity as a strong boundary
301 condition (Dirichlet). Fix the velocity and remove any
302 contribution to the residual at this node. ---*/
303 su2double zero[MAXNDIM] = {0.0};
304 nodes->SetVelocity_Old(iPoint, zero);
305
306 for (auto iDim = 0ul; iDim < nDim; iDim++)
307 LinSysRes(iPoint, nSpecies+iDim) = 0.0;
308 nodes->SetVel_ResTruncError_Zero(iPoint);
309
310 /*--- Apply viscous residual to the linear system ---*/
311 LinSysRes.SubtractBlock(iPoint, Res_Visc);
312
313 if (implicit) {
314 /*--- Enforce the no-slip boundary condition in a strong way ---*/
315 for (int iVar = nSpecies; iVar < nSpecies+nDim; iVar++) {
316 auto total_index = iPoint*nVar+iVar;
317 Jacobian.DeleteValsRowi(total_index);
318 }
319 }
320
321 }
322 END_SU2_OMP_FOR
323}

524void CNEMONSSolver::BC_IsothermalNonCatalytic_Wall(CGeometry *geometry,
525 CSolver **solver_container,
526 CNumerics *conv_numerics,
527 CNumerics *sour_numerics,
528 CConfig *config,
529 unsigned short val_marker) {
530
531 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
532 const bool ionization = config->GetIonization();
533 su2double UnitNormal[MAXNDIM] = {0.0};
534
535 if (ionization) {
536 SU2_MPI::Error("NEED TO TAKE A CLOSER LOOK AT THE JACOBIAN W/ IONIZATION",CURRENT_FUNCTION);
537 }
538
539 /*--- Define 'proportional control' constant ---*/
540 const su2double C = 5;
541
542 /*--- Identify the boundary ---*/
543 const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
544
545 /*--- Retrieve the specified wall temperature ---*/
546 const su2double Twall = config->GetIsothermal_Temperature(Marker_Tag);
547
548 su2double **Jacobian_i = nullptr;
549 if (implicit) {
550 Jacobian_i = new su2double* [nVar];
551 for (auto iVar = 0ul; iVar < nVar; iVar++)
552 Jacobian_i[iVar] = new su2double [nVar] ();
553 }
554
555 /*--- Loop over boundary points to calculate energy flux ---*/
556 SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
557 for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
558
559 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
560
561 if (!geometry->nodes->GetDomain(iPoint)) continue;
562
563 /*--- Compute dual-grid area and boundary normal ---*/
564 const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
565 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
566 for (auto iDim = 0ul; iDim < nDim; iDim++)
567 UnitNormal[iDim] = Normal[iDim]/Area;
568
569 /*--- Compute closest normal neighbor ---*/
570 const auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
571
572 /*--- Compute distance between wall & normal neighbor ---*/
573 const auto Coord_i = geometry->nodes->GetCoord(iPoint);
574 const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);
575
576 const su2double dist_ij = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
577
578 /*--- Store the corrected velocity at the wall which will
579 be zero (v = 0), unless there is grid motion (v = u_wall)---*/
580 su2double zero[MAXNDIM] = {0.0};
581 nodes->SetVelocity_Old(iPoint, zero);
582
583 /*--- Initialize viscous residual to zero ---*/
584 for (auto iVar = 0ul; iVar < nVar; iVar ++) {Res_Visc[iVar] = 0.0;}
585
586 for (auto iDim = 0ul; iDim < nDim; iDim++)
587 LinSysRes(iPoint, nSpecies+iDim) = 0.0;
588 nodes->SetVel_ResTruncError_Zero(iPoint);
589
590 /*--- Calculate the gradient of temperature ---*/
591 const su2double Ti = nodes->GetTemperature(iPoint);
592 const su2double Tj = nodes->GetTemperature(Point_Normal);
593 const su2double Tvei = nodes->GetTemperature_ve(iPoint);
594 const su2double Tvej = nodes->GetTemperature_ve(Point_Normal);
595
596 /*--- Rename variables for convenience ---*/
597 const su2double ktr = nodes->GetThermalConductivity(iPoint);
598 const su2double kve = nodes->GetThermalConductivity_ve(iPoint);
599
600 /*--- Apply to the linear system ---*/
601 Res_Visc[nSpecies+nDim] = ((ktr*(Ti-Tj) + kve*(Tvei-Tvej)) +
602 (ktr*(Twall-Ti) + kve*(Twall-Tvei))*C)*Area/dist_ij;
603 Res_Visc[nSpecies+nDim+1] = (kve*(Tvei-Tvej) + kve*(Twall-Tvei) *C)*Area/dist_ij;
604
605 /*--- Calculate Jacobian for implicit time stepping ---*/
606 if (implicit) {
607
608 /*--- Initialize Jacobian to zero ---*/
609 for (auto iVar = 0ul; iVar < nVar; iVar++)
610 for (auto jVar = 0ul; jVar < nVar; jVar++)
611 Jacobian_i[iVar][jVar] = 0.0;
612
613 /*--- Add contributions to the Jacobian from the weak enforcement of the energy equations. ---*/
614 const auto dTdU = nodes->GetdTdU(iPoint);
615 const auto dTvedU = nodes->GetdTvedU(iPoint);
616 const su2double theta = GeometryToolbox::SquaredNorm(nDim, UnitNormal);
617
618 for (auto iVar = 0ul; iVar < nVar; iVar++) {
619 Jacobian_i[nSpecies+nDim][iVar] = -(ktr*theta/dist_ij*dTdU[iVar] +
620 kve*theta/dist_ij*dTvedU[iVar])*Area;
621 Jacobian_i[nSpecies+nDim+1][iVar] = - kve*theta/dist_ij*dTvedU[iVar]*Area;
622 }
623 } // implicit
624
625 /*--- Convective and viscous contributions to the residual at the wall ---*/
626 LinSysRes.SubtractBlock(iPoint, Res_Visc);
627
628 /*--- Enforce the no-slip boundary condition in a strong way by
629 modifying the velocity-rows of the Jacobian (1 on the diagonal).
630 And add the contributions to the Jacobian due to energy. ---*/
631 if (implicit) {
632 Jacobian.SubtractBlock2Diag(iPoint, Jacobian_i);
633
634 for (auto iVar = 1u; iVar <= nDim; iVar++) {
635 auto total_index = iPoint*nVar+iVar;
636 Jacobian.DeleteValsRowi(total_index);
637 }
638 }
639 }
640 END_SU2_OMP_FOR
641
642 if (Jacobian_i)
643 for (auto iVar = 0ul; iVar < nVar; iVar++)
644 delete [] Jacobian_i[iVar];
645 delete [] Jacobian_i;
646
647}

649void CNEMONSSolver::BC_IsothermalCatalytic_Wall(CGeometry *geometry,
650 CSolver **solver_container,
651 CNumerics *conv_numerics,
652 CNumerics *sour_numerics,
653 CConfig *config,
654 unsigned short val_marker) {
655
656 /*--- Call standard isothermal BC to apply no-slip and energy b.c.'s ---*/
657 BC_IsothermalNonCatalytic_Wall(geometry, solver_container, conv_numerics,
658 sour_numerics, config, val_marker);
659
660 /*--- Local variables ---*/
661 su2double **GradY, **dVdU;
662
663 /*--- Assign booleans ---*/
664 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
665 const bool Supercatalytic_Wall = config->GetSupercatalytic_Wall();
666
667 /*--- Get universal information ---*/
668 const su2double RuSI = UNIVERSAL_GAS_CONSTANT;
669 const su2double Ru = 1000.0*RuSI;
670 const auto& Ms = FluidModel->GetSpeciesMolarMass();
671
672 /*--- Get the locations of the primitive variables ---*/
673 const unsigned short RHOS_INDEX = nodes->GetRhosIndex();
674 const unsigned short RHO_INDEX = nodes->GetRhoIndex();
675 const unsigned short T_INDEX = nodes->GetTIndex();
676 const unsigned short TVE_INDEX = nodes->GetTveIndex();
677
678 /*--- Allocate arrays ---*/
679 GradY = new su2double*[nSpecies];
680 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
681 GradY[iSpecies] = new su2double[nDim];
682 dVdU = new su2double*[nVar];
683 for (auto iVar = 0ul; iVar < nVar; iVar++)
684 dVdU[iVar] = new su2double[nVar];
685
686 /*--- Loop over all of the vertices on this boundary marker ---*/
687 for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
688
689 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
690
691 /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
692 if (geometry->nodes->GetDomain(iPoint)) {
693
694 /*--- Compute closest normal neighbor ---*/
695 const auto jPoint = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
696
697 /*--- Compute distance between wall & normal neighbor ---*/
698 const auto Coord_i = geometry->nodes->GetCoord(iPoint);
699 const auto Coord_j = geometry->nodes->GetCoord(jPoint);
700 const su2double dij = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
701
702 /*--- Compute dual-grid area and boundary normal ---*/
703 const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
704 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
705
706 /*--- Initialize the viscous residual to zero ---*/
707 for (auto iVar = 0ul; iVar < nVar; iVar++) Res_Visc[iVar] = 0.0;
708
709 /*--- Get primitive information ---*/
710 const auto& Vi = nodes->GetPrimitive(iPoint);
711 const auto& Vj = nodes->GetPrimitive(jPoint);
712 const auto& Di = nodes->GetDiffusionCoeff(iPoint);
713 const auto& eves = nodes->GetEve(iPoint);
714 const auto& hs = FluidModel->ComputeSpeciesEnthalpy(Vi[T_INDEX], Vi[TVE_INDEX], eves);
715 const su2double rho = Vi[RHO_INDEX];
716 const auto& dTdU = nodes->GetdTdU(iPoint);
717 const auto& dTvedU = nodes->GetdTvedU(iPoint);
718
719 if (Supercatalytic_Wall) {
720
721 const auto& Yst = config->GetSupercatalytic_Wall_Composition();
722
723 /*--- Calculate supplementary quantities ---*/
724 su2double dYdn = 0.0, SdYdn = 0.0;
725 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
726 dYdn = (Yst[iSpecies]-Vj[RHOS_INDEX+iSpecies]/Vj[RHO_INDEX])/dij;
727 SdYdn += rho*Di[iSpecies]*dYdn;
728 }
729
730 /*--- Calculate species residual at wall ---*/
731 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
732 dYdn = (Yst[iSpecies]-Vj[RHOS_INDEX+iSpecies]/Vj[RHO_INDEX])/dij;
733 Res_Visc[iSpecies] = -(-rho*Di[iSpecies]*dYdn+Yst[iSpecies]*SdYdn)*Area;
734 }
735
736 if (implicit) {
737 /*--- Initialize the transformation matrix ---*/
738 for (auto iVar = 0ul; iVar < nVar; iVar++)
739 for (auto jVar = 0ul; jVar < nVar; jVar++) {
740 dVdU[iVar][jVar] = 0.0;
741 Jacobian_j[iVar][jVar] = 0.0;
742 Jacobian_i[iVar][jVar] = 0.0;
743 }
744
745 /*--- Populate transformation matrix ---*/
746 // dYsdrk
747 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
748 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++)
749 dVdU[iSpecies][jSpecies] += -1.0/rho*Yst[iSpecies];
750 dVdU[iSpecies][iSpecies] += 1.0/rho;
751 }
752 for (auto iVar = 0ul; iVar < nVar; iVar++) {
753 dVdU[nSpecies+nDim][iVar] = dTdU[iVar];
754 dVdU[nSpecies+nDim+1][iVar] = dTvedU[iVar];
755 }
756
757 /*--- Calculate supplementary quantities ---*/
758 const auto& Cvtrs = FluidModel->GetSpeciesCvTraRot();
759 const auto& Cvve = nodes->GetCvve(iPoint);
760
761 /*--- Take the primitive var. Jacobian & store in Jac. jj ---*/
762 // Species mass fraction
763 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
764 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++)
765 Jacobian_j[iSpecies][jSpecies] += -Yst[iSpecies]*rho*Di[jSpecies]/dij;
766 Jacobian_j[iSpecies][iSpecies] += rho*Di[iSpecies]/dij - SdYdn;
767 }
768
769 // Temperature
770 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
771 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++) {
772 Jacobian_j[nSpecies+nDim][iSpecies] += Jacobian_j[jSpecies][iSpecies]*hs[iSpecies];
773 }
774 Jacobian_j[nSpecies+nDim][nSpecies+nDim] += Res_Visc[iSpecies]/Area*(Ru/Ms[iSpecies] +
775 Cvtrs[iSpecies] );
776 Jacobian_j[nSpecies+nDim][nSpecies+nDim+1] += Res_Visc[iSpecies]/Area*Cvve[iSpecies];
777 }
778
779 // Vib.-El. Temperature
780 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
781 for (auto jSpecies = 0ul; jSpecies < nSpecies; jSpecies++)
782 Jacobian_j[nSpecies+nDim+1][iSpecies] += Jacobian_j[jSpecies][iSpecies]*eves[iSpecies];
783 Jacobian_j[nSpecies+nDim+1][nSpecies+nDim+1] += Res_Visc[iSpecies]/Area*Cvve[iSpecies];
784 }
785
786 /*--- Multiply by the transformation matrix and store in Jac. ii ---*/
787 for (auto iVar = 0ul; iVar < nVar; iVar++)
788 for (auto jVar = 0ul; jVar < nVar; jVar++)
789 for (auto kVar = 0ul; kVar < nVar; kVar++)
790 Jacobian_i[iVar][jVar] += Jacobian_j[iVar][kVar]*dVdU[kVar][jVar]*Area;
791
792 /*--- Apply to the linear system ---*/
793 Jacobian.SubtractBlock(iPoint, iPoint, Jacobian_i);
794 }
795
796 } else {
797
798 if (implicit) {
799 SU2_MPI::Error("Implicit not currently available for partially catalytic wall.",CURRENT_FUNCTION);
800 }
801
802 /*--- Identify the boundary ---*/
803 string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
804
805 /*--- Get isothermal wall temp ----*/
806 const su2double Tw = config->GetIsothermal_Temperature(Marker_Tag);
807
808 /*--- Get wall catalytic efficiency ----*/
809 const su2double gam = config->GetCatalytic_Efficiency();
810
811 /*--- Get cataltyic reaction map ---*/
812 const auto& RxnTable = FluidModel->GetCatalyticRecombination();
813
814 /*--- Common catalytic flux factor ---*/
815 const su2double factor = gam*rho*sqrt(RuSI*Tw/2/PI_NUMBER)*Area;
816
817 /*--- Compute catalytic recombination flux ---*/
818 // Ref: 10.2514/6.2022-1636
819 // ws = gam_s*Ys*rho_wall*sqrt(Ru*Tw/(2*Pi*M_combine)*Area
820 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
821 int Index = SU2_TYPE::Int(RxnTable(iSpecies,1));
822 Res_Visc[iSpecies] = RxnTable(iSpecies,0)*factor*Vi[Index]/Vi[RHO_INDEX]*sqrt(1/Ms[Index]);
823 }
824 }
825
826 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
827 Res_Visc[nSpecies+nDim] += (Res_Visc[iSpecies]*hs[iSpecies])*Area;
828 Res_Visc[nSpecies+nDim+1] += (Res_Visc[iSpecies]*eves[iSpecies])*Area;
829 }
830
831 /*--- Viscous contribution to the residual at the wall ---*/
832 LinSysRes.SubtractBlock(iPoint, Res_Visc);
833 }
834 }
835
836 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++)
837 delete [] GradY[iSpecies];
838 delete [] GradY;
839 for (auto iVar = 0ul; iVar < nVar; iVar++)
840 delete [] dVdU[iVar];
841 delete [] dVdU;
842}

844void CNEMONSSolver::BC_Smoluchowski_Maxwell(CGeometry *geometry,
845 CSolver **solver_container,
846 CNumerics *conv_numerics,
847 CNumerics *visc_numerics,
848 CConfig *config,
849 unsigned short val_marker) {
850
851
852 const bool ionization = config->GetIonization();
853
854 if (ionization) {
855 SU2_MPI::Error("NEED TO TAKE A CLOSER LOOK AT THE JACOBIAN W/ IONIZATION", CURRENT_FUNCTION);
856 }
857
858 /*--- Define 'proportional control' constant ---*/
859 const su2double C = 1;
860
861 /*---Define under-relaxation factors --- */
862 const su2double alpha_V = 0.1;
863 const su2double alpha_T = 1.0;
864
865 /*--- Identify the boundary ---*/
866 string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
867
868 /*--- Retrieve the specified wall temperature and accomodation coefficients---*/
869 const su2double Twall = config->GetIsothermal_Temperature(Marker_Tag);
870 const su2double TMAC = 1.0;
871 const su2double TAC = 1.0;
872
873 /*--- Extract necessary indices ---*/
874 const unsigned short T_INDEX = nodes->GetTIndex();
875 const unsigned short VEL_INDEX = nodes->GetVelIndex();
876 const unsigned short TVE_INDEX = nodes->GetTveIndex();
877
878 /*--- Loop over boundary points to calculate energy flux ---*/
879 SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
880 for(auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
881
882 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
883
884 if (!geometry->nodes->GetDomain(iPoint)) continue;
885
886 /*--- Compute dual-grid area and boundary normal ---*/
887 const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
888 const su2double Area = GeometryToolbox::Norm(nDim, Normal);
889
890 su2double UnitNormal[MAXNDIM] = {0.0};
891 for (auto iDim = 0ul; iDim < nDim; iDim++)
892 UnitNormal[iDim] = Normal[iDim]/Area;
893
894 /*--- Compute closest normal neighbor ---*/
895 const auto jPoint = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
896
897 /*--- Compute distance between wall & normal neighbor ---*/
898 const auto Coord_i = geometry->nodes->GetCoord(iPoint);
899 const auto Coord_j = geometry->nodes->GetCoord(jPoint);
900
901 const su2double dij = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
902
903 /*--- Calculate Pressure ---*/
904 const su2double Pi = nodes->GetPressure(iPoint);
905
906 /*--- Calculate the gradient of temperature ---*/
907 const su2double Ti = nodes->GetTemperature(iPoint);
908 const su2double Tj = nodes->GetTemperature(jPoint);
909 const su2double Tvei = nodes->GetTemperature_ve(iPoint);
910 const su2double Tvej = nodes->GetTemperature_ve(jPoint);
911
912 /*--- Rename variables for convenience ---*/
913 const su2double ktr = nodes->GetThermalConductivity(iPoint);
914 const su2double kve = nodes->GetThermalConductivity_ve(iPoint);
915
916 /*--- Retrieve Cv*density ---*/
917 const su2double rhoCvtr = nodes->GetRhoCv_tr(iPoint);
918 const su2double rhoCvve = nodes->GetRhoCv_ve(iPoint);
919
920 /*--- Retrieve Flow Data ---*/
921 const su2double Viscosity = nodes->GetLaminarViscosity(iPoint);
922 const su2double Density = nodes->GetDensity(iPoint);
923 const su2double Gamma = nodes->GetGamma(iPoint);
924
925 /*--- Retrieve Primitive Gradients ---*/
926 const auto Grad_PrimVar = nodes->GetGradient_Primitive(iPoint);
927
928 /*--- Calculate specific gas constant --- */
929 const su2double GasConstant = FluidModel->ComputeGasConstant();
930
931 /*--- Calculate temperature gradients normal to surface---*/ //Doubt about minus sign
932 const su2double dTn = GeometryToolbox::DotProduct(nDim, Grad_PrimVar[T_INDEX], UnitNormal);
933 const su2double dTven = GeometryToolbox::DotProduct(nDim, Grad_PrimVar[TVE_INDEX], UnitNormal);
934
935 /*--- Calculate molecular mean free path ---*/
936 const su2double Lambda = Viscosity/Density*sqrt(PI_NUMBER/(2.0*GasConstant*Ti));
937
938 /*--- Calculate Temperature Slip ---*/
939 su2double Tslip = ((2.0-TAC)/TAC)*2.0*Gamma/(Gamma+1.0)/Prandtl_Lam*Lambda*dTn+Twall;
940
941 su2double Tslip_ve = Twall;
942 if (dTven !=0) Tslip_ve = (Tslip-Twall)*(kve*rhoCvtr/dTn)/(ktr*rhoCvve/dTven)+Twall;
943
944 /*--- Calculate temperature gradients tangent to surface ---*/
945 su2double Vector_Tangent_dT[MAXNDIM] = {0.0}, Vector_Tangent_dTve[MAXNDIM] = {0.0};
946 for (auto iDim = 0ul; iDim < nDim; iDim++) {
947 Vector_Tangent_dT[iDim] = Grad_PrimVar[T_INDEX][iDim] - dTn * UnitNormal[iDim];
948 Vector_Tangent_dTve[iDim] = Grad_PrimVar[TVE_INDEX][iDim] - dTven * UnitNormal[iDim];
949 }
950
951 /*--- Calculate Heatflux tangent to surface ---*/
952 su2double Vector_Tangent_HF[MAXNDIM] = {0.0};
953 for (auto iDim = 0ul; iDim < nDim; iDim++)
954 Vector_Tangent_HF[iDim] = -ktr*Vector_Tangent_dT[iDim]-kve*Vector_Tangent_dTve[iDim];
955
956 /*--- Initialize viscous residual to zero ---*/
957 for (auto iVar = 0ul; iVar < nVar; iVar ++) Res_Visc[iVar] = 0.0;
958
959 /*--- Compute wall shear stress (using the stress tensor) ---*/
960 su2double Tau[MAXNDIM][MAXNDIM] = {{0.0}};
961 CNumerics::ComputeStressTensor(nDim, Tau, Grad_PrimVar+VEL_INDEX, Viscosity);
962
963 su2double TauTangent[MAXNDIM] = {0.0};
964 GeometryToolbox::TangentProjection(nDim,Tau,UnitNormal,TauTangent);
965
966 /*--- Store the Slip Velocity at the wall */
967 for (auto iDim = 0ul; iDim < nDim; iDim++)
968 Vector[iDim] = Lambda/Viscosity*(2.0-TMAC)/TMAC*(TauTangent[iDim])-3.0/4.0*(Gamma-1.0)/Gamma*Prandtl_Lam/Pi*Vector_Tangent_HF[iDim];
969
970 /*--- Apply under-relaxation ---*/
971 Tslip = (1.0-alpha_T)*nodes->GetTemperature(iPoint)+(alpha_T)*Tslip;
972 Tslip_ve = (1.0-alpha_T)*nodes->GetTemperature_ve(iPoint)+(alpha_T)*Tslip_ve;
973
974 for (auto iDim = 0ul; iDim < nDim; iDim++){
975 Vector[iDim] = (1.0-alpha_V)*nodes->GetVelocity(iPoint,iDim)+(alpha_V)*Vector[iDim];
976 }
977
978 nodes->SetVelocity_Old(iPoint,Vector);
979
980 for (auto iDim = 0ul; iDim < nDim; iDim++) {
981 LinSysRes(iPoint, nSpecies+iDim) = 0.0;
982 nodes->SetVal_ResTruncError_Zero(iPoint,nSpecies+iDim);
983 }
984
985 /*--- Apply to the linear system ---*/
986 Res_Visc[nSpecies+nDim] = ((ktr*(Ti-Tj) + kve*(Tvei-Tvej)) +
987 (ktr*(Tslip-Ti) + kve*(Tslip_ve-Tvei))*C)*Area/dij;
988 Res_Visc[nSpecies+nDim+1] = (kve*(Tvei-Tvej) + kve*(Tslip_ve-Tvei)*C)*Area/dij;
989
990 LinSysRes.SubtractBlock(iPoint, Res_Visc);
991 }
992 END_SU2_OMP_FOR
993}
