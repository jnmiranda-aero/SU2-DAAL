60void CAvgGrad_AdjFlow::ComputeResidual(su2double *val_residual_i, su2double *val_residual_j,
61 su2double **val_Jacobian_ii, su2double **val_Jacobian_ij,
62 su2double **val_Jacobian_ji, su2double **val_Jacobian_jj, CConfig *config) {
63 unsigned short iDim, jDim;
64 su2double sq_vel_i, ViscDens_i, XiDens_i;
65 su2double sq_vel_j, ViscDens_j, XiDens_j;
66 su2double dist_ij_2, dPhiE_dn;
67
68 su2double Prandtl_Lam = config->GetPrandtl_Lam();
69 su2double Prandtl_Turb = config->GetPrandtl_Turb();
70
71 /*--- States in point i ---*/
72
73 sq_vel_i = 0.0;
74 for (iDim = 0; iDim < nDim; iDim++) {
75 Velocity_i[iDim] = V_i[iDim+1];
76 sq_vel_i += 0.5*Velocity_i[iDim]*Velocity_i[iDim];
77 }
78 Pressure_i = V_i[nDim+1];
79 Density_i = V_i[nDim+2];
80 Enthalpy_i = V_i[nDim+3];
81 SoundSpeed_i = sqrt(fabs(Pressure_i*Gamma/Density_i));
82
83 /*--- Laminar and Eddy viscosity ---*/
84
85 Laminar_Viscosity_i = V_i[nDim+5];
86 Eddy_Viscosity_i = V_i[nDim+6];
87
88 ViscDens_i = (Laminar_Viscosity_i + Eddy_Viscosity_i) / Density_i;
89 XiDens_i = Gamma*(Laminar_Viscosity_i/Prandtl_Lam + Eddy_Viscosity_i/Prandtl_Turb) / Density_i;
90
91 /*--- States in point j ---*/
92
93 sq_vel_j = 0.0;
94 for (iDim = 0; iDim < nDim; iDim++) {
95 Velocity_j[iDim] = V_j[iDim+1];
96 sq_vel_j += 0.5*Velocity_j[iDim]*Velocity_j[iDim];
97 }
98 Pressure_j = V_j[nDim+1];
99 Density_j = V_j[nDim+2];
100 Enthalpy_j = V_j[nDim+3];
101 SoundSpeed_j = sqrt(fabs(Pressure_j*Gamma/Density_j));
102
103 /*--- Laminar and Eddy viscosity ---*/
104
105 Laminar_Viscosity_j = V_j[nDim+5];
106 Eddy_Viscosity_j = V_j[nDim+6];
107
108 ViscDens_j = (Laminar_Viscosity_j + Eddy_Viscosity_j) / Density_j;
109 XiDens_j = Gamma*(Laminar_Viscosity_j/Prandtl_Lam + Eddy_Viscosity_j/Prandtl_Turb) / Density_j;
110
111 /*--- Compute vector going from iPoint to jPoint ---*/
112
113 dist_ij_2 = 0.0;
114 for (iDim = 0; iDim < nDim; iDim++) {
115 Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
116 dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
117 }
118
119 /*--- Average of the derivatives of the adjoint variables ---*/
120
121 for (iDim = 0; iDim < nDim; iDim++) {
122 Mean_GradPsiE[iDim] = 0.5*(PsiVar_Grad_i[nVar-1][iDim]+PsiVar_Grad_j[nVar-1][iDim]);
123 for (jDim = 0; jDim < nDim; jDim++)
124 Mean_GradPhi[iDim][jDim] = 0.5*(PsiVar_Grad_i[iDim+1][jDim]+PsiVar_Grad_j[iDim+1][jDim]);
125 }
126
127 dPhiE_dn = 0;
128 for (iDim = 0; iDim < nDim; iDim++)
129 dPhiE_dn += Mean_GradPsiE[iDim]*Normal[iDim];
130
131 /*--- Compute the viscous residual and jacobian ---*/
132
133 GetAdjViscousFlux_Jac(Pressure_i, Pressure_j, Density_i, Density_j,
134 ViscDens_i, ViscDens_j, Velocity_i, Velocity_j, sq_vel_i, sq_vel_j,
135 XiDens_i, XiDens_j, Mean_GradPhi, Mean_GradPsiE,
136 dPhiE_dn, Normal, Edge_Vector, dist_ij_2, val_residual_i, val_residual_j,
137 val_Jacobian_ii, val_Jacobian_ij, val_Jacobian_ji, val_Jacobian_jj,
138 implicit);
139
140}

243 /*--- Compute vector going from iPoint to jPoint ---*/
244
245 dist_ij_2 = 0.0;
246 for (iDim = 0; iDim < nDim; iDim++) {
247 Mean_Velocity[iDim] = 0.5*(Velocity_i[iDim]+Velocity_j[iDim]);
248 Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
249 dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
250 }
251
252 /*--- Mean gradient approximation. Projection of the mean gradient in the direction of the edge, weiss correction ---*/
253
254 for (iVar = 0; iVar < nVar; iVar++) {
255 Proj_Mean_GradPsiVar_Edge[iVar] = 0.0;
256 for (iDim = 0; iDim < nDim; iDim++) {
257 Mean_GradPsiVar[iVar][iDim] = 0.5*(PsiVar_Grad_i[iVar][iDim] + PsiVar_Grad_j[iVar][iDim]);
258 Proj_Mean_GradPsiVar_Edge[iVar] += Mean_GradPsiVar[iVar][iDim]*Edge_Vector[iDim];
259 }
260 for (iDim = 0; iDim < nDim; iDim++)
261 Mean_GradPsiVar[iVar][iDim] -= (Proj_Mean_GradPsiVar_Edge[iVar] -
262 (Psi_j[iVar]-Psi_i[iVar]))*Edge_Vector[iDim]/dist_ij_2;
263 }
264
265 /*--- Average of the derivatives of the adjoint variables ---*/
266
267 for (iDim = 0; iDim < nDim; iDim++) {
268 Mean_GradPsiE[iDim] = Mean_GradPsiVar[nVar-1][iDim];
269 for (jDim = 0; jDim < nDim; jDim++)
270 Mean_GradPhi[iDim][jDim] = Mean_GradPsiVar[iDim+1][jDim];
271 }
272
273 dPhiE_dn = 0;
274 for (iDim = 0; iDim < nDim; iDim++)
275 dPhiE_dn += Mean_GradPsiE[iDim]*Normal[iDim];

280 ViscDens_i, ViscDens_j, Velocity_i, Velocity_j, sq_vel_i, sq_vel_j,
281 XiDens_i, XiDens_j, Mean_GradPhi, Mean_GradPsiE,
282 dPhiE_dn, Normal, Edge_Vector, dist_ij_2, val_residual_i, val_residual_j,
283 val_Jacobian_ii, val_Jacobian_ij, val_Jacobian_ji, val_Jacobian_jj,
284 implicit);

313void CAvgGrad_AdjTurb::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i,
314 su2double **val_Jacobian_j, CConfig *config) {
315
316 bool implicit = (config->GetKind_TimeIntScheme_AdjTurb() == EULER_IMPLICIT);
317
318 su2double sigma = 2./3.;
319 su2double nu_i, nu_j, nu_e;
320 su2double dist_ij_2 = 0;
321 su2double proj_vector_ij = 0;
322 unsigned short iVar, iDim;
323
324 /*--- Compute mean effective viscosity ---*/
325 nu_i = Laminar_Viscosity_i/U_i[0];
326 nu_j = Laminar_Viscosity_j/U_j[0];
327 nu_e = 0.5*(nu_i+nu_j+ScalarVar_i[0]+ScalarVar_j[0])/sigma;
328
329 /*--- Compute vector going from iPoint to jPoint ---*/
330 for (iDim = 0; iDim < nDim; iDim++) {
331 Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
332 dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
333 proj_vector_ij += Edge_Vector[iDim]*Normal[iDim];
334 }
335 proj_vector_ij = proj_vector_ij/dist_ij_2;
336
337 /*--- Mean gradient approximation ---*/
338 for (iVar = 0; iVar < nVar; iVar++) {
339 Proj_Mean_GradTurbPsi_Edge[iVar] = 0.0;
340 for (iDim = 0; iDim < nDim; iDim++) {
341 Mean_GradTurbPsi[iVar][iDim] = 0.5*(TurbPsi_Grad_i[iVar][iDim] + TurbPsi_Grad_j[iVar][iDim]);
342 }
343
344 /*--- Projection of the corrected gradient ---*/
345 Proj_Mean_GradTurbPsi_Corrected[iVar] = 0.0;
346 for (iDim = 0; iDim < nDim; iDim++)
347 Proj_Mean_GradTurbPsi_Corrected[iVar] += Mean_GradTurbPsi[iVar][iDim]*Normal[iDim];
348 }
349
350 val_residual[0] = -nu_e*Proj_Mean_GradTurbPsi_Corrected[0];
351
352 if (implicit) {
353 val_Jacobian_i[0][0] = nu_e*proj_vector_ij;
354 val_Jacobian_j[0][0] = -nu_e*proj_vector_ij;
355 }
356
357}

507 }
508 proj_vector_ij = proj_vector_ij/dist_ij_2; // to normalize vectors
509
510 /*--- Mean gradient approximation ---*/
511 for (iVar = 0; iVar < nVar; iVar++) {
512 Proj_Mean_GradTurbPsi_Edge[iVar] = 0.0;
513 for (iDim = 0; iDim < nDim; iDim++) {
514 Mean_GradTurbPsi[iVar][iDim] = 0.5*(TurbPsi_Grad_i[iVar][iDim] + TurbPsi_Grad_j[iVar][iDim]);
515 Proj_Mean_GradTurbPsi_Edge[iVar] += Mean_GradTurbPsi[iVar][iDim]*Edge_Vector[iDim];
516 }
517
518 for (iDim = 0; iDim < nDim; iDim++)
519 Mean_GradTurbPsi[iVar][iDim] -= (Proj_Mean_GradTurbPsi_Edge[iVar] -
520 (TurbPsi_j[iVar]-TurbPsi_i[iVar]))*Edge_Vector[iDim]/dist_ij_2;
521
522 /*--- Projection of the corrected gradient ---*/
523 Proj_Mean_GradTurbPsi_Corrected[iVar] = 0.0;
524 for (iDim = 0; iDim < nDim; iDim++)
525 Proj_Mean_GradTurbPsi_Corrected[iVar] += Mean_GradTurbPsi[iVar][iDim]*Normal[iDim];
526 }
527
528 val_residual_i[0] = -nu_e_i*Proj_Mean_GradTurbPsi_Corrected[0];
529 val_residual_j[0] = nu_e_j*Proj_Mean_GradTurbPsi_Corrected[0];
530
531 if (implicit) {
532 val_Jacobian_ii[0][0] = nu_e_i*proj_vector_ij;
533 val_Jacobian_ij[0][0] = -nu_e_i*proj_vector_ij;
534 val_Jacobian_ji[0][0] = -nu_e_j*proj_vector_ij;
535 val_Jacobian_jj[0][0] = nu_e_j*proj_vector_ij;
536 }
