49 void CIsoparametric::SetTransferCoeff(const CConfig* const* config) {
50  const su2double matchingVertexTol = 1e-12; // 1um^2
51 
52  const int nProcessor = size;
53  const auto nMarkerInt = config[donorZone]->GetMarker_n_ZoneInterface() / 2;
54  const auto nDim = donor_geometry->GetnDim();
55 
56  Buffer_Receive_nVertex_Donor = new unsigned long[nProcessor];
57 
58  /*--- Make space for donor info. ---*/
59 
60  targetVertices.resize(config[targetZone]->GetnMarker_All());
61 
62  /*--- Init stats. ---*/
63  MaxDistance = 0.0;
64  ErrorCounter = 0;
65  unsigned long nGlobalVertexTarget = 0;
66 
67  /*--- Cycle over nMarkersInt interface to determine communication pattern. ---*/
68 
69  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
70  /* High level procedure:
71  * - Loop through vertices of the target grid;
72  * - Find nearest element;
73  * - Compute and set the transfer coefficients.
74  */
75 
76  /*--- On the donor side: find the tag of the boundary sharing the interface. ---*/
77  const auto markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
78 
79  /*--- On the target side: find the tag of the boundary sharing the interface. ---*/
80  const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
81 
82  /*--- Checks if the zone contains the interface, if not continue to the next step. ---*/
83  if (!CheckInterfaceBoundary(markDonor, markTarget)) continue;
84 
85  unsigned long nVertexDonor = 0, nVertexTarget = 0;
86  if (markDonor != -1) nVertexDonor = donor_geometry->GetnVertex(markDonor);
87  if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex(markTarget);
88 
89  /*--- Sets MaxLocalVertex_Donor, Buffer_Receive_nVertex_Donor. ---*/
90  Determine_ArraySize(markDonor, markTarget, nVertexDonor, nDim);
91 
92  const auto nGlobalVertexDonor =
93  accumulate(Buffer_Receive_nVertex_Donor, Buffer_Receive_nVertex_Donor + nProcessor, 0ul);
94 
95  Buffer_Send_Coord.resize(MaxLocalVertex_Donor, nDim);
96  Buffer_Send_GlobalPoint.resize(MaxLocalVertex_Donor);
97  Buffer_Receive_Coord.resize(nProcessor * MaxLocalVertex_Donor, nDim);
98  Buffer_Receive_GlobalPoint.resize(nProcessor * MaxLocalVertex_Donor);
99 
100  /*--- Collect coordinates and global point indices. ---*/
101  Collect_VertexInfo(markDonor, markTarget, nVertexDonor, nDim);
102  if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
103 
104  /*--- Compress the vertex information, and build a map of global point to "compressed
105  * index" to then reconstruct the donor elements in local index space. ---*/
106 
107  su2activematrix donorCoord(nGlobalVertexDonor, nDim);
108  vector<long> donorPoint(nGlobalVertexDonor);
109  vector<int> donorProc(nGlobalVertexDonor);
110  unordered_map<long, unsigned long> globalToLocalMap;
111 
112  auto iCount = 0ul;
113  for (int iProcessor = 0; iProcessor < nProcessor; ++iProcessor) {
114  auto offset = iProcessor * MaxLocalVertex_Donor;
115  for (auto iVertex = 0ul; iVertex < Buffer_Receive_nVertex_Donor[iProcessor]; ++iVertex) {
116  for (int iDim = 0; iDim < nDim; ++iDim) donorCoord(iCount, iDim) = Buffer_Receive_Coord(offset + iVertex, iDim);
117  donorPoint[iCount] = Buffer_Receive_GlobalPoint[offset + iVertex];
118  donorProc[iCount] = iProcessor;
119  assert((globalToLocalMap.count(donorPoint[iCount]) == 0) && "Duplicate donor point found.");
120  globalToLocalMap[donorPoint[iCount]] = iCount;
121  ++iCount;
122  }
123  }
124  assert((iCount == nGlobalVertexDonor) && "Global donor point count mismatch.");
125 
126  /*--- Collect donor element (face) information. ---*/
127 
128  vector<unsigned long> allNumElem;
129  vector<unsigned short> elemNumNodes;
130  su2matrix<long> elemIdxNodes;
131 
132  const auto nGlobalElemDonor = Collect_ElementInfo(markDonor, nDim, true, allNumElem, elemNumNodes, elemIdxNodes);
133 
134  /*--- Map the node to "local" indices and create a list of connected elements for each vertex. ---*/
135 
136  vector<vector<unsigned> > vertexElements(nGlobalVertexDonor);
137 
138  for (auto iElem = 0u; iElem < nGlobalElemDonor; ++iElem) {
139  const auto nNode = elemNumNodes[iElem];
140 
141  for (auto iNode = 0u; iNode < nNode; ++iNode) {
142  assert(globalToLocalMap.count(elemIdxNodes(iElem, iNode)) &&
143  "Unknown donor point referenced by donor element.");
144  const auto iVertex = globalToLocalMap.at(elemIdxNodes(iElem, iNode));
145  elemIdxNodes(iElem, iNode) = iVertex;
146 
147  vertexElements[iVertex].push_back(iElem);
148  }
149  }
150 
151  /*--- Compute transfer coefficients for each target point. ---*/
152  SU2_OMP_PARALLEL {
153  su2double maxDist = 0.0;
154  unsigned long errorCount = 0, totalCount = 0;
155 
156  SU2_OMP_FOR_DYN(roundUpDiv(nVertexTarget, 2 * omp_get_max_threads()))
157  for (auto iVertexTarget = 0u; iVertexTarget < nVertexTarget; ++iVertexTarget) {
158  auto& target_vertex = targetVertices[markTarget][iVertexTarget];
159  const auto iPoint = target_geometry->vertex[markTarget][iVertexTarget]->GetNode();
160 
161  if (!target_geometry->nodes->GetDomain(iPoint)) continue;
162  totalCount += 1;
163 
164  /*--- Coordinates of the target point. ---*/
165  const su2double* coord_i = target_geometry->nodes->GetCoord(iPoint);
166 
167  /*--- Find the closest donor vertex. ---*/
168  su2double minDist = 1e9;
169  unsigned iClosestVertex = 0;
170  for (auto iVertexDonor = 0u; iVertexDonor < nGlobalVertexDonor; ++iVertexDonor) {
171  su2double d = SquaredDistance(nDim, coord_i, donorCoord[iVertexDonor]);
172  if (d < minDist) {
173  minDist = d;
174  iClosestVertex = iVertexDonor;
175  }
176  }
177 
178  if (minDist < matchingVertexTol) {
179  /*--- Perfect match. ---*/
180  target_vertex.resize(1);
181  target_vertex.coefficient[0] = 1.0;
182  target_vertex.globalPoint[0] = donorPoint[iClosestVertex];
183  target_vertex.processor[0] = donorProc[iClosestVertex];
184  continue;
185  }
186 
187  /*--- Evaluate interpolation for the elements connected to the closest vertex. ---*/
188  DonorInfo donor;
189  donor.error = 2;
190  donor.distance = 1e9;
191  for (auto iElem : vertexElements[iClosestVertex]) {
192  /*--- Fetch element info. ---*/
193  DonorInfo candidate;
194  candidate.iElem = iElem;
195  const auto nNode = elemNumNodes[iElem];
196  su2double coords[4][3] = {{0.0}};
197 
198  for (auto iNode = 0u; iNode < nNode; ++iNode) {
199  const auto iVertex = elemIdxNodes(iElem, iNode);
200  for (auto iDim = 0u; iDim < nDim; ++iDim) coords[iNode][iDim] = donorCoord(iVertex, iDim);
201  }
202 
203  /*--- Compute the interpolation coefficients. ---*/
204  switch (nNode) {
205  case 2:
206  candidate.error = LineIsoparameters(coords, coord_i, candidate.isoparams);
207  break;
208  case 3:
209  candidate.error = TriangleIsoparameters(coords, coord_i, candidate.isoparams);
210  break;
211  case 4:
212  candidate.error = QuadrilateralIsoparameters(coords, coord_i, candidate.isoparams);
213  break;
214  }
215 
216  /*--- Evaluate distance from target to final mapped point. ---*/
217  su2double finalCoord[3] = {0.0};
218  for (auto iDim = 0u; iDim < nDim; ++iDim)
219  for (auto iNode = 0u; iNode < nNode; ++iNode)
220  finalCoord[iDim] += coords[iNode][iDim] * candidate.isoparams[iNode];
221 
222  candidate.distance = Distance(nDim, coord_i, finalCoord);
223 
224  /*--- Detect a very bad candidate (NaN). ---*/
225  if (candidate.distance != candidate.distance) continue;
226 
227  /*--- Check if the candidate is an improvement, update donor if so. ---*/
228  if (candidate < donor) donor = candidate;
229  }
230 
231  if (donor.error > 1) SU2_MPI::Error("Isoparametric interpolation failed, NaN detected.", CURRENT_FUNCTION);
232 
233  errorCount += donor.error;
234  maxDist = max(maxDist, donor.distance);
235 
236  const auto nNode = elemNumNodes[donor.iElem];
237 
238  target_vertex.resize(nNode);
239 
240  for (auto iNode = 0u; iNode < nNode; ++iNode) {
241  const auto iVertex = elemIdxNodes(donor.iElem, iNode);
242  target_vertex.coefficient[iNode] = donor.isoparams[iNode];
243  target_vertex.globalPoint[iNode] = donorPoint[iVertex];
244  target_vertex.processor[iNode] = donorProc[iVertex];
245  }
246  }
247  END_SU2_OMP_FOR
248  SU2_OMP_CRITICAL {
249  MaxDistance = max(MaxDistance, maxDist);
250  ErrorCounter += errorCount;
251  nGlobalVertexTarget += totalCount;
252  }
253  END_SU2_OMP_CRITICAL
254  }
255  END_SU2_OMP_PARALLEL
256 
257  } // end nMarkerInt loop
258 
259  /*--- Final reduction of statistics. ---*/
260  su2double tmp = MaxDistance;
261  unsigned long tmp1 = ErrorCounter, tmp2 = nGlobalVertexTarget;
262  SU2_MPI::Allreduce(&tmp, &MaxDistance, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
263  SU2_MPI::Allreduce(&tmp1, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
264  SU2_MPI::Allreduce(&tmp2, &nGlobalVertexTarget, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
265 
266  ErrorRate = 100 * su2double(ErrorCounter) / nGlobalVertexTarget;
267 }

269 int CIsoparametric::LineIsoparameters(const su2double X[][3], const su2double* xj, su2double* isoparams) {
270  /*--- Project the target point onto the line. ---*/
271 
272  su2double normal[2] = {0.0};
273  LineNormal(X, normal);
274  su2double xprj[2] = {0.0};
275  PointPlaneProjection<su2double, 2>(xj, X[0], normal, xprj);
276 
277  su2double l01 = Distance(2, X[0], X[1]);
278  su2double l0j = Distance(2, X[0], xprj);
279  su2double lj1 = Distance(2, xprj, X[1]);
280 
281  /*--- Detect out of bounds point. ---*/
282 
283  const int outOfBounds = (l0j + lj1) > (2 * l01);
284 
285  isoparams[0] = max(-0.5, min(lj1 / l01, 1.5));
286  isoparams[1] = 1.0 - isoparams[0];
287 
288  return outOfBounds;
289 }

291 int CIsoparametric::TriangleIsoparameters(const su2double X[][3], const su2double* xj, su2double* isoparams) {
292  /*--- The isoparameters are the solution to the determined system X^T * isoparams = xj.
293  * For which we solve the normal equations to avoid divisions by zero.
294  * This is consistent with the shape functions of the linear triangular element. ---*/
295 
296  const su2double extrapTol = -0.5;
297 
298  /*--- Project the target point onto the triangle. ---*/
299 
300  su2double normal[3] = {0.0}, xproj[3] = {0.0};
301  TriangleNormal(X, normal);
302  PointPlaneProjection<su2double, 3>(xj, X[0], normal, xproj);
303 
304  su2double A[3][3] = {{0.0}}; // = X*X^T
305  for (int i = 0; i < 3; ++i) {
306  for (int j = 0; j < 3; ++j)
307  for (int k = 0; k < 3; ++k) A[i][j] += X[i][k] * X[j][k];
308 
309  isoparams[i] = 0.0; // use isoparams as rhs
310  for (int k = 0; k < 3; ++k) isoparams[i] += X[i][k] * xproj[k];
311  }
312 
313  /*--- Solve system by in-place Gaussian elimination without pivoting. ---*/
314 
315  /*--- Transform system in Upper Matrix. ---*/
316  for (int i = 1; i < 3; ++i) {
317  for (int j = 0; j < i; ++j) {
318  su2double w = A[i][j] / A[j][j];
319  for (int k = j; k < 3; ++k) A[i][k] -= w * A[j][k];
320  isoparams[i] -= w * isoparams[j];
321  }
322  }
323  /*--- Backwards substitution. ---*/
324  for (int i = 2; i >= 0; --i) {
325  for (int j = i + 1; j < 3; ++j) isoparams[i] -= A[i][j] * isoparams[j];
326  isoparams[i] /= A[i][i];
327  }
328 
329  /*--- Detect out of bounds point. ---*/
330  const int outOfBounds = (isoparams[0] < extrapTol) || (isoparams[1] < extrapTol) || (isoparams[2] < extrapTol);
331 
332  /*--- Mitigation. ---*/
333  if (outOfBounds) {
334  /*--- Clamp. ---*/
335  su2double sum = 0.0;
336  for (int i = 0; i < 3; ++i) {
337  isoparams[i] = max(isoparams[i], extrapTol);
338  sum += isoparams[i];
339  }
340  /*--- Enforce unit sum. ---*/
341  for (int i = 0; i < 3; ++i) isoparams[i] /= sum;
342  }
343 
344  return outOfBounds;
345 }

347 int CIsoparametric::QuadrilateralIsoparameters(const su2double X[][3], const su2double* xj, su2double* isoparams) {
348  /*--- The isoparameters are the shape functions (Ni) evaluated at xj, for that we need
349  * the corresponding Xi and Eta, which are obtained by solving the overdetermined
350  * nonlinear system r = xj - X^T * Ni(Xi,Eta) = 0 via the modified Marquardt method.
351  * It is not necessary to project the point as minimizing ||r|| is equivalent. ---*/
352 
353  const su2double extrapTol = 3.0;
354 
355  constexpr int NITER = 20;
356  const su2double tol = 1e-10, lambda = 0.05;
357 
358  su2double Xi = 0.0, Eta = 0.0, eps;
359 
360  /*--- Finding Xi and Eta is a "third order" effect that we do not
361  * differentiate (also because we need to iterate). ---*/
362 
363  const bool wasActive = AD::BeginPassive();
364 
365  for (int iter = 0; iter < NITER; ++iter) {
366  /*--- Evaluate the residual. ---*/
367  su2double r[3] = {xj[0], xj[1], xj[2]};
368  su2double Ni[4] = {0.0};
369  CQUAD4::ShapeFunctions(Xi, Eta, Ni);
370  for (int i = 0; i < 3; ++i)
371  for (int j = 0; j < 4; ++j) r[i] -= X[j][i] * Ni[j];
372 
373  /*--- Evaluate the residual Jacobian. ---*/
374  su2double dNi[4][2] = {{0.0}};
375  CQUAD4::ShapeFunctionJacobian(Xi, Eta, dNi);
376 
377  su2double jac[3][2] = {{0.0}};
378  for (int i = 0; i < 3; ++i)
379  for (int j = 0; j < 2; ++j)
380  for (int k = 0; k < 4; ++k) jac[i][j] -= X[k][i] * dNi[k][j];
381 
382  /*--- Compute the correction (normal equations and Cramer's rule). ---*/
383  su2double A[2][2] = {{0.0}}, b[2] = {0.0};
384  for (int i = 0; i < 2; ++i) {
385  for (int j = i; j < 2; ++j)
386  for (int k = 0; k < 3; ++k) A[i][j] += jac[k][i] * jac[k][j];
387 
388  A[i][i] *= (1.0 + lambda);
389 
390  for (int k = 0; k < 3; ++k) b[i] += jac[k][i] * r[k];
391  }
392  A[1][0] = A[0][1];
393 
394  su2double detA = 1.0 / (A[0][0] * A[1][1] - A[0][1] * A[1][0]);
395  su2double dXi = (b[0] * A[1][1] - b[1] * A[0][1]) * detA;
396  su2double dEta = (A[0][0] * b[1] - A[1][0] * b[0]) * detA;
397  Xi -= dXi;
398  Eta -= dEta;
399 
400  eps = fabs(dXi) + fabs(dEta);
401  if (eps < tol) break;
402  }
403 
404  AD::EndPassive(wasActive);
405 
406  int outOfBounds = 0;
407 
408  if (eps > 0.01) {
409  /*--- Iteration diverged, hard fallback. ---*/
410  Xi = Eta = 0.0;
411  outOfBounds = 1;
412  } else {
413  /*--- Check bounds. ---*/
414  outOfBounds = (fabs(Xi) > extrapTol) || (fabs(Eta) > extrapTol);
415 
416  /*--- Mitigate by clamping coordinates. ---*/
417  if (outOfBounds) {
418  Xi = max(-extrapTol, min(Xi, extrapTol));
419  Eta = max(-extrapTol, min(Eta, extrapTol));
420  }
421  }
422 
423  /*--- Evaluate isoparameters at final Xi and Eta. ---*/
424  CQUAD4::ShapeFunctions(Xi, Eta, isoparams);
425 
426  return outOfBounds;
427 }
