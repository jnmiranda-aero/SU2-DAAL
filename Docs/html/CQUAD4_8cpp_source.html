30CQUAD4::CQUAD4() : CElementWithKnownSizes<NGAUSS, NNODE, NDIM>() {
31 /*--- Gauss coordinates and weights ---*/
32
33 su2double oneOnSqrt3 = 0.577350269189626;
34
35 GaussCoord[0][0] = -oneOnSqrt3;
36 GaussCoord[0][1] = -oneOnSqrt3;
37 GaussWeight(0) = 1.0;
38 GaussCoord[1][0] = oneOnSqrt3;
39 GaussCoord[1][1] = -oneOnSqrt3;
40 GaussWeight(1) = 1.0;
41 GaussCoord[2][0] = oneOnSqrt3;
42 GaussCoord[2][1] = oneOnSqrt3;
43 GaussWeight(2) = 1.0;
44 GaussCoord[3][0] = -oneOnSqrt3;
45 GaussCoord[3][1] = oneOnSqrt3;
46 GaussWeight(3) = 1.0;
47
48 /*--- Store the values of the shape functions and their derivatives ---*/
49
50 unsigned short iNode, iGauss, jGauss;
51 su2double Xi, Eta;
52
53 for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
54 Xi = GaussCoord[iGauss][0];
55 Eta = GaussCoord[iGauss][1];
56
57 su2double Ni[4] = {0.0};
58 ShapeFunctions(Xi, Eta, Ni);
59
60 for (jGauss = 0; jGauss < NGAUSS; jGauss++) GaussPoint[iGauss].SetNi(Ni[jGauss], jGauss);
61
62 /*--- dN/d xi, dN/d eta ---*/
63
64 ShapeFunctionJacobian(Xi, Eta, dNiXj[iGauss]);
65 }
66
67 /*--- Store the extrapolation functions (used to compute nodal stresses) ---*/
68
69 su2double ExtrapCoord[4][2], sqrt3 = 1.732050807568877;
70 ;
71
72 ExtrapCoord[0][0] = -sqrt3;
73 ExtrapCoord[0][1] = -sqrt3;
74 ExtrapCoord[1][0] = sqrt3;
75 ExtrapCoord[1][1] = -sqrt3;
76 ExtrapCoord[2][0] = sqrt3;
77 ExtrapCoord[2][1] = sqrt3;
78 ExtrapCoord[3][0] = -sqrt3;
79 ExtrapCoord[3][1] = sqrt3;
80
81 for (iNode = 0; iNode < NNODE; iNode++) {
82 Xi = ExtrapCoord[iNode][0];
83 Eta = ExtrapCoord[iNode][1];
84
85 NodalExtrap[iNode][0] = 0.25 * (1.0 - Xi) * (1.0 - Eta);
86 NodalExtrap[iNode][1] = 0.25 * (1.0 + Xi) * (1.0 - Eta);
87 NodalExtrap[iNode][2] = 0.25 * (1.0 + Xi) * (1.0 + Eta);
88 NodalExtrap[iNode][3] = 0.25 * (1.0 - Xi) * (1.0 + Eta);
89 }
90}

92su2double CQUAD4::ComputeArea(const FrameType mode) const {
93 unsigned short iDim;
94 su2double a[2] = {0.0, 0.0}, b[2] = {0.0, 0.0};
95 su2double Area = 0.0;
96
97 /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
98 for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed)---*/
99 const su2activematrix& Coord = (mode == REFERENCE) ? RefCoord : CurrentCoord;
100
101 for (iDim = 0; iDim < NDIM; iDim++) {
102 a[iDim] = Coord[0][iDim] - Coord[2][iDim];
103 b[iDim] = Coord[1][iDim] - Coord[2][iDim];
104 }
105
106 Area = 0.5 * fabs(a[0] * b[1] - a[1] * b[0]);
107
108 for (iDim = 0; iDim < NDIM; iDim++) {
109 a[iDim] = Coord[0][iDim] - Coord[3][iDim];
110 b[iDim] = Coord[2][iDim] - Coord[3][iDim];
111 }
112
113 Area += 0.5 * fabs(a[0] * b[1] - a[1] * b[0]);
114
115 return Area;
116}
