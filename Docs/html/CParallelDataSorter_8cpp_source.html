95 void CParallelDataSorter::SortOutputData() {
96 
97  const int VARS_PER_POINT = GlobalField_Counter;
98 
99  /*--- Allocate the memory that we need for receiving the conn
100  values and then cue up the non-blocking receives. Note that
101  we do not include our own rank in the communications. We will
102  directly copy our own data later. ---*/
103 
104  vector<unsigned long> idRecv(nPoint_Recv[size], 0);
105 
106 #ifdef HAVE_MPI
107  /*--- NOTE: This function calls MPI routines directly, instead of via SU2_MPI::,
108  * because it communicates passivedoubles and not AD types. This avoids some
109  * creative C++ to communicate AD types and then convert to passive. ---*/
110 
111  /*--- We need double the number of messages to send both the conn. and the global IDs. ---*/
112 
113  auto send_req = new MPI_Request[2*nSends];
114  auto recv_req = new MPI_Request[2*nRecvs];
115 
116  unsigned long iMessage = 0;
117  for (int ii=0; ii<size; ii++) {
118  if ((ii != rank) && (nPoint_Recv[ii+1] > nPoint_Recv[ii])) {
119  int ll = VARS_PER_POINT*nPoint_Recv[ii];
120  int kk = nPoint_Recv[ii+1] - nPoint_Recv[ii];
121  int count = VARS_PER_POINT*kk;
122  int source = ii;
123  int tag = ii + 1;
124  MPI_Irecv(&(dataBuffer[ll]), count, MPI_DOUBLE, source, tag,
125  SU2_MPI::GetComm(), &(recv_req[iMessage]));
126  iMessage++;
127  }
128  }
129 
130  /*--- Launch the non-blocking sends of the connectivity. ---*/
131 
132  iMessage = 0;
133  for (int ii=0; ii<size; ii++) {
134  if ((ii != rank) && (nPoint_Send[ii+1] > nPoint_Send[ii])) {
135  int ll = VARS_PER_POINT*nPoint_Send[ii];
136  int kk = nPoint_Send[ii+1] - nPoint_Send[ii];
137  int count = VARS_PER_POINT*kk;
138  int dest = ii;
139  int tag = rank + 1;
140  MPI_Isend(&(connSend[ll]), count, MPI_DOUBLE, dest, tag,
141  SU2_MPI::GetComm(), &(send_req[iMessage]));
142  iMessage++;
143  }
144  }
145 
146  /*--- Repeat the process to communicate the global IDs. ---*/
147 
148  iMessage = 0;
149  for (int ii=0; ii<size; ii++) {
150  if ((ii != rank) && (nPoint_Recv[ii+1] > nPoint_Recv[ii])) {
151  int ll = nPoint_Recv[ii];
152  int kk = nPoint_Recv[ii+1] - nPoint_Recv[ii];
153  int count = kk;
154  int source = ii;
155  int tag = ii + 1;
156  MPI_Irecv(&(idRecv[ll]), count, MPI_UNSIGNED_LONG, source, tag,
157  SU2_MPI::GetComm(), &(recv_req[iMessage+nRecvs]));
158  iMessage++;
159  }
160  }
161 
162  /*--- Launch the non-blocking sends of the global IDs. ---*/
163 
164  iMessage = 0;
165  for (int ii=0; ii<size; ii++) {
166  if ((ii != rank) && (nPoint_Send[ii+1] > nPoint_Send[ii])) {
167  int ll = nPoint_Send[ii];
168  int kk = nPoint_Send[ii+1] - nPoint_Send[ii];
169  int count = kk;
170  int dest = ii;
171  int tag = rank + 1;
172  MPI_Isend(&(idSend[ll]), count, MPI_UNSIGNED_LONG, dest, tag,
173  SU2_MPI::GetComm(), &(send_req[iMessage+nSends]));
174  iMessage++;
175  }
176  }
177 #endif
178 
179  /*--- Copy my own rank's data into the recv buffer directly. ---*/
180 
181  int mm = VARS_PER_POINT*nPoint_Recv[rank];
182  int ll = VARS_PER_POINT*nPoint_Send[rank];
183  int kk = VARS_PER_POINT*nPoint_Send[rank+1];
184 
185  for (int nn=ll; nn<kk; nn++, mm++) dataBuffer[mm] = connSend[nn];
186 
187  mm = nPoint_Recv[rank];
188  ll = nPoint_Send[rank];
189  kk = nPoint_Send[rank+1];
190 
191  for (int nn=ll; nn<kk; nn++, mm++) idRecv[mm] = idSend[nn];
192 
193  /*--- Wait for the non-blocking sends and recvs to complete. ---*/
194 
195 #ifdef HAVE_MPI
196  MPI_Status status;
197  int ind;
198 
199  int number = 2*nSends;
200  for (int ii = 0; ii < number; ii++)
201  MPI_Waitany(number, send_req, &ind, &status);
202 
203  number = 2*nRecvs;
204  for (int ii = 0; ii < number; ii++)
205  MPI_Waitany(number, recv_req, &ind, &status);
206 
207  delete [] send_req;
208  delete [] recv_req;
209 #endif
210 
211  /*--- Reorder the data in the buffer. ---*/
212 
213  vector<passivedouble> tmpBuffer(nPoint_Recv[size]);
214 
215  for (int jj = 0; jj < VARS_PER_POINT; jj++){
216  for (int ii = 0; ii < nPoint_Recv[size]; ii++){
217  tmpBuffer[idRecv[ii]] = dataBuffer[ii*VARS_PER_POINT+jj];
218  }
219  for (int ii = 0; ii < nPoint_Recv[size]; ii++){
220  dataBuffer[ii*VARS_PER_POINT+jj] = tmpBuffer[ii];
221  }
222  }
223 
224  /*--- Store the total number of local points my rank has for
225  the current section after completing the communications. ---*/
226 
227  nPoints = nPoint_Recv[size];
228 
229  /*--- Reduce the total number of points we will write in the output files. ---*/
230 
231  SU2_MPI::Allreduce(&nPoints, &nPointsGlobal, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
232 
233 }

235 void CParallelDataSorter::PrepareSendBuffers(std::vector<unsigned long>& globalID){
236 
237  unsigned long iPoint;
238  unsigned short iProcessor;
239 
240  int VARS_PER_POINT = GlobalField_Counter;
241 
242  /*--- We start with the grid nodes distributed across all procs with
243  no particular ordering assumed. We need to loop through our local partition
244  and decide how many nodes we must send to each other rank in order to
245  have all nodes sorted according to a linear partitioning of the grid
246  nodes, i.e., rank 0 holds the first ~ nGlobalPoint()/nProcessors nodes.
247  First, initialize a counter and flag. ---*/
248 
249  for (iPoint = 0; iPoint < nLocalPointsBeforeSort; iPoint++ ) {
250 
251  iProcessor = linearPartitioner.GetRankContainingIndex(globalID[iPoint]);
252 
253  /*--- If we have not visited this node yet, increment our
254  number of elements that must be sent to a particular proc. ---*/
255 
256  nPoint_Send[iProcessor+1]++;
257  }
258 
259  /*--- Communicate the number of nodes to be sent/recv'd amongst
260  all processors. After this communication, each proc knows how
261  many cells it will receive from each other processor. ---*/
262 
263  SU2_MPI::Alltoall(&(nPoint_Send[1]), 1, MPI_INT,
264  &(nPoint_Recv[1]), 1, MPI_INT, SU2_MPI::GetComm());
265 
266  /*--- Prepare to send coordinates. First check how many
267  messages we will be sending and receiving. Here we also put
268  the counters into cumulative storage format to make the
269  communications simpler. ---*/
270 
271  nSends = 0; nRecvs = 0;
272 
273  for (int ii = 0; ii < size; ii++) {
274  if ((ii != rank) && (nPoint_Send[ii+1] > 0)) nSends++;
275  if ((ii != rank) && (nPoint_Recv[ii+1] > 0)) nRecvs++;
276 
277  nPoint_Send[ii+1] += nPoint_Send[ii];
278  nPoint_Recv[ii+1] += nPoint_Recv[ii];
279  }
280 
281  /*--- Allocate memory to hold the connectivity that we are
282  sending. ---*/
283 
284  connSend = new passivedouble[VARS_PER_POINT*nPoint_Send[size]] ();
285 
286  /*--- Allocate the data buffer to hold the sorted data. We have to make it large enough
287  * to hold passivedoubles and su2doubles ---*/
288 
289  dataBuffer = new passivedouble[VARS_PER_POINT*nPoint_Recv[size]] ();
290 
291  /*--- Allocate arrays for sending the global ID. ---*/
292 
293  idSend = new unsigned long[nPoint_Send[size]]();
294 
295  /*--- Create an index variable to keep track of our index
296  positions as we load up the send buffer. ---*/
297 
298  vector<unsigned long> index(size), idIndex(size);
299 
300  for (int ii=0; ii < size; ii++) {
301  index[ii] = VARS_PER_POINT*nPoint_Send[ii];
302  idIndex[ii] = nPoint_Send[ii];
303  }
304 
305  Index = new unsigned long[nLocalPointsBeforeSort]();
306 
307  /*--- Loop through our elements and load the elems and their
308  additional data that we will send to the other procs. ---*/
309 
310  for (iPoint = 0; iPoint < nLocalPointsBeforeSort; iPoint++) {
311 
312  iProcessor = linearPartitioner.GetRankContainingIndex(globalID[iPoint]);
313 
314  /*--- Load the global ID (minus offset) for sorting the
315  points once they all reach the correct processor. ---*/
316 
317  unsigned long nn = idIndex[iProcessor];
318  idSend[nn] = globalID[iPoint] - linearPartitioner.GetFirstIndexOnRank(iProcessor);
319 
320  /*--- Store the index this point has in the send buffer ---*/
321 
322  Index[iPoint] = index[iProcessor];
323 
324  /*--- Increment the index by the message length ---*/
325 
326  index[iProcessor] += VARS_PER_POINT;
327  idIndex[iProcessor]++;
328 
329  }

333 unsigned long CParallelDataSorter::GetElemConnectivity(GEO_TYPE type, unsigned long iElem, unsigned long iNode) const {
334 
335  switch (type) {
336  case LINE:
337  return Conn_Line_Par[iElem*N_POINTS_LINE + iNode];
338  break;
339  case TRIANGLE:
340  return Conn_Tria_Par[iElem*N_POINTS_TRIANGLE + iNode];
341  break;
342  case QUADRILATERAL:
343  return Conn_Quad_Par[iElem*N_POINTS_QUADRILATERAL + iNode];
344  break;
345  case TETRAHEDRON:
346  return Conn_Tetr_Par[iElem*N_POINTS_TETRAHEDRON + iNode];
347  break;
348  case HEXAHEDRON:
349  return Conn_Hexa_Par[iElem*N_POINTS_HEXAHEDRON + iNode];
350  break;
351  case PRISM:
352  return Conn_Pris_Par[iElem*N_POINTS_PRISM + iNode];
353  break;
354  case PYRAMID:
355  return Conn_Pyra_Par[iElem*N_POINTS_PYRAMID + iNode];
356  break;
357  default:
358  break;
359  }
360 
361  SU2_MPI::Error("GEO_TYPE not found", CURRENT_FUNCTION);
362 
363  return 0;
364 }

366 void CParallelDataSorter::SetTotalElements(){
367 
368  /*--- Reduce the total number of cells we will be writing in the output files. ---*/
369 
370  SU2_MPI::Allreduce(nElemPerType.data(), nElemPerTypeGlobal.data(), N_ELEM_TYPES, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
371 
372  nElemGlobal = std::accumulate(nElemPerTypeGlobal.begin(), nElemPerTypeGlobal.end(), 0);
373  nElem = std::accumulate(nElemPerType.begin(), nElemPerType.end(), 0);
374 
375  nConn = 0;
376  nConnGlobal = 0;
377  auto addConnectivitySize = [this](GEO_TYPE elem, unsigned short nPoints){
378  nConn += GetnElem(elem)*nPoints;
379  nConnGlobal += GetnElemGlobal(elem)*nPoints;
380  };
381 
382  addConnectivitySize(LINE, N_POINTS_LINE);
383  addConnectivitySize(TRIANGLE, N_POINTS_TRIANGLE);
384  addConnectivitySize(QUADRILATERAL, N_POINTS_QUADRILATERAL);
385  addConnectivitySize(TETRAHEDRON, N_POINTS_TETRAHEDRON);
386  addConnectivitySize(HEXAHEDRON, N_POINTS_HEXAHEDRON);
387  addConnectivitySize(PRISM, N_POINTS_PRISM);
388  addConnectivitySize(PYRAMID, N_POINTS_PYRAMID);
389 
390  /*--- Communicate the number of total cells/storage that will be
391  written by each rank. After this communication, each proc knows how
392  many cells will be written before its location in the file and the
393  offsets can be correctly set. ---*/
394 
395  nElem_Send[0] = 0; nElemConn_Send[0] = 0;
396  nElem_Cum[0] = 0; nElemConn_Cum[0] = 0;
397  for (int ii=1; ii <= size; ii++) {
398  nElem_Send[ii] = int(nElem);
399  nElemConn_Send[ii] = int(nConn);
400  nElem_Cum[ii] = 0;
401  nElemConn_Cum[ii] = 0;
402  }
403 
404  /*--- Communicate the local counts to all ranks for building offsets. ---*/
405 
406  SU2_MPI::Alltoall(&(nElem_Send[1]), 1, MPI_INT,
407  &(nElem_Cum[1]), 1, MPI_INT, SU2_MPI::GetComm());
408 
409  SU2_MPI::Alltoall(&(nElemConn_Send[1]), 1, MPI_INT,
410  &(nElemConn_Cum[1]), 1, MPI_INT, SU2_MPI::GetComm());
411 
412  /*--- Put the counters into cumulative storage format. ---*/
413 
414  for (int ii = 0; ii < size; ii++) {
415  nElem_Cum[ii+1] += nElem_Cum[ii];
416  nElemConn_Cum[ii+1] += nElemConn_Cum[ii];
417  }
418 
419 }
