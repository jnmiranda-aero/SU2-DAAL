33CFlowCompOutput::CFlowCompOutput(const CConfig *config, unsigned short nDim) : CFlowOutput(config, nDim, false) {
34
35 turb_model = config->GetKind_Turb_Model();
36
37 /*--- Set the default history fields if nothing is set in the config file ---*/
38
39 if (nRequestedHistoryFields == 0){
40 RequestCommonHistory(config->GetTime_Domain());
41 nRequestedHistoryFields = requestedHistoryFields.size();
42 }
43 if (nRequestedScreenFields == 0){
44 if (config->GetTime_Domain()) requestedScreenFields.emplace_back("TIME_ITER");
45 if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
46 requestedScreenFields.emplace_back("INNER_ITER");
47 requestedScreenFields.emplace_back("RMS_DENSITY");
48 requestedScreenFields.emplace_back("RMS_MOMENTUM-X");
49 requestedScreenFields.emplace_back("RMS_MOMENTUM-Y");
50 requestedScreenFields.emplace_back("RMS_ENERGY");
51 nRequestedScreenFields = requestedScreenFields.size();
52 }
53 if (nRequestedVolumeFields == 0){
54 requestedVolumeFields.emplace_back("COORDINATES");
55 requestedVolumeFields.emplace_back("SOLUTION");
56 requestedVolumeFields.emplace_back("PRIMITIVE");
57 nRequestedVolumeFields = requestedVolumeFields.size();
58 }
59
60 if (gridMovement) {
61 auto notFound = requestedVolumeFields.end();
62 if (find(requestedVolumeFields.begin(), notFound, string("GRID_VELOCITY")) == notFound) {
63 requestedVolumeFields.emplace_back("GRID_VELOCITY");
64 nRequestedVolumeFields ++;
65 }
66 }
67
68 stringstream ss;
69 ss << "Zone " << config->GetiZone() << " (Comp. Fluid)";
70 multiZoneHeaderString = ss.str();
71
72 /*--- Set the volume filename --- */
73
74 volumeFilename = config->GetVolume_FileName();
75
76 /*--- Set the surface filename --- */
77
78 surfaceFilename = config->GetSurfCoeff_FileName();
79
80 /*--- Set the restart filename --- */
81
82 restartFilename = config->GetRestart_FileName();
83
84 /*--- Set the default convergence field --- */
85
86 if (convFields.empty() ) convFields.emplace_back("RMS_DENSITY");
87
88 if (config->GetFixed_CL_Mode()) {
89 if (std::find(convFields.begin(), convFields.end(), "LIFT") != convFields.end()) {
90 if (rank == MASTER_NODE)
91 cout<<" Fixed CL: Adding LIFT as Convergence Field to ensure convergence to target CL"<<endl;
92 convFields.emplace_back("LIFT");
93 newFunc.resize(convFields.size());
94 oldFunc.resize(convFields.size());
95 cauchySerie.resize(convFields.size(), vector<su2double>(nCauchy_Elems, 0.0));
96 }
97 }

310void CFlowCompOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
311
312 const auto* Node_Flow = solver[FLOW_SOL]->GetNodes();
313 auto* Node_Geo = geometry->nodes;
314
315 LoadCoordinates(Node_Geo->GetCoord(iPoint), iPoint);
316
317 SetVolumeOutputValue("DENSITY", iPoint, Node_Flow->GetSolution(iPoint, 0));
318 SetVolumeOutputValue("MOMENTUM-X", iPoint, Node_Flow->GetSolution(iPoint, 1));
319 SetVolumeOutputValue("MOMENTUM-Y", iPoint, Node_Flow->GetSolution(iPoint, 2));
320 if (nDim == 3){
321 SetVolumeOutputValue("MOMENTUM-Z", iPoint, Node_Flow->GetSolution(iPoint, 3));
322 SetVolumeOutputValue("ENERGY", iPoint, Node_Flow->GetSolution(iPoint, 4));
323 } else {
324 SetVolumeOutputValue("ENERGY", iPoint, Node_Flow->GetSolution(iPoint, 3));
325 }
326
327 if (gridMovement){
328 SetVolumeOutputValue("GRID_VELOCITY-X", iPoint, Node_Geo->GetGridVel(iPoint)[0]);
329 SetVolumeOutputValue("GRID_VELOCITY-Y", iPoint, Node_Geo->GetGridVel(iPoint)[1]);
330 if (nDim == 3)
331 SetVolumeOutputValue("GRID_VELOCITY-Z", iPoint, Node_Geo->GetGridVel(iPoint)[2]);
332 }
333
334 SetVolumeOutputValue("PRESSURE", iPoint, Node_Flow->GetPressure(iPoint));
335 SetVolumeOutputValue("TEMPERATURE", iPoint, Node_Flow->GetTemperature(iPoint));
336 SetVolumeOutputValue("MACH", iPoint, sqrt(Node_Flow->GetVelocity2(iPoint))/Node_Flow->GetSoundSpeed(iPoint));
337
338 const su2double factor = solver[FLOW_SOL]->GetReferenceDynamicPressure();
339 SetVolumeOutputValue("PRESSURE_COEFF", iPoint, (Node_Flow->GetPressure(iPoint) - solver[FLOW_SOL]->GetPressure_Inf())/factor);
340 SetVolumeOutputValue("VELOCITY-X", iPoint, Node_Flow->GetVelocity(iPoint, 0));
341 SetVolumeOutputValue("VELOCITY-Y", iPoint, Node_Flow->GetVelocity(iPoint, 1));
342 if (nDim == 3){
343 SetVolumeOutputValue("VELOCITY-Z", iPoint, Node_Flow->GetVelocity(iPoint, 2));
344 }
345
346 if(config->GetKind_FluidModel() == DATADRIVEN_FLUID){
347 SetVolumeOutputValue("EXTRAPOLATION", iPoint, Node_Flow->GetDataExtrapolation(iPoint));
348 SetVolumeOutputValue("FLUIDMODEL_NEWTONITER", iPoint, Node_Flow->GetNewtonSolverIterations(iPoint));
349 SetVolumeOutputValue("ENTROPY", iPoint, Node_Flow->GetEntropy(iPoint));
350 }
351
352 if (config->GetKind_Solver() == MAIN_SOLVER::RANS || config->GetKind_Solver() == MAIN_SOLVER::NAVIER_STOKES){
353 SetVolumeOutputValue("LAMINAR_VISCOSITY", iPoint, Node_Flow->GetLaminarViscosity(iPoint));
354 }
355
356 SetVolumeOutputValue("RES_DENSITY", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 0));
357 SetVolumeOutputValue("RES_MOMENTUM-X", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 1));
358 SetVolumeOutputValue("RES_MOMENTUM-Y", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 2));
359 if (nDim == 3){
360 SetVolumeOutputValue("RES_MOMENTUM-Z", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 3));
361 SetVolumeOutputValue("RES_ENERGY", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 4));
362 } else {
363 SetVolumeOutputValue("RES_ENERGY", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, 3));
364 }
365
366 if (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE && config->GetKind_SlopeLimit_Flow() != LIMITER::VAN_ALBADA_EDGE) {
367 SetVolumeOutputValue("LIMITER_VELOCITY-X", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 1));
368 SetVolumeOutputValue("LIMITER_VELOCITY-Y", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 2));
369 if (nDim == 3){
370 SetVolumeOutputValue("LIMITER_VELOCITY-Z", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 3));
371 }
372 SetVolumeOutputValue("LIMITER_PRESSURE", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, nDim+1));
373 SetVolumeOutputValue("LIMITER_DENSITY", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, nDim+2));
374 SetVolumeOutputValue("LIMITER_ENTHALPY", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, nDim+3));
375 }
376
377 if (config->GetKind_RoeLowDiss() != NO_ROELOWDISS){
378 SetVolumeOutputValue("ROE_DISSIPATION", iPoint, Node_Flow->GetRoe_Dissipation(iPoint));
379 }
380
381 LoadVolumeDataScalar(config, solver, geometry, iPoint);
382
383 LoadCommonFVMOutputs(config, geometry, iPoint);
384
385 if (config->GetTime_Domain()) {
386 LoadTimeAveragedData(iPoint, Node_Flow);
387 }
