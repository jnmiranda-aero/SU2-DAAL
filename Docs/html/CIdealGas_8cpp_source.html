30 CIdealGas::CIdealGas(su2double gamma, su2double R, bool CompEntropy) : CFluidModel() {
31  Gamma = gamma;
32  Gamma_Minus_One = Gamma - 1.0;
33  Gas_Constant = R;
34  Cp = Gamma / Gamma_Minus_One * Gas_Constant;
35  Cv = Cp - R;
36 
37  ComputeEntropy = CompEntropy;
38 }

40 void CIdealGas::SetTDState_rhoe(su2double rho, su2double e) {
41  Density = rho;
42  StaticEnergy = e;
43  Pressure = Gamma_Minus_One * Density * StaticEnergy;
44  Temperature = Gamma_Minus_One * StaticEnergy / Gas_Constant;
45  SoundSpeed2 = Gamma * Pressure / Density;
46  dPdrho_e = Gamma_Minus_One * StaticEnergy;
47  dPde_rho = Gamma_Minus_One * Density;
48  dTdrho_e = 0.0;
49  dTde_rho = Gamma_Minus_One / Gas_Constant;
50 
51  if (ComputeEntropy) Entropy = (1.0 / Gamma_Minus_One * log(Temperature) + log(1.0 / Density)) * Gas_Constant;
52 }

67 void CIdealGas::SetTDState_hs(su2double h, su2double s) {
68  su2double T = h * Gamma_Minus_One / Gas_Constant / Gamma;
69  su2double e = h / Gamma;
70  su2double v = exp(-1 / Gamma_Minus_One * log(T) + s / Gas_Constant);
71 
72  SetTDState_rhoe(1 / v, e);
73 }

75 void CIdealGas::SetTDState_Ps(su2double P, su2double s) {
76  su2double T = exp(Gamma_Minus_One / Gamma * (s / Gas_Constant + log(P) - log(Gas_Constant)));
77  su2double rho = P / (T * Gas_Constant);
78 
79  SetTDState_Prho(P, rho);
80 }

82 void CIdealGas::SetTDState_rhoT(su2double rho, su2double T) {

87 void CIdealGas::ComputeDerivativeNRBC_Prho(su2double P, su2double rho) {
88  su2double dPdT_rho, dPdrho_T, dPds_rho;
89 
90  SetTDState_Prho(P, rho);
91 
92  dPdT_rho = Gas_Constant * rho;
93  dPdrho_T = Gas_Constant * Temperature;
94 
95  dhdrho_P = -dPdrho_e / dPde_rho - P / rho / rho;
96  dhdP_rho = 1.0 / dPde_rho + 1.0 / rho;
97  dPds_rho = rho * rho * (SoundSpeed2 - dPdrho_T) / dPdT_rho;
98  dsdP_rho = 1.0 / dPds_rho;
99  dsdrho_P = -SoundSpeed2 / dPds_rho;
100 }
