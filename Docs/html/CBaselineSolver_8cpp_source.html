86  /*--- Retrieve filename from config ---*/
87 
88  if (config->GetContinuous_Adjoint() || config->GetDiscrete_Adjoint()) {
89  filename = config->GetSolution_AdjFileName();

94 
95 
96  /*--- Read only the number of variables in the restart file. ---*/
97

99

101 
102  filename = config->GetFilename(filename, ".dat", config->GetTimeIter());
103 
104  char fname[100];
105  strcpy(fname, filename.c_str());
106  int nVar_Buf = 5;
107  int var_buf[5];
108 
109 #ifndef HAVE_MPI
110 
111  /*--- Serial binary input. ---*/
112 
113  FILE *fhw;
114  fhw = fopen(fname,"rb");
115  size_t ret;
116 
117  /*--- Error check for opening the file. ---*/
118

121  }
122 
123  /*--- First, read the number of variables and points. ---*/
124

126  if (ret != (unsigned long)nVar_Buf) {
127  SU2_MPI::Error("Error reading restart file.", CURRENT_FUNCTION);
128  }
129 
130  /*--- Check that this is an SU2 binary file. SU2 binary files
131  have the hex representation of "SU2" as the first int in the file. ---*/
132 
133  if (var_buf[0] != 535532) {
134  SU2_MPI::Error(string("File ") + string(fname) + string(" is not a binary SU2 restart file.\n") +
135  string("SU2 reads/writes binary restart files by default.\n") +
136  string("Note that backward compatibility for ASCII restart files is\n") +

141 
142  fclose(fhw);
143 
144  /*--- Set the number of variables, one per field in the
145  restart file (without including the PointID) ---*/
146 
147  nVar = var_buf[1];
148 #else
149 
150  /*--- Parallel binary input using MPI I/O. ---*/
151 
152  MPI_File fhw;
153  int ierr;
154  MPI_Offset disp;
155  unsigned short iVar;
156  unsigned long index, iChar;
157  string field_buf;
158  char str_buf[CGNS_STRING_SIZE];
159 
160  /*--- All ranks open the file using MPI. ---*/
161 
162  ierr = MPI_File_open(SU2_MPI::GetComm(), fname, MPI_MODE_RDONLY, MPI_INFO_NULL, &fhw);
163 
164  /*--- Error check opening the file. ---*/
165 
166  if (ierr) {

168  }
169 
170  /*--- First, read the number of variables and points (i.e., cols and rows),
171  which we will need in order to read the file later. Also, read the
172  variable string names here. Only the master rank reads the header. ---*/
173 
174  if (rank == MASTER_NODE) {
175  MPI_File_read(fhw, var_buf, nVar_Buf, MPI_INT, MPI_STATUS_IGNORE);
176  }
177 
178  /*--- Broadcast the number of variables to all procs and store more clearly. ---*/
179 
180  SU2_MPI::Bcast(var_buf, nVar_Buf, MPI_INT, MASTER_NODE, SU2_MPI::GetComm());
181 
182  /*--- Check that this is an SU2 binary file. SU2 binary files
183  have the hex representation of "SU2" as the first int in the file. ---*/
184 
185  if (var_buf[0] != 535532) {
186  SU2_MPI::Error(string("File ") + string(fname) + string(" is not a binary SU2 restart file.\n") +
187  string("SU2 reads/writes binary restart files by default.\n") +
188  string("Note that backward compatibility for ASCII restart files is\n") +
189  string("possible with the READ_BINARY_RESTART option."), CURRENT_FUNCTION);
190  }
191 
192 
193 
194  /*--- Set the number of variables, one per field in the
195  restart file (without including the PointID) ---*/
196 
197  nVar = var_buf[1];
198 
199  /*--- Read the variable names from the file. Note that we are adopting a
200  fixed length of 33 for the string length to match with CGNS. This is
201  needed for when we read the strings later. ---*/
202 
203  char *mpi_str_buf = new char[nVar*CGNS_STRING_SIZE];
204  if (rank == MASTER_NODE) {
205  disp = nVar_Buf*sizeof(int);
206  MPI_File_read_at(fhw, disp, mpi_str_buf, nVar*CGNS_STRING_SIZE,
207  MPI_CHAR, MPI_STATUS_IGNORE);
208  }
209 
210  /*--- Broadcast the string names of the variables. ---*/
211 
212  SU2_MPI::Bcast(mpi_str_buf, nVar*CGNS_STRING_SIZE, MPI_CHAR,
213  MASTER_NODE, SU2_MPI::GetComm());
214 
215  fields.emplace_back("Point_ID");
216 
217  for (iVar = 0; iVar < nVar; iVar++) {
218  index = iVar*CGNS_STRING_SIZE;
219  field_buf.append("\"");
220  for (iChar = 0; iChar < (unsigned long)CGNS_STRING_SIZE; iChar++) {
221  str_buf[iChar] = mpi_str_buf[index + iChar];
222  }
223  field_buf.append(str_buf);
224  field_buf.append("\"");
225  fields.emplace_back(field_buf.c_str());
226  field_buf.clear();

228

230 
231  MPI_File_close(&fhw);
232 
233 #endif
234  } else {
235 
236  /*--- Multizone problems require the number of the zone to be appended. ---*/
237

239

241 
242  char fname[100];

246 #ifndef HAVE_MPI

259

269

272  string("SU2 reads/writes binary restart files by default.\n") +

279 #else

282

285 
286  /*--- All ranks open the file using MPI. ---*/

288  ierr = MPI_File_open(SU2_MPI::GetComm(), fname, MPI_MODE_RDONLY, MPI_INFO_NULL, &fhw);
289 
290  /*--- Error check opening the file. ---*/

292  if (ierr) {

294  }
295 
296  /*--- Have the master attempt to read the magic number. ---*/
297 
298  if (rank == MASTER_NODE)

301  /*--- Broadcast the number of variables to all procs and store clearly. ---*/
302 
303  SU2_MPI::Bcast(&magic_number, 1, MPI_INT, MASTER_NODE, SU2_MPI::GetComm());
304 
305  /*--- Check that this is an SU2 binary file. SU2 binary files
306  have the hex representation of "SU2" as the first int in the file. ---*/
307 
308  if (magic_number == 535532) {
309  SU2_MPI::Error(string("File ") + string(fname) + string(" is a binary SU2 restart file, expected ASCII.\n") +
310  string("SU2 reads/writes binary restart files by default.\n") +
311  string("Note that backward compatibility for ASCII restart files is\n") +
312  string("possible with the READ_BINARY_RESTART option."), CURRENT_FUNCTION);
313  }
314 
315  MPI_File_close(&fhw);
316 
317 #endif
318 
319  /*--- Open the restart file ---*/
320 
321  restart_file.open(filename.data(), ios::in);
322 
323  /*--- In case there is no restart file ---*/
324 
325  if (restart_file.fail()) {

327  }
328 
329  /*--- Identify the number of fields (and names) in the restart file ---*/
330 
331  getline (restart_file, text_line);
332 
333  fields = PrintingToolbox::split(text_line, ',');
334 
335  for (unsigned short iField = 0; iField < fields.size(); iField++){
336  PrintingToolbox::trim(fields[iField]);
337  }
338 
339  /*--- Close the file (the solution date is read later). ---*/
340 
341  restart_file.close();
342 
343  /*--- Set the number of variables, one per field in the
344  restart file (without including the PointID) ---*/
345 
346  nVar = fields.size() - 1;
347 
348  }

352 void CBaselineSolver::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter, bool val_update_geo) {
353 
354  /*--- Restart the solution from file information ---*/
355 
356  string filename;
357  unsigned long index;
358  unsigned short iDim, iVar;
359  bool adjoint = ( config->GetContinuous_Adjoint() || config->GetDiscrete_Adjoint() );
360  unsigned short iInst = config->GetiInst();
361  bool steady_restart = config->GetSteadyRestart();
362  TURB_MODEL turb_model = config->GetKind_Turb_Model();
363 
364  auto *Coord = new su2double [nDim];
365  for (iDim = 0; iDim < nDim; iDim++)
366  Coord[iDim] = 0.0;
367 
368  /*--- Skip coordinates ---*/
369 
370  unsigned short skipVars = geometry[iInst]->GetnDim();
371 
372  /*--- Retrieve filename from config ---*/
373 
374  if (adjoint) {
375  filename = config->GetSolution_AdjFileName();
376  filename = config->GetObjFunc_Extension(filename);
377  } else {
378  filename = config->GetSolution_FileName();
379  }
380 
381  filename = config->GetFilename(filename, "", val_iter);
382 
383  /*--- Output the file name to the console. ---*/
384 
385  if (rank == MASTER_NODE)
386  cout << "Reading and storing the solution from " << filename
387  << "." << endl;
388 
389  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
390 
391  if (config->GetRead_Binary_Restart()) {
392  Read_SU2_Restart_Binary(geometry[iInst], config, filename);
393  } else {
394  Read_SU2_Restart_ASCII(geometry[iInst], config, filename);
395  }
396 
397  int counter = 0;
398  long iPoint_Local = 0; unsigned long iPoint_Global = 0;
399 
400  /*--- Load data from the restart into correct containers. ---*/
401 
402  for (iPoint_Global = 0; iPoint_Global < geometry[iInst]->GetGlobal_nPointDomain(); iPoint_Global++ ) {
403 
404  /*--- Retrieve local index. If this node from the restart file lives
405  on the current processor, we will load and instantiate the vars. ---*/
406 
407  iPoint_Local = geometry[iInst]->GetGlobal_to_Local_Point(iPoint_Global);
408 
409  if (iPoint_Local > -1) {
410 
411  /*--- We need to store this point's data, so jump to the correct
412  offset in the buffer of data from the restart file and load it. ---*/
413 
414  index = counter*Restart_Vars[1];
415  for (iVar = 0; iVar < nVar; iVar++) Solution[iVar] = Restart_Data[index+iVar];
416  nodes->SetSolution(iPoint_Local,Solution);
417 
418  /*--- For dynamic meshes, read in and store the
419  grid coordinates and grid velocities for each node. ---*/
420 
421  if (dynamic_grid && val_update_geo) {
422 
423  /*--- First, remove any variables for the turbulence model that
424  appear in the restart file before the grid velocities. ---*/
425 
426  if (turb_model == TURB_MODEL::SA) {
427  index++;
428  } else if (turb_model == TURB_MODEL::SST) {
429  index+=2;
430  }
431 
432  /*--- Read in the next 2 or 3 variables which are the grid velocities ---*/
433  /*--- If we are restarting the solution from a previously computed static calculation (no grid movement) ---*/
434  /*--- the grid velocities are set to 0. This is useful for FSI computations ---*/
435 
436  su2double GridVel[3] = {0.0,0.0,0.0};
437  if (!steady_restart) {
438 
439  /*--- Rewind the index to retrieve the Coords. ---*/
440  index = counter*Restart_Vars[1];
441  for (iDim = 0; iDim < nDim; iDim++) { Coord[iDim] = Restart_Data[index+iDim]; }
442 
443  /*--- Move the index forward to get the grid velocities. ---*/
444  index = counter*Restart_Vars[1] + skipVars + nVar;
445  for (iDim = 0; iDim < nDim; iDim++) { GridVel[iDim] = Restart_Data[index+iDim]; }
446  }
447 
448  for (iDim = 0; iDim < nDim; iDim++) {
449  geometry[iInst]->nodes->SetCoord(iPoint_Local, iDim, Coord[iDim]);
450  geometry[iInst]->nodes->SetGridVel(iPoint_Local, iDim, GridVel[iDim]);
451  }
452  }
453 
454  /*--- Increment the overall counter for how many points have been loaded. ---*/
455  counter++;
456  }
457 
458  }
459 
460  /*--- MPI solution ---*/
461 
462  InitiateComms(geometry[iInst], config, MPI_QUANTITIES::SOLUTION);
463  CompleteComms(geometry[iInst], config, MPI_QUANTITIES::SOLUTION);
464 
465  /*--- Update the geometry for flows on dynamic meshes ---*/
466 
467  if (dynamic_grid && val_update_geo) {
468 
469  /*--- Communicate the new coordinates and grid velocities at the halos ---*/
470 
471  geometry[iInst]->InitiateComms(geometry[iInst], config, MPI_QUANTITIES::COORDINATES);
472  geometry[iInst]->CompleteComms(geometry[iInst], config, MPI_QUANTITIES::COORDINATES);
473 
474  geometry[iInst]->InitiateComms(geometry[iInst], config, MPI_QUANTITIES::GRID_VELOCITY);
475  geometry[iInst]->CompleteComms(geometry[iInst], config, MPI_QUANTITIES::GRID_VELOCITY);
476 
477  }
478 
479  delete [] Coord;
480 
481  /*--- Delete the class memory that is used to load the restart. ---*/
482 
483  Restart_Vars = decltype(Restart_Vars){};
484  Restart_Data = decltype(Restart_Data){};
485 }
