31 std::unique_ptr<CADTElemClass> CMeshFEM_DG::ComputeViscousWallADT(const CConfig* config) const {
32  /*--------------------------------------------------------------------------*/
33  /*--- Step 1: Create the coordinates and connectivity of the linear ---*/
34  /*--- subelements of the local boundaries that must be taken ---*/
35  /*--- into account in the wall distance computation. ---*/
36  /*--------------------------------------------------------------------------*/
37 
38  /* Initialize an array for the mesh points, which eventually contains the
39  mapping from the local nodes to the number used in the connectivity of the
40  local boundary faces. However, in a first pass it is an indicator whether
41  or not a mesh point is on a local wall boundary. */
42  vector<unsigned long> meshToSurface(meshPoints.size(), 0);
43 
44  /* Define the vectors for the connectivity of the local linear subelements,
45  the element ID's, the element type and marker ID's. */
46  vector<unsigned long> surfaceConn;
47  vector<unsigned long> elemIDs;
48  vector<unsigned short> VTK_TypeElem;
49  vector<unsigned short> markerIDs;
50 
51  /* Loop over the boundary markers. */
52  for (unsigned short iMarker = 0; iMarker < boundaries.size(); ++iMarker) {
53  if (!boundaries[iMarker].periodicBoundary) {
54  /* Check for a viscous wall. */
55  if ((config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX) ||
56  (config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL)) {
57  /* Loop over the surface elements of this marker. */
58  const vector<CSurfaceElementFEM>& surfElem = boundaries[iMarker].surfElem;
59  for (unsigned long i = 0; i < surfElem.size(); ++i) {
60  /* Set the flag of the mesh points on this surface to true. */
61  for (unsigned short j = 0; j < surfElem[i].nDOFsGrid; ++j) meshToSurface[surfElem[i].DOFsGridFace[j]] = 1;
62 
63  /* Determine the necessary data from the corresponding standard face,
64  such as the number of linear subfaces, the number of DOFs per
65  linear subface and the corresponding local connectivity. */
66  const unsigned short ind = surfElem[i].indStandardElement;
67  const unsigned short VTK_Type = standardBoundaryFacesGrid[ind].GetVTK_Type();
68  const unsigned short nSubFaces = standardBoundaryFacesGrid[ind].GetNSubFaces();
69  const unsigned short nDOFsPerFace = standardBoundaryFacesGrid[ind].GetNDOFsPerSubFace();
70  const unsigned short* connSubFaces = standardBoundaryFacesGrid[ind].GetSubFaceConn();
71 
72  /* Loop over the number of subfaces and store the required data. */
73  unsigned short ii = 0;
74  for (unsigned short j = 0; j < nSubFaces; ++j) {
75  markerIDs.push_back(iMarker);
76  VTK_TypeElem.push_back(VTK_Type);
77  elemIDs.push_back(i);
78 
79  for (unsigned short k = 0; k < nDOFsPerFace; ++k, ++ii)
80  surfaceConn.push_back(surfElem[i].DOFsGridFace[connSubFaces[ii]]);
81  }
82  }
83  }
84  }
85  }
86 
87  /*--- Create the coordinates of the local points on the viscous surfaces and
88  create the final version of the mapping from all volume points to the
89  points on the viscous surfaces. ---*/
90  vector<su2double> surfaceCoor;
91  unsigned long nVertex_SolidWall = 0;
92 
93  for (unsigned long i = 0; i < meshPoints.size(); ++i) {
94  if (meshToSurface[i]) {
95  meshToSurface[i] = nVertex_SolidWall++;
96 
97  for (unsigned short k = 0; k < nDim; ++k) surfaceCoor.push_back(meshPoints[i].coor[k]);
98  }
99  }
100 
101  /*--- Change the surface connectivity, such that it corresponds to
102  the entries in surfaceCoor rather than in meshPoints. ---*/
103  for (unsigned long i = 0; i < surfaceConn.size(); ++i) surfaceConn[i] = meshToSurface[surfaceConn[i]];
104 
105  /*--------------------------------------------------------------------------*/
106  /*--- Step 2: Build the ADT, which is an ADT of bounding boxes of the ---*/
107  /*--- surface elements. A nearest point search does not give ---*/
108  /*--- accurate results, especially not for the integration ---*/
109  /*--- points of the elements close to a wall boundary. ---*/
110  /*--------------------------------------------------------------------------*/
111 
112  /* Build the ADT. */
113  std::unique_ptr<CADTElemClass> WallADT(
114  new CADTElemClass(nDim, surfaceCoor, surfaceConn, VTK_TypeElem, markerIDs, elemIDs, true));
115 
116  return WallADT;
117 }

122 void CMeshFEM_DG::SetWallDistance(su2double val) {
123  for (unsigned long l = 0; l < nVolElemOwned; ++l) {
124  /* Get the required data from the corresponding standard element. */
125  const unsigned short ind = volElem[l].indStandardElement;
126  const unsigned short nInt = standardElementsGrid[ind].GetNIntegration();
127 
128  /* Store the number of solDOFS a bit easier. */
129  const unsigned short nDOFsSol = volElem[l].nDOFsSol;
130 
131  /* Allocate the memory for the wall distance of this element. */
132  volElem[l].wallDistance.resize(nInt, val);
133  volElem[l].wallDistanceSolDOFs.resize(nDOFsSol, val);
134  }
135 
136  for (unsigned long l = 0; l < matchingFaces.size(); ++l) {
137  /* Get the required data from the corresponding standard element. */
138  const unsigned short ind = matchingFaces[l].indStandardElement;
139  const unsigned short nInt = standardMatchingFacesGrid[ind].GetNIntegration();
140 
141  /* Allocate the memory for the wall distance for this matching face. */
142  matchingFaces[l].wallDistance.resize(nInt, val);
143  }
144 
145  for (unsigned short iMarker = 0; iMarker < boundaries.size(); ++iMarker) {
146  if (!boundaries[iMarker].periodicBoundary) {
147  /* Loop over the boundary faces and determine the wall distances
148  in the integration points. */
149  vector<CSurfaceElementFEM>& surfElem = boundaries[iMarker].surfElem;
150  for (unsigned long l = 0; l < surfElem.size(); ++l) {
151  /* Get the required data from the corresponding standard element. */
152  const unsigned short ind = surfElem[l].indStandardElement;
153  const unsigned short nInt = standardBoundaryFacesGrid[ind].GetNIntegration();
154 
155  /* Allocate the memory for the wall distance for this boundary face. */
156  surfElem[l].wallDistance.resize(nInt, val);
157  }
158  }
159  }

162 void CMeshFEM_DG::SetWallDistance(CADTElemClass* WallADT, const CConfig* config, unsigned short iZone) {
163  /*--------------------------------------------------------------------------*/
164  /*--- Step 3: Determine the wall distance of the integration points of ---*/
165  /*--- locally owned volume elements. ---*/
166  /*--------------------------------------------------------------------------*/
167 
168  /*--- Loop over the owned elements to compute the wall distance
169  in the integration points. ---*/
170  for (unsigned long l = 0; l < nVolElemOwned; ++l) {
171  /* Get the required data from the corresponding standard element. */
172  const unsigned short ind = volElem[l].indStandardElement;
173  const unsigned short nInt = standardElementsGrid[ind].GetNIntegration();
174 
175  /* Allocate the memory for the wall distance of this element. */
176  volElem[l].wallDistance.resize(nInt);
177 
178  if (!WallADT->IsEmpty()) {
179  /*--- The tree is not empty. Loop over the integration points
180  and determine the wall distance. ---*/
181  for (unsigned short i = 0; i < nInt; ++i) {
182  const su2double* coor = volElem[l].coorIntegrationPoints.data() + i * nDim;
183  unsigned short markerID;
184  unsigned long elemID;
185  int rankID;
186  su2double dist;
187  WallADT->DetermineNearestElement(coor, dist, markerID, elemID, rankID);
188 
189  volElem[l].wallDistance[i] = dist;
190  }
191  }
192  }
193 
194  /*--------------------------------------------------------------------------*/
195  /*--- Step 4: Determine the wall distance of the solution DOFs of ---*/
196  /*--- locally owned volume elements. ---*/
197  /*--------------------------------------------------------------------------*/
198 
199  /*--- Loop over the owned elements to compute the wall distance
200  in the integration points. ---*/
201  for (unsigned long l = 0; l < nVolElemOwned; ++l) {
202  /* Store the number of solDOFS a bit easier. */
203  const unsigned short nDOFsSol = volElem[l].nDOFsSol;
204 
205  /* Allocate the memory for the wall distance of the solution DOFs. */
206  volElem[l].wallDistanceSolDOFs.resize(nDOFsSol);
207 
208  if (!WallADT->IsEmpty()) {
209  /*--- The tree is not empty. Loop over the solution DOFs
210  and determine the wall distance. ---*/
211  for (unsigned short i = 0; i < nDOFsSol; ++i) {
212  const su2double* coor = volElem[l].coorSolDOFs.data() + i * nDim;
213  unsigned short markerID;
214  unsigned long elemID;
215  int rankID;
216  su2double dist;
217  WallADT->DetermineNearestElement(coor, dist, markerID, elemID, rankID);
218 
219  volElem[l].wallDistanceSolDOFs[i] = dist;
220  }
221  }
222  }
223 
224  /*--------------------------------------------------------------------------*/
225  /*--- Step 5: Determine the wall distance of the integration points of ---*/
226  /*--- the internal matching faces. ---*/
227  /*--------------------------------------------------------------------------*/
228 
229  for (unsigned long l = 0; l < matchingFaces.size(); ++l) {
230  /* Get the required data from the corresponding standard element. */
231  const unsigned short ind = matchingFaces[l].indStandardElement;
232  const unsigned short nInt = standardMatchingFacesGrid[ind].GetNIntegration();
233 
234  /* Allocate the memory for the wall distance for this matching face. */
235  matchingFaces[l].wallDistance.resize(nInt);
236 
237  if (!WallADT->IsEmpty()) {
238  /*--- The tree is not empty. Loop over the integration points
239  and determine the wall distance. */
240  for (unsigned short i = 0; i < nInt; ++i) {
241  const su2double* coor = matchingFaces[l].coorIntegrationPoints.data() + i * nDim;
242  unsigned short markerID;
243  unsigned long elemID;
244  int rankID;
245  su2double dist;
246  WallADT->DetermineNearestElement(coor, dist, markerID, elemID, rankID);
247 
248  matchingFaces[l].wallDistance[i] = dist;
249  }
250  }
251  }
252 
253  /*--------------------------------------------------------------------------*/
254  /*--- Step 6: Determine the wall distance of the integration points of ---*/
255  /*--- locally owned boundary faces. ---*/
256  /*--------------------------------------------------------------------------*/
257 
258  /*--- Loop over the boundary markers. Make sure to exclude the periodic
259  boundaries, because these are not physical. ---*/
260  for (unsigned short iMarker = 0; iMarker < boundaries.size(); ++iMarker) {
261  if (!boundaries[iMarker].periodicBoundary) {
262  /* Determine whether or not this is a viscous wall boundary. */
263  const bool viscousWall =
264  config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX || config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL;
265 
266  /* Loop over the boundary faces and determine the wall distances
267  in the integration points. */
268  vector<CSurfaceElementFEM>& surfElem = boundaries[iMarker].surfElem;
269  for (unsigned long l = 0; l < surfElem.size(); ++l) {
270  /* Get the required data from the corresponding standard element. */
271  const unsigned short ind = surfElem[l].indStandardElement;
272  const unsigned short nInt = standardBoundaryFacesGrid[ind].GetNIntegration();
273 
274  /* Allocate the memory for the wall distance for this boundary face. */
275  surfElem[l].wallDistance.resize(nInt);
276 
277  /* Check for an empty tree or a viscous wall.
278  In those case the wall distance is set to zero. */
279  if (viscousWall) {
280  /* Wall distance must be set to zero. */
281  for (unsigned short i = 0; i < nInt; ++i) surfElem[l].wallDistance[i] = 0.0;
282  } else if (!WallADT->IsEmpty()) {
283  /*--- Not a viscous wall boundary, while viscous walls are present.
284  The distance must be computed. Loop over the integration points
285  and do so. ---*/
286  for (unsigned short i = 0; i < nInt; ++i) {
287  const su2double* coor = surfElem[l].coorIntegrationPoints.data() + i * nDim;
288  unsigned short markerID;
289  unsigned long elemID;
290  int rankID;
291  su2double dist;
292  WallADT->DetermineNearestElement(coor, dist, markerID, elemID, rankID);
293 
294  surfElem[l].wallDistance[i] = dist;
295  }
296  }
297  }
298  }
