44 void CSTLFileWriter::WriteData(string val_filename){
45 
46  /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
47  val_filename.append(fileExt);
48 
49  /*--- This Write_Data routine has 3 major parts where the first two are transfered in external functions:
50  1. Prerequisite info: The parallel data-sorter distributes nodes of the primal mesh onto the processes
51  (i.e. this step is only important for parallel excecution) and not elements. For the STL file the connectivity
52  of the elements is broken/lost especially on the node-borders between processes. This information is
53  re-processed in this first (cumbersome) step.
54  2. The coordinate data for each node in a Triangle is successively written into a local array.
55  Quadrilateral surface elements are split into 2 triangles. All local arrays are gathered on the MASTER_NODE.
56  3. The MASTER_NODE only writes the data into a .stl file.
57  ---*/
58 
59  /*--- 1. Re-process the element connectivity information and store that appropriatly such that
60  this information can be accessed in step 2. It is copied from CTecplotBinaryFileWriter.cpp:WriteData() l.162 and mildly modified. ---*/
61  ReprocessElementConnectivity();
62 
63  /*--- 2. Load the coordinate data succesively in an array. 3coordinates*3nodes = 9 consecutive
64  su2doubles form a triangle ---*/
65  GatherCoordData();
66 
67  /*--- 3. The master rank alone writes the surface STL file. ---*/
68  unsigned short iPoint, iVar;
69  unsigned long iProcessor, nProcessor = size, iElem, index;
70  ofstream Surf_file;
71 
72  if (rank == MASTER_NODE) {
73 
74  /*--- For information how an ASCII .stl file is structured: https://en.wikipedia.org/wiki/STL_(file_format) ---*/
75  /*--- Open the STL file and write the header line. ---*/
76  Surf_file.precision(6);
77 
78  Surf_file.open(val_filename.c_str(), ios::out);
79  Surf_file << "solid SU2_output" << endl;
80 
81  /*--- Loop through all of the collected data and write each node's coordinate values. ---*/
82  for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
83  for (iElem = 0; iElem < buffRecvTriaCount[iProcessor]; iElem++) { // loops over nLocalTriaAll
84 
85  /*--- Write the coordinate data for each node in a triangle. ---*/
86  /*--- Every tested software recomputes the normal anyway,
87  such that this arbitray face normal does not matter. ---*/
88  Surf_file << "facet normal " << 1 << " " << 2 << " " << 3 << endl;
89  Surf_file << " outer loop" << endl; // 4 leading whitespaces
90 
91  for(iPoint = 0; iPoint < N_POINTS_TRIANGLE; iPoint++) {
92  Surf_file << " vertex"; // 8 leading whitespaces
93  index = iProcessor*max_nLocalTriaAll*N_POINTS_TRIANGLE*3 + iElem*N_POINTS_TRIANGLE*3 + iPoint*3;
94 
95  for (iVar = 0; iVar < 3; iVar++) {
96  Surf_file << " " << buffRecvCoords[index + iVar];
97  }
98  Surf_file << endl;
99  }//iPoint
100  Surf_file << " endloop" << endl; // 4 leading whitespaces
101  Surf_file << "endfacet" << endl;
102  }//iElem
103  }//iProcessor
104 
105  /*--- Close the file. ---*/
106  Surf_file << "endsolid SU2_output" << endl;
107  Surf_file.close();
108 
109  }
110 
111  /*--- Free temporary memory. ---*/
112  delete [] buffRecvCoords;
113  delete [] buffRecvTriaCount;
114 }

117 void CSTLFileWriter::ReprocessElementConnectivity(){
118 
119 
120  const vector<string>& fieldNames = dataSorter->GetFieldNames();
121 
122  /*--- Re-process the element connectivity information and store that appropriatly such that
123  this information can be accessed in step 2. It is copied from CTecplotBinaryFileWriter.cpp +162 and mildly modified. ---*/
124 
125  unsigned long nParallel_Tria = dataSorter->GetnElem(TRIANGLE),
126  nParallel_Quad = dataSorter->GetnElem(QUADRILATERAL);
127  unsigned short iVar;
128 
129  num_nodes_to_receive.resize(size, 0);
130  values_to_receive_displacements.resize(size);
131  halo_nodes.clear();
132  sorted_halo_nodes.clear();
133 
134  /* Gather a list of nodes we refer to but are not outputting, because they are not present on this rank. */
135  for (unsigned long i = 0; i < nParallel_Tria * N_POINTS_TRIANGLE; ++i)
136  if (dataSorter->FindProcessor(dataSorter->GetElemConnectivity(TRIANGLE, 0, i)-1) != rank)
137  halo_nodes.push_back(dataSorter->GetElemConnectivity(TRIANGLE, 0, i)-1);
138 
139  for (unsigned long i = 0; i < nParallel_Quad * N_POINTS_QUADRILATERAL; ++i)
140  if (dataSorter->FindProcessor(dataSorter->GetElemConnectivity(QUADRILATERAL, 0, i)-1) != rank)
141  halo_nodes.push_back(dataSorter->GetElemConnectivity(QUADRILATERAL, 0, i)-1);
142 
143  /* Sorted list of halo nodes for this MPI rank. */
144  sorted_halo_nodes.assign(halo_nodes.begin(), halo_nodes.end());
145  sort(sorted_halo_nodes.begin(), sorted_halo_nodes.end());
146 
147  /*--- We effectively tack the halo nodes onto the end of the node list for this partition.
148  TecIO will later replace them with references to nodes in neighboring partitions. */
149  num_halo_nodes = sorted_halo_nodes.size();
150  vector<long> neighbor_partitions(max<unsigned long>(1, num_halo_nodes));
151  vector<long> neighbor_nodes(max<unsigned long>(1, num_halo_nodes));
152 
153  for(unsigned long i = 0; i < num_halo_nodes; ++i) {
154  int owning_rank = dataSorter->FindProcessor(sorted_halo_nodes[i]);
155  unsigned long node_number = sorted_halo_nodes[i] - dataSorter->GetNodeBegin(owning_rank);
156  neighbor_partitions[i] = owning_rank;
157  neighbor_nodes[i] = static_cast<long>(node_number);
158  }
159 
160  /* Gather halo node data. First, tell each rank how many nodes' worth of data we need from them. */
161  for (unsigned long i = 0; i < num_halo_nodes; ++i)
162  ++num_nodes_to_receive[neighbor_partitions[i]];
163  num_nodes_to_send.resize(size);
164  SU2_MPI::Alltoall(num_nodes_to_receive.data(), 1, MPI_INT, num_nodes_to_send.data(), 1, MPI_INT, SU2_MPI::GetComm());
165 
166  /* Now send the global node numbers whose data we need,
167  and receive the same from all other ranks.
168  Each rank has globally consecutive node numbers,
169  so we can just parcel out sorted_halo_nodes for send. */
170  nodes_to_send_displacements.resize(size);
171  nodes_to_receive_displacements.resize(size);
172  nodes_to_send_displacements[0] = 0;
173  nodes_to_receive_displacements[0] = 0;
174 
175  for(int iRank = 1; iRank < size; ++iRank) {
176  nodes_to_send_displacements[iRank] = nodes_to_send_displacements[iRank - 1] + num_nodes_to_send[iRank - 1];
177  nodes_to_receive_displacements[iRank] = nodes_to_receive_displacements[iRank - 1] + num_nodes_to_receive[iRank - 1];
178  }
179 
180  int total_num_nodes_to_send = nodes_to_send_displacements[size - 1] + num_nodes_to_send[size - 1];
181 
182  nodes_to_send.resize(max<int>(1, total_num_nodes_to_send));
183  /* The terminology gets a bit confusing here. We're sending the node numbers
184  (sorted_halo_nodes) whose data we need to receive, and receiving
185  lists of nodes whose data we need to send. */
186  if (sorted_halo_nodes.empty()) sorted_halo_nodes.resize(1); /* Avoid crash. */
187  SU2_MPI::Alltoallv(sorted_halo_nodes.data(), num_nodes_to_receive.data(), nodes_to_receive_displacements.data(), MPI_UNSIGNED_LONG,
188  nodes_to_send.data(), num_nodes_to_send.data(), nodes_to_send_displacements.data(), MPI_UNSIGNED_LONG,
189  SU2_MPI::GetComm());
190 
191  /* Now actually send and receive the data */
192  data_to_send.resize(max<unsigned long>(1, total_num_nodes_to_send * fieldNames.size()));
193  halo_var_data.resize(max<unsigned long>(1, fieldNames.size() * num_halo_nodes));
194  num_values_to_send.resize(size);
195  values_to_send_displacements.resize(size);
196  num_values_to_receive.resize(size);
197  unsigned long index = 0;
198 
199  for(int iRank = 0; iRank < size; ++iRank) {
200 
201  /* We send and receive GlobalField_Counter values per node. */
202  num_values_to_send[iRank] = num_nodes_to_send[iRank] * fieldNames.size();
203  values_to_send_displacements[iRank] = nodes_to_send_displacements[iRank] * fieldNames.size();
204  num_values_to_receive[iRank] = num_nodes_to_receive[iRank] * fieldNames.size();
205  values_to_receive_displacements[iRank] = nodes_to_receive_displacements[iRank] * fieldNames.size();
206 
207  for(iVar = 0; iVar < fieldNames.size(); ++iVar) {
208  for(int iNode = 0; iNode < num_nodes_to_send[iRank]; ++iNode) {
209  unsigned long node_offset = nodes_to_send[nodes_to_send_displacements[iRank] + iNode] - dataSorter->GetNodeBegin(rank);
210  data_to_send[index++] = SU2_TYPE::GetValue(dataSorter->GetData(iVar,node_offset));
211  }
212  }
213 
214  }
215 
216  SU2_MPI::Alltoallv(data_to_send.data(), num_values_to_send.data(), values_to_send_displacements.data(), MPI_DOUBLE,
217  halo_var_data.data(), num_values_to_receive.data(), values_to_receive_displacements.data(), MPI_DOUBLE,
218  SU2_MPI::GetComm());
219 }

222 void CSTLFileWriter::GatherCoordData(){
223 
224  /*--- 2. Load the coordinate data succesively in an array. 3coordinates*3nodes = 9 consecutive
225  su2doubles form a triangle ---*/
226 
227  /*--- Routine to write the surface STL files (ASCII). We
228  assume here that, as an ASCII file, it is safer to merge the
229  surface data onto the master rank for writing for 2 reasons:
230  (a) as a surface file, the amount of data should be much less
231  than the volume solution, and (b) writing ASCII files in parallel
232  requires serializing the IO calls with barriers, which ruins
233  the performance at moderate to high rank counts. ---*/
234 
235  unsigned long nProcessor = size;
236 
237  /*--- Find the max number of surface vertices among all
238  partitions so we can set up buffers. The master node will handle
239  the writing of the STL file after gathering all of the data. ---*/
240 
241  unsigned long nLocalTria = dataSorter->GetnElem(TRIANGLE),
242  nLocalQuad = dataSorter->GetnElem(QUADRILATERAL),
243  nLocalTriaAll = nLocalTria + nLocalQuad*2; // Quad splitted into 2 tris
244 
245  if (rank == MASTER_NODE) buffRecvTriaCount = new unsigned long[nProcessor];
246 
247  /*--- Communicate the maximum of local triangles on any process to each partition and the number of local vertices on each partition
248  to the master node with collective calls. ---*/
249 
250  SU2_MPI::Allreduce(&nLocalTriaAll, &max_nLocalTriaAll, 1,
251  MPI_UNSIGNED_LONG, MPI_MAX, SU2_MPI::GetComm());
252 
253 
254  SU2_MPI::Gather(&nLocalTriaAll , 1, MPI_UNSIGNED_LONG,
255  buffRecvTriaCount, 1, MPI_UNSIGNED_LONG,
256  MASTER_NODE, SU2_MPI::GetComm());
257 
258  /*--- Allocate buffer for send/recv of the coordinate data. Only the master rank allocates buffers for the recv. ---*/
259  buffSendCoords = new su2double[max_nLocalTriaAll*N_POINTS_TRIANGLE*3]; /* Triangle has 3 Points with 3 coords each */
260  if (rank == MASTER_NODE)
261  buffRecvCoords = new su2double[nProcessor*max_nLocalTriaAll*N_POINTS_TRIANGLE*3];
262 
263  StoreCoordData(TRIANGLE, nLocalTria, 0ul);
264  StoreCoordData(QUADRILATERAL, nLocalQuad, nLocalTria*N_POINTS_TRIANGLE*3);
265 
266  /*--- Collective comms of the solution data and global IDs. ---*/
267  SU2_MPI::Gather(buffSendCoords, static_cast<int>(max_nLocalTriaAll*N_POINTS_TRIANGLE*3), MPI_DOUBLE,
268  buffRecvCoords, static_cast<int>(max_nLocalTriaAll*N_POINTS_TRIANGLE*3), MPI_DOUBLE,
269  MASTER_NODE, SU2_MPI::GetComm());
270 
271  /*--- Free temporary memory. ---*/
272  delete [] buffSendCoords;
273 }

276 void CSTLFileWriter::StoreCoordData(enum GEO_TYPE elemType,
277  unsigned long nLocalElements,
278  unsigned long startIndex) {
279 
280  unsigned long globalNodeNumber, localNodeNumber, iElem, index = startIndex;
281 
282  /*--- Create a node list that defines in which order the element points form a triangle. ---*/
283  vector<unsigned short> nodeList;
284  switch (elemType) {
285  case TRIANGLE:
286  nodeList.insert(nodeList.end(), {0,1,2} ); break;
287  case QUADRILATERAL:
288  /*--- Quads split into two Triangles. Assumes either clock- or counterclockwise rotation ---*/
289  nodeList.insert(nodeList.end(), {0,1,3, 1,2,3} ); break;
290  default:
291  SU2_MPI::Error("Unsupported element type for the STL writer." , CURRENT_FUNCTION); break;
292  }
293 
294  /*--- Write Triangle-Point-Coordinates subsequently in a local send buffer. ---*/
295  for (iElem = 0; iElem < nLocalElements; iElem++) {
296  for (auto node : nodeList) {
297  globalNodeNumber = dataSorter->GetElemConnectivity(elemType, iElem, node) - 1;
298  localNodeNumber = globalNodeNumber - dataSorter->GetNodeBegin(rank);
299 
300  for (auto iVar = 0; iVar < 3; iVar++){
301  if (dataSorter->FindProcessor(globalNodeNumber) == rank) {
302  buffSendCoords[index] = dataSorter->GetData(iVar, localNodeNumber);
303  } else {
304  buffSendCoords[index] = GetHaloNodeValue(globalNodeNumber, iVar);
305  }
306 
307  index++;
308  }//iVar
309  }//node
310  }//iElem
311 }

314 passivedouble CSTLFileWriter::GetHaloNodeValue(unsigned long global_node_number, unsigned short iVar) {
315 
316  auto it = lower_bound(sorted_halo_nodes.begin(), sorted_halo_nodes.end(), global_node_number);
317  int offset = distance(sorted_halo_nodes.begin(), it);
318  int id = 0;
319 
320  for (int iRank = 0; iRank < size; ++iRank) {
321  for (int i = 0; i < num_nodes_to_receive[iRank]; i++){
322  int displ = values_to_receive_displacements[iRank] + num_nodes_to_receive[iRank]*iVar;
323  if (id == offset)
324  return halo_var_data[displ+i];
325  id++;
326  }
327  }
328 
329  SU2_MPI::Error("STL File-Writer: Halo node not found.", CURRENT_FUNCTION);
330  return 0.0;
