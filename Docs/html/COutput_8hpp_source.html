194 struct CustomOutput {
195 /*--- First level of parsing the syntax "name : type{func}[markers];". ---*/
196 std::string name;

198 std::string func;

202 mel::ExpressionTree<passivedouble> expression;

205 static constexpr long PROBE_NOT_SETUP = -2;
206 static constexpr long PROBE_NOT_OWNED = -1;
207 long iPoint = PROBE_NOT_SETUP;
208
209 /*--- The symbols (strings) are associated with an integer index for efficiency. For evaluation this index
210 is passed to a functor that returns the value associated with the symbol. This functor is an input to "eval()"
211 and needs to be generated on-the-fly for each point. The functor approach is more generic than a pointer, for
212 example it allows wrapping the access to multiple solvers. The interpretation of these indices is dictated by
213 the functor used in eval, for example indices may be established as 32 * solver_idx + variable_idx.
214 The parts of the code that assign and interpret indices need to be in sync. ---*/
215 std::vector<unsigned long> varIndices;
216
217 /*--- Offset between varIndices of different solvers (see above). Power of 2 to make decoding faster. ---*/
218 static constexpr unsigned long MAX_VARS_PER_SOLVER = 32;
219
220 /*--- Arbitrary number to indicate that a string did not match a variable. ---*/
221 static constexpr unsigned long NOT_A_VARIABLE = MAX_SOLS * MAX_VARS_PER_SOLVER;
222
223 /*--- Other outputs can be referenced in expressions, e.g. to compute variance.
224 We store pointers to the required outputs to speed-up access. ---*/
225 std::vector<const su2double*> otherOutputs;
226
227 /*--- For discrete adjoint we may need to skip some expressions because there is one output class
228 for the primal solver and one for the discrete adjoint (each with different variables). ---*/
229 bool skip = false;
230
231 /*--- For evaluation, "vars" is a functor (i.e. has operator()) that returns the value of a variable at a given
232 point. For example, it can be a wrapper to the primitives pointer, in which case varIndices needs to be setup
233 with primitive indices. ---*/
234 template <class Variables>
235 su2double Eval(const Variables& vars) const {
236 return mel::Eval<su2double>(expression, [&](int iSymbol) {return vars(varIndices[iSymbol]);});
237 }
238 };
