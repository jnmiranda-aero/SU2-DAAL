30 CDiscAdjFEASolver::CDiscAdjFEASolver(CGeometry *geometry, CConfig *config, CSolver *direct_solver,
31  unsigned short Kind_Solver, unsigned short iMesh) : CSolver() {
32 
33  adjoint = true;
34 
35  unsigned long iPoint;
36 
37  const bool dynamic = (config->GetTime_Domain());
38 
39  nVar = dynamic? 3*direct_solver->GetnVar() : direct_solver->GetnVar();
40  nDim = geometry->GetnDim();
41 
42  /*-- Store some information about direct solver ---*/
43  this->KindDirect_Solver = Kind_Solver;
44  this->direct_solver = direct_solver;
45 
46  nMarker = config->GetnMarker_All();
47  nPoint = geometry->GetnPoint();
48  nPointDomain = geometry->GetnPointDomain();
49 
50  /*--- Define some auxiliary vectors related to the residual ---*/
51 
52  Residual_RMS.resize(nVar,1.0);
53  Residual_Max.resize(nVar,1.0);
54  Point_Max.resize(nVar,0);
55  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
56 
57  /*--- Define some auxiliary vectors related to the residual for problems with a BGS strategy---*/
58 
59  if (config->GetMultizone_Residual()) {
60 
61  Residual_BGS.resize(nVar,1.0);
62  Residual_Max_BGS.resize(nVar,1.0);
63  Point_Max_BGS.resize(nVar,0);
64  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
65  }
66 
67  /*--- Initialize the adjoint solution. ---*/
68 
69  vector<su2double> init(nVar,1e-16);
70  nodes = new CDiscAdjFEABoundVariable(init.data(), nPoint, nDim, nVar, config);
71  SetBaseClassPointerToNodes();
72 
73  /*--- Set which points are vertices and allocate boundary data. ---*/
74 
75  for (iPoint = 0; iPoint < nPoint; iPoint++)
76  for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
77  long iVertex = geometry->nodes->GetVertex(iPoint, iMarker);
78  if (iVertex >= 0) {
79  nodes->Set_isVertex(iPoint,true);
80  break;
81  }
82  }
83  nodes->AllocateBoundaryVariables(config);
84 
85  /*--- Initialize vector structures for multiple material definition ---*/
86 
87  nMPROP = config->GetnElasticityMat();
88 
89  E.resize(nMPROP);
90  Nu.resize(nMPROP);
91  Rho.resize(nMPROP);
92  Rho_DL.resize(nMPROP);
93 
94  /*--- Initialize vector structures for multiple electric regions ---*/
95 
96  de_effects = config->GetDE_Effects();
97 
98  if (de_effects) {
99  nEField = config->GetnElectric_Field();
100  EField.resize(nEField);
101  }
102 
103  /*--- Initialize vector structures for structural-based design variables ---*/
104 
105  switch (config->GetDV_FEA()) {
106  case YOUNG_MODULUS:
107  case POISSON_RATIO:
108  case DENSITY_VAL:
109  case DEAD_WEIGHT:
110  case ELECTRIC_FIELD:
111  fea_dv = true;
112  break;
113  default:
114  fea_dv = false;
115  break;
116  }
117 
118  if (fea_dv) ReadDV(config);
119 
120  SolverName = "ADJ.FEA";
121 }

166 void CDiscAdjFEASolver::RegisterVariables(CGeometry *geometry, CConfig *config, bool reset){
167 
168  /*--- Register element-based values as input ---*/
169 
170  unsigned short iVar;
171 
172  if (KindDirect_Solver == RUNTIME_FEA_SYS) {
173 
174  const bool pseudo_static = config->GetPseudoStatic();
175 
176  for (iVar = 0; iVar < nMPROP; iVar++) {
177  E[iVar] = config->GetElasticyMod(iVar);
178  Nu[iVar] = config->GetPoissonRatio(iVar);
179  Rho[iVar] = pseudo_static? 0.0 : config->GetMaterialDensity(iVar);
180  Rho_DL[iVar] = config->GetMaterialDensity(iVar);
181  }
182 
183  /*--- Read the values of the electric field ---*/
184  if (de_effects) {
185  for (iVar = 0; iVar < nEField; iVar++)
186  EField[iVar] = config->Get_Electric_Field_Mod(iVar);
187  }
188 
189  /*--- Reset index, otherwise messes up other derivatives ---*/
190  if (fea_dv) {
191  for (iVar = 0; iVar < nDV; iVar++) AD::ResetInput(DV[iVar]);
192  }
193 
194  if (!reset) {
195  E.Register();
196  Nu.Register();
197  Rho.Register();
198  Rho_DL.Register();
199  if (de_effects) EField.Register();
200  if (fea_dv) DV.Register();
201  }
202 
203  /*--- Register or reset the flow tractions ---*/
204  if (config->GetnMarker_Fluid_Load() > 0) {
205  direct_solver->GetNodes()->RegisterFlowTraction(reset);
206  }
207 
208  }
209 
210  /*--- Here it is possible to register other variables as input that influence the flow solution
211  * and thereby also the objective function. The adjoint values (i.e. the derivatives) can be
212  * extracted in the ExtractAdjointVariables routine. ---*/
213 }

225 void CDiscAdjFEASolver::ExtractAdjoint_Solution(CGeometry *geometry, CConfig *config, bool CrossTerm) {
226 
227  /*--- Set the old solution, for multi-zone problems this is done after computing the
228  * residuals, otherwise the per-zone-residuals do not make sense, as on entry Solution
229  * contains contributions from other zones but on extraction it does not. ---*/
230 
231  if (!config->GetMultizone_Problem()) nodes->Set_OldSolution();
232 
233  /*--- Extract and store the adjoint solution ---*/
234 
235  AD::BeginUseAdjoints();
236 
237  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
238  su2double Solution[MAXNVAR] = {0.0};
239  direct_solver->GetNodes()->GetAdjointSolution(iPoint,Solution);
240  nodes->SetSolution(iPoint,Solution);
241  }
242 
243  AD::EndUseAdjoints();
244 
245  if (CrossTerm) return;
246 
247  /*--- Extract and store the adjoint solution at time n (including accel. and velocity) ---*/
248 
249  if (config->GetTime_Domain()) {
250  AD::BeginUseAdjoints();
251 
252  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
253  su2double Solution[MAXNVAR] = {0.0};
254  direct_solver->GetNodes()->GetAdjointSolution_time_n(iPoint,Solution);
255  nodes->Set_Solution_time_n(iPoint,Solution);
256  }
257 
258  AD::EndUseAdjoints();
259  }
260 
261  /*--- Set the residuals ---*/
262 
263  SetResToZero();
264 
265  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
266  for (auto iVar = 0u; iVar < nVar; iVar++){
267  su2double residual = nodes->GetSolution(iPoint, iVar) - nodes->GetSolution_Old(iPoint, iVar);
268 
269  Residual_RMS[iVar] += residual*residual;
270  AddRes_Max(iVar,fabs(residual),geometry->nodes->GetGlobalIndex(iPoint),geometry->nodes->GetCoord(iPoint));
271  }
272  }
273 
274  SetResidual_RMS(geometry, config);
275 
276  SetIterLinSolver(direct_solver->System.GetIterations());
277  SetResLinSolver(direct_solver->System.GetResidual());
278 
279 }

349 void CDiscAdjFEASolver::SetSensitivity(CGeometry *geometry, CConfig *config, CSolver*){
350 
351  const bool time_domain = config->GetTime_Domain();
352 
353  /*--- Store the final material sensitivities for the time step to increment them on the next time step. ---*/
354  if (time_domain) {
355  E.Store();
356  Nu.Store();
357  Rho.Store();
358  Rho_DL.Store();
359  if (de_effects) EField.Store();
360  if (fea_dv) DV.Store();
361  }
362 
363  /*--- Extract the topology optimization density sensitivities. ---*/
364 
365  direct_solver->ExtractAdjoint_Variables(geometry, config);
366 
367  /*--- Extract the geometric sensitivities ---*/
368 
369  AD::BeginUseAdjoints();
370 
371  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
372 
373  auto Coord = geometry->nodes->GetCoord(iPoint);
374 
375  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
376 
377  su2double Sensitivity = geometry->nodes->GetAdjointSolution(iPoint, iDim);
378  AD::ResetInput(Coord[iDim]);
379 
380  if (!time_domain) {
381  nodes->SetSensitivity(iPoint, iDim, Sensitivity);
382  } else {
383  nodes->SetSensitivity(iPoint, iDim, nodes->GetSensitivity(iPoint, iDim) + Sensitivity);
384  }
385  }
386  }
387 
388  AD::EndUseAdjoints();
389 
390 }

392 void CDiscAdjFEASolver::ReadDV(const CConfig *config) {
393 
394  string filename;
395  ifstream properties_file;
396 
397  /*--- Choose the filename of the design variable ---*/
398 
399  switch (config->GetDV_FEA()) {
400  case YOUNG_MODULUS:
401  filename = "dv_young.opt";
402  break;
403  case POISSON_RATIO:
404  filename = "dv_poisson.opt";
405  break;
406  case DENSITY_VAL:
407  case DEAD_WEIGHT:
408  filename = "dv_density.opt";
409  break;
410  case ELECTRIC_FIELD:
411  filename = "dv_efield.opt";
412  break;
413  default:
414  filename = "dv.opt";
415  break;
416  }
417 
418  if (rank == MASTER_NODE) cout << "Filename: " << filename << "." << endl;
419 
420  properties_file.open(filename.data(), ios::in);
421 
422  /*--- In case there is no file, all elements get the same property (0) ---*/
423 
424  if (properties_file.fail()) {
425 
426  if (rank == MASTER_NODE)
427  cout << "There is no design variable file." << endl;
428 
429  nDV = 1;
430  DV.resize(nDV);
431  DV[0] = 1.0;
432  }
433  else{
434 
435  string text_line;
436  vector<su2double> values;
437 
438  /*--- Skip the first line: it is the header ---*/
439  getline (properties_file, text_line);
440 
441  while (getline (properties_file, text_line)) {
442  istringstream point_line(text_line);
443 
444  unsigned long index;
445  su2double value;
446  point_line >> index >> value;
447 
448  values.push_back(value);
449  }
450 
451  nDV = values.size();
452  DV.resize(nDV);
453  unsigned short iDV = 0;
454  for (const auto& x : values) DV[iDV++] = x;
455 
456  }
457 
458 }
