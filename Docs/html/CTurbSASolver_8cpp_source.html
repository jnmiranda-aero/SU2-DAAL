36  : CTurbSolver(geometry, config, false) {

54

56 
57  /*--- Single grid simulation ---*/
58

60

96  /*--- Read farfield conditions from config ---*/
97 
98  Density_Inf = config->GetDensity_FreeStreamND();
99  Viscosity_Inf = config->GetViscosity_FreeStreamND();
100 
101  /*--- Factor_nu_Inf in [3.0, 5.0] ---*/
102

104  su2double nu_tilde_Inf = Factor_nu_Inf*Viscosity_Inf/Density_Inf;
105  if (config->GetSAParsedOptions().bc) {

107  }
108 
109  Solution_Inf[0] = nu_tilde_Inf;
110 
111  /*--- Factor_nu_Engine ---*/
112  Factor_nu_Engine = config->GetNuFactor_Engine();

116  }
117 
118  /*--- Factor_nu_ActDisk ---*/
119  Factor_nu_ActDisk = config->GetNuFactor_Engine();
120  nu_tilde_ActDisk = Factor_nu_ActDisk*Viscosity_Inf/Density_Inf;
121 
122  /*--- Eddy viscosity at infinity ---*/
123  su2double Ji, Ji_3, fv1, cv1_3 = 7.1*7.1*7.1;
124  su2double muT_Inf;

128  muT_Inf = Density_Inf*fv1*nu_tilde_Inf;
129

131

141 
142  SlidingState.resize(nMarker);
143  SlidingStateNodes.resize(nMarker);
144 
145  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
146  if (config->GetMarker_All_KindBC(iMarker) == FLUID_INTERFACE) {
147  SlidingState[iMarker].resize(nVertex[iMarker], nPrimVar+1) = nullptr;
148  SlidingStateNodes[iMarker].resize(nVertex[iMarker],0);
149  }
150  }
151 
152  /*-- Allocation of inlets has to happen in derived classes (not CTurbSolver),
153  * due to arbitrary number of turbulence variables ---*/
154 
155  Inlet_TurbVars.resize(nMarker);
156  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++)
157  Inlet_TurbVars[iMarker].resize(nVertex[iMarker],nVar) = nu_tilde_Inf;
158

160 
161  const su2double CFL = config->GetCFL(MGLevel)*config->GetCFLRedCoeff_Turb();
162  for (iPoint = 0; iPoint < nPoint; iPoint++) {
163  nodes->SetLocalCFL(iPoint, CFL);

166  Max_CFL_Local = CFL;

208 void CTurbSASolver::Postprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh) {
209 
210  const su2double cv1_3 = 7.1*7.1*7.1, cR1 = 0.5, rough_const = 0.03;
211 
212  const bool neg_spalart_allmaras = config->GetSAParsedOptions().version == SA_OPTIONS::NEG;
213 
214  /*--- Compute eddy viscosity ---*/
215 
216  AD::StartNoSharedReading();
217 
218  SU2_OMP_FOR_STAT(omp_chunk_size)
219  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint ++) {
220 
221  const su2double rho = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
222  const su2double mu = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
223 
224  const su2double nu = mu/rho;
225  const su2double nu_hat = nodes->GetSolution(iPoint,0);
226  const su2double roughness = geometry->nodes->GetRoughnessHeight(iPoint);
227  const su2double dist = geometry->nodes->GetWall_Distance(iPoint) + rough_const * roughness;
228 
229  su2double Ji = nu_hat/nu;
230  if (roughness > 1.0e-10)
231  Ji += cR1*roughness/(dist+EPS);
232 
233  const su2double Ji_3 = Ji*Ji*Ji;
234  const su2double fv1 = Ji_3/(Ji_3+cv1_3);
235 
236  su2double muT = rho*fv1*nu_hat;
237 
238  if (neg_spalart_allmaras && nu_hat < 0) muT = 0.0;
239 
240  nodes->SetmuT(iPoint,muT);
241 
242  }
243  END_SU2_OMP_FOR
244 
245 
246  /*--- Compute turbulence index ---*/
247  if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE || config->GetSAParsedOptions().bc) {
248  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
249 
250  for (auto iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++){
251  if (config->GetViscous_Wall(iMarker)) {
252  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
253  for (auto iVertex = 0u; iVertex < geometry->nVertex[iMarker]; iVertex++) {
254  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
255 
256  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
257 
258  if (geometry->nodes->GetDomain(iPoint)) {
259  const auto jPoint = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
260 
261  su2double FrictionVelocity = 0.0;
262  /*--- Formulation varies for 2D and 3D problems: in 3D the friction velocity is assumed to be sqrt(mu * |Omega|)
263  (provided by the reference paper https://doi.org/10.2514/6.1992-439), whereas in 2D we have to use the
264  standard definition sqrt(c_f / rho) since Omega = 0. ---*/
265  if(nDim == 2){
266  su2double shearStress = 0.0;
267  for(auto iDim = 0u; iDim < nDim; iDim++) {
268  shearStress += pow(solver_container[FLOW_SOL]->GetCSkinFriction(iMarker, iVertex, iDim), 2.0);
269  }
270  shearStress = sqrt(shearStress);
271 
272  FrictionVelocity = sqrt(shearStress/flowNodes->GetDensity(iPoint));
273  } else {
274  su2double VorticityMag = max(GeometryToolbox::Norm(3, flowNodes->GetVorticity(iPoint)), 1e-12);
275  FrictionVelocity = sqrt(flowNodes->GetLaminarViscosity(iPoint)*VorticityMag);
276  }
277 
278  const su2double wall_dist = geometry->nodes->GetWall_Distance(jPoint);
279  const su2double Derivative = nodes->GetSolution(jPoint, 0) / wall_dist;
280  const su2double turbulence_index = Derivative / (FrictionVelocity * 0.41);
281 
282  nodes->SetTurbIndex(iPoint, turbulence_index);
283 
284  }
285  }
286  END_SU2_OMP_FOR
287  }
288  }
289  }
290 
291  AD::EndNoSharedReading();

308 void CTurbSASolver::Source_Residual(CGeometry *geometry, CSolver **solver_container,
309  CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
310 
311  bool axisymmetric = config->GetAxisymmetric();
312 
313  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
314  const bool harmonic_balance = (config->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE);
315  const bool transition_BC = config->GetSAParsedOptions().bc;
316 
317  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
318 
319  /*--- Pick one numerics object per thread. ---*/
320  auto* numerics = numerics_container[SOURCE_FIRST_TERM + omp_get_thread_num()*MAX_TERMS];
321 
322  AD::StartNoSharedReading();
323 
324  /*--- Loop over all points. ---*/
325 
326  SU2_OMP_FOR_DYN(omp_chunk_size)
327  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
328 
329  /*--- Conservative variables w/o reconstruction ---*/
330 
331  numerics->SetPrimitive(flowNodes->GetPrimitive(iPoint), nullptr);
332 
333  /*--- Gradient of the primitive and conservative variables ---*/
334 
335  numerics->SetPrimVarGradient(flowNodes->GetGradient_Primitive(iPoint), nullptr);
336 
337  /*--- Set vorticity and strain rate magnitude ---*/
338 
339  numerics->SetVorticity(flowNodes->GetVorticity(iPoint), nullptr);
340 
341  numerics->SetStrainMag(flowNodes->GetStrainMag(iPoint), 0.0);
342 
343  /*--- Turbulent variables w/o reconstruction, and its gradient ---*/
344 
345  numerics->SetScalarVar(nodes->GetSolution(iPoint), nullptr);
346  numerics->SetScalarVarGradient(nodes->GetGradient(iPoint), nullptr);
347 
348  /*--- Set volume ---*/
349 
350  numerics->SetVolume(geometry->nodes->GetVolume(iPoint));
351 
352  /*--- Get Hybrid RANS/LES Type and set the appropriate wall distance ---*/
353 
354  if (config->GetKind_HybridRANSLES() == NO_HYBRIDRANSLES) {
355 
356  /*--- For the SA model, wall roughness is accounted by modifying the computed wall distance
357  * d_new = d + 0.03 k_s
358  * where k_s is the equivalent sand grain roughness height that is specified in cfg file.
359  * For smooth walls, wall roughness is zero and computed wall distance remains the same. */
360 
361  su2double modifiedWallDistance = geometry->nodes->GetWall_Distance(iPoint);
362 
363  modifiedWallDistance += 0.03*geometry->nodes->GetRoughnessHeight(iPoint);
364 
365  /*--- Set distance to the surface ---*/
366 
367  numerics->SetDistance(modifiedWallDistance, 0.0);
368 
369  /*--- Set the roughness of the closest wall. ---*/
370 
371  numerics->SetRoughness(geometry->nodes->GetRoughnessHeight(iPoint), 0.0 );
372 
373  } else {
374 
375  /*--- Set DES length scale ---*/
376 
377  numerics->SetDistance(nodes->GetDES_LengthScale(iPoint), 0.0);
378 
379  }
380 
381  /*--- Effective Intermittency ---*/
382 
383  if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
384  numerics->SetIntermittencyEff(solver_container[TRANS_SOL]->GetNodes()->GetIntermittencyEff(iPoint));
385  numerics->SetIntermittency(solver_container[TRANS_SOL]->GetNodes()->GetSolution(iPoint, 0));
386  }
387 
388  if (axisymmetric) {
389  /*--- Set y coordinate ---*/
390  numerics->SetCoord(geometry->nodes->GetCoord(iPoint), geometry->nodes->GetCoord(iPoint));
391  }
392 
393  /*--- Compute the source term ---*/
394 
395  auto residual = numerics->ComputeResidual(config);
396 
397  /*--- Store the intermittency ---*/
398 
399  if (transition_BC || config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
400  nodes->SetIntermittency(iPoint,numerics->GetIntermittencyEff());
401  }
402 
403  /*--- Subtract residual and the Jacobian ---*/
404 
405  LinSysRes.SubtractBlock(iPoint, residual);
406 
407  if (implicit) Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
408 
409  }
410  END_SU2_OMP_FOR
411 
412  if (harmonic_balance) {
413 
414  SU2_OMP_FOR_STAT(omp_chunk_size)
415  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
416 
417  su2double Volume = geometry->nodes->GetVolume(iPoint);
418 
419  /*--- Access stored harmonic balance source term ---*/
420 
421  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
422  su2double Source = nodes->GetHarmonicBalance_Source(iPoint,iVar);
423  LinSysRes(iPoint,iVar) += Source*Volume;
424  }
425  }
426  END_SU2_OMP_FOR
427  }
428 
429  AD::EndNoSharedReading();
430 
431 }

437 void CTurbSASolver::BC_HeatFlux_Wall(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
438  CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
439 
440  /*--- Evaluate nu tilde at the closest point to the surface using the wall functions. ---*/
441 
442  if (config->GetWall_Functions()) {
443  SU2_OMP_SAFE_GLOBAL_ACCESS(SetTurbVars_WF(geometry, solver_container, config, val_marker);)
444  return;
445  }
446 
447  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
448  bool rough_wall = false;
449  string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
450  WALL_TYPE WallType;
451  su2double Roughness_Height;
452  tie(WallType, Roughness_Height) = config->GetWallRoughnessProperties(Marker_Tag);
453  if (WallType == WALL_TYPE::ROUGH) rough_wall = true;
454 
455  /*--- The dirichlet condition is used only without wall function, otherwise the
456  convergence is compromised as we are providing nu tilde values for the
457  first point of the wall ---*/
458 
459  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
460  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
461 
462  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
463 
464  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
465 
466  if (geometry->nodes->GetDomain(iPoint)) {
467  if (!rough_wall) {
468  for (auto iVar = 0u; iVar < nVar; iVar++)
469  nodes->SetSolution_Old(iPoint,iVar,0.0);
470 
471  LinSysRes.SetBlock_Zero(iPoint);
472 
473  /*--- Includes 1 in the diagonal ---*/
474 
475  if (implicit) Jacobian.DeleteValsRowi(iPoint);
476  } else {
477  /*--- For rough walls, the boundary condition is given by
478  * (\frac{\partial \nu}{\partial n})_wall = \frac{\nu}{0.03*k_s}
479  * where \nu is the solution variable, $n$ is the wall normal direction
480  * and k_s is the equivalent sand grain roughness specified. ---*/
481 
482  /*--- Compute dual-grid area and boundary normal ---*/
483  su2double Normal[MAXNDIM] = {0.0};
484  for (auto iDim = 0u; iDim < nDim; iDim++)
485  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
486 
487  su2double Area = GeometryToolbox::Norm(nDim, Normal);
488 
489  /*--- Get laminar_viscosity and density ---*/
490  su2double sigma = 2.0/3.0;
491  su2double laminar_viscosity = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
492  su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
493 
494  su2double nu_total = (laminar_viscosity/density + nodes->GetSolution(iPoint,0));
495 
496  su2double coeff = (nu_total/sigma);
497  su2double RoughWallBC = nodes->GetSolution(iPoint,0)/(0.03*Roughness_Height);
498 
499  su2double Res_Wall;// = new su2double [nVar];
500  Res_Wall = coeff*RoughWallBC*Area;
501  LinSysRes.SubtractBlock(iPoint, &Res_Wall);
502 
503  su2double Jacobian_i = (laminar_viscosity*Area)/(0.03*Roughness_Height*sigma);
504  Jacobian_i += 2.0*RoughWallBC*Area/sigma;
505  if (implicit) Jacobian.AddVal2Diag(iPoint, -Jacobian_i);
506  }
507  }
508  }
509  END_SU2_OMP_FOR
510 }

519 void CTurbSASolver::BC_Inlet(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
520  CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
521 
522  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
523 
524  /*--- Loop over all the vertices on this boundary marker ---*/
525 
526  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
527  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
528 
529  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
530 
531  /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
532 
533  if (geometry->nodes->GetDomain(iPoint)) {
534 
535  /*--- Normal vector for this vertex (negate for outward convention) ---*/
536 
537  su2double Normal[MAXNDIM] = {0.0};
538  for (auto iDim = 0u; iDim < nDim; iDim++)
539  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
540 
541  /*--- Allocate the value at the inlet ---*/
542 
543  auto V_inlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
544 
545  /*--- Retrieve solution at the farfield boundary node ---*/
546 
547  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
548 
549  /*--- Set various quantities in the solver class ---*/
550 
551  conv_numerics->SetPrimitive(V_domain, V_inlet);
552 
553  /*--- Non-dimensionalize Inlet_TurbVars if Inlet-Files are used. ---*/
554  su2double Inlet_Vars[MAXNVAR];
555  Inlet_Vars[0] = Inlet_TurbVars[val_marker][iVertex][0];
556  if (config->GetInlet_Profile_From_File()) {
557  Inlet_Vars[0] *= config->GetDensity_Ref() / config->GetViscosity_Ref();
558  } else {
559  /*--- Obtain fluid model for computing the nu tilde to impose at the inlet boundary. ---*/
560  CFluidModel* FluidModel = solver_container[FLOW_SOL]->GetFluidModel();
561 
562  /*--- Obtain density and laminar viscosity at inlet boundary node ---*/
563 
564  su2double Density_Inlet;
