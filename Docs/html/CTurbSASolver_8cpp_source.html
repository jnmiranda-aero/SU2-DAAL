35CTurbSASolver::CTurbSASolver(CGeometry *geometry, CConfig *config, unsigned short iMesh, CFluidModel* FluidModel)
36 : CTurbSolver(geometry, config, false) {
37 unsigned long iPoint;
38 su2double Density_Inf, Viscosity_Inf, Factor_nu_Inf, Factor_nu_Engine, Factor_nu_ActDisk;
39
40 bool multizone = config->GetMultizone_Problem();
41
42 /*--- Dimension of the problem --> dependent of the turbulent model ---*/
43
44 nVar = 1;
45 nPrimVar = 1;
46 nPoint = geometry->GetnPoint();
47 nPointDomain = geometry->GetnPointDomain();
48
49 /*--- Initialize nVarGrad for deallocation ---*/
50
51 nVarGrad = nVar;
52
53 /*--- Define geometry constants in the solver structure ---*/
54
55 nDim = geometry->GetnDim();
56
57 /*--- Single grid simulation ---*/
58
59 if (iMesh == MESH_0 || config->GetMGCycle() == FULLMG_CYCLE) {
60
61 /*--- Define some auxiliar vector related with the residual ---*/
62
63 Residual_RMS.resize(nVar,0.0);
64 Residual_Max.resize(nVar,0.0);
65 Point_Max.resize(nVar,0);
66 Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
67
68 /*--- Initialization of the structure of the whole Jacobian ---*/
69
70 if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (SA model)." << endl;
71 Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config, ReducerStrategy);
72 LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
73 LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
74 System.SetxIsZero(true);
75
76 if (ReducerStrategy)
77 EdgeFluxes.Initialize(geometry->GetnEdge(), geometry->GetnEdge(), nVar, nullptr);
78
79 if (config->GetExtraOutput()) {
80 if (nDim == 2) { nOutputVariables = 13; }
81 else if (nDim == 3) { nOutputVariables = 19; }
82 OutputVariables.Initialize(nPoint, nPointDomain, nOutputVariables, 0.0);
83 OutputHeadingNames = new string[nOutputVariables];
84 }
85
86 /*--- Initialize the BGS residuals in multizone problems. ---*/
87 if (multizone){
88 Residual_BGS.resize(nVar,0.0);
89 Residual_Max_BGS.resize(nVar,0.0);
90 Point_Max_BGS.resize(nVar,0);
91 Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
92 }
93
94 }
95
96 /*--- Read farfield conditions from config ---*/
97
98 Density_Inf = config->GetDensity_FreeStreamND();
99 Viscosity_Inf = config->GetViscosity_FreeStreamND();
100
101 /*--- Factor_nu_Inf in [3.0, 5.0] ---*/
102
103 Factor_nu_Inf = config->GetNuFactor_FreeStream();
104 su2double nu_tilde_Inf = Factor_nu_Inf*Viscosity_Inf/Density_Inf;
105 if (config->GetSAParsedOptions().bc) {
106 nu_tilde_Inf = 0.005*Factor_nu_Inf*Viscosity_Inf/Density_Inf;
107 }
108
109 Solution_Inf[0] = nu_tilde_Inf;
110
111 /*--- Factor_nu_Engine ---*/
112 Factor_nu_Engine = config->GetNuFactor_Engine();
113 nu_tilde_Engine = Factor_nu_Engine*Viscosity_Inf/Density_Inf;
114 if (config->GetSAParsedOptions().bc) {
115 nu_tilde_Engine = 0.005*Factor_nu_Engine*Viscosity_Inf/Density_Inf;
116 }
117
118 /*--- Factor_nu_ActDisk ---*/
119 Factor_nu_ActDisk = config->GetNuFactor_Engine();
120 nu_tilde_ActDisk = Factor_nu_ActDisk*Viscosity_Inf/Density_Inf;
121
122 /*--- Eddy viscosity at infinity ---*/
123 su2double Ji, Ji_3, fv1, cv1_3 = 7.1*7.1*7.1;
124 su2double muT_Inf;
125 Ji = nu_tilde_Inf/Viscosity_Inf*Density_Inf;
126 Ji_3 = Ji*Ji*Ji;
127 fv1 = Ji_3/(Ji_3+cv1_3);
128 muT_Inf = Density_Inf*fv1*nu_tilde_Inf;
129
130 /*--- Initialize the solution to the far-field state everywhere. ---*/
131
132 nodes = new CTurbSAVariable(nu_tilde_Inf, muT_Inf, nPoint, nDim, nVar, config);
133 SetBaseClassPointerToNodes();
134
135 /*--- MPI solution ---*/
136
137 InitiateComms(geometry, config, SOLUTION_EDDY);
138 CompleteComms(geometry, config, SOLUTION_EDDY);
139
140 /*--- Initializate quantities for SlidingMesh Interface ---*/
141
142 SlidingState.resize(nMarker);
143 SlidingStateNodes.resize(nMarker);
144
145 for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
146 if (config->GetMarker_All_KindBC(iMarker) == FLUID_INTERFACE) {
147 SlidingState[iMarker].resize(nVertex[iMarker], nPrimVar+1) = nullptr;
148 SlidingStateNodes[iMarker].resize(nVertex[iMarker],0);
149 }
150 }
151
152 /*-- Allocation of inlets has to happen in derived classes (not CTurbSolver),
153 * due to arbitrary number of turbulence variables ---*/
154
155 Inlet_TurbVars.resize(nMarker);
156 for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++)
157 Inlet_TurbVars[iMarker].resize(nVertex[iMarker],nVar) = nu_tilde_Inf;
158
159 /*--- Store the initial CFL number for all grid points. ---*/
160
161 const su2double CFL = config->GetCFL(MGLevel)*config->GetCFLRedCoeff_Turb();
162 for (iPoint = 0; iPoint < nPoint; iPoint++) {
163 nodes->SetLocalCFL(iPoint, CFL);
164 }
165 Min_CFL_Local = CFL;
166 Max_CFL_Local = CFL;
167 Avg_CFL_Local = CFL;
168
169 /*--- Add the solver name. ---*/
170 SolverName = "SA";
171
172}

184

208void CTurbSASolver::Postprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh) {
209
210 const su2double cv1_3 = 7.1*7.1*7.1, cR1 = 0.5, rough_const = 0.03;
211
212 const bool neg_spalart_allmaras = config->GetSAParsedOptions().version == SA_OPTIONS::NEG;
213
214 /*--- Compute eddy viscosity ---*/
215
216 AD::StartNoSharedReading();
217
218 SU2_OMP_FOR_STAT(omp_chunk_size)
219 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint ++) {
220
221 const su2double rho = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
222 const su2double mu = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
223
224 const su2double nu = mu/rho;
225 const su2double nu_hat = nodes->GetSolution(iPoint,0);
226 const su2double roughness = geometry->nodes->GetRoughnessHeight(iPoint);
227 const su2double dist = geometry->nodes->GetWall_Distance(iPoint) + rough_const * roughness;
228
229 su2double Ji = nu_hat/nu;
230 if (roughness > 1.0e-10)
231 Ji += cR1*roughness/(dist+EPS);
232
233 const su2double Ji_3 = Ji*Ji*Ji;
234 const su2double fv1 = Ji_3/(Ji_3+cv1_3);
235
236 su2double muT = rho*fv1*nu_hat;
237
238 if (neg_spalart_allmaras && nu_hat < 0) muT = 0.0;
239
240 nodes->SetmuT(iPoint,muT);
241
242 }
243 END_SU2_OMP_FOR
244
245
246 /*--- Compute turbulence index ---*/
247 if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE || config->GetSAParsedOptions().bc) {
248 auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
249
250 for (auto iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++){
251 if (config->GetViscous_Wall(iMarker)) {
252 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
253 for (auto iVertex = 0u; iVertex < geometry->nVertex[iMarker]; iVertex++) {
254 const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
255
256 /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
257
258 if (geometry->nodes->GetDomain(iPoint)) {
259 const auto jPoint = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
260
261 su2double FrictionVelocity = 0.0;
262 /*--- Formulation varies for 2D and 3D problems: in 3D the friction velocity is assumed to be sqrt(mu * |Omega|)
263 (provided by the reference paper https://doi.org/10.2514/6.1992-439), whereas in 2D we have to use the
264 standard definition sqrt(c_f / rho) since Omega = 0. ---*/
265 if(nDim == 2){
266 su2double shearStress = 0.0;
267 for(auto iDim = 0u; iDim < nDim; iDim++) {
268 shearStress += pow(solver_container[FLOW_SOL]->GetCSkinFriction(iMarker, iVertex, iDim), 2.0);
269 }
270 shearStress = sqrt(shearStress);
271
272 FrictionVelocity = sqrt(shearStress/flowNodes->GetDensity(iPoint));
273 } else {
274 su2double VorticityMag = max(GeometryToolbox::Norm(3, flowNodes->GetVorticity(iPoint)), 1e-12);
275 FrictionVelocity = sqrt(flowNodes->GetLaminarViscosity(iPoint)*VorticityMag);
276 }
277
278 const su2double wall_dist = geometry->nodes->GetWall_Distance(jPoint);
279 const su2double Derivative = nodes->GetSolution(jPoint, 0) / wall_dist;
280 const su2double turbulence_index = Derivative / (FrictionVelocity * 0.41);
281
282 nodes->SetTurbIndex(iPoint, turbulence_index);
283
284 }
285 }
286 END_SU2_OMP_FOR
287 }
288 }
289 }
290
291 AD::EndNoSharedReading();

308void CTurbSASolver::Source_Residual(CGeometry *geometry, CSolver **solver_container,
309 CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
310
311 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
312 const bool harmonic_balance = (config->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE);
313 const bool transition_BC = config->GetSAParsedOptions().bc;
314
315 auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
316
317 /*--- Pick one numerics object per thread. ---*/
318 auto* numerics = numerics_container[SOURCE_FIRST_TERM + omp_get_thread_num()*MAX_TERMS];
319
320 AD::StartNoSharedReading();
321
322 /*--- Loop over all points. ---*/
323
324 SU2_OMP_FOR_DYN(omp_chunk_size)
325 for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
326
327 /*--- Conservative variables w/o reconstruction ---*/
328
329 numerics->SetPrimitive(flowNodes->GetPrimitive(iPoint), nullptr);
330
331 /*--- Gradient of the primitive and conservative variables ---*/
332
333 numerics->SetPrimVarGradient(flowNodes->GetGradient_Primitive(iPoint), nullptr);
334
335 /*--- Set vorticity and strain rate magnitude ---*/
336
337 numerics->SetVorticity(flowNodes->GetVorticity(iPoint), nullptr);
338
339 numerics->SetStrainMag(flowNodes->GetStrainMag(iPoint), 0.0);
340
341 /*--- Turbulent variables w/o reconstruction, and its gradient ---*/
342
343 numerics->SetScalarVar(nodes->GetSolution(iPoint), nullptr);
344 numerics->SetScalarVarGradient(nodes->GetGradient(iPoint), nullptr);
345
346 /*--- Set volume ---*/
347
348 numerics->SetVolume(geometry->nodes->GetVolume(iPoint));
349
350 /*--- Get Hybrid RANS/LES Type and set the appropriate wall distance ---*/
351
352 if (config->GetKind_HybridRANSLES() == NO_HYBRIDRANSLES) {
353
354 /*--- For the SA model, wall roughness is accounted by modifying the computed wall distance
355 * d_new = d + 0.03 k_s
356 * where k_s is the equivalent sand grain roughness height that is specified in cfg file.
357 * For smooth walls, wall roughness is zero and computed wall distance remains the same. */
358
359 su2double modifiedWallDistance = geometry->nodes->GetWall_Distance(iPoint);
360
361 modifiedWallDistance += 0.03*geometry->nodes->GetRoughnessHeight(iPoint);
362
363 /*--- Set distance to the surface ---*/
364
365 numerics->SetDistance(modifiedWallDistance, 0.0);
366
367 /*--- Set the roughness of the closest wall. ---*/
368
369 numerics->SetRoughness(geometry->nodes->GetRoughnessHeight(iPoint), 0.0 );
370
371 } else {
372
373 /*--- Set DES length scale ---*/
374
375 numerics->SetDistance(nodes->GetDES_LengthScale(iPoint), 0.0);
376
377 }
378
379 /*--- Effective Intermittency ---*/
380
381 if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
382 numerics->SetIntermittencyEff(solver_container[TRANS_SOL]->GetNodes()->GetIntermittencyEff(iPoint));
383 numerics->SetIntermittency(solver_container[TRANS_SOL]->GetNodes()->GetSolution(iPoint, 0));
384 }
385
386 /*--- Compute the source term ---*/
387
388 auto residual = numerics->ComputeResidual(config);
389
390 /*--- Store the intermittency ---*/
391
392 if (transition_BC || config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
393 nodes->SetIntermittency(iPoint,numerics->GetIntermittencyEff());
394 }
395
396 /*--- Subtract residual and the Jacobian ---*/
397
398 LinSysRes.SubtractBlock(iPoint, residual);
399
400 if (implicit) Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
401
402 }
403 END_SU2_OMP_FOR
404
405 if (harmonic_balance) {
406
407 SU2_OMP_FOR_STAT(omp_chunk_size)
408 for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
409
410 su2double Volume = geometry->nodes->GetVolume(iPoint);
411
412 /*--- Access stored harmonic balance source term ---*/
413
414 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
415 su2double Source = nodes->GetHarmonicBalance_Source(iPoint,iVar);
416 LinSysRes(iPoint,iVar) += Source*Volume;
417 }
418 }
419 END_SU2_OMP_FOR
420 }
421
422 AD::EndNoSharedReading();
423
424}

430void CTurbSASolver::BC_HeatFlux_Wall(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
431 CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
432
433 /*--- Evaluate nu tilde at the closest point to the surface using the wall functions. ---*/
434
435 if (config->GetWall_Functions()) {
436 SU2_OMP_SAFE_GLOBAL_ACCESS(SetTurbVars_WF(geometry, solver_container, config, val_marker);)
437 return;
438 }
439
440 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
441 bool rough_wall = false;
442 string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
443 WALL_TYPE WallType;
444 su2double Roughness_Height;
445 tie(WallType, Roughness_Height) = config->GetWallRoughnessProperties(Marker_Tag);
446 if (WallType == WALL_TYPE::ROUGH) rough_wall = true;
447
448 /*--- The dirichlet condition is used only without wall function, otherwise the
449 convergence is compromised as we are providing nu tilde values for the
450 first point of the wall ---*/
451
452 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
453 for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
454
455 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
456
457 /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
458
459 if (geometry->nodes->GetDomain(iPoint)) {
460 if (!rough_wall) {
461 for (auto iVar = 0u; iVar < nVar; iVar++)
462 nodes->SetSolution_Old(iPoint,iVar,0.0);
463
464 LinSysRes.SetBlock_Zero(iPoint);
465
466 /*--- Includes 1 in the diagonal ---*/
467
468 if (implicit) Jacobian.DeleteValsRowi(iPoint);
469 } else {
470 /*--- For rough walls, the boundary condition is given by
471 * (\frac{\partial \nu}{\partial n})_wall = \frac{\nu}{0.03*k_s}
472 * where \nu is the solution variable, $n$ is the wall normal direction
473 * and k_s is the equivalent sand grain roughness specified. ---*/
474
475 /*--- Compute dual-grid area and boundary normal ---*/
476 su2double Normal[MAXNDIM] = {0.0};
477 for (auto iDim = 0u; iDim < nDim; iDim++)
478 Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
479
480 su2double Area = GeometryToolbox::Norm(nDim, Normal);
481
482 /*--- Get laminar_viscosity and density ---*/
483 su2double sigma = 2.0/3.0;
484 su2double laminar_viscosity = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
485 su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
486
487 su2double nu_total = (laminar_viscosity/density + nodes->GetSolution(iPoint,0));
488
489 su2double coeff = (nu_total/sigma);
490 su2double RoughWallBC = nodes->GetSolution(iPoint,0)/(0.03*Roughness_Height);
491
492 su2double Res_Wall;// = new su2double [nVar];
493 Res_Wall = coeff*RoughWallBC*Area;
494 LinSysRes.SubtractBlock(iPoint, &Res_Wall);
495
496 su2double Jacobian_i = (laminar_viscosity*Area)/(0.03*Roughness_Height*sigma);
497 Jacobian_i += 2.0*RoughWallBC*Area/sigma;
498 if (implicit) Jacobian.AddVal2Diag(iPoint, -Jacobian_i);
499 }
500 }
501 }
502 END_SU2_OMP_FOR
503}

512void CTurbSASolver::BC_Inlet(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
513 CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
514
515 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
516
517 /*--- Loop over all the vertices on this boundary marker ---*/
518
519 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
520 for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
521
522 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
523
524 /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
525
526 if (geometry->nodes->GetDomain(iPoint)) {
527
528 /*--- Normal vector for this vertex (negate for outward convention) ---*/
529
530 su2double Normal[MAXNDIM] = {0.0};
531 for (auto iDim = 0u; iDim < nDim; iDim++)
532 Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
533
534 /*--- Allocate the value at the inlet ---*/
535
536 auto V_inlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
537
538 /*--- Retrieve solution at the farfield boundary node ---*/
539
540 auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
541
542 /*--- Set various quantities in the solver class ---*/
543
544 conv_numerics->SetPrimitive(V_domain, V_inlet);
545
546 /*--- Non-dimensionalize Inlet_TurbVars if Inlet-Files are used. ---*/
547 su2double Inlet_Vars[MAXNVAR];
548 Inlet_Vars[0] = Inlet_TurbVars[val_marker][iVertex][0];
549 if (config->GetInlet_Profile_From_File()) {
550 Inlet_Vars[0] *= config->GetDensity_Ref() / config->GetViscosity_Ref();
551 } else {
552 /*--- Obtain fluid model for computing the nu tilde to impose at the inlet boundary. ---*/
553 CFluidModel* FluidModel = solver_container[FLOW_SOL]->GetFluidModel();
554
555 /*--- Obtain density and laminar viscosity at inlet boundary node ---*/
556
557 su2double Density_Inlet;
558 if (config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) {
559 Density_Inlet = V_inlet[prim_idx.Density()];
560 FluidModel->SetTDState_Prho(V_inlet[prim_idx.Pressure()], Density_Inlet);
561 } else {
562 const su2double* Scalar_Inlet = nullptr;
563 if (config->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
564 Scalar_Inlet = config->GetInlet_SpeciesVal(config->GetMarker_All_TagBound(val_marker));
565 }
566 FluidModel->SetTDState_T(V_inlet[prim_idx.Temperature()], Scalar_Inlet);
567 Density_Inlet = FluidModel->GetDensity();
568 }
569 const su2double Laminar_Viscosity_Inlet = FluidModel->GetLaminarViscosity();
570 const su2double* Turb_Properties = config->GetInlet_TurbVal(config->GetMarker_All_TagBound(val_marker));
571 const su2double Nu_Factor = Turb_Properties[0];
572 Inlet_Vars[0] = Nu_Factor * Laminar_Viscosity_Inlet / Density_Inlet;
573 if (config->GetSAParsedOptions().bc) {
574 Inlet_Vars[0] *= 0.005;
575 }
576 }
577
578 /*--- Load the inlet turbulence variable (uniform by default). ---*/
579
580 conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Inlet_Vars);
581
582 /*--- Set various other quantities in the conv_numerics class ---*/
583
584 conv_numerics->SetNormal(Normal);
585

1255void CTurbSASolver::SetTurbVars_WF(CGeometry *geometry, CSolver **solver_container,
1256 const CConfig *config, unsigned short val_marker) {
1257
1258 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1259
1260 /*--- We use a very high max nr of iterations, but we only need this the first couple of iterations ---*/
1261 const unsigned short max_iter = config->GetwallModel_MaxIter();
1262
1263 /* --- tolerance has LARGE impact on convergence, do not increase this value! --- */
1264 const su2double tol = 1e-12;
1265
1266
1267 /*--- Typical constants from boundary layer theory ---*/
1268
1269 const su2double cv1_3 = 7.1*7.1*7.1;
1270
1271 CVariable* flow_nodes = solver_container[FLOW_SOL]->GetNodes();
1272
1273 /*--- Loop over all of the vertices on this boundary marker ---*/
1274
1275 for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
1276
1277 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
1278 const auto iPoint_Neighbor = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
1279
1280 /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
1281
1282 if (geometry->nodes->GetDomain(iPoint_Neighbor)) {
1283
1284 su2double Y_Plus = solver_container[FLOW_SOL]->GetYPlus(val_marker, iVertex);
1285
1286 /*--- Do not use wall model at the ipoint when y+ < "limit" ---*/
1287
1288 if (Y_Plus < config->GetwallModel_MinYPlus()) continue;
1289
1290 su2double Lam_Visc_Normal = flow_nodes->GetLaminarViscosity(iPoint_Neighbor);
1291 su2double Density_Normal = flow_nodes->GetDensity(iPoint_Neighbor);
1292 su2double Kin_Visc_Normal = Lam_Visc_Normal/Density_Normal;
1293
1294 su2double Eddy_Visc = solver_container[FLOW_SOL]->GetEddyViscWall(val_marker, iVertex);
1295
1296 /*--- Solve for the new value of nu_tilde given the eddy viscosity and using a Newton method ---*/
1297
1298 // start with positive value of nu_til_old
1299 su2double nu_til = 0.0;
1300 su2double nu_til_old = nodes->GetSolution(iPoint,0);
1301
1302 unsigned short counter = 0;
1303 su2double diff = 1.0;
1304 su2double relax = config->GetwallModel_RelFac();
1305 while (diff > tol) {
1306 // note the error in Nichols and Nelson
1307 su2double func = pow(nu_til_old,4) - (Eddy_Visc/Density_Normal)*(pow(nu_til_old,3) + pow(Kin_Visc_Normal,3)*cv1_3);
1308 su2double func_prim = 4.0 * pow(nu_til_old,3) - 3.0*(Eddy_Visc/Density_Normal)*pow(nu_til_old,2);
1309
1310 // damped Newton method
1311 nu_til = nu_til_old - relax*(func/func_prim);
1312
1313 diff = fabs(nu_til-nu_til_old);
1314 nu_til_old = nu_til;
1315
1316 // sometimes we get negative values when the solution has not converged yet, we just reset the nu_tilde in that case.
1317 if (nu_til_old<tol) {
1318 relax /= 2.0;
1319 nu_til_old = nodes->GetSolution(iPoint,0)/relax;
1320 }
1321
1322 counter++;
1323 if (counter > max_iter) break;
1324 }
1325
1326 nodes->SetSolution_Old(iPoint_Neighbor, &nu_til);
1327 LinSysRes.SetBlock_Zero(iPoint_Neighbor);
1328
1329 /*--- includes 1 in the diagonal ---*/
1330
1331 if (implicit) Jacobian.DeleteValsRowi(iPoint_Neighbor);
1332 }
1333 }

1336void CTurbSASolver::SetDES_LengthScale(CSolver **solver, CGeometry *geometry, CConfig *config){
1337
1338 const auto kindHybridRANSLES = config->GetKind_HybridRANSLES();
1339
1340 const su2double constDES = config->GetConst_DES();
1341
1342 const su2double fw_star = 0.424, cv1_3 = pow(7.1, 3), k2 = pow(0.41, 2);
1343 const su2double cb1 = 0.1355, ct3 = 1.2, ct4 = 0.5;
1344 const su2double sigma = 2./3., cb2 = 0.622, f_max = 1.0, f_min = 0.1, a1 = 0.15, a2 = 0.3;
1345
1346 auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver[FLOW_SOL]->GetNodes());
1347
1348 SU2_OMP_FOR_DYN(omp_chunk_size)
1349 for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++){
1350
1351 const auto coord_i = geometry->nodes->GetCoord(iPoint);
1352 const auto nNeigh = geometry->nodes->GetnPoint(iPoint);
1353 const auto wallDistance = geometry->nodes->GetWall_Distance(iPoint);
1354 const auto velocityGrad = flowNodes->GetVelocityGradient(iPoint);
1355 const auto vorticity = flowNodes->GetVorticity(iPoint);
1356 const auto density = flowNodes->GetDensity(iPoint);
1357 const auto laminarViscosity = flowNodes->GetLaminarViscosity(iPoint);
1358 const auto eddyViscosity = nodes->GetmuT(iPoint);
1359 const su2double kinematicViscosity = laminarViscosity/density;
1360 const su2double kinematicViscosityTurb = eddyViscosity/density;
1361
1362 su2double uijuij = 0.0;
1363 for(auto iDim = 0u; iDim < nDim; iDim++){
1364 for(auto jDim = 0u; jDim < nDim; jDim++){
1365 uijuij += pow(velocityGrad[iDim][jDim], 2);
1366 }
1367 }
1368 uijuij = sqrt(fabs(uijuij));
1369 uijuij = max(uijuij,1e-10);
1370
1371 /*--- Low Reynolds number correction term ---*/
1372
1373 const su2double nu_hat = nodes->GetSolution(iPoint,0);
1374 const su2double Ji = nu_hat/kinematicViscosity;
1375 const su2double Ji_2 = Ji * Ji;
1376 const su2double Ji_3 = Ji*Ji*Ji;
1377 const su2double fv1 = Ji_3/(Ji_3+cv1_3);
1378 const su2double fv2 = 1.0 - Ji/(1.0+Ji*fv1);
1379 const su2double ft2 = ct3*exp(-ct4*Ji_2);
1380 const su2double cw1 = cb1/k2+(1.0+cb2)/sigma;
1381
1382 su2double psi_2 = (1.0 - (cb1/(cw1*k2*fw_star))*(ft2 + (1.0 - ft2)*fv2))/(fv1 * max(1.0e-10,1.0-ft2));
1383 psi_2 = min(100.0,psi_2);
1384
1385 su2double lengthScale = 0.0;
1386
1387 switch(kindHybridRANSLES){
1388 case SA_DES: {
1389 /*--- Original Detached Eddy Simulation (DES97)
1390 Spalart
1391 1997
1392 ---*/
1393
1394 const su2double maxDelta = geometry->nodes->GetMaxLength(iPoint);
1395 const su2double distDES = constDES * maxDelta;
1396 lengthScale = min(distDES,wallDistance);
1397
1398 break;
1399 }
1400 case SA_DDES: {
1401 /*--- A New Version of Detached-eddy Simulation, Resistant to Ambiguous Grid Densities.
1402 Spalart et al.
1403 Theoretical and Computational Fluid Dynamics - 2006
1404 ---*/
1405
1406 const su2double maxDelta = geometry->nodes->GetMaxLength(iPoint);
1407
1408 const su2double r_d = (kinematicViscosityTurb+kinematicViscosity)/(uijuij*k2*pow(wallDistance, 2.0));
1409 const su2double f_d = 1.0-tanh(pow(8.0*r_d,3.0));
1410
1411 const su2double distDES = constDES * maxDelta;
1412 lengthScale = wallDistance-f_d*max(0.0,(wallDistance-distDES));
1413
1414 break;
1415 }
1416 case SA_ZDES: {
1417 /*--- Recent improvements in the Zonal Detached Eddy Simulation (ZDES) formulation.
1418 Deck
1419 Theoretical and Computational Fluid Dynamics - 2012
1420 ---*/
1421
1422 const su2double deltaDDES = geometry->nodes->GetMaxLength(iPoint);
1423
1424 su2double delta[MAXNDIM] = {}, ratioOmega[MAXNDIM] = {};
1425
1426 for (const auto jPoint : geometry->nodes->GetPoints(iPoint)) {
1427 const auto coord_j = geometry->nodes->GetCoord(jPoint);
1428 for (auto iDim = 0u; iDim < nDim; iDim++){
1429 const su2double deltaAux = abs(coord_j[iDim] - coord_i[iDim]);
1430 delta[iDim] = max(delta[iDim], deltaAux);
1431 }
1432 }
1433
1434 const su2double omega = GeometryToolbox::Norm(3, vorticity);
1435
1436 for (auto iDim = 0u; iDim < 3; iDim++){
1437 ratioOmega[iDim] = vorticity[iDim]/omega;
1438 }
1439
1440 su2double maxDelta = sqrt(pow(ratioOmega[0], 2)*delta[1]*delta[2] +
1441 pow(ratioOmega[1], 2)*delta[0]*delta[2] +
1442 pow(ratioOmega[2], 2)*delta[0]*delta[1]);
1443
1444 const su2double r_d = (kinematicViscosityTurb+kinematicViscosity)/(uijuij*k2*pow(wallDistance, 2.0));
1445 const su2double f_d = 1.0-tanh(pow(8.0*r_d,3.0));
1446
1447 if (f_d < 0.99){
1448 maxDelta = deltaDDES;
1449 }
1450
1451 const su2double distDES = constDES * maxDelta;
1452 lengthScale = wallDistance-f_d*max(0.0,(wallDistance-distDES));
1453
1454 break;
1455 }
1456 case SA_EDDES: {
1457 /*--- An Enhanced Version of DES with Rapid Transition from RANS to LES in Separated Flows.
1458 Shur et al.
1459 Flow Turbulence Combust - 2015
1460 ---*/
1461
1462 su2double vortexTiltingMeasure = nodes->GetVortex_Tilting(iPoint);
1463
1464 const su2double omega = GeometryToolbox::Norm(3, vorticity);
1465
1466 su2double ratioOmega[MAXNDIM] = {};
1467
1468 for (auto iDim = 0; iDim < 3; iDim++){
1469 ratioOmega[iDim] = vorticity[iDim]/omega;
1470 }
1471
1472 const su2double deltaDDES = geometry->nodes->GetMaxLength(iPoint);
1473
1474 su2double ln_max = 0.0;
1475 for (const auto jPoint : geometry->nodes->GetPoints(iPoint)) {
1476 const auto coord_j = geometry->nodes->GetCoord(jPoint);
1477 su2double delta[MAXNDIM] = {};
1478 for (auto iDim = 0u; iDim < nDim; iDim++){
1479 delta[iDim] = fabs(coord_j[iDim] - coord_i[iDim]);
1480 }
1481 su2double ln[3];
1482 ln[0] = delta[1]*ratioOmega[2] - delta[2]*ratioOmega[1];
1483 ln[1] = delta[2]*ratioOmega[0] - delta[0]*ratioOmega[2];
1484 ln[2] = delta[0]*ratioOmega[1] - delta[1]*ratioOmega[0];
1485 const su2double aux_ln = sqrt(ln[0]*ln[0] + ln[1]*ln[1] + ln[2]*ln[2]);
1486 ln_max = max(ln_max, aux_ln);
1487 vortexTiltingMeasure += nodes->GetVortex_Tilting(jPoint);
1488 }
1489 vortexTiltingMeasure /= (nNeigh + 1);
1490
1491 const su2double f_kh = max(f_min,
1492 min(f_max,
1493 f_min + ((f_max - f_min)/(a2 - a1)) * (vortexTiltingMeasure - a1)));
1494
1495 const su2double r_d = (kinematicViscosityTurb+kinematicViscosity)/(uijuij*k2*pow(wallDistance, 2.0));
1496 const su2double f_d = 1.0-tanh(pow(8.0*r_d,3.0));
1497
1498 su2double maxDelta = (ln_max/sqrt(3.0)) * f_kh;
1499 if (f_d < 0.999){
1500 maxDelta = deltaDDES;
1501 }
1502
1503 const su2double distDES = constDES * maxDelta;
1504 lengthScale = wallDistance-f_d*max(0.0,(wallDistance-distDES));
1505
1506 break;
1507 }
1508 }
1509
1510 nodes->SetDES_LengthScale(iPoint, lengthScale);
1511
1512 }
1513 END_SU2_OMP_FOR
