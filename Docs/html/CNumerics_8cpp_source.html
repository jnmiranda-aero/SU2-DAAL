90void CNumerics::GetInviscidProjFlux(const su2double *val_density,
91 const su2double *val_velocity,
92 const su2double *val_pressure,
93 const su2double *val_enthalpy,
94 const su2double *val_normal,
95 su2double *val_Proj_Flux) const {
96
97 su2double rhou, rhov, rhow;
98
99 if (nDim == 2) {
100
101 rhou = (*val_density)*val_velocity[0];
102 rhov = (*val_density)*val_velocity[1];
103
104 val_Proj_Flux[0] = rhou*val_normal[0];
105 val_Proj_Flux[1] = (rhou*val_velocity[0]+(*val_pressure))*val_normal[0];
106 val_Proj_Flux[2] = rhou*val_velocity[1]*val_normal[0];
107 val_Proj_Flux[3] = rhou*(*val_enthalpy)*val_normal[0];
108
109 val_Proj_Flux[0] += rhov*val_normal[1];
110 val_Proj_Flux[1] += rhov*val_velocity[0]*val_normal[1];
111 val_Proj_Flux[2] += (rhov*val_velocity[1]+(*val_pressure))*val_normal[1];
112 val_Proj_Flux[3] += rhov*(*val_enthalpy)*val_normal[1];
113
114 }
115 else {
116
117 rhou = (*val_density)*val_velocity[0];
118 rhov = (*val_density)*val_velocity[1];
119 rhow = (*val_density)*val_velocity[2];
120
121 val_Proj_Flux[0] = rhou*val_normal[0];
122 val_Proj_Flux[1] = (rhou*val_velocity[0]+(*val_pressure))*val_normal[0];
123 val_Proj_Flux[2] = rhou*val_velocity[1]*val_normal[0];
124 val_Proj_Flux[3] = rhou*val_velocity[2]*val_normal[0];
125 val_Proj_Flux[4] = rhou*(*val_enthalpy)*val_normal[0];
126
127 val_Proj_Flux[0] += rhov*val_normal[1];
128 val_Proj_Flux[1] += rhov*val_velocity[0]*val_normal[1];
129 val_Proj_Flux[2] += (rhov*val_velocity[1]+(*val_pressure))*val_normal[1];
130 val_Proj_Flux[3] += rhov*val_velocity[2]*val_normal[1];
131 val_Proj_Flux[4] += rhov*(*val_enthalpy)*val_normal[1];
132
133 val_Proj_Flux[0] += rhow*val_normal[2];

135 val_Proj_Flux[2] += rhow*val_velocity[1]*val_normal[2];

175void CNumerics::GetInviscidProjJac(const su2double *val_velocity, const su2double *val_energy,
176 const su2double *val_normal, su2double val_scale,
177 su2double **val_Proj_Jac_Tensor) const {
178 const bool wasActive = AD::BeginPassive();
179 unsigned short iDim, jDim;
180 su2double sqvel, proj_vel, phi, a1, a2;
181
182 sqvel = 0.0; proj_vel = 0.0;
183 for (iDim = 0; iDim < nDim; iDim++) {
184 sqvel += val_velocity[iDim]*val_velocity[iDim];
185 proj_vel += val_velocity[iDim]*val_normal[iDim];
186 }
187
188 phi = 0.5*Gamma_Minus_One*sqvel;
189 a1 = Gamma*(*val_energy)-phi;
190 a2 = Gamma-1.0;
191
192 val_Proj_Jac_Tensor[0][0] = 0.0;
193 for (iDim = 0; iDim < nDim; iDim++)
194 val_Proj_Jac_Tensor[0][iDim+1] = val_scale*val_normal[iDim];
195 val_Proj_Jac_Tensor[0][nDim+1] = 0.0;
196
197 for (iDim = 0; iDim < nDim; iDim++) {
198 val_Proj_Jac_Tensor[iDim+1][0] = val_scale*(val_normal[iDim]*phi - val_velocity[iDim]*proj_vel);
199 for (jDim = 0; jDim < nDim; jDim++)
200 val_Proj_Jac_Tensor[iDim+1][jDim+1] = val_scale*(val_normal[jDim]*val_velocity[iDim]-a2*val_normal[iDim]*val_velocity[jDim]);
201 val_Proj_Jac_Tensor[iDim+1][iDim+1] += val_scale*proj_vel;
202 val_Proj_Jac_Tensor[iDim+1][nDim+1] = val_scale*a2*val_normal[iDim];
203 }
204
205 val_Proj_Jac_Tensor[nDim+1][0] = val_scale*proj_vel*(phi-a1);
206 for (iDim = 0; iDim < nDim; iDim++)
207 val_Proj_Jac_Tensor[nDim+1][iDim+1] = val_scale*(val_normal[iDim]*a1-a2*val_velocity[iDim]*proj_vel);
208 val_Proj_Jac_Tensor[nDim+1][nDim+1] = val_scale*Gamma*proj_vel;
209 AD::EndPassive(wasActive);
210}

213void CNumerics::GetInviscidProjJac(const su2double *val_velocity, const su2double *val_enthalpy,
214 const su2double *val_chi, const su2double *val_kappa,
215 const su2double *val_normal, su2double val_scale,
216 su2double **val_Proj_Jac_Tensor) const {
217 const bool wasActive = AD::BeginPassive();
218 unsigned short iDim, jDim;
219 su2double sqvel, proj_vel, phi, a1, a2;
220
221 sqvel = 0.0; proj_vel = 0.0;
222 for (iDim = 0; iDim < nDim; iDim++) {
223 sqvel += val_velocity[iDim]*val_velocity[iDim];
224 proj_vel += val_velocity[iDim]*val_normal[iDim];
225 }
226
227 phi = *val_chi + 0.5*sqvel*(*val_kappa);
228 a1 = *val_enthalpy;
229 a2 = *val_kappa;
230
231 val_Proj_Jac_Tensor[0][0] = 0.0;
232 for (iDim = 0; iDim < nDim; iDim++)
233 val_Proj_Jac_Tensor[0][iDim+1] = val_scale*val_normal[iDim];
234 val_Proj_Jac_Tensor[0][nDim+1] = 0.0;
235
236 for (iDim = 0; iDim < nDim; iDim++) {
237 val_Proj_Jac_Tensor[iDim+1][0] = val_scale*(val_normal[iDim]*phi - val_velocity[iDim]*proj_vel);
238 for (jDim = 0; jDim < nDim; jDim++)
239 val_Proj_Jac_Tensor[iDim+1][jDim+1] = val_scale*(val_normal[jDim]*val_velocity[iDim]-a2*val_normal[iDim]*val_velocity[jDim]);
240 val_Proj_Jac_Tensor[iDim+1][iDim+1] += val_scale*proj_vel;
241 val_Proj_Jac_Tensor[iDim+1][nDim+1] = val_scale*a2*val_normal[iDim];
242 }
243
244 val_Proj_Jac_Tensor[nDim+1][0] = val_scale*proj_vel*(phi-a1);
245 for (iDim = 0; iDim < nDim; iDim++)
246 val_Proj_Jac_Tensor[nDim+1][iDim+1] = val_scale*(val_normal[iDim]*a1-a2*val_velocity[iDim]*proj_vel);
247 val_Proj_Jac_Tensor[nDim+1][nDim+1] = val_scale*(a2+1)*proj_vel;
248 AD::EndPassive(wasActive);
249}

251void CNumerics::GetInviscidIncProjJac(const su2double *val_density, const su2double *val_velocity,
252 const su2double *val_betainc2, const su2double *val_cp,
253 const su2double *val_temperature, const su2double *val_dRhodT,
254 const su2double *val_normal, su2double val_scale,
255 su2double **val_Proj_Jac_Tensor) const {
256 const bool wasActive = AD::BeginPassive();
257 unsigned short iDim;
258 su2double proj_vel;
259
260 proj_vel = 0.0;
261 for (iDim = 0; iDim < nDim; iDim++)
262 proj_vel += val_velocity[iDim]*val_normal[iDim];
263
264 if (nDim == 2) {
265
266 val_Proj_Jac_Tensor[0][0] = val_scale*(proj_vel/(*val_betainc2));
267 val_Proj_Jac_Tensor[0][1] = val_scale*(val_normal[0]*(*val_density));
268 val_Proj_Jac_Tensor[0][2] = val_scale*(val_normal[1]*(*val_density));
269 val_Proj_Jac_Tensor[0][3] = val_scale*((*val_dRhodT)*proj_vel);
270
271 val_Proj_Jac_Tensor[1][0] = val_scale*(val_normal[0] + val_velocity[0]*proj_vel/(*val_betainc2));
272 val_Proj_Jac_Tensor[1][1] = val_scale*((*val_density)*(val_normal[0]*val_velocity[0] + proj_vel));
273 val_Proj_Jac_Tensor[1][2] = val_scale*(val_normal[1]*(*val_density)*val_velocity[0]);
274 val_Proj_Jac_Tensor[1][3] = val_scale*((*val_dRhodT)*val_velocity[0]*proj_vel);
275
276 val_Proj_Jac_Tensor[2][0] = val_scale*(val_normal[1] + val_velocity[1]*proj_vel/(*val_betainc2));
277 val_Proj_Jac_Tensor[2][1] = val_scale*(val_normal[0]*(*val_density)*val_velocity[1]);
278 val_Proj_Jac_Tensor[2][2] = val_scale*((*val_density)*(proj_vel + val_normal[1]*val_velocity[1]));
279 val_Proj_Jac_Tensor[2][3] = val_scale*((*val_dRhodT)*val_velocity[1]*proj_vel);
280
281 val_Proj_Jac_Tensor[3][0] = val_scale*((*val_cp)*(*val_temperature)*proj_vel/(*val_betainc2));
282 val_Proj_Jac_Tensor[3][1] = val_scale*((*val_cp)*(*val_temperature)*val_normal[0]*(*val_density));
283 val_Proj_Jac_Tensor[3][2] = val_scale*((*val_cp)*(*val_temperature)*val_normal[1]*(*val_density));
284 val_Proj_Jac_Tensor[3][3] = val_scale*((*val_cp)*((*val_temperature)*(*val_dRhodT) + (*val_density))*proj_vel);
285
286 } else {
287
288 val_Proj_Jac_Tensor[0][0] = val_scale*(proj_vel/(*val_betainc2));
289 val_Proj_Jac_Tensor[0][1] = val_scale*(val_normal[0]*(*val_density));
290 val_Proj_Jac_Tensor[0][2] = val_scale*(val_normal[1]*(*val_density));
291 val_Proj_Jac_Tensor[0][3] = val_scale*(val_normal[2]*(*val_density));
292 val_Proj_Jac_Tensor[0][4] = val_scale*((*val_dRhodT)*proj_vel);
293
294 val_Proj_Jac_Tensor[1][0] = val_scale*(val_normal[0] + val_velocity[0]*proj_vel/(*val_betainc2));
295 val_Proj_Jac_Tensor[1][1] = val_scale*((*val_density)*(val_normal[0]*val_velocity[0] + proj_vel));
296 val_Proj_Jac_Tensor[1][2] = val_scale*(val_normal[1]*(*val_density)*val_velocity[0]);
297 val_Proj_Jac_Tensor[1][3] = val_scale*(val_normal[2]*(*val_density)*val_velocity[0]);
298 val_Proj_Jac_Tensor[1][4] = val_scale*((*val_dRhodT)*val_velocity[0]*proj_vel);
299
300 val_Proj_Jac_Tensor[2][0] = val_scale*(val_normal[1] + val_velocity[1]*proj_vel/(*val_betainc2));
301 val_Proj_Jac_Tensor[2][1] = val_scale*(val_normal[0]*(*val_density)*val_velocity[1]);
302 val_Proj_Jac_Tensor[2][2] = val_scale*((*val_density)*(proj_vel + val_normal[1]*val_velocity[1]));
303 val_Proj_Jac_Tensor[2][3] = val_scale*(val_normal[2]*(*val_density)*val_velocity[1]);
304 val_Proj_Jac_Tensor[2][4] = val_scale*((*val_dRhodT)*val_velocity[1]*proj_vel);
305
306 val_Proj_Jac_Tensor[3][0] = val_scale*(val_normal[2] + val_velocity[2]*proj_vel/(*val_betainc2));
307 val_Proj_Jac_Tensor[3][1] = val_scale*(val_normal[0]*(*val_density)*val_velocity[2]);
308 val_Proj_Jac_Tensor[3][2] = val_scale*(val_normal[1]*(*val_density)*val_velocity[2]);
309 val_Proj_Jac_Tensor[3][3] = val_scale*((*val_density)*(proj_vel + val_normal[2]*val_velocity[2]));
310 val_Proj_Jac_Tensor[3][4] = val_scale*((*val_dRhodT)*val_velocity[2]*proj_vel);
311
312 val_Proj_Jac_Tensor[4][0] = val_scale*((*val_cp)*(*val_temperature)*proj_vel/(*val_betainc2));
313 val_Proj_Jac_Tensor[4][1] = val_scale*((*val_cp)*(*val_temperature)*val_normal[0]*(*val_density));
314 val_Proj_Jac_Tensor[4][2] = val_scale*((*val_cp)*(*val_temperature)*val_normal[1]*(*val_density));
315 val_Proj_Jac_Tensor[4][3] = val_scale*((*val_cp)*(*val_temperature)*val_normal[2]*(*val_density));
316 val_Proj_Jac_Tensor[4][4] = val_scale*((*val_cp)*((*val_temperature)*(*val_dRhodT) + (*val_density))*proj_vel);
317
318 }
319 AD::EndPassive(wasActive);

322void CNumerics::GetPreconditioner(const su2double *val_density, const su2double *val_velocity,
323 const su2double *val_betainc2, const su2double *val_cp,
324 const su2double *val_temperature, const su2double *val_drhodt,
325 su2double **val_Precon) const {
326 unsigned short iDim, jDim;
327
328 val_Precon[0][0] = 1.0/(*val_betainc2);
329 for (iDim = 0; iDim < nDim; iDim++)
330 val_Precon[iDim+1][0] = val_velocity[iDim]/(*val_betainc2);
331 val_Precon[nDim+1][0] = (*val_cp)*(*val_temperature)/(*val_betainc2);
332
333 for (jDim = 0; jDim < nDim; jDim++) {
334 val_Precon[0][jDim+1] = 0.0;
335 for (iDim = 0; iDim < nDim; iDim++) {
336 if (iDim == jDim) val_Precon[iDim+1][jDim+1] = (*val_density);
337 else val_Precon[iDim+1][jDim+1] = 0.0;
338 }
339 val_Precon[nDim+1][jDim+1] = 0.0;
340 }
341
342 val_Precon[0][nDim+1] = (*val_drhodt);
343 for (iDim = 0; iDim < nDim; iDim++)
344 val_Precon[iDim+1][nDim+1] = val_velocity[iDim]*(*val_drhodt);
345 val_Precon[nDim+1][nDim+1] = (*val_cp)*((*val_drhodt)*(*val_temperature) + (*val_density));

349void CNumerics::GetPreconditionedProjJac(const su2double *val_density, const su2double *val_lambda,
350 const su2double *val_betainc2, const su2double *val_normal,
351 su2double **val_invPrecon_A) const {
352 unsigned short iDim, jDim, kDim;
353
354 val_invPrecon_A[0][0] = val_lambda[nDim]/2.0 + val_lambda[nDim+1]/2.0;
355 for (iDim = 0; iDim < nDim; iDim++)
356 val_invPrecon_A[iDim+1][0] = val_normal[iDim]*(-val_lambda[nDim] + val_lambda[nDim+1])/(2.0*sqrt((*val_betainc2))*(*val_density));
357 val_invPrecon_A[nDim+1][0] = 0.0;
358
359 for (jDim = 0; jDim < nDim; jDim++) {
360 val_invPrecon_A[0][jDim+1] = sqrt((*val_betainc2))*val_normal[jDim]*(*val_density)*(-val_lambda[nDim] + val_lambda[nDim+1])/(2.0);
361 for (iDim = 0; iDim < nDim; iDim++) {
362 if (iDim == jDim) {
363 val_invPrecon_A[iDim+1][jDim+1] = (val_lambda[nDim] + val_lambda[nDim+1])*val_normal[iDim]*val_normal[iDim]/2.0;
364 for (kDim = 0; kDim < nDim; kDim++) {
365 if (kDim != iDim) val_invPrecon_A[iDim+1][jDim+1] += 2.0*val_lambda[0]*val_normal[kDim]*val_normal[kDim];
366 }
367 }
368 else {
369 val_invPrecon_A[iDim+1][jDim+1] = val_normal[iDim]*val_normal[jDim]*(-2.0*val_lambda[0] + val_lambda[nDim] + val_lambda[nDim+1])/(2.0);
370 }
371 }
372 val_invPrecon_A[nDim+1][jDim+1] = 0.0;
373 }
374
375 val_invPrecon_A[0][nDim+1] = 0.0;
376 for (iDim = 0; iDim < nDim; iDim++)
377 val_invPrecon_A[iDim+1][nDim+1] = 0.0;
378 val_invPrecon_A[nDim+1][nDim+1] = val_lambda[nDim-1];
379
380}

382void CNumerics::GetPMatrix(const su2double *val_density, const su2double *val_velocity,
383 const su2double *val_soundspeed, const su2double *val_normal,
384 su2double **val_p_tensor) const {
385
386 su2double sqvel, rhooc, rhoxc;
387 //su2double c2;
388
389 rhooc = *val_density / *val_soundspeed;
390 rhoxc = *val_density * *val_soundspeed;
391 //c2 = *val_soundspeed * *val_soundspeed;
392
393 if (nDim == 2) {
394
395 sqvel = val_velocity[0]*val_velocity[0]+val_velocity[1]*val_velocity[1];
396
397 val_p_tensor[0][0]=1.0;
398 val_p_tensor[0][1]=0.0;
399 val_p_tensor[0][2]=0.5*rhooc;
400 val_p_tensor[0][3]=0.5*rhooc;
401
402 val_p_tensor[1][0]=val_velocity[0];
403 val_p_tensor[1][1]=*val_density*val_normal[1];
404 val_p_tensor[1][2]=0.5*(val_velocity[0]*rhooc+val_normal[0]**val_density);
405 val_p_tensor[1][3]=0.5*(val_velocity[0]*rhooc-val_normal[0]**val_density);
406
407 val_p_tensor[2][0]=val_velocity[1];
408 val_p_tensor[2][1]=-*val_density*val_normal[0];
409 val_p_tensor[2][2]=0.5*(val_velocity[1]*rhooc+val_normal[1]**val_density);
410 val_p_tensor[2][3]=0.5*(val_velocity[1]*rhooc-val_normal[1]**val_density);
411
412 val_p_tensor[3][0]=0.5*sqvel;
413 val_p_tensor[3][1]=*val_density*val_velocity[0]*val_normal[1]-*val_density*val_velocity[1]*val_normal[0];
414 val_p_tensor[3][2]=0.5*(0.5*sqvel*rhooc+*val_density*val_velocity[0]*val_normal[0]+*val_density*val_velocity[1]*val_normal[1]+rhoxc/Gamma_Minus_One);
415 val_p_tensor[3][3]=0.5*(0.5*sqvel*rhooc-*val_density*val_velocity[0]*val_normal[0]-*val_density*val_velocity[1]*val_normal[1]+rhoxc/Gamma_Minus_One);
416
417 }
418 else {
419
420 sqvel = val_velocity[0]*val_velocity[0]+val_velocity[1]*val_velocity[1]+val_velocity[2]*val_velocity[2];
421
422 val_p_tensor[0][0]=val_normal[0];
423 val_p_tensor[0][1]=val_normal[1];
424 val_p_tensor[0][2]=val_normal[2];
425 val_p_tensor[0][3]=0.5*rhooc;
426 val_p_tensor[0][4]=0.5*rhooc;

1618 m[0] = 0.0;
