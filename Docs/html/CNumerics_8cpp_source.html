175 void CNumerics::GetInviscidProjJac(const su2double *val_velocity, const su2double *val_energy,
176  const su2double *val_normal, su2double val_scale,
177  su2double **val_Proj_Jac_Tensor) const {
178  const bool wasActive = AD::BeginPassive();
179  unsigned short iDim, jDim;
180  su2double sqvel, proj_vel, phi, a1, a2;
181 
182  sqvel = 0.0; proj_vel = 0.0;
183  for (iDim = 0; iDim < nDim; iDim++) {
184  sqvel += val_velocity[iDim]*val_velocity[iDim];
185  proj_vel += val_velocity[iDim]*val_normal[iDim];
186  }
187 
188  phi = 0.5*Gamma_Minus_One*sqvel;
189  a1 = Gamma*(*val_energy)-phi;
190  a2 = Gamma-1.0;
191 
192  val_Proj_Jac_Tensor[0][0] = 0.0;
193  for (iDim = 0; iDim < nDim; iDim++)
194  val_Proj_Jac_Tensor[0][iDim+1] = val_scale*val_normal[iDim];
195  val_Proj_Jac_Tensor[0][nDim+1] = 0.0;
196 
197  for (iDim = 0; iDim < nDim; iDim++) {
198  val_Proj_Jac_Tensor[iDim+1][0] = val_scale*(val_normal[iDim]*phi - val_velocity[iDim]*proj_vel);
199  for (jDim = 0; jDim < nDim; jDim++)
200  val_Proj_Jac_Tensor[iDim+1][jDim+1] = val_scale*(val_normal[jDim]*val_velocity[iDim]-a2*val_normal[iDim]*val_velocity[jDim]);
201  val_Proj_Jac_Tensor[iDim+1][iDim+1] += val_scale*proj_vel;
202  val_Proj_Jac_Tensor[iDim+1][nDim+1] = val_scale*a2*val_normal[iDim];
203  }
204 
205  val_Proj_Jac_Tensor[nDim+1][0] = val_scale*proj_vel*(phi-a1);
206  for (iDim = 0; iDim < nDim; iDim++)
207  val_Proj_Jac_Tensor[nDim+1][iDim+1] = val_scale*(val_normal[iDim]*a1-a2*val_velocity[iDim]*proj_vel);
208  val_Proj_Jac_Tensor[nDim+1][nDim+1] = val_scale*Gamma*proj_vel;
209  AD::EndPassive(wasActive);
210 }

214  const su2double *val_chi, const su2double *val_kappa,

240  val_Proj_Jac_Tensor[iDim+1][iDim+1] += val_scale*proj_vel;

382 void CNumerics::GetPMatrix(const su2double *val_density, const su2double *val_velocity,
383  const su2double *val_soundspeed, const su2double *val_normal,
384  su2double **val_p_tensor) const {
385 
386  su2double sqvel, rhooc, rhoxc;
387  //su2double c2;
388 
389  rhooc = *val_density / *val_soundspeed;
390  rhoxc = *val_density * *val_soundspeed;
391  //c2 = *val_soundspeed * *val_soundspeed;
392 
393  if (nDim == 2) {
394 
395  sqvel = val_velocity[0]*val_velocity[0]+val_velocity[1]*val_velocity[1];
396 
397  val_p_tensor[0][0]=1.0;
398  val_p_tensor[0][1]=0.0;
399  val_p_tensor[0][2]=0.5*rhooc;
400  val_p_tensor[0][3]=0.5*rhooc;
401 
402  val_p_tensor[1][0]=val_velocity[0];
403  val_p_tensor[1][1]=*val_density*val_normal[1];
404  val_p_tensor[1][2]=0.5*(val_velocity[0]*rhooc+val_normal[0]**val_density);
405  val_p_tensor[1][3]=0.5*(val_velocity[0]*rhooc-val_normal[0]**val_density);
406 
407  val_p_tensor[2][0]=val_velocity[1];
408  val_p_tensor[2][1]=-*val_density*val_normal[0];
409  val_p_tensor[2][2]=0.5*(val_velocity[1]*rhooc+val_normal[1]**val_density);
410  val_p_tensor[2][3]=0.5*(val_velocity[1]*rhooc-val_normal[1]**val_density);
411
