32 void CDriver::PreprocessPythonInterface(CConfig** config, CGeometry**** geometry, CSolver***** solver) {
33  int rank = MASTER_NODE;
34  SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &rank);
35 
36  /*--- Initialize boundary conditions customization, this is achieved through the Python wrapper. --- */
37  for (iZone = 0; iZone < nZone; iZone++) {
38  if (config[iZone]->GetnMarker_PyCustom() > 0) {
39  if (rank == MASTER_NODE) {
40  cout << "----------------- Python Interface Preprocessing ( Zone " << iZone << " ) -----------------\n";
41  cout << "Setting customized boundary conditions for zone " << iZone << endl;
42  }
43  for (iMesh = 0; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
44  geometry[iZone][INST_0][iMesh]->SetCustomBoundary(config[iZone]);
45  }
46  geometry[iZone][INST_0][MESH_0]->UpdateCustomBoundaryConditions(geometry[iZone][INST_0], config[iZone]);
47 
48  if (solver[iZone][INST_0][MESH_0][FLOW_SOL]) {
49  solver[iZone][INST_0][MESH_0][FLOW_SOL]->UpdateCustomBoundaryConditions(geometry[iZone][INST_0], solver[iZone][INST_0], config[iZone]);
50  }
51  }
52  }
53 }

83 void CDriver::SetInletAngle(unsigned short iMarker, passivedouble alpha) {
84  const su2double alpha_rad = alpha * PI_NUMBER / 180.0;
85 
86  const auto* geometry = geometry_container[selected_zone][INST_0][MESH_0];
87  auto* flow_solver = solver_container[selected_zone][INST_0][MESH_0][FLOW_SOL];
88 
89  for (auto iVertex = 0ul; iVertex < geometry->nVertex[iMarker]; ++iVertex) {
90  flow_solver->SetInletFlowDir(iMarker, iVertex, 0, cos(alpha_rad));
91  flow_solver->SetInletFlowDir(iMarker, iVertex, 1, sin(alpha_rad));
92  if (geometry->GetnDim() == 3) flow_solver->SetInletFlowDir(iMarker, iVertex, 2, 0);
93  }
94 }

110 void CSinglezoneDriver::SetInitialMesh() {
111  DynamicMeshUpdate(0);
112 
113  SU2_OMP_PARALLEL {
114  for (auto iMesh = 0u; iMesh <= main_config->GetnMGLevels(); iMesh++) {
115  SU2_OMP_FOR_STAT(roundUpDiv(geometry_container[selected_zone][INST_0][iMesh]->GetnPoint(), omp_get_max_threads()))
116  for (auto iPoint = 0ul; iPoint < geometry_container[selected_zone][INST_0][iMesh]->GetnPoint(); iPoint++) {
117  /*--- Overwrite fictitious velocities. ---*/
118  su2double Grid_Vel[3] = {0.0, 0.0, 0.0};
119 
120  /*--- Set the grid velocity for this coarse node. ---*/
121  geometry_container[selected_zone][INST_0][iMesh]->nodes->SetGridVel(iPoint, Grid_Vel);
122  }
123  END_SU2_OMP_FOR
124  /*--- Push back the volume. ---*/
125  geometry_container[selected_zone][INST_0][iMesh]->nodes->SetVolume_n();
126  geometry_container[selected_zone][INST_0][iMesh]->nodes->SetVolume_nM1();
127  }
128  /*--- Push back the solution so that there is no fictitious velocity at the next step. ---*/
129  solver_container[selected_zone][INST_0][MESH_0][MESH_SOL]->GetNodes()->Set_Solution_time_n();
130  solver_container[selected_zone][INST_0][MESH_0][MESH_SOL]->GetNodes()->Set_Solution_time_n1();
131  }
132  END_SU2_OMP_PARALLEL
133 }
