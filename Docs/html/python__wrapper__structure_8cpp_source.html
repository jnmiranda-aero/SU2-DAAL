32void CDriver::PreprocessPythonInterface(CConfig** config, CGeometry**** geometry, CSolver***** solver) {
33 int rank = MASTER_NODE;
34 SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &rank);
35
36 /*--- Initialize boundary conditions customization, this is achieved through the Python wrapper. --- */
37 for (iZone = 0; iZone < nZone; iZone++) {
38 if (config[iZone]->GetnMarker_PyCustom() > 0) {
39 if (rank == MASTER_NODE) cout << "----------------- Python Interface Preprocessing ( Zone " << iZone << " ) -----------------" << endl;
40
41 if (rank == MASTER_NODE) cout << "Setting customized boundary conditions for zone " << iZone << endl;
42 for (iMesh = 0; iMesh <= config[iZone]->GetnMGLevels(); iMesh++) {
43 geometry[iZone][INST_0][iMesh]->SetCustomBoundary(config[iZone]);
44 }
45 geometry[iZone][INST_0][MESH_0]->UpdateCustomBoundaryConditions(geometry[iZone][INST_0], config[iZone]);
46
47 if ((config[iZone]->GetKind_Solver() == MAIN_SOLVER::EULER) ||
48 (config[iZone]->GetKind_Solver() == MAIN_SOLVER::NAVIER_STOKES) ||
49 (config[iZone]->GetKind_Solver() == MAIN_SOLVER::RANS) ||
50 (config[iZone]->GetKind_Solver() == MAIN_SOLVER::INC_EULER) ||
51 (config[iZone]->GetKind_Solver() == MAIN_SOLVER::INC_NAVIER_STOKES) ||
52 (config[iZone]->GetKind_Solver() == MAIN_SOLVER::INC_RANS) ||
53 (config[iZone]->GetKind_Solver() == MAIN_SOLVER::NEMO_EULER) ||
54 (config[iZone]->GetKind_Solver() == MAIN_SOLVER::NEMO_NAVIER_STOKES)) {
55 solver[iZone][INST_0][MESH_0][FLOW_SOL]->UpdateCustomBoundaryConditions(geometry[iZone][INST_0], config[iZone]);
56 }
57 }
58 }
59}

114void CSinglezoneDriver::SetInitialMesh() {
115 DynamicMeshUpdate(0);
116
117 SU2_OMP_PARALLEL {
118 for (iMesh = 0u; iMesh <= main_config->GetnMGLevels(); iMesh++) {
119 SU2_OMP_FOR_STAT(roundUpDiv(geometry_container[selected_zone][INST_0][iMesh]->GetnPoint(), omp_get_max_threads()))
120 for (auto iPoint = 0ul; iPoint < geometry_container[selected_zone][INST_0][iMesh]->GetnPoint(); iPoint++) {
121 /*--- Overwrite fictitious velocities. ---*/
122 su2double Grid_Vel[3] = {0.0, 0.0, 0.0};
123
124 /*--- Set the grid velocity for this coarse node. ---*/
125 geometry_container[selected_zone][INST_0][iMesh]->nodes->SetGridVel(iPoint, Grid_Vel);
126 }
127 END_SU2_OMP_FOR
128 /*--- Push back the volume. ---*/
129 geometry_container[selected_zone][INST_0][iMesh]->nodes->SetVolume_n();
130 geometry_container[selected_zone][INST_0][iMesh]->nodes->SetVolume_nM1();
131 }
132 /*--- Push back the solution so that there is no fictitious velocity at the next step. ---*/
133 solver_container[selected_zone][INST_0][MESH_0][MESH_SOL]->GetNodes()->Set_Solution_time_n();
134 solver_container[selected_zone][INST_0][MESH_0][MESH_SOL]->GetNodes()->Set_Solution_time_n1();
135 }
136 END_SU2_OMP_PARALLEL
137}
