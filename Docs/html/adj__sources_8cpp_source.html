34void CSourceAxisymmetric_AdjFlow::ComputeResidual(su2double *val_residual, su2double **Jacobian_ii, CConfig *config) {
35
36 su2double yinv;
37 su2double Jacobian_Axisymmetric[4][4];
38
39 if (Coord_i[1] > 0.0) yinv = 1.0/Coord_i[1];
40 else yinv = 0.0;
41
42 Jacobian_Axisymmetric[0][0] = 0;
43 Jacobian_Axisymmetric[0][1] = 0;
44 Jacobian_Axisymmetric[0][2] = 1.;
45 Jacobian_Axisymmetric[0][3] = 0;
46
47 Jacobian_Axisymmetric[1][0] = -U_i[1]*U_i[2]/(U_i[0]*U_i[0]);
48 Jacobian_Axisymmetric[1][1] = U_i[2]/U_i[0];
49 Jacobian_Axisymmetric[1][2] = U_i[1]/U_i[0];
50 Jacobian_Axisymmetric[1][3] = 0;
51
52 Jacobian_Axisymmetric[2][0] = -U_i[2]*U_i[2]/(U_i[0]*U_i[0]);
53 Jacobian_Axisymmetric[2][1] = 0;
54 Jacobian_Axisymmetric[2][2] = 2*U_i[2]/U_i[0];
55 Jacobian_Axisymmetric[2][3] = 0;
56
57 Jacobian_Axisymmetric[3][0] = -Gamma*U_i[2]*U_i[3]/(U_i[0]*U_i[0]) + (Gamma-1)*U_i[2]*(U_i[1]*U_i[1]+U_i[2]*U_i[2])/(U_i[0]*U_i[0]*U_i[0]);
58 Jacobian_Axisymmetric[3][1] = -(Gamma-1)*U_i[2]*U_i[1]/(U_i[0]*U_i[0]);
59 Jacobian_Axisymmetric[3][2] = Gamma*U_i[3]/U_i[0] - 1/2*(Gamma-1)*( (U_i[1]*U_i[1]+U_i[2]*U_i[2])/(U_i[0]*U_i[0]) + 2*U_i[2]*U_i[2]/(U_i[0]*U_i[0]) );
60 Jacobian_Axisymmetric[3][3] = Gamma*U_i[2]/U_i[0];
61
62 for (int iVar=0; iVar<4; iVar++)
63 for (int jVar=0; jVar<4; jVar++)
64 Jacobian_Axisymmetric[iVar][jVar] *= yinv*Volume;
65
66 /* -- Residual = transpose(Jacobian) * psi --*/
67 for (int iVar = 0; iVar < nVar; iVar++) {
68 val_residual[iVar] = 0.0;
69 for (int jVar = 0; jVar < nVar; jVar++) {
70 val_residual[iVar] += Jacobian_Axisymmetric[jVar][iVar]*Psi_i[jVar];
71 Jacobian_ii[iVar][jVar] = Jacobian_Axisymmetric[jVar][iVar];
72 }
73 }
74}

102void CSourceConservative_AdjFlow::ComputeResidual (su2double *val_residual, CConfig *config) {
103 unsigned short iDim, jDim, iVar;
104 su2double rho, nu, Ji, fv1, fv2, Omega, Shat, dist_sq, Ji_2, Ji_3, one_o_oneplusJifv1;
105 su2double r, g, g_6, glim, dfw_g, dg_r, dr_nuhat, dr_Shat, Ms_coeff, invOmega;
106
107 su2double cv1_3 = 7.1*7.1*7.1;
108 su2double k2 = 0.41*0.41;
109 su2double cb1 = 0.1355;
110 su2double cw2 = 0.3;
111 su2double cw3_6 = pow(2.0,6.0);
112 su2double sigma = 2./3.;
113 su2double cb2 = 0.622;
114 su2double cw1 = cb1/k2+(1+cb2)/sigma;
115
116 for (iVar = 0; iVar < nVar; iVar++) {
117 Residual_i[iVar] = 0.0;
118 Residual_j[iVar] = 0.0;
119 }
120
121 /*--- iPoint ---*/
122
123 /*--- Density and velocities ---*/
124
125 rho = U_i[0];
126 for (iDim = 0; iDim < nDim; iDim++)
127 Velocity[iDim] = U_i[iDim+1]/rho;
128
129 /*--- Vorticity ---*/
130
131 Omega = (PrimVar_Grad_i[1][1]-PrimVar_Grad_i[2][0])*(PrimVar_Grad_i[1][1]-PrimVar_Grad_i[2][0]);
132 if (nDim == 3) Omega += (PrimVar_Grad_i[1][2]-PrimVar_Grad_i[3][0])*(PrimVar_Grad_i[1][2]-PrimVar_Grad_i[3][0]) +
133 (PrimVar_Grad_i[2][2]-PrimVar_Grad_i[3][1])*(PrimVar_Grad_i[2][2]-PrimVar_Grad_i[3][1]);
134 Omega = sqrt(Omega);
135 invOmega = 1.0/(Omega + TURB_EPS);
136
137 /*--- Compute Ms_coeff -> coming from partial derivatives ---*/
138
139 Ms_coeff = 0.0;
140 if (dist_i > 0) {
141 dist_sq = dist_i*dist_i;
142 nu = Laminar_Viscosity_i/rho;
143 Ji = ScalarVar_i[0]/nu;
144 Ji_2 = Ji*Ji;
145 Ji_3 = Ji_2*Ji;
146 fv1 = Ji_3/(Ji_3+cv1_3);
147 one_o_oneplusJifv1 = 1.0/(1.0+Ji*fv1);
148 fv2 = 1.0 - Ji*one_o_oneplusJifv1;
149 Shat = max(Omega + ScalarVar_i[0]*fv2/(k2*dist_sq), TURB_EPS);
150
151 r = min(ScalarVar_i[0]/(Shat*k2*dist_sq),10.);
152 g = r + cw2*(pow(r,6.)-r);
153 g_6 = pow(g,6.);
154 glim = pow((1+cw3_6)/(g_6+cw3_6),1./6.);
155
156 dfw_g = glim*cw3_6/(g_6+cw3_6);
157 dg_r = 1.0 + cw2*(6.0*pow(r,5.0)-1.0);
158 dr_nuhat = 1.0/(Shat*k2*dist_sq);
159 dr_Shat = -dr_nuhat*ScalarVar_i[0]/Shat;
160
161 Ms_coeff = (cb1*ScalarVar_i[0]-cw1*ScalarVar_i[0]*ScalarVar_i[0]/dist_sq*dfw_g*dg_r*dr_Shat);
162 }
163 Ms_coeff *= TurbPsi_i[0]*invOmega/rho;
164
165 /*--- Compute residual of iPoint ---*/
166
167 for (iDim = 0; iDim < nDim; iDim++) {
168 for (jDim = 0; jDim < nDim; jDim++) {
169 Residual_i[0] -= Ms_coeff*(Velocity[jDim]*PrimVar_Grad_i[jDim+1][iDim]*Normal[iDim] -
170 Velocity[jDim]*PrimVar_Grad_i[iDim+1][jDim]*Normal[iDim]);
171 Residual_i[iDim+1] += Ms_coeff*(PrimVar_Grad_i[iDim+1][jDim]*Normal[jDim] -
172 PrimVar_Grad_i[jDim+1][iDim]*Normal[jDim]);
173 }
174 }
175
176 /*--- jPoint ---*/
177
178 /*--- Density and velocities ---*/
179
180 rho = U_j[0];
181 for (iDim = 0; iDim < nDim; iDim++)
182 Velocity[iDim] = U_j[iDim+1]/rho;
183
184 /*--- Vorticity ---*/
185
186 Omega = (PrimVar_Grad_j[1][1]-PrimVar_Grad_j[2][0])*(PrimVar_Grad_j[1][1]-PrimVar_Grad_j[2][0]);
187 if (nDim == 3) Omega += (PrimVar_Grad_j[1][2]-PrimVar_Grad_j[3][0])*(PrimVar_Grad_j[1][2]-PrimVar_Grad_j[3][0]) +
188 (PrimVar_Grad_j[2][2]-PrimVar_Grad_j[3][1])*(PrimVar_Grad_j[2][2]-PrimVar_Grad_j[3][1]);
189 Omega = sqrt(Omega);
190 invOmega = 1.0/(Omega + TURB_EPS);
191
192 /*--- Compute Ms_coeff -> coming from partial derivatives ---*/
193
194 Ms_coeff = 0.0;
195 if (dist_j > 0) {
196 dist_sq = dist_j*dist_j;
197 nu = Laminar_Viscosity_j/rho;
198 Ji = ScalarVar_j[0]/nu;
199 Ji_2 = Ji*Ji;
200 Ji_3 = Ji_2*Ji;
201 fv1 = Ji_3/(Ji_3+cv1_3);
202 one_o_oneplusJifv1 = 1.0/(1.0+Ji*fv1);
203 fv2 = 1.0 - Ji*one_o_oneplusJifv1;
204 Shat = max(Omega + ScalarVar_j[0]*fv2/(k2*dist_sq), TURB_EPS);
205
206 r = min(ScalarVar_j[0]/(Shat*k2*dist_sq),10.);
207 g = r + cw2*(pow(r,6.)-r);
208 g_6 = pow(g,6.);
209 glim = pow((1+cw3_6)/(g_6+cw3_6),1./6.);
210
211 dfw_g = glim*cw3_6/(g_6+cw3_6);
212 dg_r = 1.0 + cw2*(6.0*pow(r,5.0)-1.0);
213 dr_nuhat = 1.0/(Shat*k2*dist_sq);
214 dr_Shat = -dr_nuhat*ScalarVar_j[0]/Shat;
215
216 Ms_coeff = (cb1*ScalarVar_j[0]-cw1*ScalarVar_j[0]*ScalarVar_j[0]/dist_sq*dfw_g*dg_r*dr_Shat);
217 }
218 Ms_coeff *= TurbPsi_j[0]*invOmega/rho;
219
220 /*--- Compute residual of jPoint ---*/
221
222 for (iDim = 0; iDim < nDim; iDim++) {
223 for (jDim = 0; jDim < nDim; jDim++) {
224 Residual_j[0] -= Ms_coeff*(Velocity[jDim]*PrimVar_Grad_j[jDim+1][iDim]*Normal[iDim] -
225 Velocity[jDim]*PrimVar_Grad_j[iDim+1][jDim]*Normal[iDim]);
226 Residual_j[iDim+1] += Ms_coeff*(PrimVar_Grad_j[iDim+1][jDim]*Normal[jDim] -
227 PrimVar_Grad_j[jDim+1][iDim]*Normal[jDim]);
228 }
229 }
230
231 /*--- Compute the mean residual ---*/
232
233 for (iVar = 0; iVar < nVar; iVar++)
234 val_residual[iVar] = 0.5*(Residual_i[iVar] + Residual_j[iVar]);
235
236}

243void CSourceRotatingFrame_AdjFlow::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i, CConfig *config) {
244
245 unsigned short iDim, iVar, jVar;
246 su2double Omega[3] = {0,0,0}, Phi[3] = {0,0,0};
247 bool implicit = (config->GetKind_TimeIntScheme_AdjFlow() == EULER_IMPLICIT);
248
249 /*--- Retrieve the angular velocity vector from config. ---*/
250
251 for (iDim = 0; iDim < 3; iDim++){
252 Omega[iDim] = config->GetRotation_Rate(iDim)/config->GetOmega_Ref();
253 }
254
255 /*--- Get the adjoint velocity vector at the current node. ---*/
256
257 for (iDim = 0; iDim < nDim; iDim++)
258 Phi[iDim] = Psi_i[iDim+1];
259
260 /*--- Compute the source term as the Jacobian of the rotating frame
261 source term multiplied by the adjoint state and the dual cell volume. ---*/
262
263 if (nDim == 2) {
264 val_residual[0] = 0.0;
265 val_residual[1] = Omega[2]*Phi[1]*Volume;
266 val_residual[2] = -Omega[2]*Phi[0]*Volume;
267 val_residual[3] = 0.0;
268 } else {
269 val_residual[0] = 0.0;
270 val_residual[1] = (Omega[2]*Phi[1] - Omega[1]*Phi[2])*Volume;
271 val_residual[2] = (Omega[0]*Phi[2] - Omega[2]*Phi[0])*Volume;
272 val_residual[3] = (Omega[1]*Phi[0] - Omega[0]*Phi[1])*Volume;
273 val_residual[4] = 0.0;
274 }
275
276 /*--- Calculate the source term Jacobian ---*/
277
278 if (implicit) {
279 for (iVar = 0; iVar < nVar; iVar++)
280 for (jVar = 0; jVar < nVar; jVar++)
281 val_Jacobian_i[iVar][jVar] = 0.0;
282 if (nDim == 2) {
283 val_Jacobian_i[1][2] = Omega[2]*Volume;
284 val_Jacobian_i[2][1] = -Omega[2]*Volume;
285 } else {
286 val_Jacobian_i[1][2] = Omega[2]*Volume;
287 val_Jacobian_i[1][3] = -Omega[1]*Volume;
288 val_Jacobian_i[2][1] = -Omega[2]*Volume;
289 val_Jacobian_i[2][3] = Omega[0]*Volume;
290 val_Jacobian_i[3][1] = Omega[1]*Volume;
291 val_Jacobian_i[3][2] = -Omega[0]*Volume;
292 }
293 }

355void CSourceViscous_AdjFlow::ComputeResidual (su2double *val_residual, CConfig *config) {
356
357 unsigned short iDim, jDim;
358
359// su2double Temperature = V_i[0];
360 su2double Pressure = V_i[nDim+1];
361 su2double Density = V_i[nDim+2];
362// su2double Enthalpy = V_i[nDim+3];
363 su2double Laminar_Viscosity = V_i[nDim+5];
364 su2double Eddy_Viscosity = V_i[nDim+6];
365
366// su2double Energy = Enthalpy - Pressure/Density;
367 su2double invDensity = 1.0/Density;
368 su2double invDensitysq = invDensity*invDensity;
369 su2double invDensitycube = invDensitysq*invDensity;
370 su2double Prandtl_Lam = config->GetPrandtl_Lam();
371 su2double Prandtl_Turb = config->GetPrandtl_Turb();
372 su2double mu_tot_1 = Laminar_Viscosity + Eddy_Viscosity;
373 su2double mu_tot_2 = Laminar_Viscosity/Prandtl_Lam + Eddy_Viscosity/Prandtl_Turb;
374// su2double Gas_Constant = config->GetGas_ConstantND();
375
376 /*--- Required gradients of the flow variables, point j ---*/
377
378 for (iDim = 0; iDim < nDim; iDim++) {
379
380 /*--- Gradient density ---*/
381
382 GradDensity[iDim] = PrimVar_Grad_i[nDim+2][iDim];
383
384 /*--- Gradient (1/rho) ---*/
385
386 GradInvDensity[iDim] = -GradDensity[iDim]*invDensitysq;
387
388 /*--- Computation of the derivatives of P/(Density^2) ---*/
389
390 dPoDensity2[iDim] = (PrimVar_Grad_i[nVar-1][iDim]*Density - 2.0*GradDensity[iDim]*Pressure)*invDensitycube;
391
392 /*--- Abbreviations: alpha, beta, sigma_5_vec ---*/
393
394 alpha[iDim] = Gamma*mu_tot_2*GradInvDensity[iDim];
395 beta[iDim] = Gamma*mu_tot_2*dPoDensity2[iDim]/Gamma_Minus_One;
396 Sigma_5_vec[iDim] = Gamma*mu_tot_2*PsiVar_Grad_i[nVar-1][iDim];
397
398 }
399
400 /*--- Definition of tensors and derivatives of velocity over density ---*/
401
402 su2double div_vel = 0.0, div_phi = 0.0, vel_gradpsi5 = 0.0;
403 for (iDim = 0; iDim < nDim; iDim++) {
404 div_vel += PrimVar_Grad_i[iDim+1][iDim];
405 div_phi += PsiVar_Grad_i[iDim+1][iDim];
406 vel_gradpsi5 += V_i[iDim+1]*PsiVar_Grad_i[nVar-1][iDim];
407 for (jDim = 0; jDim < nDim; jDim++) {
408 sigma[iDim][jDim] = mu_tot_1*(PrimVar_Grad_i[iDim+1][jDim]+PrimVar_Grad_i[jDim+1][iDim]);
409 Sigma_phi[iDim][jDim] = mu_tot_1*(PsiVar_Grad_i[iDim+1][jDim]+PsiVar_Grad_i[jDim+1][iDim]);
410 Sigma_5_Tensor[iDim][jDim] = mu_tot_1*(V_i[jDim+1]*PsiVar_Grad_i[nVar-1][iDim]+V_i[iDim+1]*PsiVar_Grad_i[nVar-1][jDim]);
411 GradVel_o_Rho[iDim][jDim] = (PrimVar_Grad_i[iDim+1][jDim]*Density - V_i[iDim+1]*GradDensity[jDim])*invDensitysq;
412 }
413 }
414
415 for (iDim = 0; iDim < nDim; iDim++) {
416 sigma[iDim][iDim] -= TWO3*mu_tot_1*div_vel;
417 Sigma_phi[iDim][iDim] -= TWO3*mu_tot_1*div_phi;
418 Sigma_5_Tensor[iDim][iDim] -= TWO3*mu_tot_1*vel_gradpsi5;
419 }
420
421 for (iDim = 0; iDim < nDim; iDim++) {
422 for (jDim = 0; jDim < nDim; jDim++) {
423 Sigma[iDim][jDim] = Sigma_phi[iDim][jDim] + Sigma_5_Tensor[iDim][jDim];
424 }
425 }
426
427 /*--- Vector-Tensors products ---*/
428
429 su2double gradT_gradpsi5 = 0.0, sigma_gradpsi = 0.0, vel_sigma_gradpsi5 = 0.0;
430 for (iDim = 0; iDim < nDim; iDim++) {
431 gradT_gradpsi5 += PrimVar_Grad_i[0][iDim]*PsiVar_Grad_i[nVar-1][iDim];
432 for (jDim = 0; jDim < nDim; jDim++) {
433 sigma_gradpsi += sigma[iDim][jDim]*PsiVar_Grad_i[jDim+1][iDim];
434 vel_sigma_gradpsi5 += V_i[iDim+1]*sigma[iDim][jDim]*PsiVar_Grad_i[nVar-1][jDim];
435 }
436 }
437
438 /*--- Residuals ---*/
439
440 su2double alpha_gradpsi5 = 0.0, beta_gradpsi5 = 0.0, Sigma_gradvel_o_rho = 0.0, Sigma5_vel_gradvel = 0.0, sq_vel = 0.0;
441 for (iDim = 0; iDim < nDim; iDim++) {
442 alpha_gradpsi5 += alpha[iDim]*PsiVar_Grad_i[nVar-1][iDim];
443 beta_gradpsi5 += beta[iDim]*PsiVar_Grad_i[nVar-1][iDim];
444 for (jDim = 0; jDim < nDim; jDim++) {
445 Sigma_gradvel_o_rho += Sigma[iDim][jDim]*GradVel_o_Rho[iDim][jDim];
446 Sigma5_vel_gradvel += Sigma_5_vec[iDim]*(V_i[jDim+1]*PrimVar_Grad_i[jDim+1][iDim]);
447 }
448 sq_vel += V_i[iDim+1]*V_i[iDim+1];
449 }
450
451 val_residual[0] = (-vel_sigma_gradpsi5*invDensity + 0.5*sq_vel*alpha_gradpsi5 -
452 beta_gradpsi5) * Volume;
453 for (iDim = 0; iDim < nDim; iDim++) {
454 val_residual[iDim+1] = 0.0;
455 for (jDim = 0; jDim < nDim; jDim++) {
456 val_residual[iDim+1] += (sigma[iDim][jDim]*PsiVar_Grad_i[nVar-1][jDim]*invDensity - V_i[iDim+1]*alpha[jDim]*PsiVar_Grad_i[nVar-1][jDim]) * Volume;
457 }
458 }
459 val_residual[nVar-1] = alpha_gradpsi5 * Volume;
460
461// val_residual[0] += (Sigma5_vel_gradvel*invDensity - Sigma_gradvel_o_rho) * Volume;
462// for (iDim = 0; iDim < nDim; iDim++) {
463// for (jDim = 0; jDim < nDim; jDim++) {
464// val_residual[iDim+1] += (Sigma[iDim][jDim]*GradInvDensity[jDim] -
465// Sigma_5_vec[jDim]*PrimVar_Grad_i[iDim+1][jDim]*invDensity) * Volume;
466// }
467// }
468
469 /*--- Laminar viscosity sensitivity for NS ---*/
470
471 if (config->GetKind_Solver() != MAIN_SOLVER::ADJ_RANS) {
472
473// su2double Temperature_Ref = config->GetTemperature_Ref();
474// su2double Temperature_Dim = Temperature*Temperature_Ref;
475//
476// su2double S = 0.0;
477// if (config->GetSystemMeasurements() == SI) { S = 110.4; }
478// if (config->GetSystemMeasurements() == US) { S = 198.72; }
479// su2double dVisc_T = ((Laminar_Viscosity)/(2.0*Temperature_Dim*(Temperature_Dim + S)))*(Temperature_Dim + 3.0*S)*Temperature_Ref;

620void CSourceConservative_AdjTurb::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i, su2double **val_Jacobian_j, CConfig *config) {
621
622 /*--- SOURCE term --> \nabla ( \psi_\mu \B7 E^{s} )
623 E^{s} = 2 c_{b2}/\sigma \nabla \hat{nu} ---*/
624
625 unsigned short iDim;
626 bool implicit = (config->GetKind_TimeIntScheme_AdjTurb() == EULER_IMPLICIT);
627
628 su2double cb2 = 0.622;
629 su2double sigma = 2./3.;
630 su2double coeff = 2.0*cb2/sigma;
631 su2double E_ij, proj_TurbVar_Grad_i, proj_TurbVar_Grad_j;
632
633 E_ij = 0.0; proj_TurbVar_Grad_i = 0.0; proj_TurbVar_Grad_j = 0.0;
634 for (iDim = 0; iDim < nDim; iDim++) {
635 proj_TurbVar_Grad_i += coeff*ScalarVar_Grad_i[0][iDim]*Normal[iDim];
636 proj_TurbVar_Grad_j += coeff*ScalarVar_Grad_j[0][iDim]*Normal[iDim];
637 E_ij += 0.5*(TurbPsi_i[0]*proj_TurbVar_Grad_i + TurbPsi_j[0]*proj_TurbVar_Grad_j);
638 }
639
640 val_residual[0] = E_ij;
641
642 if (implicit) {
643 val_Jacobian_i[0][0] = 0.5*proj_TurbVar_Grad_i;
644 val_Jacobian_j[0][0] = 0.5*proj_TurbVar_Grad_j;
645 }
646}

668void CSourcePieceWise_AdjTurb::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i, su2double **val_Jacobian_j, CConfig *config) {
669 unsigned short iDim, jDim;
670
671 bool implicit = (config->GetKind_TimeIntScheme_AdjTurb() == EULER_IMPLICIT);
672 su2double Prandtl_Turb = config->GetPrandtl_Turb();
673
674 val_residual[0] = 0.0;
675 if (implicit)
676 val_Jacobian_i[0][0] = 0.0;
677
678 if (dist_i > 0.0) {
679
680 /*--- Computation of Vorticity and Divergence of velocity ---*/
681 su2double div_vel = 0;
682 for (iDim = 0; iDim < nDim; iDim++) {
683 Velocity[iDim] = U_i[iDim+1]/U_i[0];
684 div_vel += PrimVar_Grad_i[iDim+1][iDim];
685 }
686
687 su2double Vorticity = (PrimVar_Grad_i[2][0]-PrimVar_Grad_i[1][1])*(PrimVar_Grad_i[2][0]-PrimVar_Grad_i[1][1]);
688 if (nDim == 3)
689 Vorticity += ( (PrimVar_Grad_i[3][1]-PrimVar_Grad_i[2][2])*(PrimVar_Grad_i[3][1]-PrimVar_Grad_i[2][2]) +
690 (PrimVar_Grad_i[1][2]-PrimVar_Grad_i[3][0])*(PrimVar_Grad_i[1][2]-PrimVar_Grad_i[3][0]) );
691 Vorticity = sqrt(Vorticity);
692
693 /*--- FIRST PART: -Bs*TurbPsi_i ---*/
694 /*--- CLOUSURE CONSTANTS ---*/
695 su2double cv1 = 7.1;
696 su2double cv1_3 = cv1*cv1*cv1;
697 su2double k = 0.41;
698 su2double k2 = k*k;
699 su2double cb1 = 0.1355;
700 su2double cw2 = 0.3;
701 su2double cw3_6 = pow(2.0,6.0);
702 su2double sigma = 2./3.;
703 su2double cb2 = 0.622;
704 su2double cw1 = cb1/k2+(1+cb2)/sigma;
705
706 su2double nu, Ji, fv1, fv2, Shat, dist_0_2, Ji_2, Ji_3, one_o_oneplusJifv1;
707 su2double r, g, g_6, glim, fw;
708 su2double dTs_nuhat, dTs_Shat, dShat_nuhat, dTs_fw, dfw_g, dg_r, dr_nuhat, dr_Shat;
709 su2double dShat_fv2, dfv2_fv1, dfv1_Ji, dJi_nuhat, dfv2_Ji;
710 su2double Bs;
711
712 dist_0_2 = dist_i*dist_i;
713 nu = Laminar_Viscosity_i/U_i[0];
714 Ji = ScalarVar_i[0]/nu;
715 Ji_2 = Ji*Ji;
716 Ji_3 = Ji_2*Ji;
717 fv1 = Ji_3/(Ji_3+cv1_3);
718 one_o_oneplusJifv1 = 1.0/(1.0+Ji*fv1);
719 fv2 = 1.0 - Ji*one_o_oneplusJifv1;
720 Shat = max(Vorticity + ScalarVar_i[0]*fv2/(k2*dist_0_2), TURB_EPS);
721
722 // r = ScalarVar_i[0]/(Shat*k2*dist_0_2);
723 r = min(ScalarVar_i[0]/(Shat*k2*dist_0_2),10.);
724 g = r + cw2*(pow(r,6.)-r);
725 g_6 = pow(g,6.);
726 glim = pow((1+cw3_6)/(g_6+cw3_6),1./6.);
727 fw = g*glim;
728
729 dTs_nuhat = cb1*Shat-2.0*cw1*fw*ScalarVar_i[0]/dist_0_2;
730 dTs_Shat = cb1*ScalarVar_i[0];
731 dTs_fw = -cw1*ScalarVar_i[0]*ScalarVar_i[0]/dist_0_2;
732 dfw_g = glim*cw3_6/(g_6+cw3_6);
733 dg_r = 1.0 + cw2*(6.0*pow(r,5.0)-1.0);
734 dr_nuhat = 1.0/(Shat*k2*dist_0_2);
735 dr_Shat = -dr_nuhat*ScalarVar_i[0]/Shat;
736
737 dShat_nuhat = fv2/(k2*dist_0_2);
738 dShat_fv2 = ScalarVar_i[0]/(k2*dist_0_2);
739 dfv2_fv1 = Ji_2*one_o_oneplusJifv1*one_o_oneplusJifv1;
740 dfv1_Ji = 3.0*cv1_3*Ji_2/((Ji_3+cv1_3)*(Ji_3+cv1_3));
741 dJi_nuhat = 1.0/nu;
742 dfv2_Ji = -one_o_oneplusJifv1*one_o_oneplusJifv1;
743 dShat_nuhat += dShat_fv2*(dfv2_fv1*dfv1_Ji+dfv2_Ji)*dJi_nuhat;
744
745 Bs = dTs_nuhat; // nu_hat term
746 Bs += dTs_Shat*dShat_nuhat; // S_hat term
747 Bs += dTs_fw*dfw_g*dg_r*(dr_nuhat+dr_Shat*dShat_nuhat); // fw terms
748
749 val_residual[0] = -Bs*TurbPsi_i[0]*Volume;
750
751 if (implicit)
752 val_Jacobian_i[0][0] = -Bs*Volume;
753
754 /*---SECOND PART: \partial_nu_hat mu^k F^{vk} cdot \grad Psi ---*/
755 su2double dEddyVisc_nuhat;
756 if (!config->GetFrozen_Visc_Cont())
757 dEddyVisc_nuhat = U_i[0]*fv1*(1.0 + 3.0*cv1_3/(Ji_3+cv1_3));
758 else
759 dEddyVisc_nuhat = 0;
760
761 for (iDim = 0; iDim < nDim; iDim++) {
762 for (jDim = 0; jDim < nDim; jDim++)
763 tau[iDim][jDim] = PrimVar_Grad_i[iDim+1][jDim] + PrimVar_Grad_i[jDim+1][iDim];
764 tau[iDim][iDim] -= TWO3*div_vel;
765 }
766
767 su2double Gas_Constant = config->GetGas_ConstantND();
768 su2double Cp = (Gamma/Gamma_Minus_One)*Gas_Constant;
769 su2double tau_gradphi = 0.0, vel_tau_gradpsi5 = 0.0, gradT_gradpsi5 = 0.0;
770
771 for (iDim = 0; iDim < nDim; iDim++) {
772 gradT_gradpsi5 += PrimVar_Grad_i[0][iDim]*PsiVar_Grad_i[nVar-1][iDim];
773 for (jDim = 0; jDim < nDim; jDim++) {
774 tau_gradphi += tau[iDim][jDim]*PsiVar_Grad_i[iDim+1][jDim];
775 vel_tau_gradpsi5 += Velocity[iDim]*tau[iDim][jDim]*PsiVar_Grad_i[nVar-1][jDim];
776 }
777 }
778 val_residual[0] += (tau_gradphi + vel_tau_gradpsi5 + Cp/Prandtl_Turb*gradT_gradpsi5)*dEddyVisc_nuhat*Volume;
779
780 }
781}
