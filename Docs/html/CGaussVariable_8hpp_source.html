32/*!
33 * \class CGaussVariable
34 * \ingroup FemAlgos
35 * \brief Main class for defining the gaussian points.
36 * \version 8.0.1 "Harrier"
37 */
38class CGaussVariable {
39 protected:
40 su2activematrix GradNi_Xj; /*!< \brief Gradient of the shape functions N[i] wrt the reference configuration. */
41 su2activematrix GradNi_xj; /*!< \brief Gradient of the shape functions N[i] wrt the current configuration. */
42 su2activevector Ni; /*!< \brief Shape functions N[i] at the gaussian point. */
43 su2double J_X = 0.0; /*!< \brief Element Jacobian evaluated at this Gauss Point wrt the reference configuration. */
44 su2double J_x = 0.0; /*!< \brief Element Jacobian evaluated at this Gauss Point wrt the current configuration. */
45 unsigned short iGaussPoint = 0; /*!< \brief Identifier of the Gauss point considered. */
46
47 public:
48 /*!
49 * \brief Deleted default constructor as this class does not allow resizing once created.
50 */
51 CGaussVariable() = delete;
52
53 /*!
54 * \brief Class constructor
55 * \param[in] val_iGauss - ID of the Gaussian Point
56 * \param[in] val_nDim - Number of dimensions of the problem.
57 * \param[in] config - Definition of the particular problem.
58 */
59 CGaussVariable(unsigned short val_iGauss, unsigned short val_nDim, unsigned short val_nNodes)
60 : J_X(0.0), J_x(0.0), iGaussPoint(val_iGauss) {
61 /* --- For the structural mechanics solver the dimensions (nNodes x nDim) are sufficient.
62 * For the Sobolev smoothing solver dimensions (nNodes x (nDim+1)) are necessary
63 * when working on a curved design surface embedded in 3D. ---*/
64 GradNi_Xj.resize(val_nNodes, val_nDim + 1) = su2double(0.0);
65 GradNi_xj = GradNi_Xj;
66
67 Ni.resize(val_nNodes) = su2double(0.0);
68 }
69
70 /*!
71 * \brief Destructor of the class.
72 */
73 ~CGaussVariable(void) = default;
74
75 inline void SetGradNi_Xj(su2double val_GradNi_Xj, unsigned short val_iDim, unsigned short val_Ni) {
76 GradNi_Xj(val_Ni, val_iDim) = val_GradNi_Xj;
77 }
78
79 inline void SetGradNi_xj(su2double val_GradNi_xj, unsigned short val_iDim, unsigned short val_Ni) {
80 GradNi_xj(val_Ni, val_iDim) = val_GradNi_xj;
81 }
82
83 inline void SetNi(su2double val_ShapeNi, unsigned short val_Ni) { Ni(val_Ni) = val_ShapeNi; }
84
85 inline void SetJ_X(su2double valJ_X) { J_X = valJ_X; }
86
87 inline void SetJ_x(su2double valJ_x) { J_x = valJ_x; }
88
89 inline su2double GetGradNi_Xj(unsigned short val_Ni, unsigned short val_iDim) const {
90 return GradNi_Xj(val_Ni, val_iDim);
91 }
92
93 inline su2double GetGradNi_xj(unsigned short val_Ni, unsigned short val_iDim) const {
94 return GradNi_xj(val_Ni, val_iDim);
95 }
96
97 inline su2double GetNi(unsigned short val_Ni) const { return Ni(val_Ni); }
98
99 inline su2double GetJ_X(void) const { return J_X; }
100
101 inline su2double GetJ_x(void) const { return J_x; }
102
103 inline unsigned short Get_iGauss(void) const { return iGaussPoint; }
104};
