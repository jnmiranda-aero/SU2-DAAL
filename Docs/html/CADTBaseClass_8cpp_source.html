34void CADTBaseClass::BuildADT(unsigned short nDim, unsigned long nPoints, const su2double* coor) {
35 /*--- Determine the number of leaves. It can be proved that nLeaves equals
36 nPoints-1 for an optimally balanced tree. Take the exceptional case of
37 nPoints == 1 into account and return if the tree is empty. ---*/
38 nDimADT = nDim;
39 isEmpty = false;
40 nLeaves = nPoints - 1;
41 if (nPoints <= 1) ++nLeaves;
42 if (nLeaves == 0) {
43 isEmpty = true;
44 return;
45 }
46
47 /*--- Allocate the memory for the leaves of the ADT and the minimum and
48 maximum coordinates of the leaves. Note that these coordinates are
49 stored in one vector, rather than that memory is allocated for the
50 individual leaves. ---*/
51 leaves.resize(nLeaves);
52 coorMinLeaves.resize(nDim * nLeaves);
53 coorMaxLeaves.resize(nDim * nLeaves);
54
55 /*--- Define the vectors, which control the subdivision of the leaves. ---*/
56 unsigned long nn = (nPoints + 1) / 2;
57 vector<unsigned long> pointIDs(nPoints), pointIDsNew(nPoints);
58 vector<unsigned long> nPointIDs(nn + 1), nPointIDsNew(nn + 1);
59 vector<unsigned long> curLeaf(nn), curLeafNew(nn);
60
61 /*--------------------------------------------------------------------------*/
62 /*--- Building of the actual ADT ---*/
63 /*--------------------------------------------------------------------------*/
64
65 /*--- Initialize the arrays pointIDs, nPointIDs and curLeaf such that all
66 points belong to the root leaf. Also set the counters nLeavesToDivide
67 and nLeavesTot. ---*/
68 nPointIDs[0] = 0;
69 nPointIDs[1] = nPoints;
70 curLeaf[0] = 0;
71
72 for (unsigned long i = 0; i < nPoints; ++i) pointIDs[i] = i;
73
74 unsigned long nLeavesToDivide = 1, nLeavesTot = 1;
75
76 /*--- Loop to subdivide the leaves. The division is such that the ADT is
77 optimally balanced. ---*/
78 for (;;) {
79 /* Criterion to exit the loop. */
80 if (nLeavesToDivide == 0) break;
81
82 /* Initializations for the next round of subdivisions. */
83 unsigned long nLeavesToDivideNew = 0;
84 nPointIDsNew[0] = 0;
85
86 /*--- Loop over the current number of leaves to be divided. ---*/
87 for (unsigned long i = 0; i < nLeavesToDivide; ++i) {
88 /* Store the number of points present in the leaf in nn and the
89 current leaf number in mm. */
90 nn = nPointIDs[i + 1] - nPointIDs[i];
91 unsigned long mm = curLeaf[i];
92
93 /*--- Set the pointers for the coordinates of the leaf to the correct
94 locations in the vectors coorMinLeaves and coorMaxLeaves and
95 determine the bounding box coordinates of this leaf. ---*/
96 leaves[mm].xMin = coorMinLeaves.data() + nDim * mm;
97 leaves[mm].xMax = coorMaxLeaves.data() + nDim * mm;
98
99 unsigned long ll = nDim * pointIDs[nPointIDs[i]];
100 for (unsigned short l = 0; l < nDim; ++l) leaves[mm].xMin[l] = leaves[mm].xMax[l] = coor[ll + l];
101
102 for (unsigned long j = (nPointIDs[i] + 1); j < nPointIDs[i + 1]; ++j) {
103 ll = nDim * pointIDs[j];
104 for (unsigned short l = 0; l < nDim; ++l) {
105 leaves[mm].xMin[l] = min(leaves[mm].xMin[l], coor[ll + l]);
106 leaves[mm].xMax[l] = max(leaves[mm].xMax[l], coor[ll + l]);
107 }
108 }
109
110 /*--- Determine the split direction for this leaf. The splitting is done
111 in such a way that isotropy is reached as quickly as possible.
112 Hence the split direction is the direction of the largest dimension
113 of the leaf. ---*/
114 unsigned short splitDir = 0;
115 su2double distMax = -1.0;
116 for (unsigned short l = 0; l < nDim; ++l) {
117 const su2double dist = leaves[mm].xMax[l] - leaves[mm].xMin[l];
118 if (dist > distMax) {
119 distMax = dist;
120 splitDir = l;
121 }
122 }
123
124 /* Sort the points of the current leaf in increasing order. The sorting
125 is based on the coordinate in the split direction, for which the
126 functor CADTComparePointClass is used. */
127 sort(pointIDs.data() + nPointIDs[i], pointIDs.data() + nPointIDs[i + 1],
128 CADTComparePointClass(coor, splitDir, nDim));
129
130 /* Determine the index of the node, which is approximately central
131 in this leave. */
132 leaves[mm].centralNodeID = pointIDs[nPointIDs[i] + nn / 2];
133
134 /*--- Determine the situation of the leaf. It is either a terminal leaf
135 or a leaf that must be subdivided. ---*/
136 if (nn <= 2) {
137 /* Terminal leaf. Store the ID's of the points as children and
138 indicate that the children are terminal. */
139 leaves[mm].children[0] = pointIDs[nPointIDs[i]];
140 leaves[mm].children[1] = pointIDs[nPointIDs[i + 1] - 1];
141
142 leaves[mm].childrenAreTerminal[0] = true;
143 leaves[mm].childrenAreTerminal[1] = true;
144 } else {
145 /* The leaf must be divided. Determine the number of points in the
146 left leaf. This number is at least 2. The actual number stored in kk
147 is this number plus an offset. Also initialize the counter nfl, which
148 is used to store the bounding boxes in the arrays for the new round. */
149 unsigned long kk = (nn + 1) / 2 + nPointIDs[i];
150 unsigned long nfl = nPointIDsNew[nLeavesToDivideNew];
151
152 /* Copy the ID's of the left points into pointIDsNew. Also update the
153 corresponding entry in nPointIDsNew. */
154 for (unsigned long k = nPointIDs[i]; k < kk; ++k) pointIDsNew[nfl++] = pointIDs[k];
155
156 nPointIDsNew[nLeavesToDivideNew + 1] = nfl;
157
158 /* Store the leaf info in the tree and in the leafs for next round and
159 update nLeavesToDivideNew and nLeavesTot. */
160 leaves[mm].children[0] = nLeavesTot;
161 leaves[mm].childrenAreTerminal[0] = false;
162
163 curLeafNew[nLeavesToDivideNew] = nLeavesTot;
164 ++nLeavesToDivideNew;
165 ++nLeavesTot;
166
167 /*--- The right leaf will only be created if it has more than one point
168 in it, i.e. if the original leaf has more than three points.
169 If the new leaf only has one point in it, it is not created.
170 Instead, the point is stored in the current leaf. ---*/
171 if (nn == 3) {
172 /* Only three points present in the current leaf. The right leaf is
173 not created and the last point is stored as the second child of
174 the current leaf. */
175 leaves[mm].children[1] = pointIDs[nPointIDs[i + 1] - 1];
176 leaves[mm].childrenAreTerminal[1] = true;
177 } else {
178 /* More than 3 points are present and thus the right leaf is created.
179 Copy the ID's from pointIDs into pointIDsNew and update the
180 counters for the new round. */
181 unsigned long nfr = nPointIDsNew[nLeavesToDivideNew];
182
183 for (unsigned long k = kk; k < nPointIDs[i + 1]; ++k) pointIDsNew[nfr++] = pointIDs[k];
184
185 nPointIDsNew[nLeavesToDivideNew + 1] = nfr;
186
187 /* Store the leaf info in the tree and in the leaves for next round and
188 update nLeavesToDivideNew and nLeavesTot. */
189 leaves[mm].children[1] = nLeavesTot;
190 leaves[mm].childrenAreTerminal[1] = false;
191
192 curLeafNew[nLeavesToDivideNew] = nLeavesTot;
193 ++nLeavesToDivideNew;
194 ++nLeavesTot;
195 }
196 }
197 }
198
199 /* Set the data for the next round. */
200 nLeavesToDivide = nLeavesToDivideNew;
201
202 for (unsigned long i = 0; i <= nLeavesToDivide; ++i) nPointIDs[i] = nPointIDsNew[i];
203 for (unsigned long i = 0; i < nLeavesToDivide; ++i) curLeaf[i] = curLeafNew[i];
204 for (unsigned long i = 0; i < nPointIDs[nLeavesToDivide]; ++i) pointIDs[i] = pointIDsNew[i];
205 }
206}
