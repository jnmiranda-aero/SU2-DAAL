94 su2vector<unsigned long> Donor_nLinkedNodes;
95 su2vector<unsigned long> Donor_StartLinkedNodes;

101 su2activematrix DonorPoint_Coord;
102
103 targetVertices.resize(config[targetZone]->GetnMarker_All());
104
105 /* 1 - Variable pre-processing */
106
107 const unsigned short nDim = donor_geometry->GetnDim();
108
109 /*--- Setting up auxiliary vectors ---*/
110
111 Donor_Vect = nullptr;
112 Coeff_Vect = nullptr;
113 storeProc = nullptr;
114
115 tmp_Donor_Vect = nullptr;
116 tmp_Coeff_Vect = nullptr;
117 tmp_storeProc = nullptr;
118
119 Normal = new su2double[nDim];
120 Direction = new su2double[nDim];
121
122 /* 2 - Find boundary tag between touching grids */
123
124 /*--- Number of markers on the FSI interface ---*/
125 nMarkerInt = (int)(config[donorZone]->GetMarker_n_ZoneInterface()) / 2;
126
127 /*--- For the number of markers on the interface... ---*/
128 for (iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
129 /*--- On the donor side: find the tag of the boundary sharing the interface ---*/
130 markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
131
132 /*--- On the target side: find the tag of the boundary sharing the interface ---*/
133 markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
134
135 /*--- Checks if the zone contains the interface, if not continue to the next step ---*/
136 if (!CheckInterfaceBoundary(markDonor, markTarget)) continue;
137
138 nVertexTarget = 0;
139 if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex(markTarget);
140
141 /*
142 3 -Reconstruct the boundaries from parallel partitioning
143 */
144
145 /*--- Target boundary ---*/
146 ReconstructBoundary(targetZone, markTarget);
147
148 nGlobalVertex_Target = nGlobalVertex;
149
150 TargetPoint_Coord = Buffer_Receive_Coord;
151 Target_GlobalPoint = Buffer_Receive_GlobalPoint;
152 Target_nLinkedNodes = Buffer_Receive_nLinkedNodes;
153 Target_StartLinkedNodes = Buffer_Receive_StartLinkedNodes;
154 Target_LinkedNodes = Buffer_Receive_LinkedNodes;
155
156 /*--- Donor boundary ---*/
157 ReconstructBoundary(donorZone, markDonor);
158
159 nGlobalVertex_Donor = nGlobalVertex;
160
161 DonorPoint_Coord = Buffer_Receive_Coord;
162 Donor_GlobalPoint = Buffer_Receive_GlobalPoint;
163 Donor_nLinkedNodes = Buffer_Receive_nLinkedNodes;
164 Donor_StartLinkedNodes = Buffer_Receive_StartLinkedNodes;
165 Donor_LinkedNodes = Buffer_Receive_LinkedNodes;
166 Donor_Proc = Buffer_Receive_Proc;
167
168 /*--- Starts building the supermesh layer (2D or 3D) ---*/
169 /* - For each target node, it first finds the closest donor point
170 * - Then it creates the supermesh in the close proximity of the target point:
171 * - Starting from the closest donor node, it expands the supermesh by including
172 * donor elements neighboring the initial one, until the overall target area is fully covered.
173 */
174 if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
175
176 if (nDim == 2) {
177 target_iMidEdge_point = new su2double[nDim];
178 target_jMidEdge_point = new su2double[nDim];
179
180 donor_iMidEdge_point = new su2double[nDim];
181 donor_jMidEdge_point = new su2double[nDim];
182
183 /*--- Starts with supermesh reconstruction ---*/
184
185 target_segment = new unsigned long[2];
186
187 for (iVertex = 0; iVertex < nVertexTarget; iVertex++) {
188 nDonorPoints = 0;
189
190 /*--- Stores coordinates of the target node ---*/
191
192 target_iPoint = target_geometry->vertex[markTarget][iVertex]->GetNode();
193
194 if (target_geometry->nodes->GetDomain(target_iPoint)) {
195 Coord_i = target_geometry->nodes->GetCoord(target_iPoint);
196
197 /*--- Brute force to find the closest donor_node ---*/
198
199 mindist = 1E6;
200 donor_StartIndex = 0;
201
202 for (donor_iPoint = 0; donor_iPoint < nGlobalVertex_Donor; donor_iPoint++) {
203 Coord_j = DonorPoint_Coord[donor_iPoint];
204
205 dist = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
206
207 if (dist < mindist) {
208 mindist = dist;
209 donor_StartIndex = donor_iPoint;
210 }
211
212 if (dist == 0.0) {
213 donor_StartIndex = donor_iPoint;
214 break;
215 }
216 }
217
218 donor_iPoint = donor_StartIndex;
219 donor_OldiPoint = donor_iPoint;
220
221 /*--- Contruct information regarding the target cell ---*/
222
223 auto dPoint = target_geometry->nodes->GetGlobalIndex(target_iPoint);
224 for (jVertexTarget = 0; jVertexTarget < nGlobalVertex_Target; jVertexTarget++)
225 if (dPoint == Target_GlobalPoint[jVertexTarget]) break;
226
227 if (Target_nLinkedNodes[jVertexTarget] == 1) {
228 target_segment[0] = Target_LinkedNodes[Target_StartLinkedNodes[jVertexTarget]];
229 target_segment[1] = jVertexTarget;
230 } else {
231 target_segment[0] = Target_LinkedNodes[Target_StartLinkedNodes[jVertexTarget]];
232 target_segment[1] = Target_LinkedNodes[Target_StartLinkedNodes[jVertexTarget] + 1];
233 }
234
235 dTMP = 0;
236 for (iDim = 0; iDim < nDim; iDim++) {
237 target_iMidEdge_point[iDim] =
238 (TargetPoint_Coord(target_segment[0], iDim) + target_geometry->nodes->GetCoord(target_iPoint, iDim)) /
239 2.;
240 target_jMidEdge_point[iDim] =
241 (TargetPoint_Coord(target_segment[1], iDim) + target_geometry->nodes->GetCoord(target_iPoint, iDim)) /
242 2.;
243
244 Direction[iDim] = target_jMidEdge_point[iDim] - target_iMidEdge_point[iDim];
245 dTMP += Direction[iDim] * Direction[iDim];
246 }
247
248 dTMP = sqrt(dTMP);
249 for (iDim = 0; iDim < nDim; iDim++) Direction[iDim] /= dTMP;
250
251 length = GeometryToolbox::Distance(nDim, target_iMidEdge_point, target_jMidEdge_point);
252
253 check = false;
254
255 /*--- Proceeds along the forward direction (depending on which connected boundary node is found first) ---*/
256
257 while (!check) {
258 /*--- Proceeds until the value of the intersection area is null ---*/
259
260 if (Donor_nLinkedNodes[donor_iPoint] == 1) {
261 donor_forward_point = Donor_LinkedNodes[Donor_StartLinkedNodes[donor_iPoint]];
262 donor_backward_point = donor_iPoint;
263 } else {
264 uptr = &Donor_LinkedNodes[Donor_StartLinkedNodes[donor_iPoint]];
265
266 if (donor_OldiPoint != uptr[0]) {
267 donor_forward_point = uptr[0];
268 donor_backward_point = uptr[1];
269 } else {
270 donor_forward_point = uptr[1];
271 donor_backward_point = uptr[0];
272 }
273 }
274
275 if (donor_iPoint >= nGlobalVertex_Donor) {
276 check = true;
277 continue;
278 }
279
280 for (iDim = 0; iDim < nDim; iDim++) {
281 donor_iMidEdge_point[iDim] =
282 (DonorPoint_Coord(donor_forward_point, iDim) + DonorPoint_Coord(donor_iPoint, iDim)) / 2.;
283 donor_jMidEdge_point[iDim] =
284 (DonorPoint_Coord(donor_backward_point, iDim) + DonorPoint_Coord(donor_iPoint, iDim)) / 2.;
285 }
286
287 LineIntersectionLength =
288 ComputeLineIntersectionLength(nDim, target_iMidEdge_point, target_jMidEdge_point, donor_iMidEdge_point,
289 donor_jMidEdge_point, Direction);
290
291 if (LineIntersectionLength == 0.0) {
292 check = true;
293 continue;
294 }
295
296 /*--- In case the element intersects the target cell, update the auxiliary communication data structure
297 * ---*/
298
299 tmp_Coeff_Vect = new su2double[nDonorPoints + 1];
300 tmp_Donor_Vect = new unsigned long[nDonorPoints + 1];
301 tmp_storeProc = new unsigned long[nDonorPoints + 1];
302
303 for (iDonor = 0; iDonor < nDonorPoints; iDonor++) {
304 tmp_Donor_Vect[iDonor] = Donor_Vect[iDonor];
305 tmp_Coeff_Vect[iDonor] = Coeff_Vect[iDonor];
306 tmp_storeProc[iDonor] = storeProc[iDonor];
307 }
308
309 tmp_Donor_Vect[nDonorPoints] = donor_iPoint;
310 tmp_Coeff_Vect[nDonorPoints] = LineIntersectionLength / length;
311 tmp_storeProc[nDonorPoints] = Donor_Proc[donor_iPoint];
312
313 delete[] Donor_Vect;
314 delete[] Coeff_Vect;
315 delete[] storeProc;
316
317 Donor_Vect = tmp_Donor_Vect;
318 Coeff_Vect = tmp_Coeff_Vect;
319 storeProc = tmp_storeProc;
320
321 donor_OldiPoint = donor_iPoint;
322 donor_iPoint = donor_forward_point;
323
324 nDonorPoints++;
325 }
326
327 if (Donor_nLinkedNodes[donor_StartIndex] == 2) {
328 check = false;
329
330 uptr = &Donor_LinkedNodes[Donor_StartLinkedNodes[donor_StartIndex]];
331
332 donor_iPoint = uptr[1];
333 donor_OldiPoint = donor_StartIndex;
334 } else
335 check = true;
336
337 /*--- Proceeds along the backward direction (depending on which connected boundary node is found first) ---*/
338
339 while (!check) {
340 /*--- Proceeds until the value of the intersection length is null ---*/
341 if (Donor_nLinkedNodes[donor_iPoint] == 1) {
342 donor_forward_point = donor_OldiPoint;
343 donor_backward_point = donor_iPoint;
344 } else {
345 uptr = &Donor_LinkedNodes[Donor_StartLinkedNodes[donor_iPoint]];
346
347 if (donor_OldiPoint != uptr[0]) {
348 donor_forward_point = uptr[0];
349 donor_backward_point = uptr[1];
350 } else {
351 donor_forward_point = uptr[1];
352 donor_backward_point = uptr[0];
353 }
354 }
355
356 if (donor_iPoint >= nGlobalVertex_Donor) {
357 check = true;
358 continue;
359 }
360
361 for (iDim = 0; iDim < nDim; iDim++) {
362 donor_iMidEdge_point[iDim] =
363 (DonorPoint_Coord(donor_forward_point, iDim) + DonorPoint_Coord(donor_iPoint, iDim)) / 2.;
364 donor_jMidEdge_point[iDim] =
365 (DonorPoint_Coord(donor_backward_point, iDim) + DonorPoint_Coord(donor_iPoint, iDim)) / 2.;
366 }
367
368 LineIntersectionLength =
369 ComputeLineIntersectionLength(nDim, target_iMidEdge_point, target_jMidEdge_point, donor_iMidEdge_point,
370 donor_jMidEdge_point, Direction);
371
372 if (LineIntersectionLength == 0.0) {
373 check = true;
374 continue;
375 }
376
377 /*--- In case the element intersects the target cell, update the auxiliary communication data structure
378 * ---*/
379
380 tmp_Coeff_Vect = new su2double[nDonorPoints + 1];
381 tmp_Donor_Vect = new unsigned long[nDonorPoints + 1];
382 tmp_storeProc = new unsigned long[nDonorPoints + 1];
383
384 for (iDonor = 0; iDonor < nDonorPoints; iDonor++) {
385 tmp_Donor_Vect[iDonor] = Donor_Vect[iDonor];
386 tmp_Coeff_Vect[iDonor] = Coeff_Vect[iDonor];
387 tmp_storeProc[iDonor] = storeProc[iDonor];
388 }
389
390 tmp_Coeff_Vect[nDonorPoints] = LineIntersectionLength / length;
391 tmp_Donor_Vect[nDonorPoints] = donor_iPoint;
392 tmp_storeProc[nDonorPoints] = Donor_Proc[donor_iPoint];
393
394 delete[] Donor_Vect;
395 delete[] Coeff_Vect;
396 delete[] storeProc;
397
398 Donor_Vect = tmp_Donor_Vect;
399 Coeff_Vect = tmp_Coeff_Vect;
400 storeProc = tmp_storeProc;
401
402 donor_OldiPoint = donor_iPoint;
403 donor_iPoint = donor_forward_point;
404
405 nDonorPoints++;
406 }
407
408 /*--- Set the communication data structure and copy data from the auxiliary vectors ---*/
409
410 targetVertices[markTarget][iVertex].resize(nDonorPoints);
411
412 for (iDonor = 0; iDonor < nDonorPoints; iDonor++) {
413 targetVertices[markTarget][iVertex].coefficient[iDonor] = Coeff_Vect[iDonor];
414 targetVertices[markTarget][iVertex].globalPoint[iDonor] = Donor_GlobalPoint[Donor_Vect[iDonor]];
415 targetVertices[markTarget][iVertex].processor[iDonor] = storeProc[iDonor];
416 }
417 }
418 }
419
420 delete[] target_segment;
421
422 delete[] target_iMidEdge_point;
423 delete[] target_jMidEdge_point;
424
425 delete[] donor_iMidEdge_point;
426 delete[] donor_jMidEdge_point;
427 } else {
428 /* --- 3D geometry, creates a superficial super-mesh --- */
429
430 for (iVertex = 0; iVertex < nVertexTarget; iVertex++) {
431 nDonorPoints = 0;
432
433 /*--- Stores coordinates of the target node ---*/
434
435 target_iPoint = target_geometry->vertex[markTarget][iVertex]->GetNode();
436
437 if (!target_geometry->nodes->GetDomain(target_iPoint)) continue;
438
439 Coord_i = target_geometry->nodes->GetCoord(target_iPoint);
440
441 target_geometry->vertex[markTarget][iVertex]->GetNormal(Normal);
442
443 /*--- The value of Area computed here includes also portion of boundary belonging to different marker ---*/
444 Area = GeometryToolbox::Norm(nDim, Normal);
445
446 for (iDim = 0; iDim < nDim; iDim++) Normal[iDim] /= Area;
447
448 for (iDim = 0; iDim < nDim; iDim++) Coord_i[iDim] = target_geometry->nodes->GetCoord(target_iPoint, iDim);
449
450 auto dPoint = target_geometry->nodes->GetGlobalIndex(target_iPoint);
451 for (target_iPoint = 0; target_iPoint < nGlobalVertex_Target; target_iPoint++) {
452 if (dPoint == Target_GlobalPoint[target_iPoint]) break;
453 }
454
455 /*--- Build local surface dual mesh for target element ---*/
456
457 nEdges_target = Target_nLinkedNodes[target_iPoint];
458
459 nNode_target = 2 * (nEdges_target + 1);
460
461 target_element = new su2double*[nNode_target];
462 for (ii = 0; ii < nNode_target; ii++) target_element[ii] = new su2double[nDim];
463
464 nNode_target = Build_3D_surface_element(Target_LinkedNodes, Target_StartLinkedNodes, Target_nLinkedNodes,
465 TargetPoint_Coord, target_iPoint, target_element);
466
467 /*--- Brute force to find the closest donor_node ---*/
468
469 mindist = 1E6;
470 donor_StartIndex = 0;
471
472 for (donor_iPoint = 0; donor_iPoint < nGlobalVertex_Donor; donor_iPoint++) {
473 Coord_j = DonorPoint_Coord[donor_iPoint];
474
475 dist = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
476
477 if (dist < mindist) {
478 mindist = dist;
479 donor_StartIndex = donor_iPoint;
480 }
481
482 if (dist == 0.0) {
483 donor_StartIndex = donor_iPoint;
484 break;
485 }
486 }
487
488 donor_iPoint = donor_StartIndex;
489
490 nEdges_donor = Donor_nLinkedNodes[donor_iPoint];
491
492 donor_element = new su2double*[2 * nEdges_donor + 2];
493 for (ii = 0; ii < 2 * nEdges_donor + 2; ii++) donor_element[ii] = new su2double[nDim];
494
495 nNode_donor = Build_3D_surface_element(Donor_LinkedNodes, Donor_StartLinkedNodes, Donor_nLinkedNodes,
496 DonorPoint_Coord, donor_iPoint, donor_element);
497
498 Area = 0;
499 for (ii = 1; ii < nNode_target - 1; ii++) {
500 for (jj = 1; jj < nNode_donor - 1; jj++) {
501 Area += Compute_Triangle_Intersection(target_element[0], target_element[ii], target_element[ii + 1],
502 donor_element[0], donor_element[jj], donor_element[jj + 1], Normal);
503 }
504 }
505
506 for (ii = 0; ii < 2 * nEdges_donor + 2; ii++) delete[] donor_element[ii];
507 delete[] donor_element;
508
509 nDonorPoints = 1;
510
511 /*--- In case the element intersect the target cell update the auxiliary communication data structure ---*/
512
513 Coeff_Vect = new su2double[nDonorPoints];
514 Donor_Vect = new unsigned long[nDonorPoints];
515 storeProc = new unsigned long[nDonorPoints];
516
517 Coeff_Vect[0] = Area;
518 Donor_Vect[0] = donor_iPoint;
519 storeProc[0] = Donor_Proc[donor_iPoint];
520
521 alreadyVisitedDonor = new unsigned long[1];
522
523 alreadyVisitedDonor[0] = donor_iPoint;
524 nAlreadyVisited = 1;
525 StartVisited = 0;
526
527 Area_old = -1;
528
529 while (Area > Area_old) {
530 /*
531 * - Starting from the closest donor_point, it expands the supermesh by a countour search pattern.
532 * - The closest donor element becomes the core, at each iteration a new layer of elements around the core is
533 * taken into account
534 */
535
536 Area_old = Area;
537
538 ToVisit = nullptr;
539 nToVisit = 0;
540
541 for (iNodeVisited = StartVisited; iNodeVisited < nAlreadyVisited; iNodeVisited++) {
542 vPoint = alreadyVisitedDonor[iNodeVisited];
543
544 nEdgeVisited = Donor_nLinkedNodes[vPoint];
545
546 for (iEdgeVisited = 0; iEdgeVisited < nEdgeVisited; iEdgeVisited++) {
547 donor_iPoint = Donor_LinkedNodes[Donor_StartLinkedNodes[vPoint] + iEdgeVisited];
548
549 /*--- Check if the node to visit is already listed in the data structure to avoid double visits ---*/
550
551 check = false;
552
553 for (jj = 0; jj < nAlreadyVisited; jj++) {
554 if (donor_iPoint == alreadyVisitedDonor[jj]) {
555 check = true;
556 break;
557 }
558 }
559
560 if (check == 0 && ToVisit != nullptr) {
561 for (jj = 0; jj < nToVisit; jj++)
562 if (donor_iPoint == ToVisit[jj]) {
563 check = true;
564 break;
565 }
566 }
567
568 if (check == 0) {
569 /*--- If the node was not already visited, visit it and list it into data structure ---*/
570
571 tmpVect = new unsigned long[nToVisit + 1];
572
573 for (jj = 0; jj < nToVisit; jj++) tmpVect[jj] = ToVisit[jj];
574 tmpVect[nToVisit] = donor_iPoint;
575
576 delete[] ToVisit;
577
578 ToVisit = tmpVect;
579 tmpVect = nullptr;
580
581 nToVisit++;
582
583 /*--- Find the value of the intersection area between the current donor element and the target element
584 * --- */
585
586 nEdges_donor = Donor_nLinkedNodes[donor_iPoint];
587
588 donor_element = new su2double*[2 * nEdges_donor + 2];
589 for (ii = 0; ii < 2 * nEdges_donor + 2; ii++) donor_element[ii] = new su2double[nDim];
590
591 nNode_donor = Build_3D_surface_element(Donor_LinkedNodes, Donor_StartLinkedNodes, Donor_nLinkedNodes,
592 DonorPoint_Coord, donor_iPoint, donor_element);
593
594 tmp_Area = 0;
595 for (ii = 1; ii < nNode_target - 1; ii++)
596 for (jj = 1; jj < nNode_donor - 1; jj++)
597 tmp_Area += Compute_Triangle_Intersection(target_element[0], target_element[ii],
598 target_element[ii + 1], donor_element[0],
599 donor_element[jj], donor_element[jj + 1], Normal);
600
601 for (ii = 0; ii < 2 * nEdges_donor + 2; ii++) delete[] donor_element[ii];
602 delete[] donor_element;
603
604 /*--- In case the element intersect the target cell update the auxiliary communication data structure
605 * ---*/
606
607 tmp_Coeff_Vect = new su2double[nDonorPoints + 1];
608 tmp_Donor_Vect = new unsigned long[nDonorPoints + 1];
609 tmp_storeProc = new unsigned long[nDonorPoints + 1];
610
611 for (iDonor = 0; iDonor < nDonorPoints; iDonor++) {
612 tmp_Donor_Vect[iDonor] = Donor_Vect[iDonor];
613 tmp_Coeff_Vect[iDonor] = Coeff_Vect[iDonor];
614 tmp_storeProc[iDonor] = storeProc[iDonor];
615 }
616
617 tmp_Coeff_Vect[nDonorPoints] = tmp_Area;
618 tmp_Donor_Vect[nDonorPoints] = donor_iPoint;
619 tmp_storeProc[nDonorPoints] = Donor_Proc[donor_iPoint];
620
621 delete[] Donor_Vect;
622 delete[] Coeff_Vect;
623 delete[] storeProc;
624
625 Donor_Vect = tmp_Donor_Vect;
626 Coeff_Vect = tmp_Coeff_Vect;
627 storeProc = tmp_storeProc;
628
629 tmp_Coeff_Vect = nullptr;
630 tmp_Donor_Vect = nullptr;
631 tmp_storeProc = nullptr;
632
633 nDonorPoints++;
634
635 Area += tmp_Area;
636 }
637 }
638 }
639
640 /*--- Update auxiliary data structure ---*/
641
642 StartVisited = nAlreadyVisited;
643
644 tmpVect = new unsigned long[nAlreadyVisited + nToVisit];
645
646 for (jj = 0; jj < nAlreadyVisited; jj++) tmpVect[jj] = alreadyVisitedDonor[jj];
647
648 for (jj = 0; jj < nToVisit; jj++) tmpVect[nAlreadyVisited + jj] = ToVisit[jj];
649
650 delete[] alreadyVisitedDonor;
651
652 alreadyVisitedDonor = tmpVect;
653
654 nAlreadyVisited += nToVisit;
655
656 delete[] ToVisit;
657 }
658
659 delete[] alreadyVisitedDonor;
660
661 /*--- Set the communication data structure and copy data from the auxiliary vectors ---*/
662
663 targetVertices[markTarget][iVertex].resize(nDonorPoints);
664
665 for (iDonor = 0; iDonor < nDonorPoints; iDonor++) {
666 targetVertices[markTarget][iVertex].coefficient[iDonor] = Coeff_Vect[iDonor] / Area;
667 targetVertices[markTarget][iVertex].globalPoint[iDonor] = Donor_GlobalPoint[Donor_Vect[iDonor]];
668 targetVertices[markTarget][iVertex].processor[iDonor] = storeProc[iDonor];
669 }
670
671 for (ii = 0; ii < 2 * nEdges_target + 2; ii++) delete[] target_element[ii];
672 delete[] target_element;
673
674 delete[] Donor_Vect;
675 Donor_Vect = nullptr;
676 delete[] Coeff_Vect;
677 Coeff_Vect = nullptr;
678 delete[] storeProc;
679 storeProc = nullptr;
680 }
681 }
682 }
683
684 delete[] Normal;
685 delete[] Direction;
686
687 delete[] Donor_Vect;
688 delete[] Coeff_Vect;
689 delete[] storeProc;
690}

692int CSlidingMesh::Build_3D_surface_element(const su2vector<unsigned long>& map,
693 const su2vector<unsigned long>& startIndex,
694 const su2vector<unsigned long>& nNeighbor, su2activematrix const& coord,
695 unsigned long centralNode, su2double** element) {
696 /*--- Given a node "centralNode", this routines reconstruct the vertex centered
697 * surface element around the node and store it into "element" ---*/
698
699 constexpr unsigned short nDim = 3;
700
701 const unsigned long* OuterNodes;
702
703 /* --- Store central node as element first point --- */
704
705 for (unsigned short iDim = 0; iDim < nDim; iDim++) element[0][iDim] = coord(centralNode, iDim);
706
707 unsigned long nOuterNodes = nNeighbor[centralNode];
708
709 OuterNodes = &map[startIndex[centralNode]];
710
711 // For each neighbor n of centralNode, store <=2 neighbors of centralNode that are neighbors of n.
712 su2matrix<int> OuterNodesNeighbour(nOuterNodes, 2);
713 OuterNodesNeighbour = -1;
714 // Typically there are exactly 2 such neighbors, and all the neighbors of centralNode can be
715 // arranged into a closed chain. However at 1D boundaries of 2D markers, the neighbors might
716 // be an open chain, with the two ends having only 1 such neighbor.
717 // StartNode is a node where we can start to iterate through the chain. I.e. it's any node in
718 // case of a closed chain, or one of the two ends in case of an open chain.
719 int StartNode = 0;
720 for (unsigned long iNode = 0; iNode < nOuterNodes; iNode++) {
721 int count = 0; // number of neighboring outer nodes already found
722 const unsigned long iPoint = OuterNodes[iNode];
723 const unsigned long* ptr = &map[startIndex[iPoint]];
724
725 for (unsigned long jNode = 0; jNode < nNeighbor[iPoint]; jNode++) {
726 const unsigned long jPoint = ptr[jNode];
727 for (unsigned long kNode = 0; kNode < nOuterNodes; kNode++) {
728 if (jPoint == OuterNodes[kNode] && jPoint != centralNode) {
729 OuterNodesNeighbour(iNode, count) = static_cast<int>(kNode);
730 count++;
731 break;
732 }
733 }
734 }
735 if (count == 1) StartNode = static_cast<int>(iNode);
736 }
737
738 /* --- Build element, starts from one outer node and loops along the external edges until the element is reconstructed
739 * --- */
740
741 int CurrentNode = StartNode;
742 int NextNode = OuterNodesNeighbour(CurrentNode, 0);
743 unsigned long iElementNode = 1;
744
745 while (NextNode !=
746 -1) { // We finished iterating through the chain if it is an open chain and we reached the other end.
747
748 for (unsigned short iDim = 0; iDim < nDim; iDim++)
749 element[iElementNode][iDim] = (element[0][iDim] + coord(OuterNodes[CurrentNode], iDim)) / 2.;
750
751 iElementNode++;
752
753 for (unsigned short iDim = 0; iDim < nDim; iDim++)
754 element[iElementNode][iDim] =
755 (element[0][iDim] + coord[OuterNodes[CurrentNode]][iDim] + coord(OuterNodes[NextNode], iDim)) / 3.;
756 iElementNode++;
757
758 // "Place the next domino piece in the correct orientation."
759 if (OuterNodesNeighbour(NextNode, 0) == CurrentNode) {
760 CurrentNode = NextNode;
761 NextNode = OuterNodesNeighbour(NextNode, 1);
762 } else {
763 CurrentNode = NextNode;
764 NextNode = OuterNodesNeighbour(NextNode, 0);
765 }
766
767 // We finished iterating through the chain if it is closed and we reached the beginning again.
768 if (CurrentNode == StartNode) break;
769 }
770
771 if (CurrentNode ==
772 StartNode) { // This is a closed element, so add again element 1 to the end of the structure, useful later
773 for (unsigned short iDim = 0; iDim < nDim; iDim++) element[iElementNode][iDim] = element[1][iDim];
774 iElementNode++;
775 } else {
776 for (unsigned short iDim = 0; iDim < nDim; iDim++)
777 element[iElementNode][iDim] = (element[0][iDim] + coord(OuterNodes[CurrentNode], iDim)) / 2.;
778 iElementNode++;
779 }
780
781 return static_cast<int>(iElementNode);
782}

784su2double CSlidingMesh::ComputeLineIntersectionLength(unsigned short nDim, const su2double* A1, const su2double* A2,
785 const su2double* B1, const su2double* B2,
786 const su2double* Direction) {
787 /*--- Given 2 segments, each defined by 2 points, it projects them along a given direction
788 * and it computes the length of the segment resulting from their intersection ---*/
789 /*--- The algorithm works for both 2D and 3D problems ---*/
790
791 unsigned short iDim;
792
793 su2double dotA2, dotB1, dotB2;
794
795 dotA2 = 0;
796 for (iDim = 0; iDim < nDim; iDim++) dotA2 += (A2[iDim] - A1[iDim]) * Direction[iDim];
797
798 if (dotA2 >= 0) {
799 dotB1 = 0;
800 dotB2 = 0;
801 for (iDim = 0; iDim < nDim; iDim++) {
802 dotB1 += (B1[iDim] - A1[iDim]) * Direction[iDim];
803 dotB2 += (B2[iDim] - A1[iDim]) * Direction[iDim];
804 }
805 } else {
806 dotA2 *= -1;
807
808 dotB1 = 0;
809 dotB2 = 0;
810 for (iDim = 0; iDim < nDim; iDim++) {
811 dotB1 -= (B1[iDim] - A1[iDim]) * Direction[iDim];
812 dotB2 -= (B2[iDim] - A1[iDim]) * Direction[iDim];
813 }
814 }
815
816 if (dotB1 >= 0 && dotB1 <= dotA2) {
817 if (dotB2 < 0) return fabs(dotB1);
818 if (dotB2 > dotA2) return fabs(dotA2 - dotB1);
819
820 return fabs(dotB1 - dotB2);
821 }
822
823 if (dotB2 >= 0 && dotB2 <= dotA2) {
824 if (dotB1 < 0) return fabs(dotB2);
825 if (dotB1 > dotA2) return fabs(dotA2 - dotB2);
826 }
827
828 if ((dotB1 <= 0 && dotA2 <= dotB2) || (dotB2 <= 0 && dotA2 <= dotB1)) return fabs(dotA2);
829
830 return 0.0;
831}

833su2double CSlidingMesh::Compute_Triangle_Intersection(const su2double* A1, const su2double* A2, const su2double* A3,
834 const su2double* B1, const su2double* B2, const su2double* B3,
835 const su2double* Direction) {
836 /* --- This routine is ONLY for 3D grids --- */
837 /* --- Projects triangle points onto a plane, specified by its normal "Direction", and calls the
838 * ComputeIntersectionArea routine --- */
839
840 unsigned short iDim;
841 constexpr unsigned short nDim = 3;
842
843 su2double I[3], J[3], K[3];
844 su2double a1[3], a2[3], a3[3];
845 su2double b1[3], b2[3], b3[3];
846 su2double m1, m2;
847
848 /* --- Reference frame is determined by: x = A1A2 y = x ^ ( -Direction ) --- */
849
850 for (iDim = 0; iDim < 3; iDim++) {
851 a1[iDim] = 0;
852 a2[iDim] = 0;
853 a3[iDim] = 0;
854
855 b1[iDim] = 0;
856 b2[iDim] = 0;
857 b3[iDim] = 0;
858 }
859
860 m1 = 0;
861 for (iDim = 0; iDim < nDim; iDim++) {
862 K[iDim] = Direction[iDim];
863
864 m1 += K[iDim] * K[iDim];
865 }
866
867 for (iDim = 0; iDim < nDim; iDim++) K[iDim] /= sqrt(m1);
868
869 m2 = 0;
870 for (iDim = 0; iDim < nDim; iDim++) m2 += (A2[iDim] - A1[iDim]) * K[iDim];
871
872 m1 = 0;
873 for (iDim = 0; iDim < nDim; iDim++) {
874 I[iDim] = (A2[iDim] - A1[iDim]) - m2 * K[iDim];
875 m1 += I[iDim] * I[iDim];
876 }
877
878 for (iDim = 0; iDim < nDim; iDim++) I[iDim] /= sqrt(m1);
879
880 // Cross product to find Y
881 J[0] = K[1] * I[2] - K[2] * I[1];
882 J[1] = -(K[0] * I[2] - K[2] * I[0]);
883 J[2] = K[0] * I[1] - K[1] * I[0];
884
885 /* --- Project all points on the plane specified by Direction and change their reference frame taking A1 as origin ---
886 */
887
888 for (iDim = 0; iDim < nDim; iDim++) {
889 a2[0] += (A2[iDim] - A1[iDim]) * I[iDim];
890 a2[1] += (A2[iDim] - A1[iDim]) * J[iDim];
891 a2[2] += (A2[iDim] - A1[iDim]) * K[iDim];
892
893 a3[0] += (A3[iDim] - A1[iDim]) * I[iDim];
894 a3[1] += (A3[iDim] - A1[iDim]) * J[iDim];
895 a3[2] += (A3[iDim] - A1[iDim]) * K[iDim];
896
897 b1[0] += (B1[iDim] - A1[iDim]) * I[iDim];
898 b1[1] += (B1[iDim] - A1[iDim]) * J[iDim];
899 b1[2] += (B1[iDim] - A1[iDim]) * K[iDim];
900
901 b2[0] += (B2[iDim] - A1[iDim]) * I[iDim];
902 b2[1] += (B2[iDim] - A1[iDim]) * J[iDim];
903 b2[2] += (B2[iDim] - A1[iDim]) * K[iDim];
904
905 b3[0] += (B3[iDim] - A1[iDim]) * I[iDim];
906 b3[1] += (B3[iDim] - A1[iDim]) * J[iDim];
907 b3[2] += (B3[iDim] - A1[iDim]) * K[iDim];
908 }
909
910 /*--- Compute intersection area ---*/
911
912 return ComputeIntersectionArea(a1, a2, a3, b1, b2, b3);
913}

915su2double CSlidingMesh::ComputeIntersectionArea(const su2double* P1, const su2double* P2, const su2double* P3,
916 const su2double* Q1, const su2double* Q2, const su2double* Q3) {
917 /* --- This routines computes the area of the polygonal element generated by the superimposition of 2 planar triangle
918 * --- */
919 /* --- The 2 triangle must lie on the same plane --- */
920
921 unsigned short iDim, nPoints = 0, i, j, k;
922 unsigned short min_theta_index;
923
924 su2double points[16][2], IntersectionPoint[2], theta[6];
925 su2double TriangleP[4][2], TriangleQ[4][2];
926 su2double Area, det, dot1, dot2, dtmp, min_theta;
927
928 constexpr unsigned short nDim = 2;
929
930 for (iDim = 0; iDim < nDim; iDim++) {
931 TriangleP[0][iDim] = 0;
932 TriangleP[1][iDim] = P2[iDim] - P1[iDim];
933 TriangleP[2][iDim] = P3[iDim] - P1[iDim];
934 TriangleP[3][iDim] = 0;
935
936 TriangleQ[0][iDim] = Q1[iDim] - P1[iDim];
937 TriangleQ[1][iDim] = Q2[iDim] - P1[iDim];
938 TriangleQ[2][iDim] = Q3[iDim] - P1[iDim];
939 TriangleQ[3][iDim] = Q1[iDim] - P1[iDim];
940 }
941
942 for (j = 0; j < 3; j++) {
943 if (CheckPointInsideTriangle(TriangleP[j], TriangleQ[0], TriangleQ[1], TriangleQ[2])) {
944 // Then P1 is also inside triangle Q, so store it
945 for (iDim = 0; iDim < nDim; iDim++) points[nPoints][iDim] = TriangleP[j][iDim];
946
947 nPoints++;
948 }
949 }
950
951 for (j = 0; j < 3; j++) {
952 if (CheckPointInsideTriangle(TriangleQ[j], TriangleP[0], TriangleP[1], TriangleP[2])) {
953 // Then Q1 is also inside triangle P, so store it
954 for (iDim = 0; iDim < nDim; iDim++) points[nPoints][iDim] = TriangleQ[j][iDim];
955
956 nPoints++;
957 }
958 }
959
960 // Compute all edge intersections
961
962 for (j = 0; j < 3; j++) {
963 for (i = 0; i < 3; i++) {
964 det = (TriangleP[j][0] - TriangleP[j + 1][0]) * (TriangleQ[i][1] - TriangleQ[i + 1][1]) -
965 (TriangleP[j][1] - TriangleP[j + 1][1]) * (TriangleQ[i][0] - TriangleQ[i + 1][0]);
966
967 if (det != 0.0) {
968 ComputeLineIntersectionPoint(TriangleP[j], TriangleP[j + 1], TriangleQ[i], TriangleQ[i + 1], IntersectionPoint);
969
970 dot1 = 0;
971 dot2 = 0;
972 for (iDim = 0; iDim < nDim; iDim++) {
973 dot1 += (TriangleP[j][iDim] - IntersectionPoint[iDim]) * (TriangleP[j + 1][iDim] - IntersectionPoint[iDim]);
974 dot2 += (TriangleQ[i][iDim] - IntersectionPoint[iDim]) * (TriangleQ[i + 1][iDim] - IntersectionPoint[iDim]);
975 }
976
977 if (dot1 <= 0 && dot2 <= 0) { // It found one intersection
978
979 // Store temporarily the intersection point
980
981 for (iDim = 0; iDim < nDim; iDim++) points[nPoints][iDim] = IntersectionPoint[iDim];
982
983 nPoints++;
984 }
985 }
986 }
987 }
988
989 // Remove double points, if any
990
991 for (i = 0; i < nPoints; i++) {
992 for (j = i + 1; j < nPoints; j++) {
993 if (points[j][0] == points[i][0] && points[j][1] == points[i][1]) {
994 for (k = j; k < nPoints - 1; k++) {
995 points[k][0] = points[k + 1][0];
996 points[k][1] = points[k + 1][1];
997 }
998 nPoints--;
999 j--;
1000 }
1001 }
1002 }
1003
1004 // Re-order nodes
1005
1006 for (i = 1; i < nPoints; i++) { // Change again reference frame
1007 for (iDim = 0; iDim < nDim; iDim++) points[i][iDim] -= points[0][iDim];
1008
1009 // Compute polar azimuth for each node but the first
1010 theta[i] = atan2(points[i][1], points[i][0]);
1011 }
1012
1013 for (iDim = 0; iDim < nDim; iDim++) points[0][iDim] = 0;
1014
1015 for (i = 1; i < nPoints; i++) {
1016 min_theta = theta[i];
1017 min_theta_index = 0;
1018
1019 for (j = i + 1; j < nPoints; j++) {
1020 if (theta[j] < min_theta) {
1021 min_theta = theta[j];
1022 min_theta_index = j;
1023 }
1024 }
1025
1026 if (min_theta_index != 0) {
1027 dtmp = theta[i];
1028 theta[i] = theta[min_theta_index];
1029 theta[min_theta_index] = dtmp;
1030
1031 dtmp = points[i][0];
1032 points[i][0] = points[min_theta_index][0];
1033 points[min_theta_index][0] = dtmp;
1034
1035 dtmp = points[i][1];
1036 points[i][1] = points[min_theta_index][1];
1037 points[min_theta_index][1] = dtmp;
1038 }
1039 }
1040
1041 // compute area using cross product rule, points position are referred to the
1042 // 2-dimensional, local, reference frame centered in points[0]
1043
1044 Area = 0;
1045
1046 if (nPoints > 2) {
1047 for (i = 1; i < nPoints - 1; i++) {
1048 // Ax*By
1049 Area += (points[i][0] - points[0][0]) * (points[i + 1][1] - points[0][1]);
1050
1051 // Ay*Bx
1052 Area -= (points[i][1] - points[0][1]) * (points[i + 1][0] - points[0][0]);
1053 }
1054 }
1055
1056 return fabs(Area) / 2;
1057}

1059void CSlidingMesh::ComputeLineIntersectionPoint(const su2double* A1, const su2double* A2, const su2double* B1,
1060 const su2double* B2, su2double* IntersectionPoint) {
1061 /* --- Uses determinant rule to compute the intersection point between 2 straight segments --- */
1062 /* This works only for lines on a 2D plane, A1, A2 and B1, B2 are respectively the head and the tail points of each
1063 * segment, since they're on a 2D plane they are defined by a 2-elements array containing their coordinates */
1064
1065 su2double det;
1066
1067 det = (A1[0] - A2[0]) * (B1[1] - B2[1]) - (A1[1] - A2[1]) * (B1[0] - B2[0]);
1068
1069 if (det != 0.0) { // else there is no intersection point
1070 IntersectionPoint[0] =
1071 ((A1[0] * A2[1] - A1[1] * A2[0]) * (B1[0] - B2[0]) - (B1[0] * B2[1] - B1[1] * B2[0]) * (A1[0] - A2[0])) / det;
1072 IntersectionPoint[1] =
1073 ((A1[0] * A2[1] - A1[1] * A2[0]) * (B1[1] - B2[1]) - (B1[0] * B2[1] - B1[1] * B2[0]) * (A1[1] - A2[1])) / det;
1074 }
1075}

1077bool CSlidingMesh::CheckPointInsideTriangle(const su2double* Point, const su2double* T1, const su2double* T2,
1078 const su2double* T3) {
1079 /* --- Check whether a point "Point" lies inside or outside a triangle defined by 3 points "T1", "T2", "T3" --- */
1080 /* For each edge it checks on which side the point lies:
1081 * - Computes the unit vector pointing at the internal side of the edge
1082 * - Comutes the vector that connects the point to a point along the edge
1083 * - If the dot product is positive it means that the point is on the internal side of the edge
1084 * - If the check is positive for all the 3 edges, then the point lies within the triangle
1085 */
1086
1087 unsigned short iDim, check = 0;
1088
1089 su2double vect1[2], vect2[2], r[2];
1090 su2double dot;
1091
1092 constexpr unsigned short nDim = 2;
1093
1094 /* --- Check first edge --- */
1095
1096 dot = 0;
1097 for (iDim = 0; iDim < nDim; iDim++) {
1098 vect1[iDim] = T3[iDim] - T1[iDim]; // vec 1 is aligned to the edge
1099 vect2[iDim] = T2[iDim] - T1[iDim]; // vect 2 is the vector connecting one edge point to the third triangle vertex
1100
1101 r[iDim] = Point[iDim] - T1[iDim]; // Connects point to vertex T1
1102
1103 dot += vect2[iDim] * vect2[iDim];
1104 }
1105 dot = sqrt(dot);
1106
1107 for (iDim = 0; iDim < nDim; iDim++) vect2[iDim] /= dot;
1108
1109 dot = 0;
1110 for (iDim = 0; iDim < nDim; iDim++) dot += vect1[iDim] * vect2[iDim];
1111
1112 for (iDim = 0; iDim < nDim; iDim++)
1113 vect1[iDim] = T3[iDim] - (T1[iDim] + dot * vect2[iDim]); // Computes the inward unit vector
1114
1115 dot = 0;
1116 for (iDim = 0; iDim < nDim; iDim++) // Checs that the point lies on the internal plane
1117 dot += vect1[iDim] * r[iDim];
1118
1119 if (dot >= 0) check++;
1120
1121 /* --- Check second edge --- */
1122
1123 dot = 0;
1124 for (iDim = 0; iDim < nDim; iDim++) {
1125 vect1[iDim] = T1[iDim] - T2[iDim];
1126 vect2[iDim] = T3[iDim] - T2[iDim];
1127
1128 r[iDim] = Point[iDim] - T2[iDim];
1129
1130 dot += vect2[iDim] * vect2[iDim];
1131 }
1132 dot = sqrt(dot);
1133
1134 for (iDim = 0; iDim < nDim; iDim++) vect2[iDim] /= dot;
1135
1136 dot = 0;
1137 for (iDim = 0; iDim < nDim; iDim++) dot += vect1[iDim] * vect2[iDim];
1138
1139 for (iDim = 0; iDim < nDim; iDim++) vect1[iDim] = T1[iDim] - (T2[iDim] + dot * vect2[iDim]);
1140
1141 dot = 0;
1142 for (iDim = 0; iDim < nDim; iDim++) dot += vect1[iDim] * r[iDim];
1143
1144 if (dot >= 0) check++;
1145
1146 /* --- Check third edge --- */
1147
1148 dot = 0;
1149 for (iDim = 0; iDim < nDim; iDim++) {
1150 vect1[iDim] = T2[iDim] - T3[iDim];
1151 vect2[iDim] = T1[iDim] - T3[iDim];
1152
1153 r[iDim] = Point[iDim] - T3[iDim];
1154
1155 dot += vect2[iDim] * vect2[iDim];
1156 }
1157 dot = sqrt(dot);
1158
1159 for (iDim = 0; iDim < nDim; iDim++) vect2[iDim] /= dot;
1160
1161 dot = 0;
1162 for (iDim = 0; iDim < nDim; iDim++) dot += vect1[iDim] * vect2[iDim];
1163
1164 for (iDim = 0; iDim < nDim; iDim++) vect1[iDim] = T2[iDim] - (T3[iDim] + dot * vect2[iDim]);
1165
1166 dot = 0;
1167 for (iDim = 0; iDim < nDim; iDim++) dot += vect1[iDim] * r[iDim];
1168
1169 if (dot >= 0) check++;
1170
1171 return (check == 3);
1172}
