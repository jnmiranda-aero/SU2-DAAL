94  /*!
95  * \brief Heat-specific specific steps in the ComputeResidual method
96  * \param[in] config - Definition of the particular problem.
97  */
98  void FinishResidualCalc(const CConfig* config) override {
99  const su2double Thermal_Diffusivity_Mean = 0.5 * (*Diffusion_Coeff_i + *Diffusion_Coeff_j);
100 
101  Flux[0] = Thermal_Diffusivity_Mean * Proj_Mean_GradScalarVar[0];
102 
103  /*--- Use TSL for Jacobians. ---*/
104  Jacobian_i[0][0] = -Thermal_Diffusivity_Mean * proj_vector_ij;
105  Jacobian_j[0][0] = Thermal_Diffusivity_Mean * proj_vector_ij;
106  }
107 };
