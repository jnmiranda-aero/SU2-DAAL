35/*! \return ||a-b||^2 */
36template <class T, class U, typename Int>
37inline T SquaredDistance(Int nDim, const T* a, const U* b) {
38 T d(0);
39 for (Int i = 0; i < nDim; i++) d += pow(a[i] - b[i], 2);
40 return d;
41}

43/*! \return ||a-b|| */
44template <class T, class U, typename Int>
45inline T Distance(Int nDim, const T* a, const U* b) {
46 return sqrt(SquaredDistance(nDim, a, b));
47}

63template <class T, typename Int>
64inline T DotProduct(Int nDim, const T* a, const T* b) {

82/*! \brief c = a x b */
83template <class T>
84inline void CrossProduct(const T* a, const T* b, T* c) {
85 c[0] = a[1] * b[2] - a[2] * b[1];
86 c[1] = a[2] * b[0] - a[0] * b[2];
87 c[2] = a[0] * b[1] - a[1] * b[0];
88}

90/*!
91 * \brief Compute the coordinate (c) where the line defined by coordinate l0 and
92 * direction d intersects the plane defined by point p0 and normal n.
93 * \return The intersection distance.
94 */
95template <class T, int nDim>
96inline T LinePlaneIntersection(const T* l0, const T* d, const T* p0, const T* n, T* c) {
97 T dist[nDim] = {0.0};
98 Distance(nDim, p0, l0, dist);
99 T alpha = DotProduct(nDim, dist, n) / DotProduct(nDim, d, n);
100 for (int iDim = 0; iDim < nDim; ++iDim) c[iDim] = l0[iDim] + alpha * d[iDim];
101 return fabs(alpha) * Norm(nDim, d);
102}

104/*!
105 * \brief Compute the coordinate (c) where point p1 intersects the plane defined
106 * by point p0 and normal n if projected perpendicular to it.
107 * \return The normal distance.
108 */
109template <class T, int nDim>
110inline T PointPlaneProjection(const T* p1, const T* p0, const T* n, T* c) {
111 return LinePlaneIntersection<T, nDim>(p1, n, p0, n, c);
112}

114/*! \brief Set U as the normal to a 2D line defined by coords[iPoint][iDim]. */
115template <class T, class U>
116inline void LineNormal(const T& coords, U* normal) {
117 normal[0] = coords[0][1] - coords[1][1];
118 normal[1] = coords[1][0] - coords[0][0];
119}

121/*! \brief Normal vector of a triangle, cross product of two sides. */
122template <class T, class U>
123inline void TriangleNormal(const T& coords, U* normal) {
124 U a[3], b[3];
125
126 for (int iDim = 0; iDim < 3; iDim++) {
127 a[iDim] = coords[1][iDim] - coords[0][iDim];
128 b[iDim] = coords[2][iDim] - coords[0][iDim];
129 }
130
131 CrossProduct(a, b, normal);
132 normal[0] *= 0.5;
133 normal[1] *= 0.5;
134 normal[2] *= 0.5;
135}

137/*! \brief Normal vector of a quadrilateral, cross product of the two diagonals. */
138template <class T, class U>
139inline void QuadrilateralNormal(const T& coords, U* normal) {
140 U a[3], b[3];
141
142 for (int iDim = 0; iDim < 3; iDim++) {
143 a[iDim] = coords[2][iDim] - coords[0][iDim];
144 b[iDim] = coords[3][iDim] - coords[1][iDim];
145 }
146
147 CrossProduct(a, b, normal);
148 normal[0] *= 0.5;
149 normal[1] *= 0.5;
150 normal[2] *= 0.5;
151}

153/*! \brief Signed distance from a point to a plane defined by 3 coordinates. */
154template <class T, class U>
155inline U PointToPlaneDistance(const T& coords, const U* point) {
156 U normal[3], distance[3];
157 TriangleNormal(coords, normal);
158 Distance(3, point, coords[0], distance);
159 return DotProduct(3, distance, normal) / Norm(3, normal);
160}

162/*!
163 * \brief Compute a 3D rotation matrix.
164 * \note The implicit ordering is rotation about the x, y, and then z axis.
165 */
166template <class Scalar, class Matrix>
167inline void RotationMatrix(Scalar theta, Scalar phi, Scalar psi, Matrix& mat) {
168 Scalar cosTheta = cos(theta);
169 Scalar cosPhi = cos(phi);
170 Scalar cosPsi = cos(psi);
171 Scalar sinTheta = sin(theta);
172 Scalar sinPhi = sin(phi);
173 Scalar sinPsi = sin(psi);
174
175 mat[0][0] = cosPhi * cosPsi;
176 mat[1][0] = cosPhi * sinPsi;
177 mat[2][0] = -sinPhi;
178
179 mat[0][1] = sinTheta * sinPhi * cosPsi - cosTheta * sinPsi;
180 mat[1][1] = sinTheta * sinPhi * sinPsi + cosTheta * cosPsi;
181 mat[2][1] = sinTheta * cosPhi;
182
183 mat[0][2] = cosTheta * sinPhi * cosPsi + sinTheta * sinPsi;
184 mat[1][2] = cosTheta * sinPhi * sinPsi - sinTheta * cosPsi;
185 mat[2][2] = cosTheta * cosPhi;
186}

188/*! \brief Compute a 2D rotation matrix. */
189template <class Scalar, class Matrix>
190inline void RotationMatrix(Scalar psi, Matrix& mat) {
191 Scalar cosPsi = cos(psi);
192 Scalar sinPsi = sin(psi);
193
194 mat[0][0] = cosPsi;
195 mat[0][1] = -sinPsi;
196 mat[1][0] = sinPsi;
197 mat[1][1] = cosPsi;
198}

200/*! \brief Apply a rotation matrix (R) about origin (O) to a point at
201 * distance (d) from it to obtain new coordinate (c). */
202template <class Scalar, int nDim>
203inline void Rotate(const Scalar R[][nDim], const Scalar* O, const Scalar* d, Scalar* c) {
204 for (int iDim = 0; iDim < nDim; ++iDim) {
205 c[iDim] = O[iDim];
206 for (int k = 0; k < nDim; ++k) c[iDim] += R[iDim][k] * d[k];
207 }
208}

210/*! \brief Tangent projection */
211template <class Mat, class Scalar, class Int>
212inline void TangentProjection(Int nDim, const Mat& tensor, const Scalar* vector, Scalar* proj) {
213 for (Int iDim = 0; iDim < nDim; iDim++) proj[iDim] = DotProduct(nDim, tensor[iDim], vector);
214
215 auto normalProj = DotProduct(nDim, proj, vector);
216
217 for (Int iDim = 0; iDim < nDim; iDim++) proj[iDim] -= normalProj * vector[iDim];
218}
