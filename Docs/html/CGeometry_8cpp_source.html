139 void CGeometry::PreprocessP2PComms(CGeometry* geometry, CConfig* config) {
140  /*--- We start with the send and receive lists already available in
141  the form of SEND_RECEIVE boundary markers. We will loop through
142  these markers and establish the neighboring ranks and number of
143  send/recv points per pair. We will store this information and set
144  up persistent data structures so that we can reuse them throughout
145  the calculation for any point-to-point communications. The goal
146  is to break the non-blocking comms into InitiateComms() and
147  CompleteComms() in separate routines so that we can overlap the
148  communication and computation to hide the communication latency. ---*/
149 
150  /*--- Local variables. ---*/
151 
152  unsigned short iMarker;
153  unsigned long nVertexS, nVertexR, iVertex, MarkerS, MarkerR;
154 
155  int iRank, iSend, iRecv, count;
156 
157  /*--- Create some temporary structures for tracking sends/recvs. ---*/
158 
159  int* nPoint_Send_All = new int[size + 1];
160  nPoint_Send_All[0] = 0;
161  int* nPoint_Recv_All = new int[size + 1];
162  nPoint_Recv_All[0] = 0;
163  int* nPoint_Flag = new int[size];
164 
165  for (iRank = 0; iRank < size; iRank++) {
166  nPoint_Send_All[iRank] = 0;
167  nPoint_Recv_All[iRank] = 0;
168  nPoint_Flag[iRank] = -1;
169  }
170  nPoint_Send_All[size] = 0;
171  nPoint_Recv_All[size] = 0;
172 
173  /*--- Loop through all of our SEND_RECEIVE markers and track
174  our sends with each rank. ---*/
175 
176  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
177  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
178  /*--- Get the destination rank and number of points to send. ---*/
179 
180  iRank = config->GetMarker_All_SendRecv(iMarker) - 1;
181  nVertexS = geometry->nVertex[iMarker];
182 
183  /*--- If we have not visited this element yet, increment our
184  number of elements that must be sent to a particular proc. ---*/
185 
186  if ((nPoint_Flag[iRank] != (int)iMarker)) {
187  nPoint_Flag[iRank] = (int)iMarker;
188  nPoint_Send_All[iRank + 1] += nVertexS;
189  }
190  }
191  }
192 
193  delete[] nPoint_Flag;
194 
195  /*--- Communicate the number of points to be sent/recv'd amongst
196  all processors. After this communication, each proc knows how
197  many cells it will receive from each other processor. ---*/
198 
199  SU2_MPI::Alltoall(&(nPoint_Send_All[1]), 1, MPI_INT, &(nPoint_Recv_All[1]), 1, MPI_INT, SU2_MPI::GetComm());
200 
201  /*--- Prepare to send connectivities. First check how many
202  messages we will be sending and receiving. Here we also put
203  the counters into cumulative storage format to make the
204  communications simpler. ---*/
205 
206  nP2PSend = 0;
207  nP2PRecv = 0;
208 
209  for (iRank = 0; iRank < size; iRank++) {
210  if ((iRank != rank) && (nPoint_Send_All[iRank + 1] > 0)) nP2PSend++;
211  if ((iRank != rank) && (nPoint_Recv_All[iRank + 1] > 0)) nP2PRecv++;
212 
213  nPoint_Send_All[iRank + 1] += nPoint_Send_All[iRank];
214  nPoint_Recv_All[iRank + 1] += nPoint_Recv_All[iRank];
215  }
216 
217  /*--- Allocate only as much memory as we need for the P2P neighbors. ---*/
218 
219  nPoint_P2PSend = new int[nP2PSend + 1];
220  nPoint_P2PSend[0] = 0;
221  nPoint_P2PRecv = new int[nP2PRecv + 1];
222  nPoint_P2PRecv[0] = 0;
223 
224  Neighbors_P2PSend = new int[nP2PSend];
225  Neighbors_P2PRecv = new int[nP2PRecv];
226 
227  iSend = 0;
228  iRecv = 0;
229  for (iRank = 0; iRank < size; iRank++) {
230  if ((nPoint_Send_All[iRank + 1] > nPoint_Send_All[iRank]) && (iRank != rank)) {
231  Neighbors_P2PSend[iSend] = iRank;
232  nPoint_P2PSend[iSend + 1] = nPoint_Send_All[iRank + 1];
233  iSend++;
234  }
235 
236  if ((nPoint_Recv_All[iRank + 1] > nPoint_Recv_All[iRank]) && (iRank != rank)) {
237  Neighbors_P2PRecv[iRecv] = iRank;
238  nPoint_P2PRecv[iRecv + 1] = nPoint_Recv_All[iRank + 1];
239  iRecv++;
240  }
241  }
242 
243  /*--- Create a reverse mapping of the message to the rank so that we
244  can quickly access the correct data in the buffers when receiving
245  messages dynamically. ---*/
246 
247  P2PSend2Neighbor.clear();
248  for (iSend = 0; iSend < nP2PSend; iSend++) P2PSend2Neighbor[Neighbors_P2PSend[iSend]] = iSend;
249 
250  P2PRecv2Neighbor.clear();
251  for (iRecv = 0; iRecv < nP2PRecv; iRecv++) P2PRecv2Neighbor[Neighbors_P2PRecv[iRecv]] = iRecv;
252 
253  delete[] nPoint_Send_All;
254  delete[] nPoint_Recv_All;
255 
256  /*--- Allocate the memory that we need for receiving the conn
257  values and then cue up the non-blocking receives. Note that
258  we do not include our own rank in the communications. We will
259  directly copy our own data later. ---*/
260 
261  Local_Point_P2PSend = nullptr;
262  Local_Point_P2PSend = new unsigned long[nPoint_P2PSend[nP2PSend]];
263  for (iSend = 0; iSend < nPoint_P2PSend[nP2PSend]; iSend++) Local_Point_P2PSend[iSend] = 0;
264 
265  Local_Point_P2PRecv = nullptr;
266  Local_Point_P2PRecv = new unsigned long[nPoint_P2PRecv[nP2PRecv]];
267  for (iRecv = 0; iRecv < nPoint_P2PRecv[nP2PRecv]; iRecv++) Local_Point_P2PRecv[iRecv] = 0;
268 
269  /*--- We allocate the memory for communicating values in a later step
270  once we know the maximum packet size that we need to communicate. This
271  memory is deallocated and reallocated automatically in the case that
272  the previously allocated memory is not sufficient. ---*/
273 
274  bufD_P2PSend = nullptr;
275  bufD_P2PRecv = nullptr;
276 
277  bufS_P2PSend = nullptr;
278  bufS_P2PRecv = nullptr;
279 
280  /*--- Allocate memory for the MPI requests if we need to communicate. ---*/
281 
282  if (nP2PSend > 0) {
283  req_P2PSend = new SU2_MPI::Request[nP2PSend];
284  }
285  if (nP2PRecv > 0) {
286  req_P2PRecv = new SU2_MPI::Request[nP2PRecv];
287  }
288 
289  /*--- Build lists of local index values for send. ---*/
290 
291  count = 0;
292  for (iSend = 0; iSend < nP2PSend; iSend++) {
293  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
294  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
295  MarkerS = iMarker;
296  nVertexS = geometry->nVertex[MarkerS];
297  iRank = config->GetMarker_All_SendRecv(MarkerS) - 1;
298 
299  if (iRank == Neighbors_P2PSend[iSend]) {
300  for (iVertex = 0; iVertex < nVertexS; iVertex++) {
301  Local_Point_P2PSend[count] = geometry->vertex[MarkerS][iVertex]->GetNode();
302  count++;
303  }
304  }
305  }
306  }
307  }
308 
309  /*--- Build lists of local index values for receive. ---*/
310 
311  count = 0;
312  for (iRecv = 0; iRecv < nP2PRecv; iRecv++) {
313  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
314  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
315  MarkerR = iMarker + 1;
316  nVertexR = geometry->nVertex[MarkerR];
317  iRank = abs(config->GetMarker_All_SendRecv(MarkerR)) - 1;
318 
319  if (iRank == Neighbors_P2PRecv[iRecv]) {
320  for (iVertex = 0; iVertex < nVertexR; iVertex++) {
321  Local_Point_P2PRecv[count] = geometry->vertex[MarkerR][iVertex]->GetNode();
322  count++;
323  }
324  }
325  }
326  }
327  }
328 
329  /*--- In the future, some additional data structures could be created
330  here to separate the interior and boundary nodes in order to help
331  further overlap computation and communication. ---*/
332 }

334 void CGeometry::AllocateP2PComms(unsigned short countPerPoint) {
335  /*--- This routine is activated whenever we attempt to perform
336  a point-to-point MPI communication with our neighbors but the
337  memory buffer allocated is not large enough for the packet size.
338  Therefore, we deallocate the previously allocated space and
339  reallocate a large enough array. Note that after the first set
340  communications, this routine will not need to be called again. ---*/
341 
342  if (countPerPoint <= maxCountPerPoint) return;
343 
344  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
345  /*--- Store the larger packet size to the class data. ---*/
346 
347  maxCountPerPoint = countPerPoint;
348 
349  /*-- Deallocate and reallocate our su2double cummunication memory. ---*/
350 
351  delete[] bufD_P2PSend;
352  bufD_P2PSend = new su2double[maxCountPerPoint * nPoint_P2PSend[nP2PSend]]();
353 
354  delete[] bufD_P2PRecv;
355  bufD_P2PRecv = new su2double[maxCountPerPoint * nPoint_P2PRecv[nP2PRecv]]();
356 
357  delete[] bufS_P2PSend;
358  bufS_P2PSend = new unsigned short[maxCountPerPoint * nPoint_P2PSend[nP2PSend]]();
359 
360  delete[] bufS_P2PRecv;
361  bufS_P2PRecv = new unsigned short[maxCountPerPoint * nPoint_P2PRecv[nP2PRecv]]();
362  }
363  END_SU2_OMP_SAFE_GLOBAL_ACCESS
364 }

366 void CGeometry::PostP2PRecvs(CGeometry* geometry, const CConfig* config, unsigned short commType,
367  unsigned short countPerPoint, bool val_reverse) const {
368  /*--- Launch the non-blocking recv's first. Note that we have stored
369  the counts and sources, so we can launch these before we even load
370  the data and send from the neighbor ranks. ---*/
371 
372  SU2_OMP_MASTER
373  for (int iRecv = 0; iRecv < nP2PRecv; iRecv++) {
374  const auto iMessage = iRecv;
375 
376  /*--- In some instances related to the adjoint solver, we need
377  to reverse the direction of communications such that the normal
378  send nodes become the recv nodes and vice-versa. ---*/
379 
380  if (val_reverse) {
381  /*--- Compute our location in the buffer using the send data
382  structure since we are reversing the comms. ---*/
383 
384  auto offset = countPerPoint * nPoint_P2PSend[iRecv];
385 
386  /*--- Take advantage of cumulative storage format to get the number
387  of elems that we need to recv. Note again that we select the send
388  points here as the recv points. ---*/
389 
390  auto nPointP2P = nPoint_P2PSend[iRecv + 1] - nPoint_P2PSend[iRecv];
391 
392  /*--- Total count can include multiple pieces of data per element. ---*/
393 
394  auto count = countPerPoint * nPointP2P;
395 
396  /*--- Get the rank from which we receive the message. Note again
397  that we use the send rank as the source instead of the recv rank. ---*/
398 
399  auto source = Neighbors_P2PSend[iRecv];
400  auto tag = source + 1;
401 
402  /*--- Post non-blocking recv for this proc. Note that we use the
403  send buffer here too. This is important to make sure the arrays
404  are the correct size. ---*/
405 
406  switch (commType) {
407  case COMM_TYPE_DOUBLE:
408  SU2_MPI::Irecv(&(bufD_P2PSend[offset]), count, MPI_DOUBLE, source, tag, SU2_MPI::GetComm(),
409  &(req_P2PRecv[iRecv]));
410  break;
411  case COMM_TYPE_UNSIGNED_SHORT:
412  SU2_MPI::Irecv(&(bufS_P2PSend[offset]), count, MPI_UNSIGNED_SHORT, source, tag, SU2_MPI::GetComm(),
413  &(req_P2PRecv[iRecv]));
414  break;
415  default:
416  SU2_MPI::Error("Unrecognized data type for point-to-point MPI comms.", CURRENT_FUNCTION);
417  break;
418  }
419 
420  } else {
421  /*--- Compute our location in the recv buffer. ---*/
422 
423  auto offset = countPerPoint * nPoint_P2PRecv[iRecv];
424 
425  /*--- Take advantage of cumulative storage format to get the number
426  of elems that we need to recv. ---*/
427 
428  auto nPointP2P = nPoint_P2PRecv[iRecv + 1] - nPoint_P2PRecv[iRecv];
429 
430  /*--- Total count can include multiple pieces of data per element. ---*/
431 
432  auto count = countPerPoint * nPointP2P;
433 
434  /*--- Get the rank from which we receive the message. ---*/
435 
436  auto source = Neighbors_P2PRecv[iRecv];
437  auto tag = source + 1;
438 
439  /*--- Post non-blocking recv for this proc. ---*/
440 
441  switch (commType) {
442  case COMM_TYPE_DOUBLE:
443  SU2_MPI::Irecv(&(bufD_P2PRecv[offset]), count, MPI_DOUBLE, source, tag, SU2_MPI::GetComm(),
444  &(req_P2PRecv[iMessage]));
445  break;
446  case COMM_TYPE_UNSIGNED_SHORT:
447  SU2_MPI::Irecv(&(bufS_P2PRecv[offset]), count, MPI_UNSIGNED_SHORT, source, tag, SU2_MPI::GetComm(),
448  &(req_P2PRecv[iMessage]));
449  break;
450  default:
451  SU2_MPI::Error("Unrecognized data type for point-to-point MPI comms.", CURRENT_FUNCTION);
452  break;
453  }
454  }
455  }
456  END_SU2_OMP_MASTER
457 }

578 void CGeometry::InitiateComms(CGeometry* geometry, const CConfig* config, MPI_QUANTITIES commType) const {
579  if (nP2PSend == 0) return;
580 
581  /*--- Local variables ---*/
582 
583  unsigned short iDim;
584  unsigned short COUNT_PER_POINT = 0;
585  unsigned short MPI_TYPE = 0;
586 
587  unsigned long iPoint, msg_offset, buf_offset;
588 
589  int iMessage, iSend, nSend;
590 
591  /*--- Set the size of the data packet and type depending on quantity. ---*/
592 
593  GetCommCountAndType(config, commType, COUNT_PER_POINT, MPI_TYPE);
594 
595  /*--- Check to make sure we have created a large enough buffer
596  for these comms during preprocessing. This is only for the su2double
597  buffer. It will be reallocated whenever we find a larger count
598  per point. After the first cycle of comms, this should be inactive. ---*/
599 
600  geometry->AllocateP2PComms(COUNT_PER_POINT);
601 
602  /*--- Set some local pointers to make access simpler. ---*/
603 
604  su2double* bufDSend = geometry->bufD_P2PSend;
605  unsigned short* bufSSend = geometry->bufS_P2PSend;
606 
607  su2double* vector = nullptr;
608 
609  /*--- Load the specified quantity from the solver into the generic
610  communication buffer in the geometry class. ---*/
611 
612  /*--- Post all non-blocking recvs first before sends. ---*/
613 
614  geometry->PostP2PRecvs(geometry, config, MPI_TYPE, COUNT_PER_POINT, false);
615 
616  for (iMessage = 0; iMessage < nP2PSend; iMessage++) {
617  /*--- Get the offset in the buffer for the start of this message. ---*/
618 
619  msg_offset = nPoint_P2PSend[iMessage];
620 
621  /*--- Total count can include multiple pieces of data per element. ---*/
622 
623  nSend = (nPoint_P2PSend[iMessage + 1] - nPoint_P2PSend[iMessage]);
624 
625  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
626  for (iSend = 0; iSend < nSend; iSend++) {
627  /*--- Get the local index for this communicated data. ---*/
628 
629  iPoint = geometry->Local_Point_P2PSend[msg_offset + iSend];
630 
631  /*--- Compute the offset in the recv buffer for this point. ---*/
632 
633  buf_offset = (msg_offset + iSend) * COUNT_PER_POINT;
634 
635  switch (commType) {
636  case MPI_QUANTITIES::COORDINATES:
637  vector = nodes->GetCoord(iPoint);
638  for (iDim = 0; iDim < nDim; iDim++) bufDSend[buf_offset + iDim] = vector[iDim];
639  break;
640  case MPI_QUANTITIES::GRID_VELOCITY:
641  vector = nodes->GetGridVel(iPoint);
642  for (iDim = 0; iDim < nDim; iDim++) bufDSend[buf_offset + iDim] = vector[iDim];
643  break;
644  case MPI_QUANTITIES::COORDINATES_OLD:
645  vector = nodes->GetCoord_n(iPoint);
646  for (iDim = 0; iDim < nDim; iDim++) {
647  bufDSend[buf_offset + iDim] = vector[iDim];
648  }
649  if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND) {
650  vector = nodes->GetCoord_n1(iPoint);
651  for (iDim = 0; iDim < nDim; iDim++) {
652  bufDSend[buf_offset + nDim + iDim] = vector[iDim];
653  }
654  }
655  break;
656  case MPI_QUANTITIES::MAX_LENGTH:
657  bufDSend[buf_offset] = nodes->GetMaxLength(iPoint);
658  break;
659  case MPI_QUANTITIES::NEIGHBORS:
660  bufSSend[buf_offset] = geometry->nodes->GetnNeighbor(iPoint);
661  break;
662  default:
663  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
664  break;
665  }
666  }
667  END_SU2_OMP_FOR
668 
669  /*--- Launch the point-to-point MPI send for this message. ---*/
670 
671  geometry->PostP2PSends(geometry, config, MPI_TYPE, COUNT_PER_POINT, iMessage, false);
672  }

675 void CGeometry::CompleteComms(CGeometry* geometry, const CConfig* config, MPI_QUANTITIES commType) {
676  if (nP2PRecv == 0) return;
677 
678  /*--- Local variables ---*/
679 
680  unsigned short iDim, COUNT_PER_POINT = 0, MPI_TYPE = 0;
681  unsigned long iPoint, iRecv, nRecv, msg_offset, buf_offset;
682 
683  int ind, source, iMessage, jRecv;
684 
685  /*--- Global status so all threads can see the result of Waitany. ---*/
686  static SU2_MPI::Status status;
687 
688  /*--- Set the size of the data packet and type depending on quantity. ---*/
689 
690  GetCommCountAndType(config, commType, COUNT_PER_POINT, MPI_TYPE);
691 
692  /*--- Set some local pointers to make access simpler. ---*/
693 
694  const su2double* bufDRecv = geometry->bufD_P2PRecv;
695  const unsigned short* bufSRecv = geometry->bufS_P2PRecv;
696 
697  /*--- Store the data that was communicated into the appropriate
698  location within the local class data structures. Note that we
699  recv and store the data in any order to take advantage of the
700  non-blocking comms. ---*/
701 
702  for (iMessage = 0; iMessage < nP2PRecv; iMessage++) {
703  /*--- For efficiency, recv the messages dynamically based on
704  the order they arrive. ---*/
705 
706  SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitany(nP2PRecv, req_P2PRecv, &ind, &status);)
707 
708  /*--- Once we have recv'd a message, get the source rank. ---*/
709 
710  source = status.MPI_SOURCE;
711 
712  /*--- We know the offsets based on the source rank. ---*/
713 
714  jRecv = P2PRecv2Neighbor[source];
715 
716  /*--- Get the offset in the buffer for the start of this message. ---*/
717 
718  msg_offset = nPoint_P2PRecv[jRecv];
719 
720  /*--- Get the number of packets to be received in this message. ---*/
721 
722  nRecv = nPoint_P2PRecv[jRecv + 1] - nPoint_P2PRecv[jRecv];
723 
724  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
725  for (iRecv = 0; iRecv < nRecv; iRecv++) {
726  /*--- Get the local index for this communicated data. ---*/
727 
728  iPoint = geometry->Local_Point_P2PRecv[msg_offset + iRecv];
729 
730  /*--- Compute the total offset in the recv buffer for this point. ---*/
731 
732  buf_offset = (msg_offset + iRecv) * COUNT_PER_POINT;
733 
734  /*--- Store the data correctly depending on the quantity. ---*/
735 
736  switch (commType) {
737  case MPI_QUANTITIES::COORDINATES:
738  for (iDim = 0; iDim < nDim; iDim++) nodes->SetCoord(iPoint, iDim, bufDRecv[buf_offset + iDim]);
739  break;
740  case MPI_QUANTITIES::GRID_VELOCITY:
741  for (iDim = 0; iDim < nDim; iDim++) nodes->SetGridVel(iPoint, iDim, bufDRecv[buf_offset + iDim]);
742  break;
743  case MPI_QUANTITIES::COORDINATES_OLD:
744  nodes->SetCoord_n(iPoint, &bufDRecv[buf_offset]);
745  if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND)
746  nodes->SetCoord_n1(iPoint, &bufDRecv[buf_offset + nDim]);
747  break;
748  case MPI_QUANTITIES::MAX_LENGTH:
749  nodes->SetMaxLength(iPoint, bufDRecv[buf_offset]);
750  break;
751  case MPI_QUANTITIES::NEIGHBORS:
752  nodes->SetnNeighbor(iPoint, bufSRecv[buf_offset]);
753  break;
754  default:
755  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
756  break;
757  }
758  }
759  END_SU2_OMP_FOR
760  }
761 
762  /*--- Verify that all non-blocking point-to-point sends have finished.
763  Note that this should be satisfied, as we have received all of the
764  data in the loop above at this point. ---*/
765 
766 #ifdef HAVE_MPI
767  SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitall(nP2PSend, req_P2PSend, MPI_STATUS_IGNORE);)
768 #endif
769 }

771 void CGeometry::PreprocessPeriodicComms(CGeometry* geometry, CConfig* config) {
772  /*--- We start with the send and receive lists already available in
773  the form of stored periodic point-donor pairs. We will loop through
774  these markers and establish the neighboring ranks and number of
775  send/recv points per pair. We will store this information and set
776  up persistent data structures so that we can reuse them throughout
777  the calculation for any periodic boundary communications. The goal
778  is to break the non-blocking comms into InitiatePeriodicComms() and
779  CompletePeriodicComms() in separate routines so that we can overlap the
780  communication and computation to hide the communication latency. ---*/
781 
782  /*--- Local variables. ---*/
783 
784  unsigned short iMarker;
785  unsigned long iPoint, iVertex, iPeriodic;
786 
787  int iRank, iSend, iRecv, ii, jj;
788 
789  /*--- Create some temporary structures for tracking sends/recvs. ---*/
790 
791  int* nPoint_Send_All = new int[size + 1];
792  nPoint_Send_All[0] = 0;
793  int* nPoint_Recv_All = new int[size + 1];
794  nPoint_Recv_All[0] = 0;
795  int* nPoint_Flag = new int[size];
796 
797  for (iRank = 0; iRank < size; iRank++) {
798  nPoint_Send_All[iRank] = 0;
799  nPoint_Recv_All[iRank] = 0;
800  nPoint_Flag[iRank] = -1;
801  }
802  nPoint_Send_All[size] = 0;
803  nPoint_Recv_All[size] = 0;
804 
805  /*--- Loop through all of our periodic markers and track
806  our sends with each rank. ---*/
807 
808  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
809  if (config->GetMarker_All_KindBC(iMarker) == PERIODIC_BOUNDARY) {
810  iPeriodic = config->GetMarker_All_PerBound(iMarker);
811  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
812  /*--- Get the current periodic point index. We only communicate
813  the owned nodes on a rank, as the MPI comms will take care of
814  the halos after completing the periodic comms. ---*/
815 
816  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
817 
818  if (geometry->nodes->GetDomain(iPoint)) {
819  /*--- Get the rank that holds the matching periodic point
820  on the other marker in the periodic pair. ---*/
821 
822  iRank = (int)geometry->vertex[iMarker][iVertex]->GetDonorProcessor();
823 
824  /*--- If we have not visited this point last, increment our
825  number of points that must be sent to a particular proc. ---*/
826 
827  if ((nPoint_Flag[iRank] != (int)iPoint)) {
828  nPoint_Flag[iRank] = (int)iPoint;
829  nPoint_Send_All[iRank + 1] += 1;
830  }
831  }
832  }
833  }
834  }
835 
836  delete[] nPoint_Flag;
837 
838  /*--- Communicate the number of points to be sent/recv'd amongst
839  all processors. After this communication, each proc knows how
840  many periodic points it will receive from each other processor. ---*/
841 
842  SU2_MPI::Alltoall(&(nPoint_Send_All[1]), 1, MPI_INT, &(nPoint_Recv_All[1]), 1, MPI_INT, SU2_MPI::GetComm());
843 
844  /*--- Check how many messages we will be sending and receiving.
845  Here we also put the counters into cumulative storage format to
846  make the communications simpler. Note that we are allowing each
847  rank to communicate to themselves in these counters, although
848  it will not be done through MPI. ---*/
849 
850  nPeriodicSend = 0;
851  nPeriodicRecv = 0;
852 
853  for (iRank = 0; iRank < size; iRank++) {
854  if ((nPoint_Send_All[iRank + 1] > 0)) nPeriodicSend++;
855  if ((nPoint_Recv_All[iRank + 1] > 0)) nPeriodicRecv++;
856 
857  nPoint_Send_All[iRank + 1] += nPoint_Send_All[iRank];
858  nPoint_Recv_All[iRank + 1] += nPoint_Recv_All[iRank];
859  }
860 
861  /*--- Allocate only as much memory as needed for the periodic neighbors. ---*/
862 
863  nPoint_PeriodicSend = new int[nPeriodicSend + 1];
864  nPoint_PeriodicSend[0] = 0;
865  nPoint_PeriodicRecv = new int[nPeriodicRecv + 1];
866  nPoint_PeriodicRecv[0] = 0;
867 
868  Neighbors_PeriodicSend = new int[nPeriodicSend];
869  Neighbors_PeriodicRecv = new int[nPeriodicRecv];
870 
871  iSend = 0;
872  iRecv = 0;
873  for (iRank = 0; iRank < size; iRank++) {
874  if ((nPoint_Send_All[iRank + 1] > nPoint_Send_All[iRank])) {
875  Neighbors_PeriodicSend[iSend] = iRank;
876  nPoint_PeriodicSend[iSend + 1] = nPoint_Send_All[iRank + 1];
877  iSend++;
878  }
879  if ((nPoint_Recv_All[iRank + 1] > nPoint_Recv_All[iRank])) {
880  Neighbors_PeriodicRecv[iRecv] = iRank;
881  nPoint_PeriodicRecv[iRecv + 1] = nPoint_Recv_All[iRank + 1];
882  iRecv++;
883  }
884  }
885 
886  /*--- Create a reverse mapping of the message to the rank so that we
887  can quickly access the correct data in the buffers when receiving
888  messages dynamically later during the iterations. ---*/
889 
890  PeriodicSend2Neighbor.clear();
891  for (iSend = 0; iSend < nPeriodicSend; iSend++) PeriodicSend2Neighbor[Neighbors_PeriodicSend[iSend]] = iSend;
892 
893  PeriodicRecv2Neighbor.clear();
894  for (iRecv = 0; iRecv < nPeriodicRecv; iRecv++) PeriodicRecv2Neighbor[Neighbors_PeriodicRecv[iRecv]] = iRecv;
895 
896  delete[] nPoint_Send_All;
897  delete[] nPoint_Recv_All;
898 
899  /*--- Allocate the memory to store the local index values for both
900  the send and receive periodic points and periodic index. ---*/
901 
902  Local_Point_PeriodicSend = nullptr;
903  Local_Point_PeriodicSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend]];
904  for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend]; iSend++) Local_Point_PeriodicSend[iSend] = 0;
905 
906  Local_Marker_PeriodicSend = nullptr;
907  Local_Marker_PeriodicSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend]];
908  for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend]; iSend++) Local_Marker_PeriodicSend[iSend] = 0;
909 
910  Local_Point_PeriodicRecv = nullptr;
911  Local_Point_PeriodicRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv]];
912  for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++) Local_Point_PeriodicRecv[iRecv] = 0;
913 
914  Local_Marker_PeriodicRecv = nullptr;
915  Local_Marker_PeriodicRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv]];
916  for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++) Local_Marker_PeriodicRecv[iRecv] = 0;
917 
918  /*--- We allocate the buffers for communicating values in a later step
919  once we know the maximum packet size that we need to communicate. This
920  memory is deallocated and reallocated automatically in the case that
921  the previously allocated memory is not sufficient. ---*/
922 
923  bufD_PeriodicSend = nullptr;
924  bufD_PeriodicRecv = nullptr;
925 
926  bufS_PeriodicSend = nullptr;
927  bufS_PeriodicRecv = nullptr;
928 
929  /*--- Allocate memory for the MPI requests if we need to communicate. ---*/
930 
931  if (nPeriodicSend > 0) {
932  req_PeriodicSend = new SU2_MPI::Request[nPeriodicSend];
933  }
934  if (nPeriodicRecv > 0) {
935  req_PeriodicRecv = new SU2_MPI::Request[nPeriodicRecv];
936  }
937 
938  /*--- Allocate arrays for sending the periodic point index and marker
939  index to the recv rank so that it can store the local values. Therefore,
940  the recv rank can quickly loop through the buffers to unpack the data. ---*/
941 
942  unsigned short nPackets = 2;
943  auto* idSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend] * nPackets];
944  for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend] * nPackets; iSend++) idSend[iSend] = 0;
945 
946  /*--- Build the lists of local index and periodic marker index values. ---*/
947 
948  ii = 0;
949  jj = 0;
950  for (iSend = 0; iSend < nPeriodicSend; iSend++) {
951  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
952  if (config->GetMarker_All_KindBC(iMarker) == PERIODIC_BOUNDARY) {
953  iPeriodic = config->GetMarker_All_PerBound(iMarker);
954  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
955  /*--- Get the current periodic point index. We only communicate
956  the owned nodes on a rank, as the MPI comms will take care of
957  the halos after completing the periodic comms. ---*/
958 
959  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
960 
961  if (geometry->nodes->GetDomain(iPoint)) {
962  /*--- Get the rank that holds the matching periodic point
963  on the other marker in the periodic pair. ---*/
964 
965  iRank = (int)geometry->vertex[iMarker][iVertex]->GetDonorProcessor();
966 
967  /*--- If the rank for the current periodic point matches the
968  rank of the current send message, then store the local point
969  index on the matching periodic point and the periodic marker
970  index to be communicated to the recv rank. ---*/
971 
972  if (iRank == Neighbors_PeriodicSend[iSend]) {
973  Local_Point_PeriodicSend[ii] = iPoint;
974  Local_Marker_PeriodicSend[ii] = (unsigned long)iMarker;
975  jj = ii * nPackets;
976  idSend[jj] = geometry->vertex[iMarker][iVertex]->GetDonorPoint();
977  jj++;
978  idSend[jj] = (unsigned long)iPeriodic;
979  ii++;
980  }
981  }
982  }
983  }
984  }
985  }
986 
987  /*--- Allocate arrays for receiving the periodic point index and marker
988  index to the recv rank so that it can store the local values. ---*/
989 
990  auto* idRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv] * nPackets];
991  for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv] * nPackets; iRecv++) idRecv[iRecv] = 0;
992 
993 #ifdef HAVE_MPI
994 
995  int iMessage, offset, count, source, dest, tag;
996 
997  /*--- Launch the non-blocking recv's first. Note that we have stored
998  the counts and sources, so we can launch these before we even load
999  the data and send from the periodically matching ranks. ---*/
1000 
1001  iMessage = 0;
1002  for (iRecv = 0; iRecv < nPeriodicRecv; iRecv++) {
1003  /*--- Compute our location in the recv buffer. ---*/
1004 
1005  offset = nPackets * nPoint_PeriodicRecv[iRecv];
1006 
1007  /*--- Take advantage of cumulative storage format to get the number
1008  of elems that we need to recv. ---*/
1009 
1010  count = nPackets * (nPoint_PeriodicRecv[iRecv + 1] - nPoint_PeriodicRecv[iRecv]);
1011 
1012  /*--- Get the rank from which we receive the message. ---*/
1013 
1014  source = Neighbors_PeriodicRecv[iRecv];
1015  tag = source + 1;
1016 
1017  /*--- Post non-blocking recv for this proc. ---*/
1018 
1019  SU2_MPI::Irecv(&(static_cast<unsigned long*>(idRecv)[offset]), count, MPI_UNSIGNED_LONG, source, tag,
1020  SU2_MPI::GetComm(), &(req_PeriodicRecv[iMessage]));
1021 
1022  /*--- Increment message counter. ---*/
1023 
1024  iMessage++;
1025  }
1026 
1027  /*--- Post the non-blocking sends. ---*/
1028 
1029  iMessage = 0;
1030  for (iSend = 0; iSend < nPeriodicSend; iSend++) {
1031  /*--- Compute our location in the send buffer. ---*/
1032 
1033  offset = nPackets * nPoint_PeriodicSend[iSend];
1034 
1035  /*--- Take advantage of cumulative storage format to get the number
1036  of points that we need to send. ---*/
1037 
1038  count = nPackets * (nPoint_PeriodicSend[iSend + 1] - nPoint_PeriodicSend[iSend]);
1039 
1040  /*--- Get the rank to which we send the message. ---*/
1041 
1042  dest = Neighbors_PeriodicSend[iSend];
1043  tag = rank + 1;
1044 
1045  /*--- Post non-blocking send for this proc. ---*/
1046 
1047  SU2_MPI::Isend(&(static_cast<unsigned long*>(idSend)[offset]), count, MPI_UNSIGNED_LONG, dest, tag,
1048  SU2_MPI::GetComm(), &(req_PeriodicSend[iMessage]));
1049 
1050  /*--- Increment message counter. ---*/
1051 
1052  iMessage++;
1053  }
1054 
1055  /*--- Wait for the non-blocking comms to complete. ---*/
1056 
1057  SU2_MPI::Waitall(nPeriodicSend, req_PeriodicSend, MPI_STATUS_IGNORE);
1058  SU2_MPI::Waitall(nPeriodicRecv, req_PeriodicRecv, MPI_STATUS_IGNORE);
1059 
1060 #else
1061 
1062  /*--- Copy my own rank's data into the recv buffer directly in serial. ---*/
1063 
1064  int myStart, myFinal;
1065  for (int val_iSend = 0; val_iSend < nPeriodicSend; val_iSend++) {
1066  iRank = geometry->PeriodicRecv2Neighbor[rank];
1067  iRecv = geometry->nPoint_PeriodicRecv[iRank] * nPackets;
1068  myStart = nPoint_PeriodicSend[val_iSend] * nPackets;
1069  myFinal = nPoint_PeriodicSend[val_iSend + 1] * nPackets;
1070  for (iSend = myStart; iSend < myFinal; iSend++) {
1071  idRecv[iRecv] = idSend[iSend];
1072  iRecv++;
1073  }
1074  }
1075 
1076 #endif
1077 
1078  /*--- Store the local periodic point and marker index values in our
1079  data structures so we can quickly unpack data during the iterations. ---*/
1080 
1081  ii = 0;
1082  for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++) {
1083  Local_Point_PeriodicRecv[iRecv] = idRecv[ii];
1084  ii++;
1085  Local_Marker_PeriodicRecv[iRecv] = idRecv[ii];
1086  ii++;
1087  }
1088 
1089  delete[] idSend;
1090  delete[] idRecv;
1091 }

1130 void CGeometry::PostPeriodicRecvs(CGeometry* geometry, const CConfig* config, unsigned short commType,
1131  unsigned short countPerPeriodicPoint) {
1132  /*--- In parallel, communicate the data with non-blocking send/recv. ---*/
1133 
1134 #ifdef HAVE_MPI
1135 
1136  /*--- Launch the non-blocking recv's first. Note that we have stored
1137  the counts and sources, so we can launch these before we even load
1138  the data and send from the neighbor ranks. ---*/
1139 
1140  SU2_OMP_MASTER
1141  for (int iRecv = 0; iRecv < nPeriodicRecv; iRecv++) {
1142  /*--- Compute our location in the recv buffer. ---*/
1143 
1144  auto offset = countPerPeriodicPoint * nPoint_PeriodicRecv[iRecv];
1145 
1146  /*--- Take advantage of cumulative storage format to get the number
1147  of elems that we need to recv. ---*/
1148 
1149  auto nPointPeriodic = nPoint_PeriodicRecv[iRecv + 1] - nPoint_PeriodicRecv[iRecv];
1150 
1151  /*--- Total count can include multiple pieces of data per element. ---*/
1152 
1153  auto count = countPerPeriodicPoint * nPointPeriodic;
1154 
1155  /*--- Get the rank from which we receive the message. ---*/
1156 
1157  auto source = Neighbors_PeriodicRecv[iRecv];

2902 void CGeometry::FilterValuesAtElementCG(const vector<su2double>& filter_radius,

2904  const unsigned short search_limit, su2double* values) const {
2905  /*--- Apply a filter to "input_values". The filter is an averaging process over the neighbourhood
2906  of each element, which is a circle in 2D and a sphere in 3D of radius "filter_radius".
2907  The filter is characterized by its kernel, i.e. how the weights are computed. Multiple kernels
2908  can be specified in which case they are applied sequentially (each one being applied to the
2909  output values of the previous filter. ---*/
2910 
2911  /*--- Check if we need to do any work. ---*/
2912  if (kernels.empty()) return;
2913 
2914  /*--- FIRST: Gather the adjacency matrix, element centroids, volumes, and values on every
2915  processor, this is required because the filter reaches far into adjacent partitions. ---*/
2916 
2917  /*--- Adjacency matrix ---*/
2918  vector<unsigned long> neighbour_start;
2919  long* neighbour_idx = nullptr;
2920  GetGlobalElementAdjacencyMatrix(neighbour_start, neighbour_idx);
2921 
2922  /*--- Element centroids and volumes. ---*/
2923  auto *cg_elem = new su2double[Global_nElemDomain * nDim], *vol_elem = new su2double[Global_nElemDomain];
2924 #ifdef HAVE_MPI
2925  /*--- Number of subdomain each point is part of. ---*/
2926  vector<char> halo_detect(Global_nElemDomain);
2927 #endif
2928 
2929  /*--- Inputs of a filter stage, like with CG and volumes, each processor needs to see everything. ---*/
2930  auto* work_values = new su2double[Global_nElemDomain];
2931 
2932  /*--- When gathering the neighborhood of each element we use a vector of booleans to indicate
2933  whether an element is already added to the list of neighbors (one vector per thread). ---*/
2934  vector<vector<bool>> is_neighbor(omp_get_max_threads());
2935 
2936  /*--- Begin OpenMP parallel section, count total number of searches for which
2937  the recursion limit is reached and the full neighborhood is not considered. ---*/
2938  unsigned long limited_searches = 0;
2939 
2940  SU2_OMP_PARALLEL_(reduction(+ : limited_searches)) {
2941  /*--- Initialize ---*/
2942  SU2_OMP_FOR_STAT(256)
2943  for (auto iElem = 0ul; iElem < Global_nElemDomain; ++iElem) {
2944  for (unsigned short iDim = 0; iDim < nDim; ++iDim) cg_elem[nDim * iElem + iDim] = 0.0;
2945  vol_elem[iElem] = 0.0;
2946  }
2947  END_SU2_OMP_FOR
2948 
2949  /*--- Populate ---*/
2950  SU2_OMP_FOR_STAT(256)
2951  for (auto iElem = 0ul; iElem < nElem; ++iElem) {
2952  auto iElem_global = elem[iElem]->GetGlobalIndex();
2953  for (unsigned short iDim = 0; iDim < nDim; ++iDim) cg_elem[nDim * iElem_global + iDim] = elem[iElem]->GetCG(iDim);
2954  vol_elem[iElem_global] = elem[iElem]->GetVolume();
2955  }
2956  END_SU2_OMP_FOR
2957

2960  reduction using MPI_SUM, which is required to maintain differentiabillity. ---*/

2989  su2double numRepeat = halo_detect[iElem];

2994 #endif

3052

3056  case ENUM_FILTER_KERNEL::CONICAL_WEIGHT:
3057  case ENUM_FILTER_KERNEL::GAUSSIAN_WEIGHT:

3080  denominator += weight;
3081  }

3086  case ENUM_FILTER_KERNEL::DILATE_MORPH:
3087  case ENUM_FILTER_KERNEL::ERODE_MORPH:
3088

3757  for (auto iMarker = 0u; iMarker < config->GetnMarker_All(); iMarker++) {

3767  }
3768  }
3769  li.linelets.resize(nLinelet);
3770 
3771  /*--- Create the linelet structure. ---*/
3772 
3773  nLinelet = 0;
3774  for (auto& linelet : li.linelets) {
3775  while (linelet.size() < CLineletInfo::MAX_LINELET_POINTS) {
3776  const auto iPoint = linelet.back();
3777 
3778  /*--- Compute the value of the max and min weights to detect if this region is isotropic. ---*/
3779 
3780  su2double max_weight = 0.0, min_weight = std::numeric_limits<su2double>::max();
3781  for (auto iNode = 0u; iNode < nodes->GetnPoint(iPoint); iNode++) {
3782  const auto jPoint = nodes->GetPoint(iPoint, iNode);
3783  const auto iEdge = nodes->GetEdge(iPoint, iNode);
3784  const auto* normal = edges->GetNormal(iEdge);

3786  const su2double volume_iPoint = nodes->GetVolume(iPoint);
3787  const su2double volume_jPoint = nodes->GetVolume(jPoint);
3788  const su2double weight = 0.5 * area * (1.0 / volume_iPoint + 1.0 / volume_jPoint);
3789  max_weight = max(max_weight, weight);
3790  min_weight = min(min_weight, weight);
3791  }
3792 
3793  /*--- Isotropic, stop this linelet. ---*/

3795 
3796  /*--- Otherwise, add the closest valid neighbor. ---*/
3797 
3798  su2double min_dist2 = std::numeric_limits<su2double>::max();
3799  auto next_Point = iPoint;
3800  const auto* iCoord = nodes->GetCoord(iPoint);
3801 
3802  for (const auto jPoint : nodes->GetPoints(iPoint)) {
3803  if (li.lineletIdx[jPoint] == CLineletInfo::NO_LINELET && nodes->GetDomain(jPoint)) {
3804  const auto* jCoord = nodes->GetCoord(jPoint);

3806  su2double cosTheta = 1;
3807  if (linelet.size() > 1) {

3809  su2double dij[3] = {0.0}, dki[3] = {0.0};

3811  GeometryToolbox::Distance(nDim, jCoord, iCoord, dij);
3812  cosTheta = GeometryToolbox::DotProduct(3, dki, dij) / sqrt(d2 * GeometryToolbox::SquaredNorm(nDim, dki));

3818  }
3819  }
3820

3823

3825  li.lineletIdx[next_Point] = nLinelet;
3826  }
3827  ++nLinelet;
3828

3854  for (const auto jPoint : nodes->GetPoints(iPoint)) {

3870  sortedLinelets.reserve(nLinelet);
3871  li.colorOffsets.reserve(nColors + 1);
3872  li.colorOffsets.push_back(0);

3876  /*--- Store the new linelet index for its points. ---*/
3877  for (const auto iPoint : li.linelets[iLine]) {
3878  li.lineletIdx[iPoint] = sortedLinelets.size();
3879  }
3880  sortedLinelets.push_back(std::move(li.linelets[iLine]));
3881  }
3882  li.colorOffsets.push_back(sortedLinelets.size());
3883  }

3899  }
