139void CGeometry::PreprocessP2PComms(CGeometry* geometry, CConfig* config) {
140 /*--- We start with the send and receive lists already available in
141 the form of SEND_RECEIVE boundary markers. We will loop through
142 these markers and establish the neighboring ranks and number of
143 send/recv points per pair. We will store this information and set
144 up persistent data structures so that we can reuse them throughout
145 the calculation for any point-to-point communications. The goal
146 is to break the non-blocking comms into InitiateComms() and
147 CompleteComms() in separate routines so that we can overlap the
148 communication and computation to hide the communication latency. ---*/
149
150 /*--- Local variables. ---*/
151
152 unsigned short iMarker;
153 unsigned long nVertexS, nVertexR, iVertex, MarkerS, MarkerR;
154
155 int iRank, iSend, iRecv, count;
156
157 /*--- Create some temporary structures for tracking sends/recvs. ---*/
158
159 int* nPoint_Send_All = new int[size + 1];
160 nPoint_Send_All[0] = 0;
161 int* nPoint_Recv_All = new int[size + 1];
162 nPoint_Recv_All[0] = 0;
163 int* nPoint_Flag = new int[size];
164
165 for (iRank = 0; iRank < size; iRank++) {
166 nPoint_Send_All[iRank] = 0;
167 nPoint_Recv_All[iRank] = 0;
168 nPoint_Flag[iRank] = -1;
169 }
170 nPoint_Send_All[size] = 0;
171 nPoint_Recv_All[size] = 0;
172
173 /*--- Loop through all of our SEND_RECEIVE markers and track
174 our sends with each rank. ---*/
175
176 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
177 if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
178 /*--- Get the destination rank and number of points to send. ---*/
179
180 iRank = config->GetMarker_All_SendRecv(iMarker) - 1;
181 nVertexS = geometry->nVertex[iMarker];
182
183 /*--- If we have not visited this element yet, increment our
184 number of elements that must be sent to a particular proc. ---*/
185
186 if ((nPoint_Flag[iRank] != (int)iMarker)) {
187 nPoint_Flag[iRank] = (int)iMarker;
188 nPoint_Send_All[iRank + 1] += nVertexS;
189 }
190 }
191 }
192
193 delete[] nPoint_Flag;
194
195 /*--- Communicate the number of points to be sent/recv'd amongst
196 all processors. After this communication, each proc knows how
197 many cells it will receive from each other processor. ---*/
198
199 SU2_MPI::Alltoall(&(nPoint_Send_All[1]), 1, MPI_INT, &(nPoint_Recv_All[1]), 1, MPI_INT, SU2_MPI::GetComm());
200
201 /*--- Prepare to send connectivities. First check how many
202 messages we will be sending and receiving. Here we also put
203 the counters into cumulative storage format to make the
204 communications simpler. ---*/
205
206 nP2PSend = 0;
207 nP2PRecv = 0;
208
209 for (iRank = 0; iRank < size; iRank++) {
210 if ((iRank != rank) && (nPoint_Send_All[iRank + 1] > 0)) nP2PSend++;
211 if ((iRank != rank) && (nPoint_Recv_All[iRank + 1] > 0)) nP2PRecv++;
212
213 nPoint_Send_All[iRank + 1] += nPoint_Send_All[iRank];
214 nPoint_Recv_All[iRank + 1] += nPoint_Recv_All[iRank];
215 }
216
217 /*--- Allocate only as much memory as we need for the P2P neighbors. ---*/
218
219 nPoint_P2PSend = new int[nP2PSend + 1];
220 nPoint_P2PSend[0] = 0;
221 nPoint_P2PRecv = new int[nP2PRecv + 1];
222 nPoint_P2PRecv[0] = 0;
223
224 Neighbors_P2PSend = new int[nP2PSend];
225 Neighbors_P2PRecv = new int[nP2PRecv];
226
227 iSend = 0;
228 iRecv = 0;
229 for (iRank = 0; iRank < size; iRank++) {
230 if ((nPoint_Send_All[iRank + 1] > nPoint_Send_All[iRank]) && (iRank != rank)) {
231 Neighbors_P2PSend[iSend] = iRank;
232 nPoint_P2PSend[iSend + 1] = nPoint_Send_All[iRank + 1];
233 iSend++;
234 }
235
236 if ((nPoint_Recv_All[iRank + 1] > nPoint_Recv_All[iRank]) && (iRank != rank)) {
237 Neighbors_P2PRecv[iRecv] = iRank;
238 nPoint_P2PRecv[iRecv + 1] = nPoint_Recv_All[iRank + 1];
239 iRecv++;
240 }
241 }
242
243 /*--- Create a reverse mapping of the message to the rank so that we
244 can quickly access the correct data in the buffers when receiving
245 messages dynamically. ---*/
246
247 P2PSend2Neighbor.clear();
248 for (iSend = 0; iSend < nP2PSend; iSend++) P2PSend2Neighbor[Neighbors_P2PSend[iSend]] = iSend;
249
250 P2PRecv2Neighbor.clear();
251 for (iRecv = 0; iRecv < nP2PRecv; iRecv++) P2PRecv2Neighbor[Neighbors_P2PRecv[iRecv]] = iRecv;
252
253 delete[] nPoint_Send_All;
254 delete[] nPoint_Recv_All;
255
256 /*--- Allocate the memory that we need for receiving the conn
257 values and then cue up the non-blocking receives. Note that
258 we do not include our own rank in the communications. We will
259 directly copy our own data later. ---*/
260
261 Local_Point_P2PSend = nullptr;
262 Local_Point_P2PSend = new unsigned long[nPoint_P2PSend[nP2PSend]];
263 for (iSend = 0; iSend < nPoint_P2PSend[nP2PSend]; iSend++) Local_Point_P2PSend[iSend] = 0;
264
265 Local_Point_P2PRecv = nullptr;
266 Local_Point_P2PRecv = new unsigned long[nPoint_P2PRecv[nP2PRecv]];
267 for (iRecv = 0; iRecv < nPoint_P2PRecv[nP2PRecv]; iRecv++) Local_Point_P2PRecv[iRecv] = 0;
268
269 /*--- We allocate the memory for communicating values in a later step
270 once we know the maximum packet size that we need to communicate. This
271 memory is deallocated and reallocated automatically in the case that
272 the previously allocated memory is not sufficient. ---*/
273
274 bufD_P2PSend = nullptr;
275 bufD_P2PRecv = nullptr;
276
277 bufS_P2PSend = nullptr;
278 bufS_P2PRecv = nullptr;
279
280 /*--- Allocate memory for the MPI requests if we need to communicate. ---*/
281
282 if (nP2PSend > 0) {
283 req_P2PSend = new SU2_MPI::Request[nP2PSend];
284 }
285 if (nP2PRecv > 0) {
286 req_P2PRecv = new SU2_MPI::Request[nP2PRecv];
287 }
288
289 /*--- Build lists of local index values for send. ---*/
290
291 count = 0;
292 for (iSend = 0; iSend < nP2PSend; iSend++) {
293 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
294 if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
295 MarkerS = iMarker;
296 nVertexS = geometry->nVertex[MarkerS];
297 iRank = config->GetMarker_All_SendRecv(MarkerS) - 1;
298
299 if (iRank == Neighbors_P2PSend[iSend]) {
300 for (iVertex = 0; iVertex < nVertexS; iVertex++) {
301 Local_Point_P2PSend[count] = geometry->vertex[MarkerS][iVertex]->GetNode();
302 count++;
303 }
304 }
305 }
306 }
307 }
308
309 /*--- Build lists of local index values for receive. ---*/
310
311 count = 0;
312 for (iRecv = 0; iRecv < nP2PRecv; iRecv++) {
313 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
314 if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) && (config->GetMarker_All_SendRecv(iMarker) > 0)) {
315 MarkerR = iMarker + 1;
316 nVertexR = geometry->nVertex[MarkerR];
317 iRank = abs(config->GetMarker_All_SendRecv(MarkerR)) - 1;
318
319 if (iRank == Neighbors_P2PRecv[iRecv]) {
320 for (iVertex = 0; iVertex < nVertexR; iVertex++) {
321 Local_Point_P2PRecv[count] = geometry->vertex[MarkerR][iVertex]->GetNode();
322 count++;
323 }
324 }
325 }
326 }
327 }
328
329 /*--- In the future, some additional data structures could be created
330 here to separate the interior and boundary nodes in order to help
331 further overlap computation and communication. ---*/
332}

334void CGeometry::AllocateP2PComms(unsigned short countPerPoint) {
335 /*--- This routine is activated whenever we attempt to perform
336 a point-to-point MPI communication with our neighbors but the
337 memory buffer allocated is not large enough for the packet size.
338 Therefore, we deallocate the previously allocated space and
339 reallocate a large enough array. Note that after the first set
340 communications, this routine will not need to be called again. ---*/
341
342 if (countPerPoint <= maxCountPerPoint) return;
343
344 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
345 /*--- Store the larger packet size to the class data. ---*/
346
347 maxCountPerPoint = countPerPoint;
348
349 /*-- Deallocate and reallocate our su2double cummunication memory. ---*/
350
351 delete[] bufD_P2PSend;
352 bufD_P2PSend = new su2double[maxCountPerPoint * nPoint_P2PSend[nP2PSend]]();
353
354 delete[] bufD_P2PRecv;
355 bufD_P2PRecv = new su2double[maxCountPerPoint * nPoint_P2PRecv[nP2PRecv]]();
356
357 delete[] bufS_P2PSend;
358 bufS_P2PSend = new unsigned short[maxCountPerPoint * nPoint_P2PSend[nP2PSend]]();
359
360 delete[] bufS_P2PRecv;
361 bufS_P2PRecv = new unsigned short[maxCountPerPoint * nPoint_P2PRecv[nP2PRecv]]();
362 }
363 END_SU2_OMP_SAFE_GLOBAL_ACCESS
364}

366void CGeometry::PostP2PRecvs(CGeometry* geometry, const CConfig* config, unsigned short commType,
367 unsigned short countPerPoint, bool val_reverse) const {
368 /*--- Launch the non-blocking recv's first. Note that we have stored
369 the counts and sources, so we can launch these before we even load
370 the data and send from the neighbor ranks. ---*/
371
372 SU2_OMP_MASTER
373 for (int iRecv = 0; iRecv < nP2PRecv; iRecv++) {
374 const auto iMessage = iRecv;
375
376 /*--- In some instances related to the adjoint solver, we need
377 to reverse the direction of communications such that the normal
378 send nodes become the recv nodes and vice-versa. ---*/
379
380 if (val_reverse) {
381 /*--- Compute our location in the buffer using the send data
382 structure since we are reversing the comms. ---*/
383
384 auto offset = countPerPoint * nPoint_P2PSend[iRecv];
385
386 /*--- Take advantage of cumulative storage format to get the number
387 of elems that we need to recv. Note again that we select the send
388 points here as the recv points. ---*/
389
390 auto nPointP2P = nPoint_P2PSend[iRecv + 1] - nPoint_P2PSend[iRecv];
391
392 /*--- Total count can include multiple pieces of data per element. ---*/
393
394 auto count = countPerPoint * nPointP2P;
395
396 /*--- Get the rank from which we receive the message. Note again
397 that we use the send rank as the source instead of the recv rank. ---*/
398
399 auto source = Neighbors_P2PSend[iRecv];
400 auto tag = source + 1;
401
402 /*--- Post non-blocking recv for this proc. Note that we use the
403 send buffer here too. This is important to make sure the arrays
404 are the correct size. ---*/
405
406 switch (commType) {
407 case COMM_TYPE_DOUBLE:
408 SU2_MPI::Irecv(&(bufD_P2PSend[offset]), count, MPI_DOUBLE, source, tag, SU2_MPI::GetComm(),
409 &(req_P2PRecv[iRecv]));
410 break;
411 case COMM_TYPE_UNSIGNED_SHORT:
412 SU2_MPI::Irecv(&(bufS_P2PSend[offset]), count, MPI_UNSIGNED_SHORT, source, tag, SU2_MPI::GetComm(),
413 &(req_P2PRecv[iRecv]));
414 break;
415 default:
416 SU2_MPI::Error("Unrecognized data type for point-to-point MPI comms.", CURRENT_FUNCTION);
417 break;
418 }
419
420 } else {
421 /*--- Compute our location in the recv buffer. ---*/
422
423 auto offset = countPerPoint * nPoint_P2PRecv[iRecv];
424
425 /*--- Take advantage of cumulative storage format to get the number
426 of elems that we need to recv. ---*/
427
428 auto nPointP2P = nPoint_P2PRecv[iRecv + 1] - nPoint_P2PRecv[iRecv];
429
430 /*--- Total count can include multiple pieces of data per element. ---*/
431
432 auto count = countPerPoint * nPointP2P;
433
434 /*--- Get the rank from which we receive the message. ---*/
435
436 auto source = Neighbors_P2PRecv[iRecv];
437 auto tag = source + 1;
438
439 /*--- Post non-blocking recv for this proc. ---*/
440
441 switch (commType) {
442 case COMM_TYPE_DOUBLE:
443 SU2_MPI::Irecv(&(bufD_P2PRecv[offset]), count, MPI_DOUBLE, source, tag, SU2_MPI::GetComm(),
444 &(req_P2PRecv[iMessage]));
445 break;
446 case COMM_TYPE_UNSIGNED_SHORT:
447 SU2_MPI::Irecv(&(bufS_P2PRecv[offset]), count, MPI_UNSIGNED_SHORT, source, tag, SU2_MPI::GetComm(),
448 &(req_P2PRecv[iMessage]));
449 break;
450 default:
451 SU2_MPI::Error("Unrecognized data type for point-to-point MPI comms.", CURRENT_FUNCTION);
452 break;
453 }
454 }
455 }
456 END_SU2_OMP_MASTER
457}

549 case COORDINATES:

578void CGeometry::InitiateComms(CGeometry* geometry, const CConfig* config, unsigned short commType) const {
579 if (nP2PSend == 0) return;
580
581 /*--- Local variables ---*/
582
583 unsigned short iDim;
584 unsigned short COUNT_PER_POINT = 0;
585 unsigned short MPI_TYPE = 0;
586
587 unsigned long iPoint, msg_offset, buf_offset;
588
589 int iMessage, iSend, nSend;
590
591 /*--- Set the size of the data packet and type depending on quantity. ---*/
592
593 GetCommCountAndType(config, commType, COUNT_PER_POINT, MPI_TYPE);
594
595 /*--- Check to make sure we have created a large enough buffer
596 for these comms during preprocessing. This is only for the su2double
597 buffer. It will be reallocated whenever we find a larger count
598 per point. After the first cycle of comms, this should be inactive. ---*/
599
600 geometry->AllocateP2PComms(COUNT_PER_POINT);
601
602 /*--- Set some local pointers to make access simpler. ---*/
603
604 su2double* bufDSend = geometry->bufD_P2PSend;
605 unsigned short* bufSSend = geometry->bufS_P2PSend;
606
607 su2double* vector = nullptr;
608
609 /*--- Load the specified quantity from the solver into the generic
610 communication buffer in the geometry class. ---*/
611
612 /*--- Post all non-blocking recvs first before sends. ---*/
613
614 geometry->PostP2PRecvs(geometry, config, MPI_TYPE, COUNT_PER_POINT, false);
615
616 for (iMessage = 0; iMessage < nP2PSend; iMessage++) {
617 /*--- Get the offset in the buffer for the start of this message. ---*/
618
619 msg_offset = nPoint_P2PSend[iMessage];
620
621 /*--- Total count can include multiple pieces of data per element. ---*/
622
623 nSend = (nPoint_P2PSend[iMessage + 1] - nPoint_P2PSend[iMessage]);
624
625 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
626 for (iSend = 0; iSend < nSend; iSend++) {
627 /*--- Get the local index for this communicated data. ---*/
628
629 iPoint = geometry->Local_Point_P2PSend[msg_offset + iSend];
630
631 /*--- Compute the offset in the recv buffer for this point. ---*/
632
633 buf_offset = (msg_offset + iSend) * COUNT_PER_POINT;
634
635 switch (commType) {
636 case COORDINATES:
637 vector = nodes->GetCoord(iPoint);
638 for (iDim = 0; iDim < nDim; iDim++) bufDSend[buf_offset + iDim] = vector[iDim];
639 break;
640 case GRID_VELOCITY:
641 vector = nodes->GetGridVel(iPoint);
642 for (iDim = 0; iDim < nDim; iDim++) bufDSend[buf_offset + iDim] = vector[iDim];
643 break;
644 case COORDINATES_OLD:
645 vector = nodes->GetCoord_n(iPoint);
646 for (iDim = 0; iDim < nDim; iDim++) {
647 bufDSend[buf_offset + iDim] = vector[iDim];
648 }
649 if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND) {
650 vector = nodes->GetCoord_n1(iPoint);
651 for (iDim = 0; iDim < nDim; iDim++) {
652 bufDSend[buf_offset + nDim + iDim] = vector[iDim];
653 }
654 }
655 break;
656 case MAX_LENGTH:
657 bufDSend[buf_offset] = nodes->GetMaxLength(iPoint);
658 break;
659 case NEIGHBORS:
660 bufSSend[buf_offset] = geometry->nodes->GetnNeighbor(iPoint);
661 break;
662 default:
663 SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
664 break;
665 }
666 }
667 END_SU2_OMP_FOR
668
669 /*--- Launch the point-to-point MPI send for this message. ---*/
670
671 geometry->PostP2PSends(geometry, config, MPI_TYPE, COUNT_PER_POINT, iMessage, false);
672 }

675void CGeometry::CompleteComms(CGeometry* geometry, const CConfig* config, unsigned short commType) {
676 if (nP2PRecv == 0) return;
677
678 /*--- Local variables ---*/
679
680 unsigned short iDim, COUNT_PER_POINT = 0, MPI_TYPE = 0;
681 unsigned long iPoint, iRecv, nRecv, msg_offset, buf_offset;
682
683 int ind, source, iMessage, jRecv;
684
685 /*--- Global status so all threads can see the result of Waitany. ---*/
686 static SU2_MPI::Status status;
687
688 /*--- Set the size of the data packet and type depending on quantity. ---*/
689
690 GetCommCountAndType(config, commType, COUNT_PER_POINT, MPI_TYPE);
691
692 /*--- Set some local pointers to make access simpler. ---*/
693
694 const su2double* bufDRecv = geometry->bufD_P2PRecv;
695 const unsigned short* bufSRecv = geometry->bufS_P2PRecv;
696
697 /*--- Store the data that was communicated into the appropriate
698 location within the local class data structures. Note that we
699 recv and store the data in any order to take advantage of the
700 non-blocking comms. ---*/
701
702 for (iMessage = 0; iMessage < nP2PRecv; iMessage++) {
703 /*--- For efficiency, recv the messages dynamically based on
704 the order they arrive. ---*/
705
706 SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitany(nP2PRecv, req_P2PRecv, &ind, &status);)
707
708 /*--- Once we have recv'd a message, get the source rank. ---*/
709
710 source = status.MPI_SOURCE;
711
712 /*--- We know the offsets based on the source rank. ---*/
713
714 jRecv = P2PRecv2Neighbor[source];
715
716 /*--- Get the offset in the buffer for the start of this message. ---*/
717
718 msg_offset = nPoint_P2PRecv[jRecv];
719
720 /*--- Get the number of packets to be received in this message. ---*/
721
722 nRecv = nPoint_P2PRecv[jRecv + 1] - nPoint_P2PRecv[jRecv];
723
724 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
725 for (iRecv = 0; iRecv < nRecv; iRecv++) {
726 /*--- Get the local index for this communicated data. ---*/
727
728 iPoint = geometry->Local_Point_P2PRecv[msg_offset + iRecv];
729
730 /*--- Compute the total offset in the recv buffer for this point. ---*/
731
732 buf_offset = (msg_offset + iRecv) * COUNT_PER_POINT;
733
734 /*--- Store the data correctly depending on the quantity. ---*/
735
736 switch (commType) {
737 case COORDINATES:
738 for (iDim = 0; iDim < nDim; iDim++) nodes->SetCoord(iPoint, iDim, bufDRecv[buf_offset + iDim]);
739 break;
740 case GRID_VELOCITY:
741 for (iDim = 0; iDim < nDim; iDim++) nodes->SetGridVel(iPoint, iDim, bufDRecv[buf_offset + iDim]);
742 break;
743 case COORDINATES_OLD:
744 nodes->SetCoord_n(iPoint, &bufDRecv[buf_offset]);
745 if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND)
746 nodes->SetCoord_n1(iPoint, &bufDRecv[buf_offset + nDim]);
747 break;
748 case MAX_LENGTH:
749 nodes->SetMaxLength(iPoint, bufDRecv[buf_offset]);
750 break;
751 case NEIGHBORS:
752 nodes->SetnNeighbor(iPoint, bufSRecv[buf_offset]);
753 break;
754 default:
755 SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
756 break;
757 }
758 }
759 END_SU2_OMP_FOR
760 }
761
762 /*--- Verify that all non-blocking point-to-point sends have finished.
763 Note that this should be satisfied, as we have received all of the
764 data in the loop above at this point. ---*/
765
766#ifdef HAVE_MPI
767 SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitall(nP2PSend, req_P2PSend, MPI_STATUS_IGNORE);)
768#endif
769}

771void CGeometry::PreprocessPeriodicComms(CGeometry* geometry, CConfig* config) {
772 /*--- We start with the send and receive lists already available in
773 the form of stored periodic point-donor pairs. We will loop through
774 these markers and establish the neighboring ranks and number of
775 send/recv points per pair. We will store this information and set
776 up persistent data structures so that we can reuse them throughout
777 the calculation for any periodic boundary communications. The goal
778 is to break the non-blocking comms into InitiatePeriodicComms() and
779 CompletePeriodicComms() in separate routines so that we can overlap the
780 communication and computation to hide the communication latency. ---*/
781
782 /*--- Local variables. ---*/
783
784 unsigned short iMarker;
785 unsigned long iPoint, iVertex, iPeriodic;
786
787 int iRank, iSend, iRecv, ii, jj;
788
789 /*--- Create some temporary structures for tracking sends/recvs. ---*/
790
791 int* nPoint_Send_All = new int[size + 1];
792 nPoint_Send_All[0] = 0;
793 int* nPoint_Recv_All = new int[size + 1];
794 nPoint_Recv_All[0] = 0;
795 int* nPoint_Flag = new int[size];
796
797 for (iRank = 0; iRank < size; iRank++) {
798 nPoint_Send_All[iRank] = 0;
799 nPoint_Recv_All[iRank] = 0;
800 nPoint_Flag[iRank] = -1;
801 }
802 nPoint_Send_All[size] = 0;
803 nPoint_Recv_All[size] = 0;
804
805 /*--- Loop through all of our periodic markers and track
806 our sends with each rank. ---*/
807
808 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
809 if (config->GetMarker_All_KindBC(iMarker) == PERIODIC_BOUNDARY) {
810 iPeriodic = config->GetMarker_All_PerBound(iMarker);
811 for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
812 /*--- Get the current periodic point index. We only communicate
813 the owned nodes on a rank, as the MPI comms will take care of
814 the halos after completing the periodic comms. ---*/
815
816 iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
817
818 if (geometry->nodes->GetDomain(iPoint)) {
819 /*--- Get the rank that holds the matching periodic point
820 on the other marker in the periodic pair. ---*/
821
822 iRank = (int)geometry->vertex[iMarker][iVertex]->GetDonorProcessor();
823
824 /*--- If we have not visited this point last, increment our
825 number of points that must be sent to a particular proc. ---*/
826
827 if ((nPoint_Flag[iRank] != (int)iPoint)) {
828 nPoint_Flag[iRank] = (int)iPoint;
829 nPoint_Send_All[iRank + 1] += 1;
830 }
831 }
832 }
833 }
834 }
835
836 delete[] nPoint_Flag;
837
838 /*--- Communicate the number of points to be sent/recv'd amongst
839 all processors. After this communication, each proc knows how
840 many periodic points it will receive from each other processor. ---*/
841
842 SU2_MPI::Alltoall(&(nPoint_Send_All[1]), 1, MPI_INT, &(nPoint_Recv_All[1]), 1, MPI_INT, SU2_MPI::GetComm());
843
844 /*--- Check how many messages we will be sending and receiving.
845 Here we also put the counters into cumulative storage format to
846 make the communications simpler. Note that we are allowing each
847 rank to communicate to themselves in these counters, although
848 it will not be done through MPI. ---*/
849
850 nPeriodicSend = 0;
851 nPeriodicRecv = 0;
852
853 for (iRank = 0; iRank < size; iRank++) {
854 if ((nPoint_Send_All[iRank + 1] > 0)) nPeriodicSend++;
855 if ((nPoint_Recv_All[iRank + 1] > 0)) nPeriodicRecv++;
856
857 nPoint_Send_All[iRank + 1] += nPoint_Send_All[iRank];
858 nPoint_Recv_All[iRank + 1] += nPoint_Recv_All[iRank];
859 }
860
861 /*--- Allocate only as much memory as needed for the periodic neighbors. ---*/
862
863 nPoint_PeriodicSend = new int[nPeriodicSend + 1];
864 nPoint_PeriodicSend[0] = 0;
865 nPoint_PeriodicRecv = new int[nPeriodicRecv + 1];
866 nPoint_PeriodicRecv[0] = 0;
867
868 Neighbors_PeriodicSend = new int[nPeriodicSend];
869 Neighbors_PeriodicRecv = new int[nPeriodicRecv];
870
871 iSend = 0;
872 iRecv = 0;
873 for (iRank = 0; iRank < size; iRank++) {
874 if ((nPoint_Send_All[iRank + 1] > nPoint_Send_All[iRank])) {
875 Neighbors_PeriodicSend[iSend] = iRank;
876 nPoint_PeriodicSend[iSend + 1] = nPoint_Send_All[iRank + 1];
877 iSend++;
878 }
879 if ((nPoint_Recv_All[iRank + 1] > nPoint_Recv_All[iRank])) {
880 Neighbors_PeriodicRecv[iRecv] = iRank;
881 nPoint_PeriodicRecv[iRecv + 1] = nPoint_Recv_All[iRank + 1];
882 iRecv++;
883 }
884 }
885
886 /*--- Create a reverse mapping of the message to the rank so that we
887 can quickly access the correct data in the buffers when receiving
888 messages dynamically later during the iterations. ---*/
889
890 PeriodicSend2Neighbor.clear();
891 for (iSend = 0; iSend < nPeriodicSend; iSend++) PeriodicSend2Neighbor[Neighbors_PeriodicSend[iSend]] = iSend;
892
893 PeriodicRecv2Neighbor.clear();
894 for (iRecv = 0; iRecv < nPeriodicRecv; iRecv++) PeriodicRecv2Neighbor[Neighbors_PeriodicRecv[iRecv]] = iRecv;
895
896 delete[] nPoint_Send_All;
897 delete[] nPoint_Recv_All;
898
899 /*--- Allocate the memory to store the local index values for both
900 the send and receive periodic points and periodic index. ---*/
901
902 Local_Point_PeriodicSend = nullptr;
903 Local_Point_PeriodicSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend]];
904 for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend]; iSend++) Local_Point_PeriodicSend[iSend] = 0;
905
906 Local_Marker_PeriodicSend = nullptr;
907 Local_Marker_PeriodicSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend]];
908 for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend]; iSend++) Local_Marker_PeriodicSend[iSend] = 0;
909
910 Local_Point_PeriodicRecv = nullptr;
911 Local_Point_PeriodicRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv]];
912 for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++) Local_Point_PeriodicRecv[iRecv] = 0;
913
914 Local_Marker_PeriodicRecv = nullptr;
915 Local_Marker_PeriodicRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv]];
916 for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++) Local_Marker_PeriodicRecv[iRecv] = 0;
917
918 /*--- We allocate the buffers for communicating values in a later step
919 once we know the maximum packet size that we need to communicate. This
920 memory is deallocated and reallocated automatically in the case that
921 the previously allocated memory is not sufficient. ---*/
922
923 bufD_PeriodicSend = nullptr;
924 bufD_PeriodicRecv = nullptr;
925
926 bufS_PeriodicSend = nullptr;
927 bufS_PeriodicRecv = nullptr;
928
929 /*--- Allocate memory for the MPI requests if we need to communicate. ---*/
930
931 if (nPeriodicSend > 0) {
932 req_PeriodicSend = new SU2_MPI::Request[nPeriodicSend];
933 }
934 if (nPeriodicRecv > 0) {
935 req_PeriodicRecv = new SU2_MPI::Request[nPeriodicRecv];
936 }
937
938 /*--- Allocate arrays for sending the periodic point index and marker
939 index to the recv rank so that it can store the local values. Therefore,
940 the recv rank can quickly loop through the buffers to unpack the data. ---*/
941
942 unsigned short nPackets = 2;
943 auto* idSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend] * nPackets];
944 for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend] * nPackets; iSend++) idSend[iSend] = 0;
945
946 /*--- Build the lists of local index and periodic marker index values. ---*/
947
948 ii = 0;
949 jj = 0;
950 for (iSend = 0; iSend < nPeriodicSend; iSend++) {
951 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
952 if (config->GetMarker_All_KindBC(iMarker) == PERIODIC_BOUNDARY) {
953 iPeriodic = config->GetMarker_All_PerBound(iMarker);
954 for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
955 /*--- Get the current periodic point index. We only communicate
956 the owned nodes on a rank, as the MPI comms will take care of
957 the halos after completing the periodic comms. ---*/
958
959 iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
960
961 if (geometry->nodes->GetDomain(iPoint)) {
962 /*--- Get the rank that holds the matching periodic point
963 on the other marker in the periodic pair. ---*/
964
965 iRank = (int)geometry->vertex[iMarker][iVertex]->GetDonorProcessor();
966
967 /*--- If the rank for the current periodic point matches the
968 rank of the current send message, then store the local point
969 index on the matching periodic point and the periodic marker
970 index to be communicated to the recv rank. ---*/
971
972 if (iRank == Neighbors_PeriodicSend[iSend]) {
973 Local_Point_PeriodicSend[ii] = iPoint;
974 Local_Marker_PeriodicSend[ii] = (unsigned long)iMarker;
975 jj = ii * nPackets;
976 idSend[jj] = geometry->vertex[iMarker][iVertex]->GetDonorPoint();
977 jj++;
978 idSend[jj] = (unsigned long)iPeriodic;
979 ii++;
980 }
981 }
982 }
983 }
984 }
985 }
986
987 /*--- Allocate arrays for receiving the periodic point index and marker
988 index to the recv rank so that it can store the local values. ---*/
989
990 auto* idRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv] * nPackets];
991 for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv] * nPackets; iRecv++) idRecv[iRecv] = 0;
992
993#ifdef HAVE_MPI
994
995 int iMessage, offset, count, source, dest, tag;
996
997 /*--- Launch the non-blocking recv's first. Note that we have stored
998 the counts and sources, so we can launch these before we even load
999 the data and send from the periodically matching ranks. ---*/
1000
1001 iMessage = 0;
1002 for (iRecv = 0; iRecv < nPeriodicRecv; iRecv++) {
1003 /*--- Compute our location in the recv buffer. ---*/
1004
1005 offset = nPackets * nPoint_PeriodicRecv[iRecv];
1006
1007 /*--- Take advantage of cumulative storage format to get the number
1008 of elems that we need to recv. ---*/
1009
1010 count = nPackets * (nPoint_PeriodicRecv[iRecv + 1] - nPoint_PeriodicRecv[iRecv]);
1011
1012 /*--- Get the rank from which we receive the message. ---*/
1013
1014 source = Neighbors_PeriodicRecv[iRecv];
1015 tag = source + 1;
1016
1017 /*--- Post non-blocking recv for this proc. ---*/
1018
1019 SU2_MPI::Irecv(&(static_cast<unsigned long*>(idRecv)[offset]), count, MPI_UNSIGNED_LONG, source, tag,
1020 SU2_MPI::GetComm(), &(req_PeriodicRecv[iMessage]));
1021
1022 /*--- Increment message counter. ---*/
1023
1024 iMessage++;
1025 }
1026
1027 /*--- Post the non-blocking sends. ---*/
1028
1029 iMessage = 0;
1030 for (iSend = 0; iSend < nPeriodicSend; iSend++) {
1031 /*--- Compute our location in the send buffer. ---*/
1032
1033 offset = nPackets * nPoint_PeriodicSend[iSend];
1034
1035 /*--- Take advantage of cumulative storage format to get the number
1036 of points that we need to send. ---*/
1037
1038 count = nPackets * (nPoint_PeriodicSend[iSend + 1] - nPoint_PeriodicSend[iSend]);
1039
1040 /*--- Get the rank to which we send the message. ---*/
1041
1042 dest = Neighbors_PeriodicSend[iSend];
1043 tag = rank + 1;
1044
1045 /*--- Post non-blocking send for this proc. ---*/
1046
1047 SU2_MPI::Isend(&(static_cast<unsigned long*>(idSend)[offset]), count, MPI_UNSIGNED_LONG, dest, tag,
1048 SU2_MPI::GetComm(), &(req_PeriodicSend[iMessage]));
1049
1050 /*--- Increment message counter. ---*/
1051
1052 iMessage++;
1053 }
1054
1055 /*--- Wait for the non-blocking comms to complete. ---*/
1056
1057 SU2_MPI::Waitall(nPeriodicSend, req_PeriodicSend, MPI_STATUS_IGNORE);
1058 SU2_MPI::Waitall(nPeriodicRecv, req_PeriodicRecv, MPI_STATUS_IGNORE);
1059
1060#else
1061
1062 /*--- Copy my own rank's data into the recv buffer directly in serial. ---*/
1063
1064 int myStart, myFinal;
1065 for (int val_iSend = 0; val_iSend < nPeriodicSend; val_iSend++) {
1066 iRank = geometry->PeriodicRecv2Neighbor[rank];
1067 iRecv = geometry->nPoint_PeriodicRecv[iRank] * nPackets;
1068 myStart = nPoint_PeriodicSend[val_iSend] * nPackets;
1069 myFinal = nPoint_PeriodicSend[val_iSend + 1] * nPackets;
1070 for (iSend = myStart; iSend < myFinal; iSend++) {
1071 idRecv[iRecv] = idSend[iSend];
1072 iRecv++;
1073 }
1074 }
1075
1076#endif
1077
1078 /*--- Store the local periodic point and marker index values in our
1079 data structures so we can quickly unpack data during the iterations. ---*/
1080
1081 ii = 0;
1082 for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++) {
1083 Local_Point_PeriodicRecv[iRecv] = idRecv[ii];
1084 ii++;
1085 Local_Marker_PeriodicRecv[iRecv] = idRecv[ii];
1086 ii++;
1087 }
1088
1089 delete[] idSend;
1090 delete[] idRecv;
1091}

1093void CGeometry::AllocatePeriodicComms(unsigned short countPerPeriodicPoint) {
1094 /*--- This routine is activated whenever we attempt to perform
1095 a periodic MPI communication with our neighbors but the
1096 memory buffer allocated is not large enough for the packet size.
1097 Therefore, we deallocate the previously allocated arrays and
1098 reallocate a large enough array. Note that after the first set
1099 communications, this routine will not need to be called again. ---*/
1100
1101 if (countPerPeriodicPoint <= maxCountPerPeriodicPoint) return;
1102
1103 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
1104 /*--- Store the larger packet size to the class data. ---*/
1105
1106 maxCountPerPeriodicPoint = countPerPeriodicPoint;
1107
1108 /*--- Store the total size of the send/recv arrays for clarity. ---*/
1109
1110 auto nSend = countPerPeriodicPoint * nPoint_PeriodicSend[nPeriodicSend];
1111 auto nRecv = countPerPeriodicPoint * nPoint_PeriodicRecv[nPeriodicRecv];
1112
1113 /*-- Deallocate and reallocate our cummunication memory. ---*/
1114
1115 delete[] bufD_PeriodicSend;
1116 bufD_PeriodicSend = new su2double[nSend]();
1117
1118 delete[] bufD_PeriodicRecv;
1119 bufD_PeriodicRecv = new su2double[nRecv]();
1120
1121 delete[] bufS_PeriodicSend;
1122 bufS_PeriodicSend = new unsigned short[nSend]();
1123
1124 delete[] bufS_PeriodicRecv;
1125 bufS_PeriodicRecv = new unsigned short[nRecv]();
1126 }
1127 END_SU2_OMP_SAFE_GLOBAL_ACCESS
1128}

1133
1134#ifdef HAVE_MPI
1135

1137 the counts and sources, so we can launch these before we even load

1139
1140 SU2_OMP_MASTER
1141 for (int iRecv = 0; iRecv < nPeriodicRecv; iRecv++) {
1142 /*--- Compute our location in the recv buffer. ---*/

1181void CGeometry::PostPeriodicSends(CGeometry* geometry, const CConfig* config, unsigned short commType,
1182 unsigned short countPerPeriodicPoint, int val_iSend) const {
1183 /*--- In parallel, communicate the data with non-blocking send/recv. ---*/
1184
1185#ifdef HAVE_MPI
1186 SU2_OMP_MASTER {
1187 /*--- Post the non-blocking send as soon as the buffer is loaded. ---*/
1188
1189 /*--- Compute our location in the send buffer. ---*/
1190
1191 auto offset = countPerPeriodicPoint * nPoint_PeriodicSend[val_iSend];
1192
1193 /*--- Take advantage of cumulative storage format to get the number
1194 of points that we need to send. ---*/
1195
1196 auto nPointPeriodic = (nPoint_PeriodicSend[val_iSend + 1] - nPoint_PeriodicSend[val_iSend]);
1197
1198 /*--- Total count can include multiple pieces of data per element. ---*/
1199
1200 auto count = countPerPeriodicPoint * nPointPeriodic;
1201
1202 /*--- Get the rank to which we send the message. ---*/
1203
1204 auto dest = Neighbors_PeriodicSend[val_iSend];
1205 auto tag = rank + 1;
1206
1207 /*--- Post non-blocking send for this proc. ---*/
1208
1209 switch (commType) {
1210 case COMM_TYPE_DOUBLE:
1211 SU2_MPI::Isend(&(static_cast<su2double*>(bufD_PeriodicSend)[offset]), count, MPI_DOUBLE, dest, tag,
1212 SU2_MPI::GetComm(), &(req_PeriodicSend[val_iSend]));
1213 break;
1214 case COMM_TYPE_UNSIGNED_SHORT:
1215 SU2_MPI::Isend(&(static_cast<unsigned short*>(bufS_PeriodicSend)[offset]), count, MPI_UNSIGNED_SHORT, dest, tag,
1216 SU2_MPI::GetComm(), &(req_PeriodicSend[val_iSend]));
1217 break;
1218 default:
1219 SU2_MPI::Error("Unrecognized data type for periodic MPI comms.", CURRENT_FUNCTION);
1220 break;
1221 }
1222 }
1223 END_SU2_OMP_MASTER
1224#else
1225

1353bool CGeometry::SegmentIntersectsPlane(const su2double* Segment_P0, const su2double* Segment_P1, su2double Variable_P0,
1354 su2double Variable_P1, const su2double* Plane_P0, const su2double* Plane_Normal,
1355 su2double* Intersection, su2double& Variable_Interp) {
1356 su2double u[3], v[3], Denominator, Numerator, Aux, ModU;
1357 su2double epsilon =
1358 1E-6; // An epsilon is added to eliminate, as much as possible, the posibility of a line that intersects a point
1359 unsigned short iDim;
1360
1361 for (iDim = 0; iDim < 3; iDim++) {
1362 u[iDim] = Segment_P1[iDim] - Segment_P0[iDim];
1363 v[iDim] = (Plane_P0[iDim] + epsilon) - Segment_P0[iDim];
1364 }
1365
1366 ModU = sqrt(u[0] * u[0] + u[1] * u[1] + u[2] * u[2]);
1367
1368 Numerator =
1369 (Plane_Normal[0] + epsilon) * v[0] + (Plane_Normal[1] + epsilon) * v[1] + (Plane_Normal[2] + epsilon) * v[2];
1370 Denominator =
1371 (Plane_Normal[0] + epsilon) * u[0] + (Plane_Normal[1] + epsilon) * u[1] + (Plane_Normal[2] + epsilon) * u[2];
1372
1373 if (fabs(Denominator) <= 0.0) return (false); // No intersection.
1374
1375 Aux = Numerator / Denominator;
1376
1377 if (Aux < 0.0 || Aux > 1.0) return (false); // No intersection.
1378
1379 for (iDim = 0; iDim < 3; iDim++) Intersection[iDim] = Segment_P0[iDim] + Aux * u[iDim];
1380
1381 /*--- Check that the intersection is in the segment ---*/
1382
1383 for (iDim = 0; iDim < 3; iDim++) {
1384 u[iDim] = Segment_P0[iDim] - Intersection[iDim];
1385 v[iDim] = Segment_P1[iDim] - Intersection[iDim];
1386 }
1387
1388 Variable_Interp = Variable_P0 + (Variable_P1 - Variable_P0) * sqrt(u[0] * u[0] + u[1] * u[1] + u[2] * u[2]) / ModU;
1389
1390 Denominator =
1391 (Plane_Normal[0] + epsilon) * u[0] + (Plane_Normal[1] + epsilon) * u[1] + (Plane_Normal[2] + epsilon) * u[2];
1392 Numerator =
1393 (Plane_Normal[0] + epsilon) * v[0] + (Plane_Normal[1] + epsilon) * v[1] + (Plane_Normal[2] + epsilon) * v[2];
1394
1395 Aux = Numerator * Denominator;
1396
1397 if (Aux > 0.0) return (false); // Intersection outside the segment.
1398
1399 return (true);
1400}

1402bool CGeometry::RayIntersectsTriangle(const su2double orig[3], const su2double dir[3], const su2double vert0[3],
1403 const su2double vert1[3], const su2double vert2[3], su2double* intersect) {
1404 const passivedouble epsilon = 0.000001;
1405 su2double edge1[3], edge2[3], tvec[3], pvec[3], qvec[3];
1406 su2double det, inv_det, t, u, v;
1407
1408 /*--- Find vectors for two edges sharing vert0 ---*/
1409
1410 GeometryToolbox::Distance(3, vert1, vert0, edge1);
1411 GeometryToolbox::Distance(3, vert2, vert0, edge2);
1412
1413 /*--- Begin calculating determinant - also used to calculate U parameter ---*/
1414
1415 GeometryToolbox::CrossProduct(dir, edge2, pvec);
1416
1417 /*--- If determinant is near zero, ray lies in plane of triangle ---*/
1418
1419 det = GeometryToolbox::DotProduct(3, edge1, pvec);
1420
1421 if (fabs(det) < epsilon) return (false);
1422
1423 inv_det = 1.0 / det;
1424
1425 /*--- Calculate distance from vert0 to ray origin ---*/
1426
1427 GeometryToolbox::Distance(3, orig, vert0, tvec);
1428
1429 /*--- Calculate U parameter and test bounds ---*/
1430
1431 u = inv_det * GeometryToolbox::DotProduct(3, tvec, pvec);
1432
1433 if (u < 0.0 || u > 1.0) return (false);
1434
1435 /*--- prepare to test V parameter ---*/
1436
1437 GeometryToolbox::CrossProduct(tvec, edge1, qvec);
1438
1439 /*--- Calculate V parameter and test bounds ---*/
1440
1441 v = inv_det * GeometryToolbox::DotProduct(3, dir, qvec);
1442
1443 if (v < 0.0 || u + v > 1.0) return (false);
1444
1445 /*--- Calculate t, ray intersects triangle ---*/
1446
1447 t = inv_det * GeometryToolbox::DotProduct(3, edge2, qvec);
1448
1449 /*--- Compute the intersection point in cartesian coordinates ---*/
1450
1451 intersect[0] = orig[0] + (t * dir[0]);
1452 intersect[1] = orig[1] + (t * dir[1]);
1453 intersect[2] = orig[2] + (t * dir[2]);
1454
1455 return (true);
1456}

1458bool CGeometry::SegmentIntersectsLine(const su2double point0[2], const su2double point1[2], const su2double vert0[2],
1459 const su2double vert1[2]) {
1460 su2double det, diff0_A, diff0_B, diff1_A, diff1_B, intersect[2];
1461
1462 diff0_A = point0[0] - point1[0];
1463 diff1_A = point0[1] - point1[1];
1464
1465 diff0_B = vert0[0] - vert1[0];
1466 diff1_B = vert0[1] - vert1[1];
1467
1468 det = (diff0_A) * (diff1_B) - (diff1_A) * (diff0_B);
1469
1470 if (det == 0) return false;
1471
1472 /*--- Compute point of intersection ---*/
1473
1474 intersect[0] = ((point0[0] * point1[1] - point0[1] * point1[0]) * diff0_B -
1475 (vert0[0] * vert1[1] - vert0[1] * vert1[0]) * diff0_A) /
1476 det;
1477
1478 intersect[1] = ((point0[0] * point1[1] - point0[1] * point1[0]) * diff1_B -
1479 (vert0[0] * vert1[1] - vert0[1] * vert1[0]) * diff1_A) /
1480 det;
1481
1482 /*--- Check that the point is between the two surface points ---*/
1483
1484 su2double dist0, dist1, length;
1485
1486 dist0 =
1487 (intersect[0] - point0[0]) * (intersect[0] - point0[0]) + (intersect[1] - point0[1]) * (intersect[1] - point0[1]);
1488
1489 dist1 =
1490 (intersect[0] - point1[0]) * (intersect[0] - point1[0]) + (intersect[1] - point1[1]) * (intersect[1] - point1[1]);
1491
1492 length = diff0_A * diff0_A + diff1_A * diff1_A;
1493
1494 return (dist0 <= length) && (dist1 <= length);
1495}

2559

2707 Length_U = sqrt(Length_U);
2708 Length_V = sqrt(Length_V);
2709 Length_W = sqrt(Length_W);
2710 for (iDim = 0; iDim < nDim; iDim++) {
2711 U[iDim] /= Length_U;
2712 V[iDim] /= Length_V;
2713 CosValue += U[iDim] * V[iDim];
2714 }
2715 if (CosValue >= 1.0) CosValue = 1.0;
2716 if (CosValue <= -1.0) CosValue = -1.0;
2717
2718 Angle_Value = acos(CosValue);
2719 Area_Vertex[iPoint] += Length_W;
2720 Angle_Defect[iPoint] -= Angle_Value;
2721 if (Angle_Alpha[iEdge] == 0.0)
2722 Angle_Alpha[iEdge] = Angle_Value;
2723 else
2724 Angle_Beta[iEdge] = Angle_Value;
2725 }
2726 }
2727 }
2728 }
2729 }
2730
2731 /*--- Compute mean curvature ---*/
2732 for (iMarker = 0; iMarker < nMarker; iMarker++) {
2733 if (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) {
2734 for (iElem_Bound = 0; iElem_Bound < nElem_Bound[iMarker]; iElem_Bound++) {
2735 if (bound[iMarker][iElem_Bound]->GetVTK_Type() == TRIANGLE) {
2736 for (iNode = 0; iNode < bound[iMarker][iElem_Bound]->GetnNodes(); iNode++) {
2737 iPoint = bound[iMarker][iElem_Bound]->GetNode(iNode);
2738
2739 for (iNeighbor_Nodes = 0; iNeighbor_Nodes < bound[iMarker][iElem_Bound]->GetnNeighbor_Nodes(iNode);
2740 iNeighbor_Nodes++) {
2741 Neighbor_Node = bound[iMarker][iElem_Bound]->GetNeighbor_Nodes(iNode, iNeighbor_Nodes);
2742 jPoint = bound[iMarker][iElem_Bound]->GetNode(Neighbor_Node);
2743
2744 iEdge = FindEdge(iPoint, jPoint);
2745
2746 if (Check_Edge[iEdge]) {
2747 Check_Edge[iEdge] = false;
2748
2749 if (tan(Angle_Alpha[iEdge]) != 0.0)
2750 cot_alpha = 1.0 / tan(Angle_Alpha[iEdge]);
2751 else
2752 cot_alpha = 0.0;
2753 if (tan(Angle_Beta[iEdge]) != 0.0)
2754 cot_beta = 1.0 / tan(Angle_Beta[iEdge]);
2755 else
2756 cot_beta = 0.0;
2757
2758 /*--- iPoint, and jPoint ---*/
2759 for (iDim = 0; iDim < nDim; iDim++) {
2760 if (Area_Vertex[iPoint] != 0.0)
2761 NormalMeanK[iPoint][iDim] += 3.0 * (cot_alpha + cot_beta) *
2762 (nodes->GetCoord(iPoint, iDim) - nodes->GetCoord(jPoint, iDim)) /
2763 Area_Vertex[iPoint];
2764 if (Area_Vertex[jPoint] != 0.0)
2765 NormalMeanK[jPoint][iDim] += 3.0 * (cot_alpha + cot_beta) *
2766 (nodes->GetCoord(jPoint, iDim) - nodes->GetCoord(iPoint, iDim)) /
2767 Area_Vertex[jPoint];
2768 }
2769 }
2770 }
2771 }
2772 }
2773 }
2774 }
2775 }
2776
2777 /*--- Compute Gauss, mean, max and min principal curvature,
2778 and set the list of critical points ---*/
2779
2780 for (iMarker = 0; iMarker < nMarker; iMarker++) {
2781 if (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) {
2782 for (iVertex = 0; iVertex < nVertex[iMarker]; iVertex++) {
2783 iPoint = vertex[iMarker][iVertex]->GetNode();
2784
2785 if (nodes->GetDomain(iPoint)) {
2786 if (Area_Vertex[iPoint] != 0.0)
2787 GaussK = 3.0 * Angle_Defect[iPoint] / Area_Vertex[iPoint];
2788 else
2789 GaussK = 0.0;
2790
2791 MeanK = 0.0;
2792 for (iDim = 0; iDim < nDim; iDim++) MeanK += NormalMeanK[iPoint][iDim] * NormalMeanK[iPoint][iDim];
2793 MeanK = sqrt(MeanK);
2794
2795 delta = max((MeanK * MeanK - GaussK), 0.0);
2796
2797 MaxPrinK = MeanK + sqrt(delta);
2798
2799 /*--- Store the curvature value ---*/
2800 K[iPoint] = MaxPrinK;
2801 nodes->SetCurvature(iPoint, K[iPoint]);
2802 }
2803 }
2804 }
2805 }
2806
2807 delete[] Angle_Defect;
2808 delete[] Area_Vertex;
2809 delete[] Angle_Alpha;
2810 delete[] Angle_Beta;
2811 delete[] Check_Edge;
2812
2813 for (iPoint = 0; iPoint < nPoint; iPoint++) delete[] NormalMeanK[iPoint];
2814 delete[] NormalMeanK;
2815 }
2816
2817 /*--- Sharp edge detection is based in the statistical
2818 distribution of the curvature ---*/
2819
2820 MaxK = K[0];
2821 MinK = K[0];
2822 MeanK = 0.0;
2823 TotalnPointDomain = 0;
2824 for (iMarker = 0; iMarker < nMarker; iMarker++) {
2825 if (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) {
2826 for (iVertex = 0; iVertex < nVertex[iMarker]; iVertex++) {
2827 iPoint = vertex[iMarker][iVertex]->GetNode();
2828 if (nodes->GetDomain(iPoint)) {
2829 MaxK = max(MaxK, fabs(K[iPoint]));
2830 MinK = min(MinK, fabs(K[iPoint]));
2831 MeanK += fabs(K[iPoint]);
2832 TotalnPointDomain++;
2833 }
2834 }
2835 }
2836 }
2837
2838 su2double MyMeanK = MeanK;
2839 MeanK = 0.0;
2840 su2double MyMaxK = MaxK;
2841 MaxK = 0.0;
2842 unsigned long MynPointDomain = TotalnPointDomain;
2843 TotalnPointDomain = 0;
2844 SU2_MPI::Allreduce(&MyMeanK, &MeanK, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
2845 SU2_MPI::Allreduce(&MyMaxK, &MaxK, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
2846 SU2_MPI::Allreduce(&MynPointDomain, &TotalnPointDomain, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
2847
2848 /*--- Compute the mean ---*/
2849 MeanK /= su2double(TotalnPointDomain);
2850
2851 /*--- Compute the standard deviation ---*/
2852 SigmaK = 0.0;
2853 for (iMarker = 0; iMarker < nMarker; iMarker++) {
2854 if (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) {
2855 for (iVertex = 0; iVertex < nVertex[iMarker]; iVertex++) {
2856 iPoint = vertex[iMarker][iVertex]->GetNode();
2857 if (nodes->GetDomain(iPoint)) {
2858 SigmaK += (fabs(K[iPoint]) - MeanK) * (fabs(K[iPoint]) - MeanK);
2859 }
2860 }
2861 }
2862 }
2863
2864 su2double MySigmaK = SigmaK;
2865 SigmaK = 0.0;
2866 SU2_MPI::Allreduce(&MySigmaK, &SigmaK, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
2867
2868 SigmaK = sqrt(SigmaK / su2double(TotalnPointDomain));
2869
2870 if (rank == MASTER_NODE)
2871 cout << "Max K: " << MaxK << ". Mean K: " << MeanK << ". Standard deviation K: " << SigmaK << "." << endl;
2872
2873 Point_Critical.clear();
2874
2875 for (iMarker = 0; iMarker < nMarker; iMarker++) {
2876 if (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) {
2877 for (iVertex = 0; iVertex < nVertex[iMarker]; iVertex++) {
2878 iPoint = vertex[iMarker][iVertex]->GetNode();
2879 if (nodes->GetDomain(iPoint)) {
2880 if (fabs(K[iPoint]) > MeanK + config->GetRefSharpEdges() * SigmaK) {
2881 Point_Critical.push_back(iPoint);
2882 }
2883 }
2884 }
2885 }
2886 }
2887
2888 /*--- Variables and buffers needed for MPI ---*/
2889
2890 Buffer_Send_nVertex = new unsigned long[1];
2891 Buffer_Receive_nVertex = new unsigned long[size];
2892
2893 /*--- Count the total number of critical edge nodes. ---*/
2894
2895 nLocalVertex = Point_Critical.size();
2896 Buffer_Send_nVertex[0] = nLocalVertex;
2897
2898 /*--- Communicate to all processors the total number of critical edge nodes. ---*/
2899
2900 MaxLocalVertex = 0;
2901 SU2_MPI::Allreduce(&nLocalVertex, &MaxLocalVertex, 1, MPI_UNSIGNED_LONG, MPI_MAX, SU2_MPI::GetComm());
2902 SU2_MPI::Allgather(Buffer_Send_nVertex, 1, MPI_UNSIGNED_LONG, Buffer_Receive_nVertex, 1, MPI_UNSIGNED_LONG,
2903 SU2_MPI::GetComm());
2904
2905 /*--- Create and initialize to zero some buffers to hold the coordinates
2906 of the boundary nodes that are communicated from each partition (all-to-all). ---*/
2907
2908 const unsigned long nBuffer = MaxLocalVertex * nDim;
2909 Buffer_Send_Coord = new su2double[nBuffer]();
2910 Buffer_Receive_Coord = new su2double[size * nBuffer];
2911
2912 /*--- Retrieve and store the coordinates of the sharp edges boundary nodes on
2913 the local partition and broadcast them to all partitions. ---*/
2914
2915 for (iVertex = 0; iVertex < Point_Critical.size(); iVertex++) {
2916 iPoint = Point_Critical[iVertex];
2917 for (iDim = 0; iDim < nDim; iDim++) Buffer_Send_Coord[iVertex * nDim + iDim] = nodes->GetCoord(iPoint, iDim);
2918 }
2919
2920 SU2_MPI::Allgather(Buffer_Send_Coord, nBuffer, MPI_DOUBLE, Buffer_Receive_Coord, nBuffer, MPI_DOUBLE,
2921 SU2_MPI::GetComm());
2922
2923 /*--- Loop over all interior mesh nodes on the local partition and compute
2924 the distances to each of the no-slip boundary nodes in the entire mesh.
2925 Store the minimum distance to the wall for each interior mesh node. ---*/
2926
2927 for (iPoint = 0; iPoint < GetnPoint(); iPoint++) {
2928 Coord = nodes->GetCoord(iPoint);
2929
2930 MinDist = 1E20;
2931 for (int iProcessor = 0; iProcessor < size; iProcessor++) {
2932 for (iVertex = 0; iVertex < Buffer_Receive_nVertex[iProcessor]; iVertex++) {
2933 Dist = 0.0;
2934 for (iDim = 0; iDim < nDim; iDim++) {
2935 Dist += (Coord[iDim] - Buffer_Receive_Coord[(iProcessor * MaxLocalVertex + iVertex) * nDim + iDim]) *
2936 (Coord[iDim] - Buffer_Receive_Coord[(iProcessor * MaxLocalVertex + iVertex) * nDim + iDim]);
2937 }
2938 if (Dist != 0.0)
2939 Dist = sqrt(Dist);
2940 else
2941 Dist = 0.0;
2942 if (Dist < MinDist) MinDist = Dist;
2943 }
2944 }
2945 nodes->SetSharpEdge_Distance(iPoint, MinDist);
2946 }
2947
2948 /*--- Deallocate Max curvature ---*/
2949 delete[] K;
2950
2951 /*--- Deallocate the buffers needed for the MPI communication. ---*/
2952 delete[] Buffer_Send_Coord;
2953 delete[] Buffer_Receive_Coord;
2954 delete[] Buffer_Send_nVertex;
2955 delete[] Buffer_Receive_nVertex;
2956}

2958void CGeometry::FilterValuesAtElementCG(const vector<su2double>& filter_radius,
2959 const vector<pair<ENUM_FILTER_KERNEL, su2double>>& kernels,
2960 const unsigned short search_limit, su2double* values) const {
2961 /*--- Apply a filter to "input_values". The filter is an averaging process over the neighbourhood
2962 of each element, which is a circle in 2D and a sphere in 3D of radius "filter_radius".
2963 The filter is characterized by its kernel, i.e. how the weights are computed. Multiple kernels
2964 can be specified in which case they are applied sequentially (each one being applied to the
2965 output values of the previous filter. ---*/
2966
2967 /*--- Check if we need to do any work. ---*/
2968 if (kernels.empty()) return;
2969
2970 /*--- FIRST: Gather the adjacency matrix, element centroids, volumes, and values on every
2971 processor, this is required because the filter reaches far into adjacent partitions. ---*/
2972
2973 /*--- Adjacency matrix ---*/
2974 vector<unsigned long> neighbour_start;
2975 long* neighbour_idx = nullptr;
2976 GetGlobalElementAdjacencyMatrix(neighbour_start, neighbour_idx);
2977
2978 /*--- Element centroids and volumes. ---*/
2979 auto *cg_elem = new su2double[Global_nElemDomain * nDim], *vol_elem = new su2double[Global_nElemDomain];
2980#ifdef HAVE_MPI
2981 /*--- Number of subdomain each point is part of. ---*/
2982 vector<char> halo_detect(Global_nElemDomain);
2983#endif
2984
2985 /*--- Inputs of a filter stage, like with CG and volumes, each processor needs to see everything. ---*/
2986 auto* work_values = new su2double[Global_nElemDomain];
2987
2988 /*--- When gathering the neighborhood of each element we use a vector of booleans to indicate
2989 whether an element is already added to the list of neighbors (one vector per thread). ---*/
2990 vector<vector<bool>> is_neighbor(omp_get_max_threads());
2991
2992 /*--- Begin OpenMP parallel section, count total number of searches for which
2993 the recursion limit is reached and the full neighborhood is not considered. ---*/
2994 unsigned long limited_searches = 0;
2995
2996 SU2_OMP_PARALLEL_(reduction(+ : limited_searches)) {
2997 /*--- Initialize ---*/
2998 SU2_OMP_FOR_STAT(256)
2999 for (auto iElem = 0ul; iElem < Global_nElemDomain; ++iElem) {
3000 for (unsigned short iDim = 0; iDim < nDim; ++iDim) cg_elem[nDim * iElem + iDim] = 0.0;
3001 vol_elem[iElem] = 0.0;
3002 }
3003 END_SU2_OMP_FOR
3004
3005 /*--- Populate ---*/
3006 SU2_OMP_FOR_STAT(256)
3007 for (auto iElem = 0ul; iElem < nElem; ++iElem) {
3008 auto iElem_global = elem[iElem]->GetGlobalIndex();
3009 for (unsigned short iDim = 0; iDim < nDim; ++iDim) cg_elem[nDim * iElem_global + iDim] = elem[iElem]->GetCG(iDim);
3010 vol_elem[iElem_global] = elem[iElem]->GetVolume();
3011 }
3012 END_SU2_OMP_FOR
3013
3014#ifdef HAVE_MPI
3015 /*--- Account for the duplication introduced by the halo elements and the
3016 reduction using MPI_SUM, which is required to maintain differentiabillity. ---*/
3017 SU2_OMP_FOR_STAT(256)
3018 for (auto iElem = 0ul; iElem < Global_nElemDomain; ++iElem) halo_detect[iElem] = 0;
3019 END_SU2_OMP_FOR
3020
3021 SU2_OMP_FOR_STAT(256)
3022 for (auto iElem = 0ul; iElem < nElem; ++iElem) halo_detect[elem[iElem]->GetGlobalIndex()] = 1;
3023 END_SU2_OMP_FOR
3024
3025 /*--- Share with all processors ---*/
3026 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
3027 auto* dbl_buffer = new su2double[Global_nElemDomain * nDim];
3028 SU2_MPI::Allreduce(cg_elem, dbl_buffer, Global_nElemDomain * nDim, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
3029 swap(dbl_buffer, cg_elem);
3030 delete[] dbl_buffer;
3031
3032 dbl_buffer = new su2double[Global_nElemDomain];
3033 SU2_MPI::Allreduce(vol_elem, dbl_buffer, Global_nElemDomain, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
3034 swap(dbl_buffer, vol_elem);
3035 delete[] dbl_buffer;
3036
3037 vector<char> char_buffer(Global_nElemDomain);
3038 MPI_Allreduce(halo_detect.data(), char_buffer.data(), Global_nElemDomain, MPI_CHAR, MPI_SUM, SU2_MPI::GetComm());
3039 halo_detect.swap(char_buffer);
3040 }
3041 END_SU2_OMP_SAFE_GLOBAL_ACCESS
3042
3043 SU2_OMP_FOR_STAT(256)
3044 for (auto iElem = 0ul; iElem < Global_nElemDomain; ++iElem) {
3045 su2double numRepeat = halo_detect[iElem];
3046 for (unsigned short iDim = 0; iDim < nDim; ++iDim) cg_elem[nDim * iElem + iDim] /= numRepeat;
3047 vol_elem[iElem] /= numRepeat;
3048 }
3049 END_SU2_OMP_FOR
3050#endif
3051
3052 /*--- SECOND: Each processor performs the average for its elements. For each
3053 element we look for neighbours of neighbours of... until the distance to the
3054 closest newly found one is greater than the filter radius. ---*/
3055
3056 is_neighbor[omp_get_thread_num()].resize(Global_nElemDomain, false);
3057
3058 for (unsigned long iKernel = 0; iKernel < kernels.size(); ++iKernel) {
3059 auto kernel_type = kernels[iKernel].first;
3060 su2double kernel_param = kernels[iKernel].second;
3061 su2double kernel_radius = filter_radius[iKernel];
3062
3063 /*--- Synchronize work values ---*/
3064 /*--- Initialize ---*/
3065 SU2_OMP_FOR_STAT(256)
3066 for (auto iElem = 0ul; iElem < Global_nElemDomain; ++iElem) work_values[iElem] = 0.0;
3067 END_SU2_OMP_FOR
3068
3069 /*--- Populate ---*/
3070 SU2_OMP_FOR_STAT(256)
3071 for (auto iElem = 0ul; iElem < nElem; ++iElem) work_values[elem[iElem]->GetGlobalIndex()] = values[iElem];
3072 END_SU2_OMP_FOR
3073
3074#ifdef HAVE_MPI
3075 /*--- Share with all processors ---*/
3076 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
3077 auto* buffer = new su2double[Global_nElemDomain];
3078 SU2_MPI::Allreduce(work_values, buffer, Global_nElemDomain, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
3079 swap(buffer, work_values);
3080 delete[] buffer;
3081 }
3082 END_SU2_OMP_SAFE_GLOBAL_ACCESS
3083
3084 /*--- Account for duplication ---*/
3085 SU2_OMP_FOR_STAT(256)
3086 for (auto iElem = 0ul; iElem < Global_nElemDomain; ++iElem) {
3087 su2double numRepeat = halo_detect[iElem];
3088 work_values[iElem] /= numRepeat;
3089 }
3090 END_SU2_OMP_FOR
3091#endif
3092
3093 /*--- Filter ---*/
3094 SU2_OMP_FOR_DYN(128)
3095 for (auto iElem = 0ul; iElem < nElem; ++iElem) {
3096 int thread = omp_get_thread_num();
3097
3098 /*--- Center of the search ---*/
3099 auto iElem_global = elem[iElem]->GetGlobalIndex();
3100
3101 /*--- Find the neighbours of iElem ---*/
3102 vector<long> neighbours;
3103 limited_searches +=
3104 !GetRadialNeighbourhood(iElem_global, SU2_TYPE::GetValue(kernel_radius), search_limit, neighbour_start,
3105 neighbour_idx, cg_elem, neighbours, is_neighbor[thread]);
3106 /*--- Apply the kernel ---*/
3107 su2double weight = 0.0, numerator = 0.0, denominator = 0.0;
3108
3109 switch (kernel_type) {
3110 /*--- distance-based kernels (weighted averages) ---*/
3111 case ENUM_FILTER_KERNEL::CONSTANT_WEIGHT:
3112 case ENUM_FILTER_KERNEL::CONICAL_WEIGHT:
3113 case ENUM_FILTER_KERNEL::GAUSSIAN_WEIGHT:
3114
3115 for (auto idx : neighbours) {
3116 su2double distance = 0.0;
3117 for (unsigned short iDim = 0; iDim < nDim; ++iDim)
3118 distance += pow(cg_elem[nDim * iElem_global + iDim] - cg_elem[nDim * idx + iDim], 2);
3119 distance = sqrt(distance);
3120
3121 switch (kernel_type) {
3122 case ENUM_FILTER_KERNEL::CONSTANT_WEIGHT:
3123 weight = 1.0;
3124 break;
3125 case ENUM_FILTER_KERNEL::CONICAL_WEIGHT:
3126 weight = kernel_radius - distance;
3127 break;
3128 case ENUM_FILTER_KERNEL::GAUSSIAN_WEIGHT:
3129 weight = exp(-0.5 * pow(distance / kernel_param, 2));
3130 break;
3131 default:
3132 break;
3133 }
3134 weight *= vol_elem[idx];
3135 numerator += weight * work_values[idx];
3136 denominator += weight;
3137 }
3138 values[iElem] = numerator / denominator;
3139 break;
3140
3141 /*--- morphology kernels (image processing) ---*/
3142 case ENUM_FILTER_KERNEL::DILATE_MORPH:
3143 case ENUM_FILTER_KERNEL::ERODE_MORPH:
3144
3145 for (auto idx : neighbours) {
3146 switch (kernel_type) {
3147 case ENUM_FILTER_KERNEL::DILATE_MORPH:
3148 numerator += exp(kernel_param * work_values[idx]);
3149 break;
3150 case ENUM_FILTER_KERNEL::ERODE_MORPH:
3151 numerator += exp(kernel_param * (1.0 - work_values[idx]));
3152 break;
3153 default:
3154 break;
3155 }
3156 denominator += 1.0;
3157 }
3158 values[iElem] = log(numerator / denominator) / kernel_param;
3159 if (kernel_type == ENUM_FILTER_KERNEL::ERODE_MORPH) values[iElem] = 1.0 - values[iElem];
3160 break;
3161
3162 default:
3163 SU2_MPI::Error("Unknown type of filter kernel", CURRENT_FUNCTION);
3164 }
3165 }
3166 END_SU2_OMP_FOR
3167 }
3168 }
3169 END_SU2_OMP_PARALLEL
3170
3171 limited_searches /= kernels.size();
3172
3173 unsigned long tmp = limited_searches;
3174 SU2_MPI::Reduce(&tmp, &limited_searches, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE, SU2_MPI::GetComm());
3175
3176 if (rank == MASTER_NODE && limited_searches > 0)
3177 cout << "Warning: The filter radius was limited for " << limited_searches << " elements ("
3178 << limited_searches / (0.01 * Global_nElemDomain) << "%).\n";
3179
3180 delete[] neighbour_idx;
3181 delete[] cg_elem;
3182 delete[] vol_elem;
3183 delete[] work_values;
3184}

3186void CGeometry::GetGlobalElementAdjacencyMatrix(vector<unsigned long>& neighbour_start, long*& neighbour_idx) const {
3187 if (neighbour_idx != nullptr)
3188 SU2_MPI::Error("neighbour_idx is expected to be NULL, stopping to avoid a potential memory leak", CURRENT_FUNCTION);
3189
3190 /*--- Determine how much space we need for the adjacency matrix by counting the
3191 neighbours of each element, i.e. its number of faces---*/
3192 auto* nFaces_elem = new unsigned short[Global_nElemDomain];
3193
3194 SU2_OMP_PARALLEL {
3195 SU2_OMP_FOR_STAT(256)
3196 for (auto iElem = 0ul; iElem < Global_nElemDomain; ++iElem) nFaces_elem[iElem] = 0;
3197 END_SU2_OMP_FOR
3198
3199 SU2_OMP_FOR_STAT(256)
3200 for (auto iElem = 0ul; iElem < nElem; ++iElem) {
3201 auto iElem_global = elem[iElem]->GetGlobalIndex();
3202 nFaces_elem[iElem_global] = elem[iElem]->GetnFaces();
3203 }
3204 END_SU2_OMP_FOR
3205 }
3206 END_SU2_OMP_PARALLEL
3207#ifdef HAVE_MPI
3208 /*--- Share with all processors ---*/
3209 {
3210 auto* buffer = new unsigned short[Global_nElemDomain];
3211 MPI_Allreduce(nFaces_elem, buffer, Global_nElemDomain, MPI_UNSIGNED_SHORT, MPI_MAX, SU2_MPI::GetComm());
3212 /*--- swap pointers and delete old data to keep the same variable name after reduction ---*/
3213 swap(buffer, nFaces_elem);
3214 delete[] buffer;
3215 }
3216#endif
3217
3218 /*--- Vector with the addresses of the start of the neighbours of a given element.
3219 This is generated by a cumulative sum of the neighbour count. ---*/
3220 neighbour_start.resize(Global_nElemDomain + 1);
3221
3222 neighbour_start[0] = 0;
3223 for (auto iElem = 0ul; iElem < Global_nElemDomain; ++iElem) {
3224 neighbour_start[iElem + 1] = neighbour_start[iElem] + nFaces_elem[iElem];
3225 }
3226 delete[] nFaces_elem;
3227
3228 /*--- Allocate ---*/
3229 unsigned long matrix_size = neighbour_start[Global_nElemDomain];
3230 neighbour_idx = new long[matrix_size];
3231
3232 SU2_OMP_PARALLEL {
3233 /*--- Initialize ---*/
3234 SU2_OMP_FOR_STAT(256)
3235 for (auto iElem = 0ul; iElem < matrix_size; ++iElem) neighbour_idx[iElem] = -1;
3236 END_SU2_OMP_FOR
3237
3238 /*--- Populate ---*/
3239 SU2_OMP_FOR_STAT(128)
3240 for (auto iElem = 0ul; iElem < nElem; ++iElem) {
3241 auto iElem_global = elem[iElem]->GetGlobalIndex();
3242 auto start_pos = neighbour_start[iElem_global];
3243
3244 for (unsigned short iFace = 0; iFace < elem[iElem]->GetnFaces(); ++iFace) {
3245 long neighbour = elem[iElem]->GetNeighbor_Elements(iFace);
3246
3247 if (neighbour >= 0) {
3248 neighbour_idx[start_pos + iFace] = elem[neighbour]->GetGlobalIndex();
3249 }
3250 }
3251 }
3252 END_SU2_OMP_FOR
3253 }
3254 END_SU2_OMP_PARALLEL
3255#ifdef HAVE_MPI
3256 /*--- Share with all processors ---*/
3257 {
3258 long* buffer = new long[matrix_size];
3259 MPI_Allreduce(neighbour_idx, buffer, matrix_size, MPI_LONG, MPI_MAX, SU2_MPI::GetComm());
3260 swap(buffer, neighbour_idx);
3261 delete[] buffer;
3262 }

3266bool CGeometry::GetRadialNeighbourhood(const unsigned long iElem_global, const passivedouble radius,
3267 size_t search_limit, const vector<unsigned long>& neighbour_start,
3268 const long* neighbour_idx, const su2double* cg_elem, vector<long>& neighbours,
3269 vector<bool>& is_neighbor) const {
3270 /*--- Validate inputs if we are debugging. ---*/
3271 assert(neighbour_start.size() == Global_nElemDomain + 1 && neighbour_idx != nullptr && cg_elem != nullptr &&
3272 is_neighbor.size() == Global_nElemDomain && "invalid inputs");
3273
3274 /*--- 0 search_limit means "unlimited" (it will probably
3275 stop once it gathers the entire domain, probably). ---*/
3276 if (!search_limit) search_limit = numeric_limits<size_t>::max();
3277
3278 /*--- Center of the search ---*/
3279 neighbours.clear();
3280 neighbours.push_back(iElem_global);
3281 is_neighbor[iElem_global] = true;
3282
3283 passivedouble X0[3] = {0.0, 0.0, 0.0};
3284 for (unsigned short iDim = 0; iDim < nDim; ++iDim) X0[iDim] = SU2_TYPE::GetValue(cg_elem[nDim * iElem_global + iDim]);
3285
3286 /*--- Loop stops when "neighbours" stops changing size, or degree reaches limit. ---*/
3287 bool finished = false;
3288 for (size_t degree = 0, start = 0; degree < search_limit && !finished; ++degree) {
3289 /*--- For each element of the last degree added consider its immediate
3290 neighbours, that are not already neighbours, as candidates. ---*/
3291 vector<long> candidates;
3292
3293 for (auto it = neighbours.begin() + start; it != neighbours.end(); ++it) {
3294 /*--- scan row of the adjacency matrix of element *it ---*/
3295 for (auto i = neighbour_start[*it]; i < neighbour_start[(*it) + 1]; ++i) {
3296 auto idx = neighbour_idx[i];
3297 if (idx >= 0)
3298 if (!is_neighbor[idx]) {
3299 candidates.push_back(idx);
3300 /*--- mark as neighbour for now to avoid duplicate candidates. ---*/
3301 is_neighbor[idx] = true;
3302 }
3303 }
3304 }
3305 /*--- update start position to fetch next degree candidates. ---*/
3306 start = neighbours.size();
3307
3308 /*--- Add candidates within "radius" of X0, if none qualifies we are "finished". ---*/
3309 finished = true;
3310 for (auto idx : candidates) {
3311 /*--- passivedouble as we only need to compare "distance". ---*/
3312 passivedouble distance = 0.0;
3313 for (unsigned short iDim = 0; iDim < nDim; ++iDim)
3314 distance += pow(X0[iDim] - SU2_TYPE::GetValue(cg_elem[nDim * idx + iDim]), 2);
3315
3316 if (distance < pow(radius, 2)) {
3317 neighbours.push_back(idx);
3318 finished = false;
3319 }
3320 /*--- not a neighbour in the end. ---*/
3321 else
3322 is_neighbor[idx] = false;
3323 }
3324 }
3325 /*--- Restore the state of the working vector for next call. ---*/
3326 for (auto idx : neighbours) is_neighbor[idx] = false;
3327
3328 return finished;
3329}

3618 /*--- When not using threading use the natural coloring to reduce overhead. ---*/

3623 }
3624

3644 auto nextEdgeColorGroupSize = edgeColorGroupSize; /* next value that we are going to try */

3903

3906 for (auto iLine = 0ul; iLine < nLinelet; ++iLine) {

3908 for (const auto iPoint : li.linelets[iLine]) {
3909 neighbors.insert(iPoint);
3910 for (const auto jPoint : nodes->GetPoints(iPoint)) {
3911 neighbors.insert(jPoint);
3912 }
3913 }
3914 adjacency[iLine].reserve(neighbors.size());
3915 for (const auto iPoint : neighbors) {
3916 adjacency[iLine].push_back(iPoint);
3917 }
3918 }

3922 const auto nColors = coloring.getOuterSize();

3926 sortedLinelets.reserve(nLinelet);
3927 li.colorOffsets.reserve(nColors + 1);
3928 li.colorOffsets.push_back(0);
3929
3930 for (auto iColor = 0ul; iColor < nColors; ++iColor) {
3931 for (const auto iLine : coloring.getInnerIter(iColor)) {
3932 /*--- Store the new linelet index for its points. ---*/
3933 for (const auto iPoint : li.linelets[iLine]) {
3934 li.lineletIdx[iPoint] = sortedLinelets.size();
3935 }
3936 sortedLinelets.push_back(std::move(li.linelets[iLine]));
3937 }
3938 li.colorOffsets.push_back(sortedLinelets.size());
3939 }
3940 li.linelets = std::move(sortedLinelets);
3941
3942 /*--- For visualization, offset colors to avoid coloring across ranks. ---*/
3943 std::vector<unsigned long> allNColors(size);
3944 SU2_MPI::Allgather(&nColors, 1, MPI_UNSIGNED_LONG, allNColors.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
3945 unsigned long offset = 0;
3946 for (int i = 0; i < rank; ++i) offset += allNColors[i];
3947
3948 /*--- Finally, transfer colors to points, using "0" as "no linelet". ---*/
3949 li.lineletColor.resize(nPoint, 0);
3950 for (auto iColor = 0ul; iColor < nColors; ++iColor) {
3951 for (auto iLine = li.colorOffsets[iColor]; iLine < li.colorOffsets[iColor + 1]; ++iLine) {
3952 for (const auto iPoint : li.linelets[iLine]) {
3953 li.lineletColor[iPoint] = 1 + offset + iColor;
3954 }
3955 }
3956 }

3958 return li;
