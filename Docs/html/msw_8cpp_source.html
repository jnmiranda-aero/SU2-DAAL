109CNumerics::ResidualType<> CUpwMSW_NEMO::ComputeResidual(const CConfig *config) {
110
111 /*--- Set parameters in the numerical method ---*/
112 const su2double alpha = 5.0;
113 const su2double epsilon = 0.0;
114
115 /*--- Calculate supporting geometry parameters ---*/
116 Area = GeometryToolbox::Norm(nDim, Normal);
117
118 for (auto iDim = 0ul; iDim < nDim; iDim++)
119 UnitNormal[iDim] = Normal[iDim]/Area;
120
121 /*--- Initialize flux & Jacobian vectors ---*/
122 for (auto iVar = 0ul; iVar < nVar; iVar++) {
123 Fc_i[iVar] = 0.0;
124 Fc_j[iVar] = 0.0;
125 }
126 if (implicit) {
127 for (auto iVar = 0ul; iVar < nVar; iVar++) {
128 for (auto jVar = 0ul; jVar < nVar; jVar++) {
129 Jacobian_i[iVar][jVar] = 0.0;
130 Jacobian_j[iVar][jVar] = 0.0;
131 }
132 }
133 }
134
135 /*--- Load variables from nodes i & j ---*/
136 for (auto iSpecies = 0ul; iSpecies < nSpecies; iSpecies++) {
137 rhos_i[iSpecies] = V_i[RHOS_INDEX+iSpecies];
138 rhos_j[iSpecies] = V_j[RHOS_INDEX+iSpecies];
139 }
140 for (auto iDim = 0ul; iDim < nDim; iDim++) {
141 Velocity_i[iDim] = V_i[VEL_INDEX+iDim];
142 Velocity_j[iDim] = V_j[VEL_INDEX+iDim];
143 }
144 Pressure_i = V_i[P_INDEX];
145 Pressure_j = V_j[P_INDEX];
146
147 /*--- Calculate velocity quantities ---*/
148 ProjVelocity_i = GeometryToolbox::DotProduct(nDim, Velocity_i, UnitNormal);
149 ProjVelocity_j = GeometryToolbox::DotProduct(nDim, Velocity_j, UnitNormal);
150
151 /*--- Calculate the state weighting function ---*/
152 const su2double dp = fabs(Pressure_j-Pressure_i) / min(Pressure_j, Pressure_i);
153 const su2double w = 0.5 * (1.0/(pow(alpha*dp,2.0) +1.0));
154 const su2double onemw = 1.0 - w;
155
156 /*--- Calculate weighted state vector (*) for i & j ---*/
157 for (auto iVar = 0ul; iVar < nVar; iVar++) {
158 Ust_i[iVar] = onemw*U_i[iVar] + w*U_j[iVar];
159 Ust_j[iVar] = onemw*U_j[iVar] + w*U_i[iVar];
160 }
161 for (auto iVar = 0ul; iVar < nPrimVar; iVar++) {
162 Vst_i[iVar] = onemw*V_i[iVar] + w*V_j[iVar];
163 Vst_j[iVar] = onemw*V_j[iVar] + w*V_i[iVar];
164 }
165 const su2double ProjVelst_i = onemw*ProjVelocity_i + w*ProjVelocity_j;
166 const su2double ProjVelst_j = onemw*ProjVelocity_j + w*ProjVelocity_i;
167
168 const auto& eves_st_i = fluidmodel->ComputeSpeciesEve(Vst_i[TVE_INDEX]);
169 const auto& eves_st_j = fluidmodel->ComputeSpeciesEve(Vst_j[TVE_INDEX]);
170
171 fluidmodel->ComputedPdU(Vst_i, eves_st_i, dPdUst_i);
172 fluidmodel->ComputedPdU(Vst_j, eves_st_j, dPdUst_j);
173
174 /*--- Flow eigenvalues at i (Lambda+) ---*/
175 for (auto iVar = 0; iVar < nSpecies+nDim-1; iVar++)
176 Lambda_i[iVar] = 0.5*(ProjVelst_i + sqrt(ProjVelst_i*ProjVelst_i +
177 epsilon*epsilon));
178 Lambda_i[nSpecies+nDim-1] = 0.5*(ProjVelst_i + Vst_i[A_INDEX] +
179 sqrt((ProjVelst_i + Vst_i[A_INDEX])*
180 (ProjVelst_i + Vst_i[A_INDEX])+
181 epsilon*epsilon));
182 Lambda_i[nSpecies+nDim] = 0.5*(ProjVelst_i - Vst_i[A_INDEX] +
183 sqrt((ProjVelst_i - Vst_i[A_INDEX])*
184 (ProjVelst_i - Vst_i[A_INDEX])+
185 epsilon*epsilon));
186 Lambda_i[nSpecies+nDim+1] = 0.5*(ProjVelst_i + sqrt(ProjVelst_i*ProjVelst_i +
187 epsilon*epsilon));
188
189 /*--- Compute projected P, invP, and Lambda ---*/
190 su2double l[MAXNDIM], m[MAXNDIM];
191 CreateBasis(UnitNormal,l,m);
192 GetPMatrix(Ust_i, Vst_i, dPdUst_i, UnitNormal, l, m, P_Tensor);
193 GetPMatrix_inv(Ust_i, Vst_i, dPdUst_i, UnitNormal, l, m, invP_Tensor);
194
195 /*--- Projected flux (f+) at i ---*/
196 su2double Proj_ModJac_Tensor_i;
197 for (auto iVar = 0ul; iVar < nVar; iVar++) {
198 for (auto jVar = 0ul; jVar < nVar; jVar++) {
199 Proj_ModJac_Tensor_i = 0.0;
200
201 /*--- Compute Proj_ModJac_Tensor = P x Lambda+ x inverse P ---*/
202 for (auto kVar = 0ul; kVar < nVar; kVar++)
203 Proj_ModJac_Tensor_i += P_Tensor[iVar][kVar]*Lambda_i[kVar]*invP_Tensor[kVar][jVar];
204 Fc_i[iVar] += Proj_ModJac_Tensor_i*U_i[jVar]*Area;
205 if (implicit)
206 Jacobian_i[iVar][jVar] += Proj_ModJac_Tensor_i*Area;
207 }
208 }
209
210 /*--- Flow eigenvalues at j (Lambda-) ---*/
211 for (auto iVar = 0; iVar < nSpecies+nDim-1; iVar++)
212 Lambda_j[iVar] = 0.5*(ProjVelst_j - sqrt(ProjVelst_j*ProjVelst_j +
213 epsilon*epsilon));
214 Lambda_j[nSpecies+nDim-1] = 0.5*(ProjVelst_j + Vst_j[A_INDEX] -
215 sqrt((ProjVelst_j + Vst_j[A_INDEX])*
216 (ProjVelst_j + Vst_j[A_INDEX])+
217 epsilon*epsilon));
218 Lambda_j[nSpecies+nDim] = 0.5*(ProjVelst_j - Vst_j[A_INDEX] -
219 sqrt((ProjVelst_j - Vst_j[A_INDEX])*
220 (ProjVelst_j - Vst_j[A_INDEX])+
221 epsilon*epsilon) );
222 Lambda_j[nSpecies+nDim+1] = 0.5*(ProjVelst_j - sqrt(ProjVelst_j*ProjVelst_j+
223 epsilon*epsilon));
224
225 /*--- Compute projected P, invP, and Lambda ---*/
226 CreateBasis(UnitNormal,l,m);
227 GetPMatrix(Ust_j, Vst_j, dPdUst_j, UnitNormal, l, m, P_Tensor);
228 GetPMatrix_inv(Ust_j, Vst_j, dPdUst_j, UnitNormal, l, m, invP_Tensor);
229
230 /*--- Projected flux (f-) ---*/
231 su2double Proj_ModJac_Tensor_j;
232 for (auto iVar = 0ul; iVar < nVar; iVar++) {
233 for (auto jVar = 0ul; jVar < nVar; jVar++) {
234 Proj_ModJac_Tensor_j = 0.0;
235
236 /*--- Compute Proj_ModJac_Tensor = P x Lambda- x inverse P ---*/
237 for (auto kVar = 0ul; kVar < nVar; kVar++)
238 Proj_ModJac_Tensor_j += P_Tensor[iVar][kVar]*Lambda_j[kVar]*invP_Tensor[kVar][jVar];
239 Fc_j[iVar] += Proj_ModJac_Tensor_j*U_j[jVar]*Area;
240 if (implicit)
241 Jacobian_j[iVar][jVar] += Proj_ModJac_Tensor_j*Area;
242 }
243 }
244
245 /*--- Flux splitting ---*/
246 for (auto iVar = 0ul; iVar < nVar; iVar++) {
247 Flux[iVar] = Fc_i[iVar]+Fc_j[iVar];
248 }
249
250 return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
251}
