30CPYRAM5::CPYRAM5() : CElementWithKnownSizes<NGAUSS, NNODE, NDIM>() {
31 /*--- Gauss coordinates and weights ---*/
32
33 GaussCoord[0][0] = 0.5;
34 GaussCoord[0][1] = 0.0;
35 GaussCoord[0][2] = 0.1531754163448146;
36 GaussWeight(0) = 2.0 / 15.0;
37 GaussCoord[1][0] = 0.0;
38 GaussCoord[1][1] = 0.5;
39 GaussCoord[1][2] = 0.1531754163448146;
40 GaussWeight(1) = 2.0 / 15.0;
41 GaussCoord[2][0] = -0.5;
42 GaussCoord[2][1] = 0.0;
43 GaussCoord[2][2] = 0.1531754163448146;
44 GaussWeight(2) = 2.0 / 15.0;
45 GaussCoord[3][0] = 0.0;
46 GaussCoord[3][1] = -0.5;
47 GaussCoord[3][2] = 0.1531754163448146;
48 GaussWeight(3) = 2.0 / 15.0;
49 GaussCoord[4][0] = 0.0;
50 GaussCoord[4][1] = 0.0;
51 GaussCoord[4][2] = 0.6372983346207416;
52 GaussWeight(4) = 2.0 / 15.0;
53
54 /*--- Store the values of the shape functions and their derivatives ---*/
55
56 unsigned short iNode, iGauss;
57 su2double Xi, Eta, Zeta, val_Ni;
58
59 for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
60 Xi = GaussCoord[iGauss][0];
61 Eta = GaussCoord[iGauss][1];
62 Zeta = GaussCoord[iGauss][2];
63
64 val_Ni = 0.25 * (-Xi + Eta + Zeta - 1.0) * (-Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
65 GaussPoint[iGauss].SetNi(val_Ni, 0);
66 val_Ni = 0.25 * (-Xi - Eta + Zeta - 1.0) * (Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
67 GaussPoint[iGauss].SetNi(val_Ni, 1);
68 val_Ni = 0.25 * (Xi + Eta + Zeta - 1.0) * (Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
69 GaussPoint[iGauss].SetNi(val_Ni, 2);
70 val_Ni = 0.25 * (Xi + Eta + Zeta - 1.0) * (-Xi + Eta + Zeta - 1.0) / (1.0 - Zeta);
71 GaussPoint[iGauss].SetNi(val_Ni, 3);
72 val_Ni = Zeta;
73 GaussPoint[iGauss].SetNi(val_Ni, 4);
74
75 /*--- dN/d xi ---*/
76
77 dNiXj[iGauss][0][0] = 0.5 * (Zeta - Xi - 1.0) / (Zeta - 1.0);
78 dNiXj[iGauss][1][0] = 0.5 * Xi / (Zeta - 1.0);
79 dNiXj[iGauss][2][0] = 0.5 * (1.0 - Zeta - Xi) / (Zeta - 1.0);
80 dNiXj[iGauss][3][0] = dNiXj[iGauss][1][0];
81 dNiXj[iGauss][4][0] = 0.0;
82
83 /*--- dN/d eta ---*/
84
85 dNiXj[iGauss][0][1] = 0.5 * Eta / (Zeta - 1.0);
86 dNiXj[iGauss][1][1] = 0.5 * (Zeta - Eta - 1.0) / (Zeta - 1.0);
87 dNiXj[iGauss][2][1] = dNiXj[iGauss][0][1];
88 dNiXj[iGauss][3][1] = 0.5 * (1.0 - Zeta - Eta) / (Zeta - 1.0);
89 dNiXj[iGauss][4][1] = 0.0;
90
91 /*--- dN/d zeta ---*/
92
93 dNiXj[iGauss][0][2] =
94 0.25 * (-1.0 + 2.0 * Zeta - Zeta * Zeta - Eta * Eta + Xi * Xi) / ((1.0 - Zeta) * (1.0 - Zeta));
95 dNiXj[iGauss][1][2] =
96 0.25 * (-1.0 + 2.0 * Zeta - Zeta * Zeta + Eta * Eta - Xi * Xi) / ((1.0 - Zeta) * (1.0 - Zeta));
97 dNiXj[iGauss][2][2] = dNiXj[iGauss][0][2];
98 dNiXj[iGauss][3][2] = dNiXj[iGauss][1][2];
99 dNiXj[iGauss][4][2] = 1.0;
100 }
101
102 /*--- Store the extrapolation functions ---*/
103
104 su2double ExtrapCoord[5][3];
105
106 ExtrapCoord[0][0] = 2.0;
107 ExtrapCoord[0][1] = 0.0;
108 ExtrapCoord[0][2] = -0.316397779494322;
109 ExtrapCoord[1][0] = 0.0;
110 ExtrapCoord[1][1] = 2.0;
111 ExtrapCoord[1][2] = -0.316397779494322;
112 ExtrapCoord[2][0] = -2.0;
113 ExtrapCoord[2][1] = 0.0;
114 ExtrapCoord[2][2] = -0.316397779494322;
115 ExtrapCoord[3][0] = 0.0;
116 ExtrapCoord[3][1] = -2.0;
117 ExtrapCoord[3][2] = -0.316397779494322;
118 ExtrapCoord[4][0] = 0.0;
119 ExtrapCoord[4][1] = 0.0;
120 ExtrapCoord[4][2] = 1.749193338482970;
121
122 for (iNode = 0; iNode < NNODE; iNode++) {
123 Xi = ExtrapCoord[iNode][0];
124 Eta = ExtrapCoord[iNode][1];
125 Zeta = ExtrapCoord[iNode][2];
126
127 NodalExtrap[iNode][0] = 0.25 * (-Xi + Eta + Zeta - 1.0) * (-Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
128 NodalExtrap[iNode][1] = 0.25 * (-Xi - Eta + Zeta - 1.0) * (Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
129 NodalExtrap[iNode][2] = 0.25 * (Xi + Eta + Zeta - 1.0) * (Xi - Eta + Zeta - 1.0) / (1.0 - Zeta);
130 NodalExtrap[iNode][3] = 0.25 * (Xi + Eta + Zeta - 1.0) * (-Xi + Eta + Zeta - 1.0) / (1.0 - Zeta);
131 NodalExtrap[iNode][4] = Zeta;
132 }
133}

135su2double CPYRAM5::ComputeVolume(const FrameType mode) const {
136 unsigned short iDim;
137 su2double r1[3] = {0.0, 0.0, 0.0}, r2[3] = {0.0, 0.0, 0.0}, r3[3] = {0.0, 0.0, 0.0},
138 CrossProduct[3] = {0.0, 0.0, 0.0};
139 su2double Volume = 0.0;
140
141 /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
142 for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed)---*/
143 const su2activematrix& Coord = (mode == REFERENCE) ? RefCoord : CurrentCoord;
144
145 for (iDim = 0; iDim < NDIM; iDim++) {
146 r1[iDim] = Coord[1][iDim] - Coord[0][iDim];
147 r2[iDim] = Coord[2][iDim] - Coord[0][iDim];
148 r3[iDim] = Coord[4][iDim] - Coord[0][iDim];
149 }
150
151 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
152 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
153 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
154
155 Volume = fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
156
157 for (iDim = 0; iDim < NDIM; iDim++) {
158 r1[iDim] = Coord[2][iDim] - Coord[0][iDim];
159 r2[iDim] = Coord[3][iDim] - Coord[0][iDim];
160 r3[iDim] = Coord[4][iDim] - Coord[0][iDim];
161 }
162
163 CrossProduct[0] = (r1[1] * r2[2] - r1[2] * r2[1]) * r3[0];
164 CrossProduct[1] = (r1[2] * r2[0] - r1[0] * r2[2]) * r3[1];
165 CrossProduct[2] = (r1[0] * r2[1] - r1[1] * r2[0]) * r3[2];
166
167 Volume += fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2]) / 6.0;
168
169 return Volume;
170}
