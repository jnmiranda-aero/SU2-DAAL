37 class CSutherland final : public CViscosityModel {
38  public:
39  /*!
40  * \brief Constructor of the class.
41  */
42  CSutherland(su2double mu_ref, su2double t_ref, su2double s) : mu_ref_(mu_ref), t_ref_(t_ref), s_(s) {}
43 
44  /*!
45  * \brief Set Viscosity.
46  */
47  void SetViscosity(su2double t, su2double rho) override {
48  const su2double t_ref_inv = 1.0 / t_ref_;
49  const su2double t_nondim = t_ref_inv * t;
50  mu_ = mu_ref_ * t_nondim * sqrt(t_nondim) * ((t_ref_ + s_) / (t + s_));
51 
52  /*--- Set Viscosity Derivatives. ---*/
53  const su2double ts_inv = 1.0 / (t + s_);
54  dmudrho_t_ = 0.0;
55  dmudt_rho_ = mu_ref_ * (t_ref_ + s_) * ts_inv * sqrt(t_nondim) * (1.5 * t_ref_inv - t_nondim * ts_inv);
56  }
57 
58  private:
59  const su2double mu_ref_{0.0}; /*!< \brief Internal Energy. */
60  const su2double t_ref_{0.0}; /*!< \brief DpDd_e. */
61  const su2double s_{0.0}; /*!< \brief DpDe_d. */
62 };
