67template<size_t nDim, class ForwardIterator, class T>
68FORCEINLINE Double dot(ForwardIterator iterator, const T* ptr) {
69 Double sum = 0.0;
70 for (size_t iDim = 0; iDim < nDim; ++iDim) {
71 sum += *(iterator++) * ptr[iDim];
72 }
73 return sum;
74}

112/*!
113 * \brief Tangential projection.
114 */
115template<size_t nDim>
116FORCEINLINE VectorDbl<nDim> tangentProjection(const MatrixDbl<nDim>& tensor,
117 const VectorDbl<nDim>& unitVector) {
118 VectorDbl<nDim> proj;
119 for (size_t iDim = 0; iDim < nDim; ++iDim)
120 proj(iDim) = dot(tensor[iDim], unitVector);
121
122 Double normalProj = dot(proj, unitVector);
123
124 for (size_t iDim = 0; iDim < nDim; ++iDim)
125 proj(iDim) -= normalProj * unitVector(iDim);
126
127 return proj;
128}

171template<class Container>
172FORCEINLINE Double gatherVariables(Int iPoint, const Container& vars) {
173 Double x;
174 for (size_t k=0; k<Double::Size; ++k) {
175 AD::SetPreaccIn(get(vars, iPoint[k]));
176 x[k] = get(vars, iPoint[k]);
177 }
178 return x;
179}

181template<size_t nVar, class Container>
182FORCEINLINE VectorDbl<nVar> gatherVariables(Int iPoint, const Container& vars) {
183 VectorDbl<nVar> x;
184 for (size_t i=0; i<nVar; ++i) {
185 for (size_t k=0; k<Double::Size; ++k) {

217/*!
218 * \brief Distance vector, from point i to point j.
219 */
220template<size_t nDim, class Container>
221FORCEINLINE VectorDbl<nDim> distanceVector(Int iPoint, Int jPoint,
222 const Container& coords) {
223 auto coord_i = gatherVariables<nDim>(iPoint, coords);
224 auto coord_j = gatherVariables<nDim>(jPoint, coords);
225 VectorDbl<nDim> vector_ij;
226 for (size_t iDim = 0; iDim < nDim; ++iDim) {
227 vector_ij(iDim) = coord_j(iDim) - coord_i(iDim);
228 }
229 return vector_ij;
230}

232/*!
233 * \brief Update the matrix and right-hand-side of a linear system.
234 */
235template<size_t nVar>
236FORCEINLINE void updateLinearSystem(Int iEdge,
237 Int iPoint,
238 Int jPoint,
239 bool implicit,
240 UpdateType updateType,
241 Double updateMask,
242 const VectorDbl<nVar>& flux,
243 const MatrixDbl<nVar>& jac_i,
244 const MatrixDbl<nVar>& jac_j,
245 CSysVector<su2double>& vector,
246 SparseMatrixType& matrix) {
247 if (updateType == UpdateType::COLORING) {
248 vector.UpdateBlocks(iPoint, jPoint, flux, updateMask);
249 if(implicit) {
250 auto wasActive = AD::BeginPassive();
251 matrix.UpdateBlocks(iEdge, iPoint, jPoint, jac_i, jac_j, updateMask);
252 AD::EndPassive(wasActive);
253 }
254 }
255 else {
256 vector.SetBlock(iEdge, flux, updateMask);
257 if(implicit) {
258 auto wasActive = AD::BeginPassive();
259 matrix.SetBlocks(iEdge, jac_i, jac_j, updateMask);
260 AD::EndPassive(wasActive);
261 }
262 }
