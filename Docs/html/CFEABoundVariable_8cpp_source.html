38 void CFEABoundVariable::AllocateBoundaryVariables(CConfig *config) {
39 
40  if (VertexMap.GetIsValid()) return; // nothing to do
41 
42  /*--- Count number of vertices and build map ---*/
43 
44  unsigned long nBoundPt = VertexMap.Build();
45 
46  /*--- Allocate ---*/
47 
48  fsi_analysis = config->GetFSI_Simulation();
49 
50  /*--- Surface residual ---*/
51  Residual_Ext_Surf.resize(nBoundPt,nVar) = su2double(0.0);
52 
53  /*--- Flow traction ---*/
54  if (fsi_analysis) FlowTraction.resize(nBoundPt,nVar) = su2double(0.0);
55 
56  /*--- Generalized alpha integration method requires storing the old residuals ---*/
57  if (config->GetKind_TimeIntScheme_FEA() == STRUCT_TIME_INT::GENERALIZED_ALPHA) {
58  Residual_Ext_Surf_n.resize(nBoundPt,nVar) = su2double(0.0);
59 
60  if (fsi_analysis) FlowTraction_n.resize(nBoundPt,nVar) = su2double(0.0);
61  }
62 }
