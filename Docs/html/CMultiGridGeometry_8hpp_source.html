69 /*!
70 * \brief Set a representative wall value of the agglomerated control volumes on a particular boundary marker.
71 * \param[in] fine_grid - Geometrical definition of the problem.
72 * \param[in] val_marker - Index of the boundary marker.
73 * \param[in] wall_quantity - Object with methods Get(iVertex_fine) and Set(iVertex_coarse, val).
74 */
75 template <class T>
76 void SetMultiGridWallQuantity(const CGeometry* fine_grid, unsigned short val_marker, T& wall_quantity) {
77 for (auto iVertex = 0ul; iVertex < nVertex[val_marker]; iVertex++) {
78 const auto Point_Coarse = vertex[val_marker][iVertex]->GetNode();
79
80 if (!nodes->GetDomain(Point_Coarse)) continue;
81
82 su2double Area_Parent = 0.0;
83
84 /*--- Compute area parent by taking into account only volumes that are on the marker. ---*/
85 for (auto iChildren = 0u; iChildren < nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
86 const auto Point_Fine = nodes->GetChildren_CV(Point_Coarse, iChildren);
87 const auto isVertex =
88 fine_grid->nodes->GetDomain(Point_Fine) && (fine_grid->nodes->GetVertex(Point_Fine, val_marker) != -1);
89 if (isVertex) {
90 Area_Parent += fine_grid->nodes->GetVolume(Point_Fine);
91 }
92 }
93
94 su2double Quantity_Coarse = 0.0;
95
96 /*--- Loop again to average coarser value. ---*/
97 for (auto iChildren = 0u; iChildren < nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
98 const auto Point_Fine = nodes->GetChildren_CV(Point_Coarse, iChildren);
99 const auto isVertex =
100 fine_grid->nodes->GetDomain(Point_Fine) && (fine_grid->nodes->GetVertex(Point_Fine, val_marker) != -1);
101 if (isVertex) {
102 const auto Vertex_Fine = fine_grid->nodes->GetVertex(Point_Fine, val_marker);
103 const auto Area_Children = fine_grid->nodes->GetVolume(Point_Fine);
104 Quantity_Coarse += wall_quantity.Get(Vertex_Fine) * Area_Children / Area_Parent;
105 }
106 }
107
108 /*--- Set the value at the coarse level. ---*/
109 wall_quantity.Set(iVertex, Quantity_Coarse);
110 }
111 }
