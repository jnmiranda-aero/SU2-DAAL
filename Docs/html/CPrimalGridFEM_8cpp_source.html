30CPrimalGridFEM::CPrimalGridFEM(unsigned long val_elemGlobalID, unsigned short val_VTK_Type,
31 unsigned short val_nPolyGrid, unsigned short val_nPolySol, unsigned short val_nDOFsGrid,
32 unsigned short val_nDOFsSol, unsigned long val_offDOfsSol, std::istringstream& elem_line)
33 : CPrimalGrid(true, val_nDOFsGrid, nFacesOfElementType(val_VTK_Type)) {
34 /*--- Store the integer data in the member variables of this object. ---*/
35 VTK_Type = val_VTK_Type;
36 nFaces = nFacesOfElementType(VTK_Type);
37
38 nPolyGrid = val_nPolyGrid;
39 nPolySol = val_nPolySol;
40 nDOFsGrid = val_nDOFsGrid;
41 nDOFsSol = val_nDOFsSol;
42
43 elemIDGlobal = val_elemGlobalID;
44 offsetDOFsSolGlobal = val_offDOfsSol;
45
46 /*--- Read face structure of the element from elem_line. ---*/
47
48 for (unsigned short i = 0; i < nDOFsGrid; i++) elem_line >> Nodes[i];
49
50 /*--- If a linear element is used, the node numbering for non-simplices
51 must be adapted. The reason is that compatability with the original
52 SU2 format is maintained for linear elements, but for the FEM solver
53 the nodes of the elements are stored row-wise. ---*/
54 if (nPolyGrid == 1) {
55 switch (VTK_Type) {
56 case QUADRILATERAL:
57 std::swap(Nodes[2], Nodes[3]);
58 break;
59
60 case HEXAHEDRON:
61 std::swap(Nodes[2], Nodes[3]);
62 std::swap(Nodes[6], Nodes[7]);
63 break;
64
65 case PYRAMID:
66 std::swap(Nodes[2], Nodes[3]);
67 break;
68 }
69 }
70}

92void CPrimalGridFEM::GetLocalCornerPointsAllFaces(unsigned short elementType, unsigned short nPoly,
93 unsigned short nDOFs, unsigned short& numFaces,
94 unsigned short nPointsPerFace[], unsigned long faceConn[6][4]) {
95 /*--- Determine the element type and set the face data accordingly.
96 The faceConn values are local to the element. ---*/
97
98 numFaces = nFacesOfElementType(elementType);
99 unsigned short nn2, nn3, nn4;
100 switch (elementType) {
101 case TRIANGLE:
102 nPointsPerFace[0] = 2;
103 faceConn[0][0] = 0;
104 faceConn[0][1] = nPoly;
105 nPointsPerFace[1] = 2;
106 faceConn[1][0] = nPoly;
107 faceConn[1][1] = nDOFs - 1;
108 nPointsPerFace[2] = 2;
109 faceConn[2][0] = nDOFs - 1;
110 faceConn[2][1] = 0;
111 break;
112
113 case QUADRILATERAL:
114 nn2 = nPoly * (nPoly + 1);
115 nPointsPerFace[0] = 2;
116 faceConn[0][0] = 0;
117 faceConn[0][1] = nPoly;
118 nPointsPerFace[1] = 2;
119 faceConn[1][0] = nPoly;
120 faceConn[1][1] = nDOFs - 1;
121 nPointsPerFace[2] = 2;
122 faceConn[2][0] = nDOFs - 1;
123 faceConn[2][1] = nn2;
124 nPointsPerFace[3] = 2;
125 faceConn[3][0] = nn2;
126 faceConn[3][1] = 0;
127 break;
128
129 case TETRAHEDRON:
130 nn2 = (nPoly + 1) * (nPoly + 2) / 2 - 1;
131 nn3 = nDOFs - 1;
132 nPointsPerFace[0] = 3;
133 faceConn[0][0] = 0;
134 faceConn[0][1] = nPoly;
135 faceConn[0][2] = nn2;
136 nPointsPerFace[1] = 3;
137 faceConn[1][0] = 0;
138 faceConn[1][1] = nn3;
139 faceConn[1][2] = nPoly;
140 nPointsPerFace[2] = 3;
141 faceConn[2][0] = 0;
142 faceConn[2][1] = nn2;
143 faceConn[2][2] = nn3;
144 nPointsPerFace[3] = 3;
145 faceConn[3][0] = nPoly;
146 faceConn[3][1] = nn3;
147 faceConn[3][2] = nn2;
148 break;
149
150 case PYRAMID:
151 nn2 = (nPoly + 1) * (nPoly + 1) - 1;
152 nn3 = nn2 - nPoly;
153 nPointsPerFace[0] = 4;
154 faceConn[0][0] = 0;
155 faceConn[0][1] = nPoly;
156 faceConn[0][2] = nn2;
157 faceConn[0][3] = nn3;
158 nPointsPerFace[1] = 3;
159 faceConn[1][0] = 0;
160 faceConn[1][1] = nDOFs - 1;
161 faceConn[1][2] = nPoly;
162 nPointsPerFace[2] = 3;
163 faceConn[2][0] = nn3;
164 faceConn[2][1] = nn2;
165 faceConn[2][2] = nDOFs - 1;
166 nPointsPerFace[3] = 3;
167 faceConn[3][0] = 0;
168 faceConn[3][1] = nn3;
169 faceConn[3][2] = nDOFs - 1;
170 nPointsPerFace[4] = 3;
171 faceConn[4][0] = nPoly;
172 faceConn[4][1] = nDOFs - 1;
173 faceConn[4][2] = nn2;
174 break;
175
176 case PRISM:
177 nn2 = (nPoly + 1) * (nPoly + 2) / 2;
178 nn3 = nPoly * nn2;
179 --nn2;
180 nPointsPerFace[0] = 3;
181 faceConn[0][0] = 0;
182 faceConn[0][1] = nPoly;
183 faceConn[0][2] = nn2;
184 nPointsPerFace[1] = 3;
185 faceConn[1][0] = nn3;
186 faceConn[1][1] = nn2 + nn3;
187 faceConn[1][2] = nPoly + nn3;
188 nPointsPerFace[2] = 4;
189 faceConn[2][0] = 0;
190 faceConn[2][1] = nn3;
191 faceConn[2][2] = nPoly + nn3;
192 faceConn[2][3] = nPoly;
193 nPointsPerFace[3] = 4;
194 faceConn[3][0] = 0;
195 faceConn[3][1] = nn2;
196 faceConn[3][2] = nn2 + nn3;
197 faceConn[3][3] = nn3;
198 nPointsPerFace[4] = 4;
199 faceConn[4][0] = nPoly;
200 faceConn[4][1] = nPoly + nn3;
201 faceConn[4][2] = nn2 + nn3;
202 faceConn[4][3] = nn2;
203 break;
204
205 case HEXAHEDRON:
206 nn2 = (nPoly + 1) * (nPoly + 1);
207 nn4 = nPoly * nn2;
208 --nn2;
209 nn3 = nn2 - nPoly;
210 nPointsPerFace[0] = 4;
211 faceConn[0][0] = 0;
212 faceConn[0][1] = nPoly;
213 faceConn[0][2] = nn2;
214 faceConn[0][3] = nn3;
215 nPointsPerFace[1] = 4;
216 faceConn[1][0] = nn4;
217 faceConn[1][1] = nn3 + nn4;
218 faceConn[1][2] = nn2 + nn4;
219 faceConn[1][3] = nPoly + nn4;
220 nPointsPerFace[2] = 4;
221 faceConn[2][0] = 0;
222 faceConn[2][1] = nn4;
223 faceConn[2][2] = nPoly + nn4;
224 faceConn[2][3] = nPoly;
225 nPointsPerFace[3] = 4;
226 faceConn[3][0] = nn3;
227 faceConn[3][1] = nn2;
228 faceConn[3][2] = nn2 + nn4;
229 faceConn[3][3] = nn3 + nn4;
230 nPointsPerFace[4] = 4;
231 faceConn[4][0] = 0;
232 faceConn[4][1] = nn3;
233 faceConn[4][2] = nn3 + nn4;
234 faceConn[4][3] = nn4;
235 nPointsPerFace[5] = 4;
236 faceConn[5][0] = nPoly;
237 faceConn[5][1] = nPoly + nn4;
238 faceConn[5][2] = nn2 + nn4;
239 faceConn[5][3] = nn2;
240 break;
241 }
242}

244void CPrimalGridFEM::GetCornerPointsAllFaces(unsigned short& numFaces, unsigned short nPointsPerFace[],
245 unsigned long faceConn[6][4]) const {
246 /*--- Get the corner points of the faces local to the element. ---*/
247
248 GetLocalCornerPointsAllFaces(VTK_Type, nPolyGrid, nDOFsGrid, numFaces, nPointsPerFace, faceConn);
249
250 /*--- Convert the local values of faceConn to global values. ---*/
251
252 for (unsigned short i = 0; i < numFaces; ++i) {
253 for (unsigned short j = 0; j < nPointsPerFace[i]; ++j) {
254 unsigned long nn = faceConn[i][j];
255 faceConn[i][j] = Nodes[nn];
256 }
257 }
