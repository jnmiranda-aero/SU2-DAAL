37 void CFileReaderLUT::ReadRawLUT(const string& file_name) {
38  version_reader = "1.1.0";
39 
40  /*--- Store MPI rank. ---*/
41  rank = SU2_MPI::GetRank();
42 
43  string line;
44  string word;
45 
46  istringstream stream_names_var;
47 
48  ifstream file_stream;
49 
50  bool eoHeader = false;
51  bool eoData = false;
52  bool eoConnectivity = false;
53  bool eoHull = false;
54 
55  file_stream.open(file_name.c_str(), ifstream::in);
56 
57  if (!file_stream.is_open()) {
58  SU2_MPI::Error(string("There is no look-up-table file called ") + file_name, CURRENT_FUNCTION);
59  }
60 
61  /*--- Read header ---*/
62  SkipToFlag(file_stream, line, "<Header>");
63  table_dim = 2;
64  bool found_level_count = false;
65  while (GetNextNonEmptyLine(file_stream, line) && !eoHeader) {
66  /*--- check version_lut ---*/
67  if (line.compare("[Version]") == 0) {
68  GetNextNonEmptyLine(file_stream, line);
69  version_lut = line;
70 
71  if (version_lut.compare("1.0.1") == 0) {
72  table_dim = 2;
73  n_levels = 1;
74  found_level_count = true;
75  n_points.resize(n_levels);
76  n_triangles.resize(n_levels);
77  n_hull_points.resize(n_levels);
78  } else if (version_lut.compare("1.1.0") == 0) {
79  table_dim = 3;
80  } else {
81  SU2_MPI::Error("Version conflict between LUT reader and LUT library file.", CURRENT_FUNCTION);
82  }
83  }
84 
85  if (line.compare("[Number of table levels]") == 0) {
86  found_level_count = true;
87  GetNextNonEmptyLine(file_stream, line);
88  n_levels = stoul(line);
89 
90  n_points.resize(n_levels);
91  n_triangles.resize(n_levels);
92  n_hull_points.resize(n_levels);
93  table_levels.resize(n_levels);
94  }
95 
96  /*--- number of points in LUT ---*/
97  if (line.compare("[Number of points]") == 0) {
98  if (!found_level_count)
99  SU2_MPI::Error("Number of points provided before specifying level count.", CURRENT_FUNCTION);
100  for (unsigned long i_level = 0; i_level < n_levels; i_level++) {
101  GetNextNonEmptyLine(file_stream, line);
102  n_points[i_level] = stoi(line);
103  }
104  }
105 
106  /*--- number of triangles in LUT ---*/
107  if (line.compare("[Number of triangles]") == 0) {
108  if (!found_level_count)
109  SU2_MPI::Error("Number of triangles provided before specifying level count.", CURRENT_FUNCTION);
110  for (unsigned long i_level = 0; i_level < n_levels; i_level++) {
111  GetNextNonEmptyLine(file_stream, line);
112  n_triangles[i_level] = stoi(line);
113  }
114  }
115 
116  /*--- number of points on the hull ---*/
117  if (line.compare("[Number of hull points]") == 0) {
118  if (!found_level_count)
119  SU2_MPI::Error("Number of hull points provided before specifying level count.", CURRENT_FUNCTION);
120  for (unsigned long i_level = 0; i_level < n_levels; i_level++) {
121  GetNextNonEmptyLine(file_stream, line);
122  n_hull_points[i_level] = stoi(line);
123  }
124  }
125 
126  if (line.compare("[Table levels]") == 0) {
127  for (unsigned long i_level = 0; i_level < n_levels; i_level++) {
128  GetNextNonEmptyLine(file_stream, line);
129  table_levels[i_level] = stod(line);
130  }
131 
132  /*--- Checking whether table levels are ordered and appearance of duplicates ---*/
133  for (auto i_level = 0u; i_level < n_levels - 1; i_level++) {
134  if (table_levels[i_level] > table_levels[i_level + 1])
135  SU2_MPI::Error("Table levels are not provided in ascending order.", CURRENT_FUNCTION);
136  }
137  auto duplicate = adjacent_find(table_levels.begin(), table_levels.end()) != table_levels.end();
138  if (duplicate) SU2_MPI::Error("Duplicate table levels are present in LUT file", CURRENT_FUNCTION);
139  }
140 
141  /*--- number of variables in LUT ---*/
142  if (line.compare("[Number of variables]") == 0) {
143  GetNextNonEmptyLine(file_stream, line);
144  n_variables = stoi(line);
145  }
146 
147  /*--- variable names ---*/
148  if (line.compare("[Variable names]") == 0) {
149  names_var.resize(n_variables);
150  for (unsigned long i = 0; i < n_variables; i++) {
151  /*--- grab a single line ---*/
152  GetNextNonEmptyLine(file_stream, line);
153  names_var[i] = line.substr(line.find(':') + 1);
154  }
155  }
156 
157  /*--- check if end of header is reached ---*/
158  if (line.compare("</Header>") == 0) eoHeader = true;
159  }
160 
161  /*--- check header quantities ---*/
162  for (unsigned long i_level = 0; i_level < n_levels; i_level++) {
163  if (n_points[i_level] == 0 || n_triangles[i_level] == 0 || n_variables == 0 || n_hull_points[i_level] == 0)
164  SU2_MPI::Error(
165  "Number of points, triangles, hull points, or variables in lookup table "
166  "library header is zero.",
167  CURRENT_FUNCTION);
168  }
169 
170  /*--- check if number of variables is consistent ---*/
171  if (n_variables != names_var.size())
172  SU2_MPI::Error(
173  "Number of read variables does not match number of "
174  "variables specified in lookup table "
175  "library header.",
176  CURRENT_FUNCTION);
177 
178  /*--- now that n_variables, n_points, n_hull_points and n_variables are available, allocate memory ---*/
179  if (rank == MASTER_NODE)
180  cout << "allocating memory for the data, size = ( " << GetNVariables() << " , " << GetNPoints() << " )" << endl;
181  table_data.resize(n_levels);
182  for (unsigned long i_level = 0; i_level < n_levels; i_level++)
183  table_data[i_level].resize(GetNVariables(), GetNPoints(i_level));
184 
185  if (rank == MASTER_NODE) cout << "allocating memory for the triangles, size = " << GetNTriangles() << endl;
186  triangles.resize(n_levels);
187  for (unsigned long i_level = 0; i_level < n_levels; i_level++) triangles[i_level].resize(GetNTriangles(i_level), 3);
188 
189  if (rank == MASTER_NODE) cout << "allocating memory for the hull points, size = " << GetNHullPoints() << endl;
190  hull.resize(n_levels);
191  for (unsigned long i_level = 0; i_level < n_levels; i_level++) hull[i_level].resize(GetNHullPoints(i_level));
192 
193  /*--- flush any cout ---*/
194  if (rank == MASTER_NODE) cout << endl;
195 
196  /*--- read data block ---*/
197  if (rank == MASTER_NODE) cout << "loading data block" << endl;
198 
199  SkipToFlag(file_stream, line, "<Data>");
200  bool eoLevel = true;
201  unsigned long pointCounter = 0;
202  unsigned long levelCounter = 0;
203  while (GetNextNonEmptyLine(file_stream, line) && !eoData) {
204  /*--- check if end of data is reached ---*/
205  if (line.compare("</Data>") == 0) eoData = true;
206 
207  if (!eoData) {
208  if (line.compare("<Level>") == 0) {
209  eoLevel = false;
210  pointCounter = 0;
211  GetNextNonEmptyLine(file_stream, line);
212  }
213  if (line.compare("</Level>") == 0) {
214  eoLevel = true;
215  if (n_points[levelCounter] != pointCounter)
216  SU2_MPI::Error("Number of read points on level " + std::to_string(levelCounter) +
217  " does not match number of points "
218  "specified in lookup table library header.",
219  CURRENT_FUNCTION);
220  levelCounter++;
221  }
222  if (!eoLevel || (table_dim == 2)) {
223  /*--- one line contains values for one point for all variables ---*/
224  istringstream streamDataLine(line);
225 
226  /*--- add next line to table array ---*/
227  for (unsigned long iVar = 0; iVar < n_variables; iVar++) {
228  streamDataLine >> word;
229  passivedouble tmp = stod(word);
230  table_data[levelCounter][iVar][pointCounter] = (su2double)tmp;
231  }
232  pointCounter++;
233  }
234  }
235  }
236 
237  /*--- read connectivity ---*/
238  if (rank == MASTER_NODE) cout << "loading connectivity block" << endl;
239 
240  SkipToFlag(file_stream, line, "<Connectivity>");
241  unsigned long triCounter = 0;
242  eoLevel = true;
243  levelCounter = 0;
244  while (GetNextNonEmptyLine(file_stream, line) && !eoConnectivity) {
245  if (!line.empty() && (line[line.length() - 1] == '\n' || line[line.length() - 1] == '\r')) {
246  line.erase(line.length() - 1);
247  }
248  /*--- check if end of data is reached ---*/
249  if (line.compare("</Connectivity>") == 0) eoConnectivity = true;
250 
251  if (!eoConnectivity) {
252  if (line.compare("<Level>") == 0) {
253  eoLevel = false;
254  triCounter = 0;
255  GetNextNonEmptyLine(file_stream, line);
256  }
257  if (line.compare("</Level>") == 0) {
258  eoLevel = true;
259  if (n_triangles[levelCounter] != triCounter)
260  SU2_MPI::Error("Number of read triangles on level " + std::to_string(levelCounter) +
261  " does not match number of triangles "
262  "specified in lookup table library header.",
263  CURRENT_FUNCTION);
264  levelCounter++;
265  }
266  if (!eoLevel || (table_dim == 2)) {
267  /*--- one line contains values for one triangle (3 points) ---*/
268  istringstream streamTriLine(line);
269 
270  /*--- add next line to triangles ---*/
271  for (int iPoint = 0; iPoint < 3; iPoint++) {
272  streamTriLine >> word;
273 
274  /*--- lookup table index starts with 1, convert to c++ indexing starting with 0: ---*/
275  triangles[levelCounter][triCounter][iPoint] = stol(word) - 1;
276  }
277  triCounter++;
278  }
279  }
280  }
281 
282  /*--- read hull points ---*/
283  if (rank == MASTER_NODE) cout << "loading hull block" << endl;
284 
285  SkipToFlag(file_stream, line, "<Hull>");
286  unsigned long hullCounter = 0;
287  eoLevel = true;
288  levelCounter = 0;
289  while (GetNextNonEmptyLine(file_stream, line) && !eoHull) {
290  if (!line.empty() && (line[line.length() - 1] == '\n' || line[line.length() - 1] == '\r')) {
291  line.erase(line.length() - 1);
292  }
293  /*--- check if end of data is reached ---*/
294  if (line.compare("</Hull>") == 0) eoHull = true;
295 
296  if (!eoHull) {
297  if (line.compare("<Level>") == 0) {
298  eoLevel = false;
299  hullCounter = 0;
300  GetNextNonEmptyLine(file_stream, line);
301  }
302  if (line.compare("</Level>") == 0) {
303  eoLevel = true;
304  if (n_hull_points[levelCounter] != hullCounter)
305  SU2_MPI::Error("Number of read hull nodes on level " + std::to_string(levelCounter) +
306  " does not match number of hull nodes "
307  "specified in lookup table library header.",
308  CURRENT_FUNCTION);
309  levelCounter++;
310  }
311  if (!eoLevel || (table_dim == 2)) {
312  /*--- one line contains one point ID for one point on the hull ---*/
313  istringstream streamHullLine(line);
314 
315  streamHullLine >> word;
316 
317  /*--- Lookup table indices start with 1, convert to c++ indexing starting with 0: ---*/
318  hull[levelCounter][hullCounter] = stol(word) - 1;
319  hullCounter++;
320  }
321  }
322  }
323 
324  file_stream.close();
325 }

327 void CFileReaderLUT::SkipToFlag(ifstream& file_stream, const string& current_line, const string& flag) const {
328  string next_line;
329 
330  /*--- compare current line to flag and return if equal ---*/
331  if (current_line.compare(flag) == 0) return;
332 
333  /*--- else, search for flag ---*/
334  while (next_line.find(flag) == string::npos && !(file_stream).eof()) {
335  GetStrippedLine(file_stream, next_line);
336  }
337 
338  /*--- throw error if end of file reached ---*/
339  if ((file_stream).eof()) SU2_MPI::Error("Flag " + flag + " not found in file", CURRENT_FUNCTION);
340 }

342 bool CFileReaderLUT::GetNextNonEmptyLine(ifstream& file_stream, string& line) const {
343  /*--- get next line and save return value ---*/
344  bool return_value = GetStrippedLine(file_stream, line);
345 
346  /*--- skip empty lines ---*/
347  while (line.empty() && !(file_stream).eof()) {
348  return_value = GetStrippedLine(file_stream, line);
349  }
350 
351  /*--- return true if line is not empty, else return false ---*/
352  return return_value;
353 }

355 bool CFileReaderLUT::GetStrippedLine(ifstream& file_stream, string& line) const {
356  /*--- get next line and save return value ---*/
357  getline(file_stream, line);
358 
359  /*--- find last non-control-character character ---*/
360  size_t end = line.find_last_not_of(" \n\r\t\f\v");
361 
362  /*--- clean up line ---*/
363  line = (end == string::npos) ? "" : line.substr(0, end + 1);
364 
365  /*--- return true if line is not empty, else return false ---*/
366  return !line.empty();
367 }
