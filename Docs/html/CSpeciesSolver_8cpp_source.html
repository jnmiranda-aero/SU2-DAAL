37CSpeciesSolver::CSpeciesSolver(CGeometry* geometry, CConfig* config, unsigned short iMesh)
38 : CScalarSolver<CSpeciesVariable>(geometry, config, true) {
39
40 /*--- Dimension of the problem. ---*/
41
42 nVar = config->GetnSpecies();
43
44 Initialize(geometry, config, iMesh, nVar);
45
46 /*--- Initialize the solution to the far-field state everywhere. ---*/
47
48 nodes = new CSpeciesVariable(Solution_Inf, nPoint, nDim, nVar, config);
49 SetBaseClassPointerToNodes();
50
51 /*--- Initialize the mass diffusivity. Nondimensionalization done in the flow solver. ---*/
52 SU2_OMP_FOR_STAT(omp_chunk_size)
53 for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
54 for (auto iVar = 0u; iVar <= nVar; iVar++) {
55 const auto MassDiffusivity = config->GetDiffusivity_ConstantND();
56 nodes->SetDiffusivity(iPoint, MassDiffusivity, iVar);
57 }
58 }
59 END_SU2_OMP_FOR
60
61 /*--- MPI solution ---*/
62
63 InitiateComms(geometry, config, SOLUTION);
64 CompleteComms(geometry, config, SOLUTION);
65
66 SlidingState.resize(nMarker);
67 SlidingStateNodes.resize(nMarker);
68
69 for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
70 if (config->GetMarker_All_KindBC(iMarker) == FLUID_INTERFACE) {
71 SlidingState[iMarker].resize(nVertex[iMarker], nPrimVar+1) = nullptr;
72 SlidingStateNodes[iMarker].resize(nVertex[iMarker],0);
73 }
74 }
75
76 /*--- Store the initial CFL number for all grid points. ---*/
77
78 const su2double CFL = config->GetCFL(MGLevel) * config->GetCFLRedCoeff_Species();
79 SU2_OMP_FOR_STAT(omp_chunk_size)
80 for (auto iPoint = 0u; iPoint < nPoint; iPoint++) {
81 nodes->SetLocalCFL(iPoint, CFL);
82 }
83 END_SU2_OMP_FOR
84 Min_CFL_Local = CFL;
85 Max_CFL_Local = CFL;
86 Avg_CFL_Local = CFL;
87
88 /*--- Add the solver name. ---*/
89 SolverName = "SPECIES";
90}

93void CSpeciesSolver::Initialize(CGeometry* geometry, CConfig* config, unsigned short iMesh, unsigned short nVar) {
94 /*--- Store if an implicit scheme is used, for use during periodic boundary conditions. ---*/
95 SetImplicitPeriodic(config->GetKind_TimeIntScheme_Species() == EULER_IMPLICIT);
96
97 nPrimVar = nVar;
98
99 if (nVar > MAXNVAR)
100 SU2_MPI::Error("Increase static array size MAXNVAR for CSpeciesVariable and proceed.", CURRENT_FUNCTION);
101
102 nPoint = geometry->GetnPoint();
103 nPointDomain = geometry->GetnPointDomain();
104
105 /*--- Initialize nVarGrad for deallocation ---*/
106
107 nVarGrad = nVar;
108
109 /*--- Define geometry constants in the solver structure ---*/
110
111 nDim = geometry->GetnDim();
112
113
114if (iMesh == MESH_0 || config->GetMGCycle() == FULLMG_CYCLE) {
115
116 /*--- Define some auxiliary vector related with the residual ---*/
117
118 Residual_RMS.resize(nVar, 0.0);
119 Residual_Max.resize(nVar, 0.0);
120 Point_Max.resize(nVar, 0);
121 Point_Max_Coord.resize(nVar, nDim) = su2double(0.0);
122
123 /*--- Initialize the BGS residuals in multizone problems. ---*/
124 if (config->GetMultizone_Problem()) {
125 Residual_BGS.resize(nVar, 0.0);
126 Residual_Max_BGS.resize(nVar, 0.0);
127 Point_Max_BGS.resize(nVar, 0);
128 Point_Max_Coord_BGS.resize(nVar, nDim) = su2double(0.0);
129 }
130
131 /*--- Initialization of the structure of the whole Jacobian ---*/
132
133 if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (species transport model)." << endl;
134 Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config, ReducerStrategy);
135 LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
136 LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
137 System.SetxIsZero(true);
138
139 if (ReducerStrategy) EdgeFluxes.Initialize(geometry->GetnEdge(), geometry->GetnEdge(), nVar, nullptr);
140 }
141
142 /*--- Initialize lower and upper limits---*/
143
144 if (config->GetSpecies_Clipping()) {
145 for (auto iVar = 0u; iVar < nVar; iVar++) {
146 lowerlimit[iVar] = config->GetSpecies_Clipping_Min(iVar);
147 upperlimit[iVar] = config->GetSpecies_Clipping_Max(iVar);
148 }
149 } else {
150 for (auto iVar = 0u; iVar < nVar; iVar++) {
151 lowerlimit[iVar] = -1.0e15;
152 upperlimit[iVar] = 1.0e15;
153 }
154 }
155
156 /*--- Scalar variable state at the far-field. ---*/
157
158 for (auto iVar = 0u; iVar < nVar; iVar++) {
159 Solution_Inf[iVar] = config->GetSpecies_Init()[iVar];
160 }
161
162 /*--- Set the column number for species in inlet-files.
163 * e.g. Coords(nDim), Temp(1), VelMag(1), Normal(nDim), Turb(1 or 2), Species(arbitrary) ---*/
164 Inlet_Position = nDim + 2 + nDim + config->GetnTurbVar();
165
166 /*-- Allocation of inlet-values. Will be filled either by an inlet file,
167 * or uniformly by a uniform boundary condition. ---*/
168
169 Inlet_SpeciesVars.resize(nMarker);
170 for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
171 Inlet_SpeciesVars[iMarker].resize(nVertex[iMarker], nVar);
172 for (unsigned long iVertex = 0; iVertex < nVertex[iMarker]; ++iVertex) {
173 for (unsigned short iVar = 0; iVar < nVar; iVar++) {
174 Inlet_SpeciesVars[iMarker](iVertex, iVar) = Solution_Inf[iVar];
175 }
176 }
177 }
178}

181void CSpeciesSolver::LoadRestart(CGeometry** geometry, CSolver*** solver, CConfig* config, int val_iter,
182 bool val_update_geo) {
183 /*--- Restart the solution from file information ---*/
184
185 const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
186
187 /*--- To make this routine safe to call in parallel most of it can only be executed by one thread. ---*/
188 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
189 /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
190
191 if (config->GetRead_Binary_Restart()) {
192 Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
193 } else {
194 Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
195 }
196
197 /*--- Skip flow variables and turbulence variables. ---*/
198
199 unsigned short skipVars = nDim + solver[MESH_0][FLOW_SOL]->GetnVar() + config->GetnTurbVar();
200
201 /*--- Adjust the number of solution variables in the incompressible
202 restart. We always carry a space in nVar for the energy equation in the
203 mean flow solver, but we only write it to the restart if it is active.
204 Therefore, we must reduce skipVars here if energy is inactive so that
205 the turbulent variables are read correctly. ---*/
206
207 const bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
208 const bool energy = config->GetEnergy_Equation();
209 const bool flamelet = (config->GetKind_FluidModel() == FLUID_FLAMELET);
210 const bool weakly_coupled_heat = config->GetWeakly_Coupled_Heat();
211
212 /*--- for the flamelet model, the temperature is saved to file, but the energy equation is off ---*/
213
214 if (incompressible && ((!energy) && (!weakly_coupled_heat) && (!flamelet))) skipVars--;
215
216 /*--- Load data from the restart into correct containers. ---*/
217
218 unsigned long counter = 0;
219 for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
220 /*--- Retrieve local index. If this node from the restart file lives
221 on the current processor, we will load and instantiate the vars. ---*/
222
223 const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
224
225 if (iPoint_Local > -1) {
226 /*--- We need to store this point's data, so jump to the correct
227 offset in the buffer of data from the restart file and load it. ---*/
228
229 const auto index = counter * Restart_Vars[1] + skipVars;
230 for (auto iVar = 0u; iVar < nVar; iVar++)
231 nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index + iVar]);
232
233 /*--- Increment the overall counter for how many points have been loaded. ---*/
234 counter++;
235 }
236 }
237
238 /*--- Detect a wrong solution file ---*/
239
240 if (counter != nPointDomain) {
241 SU2_MPI::Error(string("The solution file ") + restart_filename + string(" doesn't match with the mesh file!\n") +
242 string("It could be empty lines at the end of the file."),
243 CURRENT_FUNCTION);
244 }
245
246 } // end safe global access, pre and postprocessing are thread-safe.
247 END_SU2_OMP_SAFE_GLOBAL_ACCESS
248
249 /*--- MPI solution and compute the eddy viscosity ---*/
250
251 solver[MESH_0][SPECIES_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
252 solver[MESH_0][SPECIES_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
253
254 // Flow-Pre computes/sets mixture properties
255 solver[MESH_0][FLOW_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
256 RUNTIME_FLOW_SYS, false);
257 // Update eddy-visc which needs correct mixture density and mixture lam-visc. Note that after this, another Flow-Pre
258 // at the start of the Iteration sets the updated eddy-visc into the Flow-Solvers Primitives.
259 if (config->GetKind_Turb_Model() != TURB_MODEL::NONE)
260 solver[MESH_0][TURB_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
261 // For feature_multicomp this Scalar-Pre only computes the laminar contribution to mass diffusivity
262 solver[MESH_0][SPECIES_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
263 RUNTIME_SPECIES_SYS, false);
264
265 /*--- Interpolate the solution down to the coarse multigrid levels ---*/
266
267 for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
268 MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][SPECIES_SOL]->GetNodes()->GetSolution(),
269 *geometry[iMesh], solver[iMesh][SPECIES_SOL]->GetNodes()->GetSolution());
270 solver[iMesh][SPECIES_SOL]->InitiateComms(geometry[iMesh], config, SOLUTION);
271 solver[iMesh][SPECIES_SOL]->CompleteComms(geometry[iMesh], config, SOLUTION);
272
273 solver[iMesh][FLOW_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_FLOW_SYS,
274 false);
275
276 if (config->GetKind_Turb_Model() != TURB_MODEL::NONE)
277 solver[iMesh][TURB_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
278
279 solver[iMesh][SPECIES_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
280 RUNTIME_SPECIES_SYS, false);
281 }
282
283 /*--- Go back to single threaded execution. ---*/
284 BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
285 /*--- Delete the class memory that is used to load the restart. ---*/
286
287 Restart_Vars = decltype(Restart_Vars){};
288 Restart_Data = decltype(Restart_Data){};
289 }
290 END_SU2_OMP_SAFE_GLOBAL_ACCESS
291}

293void CSpeciesSolver::Preprocessing(CGeometry* geometry, CSolver** solver_container, CConfig* config,
294 unsigned short iMesh, unsigned short iRKStep, unsigned short RunTime_EqSystem,
295 bool Output) {
296 SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetGlobalParam(config->GetKind_Solver(), RunTime_EqSystem);)
297
298 /*--- Set the laminar mass Diffusivity for the species solver. ---*/
299 SU2_OMP_FOR_STAT(omp_chunk_size)
300 for (auto iPoint = 0u; iPoint < nPoint; iPoint++) {
301 const su2double temperature = solver_container[FLOW_SOL]->GetNodes()->GetTemperature(iPoint);
302 const su2double* scalar = solver_container[SPECIES_SOL]->GetNodes()->GetSolution(iPoint);
303 solver_container[FLOW_SOL]->GetFluidModel()->SetMassDiffusivityModel(config);
304 solver_container[FLOW_SOL]->GetFluidModel()->SetTDState_T(temperature, scalar);
305 for (auto iVar = 0u; iVar <= nVar; iVar++) {
306 const su2double mass_diffusivity = solver_container[FLOW_SOL]->GetFluidModel()->GetMassDiffusivity(iVar);
307 nodes->SetDiffusivity(iPoint, mass_diffusivity, iVar);
308 }
309
310 } // iPoint
311 END_SU2_OMP_FOR
312
313 /*--- Clear Residual and Jacobian. Upwind second order reconstruction and gradients ---*/
314 CommonPreprocessing(geometry, config, Output);
315}

331void CSpeciesSolver::BC_Inlet(CGeometry* geometry, CSolver** solver_container, CNumerics* conv_numerics,
332 CNumerics* visc_numerics, CConfig* config, unsigned short val_marker) {
333
334 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
335
336 /*--- Loop over all the vertices on this boundary marker ---*/
337
338 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
339 for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
340 auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
341
342 /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
343
344 if (!geometry->nodes->GetDomain(iPoint)) continue;
345
346 /*--- Identify the boundary by string name ---*/
347 string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
348
349 if (config->GetMarker_StrongBC(Marker_Tag)==true) {
350 nodes->SetSolution_Old(iPoint, Inlet_SpeciesVars[val_marker][iVertex]);
351
352 LinSysRes.SetBlock_Zero(iPoint);
353
354 /*--- Includes 1 in the diagonal ---*/
355 for (auto iVar = 0u; iVar < nVar; iVar++) {
356 auto total_index = iPoint * nVar + iVar;
357 Jacobian.DeleteValsRowi(total_index);
358 }
359 } else { // weak BC
360 /*--- Normal vector for this vertex (negate for outward convention) ---*/
361
362 su2double Normal[MAXNDIM] = {0.0};
363 for (auto iDim = 0u; iDim < nDim; iDim++) Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
364 conv_numerics->SetNormal(Normal);
365
366 /*--- Allocate the value at the inlet ---*/
367
368 auto V_inlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
369
370 /*--- Retrieve solution at the farfield boundary node ---*/
371
372 auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
373
374 /*--- Set various quantities in the solver class ---*/
375
376 conv_numerics->SetPrimitive(V_domain, V_inlet);
377
378 /*--- Set the species variable state at the inlet. ---*/
379
380 conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Inlet_SpeciesVars[val_marker][iVertex]);
381
382 /*--- Set various other quantities in the solver class ---*/
383
384 if (dynamic_grid)
385 conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
386
387 if (conv_numerics->GetBoundedScalar()) {
388 const su2double* velocity = &V_inlet[prim_idx.Velocity()];
389 const su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
390 conv_numerics->SetMassFlux(BoundedScalarBCFlux(iPoint, implicit, density, velocity, Normal));
391 }
392
393 /*--- Compute the residual using an upwind scheme ---*/
394
395 auto residual = conv_numerics->ComputeResidual(config);
396 LinSysRes.AddBlock(iPoint, residual);
397
398 /*--- Jacobian contribution for implicit integration ---*/
399
400 if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
401
402 // Unfinished viscous contribution removed before right after d8a0da9a00. Further testing required.
403
404 }
405 }
406 END_SU2_OMP_FOR

418su2double CSpeciesSolver::GetInletAtVertex(su2double *val_inlet,
419 unsigned long val_inlet_point,
420 unsigned short val_kind_marker,
421 string val_marker,
422 const CGeometry *geometry,
423 const CConfig *config) const {
424 /*--- Local variables ---*/
425
426 unsigned short iMarker;
427 unsigned long iPoint, iVertex;
428 su2double Area = 0.0;
429 su2double Normal[3] = {0.0,0.0,0.0};
430
431 /*--- Alias positions within inlet file for readability ---*/
432
433 if (val_kind_marker == INLET_FLOW) {
434
435 for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
436 if ((config->GetMarker_All_KindBC(iMarker) == INLET_FLOW) &&
437 (config->GetMarker_All_TagBound(iMarker) == val_marker)) {
438
439 for (iVertex = 0; iVertex < nVertex[iMarker]; iVertex++) {
440
441 iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
442
443 if (iPoint == val_inlet_point) {
444
445 /*-- Compute boundary face area for this vertex. ---*/
446
447 geometry->vertex[iMarker][iVertex]->GetNormal(Normal);
448 Area = GeometryToolbox::Norm(nDim, Normal);
449
450 /*--- Access and store the inlet variables for this vertex. ---*/
451 for (unsigned short iVar = 0; iVar < nVar; iVar++)
452 val_inlet[Inlet_Position + iVar] = Inlet_SpeciesVars[iMarker][iVertex][iVar];
453
454 /*--- Exit once we find the point. ---*/
455
456 return Area;
457
458 }
459 }
460 }
461 }
462
463 }
464
465 /*--- If we don't find a match, then the child point is not on the
466 current inlet boundary marker. Return zero area so this point does
467 not contribute to the restriction operator and continue. ---*/
468
469 return Area;
470

485void CSpeciesSolver::BC_Outlet(CGeometry* geometry, CSolver** solver_container, CNumerics* conv_numerics,
486 CNumerics* visc_numerics, CConfig* config, unsigned short val_marker) {
487
488 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
489
490 /*--- Loop over all the vertices on this boundary marker ---*/
491
492 SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
493 for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
494 /*--- Strong zero flux Neumann boundary condition at the outlet ---*/
495 const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
496
497 /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
498
499 if (!geometry->nodes->GetDomain(iPoint)) continue;
500
501 /*--- Identify the boundary by string name ---*/
502 string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
503
504 if (config->GetMarker_StrongBC(Marker_Tag)==true) {
505 /*--- Allocate the value at the outlet ---*/
506 auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
507
508 nodes->SetSolution_Old(iPoint, nodes->GetSolution(Point_Normal));
509
510 LinSysRes.SetBlock_Zero(iPoint);
511
512 /*--- Includes 1 on the diagonal ---*/
513 for (auto iVar = 0u; iVar < nVar; iVar++) {
514 auto total_index = iPoint * nVar + iVar;
515 Jacobian.DeleteValsRowi(total_index);
516 }
517 } else { // weak BC
518
519 /*--- Allocate the value at the outlet ---*/
520 auto V_outlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
521
522 /*--- Retrieve solution at the farfield boundary node ---*/
523
524 auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
525
526 /*--- Set various quantities in the solver class ---*/
527
528 conv_numerics->SetPrimitive(V_domain, V_outlet);
529
530 /*--- Set the species variables. Here we use a Neumann BC such
531 that the species variable is copied from the interior of the
532 domain to the outlet before computing the residual. ---*/
533
534 conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), nodes->GetSolution(iPoint));
535
536 /*--- Set Normal (negate for outward convention) ---*/
537
538 su2double Normal[MAXNDIM] = {0.0};
539 for (auto iDim = 0u; iDim < nDim; iDim++) Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
540 conv_numerics->SetNormal(Normal);
541
542 if (dynamic_grid)
543 conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
544
545 if (conv_numerics->GetBoundedScalar()) {
546 const su2double* velocity = &V_outlet[prim_idx.Velocity()];
547 const su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
548 conv_numerics->SetMassFlux(BoundedScalarBCFlux(iPoint, implicit, density, velocity, Normal));
549 }
550
551 /*--- Compute the residual using an upwind scheme ---*/
552 auto residual = conv_numerics->ComputeResidual(config);
553 LinSysRes.AddBlock(iPoint, residual);
554
555 /*--- Jacobian contribution for implicit integration ---*/
556 if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
557
558 // Unfinished viscous contribution removed before right after d8a0da9a00. Further testing required.
559
560 }
561 }
562 END_SU2_OMP_FOR

565void CSpeciesSolver::Source_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
566 CConfig *config, unsigned short iMesh) {
567
568 const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
569 const bool axisymmetric = config->GetAxisymmetric();
570
571 if (axisymmetric) {
572 CNumerics *numerics = numerics_container[SOURCE_FIRST_TERM + omp_get_thread_num()*MAX_TERMS];
573
574 SU2_OMP_FOR_DYN(omp_chunk_size)
575 for (auto iPoint = 0u; iPoint < nPointDomain; iPoint++) {
576 /*--- Set primitive variables w/o reconstruction ---*/
577
578 numerics->SetPrimitive(solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint), nullptr);
579
580 /*--- Set scalar variables w/o reconstruction ---*/
581
