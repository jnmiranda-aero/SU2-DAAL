37 /*!
38  * \class CCenteredBase
39  * \ingroup ConvDiscr
40  * \brief Base class for Centered schemes, derived classes implement
41  * the dissipation term in a const "finalizeFlux" method.
42  * \note See CRoeBase for the role of Base.
43  */
44 template<class Derived, class Base>
45 class CCenteredBase : public Base {
46 protected:
47  using Base::nDim;
48  static constexpr size_t nVar = CCompressibleConservatives<nDim>::nVar;
49  static constexpr size_t nPrimVar = Max(Base::nPrimVar, nDim+5);
50 
51  const su2double gamma;
52  const su2double fixFactor;
53  const bool dynamicGrid;
54  const su2double stretchParam = 0.3;
55 
56  /*!
57  * \brief Constructor, store some constants and forward args to base.
58  */
59  template<class... Ts>
60  CCenteredBase(const CConfig& config, Ts&... args) : Base(config, args...),
61  gamma(config.GetGamma()),
62  fixFactor(config.GetCent_Jac_Fix_Factor()),
63  dynamicGrid(config.GetDynamic_Grid()) {
64  }

66  /*!
67  * \brief Special treatment needed to fetch integer data.
68  */
69  template<class T, size_t N>
70  FORCEINLINE static Double numNeighbor(simd::Array<T,N> idx, const CGeometry& geometry) {
71  Double n;
72  for (size_t k=0; k<N; ++k) n[k] = geometry.nodes->GetnNeighbor(idx[k]);
73  return n;

79 public:
80  /*!
81  * \brief Implementation of the base centered flux.
82  */
83  void ComputeFlux(Int iEdge,
84  const CConfig& config,
85  const CGeometry& geometry,
86  const CVariable& solution_,
87  UpdateType updateType,
88  Double updateMask,
89  CSysVector<su2double>& vector,
90  SparseMatrixType& matrix) const final {
91 
92  /*--- Start preaccumulation, inputs are registered
93  * automatically in "gatherVariables". ---*/
94  AD::StartPreacc();
95 
96  const bool implicit = (config.GetKind_TimeIntScheme() == EULER_IMPLICIT);
97  const auto& solution = static_cast<const CEulerVariable&>(solution_);
98 
99  const auto iPoint = geometry.edges->GetNode(iEdge,0);
100  const auto jPoint = geometry.edges->GetNode(iEdge,1);
101 
102  /*--- Geometric properties. ---*/
103 
104  const auto normal = gatherVariables<nDim>(iEdge, geometry.edges->GetNormal());
105  const auto area = norm(normal);
106  VectorDbl<nDim> unitNormal;
107  for (size_t iDim = 0; iDim < nDim; ++iDim) {
108  unitNormal(iDim) = normal(iDim) / area;
109  }
110 
111  /*--- Primitive variables. ---*/
112 
113  CPair<CCompressiblePrimitives<nDim,nPrimVar> > V;
114  V.i.all = gatherVariables<nPrimVar>(iPoint, solution.GetPrimitive());
115  V.j.all = gatherVariables<nPrimVar>(jPoint, solution.GetPrimitive());
116 
117  CCompressiblePrimitives<nDim,nPrimVar> avgV;
118  for (size_t iVar = 0; iVar < nPrimVar; ++iVar) {
119  avgV.all(iVar) = 0.5 * (V.i.all(iVar) + V.j.all(iVar));
120  }
121 
122  /*--- Compute conservative variables. ---*/
123 
124  CPair<CCompressibleConservatives<nDim> > U;
125  U.i = compressibleConservatives(V.i);
126  U.j = compressibleConservatives(V.j);
127 
128  auto avgU = compressibleConservatives(avgV);
129 
130  VectorDbl<nVar> diffU;
131  for (size_t iVar = 0; iVar < nVar-1; ++iVar) {
132  diffU(iVar) = U.i.all(iVar) - U.j.all(iVar);
133  }
134  diffU(nVar-1) = V.i.density()*V.i.enthalpy() - V.j.density()*V.j.enthalpy();
135 
136  /*--- Inviscid fluxes and Jacobians. ---*/
137 
138  auto flux = inviscidProjFlux(avgV, avgU, normal);
139 
140  MatrixDbl<nVar> jac_i, jac_j;
141  if (implicit) {
142  jac_i = inviscidProjJac(gamma, V.i.velocity(), U.i.energy(), normal, 0.5);
143  jac_j = inviscidProjJac(gamma, V.j.velocity(), U.j.energy(), normal, 0.5);
144  }
145 
146  /*--- Grid motion. ---*/
147 
148  Double projGridVel = 0.0;
149  if (dynamicGrid) {
150  const auto& gridVel = geometry.nodes->GetGridVel();
151  projGridVel = 0.5*(dot(gatherVariables<nDim>(iPoint,gridVel), normal)+
152  dot(gatherVariables<nDim>(jPoint,gridVel), normal));
153 
154  for (size_t iVar = 0; iVar < nVar; ++iVar) {
155  flux(iVar) -= projGridVel * avgU.all(iVar);
156  if (implicit) {
157  jac_i(iVar,iVar) -= 0.5 * projGridVel;
158  jac_j(iVar,iVar) -= 0.5 * projGridVel;
159  }
160  }
161  }
162 
163  const Double projVel = dot(avgV.velocity(), normal) - projGridVel;
164 
165  /*--- Finalize in derived class (static polymorphism). ---*/
166 
167  const auto derived = static_cast<const Derived*>(this);
168 
169  derived->finalizeFlux(flux, jac_i, jac_j, implicit, area, projVel, avgV, V,
170  diffU, iPoint, jPoint, geometry, solution, unitNormal);
171 
172  /*--- Add the contributions from the base class (static decorator). ---*/
173 
174  Base::viscousTerms(iEdge, iPoint, jPoint, avgV, V, solution_, geometry,
175  config, area, unitNormal, implicit, flux, jac_i, jac_j);
176 
177  /*--- Stop preaccumulation. ---*/
178 
179  stopPreacc(flux);
180 
181  /*--- Update the vector and system matrix. ---*/
182 
183  updateLinearSystem(iEdge, iPoint, jPoint, implicit, updateType,
184  updateMask, flux, jac_i, jac_j, vector, matrix);
185  }

210  CJSTScheme(const CConfig& config, Ts&... args) : Base(config, args...),

215  /*!
216  * \brief Updates flux and Jacobians with JST dissipation.
217  * \note "Ts" is here just in case other schemes in the family need extra args.
218  */
219  template<class PrimVarType, class... Ts>
220  FORCEINLINE void finalizeFlux(VectorDbl<nVar>& flux,
221  MatrixDbl<nVar>& jac_i,
222  MatrixDbl<nVar>& jac_j,
223  bool implicit,
224  Double area,
225  Double projVel,
226  const PrimVarType& avgV,
227  const CPair<PrimVarType>& V,
228  const VectorDbl<nVar>& diffU,
229  Int iPoint,
230  Int jPoint,
231  const CGeometry& geometry,
232  const CEulerVariable& solution,
233  Ts&...) const {
234 
235  Double lambda = abs(projVel) + avgV.speedSound()*area;
236  lambda = correctedSpectralRadius(iPoint, jPoint, lambda, stretchParam, solution);
237 
238  /*--- Compute dissipation coefficients. ---*/
239 
240  const auto ni = Base::numNeighbor(iPoint, geometry);
241  const auto nj = Base::numNeighbor(jPoint, geometry);
242  const Double sc2 = 3 * (ni+nj) / (ni*nj);
243  const Double sc4 = 0.25*pow(sc2, 2);
244 
245  const auto si = gatherVariables(iPoint, solution.GetSensor());
246  const auto sj = gatherVariables(jPoint, solution.GetSensor());
247  const Double eps2 = kappa2 * 0.5*(si+sj) * sc2;
248  const Double eps4 = fmax(0.0, kappa4-eps2) * sc4;
249 
250  /*--- Update flux and Jacobians with dissipation terms. ---*/
251 
252  const auto lapl_i = gatherVariables<nVar>(iPoint, solution.GetUndivided_Laplacian());
253  const auto lapl_j = gatherVariables<nVar>(jPoint, solution.GetUndivided_Laplacian());
254 
255  for (size_t iVar = 0; iVar < nVar; ++iVar) {
256  flux(iVar) += (eps2*diffU(iVar) - eps4*(lapl_i(iVar)-lapl_j(iVar))) * lambda;
257  }
258 
259  if (implicit) {
260  const Double dissip_i = fixFactor * (eps2 + eps4*(ni+1)) * lambda;
261  const Double dissip_j = -fixFactor * (eps2 + eps4*(nj+1)) * lambda;
262  scalarDissipationJacobian(V.i, gamma, dissip_i, jac_i);
263  scalarDissipationJacobian(V.j, gamma, dissip_j, jac_j);
264  }
265  }
266 };

296  /*!
297  * \brief Updates flux and Jacobians.
298  * \note "Ts" is here just in case other schemes in the family need extra args.
299  */
300  template<class PrimVarType, class... Ts>
301  FORCEINLINE void finalizeFlux(VectorDbl<nVar>& flux,
302  MatrixDbl<nVar>& jac_i,
303  MatrixDbl<nVar>& jac_j,
304  bool implicit,
305  Double area,
306  Double projVel,
307  const PrimVarType& avgV,
308  const CPair<PrimVarType>& V,
309  const VectorDbl<nVar>& diffU,
310  Int iPoint,
311  Int jPoint,
312  const CGeometry& geometry,
313  const CEulerVariable& solution,
314  const VectorDbl<nDim>& unitNormal,
315  Ts&...) const {
316 
317  /*--- Compute scalar dissipation. ---*/
318 
319  const auto ni = Base::numNeighbor(iPoint, geometry);
320  const auto nj = Base::numNeighbor(jPoint, geometry);
321  const Double sc2 = 3 * (ni+nj) / (ni*nj);
322  const Double sc4 = 0.25*pow(sc2, 2);
323 
324  const auto si = gatherVariables(iPoint, solution.GetSensor());
325  const auto sj = gatherVariables(jPoint, solution.GetSensor());
326  const Double eps2 = kappa2 * 0.5*(si+sj) * sc2;
327  const Double eps4 = fmax(0.0, kappa4-eps2) * sc4;
328 
329  const auto lapl_i = gatherVariables<nVar>(iPoint, solution.GetUndivided_Laplacian());
330  const auto lapl_j = gatherVariables<nVar>(jPoint, solution.GetUndivided_Laplacian());
331 
332  VectorDbl<nVar> scalarDissip;
333  for (size_t iVar = 0; iVar < nVar; ++iVar) {
334  scalarDissip(iVar) = eps2*diffU(iVar) - eps4*(lapl_i(iVar)-lapl_j(iVar));
335  }
336 
337  MatrixDbl<nVar> scalarJac;
338  if (implicit) {
339  scalarJac = Double(0.0);
340  Double factor = fixFactor * (eps2 + 0.5*eps4*(ni+nj+2));
341  scalarDissipationJacobian(avgV, gamma, factor, scalarJac);
342  }
343 
344  /*--- Compute matrix dissipation terms. ---*/
345 
346  const auto unitProjVel = dot(avgV.velocity(), unitNormal);
347 
348  auto pMat = pMatrix(gamma, avgV.density(), avgV.velocity(),
349  unitProjVel, avgV.speedSound(), unitNormal);
350 
351  auto pMatInv = pMatrixInv(gamma, avgV.density(), avgV.velocity(),
352  unitProjVel, avgV.speedSound(), unitNormal);
353 
354  /*--- Compute limited absolute eigenvalues (times area). ---*/
355 
356  VectorDbl<nVar> lambda;
357  for (size_t iDim = 0; iDim < nDim; ++iDim) {
358  lambda(iDim) = projVel;
359  }
360  lambda(nDim) = projVel + avgV.speedSound()*area;
361  lambda(nDim+1) = projVel - avgV.speedSound()*area;
362 
363  const Double maxLambda = fmax(lambda(nDim), -lambda(nDim+1));
364 
365  for (size_t iVar = 0; iVar < nVar; ++iVar) {
366  lambda(iVar) = fmax(abs(lambda(iVar)), entropyFix*maxLambda);
367  }
368 
369  /*--- Update flux and Jacobians with scaled dissipation terms. ---*/
370 
371  for (size_t iVar = 0; iVar < nVar; ++iVar) {
372  for (size_t jVar = 0; jVar < nVar; ++jVar) {
373  /*--- Matrix scaling, P x |S * Lambda| x P^-1. ---*/
374 
375  Double scale = 0.0;
376  for (size_t kVar = 0; kVar < nVar; ++kVar) {
377  scale += pMat(iVar,kVar) * lambda(kVar) * pMatInv(kVar,jVar);
378  }
379 
380  /*--- Update flux and Jacobians. ---*/
381 
382  flux(iVar) += scale * scalarDissip(jVar);
383 
384  if (implicit) {
385  for (size_t kVar = 0; kVar < nVar; ++kVar) {
386  jac_i(iVar,kVar) += scale * scalarJac(jVar,kVar);
387  jac_j(iVar,kVar) -= scale * scalarJac(jVar,kVar);
388  }
389  }
390  }
391  }
392  }
393 };
