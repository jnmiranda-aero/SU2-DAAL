277  /*!
278  * \brief Get the external of all adjoint solvers in a zone.
279  * \param[in] iZone - Index of the zone.
280  * \param[out] rhs - Object with interface (iPoint,iVar), set to -external.
281  */
282  template<class Container>
283  void GetAdjointRHS(unsigned short iZone, Container& rhs) const {
284  const auto nPoint = geometry_container[iZone][INST_0][MESH_0]->GetnPoint();
285  for (auto iSol = 0u, offset = 0u; iSol < MAX_SOLS; ++iSol) {
286  auto solver = solver_container[iZone][INST_0][MESH_0][iSol];
287  if (!(solver && solver->GetAdjoint())) continue;
288  const auto& ext = solver->GetNodes()->Get_External();
289  for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint)
290  for (auto iVar = 0ul; iVar < solver->GetnVar(); ++iVar)
291  rhs(iPoint,offset+iVar) = -SU2_TYPE::GetValue(ext(iPoint,iVar));
292  offset += solver->GetnVar();
293  }
294  }
