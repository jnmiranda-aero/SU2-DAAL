37 CInterpolator* CreateInterpolator(CGeometry**** geometry_container, const CConfig* const* config,
38  const CInterpolator* transpInterpolator, unsigned iZone, unsigned jZone,
39  bool verbose) {
40  CInterpolator* interpolator = nullptr;
41 
42  /*--- Only print information on master node. ---*/
43  verbose &= (SU2_MPI::GetRank() == MASTER_NODE);
44 
45  /*--- Type of interpolation defined by donor. ---*/
46  const auto type = config[iZone]->GetKindInterpolation();
47 
48  if (verbose) cout << " Setting coupling ";
49 
50  /*--- Conservative interpolation is not applicable to the sliding
51  * mesh approach so that case is handled first. Then we either
52  * return a CMirror if the target requires conservative inter-
53  * polation, or the type of interpolator defined by "type". ---*/
54 
55  if (type == INTERFACE_INTERPOLATOR::WEIGHTED_AVERAGE) {
56  if (verbose) cout << "using a sliding mesh approach." << endl;
57  interpolator = new CSlidingMesh(geometry_container, config, iZone, jZone);
58  } else if (config[jZone]->GetConservativeInterpolation()) {
59  if (verbose) cout << "using the mirror approach, \"transposing\" coefficients from opposite mesh." << endl;
60  interpolator = new CMirror(geometry_container, config, transpInterpolator, iZone, jZone);
61  } else {
62  switch (type) {
63  case INTERFACE_INTERPOLATOR::ISOPARAMETRIC:
64  if (verbose) cout << "using the isoparametric approach." << endl;
65  interpolator = new CIsoparametric(geometry_container, config, iZone, jZone);
66  break;
67 
68  case INTERFACE_INTERPOLATOR::NEAREST_NEIGHBOR:
69  if (verbose) cout << "using a nearest neighbor approach." << endl;
70  interpolator = new CNearestNeighbor(geometry_container, config, iZone, jZone);
71  break;
72 
73  case INTERFACE_INTERPOLATOR::RADIAL_BASIS_FUNCTION:
74  if (verbose) cout << "using a radial basis function approach." << endl;
75  interpolator = new CRadialBasisFunction(geometry_container, config, iZone, jZone);
76  break;
77 
78  default:
79  SU2_MPI::Error("Unknown type of interpolation.", CURRENT_FUNCTION);
80  }
81  }
82 
83  if (verbose) interpolator->PrintStatistics();
84 
85  return interpolator;
86 }
