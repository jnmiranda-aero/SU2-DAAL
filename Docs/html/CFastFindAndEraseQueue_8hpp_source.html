36 /*!
37  * \class CFastFindAndEraseQueue
38  * \ingroup Containers
39  * \brief A queue-type container (push back, pop front), but with
40  * fast deletion of arbitrary items (possibly in the middle).
41  * \param[in] ItemType_ - Type of the stored items.
42  * \param[in] ErasedValue_ - Value used to mark items for erasure.
43  * \param[in] CleanupThreshold - Number of marked items that triggers full cleanup.
44  * \note It would not be a good idea to use non-trivial item types.
45  */
46 template <typename ItemType_ = unsigned long, ItemType_ ErasedValue_ = std::numeric_limits<ItemType_>::max(),
47  size_t CleanupThreshold = 8192>
48 class CFastFindAndEraseQueue {
49  public:
50  enum { ErasedValue = ErasedValue_ };
51  using ItemType = ItemType_;
52  using Iterator = typename std::vector<ItemType>::const_iterator;
53 
54  private:
55  size_t erasedCounter = 0; /*!< \brief How many items have been marked since last cleanup. */
56  std::vector<ItemType> items; /*!< \brief The stored items. */
57  std::unordered_map<ItemType, size_t> indexes; /*!< \brief Map items to their location. */
58 
59  /*!
60  * \brief Cleanup, shifts non-erased items forward, re-mapping them.
61  * \note This is the performance hotspot (due to the re-mapping).
62  */
63  void cleanup() {
64  size_t idx = 0;
65  for (auto value : items) {
66  if (value != ErasedValue) {
67  items[idx] = value;
68  indexes[value] = idx;
69  ++idx;
70  }
71  }
72  items.resize(idx);
73  erasedCounter = 0;
74  }
75 
76  public:
77  /*!
78  * \brief Default construct.
79  */
80  CFastFindAndEraseQueue() = default;
81 
82  /*!
83  * \brief Construct and initialize with [0,N) range.
84  */
85  CFastFindAndEraseQueue(size_t N) {
86  items.resize(N);
87  for (size_t i = 0; i < N; ++i) {
88  items[i] = i;
89  indexes[i] = i;
90  }
91  }
92 
93  /*--- Wrapper methods for stl-type operations. ---*/
94  size_t size() const { return indexes.size(); }
95  bool empty() const { return indexes.empty(); }
96  Iterator begin() const { return items.cbegin(); }
97  Iterator end() const { return items.cend(); }
98 
99  /*!
100  * \brief Push back an item, that should be unique.
101  * \param[in] value - The new item.
102  */
103  void push_back(ItemType value) {
104  assert(value != ErasedValue && "#bad_idea");
105  assert(!indexes.count(value) && "Items must be unique.");
106  indexes[value] = items.size();
107  items.push_back(value);
108  }
109 
110  /*!
111  * \brief Get the item in front of the queue.
112  */
113  ItemType front() const {
114  for (auto value : items)
115  if (value != ErasedValue) return value;
116  assert(false && "Queue is empty.");
117  return ErasedValue;
118  }
119 
120  /*!
121  * \brief Main event, look for an item and erase it if present.
122  * \param[in] value - Item to look for.
123  * \return True, if something was erased, false otherwise.
124  */
125  bool findAndErase(const ItemType value) {
126  if (empty()) return false;
127 
128  /*--- Get the item position. ---*/
129  auto it = indexes.find(value);
130  if (it == indexes.end()) return false; // value is not here
131 
132  const auto idx = it->second;
133  assert(items[idx] != ErasedValue && "The item was already erased?!");
134 
135  /*--- Erase and cleanup items. ---*/
136  indexes.erase(it);
137  items[idx] = ErasedValue;
138  if (++erasedCounter == CleanupThreshold) cleanup();
139 
140  return true;
141  }
142 };
