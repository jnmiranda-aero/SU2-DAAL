1649  void SetMultiGridMarkerQuantity(const CGeometry* fine_grid, unsigned short val_marker, T& wall_quantity) {
1650  for (auto iVertex = 0ul; iVertex < nVertex[val_marker]; iVertex++) {
1651  const auto Point_Coarse = vertex[val_marker][iVertex]->GetNode();
1652 
1653  if (!nodes->GetDomain(Point_Coarse)) continue;
1654 
1655  su2double Area_Parent = 0.0;
1656 
1657  /*--- Compute area parent by taking into account only volumes that are on the marker. ---*/
1658  for (auto iChildren = 0u; iChildren < nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
1659  const auto Point_Fine = nodes->GetChildren_CV(Point_Coarse, iChildren);
1660  const auto isVertex =
1661  fine_grid->nodes->GetDomain(Point_Fine) && (fine_grid->nodes->GetVertex(Point_Fine, val_marker) != -1);
1662  if (isVertex) {
1663  Area_Parent += fine_grid->nodes->GetVolume(Point_Fine);
1664  }
1665  }
1666 
1667  su2double Quantity_Coarse = 0.0;
