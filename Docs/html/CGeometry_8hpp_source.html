573 * \return Index of the edge.
574 */
575 inline long FindEdge(unsigned long first_point, unsigned long second_point, bool error = true) const {
576 for (unsigned short iNode = 0; iNode < nodes->GetnPoint(first_point); iNode++) {
577 auto iPoint = nodes->GetPoint(first_point, iNode);
578 if (iPoint == second_point) return nodes->GetEdge(first_point, iNode);
579 }
580 if (error) {
581 char buf[100];
582 SPRINTF(buf, "Can't find the edge that connects %lu and %lu.", first_point, second_point);
583 SU2_MPI::Error(buf, CURRENT_FUNCTION);
584 }
585 return -1;
586 }
