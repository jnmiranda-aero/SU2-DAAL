43 su2double CBezierBlending::GetBernstein(short val_n, short val_i, su2double val_t) {
44  su2double value = 0.0;
45 
46  if (val_i > val_n) {
47  value = 0.0;
48  return value;
49  }
50 
51  if (val_i == 0) {
52  if (val_t == 0)
53  value = 1.0;
54  else if (val_t == 1)
55  value = 0.0;
56  else
57  value = Binomial(val_n, val_i) * pow(val_t, val_i) * pow(1.0 - val_t, val_n - val_i);
58  } else if (val_i == val_n) {
59  if (val_t == 0)
60  value = 0.0;
61  else if (val_t == 1)
62  value = 1.0;
63  else
64  value = pow(val_t, val_n);
65  } else {
66  if ((val_t == 0) || (val_t == 1)) value = 0.0;
67  value = Binomial(val_n, val_i) * pow(val_t, val_i) * pow(1.0 - val_t, val_n - val_i);
68  }
69 
70  return value;
71 }

77 su2double CBezierBlending::GetBernsteinDerivative(short val_n, short val_i, su2double val_t, short val_order_der) {
78  su2double value = 0.0;
79 
80  /*--- Verify this subroutine, it provides negative val_n,
81  which is a wrong value for GetBernstein ---*/
82 
83  if (val_order_der == 0) {
84  value = GetBernstein(val_n, val_i, val_t);
85  return value;
86  }
87 
88  if (val_i == 0) {
89  value = val_n * (-GetBernsteinDerivative(val_n - 1, val_i, val_t, val_order_der - 1));
90  return value;
91  }
92  if (val_n == 0) {
93  value = val_t;
94  return value;
95  } else {
96  value = val_n * (GetBernsteinDerivative(val_n - 1, val_i - 1, val_t, val_order_der - 1) -
97  GetBernsteinDerivative(val_n - 1, val_i, val_t, val_order_der - 1));
98  return value;
99  }
100 
101  return value;
102 }

104 su2double CBezierBlending::Binomial(unsigned short n, unsigned short m) {
105  unsigned short i, j;
106  su2double result;
107 
108  binomial[0] = 1.0;
109  for (i = 1; i <= n; ++i) {
110  binomial[i] = 1.0;
111  for (j = i - 1U; j > 0; --j) {
112  binomial[j] += binomial[j - 1U];
113  }
114  }
115 
116  result = binomial[m];
117  if (fabs(result) < EPS * EPS) {
118  result = 0.0;
119  }
120 
121  return result;
122 }
