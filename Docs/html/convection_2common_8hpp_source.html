35/*!
36 * \brief Unlimited reconstruction.

38template<size_t nVar, size_t nDim, class Gradient_t>
39FORCEINLINE void musclUnlimited(Int iPoint,
40 const VectorDbl<nDim>& vector_ij,
41 Double scale,
42 const Gradient_t& gradient,
43 VectorDbl<nVar>& vars) {
44 auto grad = gatherVariables<nVar,nDim>(iPoint, gradient);
45 for (size_t iVar = 0; iVar < nVar; ++iVar) {
46 vars(iVar) += scale * dot(grad[iVar], vector_ij);
47 }
48}

50/*!
51 * \brief Limited reconstruction with point-based limiter.
52 */
53template<size_t nVar, size_t nDim, class Limiter_t, class Gradient_t>
54FORCEINLINE void musclPointLimited(Int iPoint,
55 const VectorDbl<nDim>& vector_ij,
56 Double scale,
57 const Limiter_t& limiter,
58 const Gradient_t& gradient,
59 VectorDbl<nVar>& vars) {
60 auto lim = gatherVariables<nVar>(iPoint, limiter);
61 auto grad = gatherVariables<nVar,nDim>(iPoint, gradient);
62 for (size_t iVar = 0; iVar < nVar; ++iVar) {
63 vars(iVar) += lim(iVar) * scale * dot(grad[iVar], vector_ij);
64 }
65}

67/*!
68 * \brief Limited reconstruction with edge-based limiter.
69 */
70template<size_t nDim, class VarType, class Gradient_t>
71FORCEINLINE void musclEdgeLimited(Int iPoint,
72 Int jPoint,
73 const VectorDbl<nDim>& vector_ij,
74 const Gradient_t& gradient,
75 CPair<VarType>& V) {
76 constexpr size_t nVar = VarType::nVar;
77
78 auto grad_i = gatherVariables<nVar,nDim>(iPoint, gradient);
79 auto grad_j = gatherVariables<nVar,nDim>(jPoint, gradient);
80
81 for (size_t iVar = 0; iVar < nVar; ++iVar) {
82 const Double proj_i = dot(grad_i[iVar], vector_ij);
83 const Double proj_j = dot(grad_j[iVar], vector_ij);
84 const Double delta_ij = V.j.all(iVar) - V.i.all(iVar);
85 const Double delta_ij_2 = pow(delta_ij, 2) + 1e-6;
86 /// TODO: Customize the limiter function.
87 const Double lim_i = (delta_ij_2 + proj_i*delta_ij) / (pow(proj_i,2) + delta_ij_2);
88 const Double lim_j = (delta_ij_2 + proj_j*delta_ij) / (pow(proj_j,2) + delta_ij_2);
89 V.i.all(iVar) += lim_i * 0.5 * proj_i;
90 V.j.all(iVar) -= lim_j * 0.5 * proj_j;
91 }
92}

94/*!
95 * \brief Retrieve primitive variables for points i/j, reconstructing them if needed.
96 * \param[in] iEdge, iPoint, jPoint - Edge and its nodes.
97 * \param[in] muscl - If true, reconstruct, else simply fetch.
98 * \param[in] V1st - Pair of compressible flow primitives for nodes i,j.
99 * \param[in] vector_ij - Distance vector from i to j.
100 * \param[in] solution - Entire solution container (a derived CVariable).
101 * \return Pair of primitive variables.
102 */
103template<class ReconVarType, class PrimVarType, size_t nDim, class VariableType>
104FORCEINLINE CPair<ReconVarType> reconstructPrimitives(Int iEdge, Int iPoint, Int jPoint,
105 bool muscl, LIMITER limiterType,
106 const CPair<PrimVarType>& V1st,
107 const VectorDbl<nDim>& vector_ij,
108 const VariableType& solution) {
109 static_assert(ReconVarType::nVar <= PrimVarType::nVar,"");
110
111 const auto& gradients = solution.GetGradient_Reconstruction();
112 const auto& limiters = solution.GetLimiter_Primitive();
113
114 CPair<ReconVarType> V;
115
116 for (size_t iVar = 0; iVar < ReconVarType::nVar; ++iVar) {
117 V.i.all(iVar) = V1st.i.all(iVar);
118 V.j.all(iVar) = V1st.j.all(iVar);
119 }
120
121 if (muscl) {
122 switch (limiterType) {
123 case LIMITER::NONE:
124 musclUnlimited(iPoint, vector_ij, 0.5, gradients, V.i.all);
125 musclUnlimited(jPoint, vector_ij,-0.5, gradients, V.j.all);
126 break;
127 case LIMITER::VAN_ALBADA_EDGE:
128 musclEdgeLimited(iPoint, jPoint, vector_ij, gradients, V);
129 break;
130 default:
131 musclPointLimited(iPoint, vector_ij, 0.5, limiters, gradients, V.i.all);
132 musclPointLimited(jPoint, vector_ij,-0.5, limiters, gradients, V.j.all);
133 break;
134 }
135 /*--- Detect a non-physical reconstruction based on negative pressure or density. ---*/
136 const Double neg_p_or_rho = fmax(fmin(V.i.pressure(), V.j.pressure()) < 0.0,
137 fmin(V.i.density(), V.j.density()) < 0.0);
138 /*--- Test the sign of the Roe-averaged speed of sound. ---*/
139 const Double R = sqrt(abs(V.j.density() / V.i.density()));
140 /*--- Delay dividing by R+1 until comparing enthalpy and velocity magnitude. ---*/
141 const Double enthalpy = R*V.j.enthalpy() + V.i.enthalpy();
142 Double v_squared = 0.0;
143 for (size_t iDim = 0; iDim < nDim; ++iDim) {
144 v_squared += pow(R*V.j.velocity(iDim) + V.i.velocity(iDim), 2);
145 }
146 /*--- Multiply enthalpy by R+1 since v^2 was not divided by (R+1)^2.
147 * Note: a = sqrt((gamma-1) * (H - 0.5 * v^2)) ---*/
148 const Double neg_sound_speed = enthalpy * (R+1) < 0.5 * v_squared;
149
150 /*--- Revert to first order if the state is non-physical. ---*/
151 Double bad_recon = fmax(neg_p_or_rho, neg_sound_speed);
152 /*--- Handle SIMD dimensions 1 by 1. ---*/
153 for (size_t k = 0; k < Double::Size; ++k) {
154 bad_recon[k] = solution.UpdateNonPhysicalEdgeCounter(iEdge[k], bad_recon[k]);
155 }
156 for (size_t iVar = 0; iVar < ReconVarType::nVar; ++iVar) {
157 V.i.all(iVar) = bad_recon * V1st.i.all(iVar) + (1-bad_recon) * V.i.all(iVar);
158 V.j.all(iVar) = bad_recon * V1st.j.all(iVar) + (1-bad_recon) * V.j.all(iVar);
159 }
160 }
161 return V;
162}

164/*!
165 * \brief Compute and return the P tensor (compressible flow, ideal gas).
166 */
167template<size_t nDim, class RandomAccessIterator>
168FORCEINLINE MatrixDbl<nDim+2> pMatrix(Double gamma, Double density, const RandomAccessIterator& velocity,
169 Double projVel, Double speedSound, const VectorDbl<nDim>& normal) {
170 MatrixDbl<nDim+2> pMat;
171 const Double vel2 = 0.5*squaredNorm<nDim>(velocity);
172
173 if (nDim == 2) {
174 pMat(0,0) = 1.0;
175 pMat(0,1) = 0.0;
176
177 pMat(1,0) = velocity[0];
178 pMat(1,1) = density*normal(1);
179
180 pMat(2,0) = velocity[1];
181 pMat(2,1) = -density*normal(0);
182
183 pMat(3,0) = vel2;
184 pMat(3,1) = density*(velocity[0]*normal(1) - velocity[1]*normal(0));
185 }
186 else {
187 pMat(0,0) = normal(0);
188 pMat(0,1) = normal(1);
189 pMat(0,2) = normal(2);
190
191 pMat(1,0) = velocity[0]*normal(0);
192 pMat(1,1) = velocity[0]*normal(1) - density*normal(2);
193 pMat(1,2) = velocity[0]*normal(2) + density*normal(1);
194
195 pMat(2,0) = velocity[1]*normal(0) + density*normal(2);
196 pMat(2,1) = velocity[1]*normal(1);
197 pMat(2,2) = velocity[1]*normal(2) - density*normal(0);
198
199 pMat(3,0) = velocity[2]*normal(0) - density*normal(1);
200 pMat(3,1) = velocity[2]*normal(1) + density*normal(0);
201 pMat(3,2) = velocity[2]*normal(2);
202
203 pMat(4,0) = vel2*normal(0) + density*(velocity[1]*normal(2) - velocity[2]*normal(1));
204 pMat(4,1) = vel2*normal(1) - density*(velocity[0]*normal(2) - velocity[2]*normal(0));
205 pMat(4,2) = vel2*normal(2) + density*(velocity[0]*normal(1) - velocity[1]*normal(0));
206 }
207
208 /*--- Last two columns. ---*/
209
210 const Double rhoOn2 = 0.5*density;
211 const Double rhoOnTwoC = rhoOn2 / speedSound;
212 pMat(0,nDim) = rhoOnTwoC;
213 pMat(0,nDim+1) = rhoOnTwoC;
214
215 for (size_t iDim = 0; iDim < nDim; ++iDim) {
216 pMat(iDim+1,nDim) = rhoOnTwoC * velocity[iDim] + rhoOn2 * normal(iDim);
217 pMat(iDim+1,nDim+1) = rhoOnTwoC * velocity[iDim] - rhoOn2 * normal(iDim);
218 }
219
220 pMat(nDim+1,nDim) = rhoOnTwoC * vel2 + rhoOn2 * (projVel + speedSound/(gamma-1));
221 pMat(nDim+1,nDim+1) = rhoOnTwoC * vel2 - rhoOn2 * (projVel - speedSound/(gamma-1));
222
223 return pMat;
224}

226/*!
227 * \brief Compute and return the inverse P tensor (compressible flow, ideal gas).
228 */
229template<size_t nDim, class RandomAccessIterator>
230FORCEINLINE MatrixDbl<nDim+2> pMatrixInv(Double gamma, Double density, const RandomAccessIterator& velocity,
231 Double projVel, Double speedSound, const VectorDbl<nDim>& normal) {
232 MatrixDbl<nDim+2> pMatInv;
233
234 const Double c2 = pow(speedSound,2);
235 const Double vel2 = 0.5*squaredNorm<nDim>(velocity);
236 const Double oneOnRho = 1 / density;
237
238 if (nDim == 2) {
239 Double tmp = (gamma-1)/c2;
240 pMatInv(0,0) = 1.0 - tmp*vel2;
241 pMatInv(0,1) = tmp*velocity[0];
242 pMatInv(0,2) = tmp*velocity[1];
243 pMatInv(0,3) = -tmp;
244
245 pMatInv(1,0) = (normal(0)*velocity[1]-normal(1)*velocity[0])*oneOnRho;
246 pMatInv(1,1) = normal(1)*oneOnRho;
247 pMatInv(1,2) = -normal(0)*oneOnRho;
248 pMatInv(1,3) = 0.0;
249 }
250 else {
251 Double tmp = (gamma-1)/c2 * normal(0);
252 pMatInv(0,0) = normal(0) - tmp*vel2 - (normal(2)*velocity[1]-normal(1)*velocity[2])*oneOnRho;
253 pMatInv(0,1) = tmp*velocity[0];
254 pMatInv(0,2) = tmp*velocity[1] + normal(2)*oneOnRho;
255 pMatInv(0,3) = tmp*velocity[2] - normal(1)*oneOnRho;
256 pMatInv(0,4) = -tmp;
257
258 tmp = (gamma-1)/c2 * normal(1);
259 pMatInv(1,0) = normal(1) - tmp*vel2 + (normal(2)*velocity[0]-normal(0)*velocity[2])*oneOnRho;
260 pMatInv(1,1) = tmp*velocity[0] - normal(2)*oneOnRho;
261 pMatInv(1,2) = tmp*velocity[1];
262 pMatInv(1,3) = tmp*velocity[2] + normal(0)*oneOnRho;
263 pMatInv(1,4) = -tmp;
264
265 tmp = (gamma-1)/c2 * normal(2);
266 pMatInv(2,0) = normal(2) - tmp*vel2 - (normal(1)*velocity[0]-normal(0)*velocity[1])*oneOnRho;
267 pMatInv(2,1) = tmp*velocity[0] + normal(1)*oneOnRho;
268 pMatInv(2,2) = tmp*velocity[1] - normal(0)*oneOnRho;
269 pMatInv(2,3) = tmp*velocity[2];
270 pMatInv(2,4) = -tmp;
271 }
272
273 /*--- Last two rows. ---*/
274
275 const Double gamma_minus_1_on_rho_times_c = (gamma-1) / (density*speedSound);
276
277 for (size_t iVar = nDim; iVar < nDim+2; ++iVar) {
278 Double sign = (iVar==nDim)? 1 : -1;
279 pMatInv(iVar,0) = -sign*projVel*oneOnRho + gamma_minus_1_on_rho_times_c * vel2;
280 for (size_t iDim = 0; iDim < nDim; ++iDim) {
281 pMatInv(iVar,iDim+1) = sign*normal(iDim)*oneOnRho - gamma_minus_1_on_rho_times_c * velocity[iDim];
282 }
283 pMatInv(iVar,nDim+1) = gamma_minus_1_on_rho_times_c;
284 }
285
286 return pMatInv;
287}

289/*!
290 * \brief Convective projected (onto normal) flux (compressible flow).
291 */
292template<class PrimVarType, class ConsVarType, size_t nDim>
293FORCEINLINE VectorDbl<nDim+2> inviscidProjFlux(const PrimVarType& V,
294 const ConsVarType& U,
295 const VectorDbl<nDim>& normal) {
296 static_assert(ConsVarType::nVar == nDim+2,"");
297 Double mdot = dot(U.momentum(), normal);
298 VectorDbl<nDim+2> flux;
299 flux(0) = mdot;
300 for (size_t iDim = 0; iDim < nDim; ++iDim) {
301 flux(iDim+1) = mdot*V.velocity(iDim) + normal(iDim)*V.pressure();
302 }
303 flux(nDim+1) = mdot*V.enthalpy();
304 return flux;
305}

307/*!
308 * \brief Jacobian of the convective flux (compressible flow, ideal gas).
309 */
310template<size_t nDim, class RandomAccessIterator>
311FORCEINLINE MatrixDbl<nDim+2> inviscidProjJac(Double gamma, RandomAccessIterator velocity,
312 Double energy, const VectorDbl<nDim>& normal,
313 Double scale) {
314 MatrixDbl<nDim+2> jac;
315
316 Double projVel = dot(velocity, normal);
317 Double gamma_m_1 = gamma-1;
318 Double phi = 0.5*gamma_m_1*squaredNorm<nDim>(velocity);
319 Double a1 = gamma*energy - phi;
320
321 jac(0,0) = 0.0;
322 for (size_t iDim = 0; iDim < nDim; ++iDim) {
323 jac(0,iDim+1) = scale * normal(iDim);
324 }
325 jac(0,nDim+1) = 0.0;
326
327 for (size_t iDim = 0; iDim < nDim; ++iDim) {
328 jac(iDim+1,0) = scale * (normal(iDim)*phi - velocity[iDim]*projVel);
329 for (size_t jDim = 0; jDim < nDim; ++jDim) {
330 jac(iDim+1,jDim+1) = scale * (normal(jDim)*velocity[iDim] - gamma_m_1*normal(iDim)*velocity[jDim]);
331 }
332 jac(iDim+1,iDim+1) += scale * projVel;
333 jac(iDim+1,nDim+1) = scale * gamma_m_1 * normal(iDim);
334 }
335
336 jac(nDim+1,0) = scale * projVel * (phi-a1);
337 for (size_t iDim = 0; iDim < nDim; ++iDim) {
338 jac(nDim+1,iDim+1) = scale * (normal(iDim)*a1 - gamma_m_1*velocity[iDim]*projVel);
339 }
340 jac(nDim+1,nDim+1) = scale * gamma * projVel;
341
342 return jac;
343}

345/*!
346 * \brief (Low) Dissipation coefficient for Roe schemes.
347 */
348template<class VariableType>
349FORCEINLINE Double roeDissipation(Int iPoint,
350 Int jPoint,
351 ENUM_ROELOWDISS type,
352 const VariableType& solution) {
353 if (type == NO_ROELOWDISS) {
354 return 1.0;
355 }
356
357 const auto& sol = static_cast<const CNSVariable&>(solution);
358 const auto& sensor = sol.GetSensor();
359 const auto& dissip = sol.GetRoe_Dissipation();
360
361 const Double si = gatherVariables(iPoint, sensor);
362 const Double sj = gatherVariables(jPoint, sensor);
363 const Double avgSensor = 0.5 * (si + sj);
364
365 const Double di = gatherVariables(iPoint, dissip);
366 const Double dj = gatherVariables(jPoint, dissip);
367 const Double avgDissip = 0.5 * (di + dj);
368
369 /*--- A minimum level of upwinding is used to enhance stability. ---*/
370 constexpr passivedouble minDissip = 0.05;
371
372 switch (type) {
373 case FD:
374 case FD_DUCROS: {
375 Double d = fmax(minDissip, 1.0 - avgDissip);
376
377 if (type == FD_DUCROS) {
378 /*--- See Jonhsen et al. JCP 229 (2010) pag. 1234 ---*/
379 d = fmax(d, 0.05 + 0.95*(avgSensor > 0.65));
380 }
381 return d;
382 }
383 case NTS:
384 return fmax(minDissip, avgDissip);
385
386 case NTS_DUCROS:
387 /*--- See Xiao et al. INT J HEAT FLUID FL 51 (2015) pag. 141
388 * https://doi.org/10.1016/j.ijheatfluidflow.2014.10.007 ---*/
389 return fmax(minDissip, avgSensor+avgDissip - avgSensor*avgDissip);
390
391 default:
392 assert(false);
393 return 1.0;
394 }
395}

397/*!
398 * \brief Correct spectral radius (avgLambda) for stretching.
399 */
400template<class VariableType, class T>
401FORCEINLINE Double correctedSpectralRadius(Int iPoint,
402 Int jPoint,
403 Double avgLambda,
404 T stretchParam,
405 const VariableType& solution) {
406
407 const auto lambda_i = gatherVariables(iPoint, solution.GetLambda());
408 const Double phi_i = pow(0.25*lambda_i/avgLambda, stretchParam);
409
410 const auto lambda_j = gatherVariables(jPoint, solution.GetLambda());
411 const Double phi_j = pow(0.25*lambda_j/avgLambda, stretchParam);
412
413 return 4*phi_i*phi_j / (phi_i + phi_j) * avgLambda;
414}

416/*!
417 * \brief Update of a flux Jacobian due to a scalar dissipation term.
418 */
419template<class VariableType, size_t nVar>
420FORCEINLINE void scalarDissipationJacobian(const VariableType& V,
421 Double gamma,
422 Double dissipConst,
423 MatrixDbl<nVar>& jac) {
424 /*--- Diagonal entries. ---*/
425 for (size_t iVar = 0; iVar < nVar-1; ++iVar) {
426 jac(iVar,iVar) += dissipConst;
427 }
428 jac(nVar-1,nVar-1) += dissipConst * gamma;
429
430 /*--- N-1 columns of last row. ---*/
431 dissipConst *= gamma-1.0;
432 for (size_t iDim = 0; iDim < VariableType::nDim; ++iDim) {
433 jac(nVar-1,iDim+1) -= dissipConst * V.velocity(iDim);
434 jac(nVar-1,0) += dissipConst * pow(V.velocity(iDim), 2);
435 }
