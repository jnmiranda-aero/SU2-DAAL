33CHeatOutput::CHeatOutput(CConfig *config, unsigned short nDim) : CFVMOutput(config, nDim, false) {
34
35 /*--- Set the default history fields if nothing is set in the config file ---*/
36
37 if (nRequestedHistoryFields == 0){
38 RequestCommonHistory(config->GetTime_Domain());
39 nRequestedHistoryFields = requestedHistoryFields.size();
40 }
41 if (nRequestedScreenFields == 0){
42 requestedScreenFields.emplace_back("OUTER_ITER");
43 requestedScreenFields.emplace_back("INNER_ITER");
44 requestedScreenFields.emplace_back("RMS_TEMPERATURE");
45 nRequestedScreenFields = requestedScreenFields.size();
46 }
47 if (nRequestedVolumeFields == 0){
48 requestedVolumeFields.emplace_back("COORDINATES");
49 requestedVolumeFields.emplace_back("SOLUTION");
50 nRequestedVolumeFields = requestedVolumeFields.size();
51 }
52
53 stringstream ss;
54 ss << "Zone " << config->GetiZone() << " (Solid Heat)";
55 multiZoneHeaderString = ss.str();
56
57 /*--- Set the volume filename --- */
58
59 volumeFilename = config->GetVolume_FileName();
60
61 /*--- Set the surface filename --- */
62
63 surfaceFilename = config->GetSurfCoeff_FileName();
64
65 /*--- Set the restart filename --- */
66
67 restartFilename = config->GetRestart_FileName();
68
69 /*--- Set the default convergence field --- */
70
71 if (convFields.empty() ) convFields.emplace_back("RMS_TEMPERATURE");
72
73}

75void CHeatOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
76
77 CSolver* heat_solver = solver[HEAT_SOL];
78
79 SetHistoryOutputValue("TOTAL_HEATFLUX", heat_solver->GetTotal_HeatFlux());
80 SetHistoryOutputValue("MAXIMUM_HEATFLUX", heat_solver->GetTotal_MaxHeatFlux());
81 SetHistoryOutputValue("AVG_TEMPERATURE", heat_solver->GetTotal_AvgTemperature());
82 SetHistoryOutputValue("RMS_TEMPERATURE", log10(heat_solver->GetRes_RMS(0)));
83 SetHistoryOutputValue("MAX_TEMPERATURE", log10(heat_solver->GetRes_Max(0)));
84 if (multiZone)
85 SetHistoryOutputValue("BGS_TEMPERATURE", log10(heat_solver->GetRes_BGS(0)));
86
87 SetHistoryOutputValue("LINSOL_ITER", heat_solver->GetIterLinSolver());
88 SetHistoryOutputValue("LINSOL_RESIDUAL", log10(heat_solver->GetResLinSolver()));
89 SetHistoryOutputValue("CFL_NUMBER", config->GetCFL(MESH_0));
90
91 ComputeSimpleCustomOutputs(config);
92
93 /*--- Keep this as last, since it uses the history values that were set. ---*/
94 SetCustomAndComboObjectives(HEAT_SOL, config, solver);
95}

134void CHeatOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
135
136 CVariable* Node_Heat = solver[HEAT_SOL]->GetNodes();
137 const auto Node_Geo = geometry->nodes;
138
139 // Grid coordinates
140 LoadCoordinates(Node_Geo->GetCoord(iPoint), iPoint);
141
142 // SOLUTION
143 SetVolumeOutputValue("TEMPERATURE", iPoint, Node_Heat->GetSolution(iPoint, 0));
144
145 // Residuals
146 SetVolumeOutputValue("RES_TEMPERATURE", iPoint, solver[HEAT_SOL]->LinSysRes(iPoint, 0));
147
148 LoadCommonFVMOutputs(config, geometry, iPoint);
149}
