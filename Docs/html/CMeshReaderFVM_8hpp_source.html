40 * \author T. Economon
41 */
42class CMeshReaderFVM {
43 protected:
44 const int rank; /*!< \brief MPI Rank. */
45 const int size; /*!< \brief MPI Size. */
46
47 const CConfig* config = nullptr; /*!< \brief Local pointer to the config parameter object. */
48
49 unsigned short dimension = 0; /*!< \brief Dimension of the problem (2 or 3). */
50
51 unsigned long numberOfLocalPoints =
52 0; /*!< \brief Number of local grid points within the linear partition on this rank. */
53 unsigned long numberOfGlobalPoints = 0; /*!< \brief Number of global grid points within the mesh file. */
54 vector<vector<passivedouble> >
55 localPointCoordinates; /*!< \brief Vector holding the coordinates from the mesh file for the local grid points.
56 First index is dimension, second is point index. */
57
58 unsigned long numberOfLocalElements =
59 0; /*!< \brief Number of local elements within the linear partition on this rank. */
60 unsigned long numberOfGlobalElements = 0; /*!< \brief Number of global elements within the mesh file. */
61 vector<unsigned long> localVolumeElementConnectivity; /*!< \brief Vector containing the element connectivity from the

63
64 unsigned long numberOfMarkers = 0; /*!< \brief Total number of markers contained within the mesh file. */
65 vector<string> markerNames; /*!< \brief String names for all markers in the mesh file. */
66 vector<vector<unsigned long> >

68 per-marker basis. Only the master node reads and stores this connectivity. */
69
70 public:
71 /*!
72 * \brief Constructor of the CMeshReaderFVM class.
73 * \param[in] val_config - config object for the current zone.
74 * \param[in] val_iZone - Current zone index.
75 * \param[in] val_nZone - Total number of zones.

77 CMeshReaderFVM(const CConfig* val_config, unsigned short val_iZone, unsigned short val_nZone);

83 * \returns Physical dimension of the problem.

85 inline unsigned short GetDimension() const { return dimension; }
86
87 /*!
88 * \brief Get the local point coordinates (linearly partitioned).
89 * \returns Local point coordinates (linear partitioned).

92
93 /*!
94 * \brief Get the surface element connectivity for the specified marker. Only the master node owns the surface
95 * connectivity. \param[in] val_iMarker - current marker index. \returns Surface element connecitivity for a marker
96 * from the master rank.
97 */
98 inline const vector<unsigned long>& GetSurfaceElementConnectivityForMarker(int val_iMarker) const {
99 return surfaceElementConnectivity[val_iMarker];
100 }
101
102 /*!
103 * \brief Get the number surface elements for the specified marker.
104 * \param[in] val_iMarker - current marker index.
105 * \returns Number of surface elements for a marker.
106 */
107 inline unsigned long GetNumberOfSurfaceElementsForMarker(int val_iMarker) const {
108 return (unsigned long)surfaceElementConnectivity[val_iMarker].size() / SU2_CONN_SIZE;
109 }
110
111 /*!
112 * \brief Get the local volume element connectivity (linearly partitioned).

118

152 */
153 inline unsigned long GetNumberOfGlobalElements() const { return numberOfGlobalElements; }
