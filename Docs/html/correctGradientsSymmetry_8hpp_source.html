37 /*!
38  * \brief Correct the gradient on a symmetry plane.
39  * \ingroup FvmAlgos
40  * \param[in] varBegin - Start of the variables.
41  * \param[in] varEnd - End of the variables.
42  * \param[in] idxVel - Variable index where velocity gradients start (-1 if all variables are scalars).
43  * \param[in] n - Normal direction.
44  * \param[in, out] gradients - The gradients to be modified.
45  */
46 template <size_t nDim, class Matrix, class Scalar>
47 inline void correctGradient(const size_t varBegin, const size_t varEnd, const int idxVel, const Scalar* n,
48  Matrix&& gradients) {
49 
50  /*--- First we correct the part that involves velocities. ---*/
51 
52  if (idxVel >= static_cast<int>(varBegin) && idxVel + nDim <= varEnd) {
53  /*--- Normal gradient of velocity components and gradient of normal velocity. ---*/
54  su2double normalGrad[nDim] = {}, gradNormalVel[nDim] = {};
55 
56  for (size_t iDim = 0; iDim < nDim; iDim++) {
57  normalGrad[iDim] = GeometryToolbox::DotProduct(nDim, gradients[idxVel + iDim], n);
58 
59  for (size_t jDim = 0; jDim < nDim; jDim++) {
60  gradNormalVel[jDim] += n[iDim] * gradients[idxVel + iDim][jDim];
61  }
62  }
63 
64  /*--- Normal gradient of the normal velocity. ---*/
65  const su2double normalGradNormalVel = GeometryToolbox::DotProduct(nDim, n, gradNormalVel);
66 
67  /*--- Remove the tangential projection (I - n.n^T) of the normal gradients.
68  * And the normal projection (n.n^T) of the tangential gradients.
69  * dV = dV - (I - n.n^T) dV n.n^T - n.n^T dV (I - n.n^T) ---*/
70 
71  for (size_t iDim = 0; iDim < nDim; iDim++) {
72  for (size_t jDim = 0; jDim < nDim; jDim++) {
73  gradients[idxVel + iDim][jDim] -= normalGrad[iDim] * n[jDim] + n[iDim] * gradNormalVel[jDim];
74  gradients[idxVel + iDim][jDim] += 2 * n[iDim] * normalGradNormalVel * n[jDim];
75  }
76  }
77  }
78 
79  /*--- Remove the normal component for all scalars (excluding velocities). ---*/
80 
81  for (auto iVar = varBegin; iVar < varEnd; iVar++) {
82  if (idxVel != -1 && static_cast<int>(iVar) >= idxVel && iVar < idxVel + nDim) continue;
83 
84  const su2double normalGrad = GeometryToolbox::DotProduct(nDim, n, gradients[iVar]);
85  for (size_t iDim = 0; iDim < nDim; iDim++) {
86  gradients[iVar][iDim] -= normalGrad * n[iDim];
87  }
88  }
89 }
90 }

92 /*!
93  * \brief Correct gradients on symmetry and Euler (slip) markers to respect the conditions:
94  * 1. n.grad(phi) = 0
95  * 2. n.grad(v.t) = 0
96  * 3. t.grad(v.n) = 0
97  * \note See Blazek eq. 8.40.
98  * \ingroup FvmAlgos
99  * \param[in] geometry - Geometric grid properties.
100  * \param[in] config - Configuration of the problem, used to identify types of boundaries.
101  * \param[in] varBegin - Index of first variable for which to compute the gradient.
102  * \param[in] varEnd - Index of last variable for which to compute the gradient.
103  * \param[in] idxVel - Index to velocity, -1 if no velocity is present in the solver.
104  * \param[in,out] gradient - Generic object implementing operator (iPoint, iVar, iDim).
105  */
106 template <size_t nDim, class GradientType>
107 void correctGradientsSymmetry(CGeometry& geometry, const CConfig& config, const size_t varBegin,
108  const size_t varEnd, const int idxVel, GradientType& gradient) {
109 
110  /*--- Check how many symmetry planes there are. ---*/
111  std::vector<unsigned short> symMarkers;
112  for (auto iMarker = 0u; iMarker < geometry.GetnMarker(); ++iMarker) {
113  if (config.GetMarker_All_KindBC(iMarker) == SYMMETRY_PLANE ||
114  config.GetMarker_All_KindBC(iMarker) == EULER_WALL) {
115  symMarkers.push_back(iMarker);
116  }
117  }
118 
119  for (const auto iMarker : symMarkers) {
120  SU2_OMP_FOR_STAT(32)
121  for (size_t iVertex = 0; iVertex < geometry.GetnVertex(iMarker); ++iVertex) {
122 
123  const auto iPoint = geometry.vertex[iMarker][iVertex]->GetNode();
124 
125  /*--- Get the normal of the current symmetry. This may be the original normal of the vertex
126  * or a modified normal if there are intersecting symmetries. ---*/
127 
128  su2double unitNormal[nDim] = {};
129  const auto it = geometry.symmetryNormals[iMarker].find(iVertex);
130 
131  if (it != geometry.symmetryNormals[iMarker].end()) {
132  for (auto iDim = 0u; iDim < nDim; iDim++) unitNormal[iDim] = it->second[iDim];
133  } else {
134  geometry.vertex[iMarker][iVertex]->GetNormal(unitNormal);
135  const su2double area = GeometryToolbox::Norm(nDim, unitNormal);
136  for (auto iDim = 0u; iDim < nDim; iDim++) unitNormal[iDim] /= area;
137  }
138 
139  detail::correctGradient<nDim>(varBegin, varEnd, idxVel, unitNormal, gradient[iPoint]);
140 
141  } // loop over vertices
142  END_SU2_OMP_FOR
143  } // markers
144 
145 }
