32 template <class VariableType>
33 CScalarSolver<VariableType>::CScalarSolver(CGeometry* geometry, CConfig* config, bool conservative)
34  : CSolver(), Conservative(conservative),
35  prim_idx(config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE,
36  config->GetNEMOProblem(), geometry->GetnDim(), config->GetnSpecies()) {
37  nMarker = config->GetnMarker_All();
38 
39  /*--- Store the number of vertices on each marker for deallocation later ---*/
40  nVertex.resize(nMarker);
41  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) nVertex[iMarker] = geometry->nVertex[iMarker];
42 
43  /* A grid is defined as dynamic if there's rigid grid movement or grid deformation AND the problem is time domain */
44  dynamic_grid = config->GetDynamic_Grid();
45 
46 #ifdef HAVE_OMP
47  /*--- Get the edge coloring, see notes in CEulerSolver's constructor. ---*/
48  su2double parallelEff = 1.0;
49 #ifdef CODI_REVERSE_TYPE
50  /*--- For the discrete adjoint, the reducer strategy is costly. Prefer coloring, possibly with reduced edge color
51  * group size. Find the maximum edge color group size that yields an efficient coloring. Also, allow larger numbers
52  * of colors. ---*/
53  const bool relax = config->GetEdgeColoringRelaxDiscAdj();
54  const auto& coloring = geometry->GetEdgeColoring(&parallelEff, relax);
55 #else
56  const auto& coloring = geometry->GetEdgeColoring(&parallelEff);
57 #endif
58 
59  ReducerStrategy = parallelEff < COLORING_EFF_THRESH;
60 
61  if (ReducerStrategy && (coloring.getOuterSize() > 1)) geometry->SetNaturalEdgeColoring();
62 
63  if (!coloring.empty()) {
64  auto groupSize = ReducerStrategy ? 1ul : geometry->GetEdgeColorGroupSize();
65  auto nColor = coloring.getOuterSize();
66  EdgeColoring.reserve(nColor);
67 
68  for (auto iColor = 0ul; iColor < nColor; ++iColor)
69  EdgeColoring.emplace_back(coloring.innerIdx(iColor), coloring.getNumNonZeros(iColor), groupSize);
70  }
71 
72  nPoint = geometry->GetnPoint();
73  omp_chunk_size = computeStaticChunkSize(nPoint, omp_get_max_threads(), OMP_MAX_SIZE);
74 #else
75  EdgeColoring[0] = DummyGridColor<>(geometry->GetnEdge());
76 #endif
77 
78  /*--- Initialize lower and upper limits for solution clipping. Solvers might overwrite these values. ---*/
79  for (unsigned int iVar = 0; iVar < MAXNVAR; iVar++) {
80  lowerlimit[iVar] = std::numeric_limits<su2double>::lowest();
81  upperlimit[iVar] = std::numeric_limits<su2double>::max();
82  }
83 }

90 template <class VariableType>
91 void CScalarSolver<VariableType>::CommonPreprocessing(CGeometry *geometry, const CConfig *config, const bool Output) {
92  /*--- Define booleans that are solver specific through CConfig's GlobalParams which have to be set in CFluidIteration
93  * before calling these solver functions. ---*/
94  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
95  const bool muscl = config->GetMUSCL();
96  const bool limiter = (config->GetKind_SlopeLimit() != LIMITER::NONE) &&
97  (config->GetInnerIter() <= config->GetLimiterIter());
98 
99  /*--- Clear residual and system matrix, not needed for
100  * reducer strategy as we write over the entire matrix. ---*/
101  if (!ReducerStrategy && !Output) {
102  LinSysRes.SetValZero();
103  if (implicit) {
104  Jacobian.SetValZero();
105  } else {
106  SU2_OMP_BARRIER
107  }
108  }
109 
110  /*--- Upwind second order reconstruction and gradients ---*/
111 
112  if (config->GetReconstructionGradientRequired()) {
113  switch(config->GetKind_Gradient_Method_Recon()) {
114  case GREEN_GAUSS: SetSolution_Gradient_GG(geometry, config, -1, true); break;
115  case LEAST_SQUARES: SetSolution_Gradient_LS(geometry, config, -1, true); break;
116  case WEIGHTED_LEAST_SQUARES: SetSolution_Gradient_LS(geometry, config, -1, true); break;
117  }
118  }
119 
120  switch(config->GetKind_Gradient_Method()) {
121  case GREEN_GAUSS: SetSolution_Gradient_GG(geometry, config, -1); break;
122  case WEIGHTED_LEAST_SQUARES: SetSolution_Gradient_LS(geometry, config, -1); break;
123  }
124 
125  if (limiter && muscl) SetSolution_Limiter(geometry, config);
126 }

128 template <class VariableType>
129 void CScalarSolver<VariableType>::Upwind_Residual(CGeometry* geometry, CSolver** solver_container,
130  CNumerics** numerics_container, CConfig* config,
131  unsigned short iMesh) {
132  /*--- Define booleans that are solver specific through CConfig's GlobalParams which have to be set in CFluidIteration
133  * before calling these solver functions. ---*/
134  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
135  const bool muscl = config->GetMUSCL();
136  const bool limiter = (config->GetKind_SlopeLimit() != LIMITER::NONE) &&
137  (config->GetInnerIter() <= config->GetLimiterIter());
138 
139  /*--- Only reconstruct flow variables if MUSCL is on for flow (requires upwind) and turbulence. ---*/
140  const bool musclFlow = config->GetMUSCL_Flow() && muscl && (config->GetKind_ConvNumScheme_Flow() == SPACE_UPWIND);
141  /*--- Only consider flow limiters for cell-based limiters, edge-based would need to be recomputed. ---*/
142  const bool limiterFlow =
143  (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE) && (config->GetKind_SlopeLimit_Flow() != LIMITER::VAN_ALBADA_EDGE);
144 
145  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
146  const auto& edgeMassFluxes = *(solver_container[FLOW_SOL]->GetEdgeMassFluxes());
147 
148  /*--- Pick one numerics object per thread. ---*/
149  auto* numerics = numerics_container[CONV_TERM + omp_get_thread_num() * MAX_TERMS];
150 
151  /*--- Apply scalar advection correction terms for bounded scalar problems ---*/
152  const bool bounded_scalar = numerics->GetBoundedScalar();
153 
154  /*--- Static arrays of MUSCL-reconstructed flow primitives and turbulence variables (thread safety). ---*/
155  su2double solution_i[MAXNVAR] = {0.0}, flowPrimVar_i[MAXNVARFLOW] = {0.0};
156  su2double solution_j[MAXNVAR] = {0.0}, flowPrimVar_j[MAXNVARFLOW] = {0.0};
157 
158  /*--- For hybrid parallel AD, pause preaccumulation if there is shared reading of
159  * variables, otherwise switch to the faster adjoint evaluation mode. ---*/
160  bool pausePreacc = false;
161  if (ReducerStrategy)
162  pausePreacc = AD::PausePreaccumulation();
163  else
164  AD::StartNoSharedReading();
165 
166  /*--- Loop over edge colors. ---*/
167  for (auto color : EdgeColoring) {
168  /*--- Chunk size is at least OMP_MIN_SIZE and a multiple of the color group size. ---*/
169  SU2_OMP_FOR_DYN(nextMultiple(OMP_MIN_SIZE, color.groupSize))
170  for (auto k = 0ul; k < color.size; ++k) {
171  auto iEdge = color.indices[k];
172 
173  unsigned short iDim, iVar;
174 
175  /*--- Points in edge and normal vectors ---*/
176 
177  auto iPoint = geometry->edges->GetNode(iEdge, 0);
178  auto jPoint = geometry->edges->GetNode(iEdge, 1);
179 
180  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
181 
182  /*--- Primitive variables w/o reconstruction ---*/
183 
184  const auto V_i = flowNodes->GetPrimitive(iPoint);
185  const auto V_j = flowNodes->GetPrimitive(jPoint);
186  numerics->SetPrimitive(V_i, V_j);
187 
188  /*--- Scalar variables w/o reconstruction ---*/
189 
190  const auto Scalar_i = nodes->GetSolution(iPoint);
191  const auto Scalar_j = nodes->GetSolution(jPoint);
192  numerics->SetScalarVar(Scalar_i, Scalar_j);
193 
194  /*--- Grid Movement ---*/
195 
196  if (dynamic_grid) numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(jPoint));
197 
198  if (muscl || musclFlow) {
199  const su2double *Limiter_i = nullptr, *Limiter_j = nullptr;
200 
201  const auto Coord_i = geometry->nodes->GetCoord(iPoint);
202  const auto Coord_j = geometry->nodes->GetCoord(jPoint);
203 
204  su2double Vector_ij[MAXNDIM] = {0.0};
205  for (iDim = 0; iDim < nDim; iDim++) {
206  Vector_ij[iDim] = 0.5 * (Coord_j[iDim] - Coord_i[iDim]);
207  }
208 
209  if (musclFlow && !bounded_scalar) {
210  /*--- Reconstruct mean flow primitive variables, note that in bounded scalar mode this is
211  * not necessary because the edge mass flux is read directly from the flow solver, instead
212  * of being computed from the primitive flow variables. ---*/
213 
214  auto Gradient_i = flowNodes->GetGradient_Reconstruction(iPoint);
215  auto Gradient_j = flowNodes->GetGradient_Reconstruction(jPoint);
216 
217  if (limiterFlow) {
218  Limiter_i = flowNodes->GetLimiter_Primitive(iPoint);
219  Limiter_j = flowNodes->GetLimiter_Primitive(jPoint);
220  }
221 
222  for (iVar = 0; iVar < solver_container[FLOW_SOL]->GetnPrimVarGrad(); iVar++) {
223  su2double Project_Grad_i = 0.0;
224  su2double Project_Grad_j = 0.0;
225  for (iDim = 0; iDim < nDim; iDim++) {
226  Project_Grad_i += Vector_ij[iDim] * Gradient_i[iVar][iDim];
227  Project_Grad_j -= Vector_ij[iDim] * Gradient_j[iVar][iDim];
228  }
229  if (limiterFlow) {
230  Project_Grad_i *= Limiter_i[iVar];
231  Project_Grad_j *= Limiter_j[iVar];
232  }
233  flowPrimVar_i[iVar] = V_i[iVar] + Project_Grad_i;
234  flowPrimVar_j[iVar] = V_j[iVar] + Project_Grad_j;
235  }
236 
237  numerics->SetPrimitive(flowPrimVar_i, flowPrimVar_j);
238  }
239 
240  if (muscl) {
241  /*--- Reconstruct scalar variables. ---*/
242 
243  auto Gradient_i = nodes->GetGradient_Reconstruction(iPoint);
244  auto Gradient_j = nodes->GetGradient_Reconstruction(jPoint);
245 
246  if (limiter) {
247  Limiter_i = nodes->GetLimiter(iPoint);
248  Limiter_j = nodes->GetLimiter(jPoint);
249  }
250 
251  for (iVar = 0; iVar < nVar; iVar++) {
252  su2double Project_Grad_i = 0.0, Project_Grad_j = 0.0;
253  for (iDim = 0; iDim < nDim; iDim++) {
254  Project_Grad_i += Vector_ij[iDim] * Gradient_i[iVar][iDim];
255  Project_Grad_j -= Vector_ij[iDim] * Gradient_j[iVar][iDim];
256  }
257  if (limiter) {
258  Project_Grad_i *= Limiter_i[iVar];
259  Project_Grad_j *= Limiter_j[iVar];
260  }
261  solution_i[iVar] = Scalar_i[iVar] + Project_Grad_i;
262  solution_j[iVar] = Scalar_j[iVar] + Project_Grad_j;
263  }
264 
265  numerics->SetScalarVar(solution_i, solution_j);
266  }
267  }
268 
269  /*--- Convective flux ---*/
270  su2double EdgeMassFlux = 0.0;
271  if (bounded_scalar) {
272  EdgeMassFlux = edgeMassFluxes[iEdge];
273  numerics->SetMassFlux(EdgeMassFlux);
274  }
275 
276  /*--- Update convective residual value ---*/
277 
278  auto residual = numerics->ComputeResidual(config);
279 
280  if (ReducerStrategy) {
281  EdgeFluxes.SetBlock(iEdge, residual);
282  if (implicit) Jacobian.SetBlocks(iEdge, residual.jacobian_i, residual.jacobian_j);
283  } else {
284  LinSysRes.AddBlock(iPoint, residual);
285  LinSysRes.SubtractBlock(jPoint, residual);
286  if (implicit) Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
287  }
288 
289  /*--- Apply convective flux correction to negate the effects of flow divergence in case of incompressible flow.
290  * Note that for the bounded scalar model, we explicitly put div(v)=0.
291  * If the ReducerStrategy is used, the corrections need to be applied in a loop over nodes
292  * to avoid race conditions in accessing nodes shared by edges handled by different threads. ---*/
293 
294  if (bounded_scalar && !ReducerStrategy) {
295  LinSysRes.AddBlock(iPoint, nodes->GetSolution(iPoint), -EdgeMassFlux);
296  LinSysRes.AddBlock(jPoint, nodes->GetSolution(jPoint), EdgeMassFlux);
297 
298  if (implicit) {
299  Jacobian.AddVal2Diag(iPoint, -EdgeMassFlux);
300  Jacobian.AddVal2Diag(jPoint, EdgeMassFlux);
301  }
302  }
303 
304  /*--- Viscous contribution. ---*/
305 
306  Viscous_Residual(iEdge, geometry, solver_container,
307  numerics_container[VISC_TERM + omp_get_thread_num() * MAX_TERMS], config);
308  }
309  END_SU2_OMP_FOR
310  } // end color loop
311 
312  /*--- Restore preaccumulation and adjoint evaluation state. ---*/
313  AD::ResumePreaccumulation(pausePreacc);
314  if (!ReducerStrategy) AD::EndNoSharedReading();
315 
316  if (ReducerStrategy) {
317  SumEdgeFluxes(geometry);
318  if (implicit) Jacobian.SetDiagonalAsColumnSum();
319 
320  /*--- Bounded scalar correction that cannot be applied in the edge loop when using the ReducerStrategy. ---*/
321  if (bounded_scalar) {
322  SU2_OMP_FOR_STAT(omp_chunk_size)
323  for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
324  const auto* solution = nodes->GetSolution(iPoint);
325  su2double divergence = 0;
326 
327  for (auto iEdge : geometry->nodes->GetEdges(iPoint)) {
328  const auto sign = (iPoint == geometry->edges->GetNode(iEdge,0)) ? 1 : -1;
329  const su2double EdgeMassFlux = sign * edgeMassFluxes[iEdge];
330  divergence += EdgeMassFlux;
331  LinSysRes.AddBlock(iPoint, solution, -EdgeMassFlux);
332  }
333  if (implicit) {
334  Jacobian.AddVal2Diag(iPoint, -divergence);
335  }
336  }
337  END_SU2_OMP_FOR
338  }
339  }
340 }

342 template <class VariableType>
343 void CScalarSolver<VariableType>::SumEdgeFluxes(CGeometry* geometry) {
344  SU2_OMP_FOR_STAT(omp_chunk_size)
345  for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
346  LinSysRes.SetBlock_Zero(iPoint);
347 
348  for (auto iEdge : geometry->nodes->GetEdges(iPoint)) {
349  if (iPoint == geometry->edges->GetNode(iEdge, 0))
350  LinSysRes.AddBlock(iPoint, EdgeFluxes.GetBlock(iEdge));
351  else
352  LinSysRes.SubtractBlock(iPoint, EdgeFluxes.GetBlock(iEdge));
353  }
354  }
355  END_SU2_OMP_FOR
356 }

361  /*--- Complete residuals for periodic boundary conditions. We loop over
362  the periodic BCs in matching pairs so that, in the event that there are

368  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_RESIDUAL);
369  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_RESIDUAL);
370  }

373 template <class VariableType>
374 void CScalarSolver<VariableType>::BC_Far_Field(CGeometry* geometry, CSolver** solver_container,
375  CNumerics* conv_numerics, CNumerics*, CConfig *config,
376  unsigned short val_marker) {
377 
378  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
379 
380  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
381  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
382 
383  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
384 
385  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
386 
387  if (geometry->nodes->GetDomain(iPoint)) {
388 
389  /*--- Allocate the value at the infinity ---*/
390 
391  auto V_infty = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
392 
393  /*--- Retrieve solution at the farfield boundary node ---*/
394 
395  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
396 
397  /*--- Grid Movement ---*/
398 
399  if (dynamic_grid)
400  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
401 
402  conv_numerics->SetPrimitive(V_domain, V_infty);
403 
404  /*--- Set turbulent variable at the wall, and at infinity ---*/
405 
406  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Solution_Inf);
407 
408  /*--- Set Normal (it is necessary to change the sign) ---*/
409 
410  su2double Normal[MAXNDIM] = {0.0};
411  for (auto iDim = 0u; iDim < nDim; iDim++)
412  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
413  conv_numerics->SetNormal(Normal);
414 
415  if (conv_numerics->GetBoundedScalar()) {
416  const su2double* velocity = &V_infty[prim_idx.Velocity()];
417  const su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
418  conv_numerics->SetMassFlux(BoundedScalarBCFlux(iPoint, implicit, density, velocity, Normal));
419  }
420 
421  /*--- Compute residuals and Jacobians ---*/
422 
423  auto residual = conv_numerics->ComputeResidual(config);
424 
425  /*--- Add residuals and Jacobians ---*/
426 
427  LinSysRes.AddBlock(iPoint, residual);
428  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
429  }
430  }
431  END_SU2_OMP_FOR
432 }

447 template <class VariableType>
448 void CScalarSolver<VariableType>::PrepareImplicitIteration(CGeometry* geometry, CSolver** solver_container,
449  CConfig* config) {
450  /*--- Set shared residual variables to 0 and declare
451  * local ones for current thread to work on. ---*/
452 
453  SetResToZero();
454 
455  su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
456  unsigned long idxMax[MAXNVAR] = {0};
457 
458  /*--- Build implicit system ---*/
459 
460  SU2_OMP_FOR_(schedule(static, omp_chunk_size) SU2_NOWAIT)
461  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
462  /*--- Modify matrix diagonal to improve diagonal dominance. ---*/
463  const su2double dt = nodes->GetDelta_Time(iPoint);
464 
465  if (dt != 0.0) {
466  su2double Vol = geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint);
467  Jacobian.AddVal2Diag(iPoint, Vol / dt);
468  } else {
469  Jacobian.SetVal2Diag(iPoint, 1.0);
470  LinSysRes.SetBlock_Zero(iPoint);
471  }
472 
473  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
474 
475  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
476  unsigned long total_index = iPoint * nVar + iVar;
477  LinSysRes[total_index] = -LinSysRes[total_index];
478  LinSysSol[total_index] = 0.0;
479 
480  /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
481  ResidualReductions_PerThread(iPoint, iVar, LinSysRes[total_index], resRMS, resMax, idxMax);
482  }
483  }
484  END_SU2_OMP_FOR
485 
486  /*--- "Add" residuals from all threads to global residual variables. ---*/
487  ResidualReductions_FromAllThreads(geometry, config, resRMS, resMax, idxMax);
488 }

490 template <class VariableType>
491 void CScalarSolver<VariableType>::CompleteImplicitIteration(CGeometry* geometry, CSolver** solver_container,
492  CConfig* config) {
493  const bool compressible = (config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE);
494 
495  ComputeUnderRelaxationFactor(config);
496 
497  /*--- Update solution (system written in terms of increments) ---*/
498 
499  if (!adjoint) {
500  /*--- Update the scalar solution. For transport equations, where Solution is not equivalent with the transported
501  * quantity, multiply the respective factor. ---*/
502  if (Conservative) {
503  const auto flowNodes = solver_container[FLOW_SOL]->GetNodes();
504 
505  SU2_OMP_FOR_STAT(omp_chunk_size)
506  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
507  /*--- Multiply the Solution var with density to get the conservative transported quantity, if necessary. ---*/
508  const su2double density = flowNodes->GetDensity(iPoint);
509  const su2double density_old = compressible ? flowNodes->GetSolution_Old(iPoint, 0) : density;
510 
511  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
512  nodes->AddClippedSolution(iPoint, iVar, nodes->GetUnderRelaxation(iPoint) * LinSysSol(iPoint, iVar),
513  lowerlimit[iVar], upperlimit[iVar], density, density_old);
514  }
515  }
516  END_SU2_OMP_FOR
517  } else {
518  SU2_OMP_FOR_STAT(omp_chunk_size)
519  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
520  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
521  nodes->AddClippedSolution(iPoint, iVar, nodes->GetUnderRelaxation(iPoint) * LinSysSol(iPoint, iVar),
522  lowerlimit[iVar], upperlimit[iVar]);
523  }
524  }
525  END_SU2_OMP_FOR
526  }
527  }
528 
529  for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic() / 2; iPeriodic++) {
530  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_IMPLICIT);
531  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_IMPLICIT);
532  }
533 
534  InitiateComms(geometry, config, MPI_QUANTITIES::SOLUTION);
535  CompleteComms(geometry, config, MPI_QUANTITIES::SOLUTION);
536 }

538 template <class VariableType>
539 void CScalarSolver<VariableType>::ImplicitEuler_Iteration(CGeometry* geometry, CSolver** solver_container,
540  CConfig* config) {
541  PrepareImplicitIteration(geometry, solver_container, config);
542 
543  /*--- Solve or smooth the linear system. ---*/
544 
545  SU2_OMP_FOR_(schedule(static, OMP_MIN_SIZE) SU2_NOWAIT)
546  for (unsigned long iPoint = nPointDomain; iPoint < nPoint; iPoint++) {

565  CConfig* config) {
566  /*--- Local residual variables for current thread ---*/
567  su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
568  unsigned long idxMax[MAXNVAR] = {0};
569 
570  SU2_OMP_FOR_STAT(omp_chunk_size)
571  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
572  const su2double dt = nodes->GetDelta_Time(iPoint);
573  const su2double Vol = geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint);

576  /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
577  ResidualReductions_PerThread(iPoint, iVar, LinSysRes(iPoint, iVar), resRMS, resMax, idxMax);
578  /*--- Explicit Euler step: ---*/
579  LinSysSol(iPoint, iVar) = -dt / Vol * LinSysRes(iPoint, iVar);
580  }
581  }
582  END_SU2_OMP_FOR
583

585  ResidualReductions_FromAllThreads(geometry, config, resRMS, resMax, idxMax);
586 
587  /*--- Use LinSysSol for solution update. ---*/
588  CompleteImplicitIteration(geometry, solver_container, config);

591 template <class VariableType>
592 void CScalarSolver<VariableType>::SetResidual_DualTime(CGeometry* geometry, CSolver** solver_container, CConfig* config,
593  unsigned short iRKStep, unsigned short iMesh,
594  unsigned short RunTime_EqSystem) {
595  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
596  const bool first_order = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
597  const bool second_order = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
598  const bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
599 
600  /*--- Flow solution, needed to get density. ---*/
601 
602  CVariable* flowNodes = solver_container[FLOW_SOL]->GetNodes();
603 
604  /*--- Store the physical time step ---*/
605 
606  const su2double TimeStep = config->GetDelta_UnstTimeND();
607 
608  /*--- Local variables ---*/
609 
610  unsigned short iVar, iMarker, iDim, iNeigh;
611  unsigned long iPoint, jPoint, iVertex, iEdge;
612 
613  su2double *U_time_nM1 = nullptr, *U_time_n = nullptr, *U_time_nP1 = nullptr;
614  /*--- For non-Conservative scalars (e.g. SA), multiply the Primitives with 1.0 instead of Density. ---*/
615  su2double Density_nM1 = 1.0, Density_n = 1.0, Density_nP1 = 1.0;
616  su2double Volume_nM1, Volume_nP1;
617  const su2double *Normal = nullptr, *GridVel_i = nullptr, *GridVel_j = nullptr;
618  su2double Residual_GCL;
619 
620  /*--- Compute the dual time-stepping source term for static meshes ---*/
621 
622  if (!dynamic_grid) {
623  /*--- Loop over all nodes (excluding halos) ---*/
624 
625  AD::StartNoSharedReading();
626 
627  SU2_OMP_FOR_STAT(omp_chunk_size)
628  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
629  if (Conservative) {
630  if (incompressible) {
631  /*--- This is temporary and only valid for constant-density problems:
632  density could also be temperature dependent, but as it is not a part
633  of the solution vector it's neither stored for previous time steps
634  nor updated with the solution at the end of each iteration. */
635  Density_nM1 = flowNodes->GetDensity(iPoint);
636  Density_n = flowNodes->GetDensity(iPoint);
637  Density_nP1 = flowNodes->GetDensity(iPoint);
638  } else {
639  Density_nM1 = flowNodes->GetSolution_time_n1(iPoint)[0];
640  Density_n = flowNodes->GetSolution_time_n(iPoint, 0);
641  Density_nP1 = flowNodes->GetSolution(iPoint, 0);
642  }
643  }
644 
645  /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
646  we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
647  previous solutions that are stored in memory. ---*/
648 
649  U_time_nM1 = nodes->GetSolution_time_n1(iPoint);
650  U_time_n = nodes->GetSolution_time_n(iPoint);
651  U_time_nP1 = nodes->GetSolution(iPoint);
652 
653  /*--- CV volume at time n+1. As we are on a static mesh, the volume
654  of the CV will remained fixed for all time steps. ---*/
655 
656  Volume_nP1 = geometry->nodes->GetVolume(iPoint);
657 
658  /*--- Compute the dual time-stepping source term based on the chosen
659  time discretization scheme (1st- or 2nd-order).---*/
660 
661  for (iVar = 0; iVar < nVar; iVar++) {
662  if (first_order)
663  LinSysRes(iPoint, iVar) +=
664  (Density_nP1 * U_time_nP1[iVar] - Density_n * U_time_n[iVar]) * Volume_nP1 / TimeStep;
665  if (second_order)
666  LinSysRes(iPoint, iVar) += (3.0 * Density_nP1 * U_time_nP1[iVar] - 4.0 * Density_n * U_time_n[iVar] +
667  1.0 * Density_nM1 * U_time_nM1[iVar]) *
668  Volume_nP1 / (2.0 * TimeStep);
669  }
670 
671  /*--- Compute the Jacobian contribution due to the dual time source term. ---*/
672  if (implicit) {
673  if (first_order) Jacobian.AddVal2Diag(iPoint, Volume_nP1 / TimeStep);
674  if (second_order) Jacobian.AddVal2Diag(iPoint, (Volume_nP1 * 3.0) / (2.0 * TimeStep));
675  }
676  }
677  END_SU2_OMP_FOR
678 
679  AD::EndNoSharedReading();
680 
681  } else {
682  /*--- For unsteady flows on dynamic meshes (rigidly transforming or
683  dynamically deforming), the Geometric Conservation Law (GCL) should be
684  satisfied in conjunction with the ALE formulation of the governing
685  equations. The GCL prevents accuracy issues caused by grid motion, i.e.
686  a uniform free-stream should be preserved through a moving grid. First,
687  we will loop over the edges and boundaries to compute the GCL component
688  of the dual time source term that depends on grid velocities. ---*/
689 
690  SU2_OMP_FOR_STAT(omp_chunk_size)
691  for (iPoint = 0; iPoint < nPointDomain; ++iPoint) {
692  GridVel_i = geometry->nodes->GetGridVel(iPoint);
693  U_time_n = nodes->GetSolution_time_n(iPoint);
694 
695  if (Conservative) {
696  if (incompressible)
697  Density_n = flowNodes->GetDensity(iPoint); // Temporary fix
698  else
699  Density_n = flowNodes->GetSolution_time_n(iPoint, 0);
700  }
701 
702  for (iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); iNeigh++) {
703  iEdge = geometry->nodes->GetEdge(iPoint, iNeigh);
704  Normal = geometry->edges->GetNormal(iEdge);
705 
706  jPoint = geometry->nodes->GetPoint(iPoint, iNeigh);
707  GridVel_j = geometry->nodes->GetGridVel(jPoint);
708 
709  /*--- Determine whether to consider the normal outward or inward. ---*/
710  su2double dir = (iPoint < jPoint) ? 0.5 : -0.5;
711 
712  Residual_GCL = 0.0;
713  for (iDim = 0; iDim < nDim; iDim++) Residual_GCL += dir * (GridVel_i[iDim] + GridVel_j[iDim]) * Normal[iDim];
714 
715  Residual_GCL *= Density_n;
716 
717  for (iVar = 0; iVar < nVar; iVar++) LinSysRes(iPoint, iVar) += U_time_n[iVar] * Residual_GCL;
718  }
719  }
720  END_SU2_OMP_FOR
721 
722  /*--- Loop over the boundary edges ---*/
723 
724  for (iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
725  if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
726  (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
727  (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
728  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
729  for (iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
730  /*--- Get the index for node i plus the boundary face normal ---*/
731 
732  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
733  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
734 
735  /*--- Grid velocities stored at boundary node i ---*/
736 
737  GridVel_i = geometry->nodes->GetGridVel(iPoint);
738 
739  /*--- Compute the GCL term by dotting the grid velocity with the face
740  normal. The normal is negated to match the boundary convention. ---*/
741 
742  Residual_GCL = 0.0;
743  for (iDim = 0; iDim < nDim; iDim++) Residual_GCL -= 0.5 * (GridVel_i[iDim] + GridVel_i[iDim]) * Normal[iDim];
744 
745  /*--- Compute the GCL component of the source term for node i ---*/
746 
747  U_time_n = nodes->GetSolution_time_n(iPoint);
748 
749  /*--- Multiply by density at node i for the SST model ---*/
750 
751  if (Conservative) {
752  if (incompressible)
753  Density_n = flowNodes->GetDensity(iPoint); // Temporary fix
754  else
755  Density_n = flowNodes->GetSolution_time_n(iPoint, 0);
756  }
757 
758  for (iVar = 0; iVar < nVar; iVar++) LinSysRes(iPoint, iVar) += Density_n * U_time_n[iVar] * Residual_GCL;
759  }
760  END_SU2_OMP_FOR
761  }
762  }
763 
764  /*--- Loop over all nodes (excluding halos) to compute the remainder
765  of the dual time-stepping source term. ---*/
766 
767  AD::EndNoSharedReading();
768 
769  SU2_OMP_FOR_STAT(omp_chunk_size)
770  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
771  /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
772  we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
773  previous solutions that are stored in memory. ---*/
774 
775  U_time_nM1 = nodes->GetSolution_time_n1(iPoint);
776  U_time_n = nodes->GetSolution_time_n(iPoint);
777  U_time_nP1 = nodes->GetSolution(iPoint);
778 
779  /*--- CV volume at time n-1 and n+1. In the case of dynamically deforming
780  grids, the volumes will change. On rigidly transforming grids, the
781  volumes will remain constant. ---*/
782 
783  Volume_nM1 = geometry->nodes->GetVolume_nM1(iPoint);
784  Volume_nP1 = geometry->nodes->GetVolume(iPoint);
785 
786  /*--- Compute the dual time-stepping source residual. Due to the
787  introduction of the GCL term above, the remainder of the source residual
788  due to the time discretization has a new form.---*/
789 
790  if (Conservative) {
791  /*--- If this is the SST model, we need to multiply by the density
792  in order to get the conservative variables ---*/
793  if (incompressible) {
794  /*--- This is temporary and only valid for constant-density problems:
795  density could also be temperature dependent, but as it is not a part
796  of the solution vector it's neither stored for previous time steps
797  nor updated with the solution at the end of each iteration. */
798  Density_nM1 = flowNodes->GetDensity(iPoint);
799  Density_n = flowNodes->GetDensity(iPoint);
800  Density_nP1 = flowNodes->GetDensity(iPoint);
801  } else {
802  Density_nM1 = flowNodes->GetSolution_time_n1(iPoint)[0];
803  Density_n = flowNodes->GetSolution_time_n(iPoint, 0);
804  Density_nP1 = flowNodes->GetSolution(iPoint, 0);
805  }
806  }
807 
808  for (iVar = 0; iVar < nVar; iVar++) {
809  if (first_order)
810  LinSysRes(iPoint, iVar) +=
811  (Density_nP1 * U_time_nP1[iVar] - Density_n * U_time_n[iVar]) * (Volume_nP1 / TimeStep);
812  if (second_order)
813  LinSysRes(iPoint, iVar) +=
814  (Density_nP1 * U_time_nP1[iVar] - Density_n * U_time_n[iVar]) * (3.0 * Volume_nP1 / (2.0 * TimeStep)) +
815  (Density_nM1 * U_time_nM1[iVar] - Density_n * U_time_n[iVar]) * (Volume_nM1 / (2.0 * TimeStep));
816  }
817 
818  /*--- Compute the Jacobian contribution due to the dual time source term. ---*/
819  if (implicit) {
820  if (first_order) Jacobian.AddVal2Diag(iPoint, Volume_nP1 / TimeStep);
821  if (second_order) Jacobian.AddVal2Diag(iPoint, (Volume_nP1 * 3.0) / (2.0 * TimeStep));
822  }
823  }
824  END_SU2_OMP_FOR
825 
826  AD::EndNoSharedReading();
827 
828  } // end dynamic grid
829 }
